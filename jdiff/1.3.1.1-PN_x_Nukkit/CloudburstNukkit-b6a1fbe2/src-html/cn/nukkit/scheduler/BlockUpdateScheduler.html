001package cn.nukkit.scheduler;
002
003import cn.nukkit.block.Block;
004import cn.nukkit.level.Level;
005import cn.nukkit.math.AxisAlignedBB;
006import cn.nukkit.math.NukkitMath;
007import cn.nukkit.math.Vector3;
008import cn.nukkit.utils.BlockUpdateEntry;
009import com.google.common.collect.Maps;
010
011import java.util.*;
012
013public class BlockUpdateScheduler {
014    private final Level level;
015    private long lastTick;
016    private Map<Long, LinkedHashSet<BlockUpdateEntry>> queuedUpdates;
017
018    private Set<BlockUpdateEntry> pendingUpdates;
019
020    public BlockUpdateScheduler(Level level, long currentTick) {
021        queuedUpdates = Maps.newHashMap(); // Change to ConcurrentHashMap if this needs to be concurrent
022        lastTick = currentTick;
023        this.level = level;
024    }
025
026    public synchronized void tick(long currentTick) {
027        // Should only perform once, unless ticks were skipped
028        if (currentTick - lastTick < Short.MAX_VALUE) {// Arbitrary
029            for (long tick = lastTick + 1; tick <= currentTick; tick++) {
030                perform(tick);
031            }
032        } else {
033            ArrayList<Long> times = new ArrayList<>(queuedUpdates.keySet());
034            Collections.sort(times);
035            for (long tick : times) {
036                if (tick <= currentTick) {
037                    perform(tick);
038                } else {
039                    break;
040                }
041            }
042        }
043        lastTick = currentTick;
044    }
045
046    private void perform(long tick) {
047        try {
048            lastTick = tick;
049            Set<BlockUpdateEntry> updates = pendingUpdates = queuedUpdates.remove(tick);
050            if (updates != null) {
051                for (BlockUpdateEntry entry : updates) {
052                    Vector3 pos = entry.pos;
053                    if (level.isChunkLoaded(NukkitMath.floorDouble(pos.x) >> 4, NukkitMath.floorDouble(pos.z) >> 4)) {
054                        Block block = level.getBlock(entry.pos);
055
056                        if (Block.equals(block, entry.block, false)) {
057                            block.onUpdate(Level.BLOCK_UPDATE_SCHEDULED);
058                        }
059                    } else {
060                        level.scheduleUpdate(entry.block, entry.pos, 0);
061                    }
062                }
063            }
064        } finally {
065            pendingUpdates = null;
066        }
067    }
068
069    public Set<BlockUpdateEntry> getPendingBlockUpdates(AxisAlignedBB boundingBox) {
070        Set<BlockUpdateEntry> set = null;
071
072        for (Map.Entry<Long, LinkedHashSet<BlockUpdateEntry>> tickEntries : this.queuedUpdates.entrySet()) {
073            LinkedHashSet<BlockUpdateEntry> tickSet = tickEntries.getValue();
074            for (BlockUpdateEntry update : tickSet) {
075                Vector3 pos = update.pos;
076
077                if (pos.getX() >= boundingBox.getMinX() && pos.getX() < boundingBox.getMaxX() && pos.getZ() >= boundingBox.getMinZ() && pos.getZ() < boundingBox.getMaxZ()) {
078                    if (set == null) {
079                        set = new LinkedHashSet<>();
080                    }
081
082                    set.add(update);
083                }
084            }
085        }
086
087        return set;
088    }
089
090    public boolean isBlockTickPending(Vector3 pos, Block block) {
091        Set<BlockUpdateEntry> tmpUpdates = pendingUpdates;
092        if (tmpUpdates == null || tmpUpdates.isEmpty()) return false;
093        return tmpUpdates.contains(new BlockUpdateEntry(pos, block));
094    }
095
096    private long getMinTime(BlockUpdateEntry entry) {
097        return Math.max(entry.delay, lastTick + 1);
098    }
099
100    public void add(BlockUpdateEntry entry) {
101        long time = getMinTime(entry);
102        LinkedHashSet<BlockUpdateEntry> updateSet = queuedUpdates.get(time);
103        if (updateSet == null) {
104            LinkedHashSet<BlockUpdateEntry> tmp = queuedUpdates.putIfAbsent(time, updateSet = new LinkedHashSet<>());
105            if (tmp != null) updateSet = tmp;
106        }
107        updateSet.add(entry);
108    }
109
110    public boolean contains(BlockUpdateEntry entry) {
111        for (Map.Entry<Long, LinkedHashSet<BlockUpdateEntry>> tickUpdateSet : queuedUpdates.entrySet()) {
112            if (tickUpdateSet.getValue().contains(entry)) {
113                return true;
114            }
115        }
116        return false;
117    }
118
119    public boolean remove(BlockUpdateEntry entry) {
120        for (Map.Entry<Long, LinkedHashSet<BlockUpdateEntry>> tickUpdateSet : queuedUpdates.entrySet()) {
121            if (tickUpdateSet.getValue().remove(entry)) {
122                return true;
123            }
124        }
125        return false;
126    }
127
128    public boolean remove(Vector3 pos) {
129        for (Map.Entry<Long, LinkedHashSet<BlockUpdateEntry>> tickUpdateSet : queuedUpdates.entrySet()) {
130            if (tickUpdateSet.getValue().remove(pos)) {
131                return true;
132            }
133        }
134        return false;
135    }
136}
