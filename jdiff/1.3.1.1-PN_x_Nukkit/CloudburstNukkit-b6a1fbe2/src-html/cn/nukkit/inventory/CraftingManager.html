035        if (i1.getId() > i2.getId()) {

042            return -1;
043        } else return Integer.compare(i1.getCount(), i2.getCount());

164        CraftingDataPacket pk = new CraftingDataPacket();
165        pk.cleanRecipes = true;
166        for (Recipe recipe : this.getRecipes()) {
167            if (recipe instanceof ShapedRecipe) {
168                pk.addShapedRecipe((ShapedRecipe) recipe);
169            } else if (recipe instanceof ShapelessRecipe) {
170                pk.addShapelessRecipe((ShapelessRecipe) recipe);
171            }
172        }
173        for (FurnaceRecipe recipe : this.getFurnaceRecipes().values()) {
174            pk.addFurnaceRecipe(recipe);
175        }
176        for (BrewingRecipe recipe : brewingRecipes.values()) {
177            pk.addBrewingRecipe(recipe);
178        }
179        for (ContainerRecipe recipe : containerRecipes.values()) {
180            pk.addContainerRecipe(recipe);
181        }
182        pk.encode();
183        packet = pk.compress(Deflater.BEST_COMPRESSION);

281    public CraftingRecipe matchRecipe(List<Item> inputList, Item primaryOutput, List<Item> extraOutputList) {
282        //TODO: try to match special recipes before anything else (first they need to be implemented!)
283        int outputHash = getItemHash(primaryOutput);
284        if (this.shapedRecipes.containsKey(outputHash)) {
285            inputList.sort(recipeComparator);
286            UUID inputHash = getMultiItemHash(inputList);
287            Map<UUID, ShapedRecipe> recipeMap = shapedRecipes.get(outputHash);
288            if (recipeMap != null) {
289                ShapedRecipe recipe = recipeMap.get(inputHash);
290                if (recipe != null && (recipe.matchItems(inputList, extraOutputList) || matchItemsAccumulation(recipe, inputList, primaryOutput, extraOutputList))) {
291                    return recipe;
292                }
293                for (ShapedRecipe shapedRecipe : recipeMap.values()) {
294                    if (shapedRecipe.matchItems(inputList, extraOutputList) || matchItemsAccumulation(shapedRecipe, inputList, primaryOutput, extraOutputList)) {
295                        return shapedRecipe;
296                    }
297                }
298            }
299        }
300        if (shapelessRecipes.containsKey(outputHash)) {
301            inputList.sort(recipeComparator);
302            UUID inputHash = getMultiItemHash(inputList);
303            Map<UUID, ShapelessRecipe> recipes = shapelessRecipes.get(outputHash);
304            if (recipes == null) {
305                return null;
306            }
307            ShapelessRecipe recipe = recipes.get(inputHash);
308            if (recipe != null && (recipe.matchItems(inputList, extraOutputList) || matchItemsAccumulation(recipe, inputList, primaryOutput, extraOutputList))) {
309                return recipe;
310            }
311            for (ShapelessRecipe shapelessRecipe : recipes.values()) {
312                if (shapelessRecipe.matchItems(inputList, extraOutputList) || matchItemsAccumulation(shapelessRecipe, inputList, primaryOutput, extraOutputList)) {
313                    return shapelessRecipe;
314                }
315            }
316        }
317        return null;
318    }

320    private boolean matchItemsAccumulation(CraftingRecipe recipe, List<Item> inputList, Item primaryOutput, List<Item> extraOutputList) {
321        Item recipeResult = recipe.getResult();
322        if (primaryOutput.equals(recipeResult, recipeResult.hasMeta(), recipeResult.hasCompoundTag()) && primaryOutput.getCount() % recipeResult.getCount() == 0) {
323            int multiplier = primaryOutput.getCount() / recipeResult.getCount();
324            return recipe.matchItems(inputList, extraOutputList, multiplier);
325        }
326        return false;
327    }
