001package cn.nukkit.plugin;
002
003import java.io.File;
004import java.util.regex.Pattern;
005
006/**
007 * 描述一个插件加载器的接口。<br>
008 * An interface to describe a plugin loader.
009 *
010 * @author iNevet(code) @ Nukkit Project
011 * @author 粉鞋大妈(javadoc) @ Nukkit Project
012 * @see JavaPluginLoader
013 * @since Nukkit 1.0 | Nukkit API 1.0.0
014 */
015public interface PluginLoader {
016
017    /**
018     * 通过文件名字的字符串，来加载和初始化一个插件。<br>
019     * Loads and initializes a plugin by its file name.
020     *
021     * <p>这个方法应该设置好插件的相关属性。比如，插件所在的服务器对象，插件的加载器对象，插件的描述对象，插件的数据文件夹。<br>
022     * Properties for loaded plugin should be set in this method. Such as, the {@code Server} object for which this
023     * plugin is running in, the {@code PluginLoader} object for its loader, and the {@code File} object for its
024     * data folder.</p>
025     *
026     * <p>如果插件加载失败，这个方法应该返回{@code null}，或者抛出异常。<br>
027     * If the plugin loader does not load this plugin successfully, a {@code null} should be returned,
028     * or an exception should be thrown.</p>
029     *
030     * @param filename 这个插件的文件名字字符串。<br>A string of its file name.
031     * @return 加载完毕的插件的 {@code Plugin}对象。<br>The loaded plugin as a {@code Plugin} object.
032     * @throws java.lang.Exception 插件加载失败所抛出的异常。<br>Thrown when an error occurred.
033     * @see #loadPlugin(File)
034     * @see cn.nukkit.plugin.PluginBase#init(PluginLoader, cn.nukkit.Server, PluginDescription, File, File)
035     * @since Nukkit 1.0 | Nukkit API 1.0.0
036     */
037    Plugin loadPlugin(String filename) throws Exception;
038
039    /**
040     * 通过插件的 {@code File}对象，来加载和初始化一个插件。<br>
041     * Loads and initializes a plugin by a {@code File} object describes the file.
042     *
043     * <p>这个方法应该设置好插件的相关属性。比如，插件所在的服务器对象，插件的加载器对象，插件的描述对象，插件的数据文件夹。<br>
044     * Properties for loaded plugin should be set in this method. Such as, the {@code Server} object for which this
045     * plugin is running in, the {@code PluginLoader} object for its loader, and the {@code File} object for its
046     * data folder.</p>
047     *
048     * <p>如果插件加载失败，这个方法应该返回{@code null}，或者抛出异常。<br>
049     * If the plugin loader does not load this plugin successfully, a {@code null} should be returned,
050     * or an exception should be thrown.</p>
051     *
052     * @param file 这个插件的文件的 {@code File}对象。<br>A {@code File} object for this plugin.
053     * @return 加载完毕的插件的 {@code Plugin}对象。<br>The loaded plugin as a {@code Plugin} object.
054     * @throws java.lang.Exception 插件加载失败所抛出的异常。<br>Thrown when an error occurred.
055     * @see #loadPlugin(String)
056     * @see cn.nukkit.plugin.PluginBase#init(PluginLoader, cn.nukkit.Server, PluginDescription, File, File)
057     * @since Nukkit 1.0 | Nukkit API 1.0.0
058     */
059    Plugin loadPlugin(File file) throws Exception;
060
061    /**
062     * 通过插件文件名的字符串，来获得描述这个插件的 {@code PluginDescription}对象。<br>
063     * Gets a {@code PluginDescription} object describes the plugin by its file name.
064     *
065     * <p>如果插件的描述对象获取失败，这个方法应该返回{@code null}。<br>
066     * If the plugin loader does not get its description successfully, a {@code null} should be returned.</p>
067     *
068     * @param filename 这个插件的文件名字。<br>A string of its file name.
069     * @return 描述这个插件的 {@code PluginDescription}对象。<br>
070     * A {@code PluginDescription} object describes the plugin.
071     * @see #getPluginDescription(File)
072     * @since Nukkit 1.0 | Nukkit API 1.0.0
073     */
074    PluginDescription getPluginDescription(String filename);
075
076    /**
077     * 通过插件的 {@code File}对象，来获得描述这个插件的 {@code PluginDescription}对象。<br>
078     * Gets a {@code PluginDescription} object describes the plugin by a {@code File} object describes the plugin file.
079     *
080     * <p>如果插件的描述对象获取失败，这个方法应该返回{@code null}。<br>
081     * If the plugin loader does not get its description successfully, a {@code null} should be returned.</p>
082     *
083     * @param file 这个插件的文件的 {@code File}对象。<br>A {@code File} object for this plugin.
084     * @return 描述这个插件的 {@code PluginDescription}对象。<br>
085     * A {@code PluginDescription} object describes the plugin.
086     * @see #getPluginDescription(String)
087     * @since Nukkit 1.0 | Nukkit API 1.0.0
088     */
089    PluginDescription getPluginDescription(File file);
090
091    /**
092     * 返回这个插件加载器支持的文件类型。<br>
093     * Returns the file types this plugin loader supports.
094     *
095     * <p>在Nukkit读取所有插件时，插件管理器会查找所有已经安装的插件加载器，通过识别这个插件是否满足下面的条件，
096     * 来选择对应的插件加载器。<br>
097     * When Nukkit is trying to load all its plugins, the plugin manager will look for all installed plugin loader,
098     * and choose the correct one by checking if this plugin matches the filters given below.</p>
099     *
100     * <p>举个例子，识别这个文件是否以jar为扩展名，它的正则表达式是：<br>
101     * For example, to check if this file is has a "jar" extension, the regular expression should be:<br>
102     * {@code ^.+\\.jar$}<br>
103     * 所以只读取jar扩展名的插件加载器，这个函数应该写成：<br>
104     * So, for a jar-extension-only file plugin loader, this method should be:
105     * </p>
106     * <pre> {@code           @Override}
107     *      public Pattern[] getPluginFilters() {
108     *          return new Pattern[]{Pattern.compile("^.+\\.jar$")};
109     *      }
110     * </pre>
111     *
112     * @return 表达这个插件加载器支持的文件类型的正则表达式数组。<br>
113     * An array of regular expressions, that describes what kind of file this plugin loader supports.
114     * @since Nukkit 1.0 | Nukkit API 1.0.0
115     */
116    Pattern[] getPluginFilters();
117
118    /**
119     * 启用一个插件。<br>
120     * Enables a plugin.
121     *
122     * @param plugin 要被启用的插件。<br>The plugin to enable.
123     * @see #disablePlugin
124     * @since Nukkit 1.0 | Nukkit API 1.0.0
125     */
126    void enablePlugin(Plugin plugin);
127
128    /**
129     * 停用一个插件。<br>
130     * Disables a plugin.
131     *
132     * @param plugin 要被停用的插件。<br>The plugin to disable.
133     * @see #enablePlugin
134     * @since Nukkit 1.0 | Nukkit API 1.0.0
135     */
136    void disablePlugin(Plugin plugin);
137
138}
