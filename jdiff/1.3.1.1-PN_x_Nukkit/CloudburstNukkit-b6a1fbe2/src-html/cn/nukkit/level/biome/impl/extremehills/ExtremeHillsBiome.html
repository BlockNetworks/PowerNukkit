001package cn.nukkit.level.biome.impl.extremehills;
002
003import cn.nukkit.block.BlockSapling;
004import cn.nukkit.level.biome.type.GrassyBiome;
005import cn.nukkit.level.generator.populator.impl.PopulatorTree;
006
007/**
008 * author: DaPorkchop_
009 * Nukkit Project
010 * <p>
011 * make sure this is touching another extreme hills type or it'll look dumb
012 *
013 * steep mountains with flat areas between
014 */
015public class ExtremeHillsBiome extends GrassyBiome {
016    public ExtremeHillsBiome() {
017        this(true);
018    }
019
020    public ExtremeHillsBiome(boolean tree) {
021        super();
022
023        if (tree) {
024            PopulatorTree trees = new PopulatorTree(BlockSapling.SPRUCE);
025            trees.setBaseAmount(2);
026            trees.setRandomAmount(2);
027            this.addPopulator(trees);
028        }
029
030        this.setBaseHeight(1f);
031        this.setHeightVariation(0.5f);
032    }
033
034    @Override
035    public String getName() {
036        return "Extreme Hills";
037    }
038
039    @Override
040    public boolean doesOverhang() {
041        return true;
042    }
043}
