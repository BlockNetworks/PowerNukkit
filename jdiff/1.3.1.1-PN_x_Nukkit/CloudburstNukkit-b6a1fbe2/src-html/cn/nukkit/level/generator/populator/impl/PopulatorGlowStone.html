001package cn.nukkit.level.generator.populator.impl;
002
003import cn.nukkit.block.Block;
004import cn.nukkit.level.ChunkManager;
005import cn.nukkit.level.format.FullChunk;
006import cn.nukkit.level.generator.populator.type.Populator;
007import cn.nukkit.math.NukkitMath;
008import cn.nukkit.math.NukkitRandom;
009
010public class PopulatorGlowStone extends Populator {
011    @Override
012    public void populate(ChunkManager level, int chunkX, int chunkZ, NukkitRandom random, FullChunk chunk) {
013        int x = NukkitMath.randomRange(random, chunkX << 4, (chunkX << 4) + 15);
014        int z = NukkitMath.randomRange(random, chunkZ << 4, (chunkZ << 4) + 15);
015        int y = this.getHighestWorkableBlock(chunk, x & 0xF, z & 0xF);
016        if (y != -1 && level.getBlockIdAt(x, y, z) != NETHERRACK) {
017            int count = NukkitMath.randomRange(random, 40, 60);
018            for (int i = 0; i < count; i++) {
019                level.setBlockAt(x + (random.nextBoundedInt(7) - 3), y + (random.nextBoundedInt(9) - 4), z + (random.nextBoundedInt(7) - 3), GLOWSTONE);
020            }
021        }
022    }
023
024    private int getHighestWorkableBlock(FullChunk chunk, int x, int z) {
025        int y;
026        //start scanning a bit lower down to allow space for placing on top
027        for (y = 120; y >= 0; y--) {
028            int b = chunk.getBlockId(x, y, z);
029            if (b == Block.AIR) {
030                break;
031            }
032        }
033        return y == 0 ? -1 : y;
034    }
035}
