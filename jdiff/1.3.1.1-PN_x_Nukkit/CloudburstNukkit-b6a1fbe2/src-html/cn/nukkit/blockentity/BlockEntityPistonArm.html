003import cn.nukkit.entity.Entity;

005import cn.nukkit.math.AxisAlignedBB;
006import cn.nukkit.math.BlockFace;

012
013/**
014 * @author CreeperFace
015 */
016public class BlockEntityPistonArm extends BlockEntity {
017
018    public float progress = 1.0F;
019    public float lastProgress = 1.0F;
020    public BlockFace facing;
021    public boolean extending = false;
022    public boolean sticky = false;
023    public byte state = 1;
024    public byte newState = 1;
025    public Vector3 attachedBlock = null;
026    public boolean isMovable = true;
027    public boolean powered = false;
028
029    public BlockEntityPistonArm(FullChunk chunk, CompoundTag nbt) {
030        super(chunk, nbt);
031    }
032
033    @Override
034    protected void initBlockEntity() {
035        if (namedTag.contains("Progress")) {
036            this.progress = namedTag.getFloat("Progress");
037        }
038
039        if (namedTag.contains("LastProgress")) {
040            this.lastProgress = (float) namedTag.getInt("LastProgress");
041        }
042
043        if (namedTag.contains("Sticky")) {
044            this.sticky = namedTag.getBoolean("Sticky");
045        }
046
047        if (namedTag.contains("Extending")) {
048            this.extending = namedTag.getBoolean("Extending");
049        }
050
051        if (namedTag.contains("powered")) {
052            this.powered = namedTag.getBoolean("powered");
053        }
054
055        if (namedTag.contains("AttachedBlocks")) {
056            ListTag blocks = namedTag.getList("AttachedBlocks", IntTag.class);
057            if (blocks != null && blocks.size() > 0) {
058                this.attachedBlock = new Vector3((double) ((IntTag) blocks.get(0)).getData(), (double) ((IntTag) blocks.get(1)).getData(), (double) ((IntTag) blocks.get(2)).getData());
059            }
060        } else {
061            namedTag.putList(new ListTag("AttachedBlocks"));
062        }
063
064        super.initBlockEntity();
065    }
066
067    private void pushEntities() {
068        float lastProgress = this.getExtendedProgress(this.lastProgress);
069        double x = (double) (lastProgress * (float) this.facing.getXOffset());
070        double y = (double) (lastProgress * (float) this.facing.getYOffset());
071        double z = (double) (lastProgress * (float) this.facing.getZOffset());
072        AxisAlignedBB bb = new SimpleAxisAlignedBB(x, y, z, x + 1.0D, y + 1.0D, z + 1.0D);
073        Entity[] entities = this.level.getCollidingEntities(bb);
074        if (entities.length != 0) {
075
076        }
077
078    }
079
080    private float getExtendedProgress(float progress) {
081        return this.extending ? progress - 1.0F : 1.0F - progress;
082    }
083
084    public boolean isBlockEntityValid() {
085        return true;
086    }
087
088    public void saveNBT() {
089        super.saveNBT();
090        this.namedTag.putBoolean("isMovable", this.isMovable);
091        this.namedTag.putByte("State", this.state);
092        this.namedTag.putByte("NewState", this.newState);
093        this.namedTag.putFloat("Progress", this.progress);
094        this.namedTag.putFloat("LastProgress", this.lastProgress);
095        this.namedTag.putBoolean("powered", this.powered);
096    }
097
098    public CompoundTag getSpawnCompound() {
099        return (new CompoundTag()).putString("id", "PistonArm").putInt("x", (int) this.x).putInt("y", (int) this.y).putInt("z", (int) this.z);
100    }
101}
