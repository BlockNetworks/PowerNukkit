033    public BaseLang(String lang, String path, String fallback) {
034        this.langName = lang.toLowerCase();
035        boolean useFallback = !lang.equals(fallback);
036
037        if (path == null) {
038            path = "lang/";
039            this.lang = this.loadLang(this.getClass().getClassLoader().getResourceAsStream(path + this.langName + "/lang.ini"));
040            if (useFallback) this.fallbackLang = this.loadLang(this.getClass().getClassLoader().getResourceAsStream(path + fallback + "/lang.ini"));
041        } else {
042            this.lang = this.loadLang(path + this.langName + "/lang.ini");
043            if (useFallback) this.fallbackLang = this.loadLang(path + fallback + "/lang.ini");
044        }
045        if (this.fallbackLang == null) this.fallbackLang = this.lang;
046
047
048    }

066    protected Map<String, String> loadLang(String path) {
067        try {
068            String content = Utils.readFile(path);
069            Map<String, String> d = new HashMap<>();
070            for (String line : content.split("\n")) {
071                line = line.trim();
072                if (line.equals("") || line.charAt(0) == '#') {
073                    continue;
074                }
075                String[] t = line.split("=");
076                if (t.length < 2) {
077                    continue;
078                }
079                String key = t[0];
080                String value = "";
081                for (int i = 1; i < t.length - 1; i++) {
082                    value += t[i] + "=";
083                }
084                value += t[t.length - 1];
085                if (value.equals("")) {
086                    continue;
087                }
088                d.put(key, value);
089            }
090            return d;
091        } catch (IOException e) {
092            Server.getInstance().getLogger().logException(e);
093            return null;
094        }
095    }
096

139    public String translateString(String str, Object... params) {
140        if (params != null) {
141            String[] paramsToString = new String[params.length];
142            for (int i = 0; i < params.length; i++) {
143                paramsToString[i] = Objects.toString(params[i]);
144            }
145            return this.translateString(str, paramsToString, null);
146        }
147        return this.translateString(str, new String[0], null);
148    }

150    public String translateString(String str, String param, String onlyPrefix) {
151        return this.translateString(str, new String[]{param}, onlyPrefix);
152    }

154    public String translateString(String str, String[] params, String onlyPrefix) {
155        String baseText = this.get(str);
156        baseText = this.parseTranslation((baseText != null && (onlyPrefix == null || str.indexOf(onlyPrefix) == 0)) ? baseText : str, onlyPrefix);
157        for (int i = 0; i < params.length; i++) {
158            baseText = baseText.replace("{%" + i + "}", this.parseTranslation(String.valueOf(params[i])));
159        }
160
161        return baseText;
162    }

164    public String translate(TextContainer c) {
165        String baseText = this.parseTranslation(c.getText());
166        if (c instanceof TranslationContainer) {
167            baseText = this.internalGet(c.getText());
168            baseText = this.parseTranslation(baseText != null ? baseText : c.getText());
169            for (int i = 0; i < ((TranslationContainer) c).getParameters().length; i++) {
170                baseText = baseText.replace("{%" + i + "}", this.parseTranslation(((TranslationContainer) c).getParameters()[i]));
171            }
172        }
173        return baseText;
174    }

176    public String internalGet(String id) {
177        if (this.lang.containsKey(id)) {
178            return this.lang.get(id);
179        } else if (this.fallbackLang.containsKey(id)) {
180            return this.fallbackLang.get(id);
181        }
182        return null;
183    }

185    public String get(String id) {
186        if (this.lang.containsKey(id)) {
187            return this.lang.get(id);
188        } else if (this.fallbackLang.containsKey(id)) {
189            return this.fallbackLang.get(id);
190        }
191        return id;
192    }

194    protected String parseTranslation(String text) {
195        return this.parseTranslation(text, null);
196    }
197
198    protected String parseTranslation(String text, String onlyPrefix) {
199        String newString = "";
200        text = String.valueOf(text);
201
202        String replaceString = null;
203
204        int len = text.length();
205
206        for (int i = 0; i < len; ++i) {
207            char c = text.charAt(i);
208            if (replaceString != null) {
209                int ord = c;
210                if ((ord >= 0x30 && ord <= 0x39) // 0-9
211                        || (ord >= 0x41 && ord <= 0x5a) // A-Z
212                        || (ord >= 0x61 && ord <= 0x7a) || // a-z
213                        c == '.' || c == '-') {
214                    replaceString += String.valueOf(c);
215                } else {
216                    String t = this.internalGet(replaceString.substring(1));
217                    if (t != null && (onlyPrefix == null || replaceString.indexOf(onlyPrefix) == 1)) {
218                        newString += t;
219                    } else {
220                        newString += replaceString;
221                    }
222                    replaceString = null;
223                    if (c == '%') {
224                        replaceString = String.valueOf(c);
225                    } else {
226                        newString += String.valueOf(c);
227                    }
228                }
229            } else if (c == '%') {
230                replaceString = String.valueOf(c);
231            } else {
232                newString += String.valueOf(c);
233            }
234        }
235
236        if (replaceString != null) {
237            String t = this.internalGet(replaceString.substring(1));
238            if (t != null && (onlyPrefix == null || replaceString.indexOf(onlyPrefix) == 1)) {
239                newString += t;
240            } else {
241                newString += replaceString;
242            }
243        }
244        return newString;
245    }
