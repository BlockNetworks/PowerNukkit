001package cn.nukkit.entity.mob;
002
003import cn.nukkit.entity.Entity;
004import cn.nukkit.entity.data.ByteEntityData;
005import cn.nukkit.entity.weather.EntityLightningStrike;
006import cn.nukkit.event.entity.CreeperPowerEvent;
007import cn.nukkit.event.entity.EntityDamageByEntityEvent;
008import cn.nukkit.item.Item;
009import cn.nukkit.level.format.FullChunk;
010import cn.nukkit.nbt.tag.CompoundTag;
011
012import java.util.concurrent.ThreadLocalRandom;
013
014/**
015 * @author Box.
016 */
017public class EntityCreeper extends EntityMob {
018
019    public static final int NETWORK_ID = 33;
020
021    public static final int DATA_SWELL_DIRECTION = 16;
022    public static final int DATA_SWELL = 17;
023    public static final int DATA_SWELL_OLD = 18;
024    public static final int DATA_POWERED = 19;
025
026    @Override
027    public int getNetworkId() {
028        return NETWORK_ID;
029    }
030
031    @Override
032    public float getWidth() {
033        return 0.6f;
034    }
035
036    @Override
037    public float getHeight() {
038        return 1.7f;
039    }
040
041    public EntityCreeper(FullChunk chunk, CompoundTag nbt) {
042        super(chunk, nbt);
043    }
044
045    public boolean isPowered() {
046        return getDataPropertyBoolean(DATA_POWERED);
047    }
048
049    public void setPowered(EntityLightningStrike bolt) {
050        CreeperPowerEvent ev = new CreeperPowerEvent(this, bolt, CreeperPowerEvent.PowerCause.LIGHTNING);
051        this.getServer().getPluginManager().callEvent(ev);
052
053        if (!ev.isCancelled()) {
054            this.setDataProperty(new ByteEntityData(DATA_POWERED, 1));
055            this.namedTag.putBoolean("powered", true);
056        }
057    }
058
059    public void setPowered(boolean powered) {
060        CreeperPowerEvent ev = new CreeperPowerEvent(this, powered ? CreeperPowerEvent.PowerCause.SET_ON : CreeperPowerEvent.PowerCause.SET_OFF);
061        this.getServer().getPluginManager().callEvent(ev);
062
063        if (!ev.isCancelled()) {
064            this.setDataProperty(new ByteEntityData(DATA_POWERED, powered ? 1 : 0));
065            this.namedTag.putBoolean("powered", powered);
066        }
067    }
068
069    public void onStruckByLightning(Entity entity) {
070        this.setPowered(true);
071    }
072
073    @Override
074    protected void initEntity() {
075        super.initEntity();
076
077        if (this.namedTag.getBoolean("powered") || this.namedTag.getBoolean("IsPowered")) {
078            this.dataProperties.putBoolean(DATA_POWERED, true);
079        }
080        this.setMaxHealth(20);
081    }
082
083    @Override
084    public String getName() {
085        return "Creeper";
086    }
087
088    @Override
089    public Item[] getDrops() {
090        if (this.lastDamageCause instanceof EntityDamageByEntityEvent) {
091            return new Item[]{Item.get(Item.GUNPOWDER, ThreadLocalRandom.current().nextInt(2) + 1)};
092        }
093        return new Item[0];
094    }
095}
