085    //Opened API for plugins
086    public static Food registerFood(Food food, Plugin plugin) {
087        Objects.requireNonNull(food);
088        Objects.requireNonNull(plugin);
089        food.relativeIDs.forEach(n -> registryCustom.put(new NodeIDMetaPlugin(n.id, n.meta, plugin), food));
090        return food;
091    }
092
093    private static Food registerDefaultFood(Food food) {
094        food.relativeIDs.forEach(n -> registryDefault.put(n, food));
095        return food;
096    }
097
098    public static Food getByRelative(Item item) {
099        Objects.requireNonNull(item);
100        return getByRelative(item.getId(), item.getDamage());
101    }
102
103    public static Food getByRelative(Block block) {
104        Objects.requireNonNull(block);
105        return getByRelative(block.getId(), block.getDamage());
106    }
107
108    public static Food getByRelative(int relativeID, int meta) {
109        final Food[] result = {null};
110        registryCustom.forEach((n, f) -> {
111            if (n.id == relativeID && n.meta == meta && n.plugin.isEnabled()) result[0] = f;
112        });
113        if (result[0] == null) {
114            registryDefault.forEach((n, f) -> {
115                if (n.id == relativeID && n.meta == meta) result[0] = f;
116            });
117        }
118        return result[0];
119    }

121    protected int restoreFood = 0;
122    protected float restoreSaturation = 0;
123    protected final List<NodeIDMeta> relativeIDs = new ArrayList<>();
124
125    public final boolean eatenBy(Player player) {
126        PlayerEatFoodEvent event = new PlayerEatFoodEvent(player, this);
127        player.getServer().getPluginManager().callEvent(event);
128        if (event.isCancelled()) return false;
129        return event.getFood().onEatenBy(player);
130    }
131
132    protected boolean onEatenBy(Player player) {
133        player.getFoodData().addFoodLevel(this);
134        return true;
135    }
136
137    public Food addRelative(int relativeID) {
138        return addRelative(relativeID, 0);
139    }
140
141    public Food addRelative(int relativeID, int meta) {
142        NodeIDMeta node = new NodeIDMeta(relativeID, meta);
143        return addRelative(node);
144    }
145
146    private Food addRelative(NodeIDMeta node) {
147        if (!relativeIDs.contains(node)) relativeIDs.add(node);
148        return this;
149    }

169    static class NodeIDMeta {
170        final int id;
171        final int meta;
172
173        NodeIDMeta(int id, int meta) {
174            this.id = id;
175            this.meta = meta;
176        }
177    }
178
179    static class NodeIDMetaPlugin extends NodeIDMeta {
180        final Plugin plugin;
181
182        NodeIDMetaPlugin(int id, int meta, Plugin plugin) {
183            super(id, meta);
184            this.plugin = plugin;
185        }
186    }
