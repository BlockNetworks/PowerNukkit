022public abstract class BlockPistonBase extends BlockSolidMeta implements Faceable {

024    public boolean sticky;

026    public BlockPistonBase() {
027        this(0);

030    public BlockPistonBase(int meta) {
031        super(meta);
032    }

034    @Override
035    public double getResistance() {
036        return 2.5;
037    }

039    @Override
040    public double getHardness() {
041        return 0.5;
042    }

044    @Override
045    public boolean place(Item item, Block block, Block target, BlockFace face, double fx, double fy, double fz, Player player) {
046        if (Math.abs(player.x - this.x) < 2 && Math.abs(player.z - this.z) < 2) {
047            double y = player.y + player.getEyeHeight();
048
049            if (y - this.y > 2) {
050                this.setDamage(BlockFace.UP.getIndex());
051            } else if (this.y - y > 0) {
052                this.setDamage(BlockFace.DOWN.getIndex());
053            } else {
054                this.setDamage(player.getHorizontalFacing().getIndex());
055            }
056        } else {
057            this.setDamage(player.getHorizontalFacing().getIndex());
058        }
059        this.level.setBlock(block, this, true, false);
060
061        CompoundTag nbt = new CompoundTag("")
062                .putString("id", BlockEntity.PISTON_ARM)
063                .putInt("x", (int) this.x)
064                .putInt("y", (int) this.y)
065                .putInt("z", (int) this.z)
066                .putBoolean("Sticky", this.sticky);
067
068        BlockEntityPistonArm be = (BlockEntityPistonArm) BlockEntity.createBlockEntity(BlockEntity.PISTON_ARM, this.level.getChunk((int) this.x >> 4, (int) this.z >> 4), nbt);
069
070        if (be == null) return false;
071        //this.checkState();
072        return true;
073    }
074
075    @Override
076    public boolean onBreak(Item item) {
077        this.level.setBlock(this, Block.get(BlockID.AIR), true, true);
078
079        Block block = this.getSide(getFacing());
080
081        if (block instanceof BlockPistonHead && ((BlockPistonHead) block).getFacing() == this.getFacing()) {
082            block.onBreak(item);
083        }
084        return true;
085    }
086
087    public boolean isExtended() {
088        BlockFace face = getFacing();
089        Block block = getSide(face);
090        return block instanceof BlockPistonHead && ((BlockPistonHead) block).getFacing() == face;
091    }
092
093    @Override
094    public int onUpdate(int type) {
095        if (type != 6 && type != 1) {
096            return 0;
097        } else {
098            BlockEntity blockEntity = this.level.getBlockEntity(this);
099            if (blockEntity instanceof BlockEntityPistonArm) {
100                BlockEntityPistonArm arm = (BlockEntityPistonArm) blockEntity;
101                boolean powered = this.isPowered();
102                if (arm.powered != powered) {
103                    this.level.getServer().getPluginManager().callEvent(new BlockPistonChangeEvent(this, powered ? 0 : 15, powered ? 15 : 0));
104                    arm.powered = !arm.powered;
105                    if (arm.chunk != null) {
106                        arm.chunk.setChanged();
107                    }
108                }
109            }
110
111            return type;
112        }
113    }

115    private void checkState() {
116        BlockFace facing = getFacing();
117        boolean isPowered = this.isPowered();
118
119        if (isPowered && !isExtended()) {
120            if ((new BlocksCalculator(this.level, this, facing, true)).canMove()) {
121                if (!this.doMove(true)) {
122                    return;
123                }
124
125                this.getLevel().addLevelSoundEvent(this, LevelSoundEventPacket.SOUND_PISTON_OUT);
126            } else {
127            }
128        } else if (!isPowered && isExtended()) {
129            //this.level.setBlock() TODO: set piston extension?
130
131            if (this.sticky) {
132                Vector3 pos = this.add(facing.getXOffset() * 2, facing.getYOffset() * 2, facing.getZOffset() * 2);
133                Block block = this.level.getBlock(pos);
134
135                if (block.getId() == AIR) {
136                    this.level.setBlock(this.getLocation().getSide(facing), Block.get(BlockID.AIR), true, true);
137                }
138                if (canPush(block, facing.getOpposite(), false) && (!(block instanceof BlockFlowable) || block.getId() == PISTON || block.getId() == STICKY_PISTON)) {
139                    this.doMove(false);
140                }
141            } else {
142                this.level.setBlock(getLocation().getSide(facing), Block.get(BlockID.AIR), true, false);
143            }
144
145            this.getLevel().addLevelSoundEvent(this, LevelSoundEventPacket.SOUND_PISTON_IN);
146        }
147    }
148
149    public BlockFace getFacing() {
150        return BlockFace.fromIndex(this.getDamage()).getOpposite();
151    }
152
153    private boolean isPowered() {
154        BlockFace face = getFacing();
155
156        for (BlockFace side : BlockFace.values()) {
157            if (side != face && this.level.isSidePowered(this.getLocation().getSide(side), side)) {
158                return true;
159            }
160        }
161
162        if (this.level.isSidePowered(this, BlockFace.DOWN)) {
163            return true;
164        } else {
165            Vector3 pos = this.getLocation().up();
166
167            for (BlockFace side : BlockFace.values()) {
168                if (side != BlockFace.DOWN && this.level.isSidePowered(pos.getSide(side), side)) {
169                    return true;
170                }
171            }
172
173            return false;
174        }
175    }
176
177    private boolean doMove(boolean extending) {
178        Vector3 pos = this.getLocation();
179        BlockFace direction = getFacing();
180
181        if (!extending) {
182            this.level.setBlock(pos.getSide(direction), Block.get(BlockID.AIR), true, false);
183        }
184
185        BlocksCalculator calculator = new BlocksCalculator(this.level, this, direction, extending);
186
187        if (!calculator.canMove()) {
188            return false;
189        } else {
190            List<Block> blocks = calculator.getBlocksToMove();
191
192            List<Block> newBlocks = new ArrayList<>(blocks);
193
194            List<Block> destroyBlocks = calculator.getBlocksToDestroy();
195            BlockFace side = extending ? direction : direction.getOpposite();
196
197            for (int i = destroyBlocks.size() - 1; i >= 0; --i) {
198                Block block = destroyBlocks.get(i);
199                this.level.useBreakOn(block);
200            }
201
202            for (int i = blocks.size() - 1; i >= 0; --i) {
203                Block block = blocks.get(i);
204                this.level.setBlock(block, Block.get(BlockID.AIR));
205                Vector3 newPos = block.getLocation().getSide(side);
206
207                //TODO: change this to block entity
208                this.level.setBlock(newPos, newBlocks.get(i));
209            }
210
211            Vector3 pistonHead = pos.getSide(direction);
212
213            if (extending) {
214                //extension block entity
215
216                this.level.setBlock(pistonHead, Block.get(BlockID.PISTON_HEAD, this.getDamage()));
217            }
218
219            return true;
220        }
221    }
222
223    public static boolean canPush(Block block, BlockFace face, boolean destroyBlocks) {
224        if (block.canBePushed() && block.getY() >= 0 && (face != BlockFace.DOWN || block.getY() != 0) &&
225                block.getY() <= 255 && (face != BlockFace.UP || block.getY() != 255)) {
226            if (!(block instanceof BlockPistonBase)) {
227
228                if (block instanceof BlockFlowable) {
229                    return destroyBlocks;
230                }
231            } else return !((BlockPistonBase) block).isExtended();
232            return true;
233        }
234        return false;
235
236    }

238    public class BlocksCalculator {
239
240        private final Level level;
241        private final Vector3 pistonPos;
242        private final Block blockToMove;
243        private final BlockFace moveDirection;
244
245        private final List<Block> toMove = new ArrayList<>();
246        private final List<Block> toDestroy = new ArrayList<>();
247
248        public BlocksCalculator(Level level, Block pos, BlockFace facing, boolean extending) {
249            this.level = level;
250            this.pistonPos = pos.getLocation();
251
252            if (extending) {
253                this.moveDirection = facing;
254                this.blockToMove = pos.getSide(facing);
255            } else {
256                this.moveDirection = facing.getOpposite();
257                this.blockToMove = pos.getSide(facing, 2);
258            }
259        }
260
261        public boolean canMove() {
262            this.toMove.clear();
263            this.toDestroy.clear();
264            Block block = this.blockToMove;
265
266            if (!canPush(block, this.moveDirection, false)) {
267                if (block instanceof BlockFlowable) {
268                    this.toDestroy.add(this.blockToMove);
269                    return true;
270                } else {
271                    return false;
272                }
273            } else if (!this.addBlockLine(this.blockToMove)) {
274                return false;
275            } else {
276                for (Block b : this.toMove) {
277                    if (b.getId() == SLIME_BLOCK && !this.addBranchingBlocks(b)) {
278                        return false;
279                    }
280                }
281
282                return true;
283            }
284        }
285
286        private boolean addBlockLine(Block origin) {
287            Block block = origin.clone();
288
289            if (block.getId() == AIR) {
290                return true;
291            } else if (!canPush(origin, this.moveDirection, false)) {
292                return true;
293            } else if (origin.equals(this.pistonPos)) {
294                return true;
295            } else if (this.toMove.contains(origin)) {
296                return true;
297            } else {
298                int count = 1;
299
300                if (count + this.toMove.size() > 12) {
301                    return false;
302                } else {
303                    while (block.getId() == SLIME_BLOCK) {
304                        block = origin.getSide(this.moveDirection.getOpposite(), count);
305
306                        if (block.getId() == AIR || !canPush(block, this.moveDirection, false) || block.equals(this.pistonPos)) {
307                            break;
308                        }
309
310                        ++count;
311
312                        if (count + this.toMove.size() > 12) {
313                            return false;
314                        }
315                    }
316
317                    int blockCount = 0;
318
319                    for (int step = count - 1; step >= 0; --step) {
320                        this.toMove.add(block.getSide(this.moveDirection.getOpposite(), step));
321                        ++blockCount;
322                    }
323
324                    int steps = 1;
325
326                    while (true) {
327                        Block nextBlock = block.getSide(this.moveDirection, steps);
328                        int index = this.toMove.indexOf(nextBlock);
329
330                        if (index > -1) {
331                            this.reorderListAtCollision(blockCount, index);
332
333                            for (int l = 0; l <= index + blockCount; ++l) {
334                                Block b = this.toMove.get(l);
335
336                                if (b.getId() == SLIME_BLOCK && !this.addBranchingBlocks(b)) {
337                                    return false;
338                                }
339                            }
340
341                            return true;
342                        }
343
344                        if (nextBlock.getId() == AIR) {
345                            return true;
346                        }
347
348                        if (!canPush(nextBlock, this.moveDirection, true) || nextBlock.equals(this.pistonPos)) {
349                            return false;
350                        }
351
352                        if (nextBlock instanceof BlockFlowable) {
353                            this.toDestroy.add(nextBlock);
354                            return true;
355                        }
356
357                        if (this.toMove.size() >= 12) {
358                            return false;
359                        }
360
361                        this.toMove.add(nextBlock);
362                        ++blockCount;
363                        ++steps;
364                    }
365                }
366            }
367        }
368
369        private void reorderListAtCollision(int count, int index) {
370            List<Block> list = new ArrayList<>(this.toMove.subList(0, index));
371            List<Block> list1 = new ArrayList<>(this.toMove.subList(this.toMove.size() - count, this.toMove.size()));
372            List<Block> list2 = new ArrayList<>(this.toMove.subList(index, this.toMove.size() - count));
373            this.toMove.clear();
374            this.toMove.addAll(list);
375            this.toMove.addAll(list1);
376            this.toMove.addAll(list2);
377        }
378
379        private boolean addBranchingBlocks(Block block) {
380            for (BlockFace face : BlockFace.values()) {
381                if (face.getAxis() != this.moveDirection.getAxis() && !this.addBlockLine(block.getSide(face))) {
382                    return false;
383                }
384            }
385
386            return true;
387        }
388
389        public List<Block> getBlocksToMove() {
390            return this.toMove;
391        }
392
393        public List<Block> getBlocksToDestroy() {
394            return this.toDestroy;
395        }
396    }

400        return new ItemBlock(this, 0);
