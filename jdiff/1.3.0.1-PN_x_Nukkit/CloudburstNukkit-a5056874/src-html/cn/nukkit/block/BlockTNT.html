001package cn.nukkit.block;
002
003import cn.nukkit.Player;
004import cn.nukkit.entity.Entity;
005import cn.nukkit.item.Item;
006import cn.nukkit.level.Level;
007import cn.nukkit.level.Sound;
008import cn.nukkit.math.NukkitRandom;
009import cn.nukkit.nbt.tag.CompoundTag;
010import cn.nukkit.nbt.tag.DoubleTag;
011import cn.nukkit.nbt.tag.FloatTag;
012import cn.nukkit.nbt.tag.ListTag;
013import cn.nukkit.utils.BlockColor;
014
015/**
016 * Created on 2015/12/8 by xtypr.
017 * Package cn.nukkit.block in project Nukkit .
018 */
019public class BlockTNT extends BlockSolid {
020
021    public BlockTNT() {
022    }
023
024    @Override
025    public String getName() {
026        return "TNT";
027    }
028
029    @Override
030    public int getId() {
031        return TNT;
032    }
033
034    @Override
035    public double getHardness() {
036        return 0;
037    }
038
039    @Override
040    public double getResistance() {
041        return 0;
042    }
043
044    @Override
045    public boolean canBeActivated() {
046        return true;
047    }
048
049    @Override
050    public int getBurnChance() {
051        return 15;
052    }
053
054    @Override
055    public int getBurnAbility() {
056        return 100;
057    }
058
059    public void prime() {
060        this.prime(80);
061    }
062
063    public void prime(int fuse) {
064        prime(fuse, null);
065    }
066
067    public void prime(int fuse, Entity source) {
068        this.getLevel().setBlock(this, Block.get(BlockID.AIR), true);
069        double mot = (new NukkitRandom()).nextSignedFloat() * Math.PI * 2;
070        CompoundTag nbt = new CompoundTag()
071                .putList(new ListTag<DoubleTag>("Pos")
072                        .add(new DoubleTag("", this.x + 0.5))
073                        .add(new DoubleTag("", this.y))
074                        .add(new DoubleTag("", this.z + 0.5)))
075                .putList(new ListTag<DoubleTag>("Motion")
076                        .add(new DoubleTag("", -Math.sin(mot) * 0.02))
077                        .add(new DoubleTag("", 0.2))
078                        .add(new DoubleTag("", -Math.cos(mot) * 0.02)))
079                .putList(new ListTag<FloatTag>("Rotation")
080                        .add(new FloatTag("", 0))
081                        .add(new FloatTag("", 0)))
082                .putShort("Fuse", fuse);
083        Entity tnt = Entity.createEntity("PrimedTnt",
084                this.getLevel().getChunk(this.getFloorX() >> 4, this.getFloorZ() >> 4),
085                nbt, source
086        );
087        if(tnt == null) {
088            return;
089        }
090        tnt.spawnToAll();
091        this.level.addSound(this, Sound.RANDOM_FUSE);
092    }
093
094    @Override
095    public int onUpdate(int type) {
096        if ((type == Level.BLOCK_UPDATE_NORMAL || type == Level.BLOCK_UPDATE_REDSTONE) && this.level.isBlockPowered(this.getLocation())) {
097            this.prime();
098        }
099
100        return 0;
101    }
102
103    @Override
104    public boolean onActivate(Item item, Player player) {
105        if (item.getId() == Item.FLINT_STEEL) {
106            item.useOn(this);
107            this.prime(80, player);
108            return true;
109        }
110        if (item.getId() == Item.FIRE_CHARGE) {
111            if (!player.isCreative()) player.getInventory().removeItem(Item.get(Item.FIRE_CHARGE, 0, 1));
112            this.level.addSound(player, Sound.MOB_GHAST_FIREBALL);
113            this.prime(80, player);
114            return true;
115        }
116        return false;
117    }
118
119    @Override
120    public BlockColor getColor() {
121        return BlockColor.TNT_BLOCK_COLOR;
122    }
123}
