001package cn.nukkit.command.defaults;
002
003import cn.nukkit.Player;
004import cn.nukkit.command.Command;
005import cn.nukkit.command.CommandSender;
006import cn.nukkit.command.data.CommandParamType;
007import cn.nukkit.command.data.CommandParameter;
008import cn.nukkit.lang.TranslationContainer;
009import cn.nukkit.utils.TextFormat;
010
011/**
012 * Created by Snake1999 on 2016/1/22.
013 * Package cn.nukkit.command.defaults in project nukkit.
014 */
015public class XpCommand extends Command {
016    public XpCommand(String name) {
017        super(name, "%nukkit.command.xp.description", "%commands.xp.usage");
018        this.setPermission("nukkit.command.xp");
019        this.commandParameters.clear();
020        this.commandParameters.put("default", new CommandParameter[]{
021                new CommandParameter("amount|level", CommandParamType.INT, false),
022                new CommandParameter("player", CommandParamType.TARGET, true)
023        });
024    }
025
026    @Override
027    public boolean execute(CommandSender sender, String commandLabel, String[] args) {
028        if (!this.testPermission(sender)) {
029            return true;
030        }
031
032        //  "/xp <amount> [player]"  for adding exp
033        //  "/xp <amount>L [player]" for adding exp level
034        String amountString;
035        String playerName;
036        Player player;
037        if (!(sender instanceof Player)) {
038            if (args.length != 2) {
039                sender.sendMessage(new TranslationContainer("commands.generic.usage", this.usageMessage));
040                return true;
041            }
042            amountString = args[0];
043            playerName = args[1];
044            player = sender.getServer().getPlayer(playerName);
045        } else {
046            if (args.length == 1) {
047                amountString = args[0];
048                player = (Player) sender;
049            } else if (args.length == 2) {
050                amountString = args[0];
051                playerName = args[1];
052                player = sender.getServer().getPlayer(playerName);
053            } else {
054                sender.sendMessage(new TranslationContainer("commands.generic.usage", this.usageMessage));
055                return true;
056            }
057        }
058
059        if (player == null) {
060            sender.sendMessage(new TranslationContainer(TextFormat.RED + "%commands.generic.player.notFound"));
061            return true;
062        }
063
064        int amount;
065        boolean isLevel = false;
066        if (amountString.endsWith("l") || amountString.endsWith("L")) {
067            amountString = amountString.substring(0, amountString.length() - 1);
068            isLevel = true;
069        }
070
071        try {
072            amount = Integer.parseInt(amountString);
073        } catch (NumberFormatException e1) {
074            sender.sendMessage(new TranslationContainer("commands.generic.usage", this.usageMessage));
075            return true;
076        }
077
078        if (isLevel) {
079            int newLevel = player.getExperienceLevel();
080            newLevel += amount;
081            if (newLevel > 24791) newLevel = 24791;
082            if (newLevel < 0) {
083                player.setExperience(0, 0);
084            } else {
085                player.setExperience(player.getExperience(), newLevel);
086            }
087            if (amount > 0) {
088                sender.sendMessage(new TranslationContainer("commands.xp.success.levels", String.valueOf(amount), player.getName()));
089            } else {
090                sender.sendMessage(new TranslationContainer("commands.xp.success.levels.minus", String.valueOf(-amount), player.getName()));
091            }
092            return true;
093        } else {
094            if (amount < 0) {
095                sender.sendMessage(new TranslationContainer("commands.generic.usage", this.usageMessage));
096                return true;
097            }
098            player.addExperience(amount);
099            sender.sendMessage(new TranslationContainer("commands.xp.success", String.valueOf(amount), player.getName()));
100            return true;
101        }
102    }
103}
