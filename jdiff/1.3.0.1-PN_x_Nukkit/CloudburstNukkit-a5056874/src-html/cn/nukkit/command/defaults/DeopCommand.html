001package cn.nukkit.command.defaults;
002
003import cn.nukkit.IPlayer;
004import cn.nukkit.Player;
005import cn.nukkit.command.Command;
006import cn.nukkit.command.CommandSender;
007import cn.nukkit.command.data.CommandParamType;
008import cn.nukkit.command.data.CommandParameter;
009import cn.nukkit.lang.TranslationContainer;
010import cn.nukkit.utils.TextFormat;
011
012/**
013 * Created on 2015/11/12 by xtypr.
014 * Package cn.nukkit.command.defaults in project Nukkit .
015 */
016public class DeopCommand extends VanillaCommand {
017    public DeopCommand(String name) {
018        super(name, "%nukkit.command.deop.description", "%commands.deop.description");
019        this.setPermission("nukkit.command.op.take");
020        this.commandParameters.put("default", new CommandParameter[]{
021                new CommandParameter("player", CommandParamType.TARGET, false)
022        });
023    }
024
025    @Override
026    public boolean execute(CommandSender sender, String commandLabel, String[] args) {
027        if (!this.testPermission(sender)) {
028            return true;
029        }
030
031        if (args.length == 0) {
032            sender.sendMessage(new TranslationContainer("commands.generic.usage", this.usageMessage));
033
034            return false;
035        }
036
037        String playerName = args[0];
038        IPlayer player = sender.getServer().getOfflinePlayer(playerName);
039        player.setOp(false);
040
041        if (player instanceof Player) {
042            ((Player) player).sendMessage(new TranslationContainer(TextFormat.GRAY + "%commands.deop.message"));
043        }
044
045        Command.broadcastCommandMessage(sender, new TranslationContainer("commands.deop.success", new String[]{player.getName()}));
046
047        return true;
048    }
049}
