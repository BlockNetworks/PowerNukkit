001package cn.nukkit.command;
002
003/**
004 * 能监听命令执行的类实现的接口。<br>
005 * An interface what can be implemented by classes which listens command executing.
006 *
007 * @author MagicDroidX(code) @ Nukkit Project
008 * @author 粉鞋大妈(javadoc) @ Nukkit Project
009 * @see cn.nukkit.plugin.PluginBase
010 * @see cn.nukkit.command.CommandExecutor#onCommand
011 * @since Nukkit 1.0 | Nukkit API 1.0.0
012 */
013public interface CommandExecutor {
014    /**
015     * 在命令执行时会调用的方法。<br>
016     * Called when a command is executed.
017     *
018     * <p>一个命令可以是{@code /a_LABEL an_arg1 AN_ARG2...}的形式，这时{@code label}变量的值为{@code "a_label"}，
019     * {@code args}数组的元素有{@code "an_arg1","AN_ARG2",...}。注意到{@code label}变量会被转化成小写，
020     * 而{@code args}数组内字符串元素的大小写不变。<br>
021     * A command can be such a form like {@code /a_LABEL an_arg1 AN_ARG2...}. At this time, the value of
022     * variable {@code label} is {@code "a_label"}, and the values of elements of array {@code args} are
023     * {@code "an_arg1","AN_ARG2",...}. Notice that the value of variable {@code label} will be converted to
024     * lower case, but the cases of elements of array {@code args} won't change.</p>
025     *
026     * <p>关于返回值，如果返回{@code false}，Nukkit会给sender发送这个命令的使用方法等信息，来表示这个命令没有使用成功。
027     * 如果你的命令成功的发挥了作用，你应该返回{@code true}来表示这个命令已执行成功。<br>
028     * If this function returns {@code false}, Nukkit will send command usages to command sender, to explain that
029     * the command didn't work normally. If your command works properly, a {@code true} should be returned to explain
030     * that the command works.</p>
031     *
032     * <p>如果你想测试一个命令发送者是否有权限执行这个命令，
033     * 可以使用{@link cn.nukkit.command.Command#testPermissionSilent}。<br>
034     * If you want to test whether a command sender has the permission to execute a command,
035     * you can use {@link cn.nukkit.command.Command#testPermissionSilent}.</p>
036     *
037     * @param sender  这个命令的发送者，可以是玩家或控制台等。<br>
038     *                The sender of this command, this can be a player or a console.
039     * @param command 要被发送的命令。<br>
040     *                The command to send.
041     * @param label   这个命令的标签。<br>
042     *                Label of the command.
043     * @param args    这个命令的参数列表。<br>
044     *                Arguments of this command.
045     * @return 这个命令执行是否执行成功。<br>whether this command is executed successfully.
046     * @since Nukkit 1.0 | Nukkit API 1.0.0
047     */
048    boolean onCommand(CommandSender sender, Command command, String label, String[] args);
049}
