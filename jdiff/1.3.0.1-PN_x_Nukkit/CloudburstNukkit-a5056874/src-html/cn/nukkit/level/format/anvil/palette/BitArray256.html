001package cn.nukkit.level.format.anvil.palette;
002
003import cn.nukkit.utils.ThreadCache;
004
005/**
006 * @author https://github.com/boy0001/
007 */
008public final class BitArray256 {
009    private final int bitsPerEntry;
010    protected final long[] data;
011
012    public BitArray256(int bitsPerEntry) {
013        this.bitsPerEntry = bitsPerEntry;
014        int longLen = (this.bitsPerEntry * 256) >> 6;
015        this.data = new long[longLen];
016    }
017
018    public BitArray256(BitArray256 other) {
019        this.bitsPerEntry = other.bitsPerEntry;
020        this.data = other.data.clone();
021    }
022
023    public final void setAt(int index, int value) {
024        int bitIndexStart = index * bitsPerEntry;
025        int longIndexStart = bitIndexStart >> 6;
026        int localBitIndexStart = bitIndexStart & 63;
027        this.data[longIndexStart] = this.data[longIndexStart] & ~((long) ((1 << bitsPerEntry) - 1) << localBitIndexStart) | ((long) value) << localBitIndexStart;
028
029        if(localBitIndexStart > 64 - bitsPerEntry) {
030            int longIndexEnd = longIndexStart + 1;
031            int localShiftStart = 64 - localBitIndexStart;
032            int localShiftEnd = bitsPerEntry - localShiftStart;
033            this.data[longIndexEnd] = this.data[longIndexEnd] >>> localShiftEnd << localShiftEnd | (((long) value) >> localShiftStart);
034        }
035    }
036
037    public final int getAt(int index) {
038        int bitIndexStart = index * bitsPerEntry;
039
040        int longIndexStart = bitIndexStart >> 6;
041
042        int localBitIndexStart = bitIndexStart & 63;
043        if(localBitIndexStart <= 64 - bitsPerEntry) {
044            return (int)(this.data[longIndexStart] >>> localBitIndexStart & ((1 << bitsPerEntry) - 1));
045        } else {
046            int localShift = 64 - localBitIndexStart;
047            return (int) ((this.data[longIndexStart] >>> localBitIndexStart | this.data[longIndexStart + 1] << localShift) & ((1 << bitsPerEntry) - 1));
048        }
049    }
050
051    public final void fromRaw(int[] arr) {
052        for (int i = 0; i < arr.length; i++) {
053            setAt(i, arr[i]);
054        }
055    }
056
057    public BitArray256 grow(int newBitsPerEntry) {
058        int amtGrow = newBitsPerEntry - this.bitsPerEntry;
059        if (amtGrow <= 0) return this;
060        BitArray256 newBitArray = new BitArray256(newBitsPerEntry);
061
062        int[] buffer = ThreadCache.intCache256.get();
063        toRaw(buffer);
064        newBitArray.fromRaw(buffer);
065
066        return newBitArray;
067    }
068
069    public BitArray256 growSlow(int bitsPerEntry) {
070        BitArray256 newBitArray = new BitArray256(bitsPerEntry);
071        for (int i = 0; i < 256; i++) {
072            newBitArray.setAt(i, getAt(i));
073        }
074        return newBitArray;
075    }
076
077    public final int[] toRaw(int[] buffer) {
078        for (int i = 0; i < buffer.length; i++) {
079            buffer[i] = getAt(i);
080        }
081        return buffer;
082    }
083
084    public final int[] toRaw() {
085        return toRaw(new int[256]);
086    }
087
088    public BitArray256 clone() {
089        return new BitArray256(this);
090    }
091}
