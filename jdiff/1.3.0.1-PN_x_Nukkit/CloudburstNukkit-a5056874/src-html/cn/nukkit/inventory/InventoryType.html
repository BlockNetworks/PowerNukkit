006 */
007public enum InventoryType {
008    CHEST(27, "Chest", 0),
009    ENDER_CHEST(27, "Ender Chest", 0),
010    DOUBLE_CHEST(27 + 27, "Double Chest", 0),
011    PLAYER(40, "Player", -1), //36 CONTAINER, 4 ARMOR
012    FURNACE(3, "Furnace", 2),
013    CRAFTING(5, "Crafting", 1), //4 CRAFTING slots, 1 RESULT
014    WORKBENCH(10, "Crafting", 1), //9 CRAFTING slots, 1 RESULT
015    BREWING_STAND(5, "Brewing", 4), //1 INPUT, 3 POTION, 1 fuel
016    ANVIL(3, "Anvil", 5), //2 INPUT, 1 OUTPUT
017    ENCHANT_TABLE(2, "Enchant", 3), //1 INPUT/OUTPUT, 1 LAPIS
018    DISPENSER(0, "Dispenser", 6), //9 CONTAINER
019    DROPPER(9, "Dropper", 7), //9 CONTAINER
020    HOPPER(5, "Hopper", 8), //5 CONTAINER
021    UI(1, "UI", -1),
022    SHULKER_BOX(27, "Shulker Box", 0),
023    BEACON(1, "Beacon", 13),
024    OFFHAND(1, "Offhand", -1),
025    MINECART_CHEST(27, "Minecart with Chest", 0),
026    MINECART_HOPPER(5, "Minecart with Hopper", 8);
027
028    private final int size;
029    private final String title;
030    private final int typeId;
031
032    InventoryType(int defaultSize, String defaultBlockEntity, int typeId) {
033        this.size = defaultSize;
034        this.title = defaultBlockEntity;
035        this.typeId = typeId;
036    }
037
038    public int getDefaultSize() {
039        return size;
040    }
041
042    public String getDefaultTitle() {
043        return title;
044    }
045
046    public int getNetworkType() {
047        return typeId;
048    }
049}
