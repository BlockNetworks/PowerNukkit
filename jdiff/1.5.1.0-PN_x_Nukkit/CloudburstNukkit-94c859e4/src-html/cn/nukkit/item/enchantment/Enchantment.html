120    public static Enchantment get(int id) {
121        Enchantment enchantment = null;
122        if (id >= 0 && id < enchantments.length) {
123            enchantment = enchantments[id];
124        }
125        if (enchantment == null) {
126            return new UnknownEnchantment(id);
127        }
128        return enchantment;
129    }
130
131    public static Enchantment getEnchantment(int id) {
132        return get(id).clone();
133    }

135    public static Enchantment[] getEnchantments() {
136        ArrayList<Enchantment> list = new ArrayList<>();
137        for (Enchantment enchantment : enchantments) {
138            if (enchantment == null) {
139                break;
140            }
141
142            list.add(enchantment);
143        }
144
145        return list.toArray(new Enchantment[0]);
146    }

266    public static String getRandomName() {
267        int count = ThreadLocalRandom.current().nextInt(3, 6);
268        HashSet<String> set = new HashSet<>();
269        while (set.size() < count) {
270            set.add(Enchantment.words[ThreadLocalRandom.current().nextInt(0, Enchantment.words.length)]);
271        }
272
273        String[] words = set.toArray(new String[0]);
274        return String.join(" ", words);
275    }

284    public enum Rarity {
285        COMMON(10),
286        UNCOMMON(5),
287        RARE(2),
288        VERY_RARE(1);
289
290        private final int weight;
291
292        Rarity(int weight) {
293            this.weight = weight;
294        }
295
296        public int getWeight() {
297            return this.weight;
298        }
299
300        public static Rarity fromWeight(int weight) {
301            if (weight < 2) {
302                return VERY_RARE;
303            } else if (weight < 5) {
304                return RARE;
305            } else if (weight < 10) {
306                return UNCOMMON;
307            }
308            return COMMON;
309        }
