001package cn.nukkit.utils;
002
003import cn.nukkit.block.Block;
004import cn.nukkit.level.Level;
005import cn.nukkit.math.BlockFace;
006import cn.nukkit.math.Vector3;
007
008import java.util.Iterator;
009
010/**
011 * author: MagicDroidX
012 * Nukkit Project
013 */
014public class BlockIterator implements Iterator<Block> {
015    private final Level level;
016    private final int maxDistance;
017
018    private static final int gridSize = 1 << 24;
019
020    private boolean end = false;
021
022    private final Block[] blockQueue;
023    private int currentBlock = 0;
024
025    private Block currentBlockObject = null;
026    private int currentDistance;
027    private int maxDistanceInt = 0;
028
029    private int secondError;
030    private int thirdError;
031
032    private final int secondStep;
033    private final int thirdStep;
034
035    private BlockFace mainFace;
036    private BlockFace secondFace;
037    private BlockFace thirdFace;
038
039    public BlockIterator(Level level, Vector3 start, Vector3 direction) {
040        this(level, start, direction, 0);
041    }
042
043    public BlockIterator(Level level, Vector3 start, Vector3 direction, double yOffset) {
044        this(level, start, direction, yOffset, 0);
045    }
046
047    public BlockIterator(Level level, Vector3 start, Vector3 direction, double yOffset, int maxDistance) {
048        this.level = level;
049        this.maxDistance = maxDistance;
050        this.blockQueue = new Block[3];
051
052        Vector3 startClone = new Vector3(start.x, start.y, start.z);
053        startClone.y += yOffset;
054
055        this.currentDistance = 0;
056
057        double mainDirection = 0;
058        double secondDirection = 0;
059        double thirdDirection = 0;
060
061        double mainPosition = 0;
062        double secondPosition = 0;
063        double thirdPosition = 0;
064
065        Vector3 pos = new Vector3(startClone.x, startClone.y, startClone.z);
066        Block startBlock = this.level.getBlock(new Vector3(Math.floor(pos.x), Math.floor(pos.y), Math.floor(pos.z)));
067
068        if (this.getXLength(direction) > mainDirection) {
069            this.mainFace = this.getXFace(direction);
070            mainDirection = this.getXLength(direction);
071            mainPosition = this.getXPosition(direction, startClone, startBlock);
072
073            this.secondFace = this.getYFace(direction);
074            secondDirection = this.getYLength(direction);
075            secondPosition = this.getYPosition(direction, startClone, startBlock);
076
077            this.thirdFace = this.getZFace(direction);
078            thirdDirection = this.getZLength(direction);
079            thirdPosition = this.getZPosition(direction, startClone, startBlock);
080        }
081        if (this.getYLength(direction) > mainDirection) {
082            this.mainFace = this.getYFace(direction);
083            mainDirection = this.getYLength(direction);
084            mainPosition = this.getYPosition(direction, startClone, startBlock);
085
086            this.secondFace = this.getZFace(direction);
087            secondDirection = this.getZLength(direction);
088            secondPosition = this.getZPosition(direction, startClone, startBlock);
089
090            this.thirdFace = this.getXFace(direction);
091            thirdDirection = this.getXLength(direction);
092            thirdPosition = this.getXPosition(direction, startClone, startBlock);
093        }
094        if (this.getZLength(direction) > mainDirection) {
095            this.mainFace = this.getZFace(direction);
096            mainDirection = this.getZLength(direction);
097            mainPosition = this.getZPosition(direction, startClone, startBlock);
098
099            this.secondFace = this.getXFace(direction);
100            secondDirection = this.getXLength(direction);
101            secondPosition = this.getXPosition(direction, startClone, startBlock);
102
103            this.thirdFace = this.getYFace(direction);
104            thirdDirection = this.getYLength(direction);
105            thirdPosition = this.getYPosition(direction, startClone, startBlock);
106        }
107
108        double d = mainPosition / mainDirection;
109        double secondd = secondPosition - secondDirection * d;
110        double thirdd = thirdPosition - thirdDirection * d;
111
112        this.secondError = (int) Math.floor(secondd * gridSize);
113        this.secondStep = (int) Math.round(secondDirection / mainDirection * gridSize);
114        this.thirdError = (int) Math.floor(thirdd * gridSize);
115        this.thirdStep = (int) Math.round(thirdDirection / mainDirection * gridSize);
116
117        if (this.secondError + this.secondStep <= 0) {
118            this.secondError = -this.secondStep + 1;
119        }
120
121        if (this.thirdError + this.thirdStep <= 0) {
122            this.thirdError = -this.thirdStep + 1;
123        }
124
125        Block lastBlock = startBlock.getSide(this.mainFace.getOpposite());
126
127        if (this.secondError < 0) {
128            this.secondError += gridSize;
129            lastBlock = lastBlock.getSide(this.secondFace.getOpposite());
130        }
131
132        if (this.thirdError < 0) {
133            this.thirdError += gridSize;
134            lastBlock = lastBlock.getSide(this.thirdFace.getOpposite());
135        }
136
137        this.secondError -= gridSize;
138        this.thirdError -= gridSize;
139
140        this.blockQueue[0] = lastBlock;
141
142        this.currentBlock = -1;
143
144        this.scan();
145
146        boolean startBlockFound = false;
147
148        for (int cnt = this.currentBlock; cnt >= 0; --cnt) {
149            if (this.blockEquals(this.blockQueue[cnt], startBlock)) {
150                this.currentBlock = cnt;
151                startBlockFound = true;
152                break;
153            }
154        }
155
156        if (!startBlockFound) {
157            throw new IllegalStateException("Start block missed in BlockIterator");
158        }
159
160        this.maxDistanceInt = (int) Math.round(maxDistance / (Math.sqrt(mainDirection * mainDirection + secondDirection * secondDirection + thirdDirection * thirdDirection) / mainDirection));
161    }
162
163    private boolean blockEquals(Block a, Block b) {
164        return a.x == b.x && a.y == b.y && a.z == b.z;
165    }
166
167    private BlockFace getXFace(Vector3 direction) {
168        return ((direction.x) > 0) ? BlockFace.EAST : BlockFace.WEST;
169    }
170
171    private BlockFace getYFace(Vector3 direction) {
172        return ((direction.y) > 0) ? BlockFace.UP : BlockFace.DOWN;
173    }
174
175    private BlockFace getZFace(Vector3 direction) {
176        return ((direction.z) > 0) ? BlockFace.SOUTH : BlockFace.NORTH;
177    }
178
179    private double getXLength(Vector3 direction) {
180        return Math.abs(direction.x);
181    }
182
183    private double getYLength(Vector3 direction) {
184        return Math.abs(direction.y);
185    }
186
187    private double getZLength(Vector3 direction) {
188        return Math.abs(direction.z);
189    }
190
191    private double getPosition(double direction, double position, double blockPosition) {
192        return direction > 0 ? (position - blockPosition) : (blockPosition + 1 - position);
193    }
194
195    private double getXPosition(Vector3 direction, Vector3 position, Block block) {
196        return this.getPosition(direction.x, position.x, block.x);
197    }
198
199    private double getYPosition(Vector3 direction, Vector3 position, Block block) {
200        return this.getPosition(direction.y, position.y, block.y);
201    }
202
203    private double getZPosition(Vector3 direction, Vector3 position, Block block) {
204        return this.getPosition(direction.z, position.z, block.z);
205    }
206
207    @Override
208    public Block next() {
209        this.scan();
210
211        if (this.currentBlock <= -1) {
212            throw new IndexOutOfBoundsException();
213        } else {
214            this.currentBlockObject = this.blockQueue[this.currentBlock--];
215        }
216        return this.currentBlockObject;
217    }
218
219    @Override
220    public boolean hasNext() {
221        this.scan();
222        return this.currentBlock != -1;
223    }
224
225    private void scan() {
226        if (this.currentBlock >= 0) {
227            return;
228        }
229
230        if (this.maxDistance != 0 && this.currentDistance > this.maxDistanceInt) {
231            this.end = true;
232            return;
233        }
234
235        if (this.end) {
236            return;
237        }
238
239        ++this.currentDistance;
240
241        this.secondError += this.secondStep;
242        this.thirdError += this.thirdStep;
243
244        if (this.secondError > 0 && this.thirdError > 0) {
245            this.blockQueue[2] = this.blockQueue[0].getSide(this.mainFace);
246
247            if ((this.secondStep * this.thirdError) < (this.thirdStep * this.secondError)) {
248                this.blockQueue[1] = this.blockQueue[2].getSide(this.secondFace);
249                this.blockQueue[0] = this.blockQueue[1].getSide(this.thirdFace);
250            } else {
251                this.blockQueue[1] = this.blockQueue[2].getSide(this.thirdFace);
252                this.blockQueue[0] = this.blockQueue[1].getSide(this.secondFace);
253            }
254
255            this.thirdError -= gridSize;
256            this.secondError -= gridSize;
257            this.currentBlock = 2;
258        } else if (this.secondError > 0) {
259            this.blockQueue[1] = this.blockQueue[0].getSide(this.mainFace);
260            this.blockQueue[0] = this.blockQueue[1].getSide(this.secondFace);
261            this.secondError -= gridSize;
262            this.currentBlock = 1;
263        } else if (this.thirdError > 0) {
264            this.blockQueue[1] = this.blockQueue[0].getSide(this.mainFace);
265            this.blockQueue[0] = this.blockQueue[1].getSide(this.thirdFace);
266            this.thirdError -= gridSize;
267            this.currentBlock = 1;
268        } else {
269            this.blockQueue[0] = this.blockQueue[0].getSide(this.mainFace);
270            this.currentBlock = 0;
271        }
272    }
273}
