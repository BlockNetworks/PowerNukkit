202    private void decodeSkinData() {
203        JsonObject skinToken = decodeToken(new String(bs.get(bs.getLInt())));
204        if (skinToken == null) return;
205        if (skinToken.has("ClientRandomId")) this.clientId = skinToken.get("ClientRandomId").getAsLong();
206        if (skinToken.has("ServerAddress")) this.serverAddress = skinToken.get("ServerAddress").getAsString();
207        if (skinToken.has("DeviceModel")) this.deviceModel = skinToken.get("DeviceModel").getAsString();
208        if (skinToken.has("DeviceOS")) this.deviceOS = skinToken.get("DeviceOS").getAsInt();
209        if (skinToken.has("DeviceId")) this.deviceId = skinToken.get("DeviceId").getAsString();
210        if (skinToken.has("GameVersion")) this.gameVersion = skinToken.get("GameVersion").getAsString();
211        if (skinToken.has("GuiScale")) this.guiScale = skinToken.get("GuiScale").getAsInt();
212        if (skinToken.has("LanguageCode")) this.languageCode = skinToken.get("LanguageCode").getAsString();
213        if (skinToken.has("CurrentInputMode")) this.currentInputMode = skinToken.get("CurrentInputMode").getAsInt();
214        if (skinToken.has("DefaultInputMode")) this.defaultInputMode = skinToken.get("DefaultInputMode").getAsInt();
215        if (skinToken.has("UIProfile")) this.UIProfile = skinToken.get("UIProfile").getAsInt();
216        if (skinToken.has("CapeData")) this.capeData = skinToken.get("CapeData").getAsString();
217
218        this.rawData = skinToken;
219    }

221    private JsonObject decodeToken(String token) {
222        String[] base = token.split("\\.");
223        if (base.length < 2) return null;
224        String json = new String(Base64.getDecoder().decode(base[1]), StandardCharsets.UTF_8);
225        //Server.getInstance().getLogger().debug(json);
226        return new Gson().fromJson(json, JsonObject.class);
227    }
228
229    private void decodeChainData() {
230        Map<String, List<String>> map = new Gson().fromJson(new String(bs.get(bs.getLInt()), StandardCharsets.UTF_8),
231                new TypeToken<Map<String, List<String>>>() {
232                }.getType());
233        if (map.isEmpty() || !map.containsKey("chain") || map.get("chain").isEmpty()) return;
234        List<String> chains = map.get("chain");
235
236        // Validate keys
237        try {
238            xboxAuthed = verifyChain(chains);
239        } catch (Exception e) {
240            xboxAuthed = false;
241        }
242
243        for (String c : chains) {
244            JsonObject chainMap = decodeToken(c);
245            if (chainMap == null) continue;
246            if (chainMap.has("extraData")) {
247                JsonObject extra = chainMap.get("extraData").getAsJsonObject();
248                if (extra.has("displayName")) this.username = extra.get("displayName").getAsString();
249                if (extra.has("identity")) this.clientUUID = UUID.fromString(extra.get("identity").getAsString());
250                if (extra.has("XUID")) this.xuid = extra.get("XUID").getAsString();
251            }
252            if (chainMap.has("identityPublicKey"))
253                this.identityPublicKey = chainMap.get("identityPublicKey").getAsString();
254        }
255
256        if (!xboxAuthed) {
257            xuid = null;
258        }
259    }
260
261    private boolean verifyChain(List<String> chains) throws Exception {
262
263        PublicKey lastKey = null;
264        boolean mojangKeyVerified = false;
265        for (String chain: chains) {
266            JWSObject jws = JWSObject.parse(chain);
267
268            if (!mojangKeyVerified) {
269                // First chain should be signed using Mojang's private key. We'd be in big trouble if it leaked...
270                mojangKeyVerified = verify(MOJANG_PUBLIC_KEY, jws);
271            }
272
273            if (lastKey != null) {
274                if (!verify(lastKey, jws)) {
275                    throw new JOSEException("Unable to verify key in chain.");
276                }
277            }
278
279            JSONObject payload = jws.getPayload().toJSONObject();
280            String base64key = payload.getAsString("identityPublicKey");
281            if (base64key == null) {
282                throw new RuntimeException("No key found");
283            }
284            lastKey = generateKey(base64key);
285        }
286        return mojangKeyVerified;
287    }
288
289    private boolean verify(PublicKey key, JWSObject object) throws JOSEException {
290        JWSVerifier verifier = new DefaultJWSVerifierFactory().createJWSVerifier(object.getHeader(), key);
291        return object.verify(verifier);
292    }
