001package cn.nukkit.metrics;
002
003import cn.nukkit.Server;
004import cn.nukkit.utils.Config;
005
006import java.io.BufferedWriter;
007import java.io.File;
008import java.io.FileWriter;
009import java.io.IOException;
010import java.util.HashMap;
011import java.util.Map;
012import java.util.UUID;
013import java.util.regex.Matcher;
014import java.util.regex.Pattern;
015
016public class NukkitMetrics {
017    private static boolean metricsStarted = false;
018
019    private final Server server;
020
021    private boolean enabled;
022    private String serverUUID;
023    private boolean logFailedRequests;
024
025    public NukkitMetrics(Server server) {
026        this.server = server;
027
028        if (metricsStarted) {
029            return;
030        }
031
032        try {
033            this.loadConfig();
034        } catch (IOException e) {
035            e.printStackTrace();
036        }
037
038        if (enabled) {
039            Metrics metrics = new Metrics("Nukkit", serverUUID, logFailedRequests, server.getLogger());
040
041            metrics.addCustomChart(new Metrics.SingleLineChart("players", () -> server.getOnlinePlayers().size()));
042            metrics.addCustomChart(new Metrics.SimplePie("codename", server::getCodename));
043            metrics.addCustomChart(new Metrics.SimplePie("minecraft_version", server::getVersion));
044            metrics.addCustomChart(new Metrics.SimplePie("nukkit_version", server::getNukkitVersion));
045            metrics.addCustomChart(new Metrics.SimplePie("xbox_auth", () -> server.getPropertyBoolean("xbox-auth") ? "Required" : "Not required"));
046
047            metrics.addCustomChart(new Metrics.AdvancedPie("player_platform", () -> {
048                Map<String, Integer> valueMap = new HashMap<>();
049
050                server.getOnlinePlayers().forEach((uuid, player) -> {
051                    String deviceOS = mapDeviceOSToString(player.getLoginChainData().getDeviceOS());
052                    if (!valueMap.containsKey(deviceOS)) {
053                        valueMap.put(deviceOS, 1);
054                    } else {
055                        valueMap.put(deviceOS, valueMap.get(deviceOS) + 1);
056                    }
057                });
058                return valueMap;
059            }));
060
061            metrics.addCustomChart(new Metrics.AdvancedPie("player_game_version", () -> {
062                Map<String, Integer> valueMap = new HashMap<>();
063
064                server.getOnlinePlayers().forEach((uuid, player) -> {
065                    String gameVersion = player.getLoginChainData().getGameVersion();
066                    if (!valueMap.containsKey(gameVersion)) {
067                        valueMap.put(gameVersion, 1);
068                    } else {
069                        valueMap.put(gameVersion, valueMap.get(gameVersion) + 1);
070                    }
071                });
072                return valueMap;
073            }));
074
075            // The following code can be attributed to the PaperMC project
076            // https://github.com/PaperMC/Paper/blob/master/Spigot-Server-Patches/0005-Paper-Metrics.patch#L614
077            metrics.addCustomChart(new Metrics.DrilldownPie("java_version", () -> {
078                Map<String, Map<String, Integer>> map = new HashMap<>();
079                String javaVersion = System.getProperty("java.version");
080                Map<String, Integer> entry = new HashMap<>();
081                entry.put(javaVersion, 1);
082
083                // http://openjdk.java.net/jeps/223
084                // Java decided to change their versioning scheme and in doing so modified the java.version system
085                // property to return $major[.$minor][.$secuity][-ea], as opposed to 1.$major.0_$identifier
086                // we can handle pre-9 by checking if the "major" is equal to "1", otherwise, 9+
087                String majorVersion = javaVersion.split("\\.")[0];
088                String release;
089
090                int indexOf = javaVersion.lastIndexOf('.');
091
092                if (majorVersion.equals("1")) {
093                    release = "Java " + javaVersion.substring(0, indexOf);
094                } else {
095                    // of course, it really wouldn't be all that simple if they didn't add a quirk, now would it
096                    // valid strings for the major may potentially include values such as -ea to deannotate a pre release
097                    Matcher versionMatcher = Pattern.compile("\\d+").matcher(majorVersion);
098                    if (versionMatcher.find()) {
099                        majorVersion = versionMatcher.group(0);
100                    }
101                    release = "Java " + majorVersion;
102                }
103                map.put(release, entry);
104                return map;
105            }));
106
107            metricsStarted = true;
108        }
109    }
110
111    /**
112     * Loads the bStats configuration.
113     */
114    private void loadConfig() throws IOException {
115        File bStatsFolder = new File(server.getPluginPath(), "bStats");
116
117        if (!bStatsFolder.exists() && !bStatsFolder.mkdirs()) {
118            server.getLogger().warning("Failed to create bStats metrics directory");
119            return;
120        }
121
122        File configFile = new File(bStatsFolder, "config.yml");
123        if (!configFile.exists()) {
124            writeFile(configFile,
125                    "# bStats collects some data for plugin authors like how many servers are using their plugins.",
126                    "# To honor their work, you should not disable it.",
127                    "# This has nearly no effect on the server performance!",
128                    "# Check out https://bStats.org/ to learn more :)",
129                    "enabled: true",
130                    "serverUuid: \"" + UUID.randomUUID().toString() + "\"",
131                    "logFailedRequests: false");
132        }
133
134        Config config = new Config(configFile, Config.YAML);
135
136        // Load configuration
137        this.enabled = config.getBoolean("enabled", true);
138        this.serverUUID = config.getString("serverUuid");
139        this.logFailedRequests = config.getBoolean("logFailedRequests", false);
140    }
141
142    private void writeFile(File file, String... lines) throws IOException {
143        try (BufferedWriter writer = new BufferedWriter(new FileWriter(file))) {
144            for (String line : lines) {
145                writer.write(line);
146                writer.newLine();
147            }
148        }
149    }
150
151    private String mapDeviceOSToString(int os) {
152        switch (os) {
153            case 1: return "Android";
154            case 2: return "iOS";
155            case 3: return "macOS";
156            case 4: return "FireOS";
157            case 5: return "Gear VR";
158            case 6: return "Hololens";
159            case 7: return "Windows 10";
160            case 8: return "Windows";
161            case 9: return "Dedicated";
162            case 10: return "PS4";
163            case 11: return "Switch";
164            case 12: return "Switch";
165            case 13: return "Xbox One";
166            case 14: return "Windows Phone";
167        }
168        return "Unknown";
169    }
170}
