001package cn.nukkit.blockentity;
002
003import cn.nukkit.Player;
004import cn.nukkit.block.Block;
005import cn.nukkit.block.BlockID;
006import cn.nukkit.inventory.BeaconInventory;
007import cn.nukkit.item.ItemBlock;
008import cn.nukkit.level.format.FullChunk;
009import cn.nukkit.nbt.tag.CompoundTag;
010import cn.nukkit.network.protocol.LevelSoundEventPacket;
011import cn.nukkit.potion.Effect;
012
013import java.util.Map;
014
015/**
016 * author: Rover656
017 */
018public class BlockEntityBeacon extends BlockEntitySpawnable {
019
020    public BlockEntityBeacon(FullChunk chunk, CompoundTag nbt) {
021        super(chunk, nbt);
022    }
023
024    @Override
025    protected void initBlockEntity() {
026        if (!namedTag.contains("Lock")) {
027            namedTag.putString("Lock", "");
028        }
029
030        if (!namedTag.contains("Levels")) {
031            namedTag.putInt("Levels", 0);
032        }
033
034        if (!namedTag.contains("Primary")) {
035            namedTag.putInt("Primary", 0);
036        }
037
038        if (!namedTag.contains("Secondary")) {
039            namedTag.putInt("Secondary", 0);
040        }
041
042        scheduleUpdate();
043
044        super.initBlockEntity();
045    }
046
047    @Override
048    public boolean isBlockEntityValid() {
049        int blockID = getBlock().getId();
050        return blockID == Block.BEACON;
051    }
052
053    @Override
054    public CompoundTag getSpawnCompound() {
055        return new CompoundTag()
056                .putString("id", BlockEntity.BEACON)
057                .putInt("x", (int) this.x)
058                .putInt("y", (int) this.y)
059                .putInt("z", (int) this.z)
060                .putString("Lock", this.namedTag.getString("Lock"))
061                .putInt("Levels", this.namedTag.getInt("Levels"))
062                .putInt("Primary", this.namedTag.getInt("Primary"))
063                .putInt("Secondary", this.namedTag.getInt("Secondary"));
064    }
065
066    private long currentTick = 0;
067
068    @Override
069    public boolean onUpdate() {
070        //Only apply effects every 4 secs
071        if (currentTick++ % 80 != 0) {
072            return true;
073        }
074
075        int oldPowerLevel = this.getPowerLevel();
076        //Get the power level based on the pyramid
077        setPowerLevel(calculatePowerLevel());
078        int newPowerLevel = this.getPowerLevel();
079
080        //Skip beacons that do not have a pyramid or sky access
081        if (newPowerLevel < 1 || !hasSkyAccess()) {
082            if (oldPowerLevel > 0) {
083                this.getLevel().addLevelSoundEvent(this, LevelSoundEventPacket.SOUND_BEACON_DEACTIVATE);
084            }
085            return true;
086        } else if (oldPowerLevel < 1) {
087            this.getLevel().addLevelSoundEvent(this, LevelSoundEventPacket.SOUND_BEACON_ACTIVATE);
088        } else {
089            this.getLevel().addLevelSoundEvent(this, LevelSoundEventPacket.SOUND_BEACON_AMBIENT);
090        }
091
092        //Get all players in game
093        Map<Long, Player> players = this.level.getPlayers();
094
095        //Calculate vars for beacon power
096        int range = 10 + getPowerLevel() * 10;
097        int duration = 9 + getPowerLevel() * 2;
098
099        for(Map.Entry<Long, Player> entry : players.entrySet()) {
100            Player p = entry.getValue();
101
102            //If the player is in range
103            if (p.distance(this) < range) {
104                Effect e;
105
106                if (getPrimaryPower() != 0) {
107                    //Apply the primary power
108                    e = Effect.getEffect(getPrimaryPower());
109
110                    //Set duration
111                    e.setDuration(duration * 20);
112
113                    //If secondary is selected as the primary too, apply 2 amplification
114                    if (getSecondaryPower() == getPrimaryPower()) {
115                        e.setAmplifier(1);
116                    } else {
117                        e.setAmplifier(0);
118                    }
119
120                    //Hide particles
121                    e.setVisible(false);
122
123                    //Add the effect
124                    p.addEffect(e);
125                }
126
127                //If we have a secondary power as regen, apply it
128                if (getSecondaryPower() == Effect.REGENERATION) {
129                    //Get the regen effect
130                    e = Effect.getEffect(Effect.REGENERATION);
131
132                    //Set duration
133                    e.setDuration(duration * 20);
134
135                    //Regen I
136                    e.setAmplifier(1);
137
138                    //Hide particles
139                    e.setVisible(false);
140
141                    //Add effect
142                    p.addEffect(e);
143                }
144            }
145        }
146
147        return true;
148    }
149
150    private static final int POWER_LEVEL_MAX = 4;
151
152    private boolean hasSkyAccess() {
153        int tileX = getFloorX();
154        int tileY = getFloorY();
155        int tileZ = getFloorZ();
156
157        //Check every block from our y coord to the top of the world
158        for (int y = tileY + 1; y <= 255; y++) {
159            int testBlockId = level.getBlockIdAt(tileX, y, tileZ);
160            if (!Block.transparent[testBlockId]) {
161                //There is no sky access
162                return false;
163            }
164        }
165
166        return true;
167    }
168
169    private int calculatePowerLevel() {
170        int tileX = getFloorX();
171        int tileY = getFloorY();
172        int tileZ = getFloorZ();
173
174        //The power level that we're testing for
175        for (int powerLevel = 1; powerLevel <= POWER_LEVEL_MAX; powerLevel++) {
176            int queryY = tileY - powerLevel; //Layer below the beacon block
177
178            for (int queryX = tileX - powerLevel; queryX <= tileX + powerLevel; queryX++) {
179                for (int queryZ = tileZ - powerLevel; queryZ <= tileZ + powerLevel; queryZ++) {
180
181                    int testBlockId = level.getBlockIdAt(queryX, queryY, queryZ);
182                    if (
183                            testBlockId != Block.IRON_BLOCK &&
184                                    testBlockId != Block.GOLD_BLOCK &&
185                                    testBlockId != Block.EMERALD_BLOCK &&
186                                    testBlockId != Block.DIAMOND_BLOCK
187                            ) {
188                        return powerLevel - 1;
189                    }
190
191                }
192            }
193        }
194
195        return POWER_LEVEL_MAX;
196    }
197
198    public int getPowerLevel() {
199        return namedTag.getInt("Level");
200    }
201
202    public void setPowerLevel(int level) {
203        int currentLevel = getPowerLevel();
204        if (level != currentLevel) {
205            namedTag.putInt("Level", level);
206            setDirty();
207            this.spawnToAll();
208        }
209    }
210
211    public int getPrimaryPower() {
212        return namedTag.getInt("Primary");
213    }
214
215    public void setPrimaryPower(int power) {
216        int currentPower = getPrimaryPower();
217        if (power != currentPower) {
218            namedTag.putInt("Primary", power);
219            setDirty();
220            this.spawnToAll();
221        }
222    }
223
224    public int getSecondaryPower() {
225        return namedTag.getInt("Secondary");
226    }
227
228    public void setSecondaryPower(int power) {
229        int currentPower = getSecondaryPower();
230        if (power != currentPower) {
231            namedTag.putInt("Secondary", power);
232            setDirty();
233            this.spawnToAll();
234        }
235    }
236
237    @Override
238    public boolean updateCompoundTag(CompoundTag nbt, Player player) {
239        if (!nbt.getString("id").equals(BlockEntity.BEACON)) {
240            return false;
241        }
242
243        this.setPrimaryPower(nbt.getInt("primary"));
244        this.setSecondaryPower(nbt.getInt("secondary"));
245
246        this.getLevel().addLevelSoundEvent(this, LevelSoundEventPacket.SOUND_BEACON_POWER);
247
248        BeaconInventory inv = (BeaconInventory)player.getWindowById(Player.BEACON_WINDOW_ID);
249
250        inv.setItem(0, new ItemBlock(Block.get(BlockID.AIR), 0, 0));
251        return true;
252    }
253}
