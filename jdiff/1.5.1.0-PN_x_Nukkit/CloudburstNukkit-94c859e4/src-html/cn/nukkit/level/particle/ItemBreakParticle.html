001package cn.nukkit.level.particle;
002
003import cn.nukkit.item.Item;
004import cn.nukkit.math.Vector3;
005import cn.nukkit.network.protocol.DataPacket;
006import cn.nukkit.network.protocol.LevelEventPacket;
007
008/**
009 * Created on 2015/11/21 by xtypr.
010 * Package cn.nukkit.level.particle in project Nukkit .
011 */
012public class ItemBreakParticle extends Particle {
013
014    private final int data;
015
016    public ItemBreakParticle(Vector3 pos, Item item) {
017        super(pos.x, pos.y, pos.z);
018        this.data = (item.getNetworkId() << 16 | item.getDamage());
019    }
020
021    @Override
022    public DataPacket[] encode() {
023        LevelEventPacket packet = new LevelEventPacket();
024        packet.evid = (short) (LevelEventPacket.EVENT_ADD_PARTICLE_MASK | Particle.TYPE_ITEM_BREAK);
025        packet.x = (float) this.x;
026        packet.y = (float) this.y;
027        packet.z = (float) this.z;
028        packet.data = this.data;
029        return new DataPacket[]{packet};
030    }
031}
