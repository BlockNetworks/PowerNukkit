001package cn.nukkit.level;
002
003import cn.nukkit.block.Block;
004import cn.nukkit.level.format.generic.BaseFullChunk;
005
006import java.util.ArrayList;
007import java.util.List;
008import java.util.Optional;
009
010public class ListChunkManager implements ChunkManager {
011
012    private ChunkManager parent;
013    private List<Block> blocks;
014
015    public ListChunkManager(ChunkManager parent) {
016        this.parent = parent;
017        this.blocks = new ArrayList<>();
018    }
019
020    @Override
021    public int getBlockIdAt(int x, int y, int z) {
022        Optional<Block> optionalBlock = this.blocks.stream().filter(block -> block.getFloorX() == x && block.getFloorY() == y && block.getFloorZ() == z).findAny();
023        return optionalBlock.map(Block::getId).orElseGet(() -> this.parent.getBlockIdAt(x, y, z));
024    }
025
026    @Override
027    public void setBlockFullIdAt(int x, int y, int z, int fullId) {
028        this.blocks.removeIf(block -> block.getFloorX() == x && block.getFloorY() == y && block.getFloorZ() == z);
029        this.blocks.add(Block.get(fullId, null, x, y, z));
030    }
031
032    @Override
033    public void setBlockIdAt(int x, int y, int z, int id) {
034        Optional<Block> optionalBlock = this.blocks.stream().filter(block -> block.getFloorX() == x && block.getFloorY() == y && block.getFloorZ() == z).findAny();
035        Block block = optionalBlock.orElse(Block.get(this.getBlockIdAt(x, y, z), this.getBlockDataAt(x, y, z), new Position(x, y, z)));
036        this.blocks.remove(block);
037        this.blocks.add(Block.get(this.getBlockIdAt(x, y, z), this.getBlockDataAt(x, y, z), new Position(x, y, z)));
038    }
039
040    @Override
041    public void setBlockAt(int x, int y, int z, int id, int data) {
042        this.blocks.removeIf(block -> block.getFloorX() == x && block.getFloorY() == y && block.getFloorZ() == z);
043        this.blocks.add(Block.get(id, data, new Position(x, y, z)));
044    }
045
046    @Override
047    public int getBlockDataAt(int x, int y, int z) {
048        Optional<Block> optionalBlock = this.blocks.stream().filter(block -> block.getFloorX() == x && block.getFloorY() == y && block.getFloorZ() == z).findAny();
049        return optionalBlock.map(Block::getDamage).orElseGet(() -> this.parent.getBlockDataAt(x, y, z));
050    }
051
052    @Override
053    public void setBlockDataAt(int x, int y, int z, int data) {
054        Optional<Block> optionalBlock = this.blocks.stream().filter(block -> block.getFloorX() == x && block.getFloorY() == y && block.getFloorZ() == z).findAny();
055        Block block = optionalBlock.orElse(Block.get(this.getBlockIdAt(x, y, z), this.getBlockDataAt(x, y, z), new Position(x, y, z)));
056        this.blocks.remove(block);
057        block.setDamage(data);
058        this.blocks.add(block);
059    }
060
061    @Override
062    public BaseFullChunk getChunk(int chunkX, int chunkZ) {
063        return this.parent.getChunk(chunkX, chunkZ);
064    }
065
066    @Override
067    public void setChunk(int chunkX, int chunkZ) {
068        this.parent.setChunk(chunkX, chunkZ);
069    }
070
071    @Override
072    public void setChunk(int chunkX, int chunkZ, BaseFullChunk chunk) {
073        this.parent.setChunk(chunkX, chunkZ, chunk);
074    }
075
076    @Override
077    public long getSeed() {
078        return this.parent.getSeed();
079    }
080
081    public List<Block> getBlocks() {
082        return this.blocks;
083    }
084
085}
