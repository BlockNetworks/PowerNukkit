001package cn.nukkit.command.defaults;
002
003import cn.nukkit.command.Command;
004import cn.nukkit.command.CommandSender;
005import cn.nukkit.command.ConsoleCommandSender;
006import cn.nukkit.command.data.CommandParamType;
007import cn.nukkit.command.data.CommandParameter;
008import cn.nukkit.lang.TranslationContainer;
009import cn.nukkit.utils.TextFormat;
010
011import java.util.Map;
012import java.util.TreeMap;
013
014/**
015 * author: MagicDroidX
016 * Nukkit Project
017 */
018public class HelpCommand extends VanillaCommand {
019
020    public HelpCommand(String name) {
021        super(name, "%nukkit.command.help.description", "%commands.help.usage", new String[]{"?"});
022        this.setPermission("nukkit.command.help");
023        this.commandParameters.clear();
024        this.commandParameters.put("default", new CommandParameter[]{
025                new CommandParameter("page", CommandParamType.INT, true)
026        });
027    }
028
029    @Override
030    public boolean execute(CommandSender sender, String commandLabel, String[] args) {
031        if (!this.testPermission(sender)) {
032            return true;
033        }
034        StringBuilder command = new StringBuilder();
035        int pageNumber = 1;
036        int pageHeight = 5;
037        if (args.length != 0) {
038            try {
039                pageNumber = Integer.parseInt(args[args.length - 1]);
040                if (pageNumber <= 0) {
041                    pageNumber = 1;
042                }
043
044                String[] newargs = new String[args.length - 1];
045                System.arraycopy(args, 0, newargs, 0, newargs.length);
046                args = newargs;
047                /*if (args.length > 1) {
048                    args = Arrays.copyOfRange(args, 0, args.length - 2);
049                } else {
050                    args = new String[0];
051                }*/
052                for (String arg : args) {
053                    if (!command.toString().equals("")) {
054                        command.append(" ");
055                    }
056                    command.append(arg);
057                }
058            } catch (NumberFormatException e) {
059                pageNumber = 1;
060                for (String arg : args) {
061                    if (!command.toString().equals("")) {
062                        command.append(" ");
063                    }
064                    command.append(arg);
065                }
066            }
067        }
068
069        if (sender instanceof ConsoleCommandSender) {
070            pageHeight = Integer.MAX_VALUE;
071        }
072
073        if (command.toString().equals("")) {
074            Map<String, Command> commands = new TreeMap<>();
075            for (Command cmd : sender.getServer().getCommandMap().getCommands().values()) {
076                if (cmd.testPermissionSilent(sender)) {
077                    commands.put(cmd.getName(), cmd);
078                }
079            }
080            int totalPage = commands.size() % pageHeight == 0 ? commands.size() / pageHeight : commands.size() / pageHeight + 1;
081            pageNumber = Math.min(pageNumber, totalPage);
082            if (pageNumber < 1) {
083                pageNumber = 1;
084            }
085
086            sender.sendMessage(new TranslationContainer("commands.help.header", String.valueOf(pageNumber), String.valueOf(totalPage)));
087            int i = 1;
088            for (Command command1 : commands.values()) {
089                if (i >= (pageNumber - 1) * pageHeight + 1 && i <= Math.min(commands.size(), pageNumber * pageHeight)) {
090                    sender.sendMessage(TextFormat.DARK_GREEN + "/" + command1.getName() + ": " + TextFormat.WHITE + command1.getDescription());
091                }
092                i++;
093            }
094
095            return true;
096        } else {
097            Command cmd = sender.getServer().getCommandMap().getCommand(command.toString().toLowerCase());
098            if (cmd != null) {
099                if (cmd.testPermissionSilent(sender)) {
100                    String message = TextFormat.YELLOW + "--------- " + TextFormat.WHITE + " Help: /" + cmd.getName() + TextFormat.YELLOW + " ---------\n";
101                    message += TextFormat.GOLD + "Description: " + TextFormat.WHITE + cmd.getDescription() + "\n";
102                    StringBuilder usage = new StringBuilder();
103                    String[] usages = cmd.getUsage().split("\n");
104                    for (String u : usages) {
105                        if (!usage.toString().equals("")) {
106                            usage.append("\n" + TextFormat.WHITE);
107                        }
108                        usage.append(u);
109                    }
110                    message += TextFormat.GOLD + "Usage: " + TextFormat.WHITE + usage + "\n";
111                    sender.sendMessage(message);
112                    return true;
113                }
114            }
115
116            sender.sendMessage(TextFormat.RED + "No help for " + command.toString().toLowerCase());
117            return true;
118        }
119    }
120}
