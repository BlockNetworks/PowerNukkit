001package cn.nukkit.inventory.transaction.action;
002
003import cn.nukkit.Player;
004import cn.nukkit.inventory.transaction.InventoryTransaction;
005import cn.nukkit.item.Item;
006
007/**
008 * @author CreeperFace
009 */
010public abstract class InventoryAction {
011
012
013    private long creationTime;
014
015    protected Item sourceItem;
016
017    protected Item targetItem;
018
019    public InventoryAction(Item sourceItem, Item targetItem) {
020        this.sourceItem = sourceItem;
021        this.targetItem = targetItem;
022
023        this.creationTime = System.currentTimeMillis();
024    }
025
026    public long getCreationTime() {
027        return creationTime;
028    }
029
030    /**
031     * Returns the item that was present before the action took place.
032     *
033     * @return source item
034     */
035    public Item getSourceItem() {
036        return sourceItem.clone();
037    }
038
039    /**
040     * Returns the item that the action attempted to replace the source item with.
041     *
042     * @return target item
043     */
044    public Item getTargetItem() {
045        return targetItem.clone();
046    }
047
048    /**
049     * Called by inventory transactions before any actions are processed. If this returns false, the transaction will
050     * be cancelled.
051     *
052     * @param source player
053     * @return cancelled
054     */
055    public boolean onPreExecute(Player source) {
056        return true;
057    }
058
059    /**
060     * Returns whether this action is currently valid. This should perform any necessary sanity checks.
061     *
062     * @param source player
063     * @return valid
064     */
065    abstract public boolean isValid(Player source);
066
067    /**
068     * Called when the action is added to the specified InventoryTransaction.
069     *
070     * @param transaction to add
071     */
072    public void onAddToTransaction(InventoryTransaction transaction) {
073
074    }
075
076    /**
077     * Performs actions needed to complete the inventory-action server-side. Returns if it was successful. Will return
078     * false if plugins cancelled events. This will only be called if the transaction which it is part of is considered
079     * valid.
080     *
081     * @param source player
082     * @return successfully executed
083     */
084    abstract public boolean execute(Player source);
085
086    /**
087     * Performs additional actions when this inventory-action completed successfully.
088     *
089     * @param source player
090     */
091    abstract public void onExecuteSuccess(Player source);
092
093    /**
094     * Performs additional actions when this inventory-action did not complete successfully.
095     *
096     * @param source player
097     */
098    abstract public void onExecuteFail(Player source);
099}
