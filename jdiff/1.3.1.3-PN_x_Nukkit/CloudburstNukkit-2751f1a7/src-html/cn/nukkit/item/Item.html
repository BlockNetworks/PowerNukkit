390    public static Item get(int id, Integer meta, int count, byte[] tags) {
391        try {
392            Class c = list[id];
393            Item item;
394
395            if (c == null) {
396                item = new Item(id, meta, count);
397            } else if (id < 256) {
398                if (meta >= 0) {
399                    item = new ItemBlock(Block.get(id, meta), meta, count);
400                } else {
401                    item = new ItemBlock(Block.get(id), meta, count);
402                }
403            } else {
404                item = ((Item) c.getConstructor(Integer.class, int.class).newInstance(meta, count));
405            }
406
407            if (tags.length != 0) {
408                item.setCompoundTag(tags);
409            }
410
411            return item;
412        } catch (Exception e) {
413            return new Item(id, meta, count).setCompoundTag(tags);
414        }
415    }

417    public static Item fromString(String str) {
418        String[] b = str.trim().replace(' ', '_').replace("minecraft:", "").split(":");
419
420        int id = 0;
421        int meta = 0;
422
423        Pattern integerPattern = Pattern.compile("^[1-9]\\d*$");
424        if (integerPattern.matcher(b[0]).matches()) {
425            id = Integer.parseInt(b[0]);
426        } else {
427            try {
428                id = Item.class.getField(b[0].toUpperCase()).getInt(null);
429            } catch (Exception ignore) {
430            }
431        }
432
433        id = id & 0xFFFF;
434        if (b.length != 1) meta = Integer.parseInt(b[1]) & 0xFFFF;
435
436        return get(id, meta);
437    }

439    public static Item fromJson(Map<String, Object> data) {
440        String nbt = (String) data.get("nbt_b64");
441        byte[] nbtBytes;
442        if (nbt != null) {
443            nbtBytes = Base64.getDecoder().decode(nbt);
444        } else { // Support old format for backwards compat
445            nbt = (String) data.getOrDefault("nbt_hex", null);
446            if (nbt == null) {
447                nbtBytes = new byte[0];
448            } else {
449                nbtBytes = Utils.parseHexBinary(nbt);
450            }
451        }
452
453        return get(Utils.toInt(data.get("id")), Utils.toInt(data.getOrDefault("damage", 0)), Utils.toInt(data.getOrDefault("count", 1)), nbtBytes);
454    }

508    public Item setCustomBlockData(CompoundTag compoundTag) {
509        CompoundTag tags = compoundTag.copy();
510        tags.setName("BlockEntityTag");
511
512        CompoundTag tag;
513        if (!this.hasCompoundTag()) {
514            tag = new CompoundTag();
515        } else {
516            tag = this.getNamedTag();
517        }
518
519        tag.putCompound("BlockEntityTag", tags);
520        this.setNamedTag(tag);
521
522        return this;
523    }

566        for (CompoundTag entry : this.getNamedTag().getList("ench", CompoundTag.class).getAll()) {
567            if (entry.getShort("id") == id) {
568                Enchantment e = Enchantment.getEnchantment(entry.getShort("id"));
569                if (e != null) {
570                    e.setLevel(entry.getShort("lvl"), false);
571                    return e;
572                }
573            }
574        }
575
576        return null;
577    }

579    public void addEnchantment(Enchantment... enchantments) {
580        CompoundTag tag;
581        if (!this.hasCompoundTag()) {
582            tag = new CompoundTag();
583        } else {
584            tag = this.getNamedTag();
585        }
586
587        ListTag<CompoundTag> ench;
588        if (!tag.contains("ench")) {
589            ench = new ListTag<>("ench");
590            tag.putList(ench);
591        } else {
592            ench = tag.getList("ench", CompoundTag.class);
593        }
594
595        for (Enchantment enchantment : enchantments) {
596            boolean found = false;
597
598            for (int k = 0; k < ench.size(); k++) {
599                CompoundTag entry = ench.get(k);
600                if (entry.getShort("id") == enchantment.getId()) {
601                    ench.add(k, new CompoundTag()
602                            .putShort("id", enchantment.getId())
603                            .putShort("lvl", enchantment.getLevel())
604                    );
605                    found = true;
606                    break;
607                }
608            }
609
610            if (!found) {
611                ench.add(new CompoundTag()
612                        .putShort("id", enchantment.getId())
613                        .putShort("lvl", enchantment.getLevel())
614                );
615            }
616        }
617
618        this.setNamedTag(tag);
619    }

626        List<Enchantment> enchantments = new ArrayList<>();
627
628        ListTag<CompoundTag> ench = this.getNamedTag().getList("ench", CompoundTag.class);
629        for (CompoundTag entry : ench.getAll()) {
630            Enchantment e = Enchantment.getEnchantment(entry.getShort("id"));

633                enchantments.add(e);
634            }
635        }

637        return enchantments.toArray(new Enchantment[0]);

694            return this;

696
697        CompoundTag tag = this.getNamedTag();
698
699        if (tag.contains("display") && tag.get("display") instanceof CompoundTag) {

701            if (tag.getCompound("display").isEmpty()) {
702                tag.remove("display");

704
705            this.setNamedTag(tag);
706        }

723            }

763            return null;

768        }
769
770        if (this.cachedNBT != null) {
771            this.cachedNBT.setName("");
772        }
773
774        return this.cachedNBT;
775    }

994    public final boolean equals(Item item, boolean checkDamage, boolean checkCompound) {
995        if (this.getId() == item.getId() && (!checkDamage || this.getDamage() == item.getDamage())) {
996            if (checkCompound) {
997                if (Arrays.equals(this.getCompoundTag(), item.getCompoundTag())) {
998                    return true;

1000                    return this.getNamedTag().equals(item.getNamedTag());

1002            } else {
1003                return true;
1004            }
1005        }
