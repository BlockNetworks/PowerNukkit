001package cn.nukkit.level.generator.object.tree;
002
003import cn.nukkit.block.Block;
004import cn.nukkit.block.BlockID;
005import cn.nukkit.block.BlockLeaves2;
006import cn.nukkit.block.BlockWood2;
007import cn.nukkit.level.ChunkManager;
008import cn.nukkit.math.BlockFace;
009import cn.nukkit.math.NukkitRandom;
010import cn.nukkit.math.Vector3;
011
012/**
013 * Created by CreeperFace on 23. 10. 2016.
014 */
015public class ObjectDarkOakTree extends TreeGenerator {
016    private static final Block DARK_OAK_LOG = Block.get(BlockID.WOOD2, BlockWood2.DARK_OAK);
017    private static final Block DARK_OAK_LEAVES = Block.get(BlockID.LEAVES2, BlockLeaves2.DARK_OAK);
018
019    @Override
020    public boolean generate(ChunkManager level, NukkitRandom rand, Vector3 position) {
021        int i = rand.nextBoundedInt(3) + rand.nextBoundedInt(2) + 6;
022        int j = position.getFloorX();
023        int k = position.getFloorY();
024        int l = position.getFloorZ();
025
026        if (k >= 1 && k + i + 1 < 256) {
027            Vector3 blockpos = position.down();
028            int block = level.getBlockIdAt(blockpos.getFloorX(), blockpos.getFloorY(), blockpos.getFloorZ());
029
030            if (block != Block.GRASS && block != Block.DIRT) {
031                return false;
032            } else if (!this.placeTreeOfHeight(level, position, i)) {
033                return false;
034            } else {
035                this.setDirtAt(level, blockpos);
036                this.setDirtAt(level, blockpos.east());
037                this.setDirtAt(level, blockpos.south());
038                this.setDirtAt(level, blockpos.south().east());
039                BlockFace enumfacing = BlockFace.Plane.HORIZONTAL.random(rand);
040                int i1 = i - rand.nextBoundedInt(4);
041                int j1 = 2 - rand.nextBoundedInt(3);
042                int k1 = j;
043                int l1 = l;
044                int i2 = k + i - 1;
045
046                for (int j2 = 0; j2 < i; ++j2) {
047                    if (j2 >= i1 && j1 > 0) {
048                        k1 += enumfacing.getXOffset();
049                        l1 += enumfacing.getZOffset();
050                        --j1;
051                    }
052
053                    int k2 = k + j2;
054                    Vector3 blockpos1 = new Vector3(k1, k2, l1);
055                    int material = level.getBlockIdAt(blockpos1.getFloorX(), blockpos1.getFloorY(), blockpos1.getFloorZ());
056
057                    if (material == Block.AIR || material == Block.LEAVES) {
058                        this.placeLogAt(level, blockpos1);
059                        this.placeLogAt(level, blockpos1.east());
060                        this.placeLogAt(level, blockpos1.south());
061                        this.placeLogAt(level, blockpos1.east().south());
062                    }
063                }
064
065                for (int i3 = -2; i3 <= 0; ++i3) {
066                    for (int l3 = -2; l3 <= 0; ++l3) {
067                        int k4 = -1;
068                        this.placeLeafAt(level, k1 + i3, i2 + k4, l1 + l3);
069                        this.placeLeafAt(level, 1 + k1 - i3, i2 + k4, l1 + l3);
070                        this.placeLeafAt(level, k1 + i3, i2 + k4, 1 + l1 - l3);
071                        this.placeLeafAt(level, 1 + k1 - i3, i2 + k4, 1 + l1 - l3);
072
073                        if ((i3 > -2 || l3 > -1) && (i3 != -1 || l3 != -2)) {
074                            k4 = 1;
075                            this.placeLeafAt(level, k1 + i3, i2 + k4, l1 + l3);
076                            this.placeLeafAt(level, 1 + k1 - i3, i2 + k4, l1 + l3);
077                            this.placeLeafAt(level, k1 + i3, i2 + k4, 1 + l1 - l3);
078                            this.placeLeafAt(level, 1 + k1 - i3, i2 + k4, 1 + l1 - l3);
079                        }
080                    }
081                }
082
083                if (rand.nextBoolean()) {
084                    this.placeLeafAt(level, k1, i2 + 2, l1);
085                    this.placeLeafAt(level, k1 + 1, i2 + 2, l1);
086                    this.placeLeafAt(level, k1 + 1, i2 + 2, l1 + 1);
087                    this.placeLeafAt(level, k1, i2 + 2, l1 + 1);
088                }
089
090                for (int j3 = -3; j3 <= 4; ++j3) {
091                    for (int i4 = -3; i4 <= 4; ++i4) {
092                        if ((j3 != -3 || i4 != -3) && (j3 != -3 || i4 != 4) && (j3 != 4 || i4 != -3) && (j3 != 4 || i4 != 4) && (Math.abs(j3) < 3 || Math.abs(i4) < 3)) {
093                            this.placeLeafAt(level, k1 + j3, i2, l1 + i4);
094                        }
095                    }
096                }
097
098                for (int k3 = -1; k3 <= 2; ++k3) {
099                    for (int j4 = -1; j4 <= 2; ++j4) {
100                        if ((k3 < 0 || k3 > 1 || j4 < 0 || j4 > 1) && rand.nextBoundedInt(3) <= 0) {
101                            int l4 = rand.nextBoundedInt(3) + 2;
102
103                            for (int i5 = 0; i5 < l4; ++i5) {
104                                this.placeLogAt(level, new Vector3(j + k3, i2 - i5 - 1, l + j4));
105                            }
106
107                            for (int j5 = -1; j5 <= 1; ++j5) {
108                                for (int l2 = -1; l2 <= 1; ++l2) {
109                                    this.placeLeafAt(level, k1 + k3 + j5, i2, l1 + j4 + l2);
110                                }
111                            }
112
113                            for (int k5 = -2; k5 <= 2; ++k5) {
114                                for (int l5 = -2; l5 <= 2; ++l5) {
115                                    if (Math.abs(k5) != 2 || Math.abs(l5) != 2) {
116                                        this.placeLeafAt(level, k1 + k3 + k5, i2 - 1, l1 + j4 + l5);
117                                    }
118                                }
119                            }
120                        }
121                    }
122                }
123
124                return true;
125            }
126        } else {
127            return false;
128        }
129    }
130
131    private boolean placeTreeOfHeight(ChunkManager worldIn, Vector3 pos, int height) {
132        int i = pos.getFloorX();
133        int j = pos.getFloorY();
134        int k = pos.getFloorZ();
135        Vector3 blockPos = new Vector3();
136
137        for (int l = 0; l <= height + 1; ++l) {
138            int i1 = 1;
139
140            if (l == 0) {
141                i1 = 0;
142            }
143
144            if (l >= height - 1) {
145                i1 = 2;
146            }
147
148            for (int j1 = -i1; j1 <= i1; ++j1) {
149                for (int k1 = -i1; k1 <= i1; ++k1) {
150                    blockPos.setComponents(i + j1, j + l, k + k1);
151                    if (!this.canGrowInto(worldIn.getBlockIdAt(blockPos.getFloorX(), blockPos.getFloorY(), blockPos.getFloorZ()))) {
152                        return false;
153                    }
154                }
155            }
156        }
157
158        return true;
159    }
160
161    private void placeLogAt(ChunkManager worldIn, Vector3 pos) {
162        if (this.canGrowInto(worldIn.getBlockIdAt(pos.getFloorX(), pos.getFloorY(), pos.getFloorZ()))) {
163            this.setBlockAndNotifyAdequately(worldIn, pos, DARK_OAK_LOG);
164        }
165    }
166
167    private void placeLeafAt(ChunkManager worldIn, int x, int y, int z) {
168        Vector3 blockpos = new Vector3(x, y, z);
169        int material = worldIn.getBlockIdAt(blockpos.getFloorX(), blockpos.getFloorY(), blockpos.getFloorZ());
170
171        if (material == Block.AIR) {
172            this.setBlockAndNotifyAdequately(worldIn, blockpos, DARK_OAK_LEAVES);
173        }
174    }
175}
