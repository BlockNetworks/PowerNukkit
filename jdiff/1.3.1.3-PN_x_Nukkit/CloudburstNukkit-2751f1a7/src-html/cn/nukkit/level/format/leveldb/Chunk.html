056    public Chunk(LevelProvider level, int chunkX, int chunkZ, byte[] terrain, List<CompoundTag> entityData, List<CompoundTag> tileData, Map<Integer, Integer> extraData) {
057        ByteBuffer buffer = ByteBuffer.wrap(terrain).order(ByteOrder.BIG_ENDIAN);
058
059        byte[] blocks = new byte[32768];
060        buffer.get(blocks);
061
062        byte[] data = new byte[16384];
063        buffer.get(data);
064
065        byte[] skyLight = new byte[16384];
066        buffer.get(skyLight);
067
068        byte[] blockLight = new byte[16384];
069        buffer.get(blockLight);
070
071        byte[] heightMap = new byte[256];
072        for (int i = 0; i < 256; i++) {
073            heightMap[i] = buffer.get();
074        }
075
076        int[] biomeColors = new int[256];
077        for (int i = 0; i < 256; i++) {
078            biomeColors[i] = buffer.getInt();
079        }
080
081        this.provider = level;
082        if (level != null) {
083            this.providerClass = level.getClass();
084        }
085
086        this.setPosition(chunkX, chunkZ);
087
088        this.blocks = blocks;
089        this.data = data;
090        this.skyLight = skyLight;
091        this.blockLight = blockLight;
092
093        this.biomes = new byte[16 * 16];
094        if (biomeColors.length == 256) {
095            BiomePalette palette = new BiomePalette(biomeColors);
096            for (int x = 0; x < 16; x++)    {
097                for (int z = 0; z < 16; z++)    {
098                    biomes[(x << 4) | z] = (byte) (palette.get(x, z) >> 24);
099                }
100            }
101        }
102
103        if (heightMap.length == 256) {
104            this.heightMap = heightMap;
105        } else {
106            byte[] bytes = new byte[256];
107            Arrays.fill(bytes, (byte) 256);
108            this.heightMap = bytes;
109        }
110
111        this.NBTentities = entityData;
112        this.NBTtiles = tileData;
113        this.extraData = extraData;
114    }

139        int i = (x << 10) | (z << 6) | (y >> 1);
140        int old = this.data[i] & 0xff;
141        if ((y & 1) == 0) {
142            this.data[i] = (byte) ((old & 0xf0) | (data & 0x0f));
143        } else {
144            this.data[i] = (byte) (((data & 0x0f) << 4) | (old & 0x0f));
145        }
146        setChanged();

149    @Override
150    public int getFullBlock(int x, int y, int z) {
151        int i = (x << 11) | (z << 7) | y;
152        int block = this.blocks[i] & 0xff;
153        int data = this.data[i >> 1] & 0xff;
154        if ((y & 1) == 0) {
155            return (block << 4) | (data & 0x0f);
156        } else {
157            return (block << 4) | (data >> 4);
158        }
159    }

161    @Override
162    public Block getAndSetBlock(int x, int y, int z, Block block) {
163        int i = (x << 11) | (z << 7) | y;
164        boolean changed = false;
165        byte id = (byte) block.getId();
166
167        byte previousId = this.blocks[i];
168
169        if (previousId != id) {
170            this.blocks[i] = id;
171            changed = true;
172        }
173
174        int previousData;
175        i >>= 1;
176        int old = this.data[i] & 0xff;
177        if ((y & 1) == 0) {
178            previousData = old & 0x0f;
179            if (Block.hasMeta[block.getId()]) {
180                this.data[i] = (byte) ((old & 0xf0) | (block.getDamage() & 0x0f));
181                if (block.getDamage() != previousData) {
182                    changed = true;
183                }
184            }
185        } else {
186            previousData = old >> 4;
187            if (Block.hasMeta[block.getId()]) {
188                this.data[i] = (byte) (((block.getDamage() & 0x0f) << 4) | (old & 0x0f));
189                if (block.getDamage() != previousData) {
190                    changed = true;
191                }
192            }
193        }
194
195        if (changed) {
196            setChanged();
197        }
198        return Block.get(previousId, previousData);
199    }

206    @Override
207    public boolean setBlock(int x, int y, int z, int blockId, int meta) {
208        int i = (x << 11) | (z << 7) | y;
209        boolean changed = false;
210        byte id = (byte) blockId;
211        if (this.blocks[i] != id) {
212            this.blocks[i] = id;
213            changed = true;
214        }
215
216        if (Block.hasMeta[blockId]) {
217            i >>= 1;
218            int old = this.data[i] & 0xff;
219            if ((y & 1) == 0) {
220                this.data[i] = (byte) ((old & 0xf0) | (meta & 0x0f));
221                if ((old & 0x0f) != meta) {
222                    changed = true;
223                }
224            } else {
225                this.data[i] = (byte) (((meta & 0x0f) << 4) | (old & 0x0f));
226                if (meta != (old >> 4)) {
227                    changed = true;
228                }
229            }
230        }
231
232        if (changed) {
233            setChanged();
234        }
235        return changed;
236    }

331    public static Chunk fromBinary(byte[] data, LevelProvider provider) {
332        try {
333
334            int chunkX = Binary.readLInt(new byte[]{data[0], data[1], data[2], data[3]});
335            int chunkZ = Binary.readLInt(new byte[]{data[4], data[5], data[6], data[7]});
336            byte[] chunkData = Binary.subBytes(data, 8, data.length - 1);
337
338            int flags = data[data.length - 1];
339
340            List<CompoundTag> entities = new ArrayList<>();
341            List<CompoundTag> tiles = new ArrayList<>();
342
343            Map<Integer, Integer> extraDataMap = new HashMap<>();
344
345            if (provider instanceof LevelDB) {
346                byte[] entityData = ((LevelDB) provider).getDatabase().get(EntitiesKey.create(chunkX, chunkZ).toArray());
347                if (entityData != null && entityData.length > 0) {
348                    try (NBTInputStream nbtInputStream = new NBTInputStream(new ByteArrayInputStream(entityData), ByteOrder.LITTLE_ENDIAN)) {
349                        while (nbtInputStream.available() > 0) {
350                            Tag tag = Tag.readNamedTag(nbtInputStream);
351                            if (!(tag instanceof CompoundTag)) {
352                                throw new IOException("Root tag must be a named compound tag");
353                            }
354                            entities.add((CompoundTag) tag);
355                        }
356                    }
357                }
358
359                byte[] tileData = ((LevelDB) provider).getDatabase().get(TilesKey.create(chunkX, chunkZ).toArray());
360                if (tileData != null && tileData.length > 0) {
361                    try (NBTInputStream nbtInputStream = new NBTInputStream(new ByteArrayInputStream(tileData), ByteOrder.LITTLE_ENDIAN)) {
362                        while (nbtInputStream.available() > 0) {
363                            Tag tag = Tag.readNamedTag(nbtInputStream);
364                            if (!(tag instanceof CompoundTag)) {
365                                throw new IOException("Root tag must be a named compound tag");
366                            }
367                            tiles.add((CompoundTag) tag);
368                        }
369                    }
370                }
371
372                byte[] extraData = ((LevelDB) provider).getDatabase().get(ExtraDataKey.create(chunkX, chunkZ).toArray());
373                if (extraData != null && extraData.length > 0) {
374                    BinaryStream stream = new BinaryStream(tileData);
375                    int count = stream.getInt();
376                    for (int i = 0; i < count; ++i) {
377                        int key = stream.getInt();
378                        int value = stream.getShort();
379                        extraDataMap.put(key, value);
380                    }
381                }
382
383                /*if (!entities.isEmpty() || !blockEntities.isEmpty()) {
384                    CompoundTag ct = new CompoundTag();
385                    ListTag<CompoundTag> entityList = new ListTag<>("entities");
386                    ListTag<CompoundTag> tileList = new ListTag<>("blockEntities");
387
388                    entityList.list = entities;
389                    tileList.list = blockEntities;
390                    ct.putList(entityList);
391                    ct.putList(tileList);
392                    NBTIO.write(ct, new File(Nukkit.DATA_PATH + chunkX + "_" + chunkZ + ".dat"));
393                }*/
394
395
396                Chunk chunk = new Chunk(provider, chunkX, chunkZ, chunkData, entities, tiles, extraDataMap);
397
398                if ((flags & 0x01) > 0) {
399                    chunk.setGenerated();
400                }
401
402                if ((flags & 0x02) > 0) {
403                    chunk.setPopulated();
404                }
405
406                if ((flags & 0x04) > 0) {
407                    chunk.setLightPopulated();
408                }
409                return chunk;
410            }
411        } catch (Exception e) {
412            Server.getInstance().getLogger().logException(e);
413        }
414        return null;
415    }

435    public byte[] toBinary(boolean saveExtra) {
436        try {
437            LevelProvider provider = this.getProvider();
438
439            if (saveExtra && provider instanceof LevelDB) {
440
441                List<CompoundTag> entities = new ArrayList<>();
442
443                for (Entity entity : this.getEntities().values()) {
444                    if (!(entity instanceof Player) && !entity.closed) {
445                        entity.saveNBT();
446                        entities.add(entity.namedTag);
447                    }
448                }
449
450                EntitiesKey entitiesKey = EntitiesKey.create(this.getX(), this.getZ());
451                if (!entities.isEmpty()) {
452                    ((LevelDB) provider).getDatabase().put(entitiesKey.toArray(), NBTIO.write(entities));
453                } else {
454                    ((LevelDB) provider).getDatabase().delete(entitiesKey.toArray());
455                }
456
457                List<CompoundTag> tiles = new ArrayList<>();
458
459                for (BlockEntity blockEntity : this.getBlockEntities().values()) {
460                    if (!blockEntity.closed) {
461                        blockEntity.saveNBT();
462                        tiles.add(blockEntity.namedTag);
463                    }
464                }
465
466                TilesKey tilesKey = TilesKey.create(this.getX(), this.getZ());
467                if (!tiles.isEmpty()) {
468                    ((LevelDB) provider).getDatabase().put(tilesKey.toArray(), NBTIO.write(tiles));
469                } else {
470                    ((LevelDB) provider).getDatabase().delete(tilesKey.toArray());
471                }
472
473                ExtraDataKey extraDataKey = ExtraDataKey.create(this.getX(), this.getZ());
474                if (!this.getBlockExtraDataArray().isEmpty()) {
475                    BinaryStream extraData = new BinaryStream();
476                    Map<Integer, Integer> extraDataArray = this.getBlockExtraDataArray();
477                    extraData.putInt(extraDataArray.size());
478                    for (Integer key : extraDataArray.keySet()) {
479                        extraData.putInt(key);
480                        extraData.putShort(extraDataArray.get(key));
481                    }
482                    ((LevelDB) provider).getDatabase().put(extraDataKey.toArray(), extraData.getBuffer());
483                } else {
484                    ((LevelDB) provider).getDatabase().delete(extraDataKey.toArray());
485                }
486
487            }
488
489            byte[] heightMap = this.getHeightMapArray();
490
491            byte[] biomeColors = new byte[this.biomes.length * 4];
492            for (int i = 0; i < this.biomes.length; i++) {
493                byte[] bytes = Binary.writeInt(this.biomes[i] << 24);
494                biomeColors[i * 4] = bytes[0];
495                biomeColors[i * 4 + 1] = bytes[1];
496                biomeColors[i * 4 + 2] = bytes[2];
497                biomeColors[i * 4 + 3] = bytes[3];
498            }
499
500            return Binary.appendBytes(
501                    Binary.writeLInt(this.getX()),
502                    Binary.writeLInt(this.getZ()),
503                    this.getBlockIdArray(),
504                    this.getBlockDataArray(),
505                    this.getBlockSkyLightArray(),
506                    this.getBlockLightArray(),
507                    heightMap,
508                    biomeColors,
509                    new byte[]{(byte) (((this.isLightPopulated ? 0x04 : 0) | (this.isPopulated() ? 0x02 : 0) | (this.isGenerated() ? 0x01 : 0)) & 0xff)}
510            );
511        } catch (Exception e) {
512            throw new RuntimeException(e);
513        }
514    }

520    public static Chunk getEmptyChunk(int chunkX, int chunkZ, LevelProvider provider) {
521        try {
522            Chunk chunk;
523            if (provider != null) {
524                chunk = new Chunk(provider, chunkX, chunkZ, new byte[DATA_LENGTH]);
525            } else {
526                chunk = new Chunk(LevelDB.class, chunkX, chunkZ, new byte[DATA_LENGTH]);
527            }
528
529            byte[] skyLight = new byte[16384];
530            Arrays.fill(skyLight, (byte) 0xff);
531            chunk.skyLight = skyLight;
532            return chunk;
533        } catch (Exception e) {
534            throw new RuntimeException(e);
535        }
