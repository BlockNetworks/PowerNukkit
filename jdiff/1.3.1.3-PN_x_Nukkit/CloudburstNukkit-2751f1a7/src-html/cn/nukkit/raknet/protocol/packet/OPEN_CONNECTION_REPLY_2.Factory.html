001package cn.nukkit.raknet.protocol.packet;
002
003import cn.nukkit.raknet.RakNet;
004import cn.nukkit.raknet.protocol.Packet;
005
006import java.net.InetSocketAddress;
007
008/**
009 * author: MagicDroidX
010 * Nukkit Project
011 */
012public class OPEN_CONNECTION_REPLY_2 extends Packet {
013    public static final byte ID = (byte) 0x08;
014
015    @Override
016    public byte getID() {
017        return ID;
018    }
019
020    public long serverID;
021    public String clientAddress;
022    public int clientPort;
023    public short mtuSize;
024
025    @Override
026    public void encode() {
027        super.encode();
028        this.put(RakNet.MAGIC);
029        this.putLong(this.serverID);
030        this.putAddress(this.clientAddress, this.clientPort);
031        this.putShort(this.mtuSize);
032        this.putByte((byte) 0); //server security
033    }
034
035    @Override
036    public void decode() {
037        super.decode();
038        this.offset += 16; //skip magic bytes
039        this.serverID = this.getLong();
040        InetSocketAddress address = this.getAddress();
041        this.clientAddress = address.getHostString();
042        this.clientPort = address.getPort();
043        this.mtuSize = this.getSignedShort();
044    }
045
046    public static final class Factory implements Packet.PacketFactory {
047
048        @Override
049        public Packet create() {
050            return new OPEN_CONNECTION_REPLY_2();
051        }
052
053    }
054}
