001package cn.nukkit.event.server;
002
003import cn.nukkit.Player;
004import cn.nukkit.event.Cancellable;
005import cn.nukkit.event.HandlerList;
006import cn.nukkit.network.protocol.DataPacket;
007
008public class BatchPacketsEvent extends ServerEvent implements Cancellable {
009
010    private static final HandlerList handlers = new HandlerList();
011
012    public static HandlerList getHandlers() {
013        return handlers;
014    }
015
016    private Player[] players;
017    private DataPacket[] packets;
018
019    private boolean forceSync;
020
021    public BatchPacketsEvent(Player[] players, DataPacket[] packets, boolean forceSync) {
022        this.players = players;
023        this.packets = packets;
024        this.forceSync = forceSync;
025    }
026
027    public Player[] getPlayers() {
028        return players;
029    }
030
031    public DataPacket[] getPackets() {
032        return packets;
033    }
034
035    public boolean isForceSync() {
036        return forceSync;
037    }
038}
