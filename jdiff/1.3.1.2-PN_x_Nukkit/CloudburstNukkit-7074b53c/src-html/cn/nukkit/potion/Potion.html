111    public static Potion getPotion(int id) {
112        if (id >= 0 && id < potions.length && potions[id] != null) {
113            return potions[id].clone();
114        } else {
115            throw new ServerException("Effect id: " + id + " not found");
116        }
117    }

119    public static Potion getPotionByName(String name) {
120        try {
121            byte id = Potion.class.getField(name.toUpperCase()).getByte(null);
122            return getPotion(id);

124            throw new RuntimeException(e);
125        }

173    public void applyPotion(Entity entity, double health) {
174        if (!(entity instanceof EntityLiving)) {
175            return;
176        }
177
178        Effect applyEffect = getEffect(this.getId(), this.isSplash());
179
180        if (applyEffect == null) {
181            return;
182        }
183
184        if (entity instanceof Player) {
185            if (!((Player) entity).isSurvival() && !((Player) entity).isAdventure() && applyEffect.isBad()) {
186                return;
187            }
188        }
189
190        PotionApplyEvent event = new PotionApplyEvent(this, applyEffect, entity);
191
192        entity.getServer().getPluginManager().callEvent(event);
193        if (event.isCancelled()) {
194            return;
195        }
196
197        applyEffect = event.getApplyEffect();
198
199        switch (this.getId()) {
200            case INSTANT_HEALTH:
201            case INSTANT_HEALTH_II:
202                entity.heal(new EntityRegainHealthEvent(entity, (float) (health * (double) (4 << (applyEffect.getAmplifier() + 1))), EntityRegainHealthEvent.CAUSE_MAGIC));
203                break;
204            case HARMING:
205            case HARMING_II:
206                entity.attack(new EntityDamageEvent(entity, DamageCause.MAGIC, (float) (health * (double) (6 << (applyEffect.getAmplifier() + 1)))));
207                break;
208            default:
209                int duration = (int) ((isSplash() ? health : 1) * (double) applyEffect.getDuration() + 0.5);
210                applyEffect.setDuration(duration);
211                entity.addEffect(applyEffect);
212        }
213    }

224    public static Effect getEffect(int potionType, boolean isSplash) {
225        Effect effect;
226        switch (potionType) {
227            case NIGHT_VISION:
228            case NIGHT_VISION_LONG:
229                effect = Effect.getEffect(Effect.NIGHT_VISION);
230                break;
231            case INVISIBLE:
232            case INVISIBLE_LONG:
233                effect = Effect.getEffect(Effect.INVISIBILITY);
234                break;
235            case LEAPING:
236            case LEAPING_LONG:
237            case LEAPING_II:
238                effect = Effect.getEffect(Effect.JUMP);
239                break;
240            case FIRE_RESISTANCE:
241            case FIRE_RESISTANCE_LONG:
242                effect = Effect.getEffect(Effect.FIRE_RESISTANCE);
243                break;
244            case SPEED:
245            case SPEED_LONG:
246            case SPEED_II:
247                effect = Effect.getEffect(Effect.SPEED);
248                break;
249            case SLOWNESS:
250            case SLOWNESS_LONG:
251                effect = Effect.getEffect(Effect.SLOWNESS);
252                break;
253            case WATER_BREATHING:
254            case WATER_BREATHING_LONG:
255                effect = Effect.getEffect(Effect.WATER_BREATHING);
256                break;
257            case INSTANT_HEALTH:
258            case INSTANT_HEALTH_II:
259                return Effect.getEffect(Effect.HEALING);
260            case HARMING:
261            case HARMING_II:
262                return Effect.getEffect(Effect.HARMING);
263            case POISON:
264            case POISON_LONG:
265            case POISON_II:
266                effect = Effect.getEffect(Effect.POISON);
267                break;
268            case REGENERATION:
269            case REGENERATION_LONG:
270            case REGENERATION_II:
271                effect = Effect.getEffect(Effect.REGENERATION);
272                break;
273            case STRENGTH:
274            case STRENGTH_LONG:
275            case STRENGTH_II:
276                effect = Effect.getEffect(Effect.STRENGTH);
277                break;
278            case WEAKNESS:
279            case WEAKNESS_LONG:
280                effect = Effect.getEffect(Effect.WEAKNESS);
281                break;
282            case WITHER_II:
283                effect = Effect.getEffect(Effect.WITHER);
284                break;
285            default:
286                return null;
287        }
288
289        if (getLevel(potionType) > 1) {
290            effect.setAmplifier(1);
291        }
292
293        if (!isInstant(potionType)) {
294            effect.setDuration(20 * getApplySeconds(potionType, isSplash));
295        }
296
297        return effect;
298    }
