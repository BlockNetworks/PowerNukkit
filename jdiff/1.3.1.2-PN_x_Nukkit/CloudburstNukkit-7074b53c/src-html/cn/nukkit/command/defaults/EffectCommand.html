001package cn.nukkit.command.defaults;
002
003import cn.nukkit.Player;
004import cn.nukkit.command.Command;
005import cn.nukkit.command.CommandSender;
006import cn.nukkit.command.data.CommandParamType;
007import cn.nukkit.command.data.CommandParameter;
008import cn.nukkit.lang.TranslationContainer;
009import cn.nukkit.potion.Effect;
010import cn.nukkit.potion.InstantEffect;
011import cn.nukkit.utils.ServerException;
012import cn.nukkit.utils.TextFormat;
013
014/**
015 * Created by Snake1999 and Pub4Game on 2016/1/23.
016 * Package cn.nukkit.command.defaults in project nukkit.
017 */
018public class EffectCommand extends Command {
019    public EffectCommand(String name) {
020        super(name, "%nukkit.command.effect.description", "%commands.effect.usage");
021        this.setPermission("nukkit.command.effect");
022        this.commandParameters.clear();
023        this.commandParameters.put("default", new CommandParameter[]{
024                new CommandParameter("player", CommandParamType.TARGET, false),
025                new CommandParameter("effect", CommandParamType.STRING, false), //Do not use Enum here because of buggy behavior
026                new CommandParameter("seconds", CommandParamType.INT, true),
027                new CommandParameter("amplifier", true),
028                new CommandParameter("hideParticle", true, new String[]{"true", "false"})
029        });
030        this.commandParameters.put("clear", new CommandParameter[]{
031                new CommandParameter("player", CommandParamType.TARGET, false),
032                new CommandParameter("clear", new String[]{"clear"})
033        });
034    }
035
036    @Override
037    public boolean execute(CommandSender sender, String commandLabel, String[] args) {
038        if (!this.testPermission(sender)) {
039            return true;
040        }
041        if (args.length < 2) {
042            sender.sendMessage(new TranslationContainer("commands.generic.usage", this.usageMessage));
043            return true;
044        }
045        Player player = sender.getServer().getPlayer(args[0]);
046        if (player == null) {
047            sender.sendMessage(new TranslationContainer(TextFormat.RED + "%commands.generic.player.notFound"));
048            return true;
049        }
050        if (args[1].equalsIgnoreCase("clear")) {
051            for (Effect effect : player.getEffects().values()) {
052                player.removeEffect(effect.getId());
053            }
054            sender.sendMessage(new TranslationContainer("commands.effect.success.removed.all", player.getDisplayName()));
055            return true;
056        }
057        Effect effect;
058        try {
059            effect = Effect.getEffect(Integer.parseInt(args[1]));
060        } catch (NumberFormatException | ServerException a) {
061            try {
062                effect = Effect.getEffectByName(args[1]);
063            } catch (Exception e) {
064                sender.sendMessage(new TranslationContainer("commands.effect.notFound", args[1]));
065                return true;
066            }
067        }
068        int duration = 300;
069        int amplification = 0;
070        if (args.length >= 3) {
071            try {
072                duration = Integer.parseInt(args[2]);
073            } catch (NumberFormatException a) {
074                sender.sendMessage(new TranslationContainer("commands.generic.usage", this.usageMessage));
075                return true;
076            }
077            if (!(effect instanceof InstantEffect)) {
078                duration *= 20;
079            }
080        } else if (effect instanceof InstantEffect) {
081            duration = 1;
082        }
083        if (args.length >= 4) {
084            try {
085                amplification = Integer.parseInt(args[3]);
086            } catch (NumberFormatException a) {
087                sender.sendMessage(new TranslationContainer("commands.generic.usage", this.usageMessage));
088                return true;
089            }
090        }
091        if (args.length >= 5) {
092            String v = args[4].toLowerCase();
093            if (v.matches("(?i)|on|true|t|1")) {
094                effect.setVisible(false);
095            }
096        }
097        if (duration == 0) {
098            if (!player.hasEffect(effect.getId())) {
099                if (player.getEffects().size() == 0) {
100                    sender.sendMessage(new TranslationContainer("commands.effect.failure.notActive.all", player.getDisplayName()));
101                } else {
102                    sender.sendMessage(new TranslationContainer("commands.effect.failure.notActive", effect.getName(), player.getDisplayName()));
103                }
104                return true;
105            }
106            player.removeEffect(effect.getId());
107            sender.sendMessage(new TranslationContainer("commands.effect.success.removed", effect.getName(), player.getDisplayName()));
108        } else {
109            effect.setDuration(duration).setAmplifier(amplification);
110            player.addEffect(effect);
111            Command.broadcastCommandMessage(sender, new TranslationContainer("%commands.effect.success", effect.getName(), String.valueOf(effect.getAmplifier()), player.getDisplayName(), String.valueOf(effect.getDuration() / 20)));
112        }
113        return true;
114    }
115}
