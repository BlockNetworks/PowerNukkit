001package cn.nukkit.level.generator.populator.type;
002
003import cn.nukkit.block.BlockID;
004import cn.nukkit.level.ChunkManager;
005import cn.nukkit.level.format.FullChunk;
006import cn.nukkit.math.NukkitRandom;
007
008/**
009 * author: MagicDroidX
010 * Nukkit Project
011 */
012public abstract class Populator implements BlockID {
013    public abstract void populate(ChunkManager level, int chunkX, int chunkZ, NukkitRandom random, FullChunk chunk);
014
015    protected int getHighestWorkableBlock(ChunkManager level, int x, int z, FullChunk chunk)    {
016        return chunk.getHighestBlockAt(x & 0xF, z & 0xF);
017    }
018}
