001package cn.nukkit.level.generator.populator.impl;
002
003import cn.nukkit.block.BlockID;
004import cn.nukkit.level.ChunkManager;
005import cn.nukkit.level.biome.EnumBiome;
006import cn.nukkit.level.format.FullChunk;
007import cn.nukkit.level.biome.Biome;
008import cn.nukkit.level.biome.type.CoveredBiome;
009import cn.nukkit.level.generator.Normal;
010import cn.nukkit.level.generator.populator.type.Populator;
011import cn.nukkit.math.NukkitRandom;
012
013/**
014 * author: DaPorkchop_
015 * Nukkit Project
016 */
017public class PopulatorGroundCover extends Populator {
018    public static final int STONE = BlockID.STONE << 4;
019
020    @Override
021    public void populate(ChunkManager level, int chunkX, int chunkZ, NukkitRandom random, FullChunk chunk) {
022        int realX = chunkX << 4;
023        int realZ = chunkZ << 4;
024        for (int x = 0; x < 16; ++x) {
025            for (int z = 0; z < 16; ++z) {
026                Biome realBiome = EnumBiome.getBiome(chunk.getBiomeId(x, z));
027                if (realBiome instanceof CoveredBiome) {
028                    final CoveredBiome biome = (CoveredBiome) realBiome;
029                    //just in case!
030                    synchronized (biome.synchronizeCover) {
031                        biome.preCover(realX | x, realZ | z);
032                        int coverBlock = biome.getCoverBlock() << 4;
033
034                        boolean hasCovered = false;
035                        int realY;
036                        //start one below build limit in case of cover blocks
037                        for (int y = 254; y > 32; y--) {
038                            if (chunk.getFullBlock(x, y, z) == STONE) {
039                                COVER:
040                                if (!hasCovered) {
041                                    if (y >= Normal.seaHeight) {
042                                        chunk.setFullBlockId(x, y + 1, z, coverBlock);
043                                        int surfaceDepth = biome.getSurfaceDepth(y);
044                                        for (int i = 0; i < surfaceDepth; i++) {
045                                            realY = y - i;
046                                            if (chunk.getFullBlock(x, realY, z) == STONE) {
047                                                chunk.setFullBlockId(x, realY, z, (biome.getSurfaceBlock(realY) << 4) | biome.getSurfaceMeta(realY));
048                                            } else break COVER;
049                                        }
050                                        y -= surfaceDepth;
051                                    }
052                                    int groundDepth = biome.getGroundDepth(y);
053                                    for (int i = 0; i < groundDepth; i++) {
054                                        realY = y - i;
055                                        if (chunk.getFullBlock(x, realY, z) == STONE) {
056                                            chunk.setFullBlockId(x, realY, z, (biome.getGroundBlock(realY) << 4) | biome.getGroundMeta(realY));
057                                        } else break COVER;
058                                    }
059                                    //don't take all of groundDepth away because we do y-- in the loop
060                                    y -= groundDepth - 1;
061                                }
062                                hasCovered = true;
063                            } else {
064                                if (hasCovered) {
065                                    //reset it if this isn't a valid stone block (allows us to place ground cover on top and below overhangs)
066                                    hasCovered = false;
067                                }
068                            }
069                        }
070                    }
071                }
072            }
073        }
074    }
075}
