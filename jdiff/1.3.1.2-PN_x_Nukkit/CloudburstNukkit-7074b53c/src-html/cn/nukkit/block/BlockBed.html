067    @Override
068    public boolean onActivate(Item item, Player player) {
069
070        if (this.level.getDimension() == Level.DIMENSION_NETHER || this.level.getDimension() == Level.DIMENSION_THE_END) {
071            CompoundTag tag = EntityPrimedTNT.getDefaultNBT(this).putShort("Fuse", 0);
072            new EntityPrimedTNT(this.level.getChunk(this.getFloorX() >> 4, this.getFloorZ() >> 4), tag);
073            return true;
074        }
075
076        int time = this.getLevel().getTime() % Level.TIME_FULL;
077
078        boolean isNight = (time >= Level.TIME_NIGHT && time < Level.TIME_SUNRISE);
079
080        if (player != null && !isNight) {
081            player.sendMessage(new TranslationContainer("tile.bed.noSleep"));
082            return true;
083        }
084
085        Block blockNorth = this.north();
086        Block blockSouth = this.south();
087        Block blockEast = this.east();
088        Block blockWest = this.west();
089
090        Block b;
091        if ((this.getDamage() & 0x08) == 0x08) {
092            b = this;
093        } else {
094            if (blockNorth.getId() == this.getId() && (blockNorth.getDamage() & 0x08) == 0x08) {
095                b = blockNorth;
096            } else if (blockSouth.getId() == this.getId() && (blockSouth.getDamage() & 0x08) == 0x08) {
097                b = blockSouth;
098            } else if (blockEast.getId() == this.getId() && (blockEast.getDamage() & 0x08) == 0x08) {
099                b = blockEast;
100            } else if (blockWest.getId() == this.getId() && (blockWest.getDamage() & 0x08) == 0x08) {
101                b = blockWest;
102            } else {
103                if (player != null) {
104                    player.sendMessage(new TranslationContainer("tile.bed.notValid"));
105                }
106
107                return true;
108            }
109        }
110
111        if (player != null && !player.sleepOn(b)) {
112            player.sendMessage(new TranslationContainer("tile.bed.occupied"));
113        }
114
115
116        return true;
117    }
118
119    @Override
120    public boolean place(Item item, Block block, Block target, BlockFace face, double fx, double fy, double fz, Player player) {
121        Block down = this.down();
122        if (!down.isTransparent()) {
123            Block next = this.getSide(player.getDirection());
124            Block downNext = next.down();
125
126            if (next.canBeReplaced() && !downNext.isTransparent()) {
127                int meta = player.getDirection().getHorizontalIndex();
128
129                this.getLevel().setBlock(block, Block.get(this.getId(), meta), true, true);
130                this.getLevel().setBlock(next, Block.get(this.getId(), meta | 0x08), true, true);
131
132                createBlockEntity(this, item.getDamage());
133                createBlockEntity(next, item.getDamage());
134                return true;
135            }
136        }
137
138        return false;
139    }
140
141    @Override
142    public boolean onBreak(Item item) {
143        Block blockNorth = this.north(); //Gets the blocks around them
144        Block blockSouth = this.south();
145        Block blockEast = this.east();
146        Block blockWest = this.west();
147
148        if ((this.getDamage() & 0x08) == 0x08) { //This is the Top part of bed
149            if (blockNorth.getId() == BED_BLOCK && (blockNorth.getDamage() & 0x08) != 0x08) { //Checks if the block ID&&meta are right
150                this.getLevel().setBlock(blockNorth, Block.get(BlockID.AIR), true, true);
151            } else if (blockSouth.getId() == BED_BLOCK && (blockSouth.getDamage() & 0x08) != 0x08) {
152                this.getLevel().setBlock(blockSouth, Block.get(BlockID.AIR), true, true);
153            } else if (blockEast.getId() == BED_BLOCK && (blockEast.getDamage() & 0x08) != 0x08) {
154                this.getLevel().setBlock(blockEast, Block.get(BlockID.AIR), true, true);
155            } else if (blockWest.getId() == BED_BLOCK && (blockWest.getDamage() & 0x08) != 0x08) {
156                this.getLevel().setBlock(blockWest, Block.get(BlockID.AIR), true, true);
157            }
158        } else { //Bottom Part of Bed
159            if (blockNorth.getId() == this.getId() && (blockNorth.getDamage() & 0x08) == 0x08) {
160                this.getLevel().setBlock(blockNorth, Block.get(BlockID.AIR), true, true);
161            } else if (blockSouth.getId() == this.getId() && (blockSouth.getDamage() & 0x08) == 0x08) {
162                this.getLevel().setBlock(blockSouth, Block.get(BlockID.AIR), true, true);
163            } else if (blockEast.getId() == this.getId() && (blockEast.getDamage() & 0x08) == 0x08) {
164                this.getLevel().setBlock(blockEast, Block.get(BlockID.AIR), true, true);
165            } else if (blockWest.getId() == this.getId() && (blockWest.getDamage() & 0x08) == 0x08) {
166                this.getLevel().setBlock(blockWest, Block.get(BlockID.AIR), true, true);
167            }
168        }
169
170        this.getLevel().setBlock(this, Block.get(BlockID.AIR), true, false); // Do not update both parts to prevent duplication bug if there is two fallable blocks top of the bed
171
172        return true;
173    }
174
175    private void createBlockEntity(Vector3 pos, int color) {
176        CompoundTag nbt = BlockEntity.getDefaultCompound(pos, BlockEntity.BED);
177        nbt.putByte("color", color);
178
179        BlockEntity.createBlockEntity(BlockEntity.BED, this.level.getChunk(pos.getFloorX() >> 4, pos.getFloorZ() >> 4), nbt);
180    }

182    @Override
183    public Item toItem() {
184        return new ItemBed(this.getDyeColor().getWoolData());
185    }
186
187    @Override
188    public BlockColor getColor() {
189        return this.getDyeColor().getColor();
190    }
191
192    public DyeColor getDyeColor() {
193        if (this.level != null) {
194            BlockEntity blockEntity = this.level.getBlockEntity(this);
195
196            if (blockEntity instanceof BlockEntityBed) {
197                return ((BlockEntityBed) blockEntity).getDyeColor();
198            }
199        }
200
201        return DyeColor.WHITE;
202    }
203
204    @Override
205    public BlockFace getBlockFace() {
206        return BlockFace.fromHorizontalIndex(this.getDamage() & 0x7);
207    }
