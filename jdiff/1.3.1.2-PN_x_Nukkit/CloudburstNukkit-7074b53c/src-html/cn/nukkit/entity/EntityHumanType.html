027public abstract class EntityHumanType extends EntityCreature implements InventoryHolder {
028
029    protected PlayerInventory inventory;
030    protected PlayerEnderChestInventory enderChestInventory;
031    protected PlayerOffhandInventory offhandInventory;
032
033    public EntityHumanType(FullChunk chunk, CompoundTag nbt) {
034        super(chunk, nbt);
035    }
036
037    @Override
038    public PlayerInventory getInventory() {
039        return inventory;
040    }
041
042    public PlayerEnderChestInventory getEnderChestInventory() {
043        return enderChestInventory;
044    }
045
046    public PlayerOffhandInventory getOffhandInventory() {
047        return offhandInventory;
048    }
049
050    @Override
051    protected void initEntity() {
052        this.inventory = new PlayerInventory(this);
053        this.offhandInventory = new PlayerOffhandInventory(this);
054
055        if (this.namedTag.contains("Inventory") && this.namedTag.get("Inventory") instanceof ListTag) {
056            ListTag<CompoundTag> inventoryList = this.namedTag.getList("Inventory", CompoundTag.class);
057            for (CompoundTag item : inventoryList.getAll()) {
058                int slot = item.getByte("Slot");
059                if (slot >= 0 && slot < 9) { //hotbar
060                    //Old hotbar saving stuff, remove it (useless now)
061                    inventoryList.remove(item);
062                } else if (slot >= 100 && slot < 104) {
063                    this.inventory.setItem(this.inventory.getSize() + slot - 100, NBTIO.getItemHelper(item));
064                } else if (slot == -106) {
065                    this.offhandInventory.setItem(0, NBTIO.getItemHelper(item));
066                } else {
067                    this.inventory.setItem(slot - 9, NBTIO.getItemHelper(item));
068                }
069            }
070        }
071
072        this.enderChestInventory = new PlayerEnderChestInventory(this);
073
074        if (this.namedTag.contains("EnderItems") && this.namedTag.get("EnderItems") instanceof ListTag) {
075            ListTag<CompoundTag> inventoryList = this.namedTag.getList("EnderItems", CompoundTag.class);
076            for (CompoundTag item : inventoryList.getAll()) {
077                this.enderChestInventory.setItem(item.getByte("Slot"), NBTIO.getItemHelper(item));
078            }
079        }
080
081        super.initEntity();
082    }
083
084    @Override
085    public void saveNBT() {
086        super.saveNBT();
087
088        ListTag<CompoundTag> inventoryTag = null;
089        if (this.inventory != null) {
090            inventoryTag = new ListTag<>("Inventory");
091            this.namedTag.putList(inventoryTag);
092
093            for (int slot = 0; slot < 9; ++slot) {
094                inventoryTag.add(new CompoundTag()
095                        .putByte("Count", 0)
096                        .putShort("Damage", 0)
097                        .putByte("Slot", slot)
098                        .putByte("TrueSlot", -1)
099                        .putShort("id", 0)
100                );
101            }
102
103            int slotCount = Player.SURVIVAL_SLOTS + 9;
104            for (int slot = 9; slot < slotCount; ++slot) {
105                Item item = this.inventory.getItem(slot - 9);
106                inventoryTag.add(NBTIO.putItemHelper(item, slot));
107            }
108
109            for (int slot = 100; slot < 104; ++slot) {
110                Item item = this.inventory.getItem(this.inventory.getSize() + slot - 100);
111                if (item != null && item.getId() != Item.AIR) {
112                    inventoryTag.add(NBTIO.putItemHelper(item, slot));
113                }
114            }
115        }
116
117        if (this.offhandInventory != null) {
118            Item item = this.offhandInventory.getItem(0);
119            if (item.getId() != Item.AIR) {
120                if (inventoryTag == null) {
121                    inventoryTag = new ListTag<>("Inventory");
122                    this.namedTag.putList(inventoryTag);
123                }
124                inventoryTag.add(NBTIO.putItemHelper(item, -106));
125            }
126        }
127
128        this.namedTag.putList(new ListTag<CompoundTag>("EnderItems"));
129        if (this.enderChestInventory != null) {
130            for (int slot = 0; slot < 27; ++slot) {
131                Item item = this.enderChestInventory.getItem(slot);
132                if (item != null && item.getId() != Item.AIR) {
133                    this.namedTag.getList("EnderItems", CompoundTag.class).add(NBTIO.putItemHelper(item, slot));
134                }
135            }
136        }
137    }
138
139    @Override
140    public Item[] getDrops() {
141        if (this.inventory != null) {
142            List<Item> drops = new ArrayList<>(this.inventory.getContents().values());
143            drops.addAll(this.offhandInventory.getContents().values());
144            return drops.toArray(new Item[0]);
145        }
146        return new Item[0];
147    }
148
149    @Override
150    public boolean attack(EntityDamageEvent source) {
151        if (this.isClosed() || !this.isAlive()) {
152            return false;
153        }
154
155        if (source.getCause() != DamageCause.VOID && source.getCause() != DamageCause.CUSTOM && source.getCause() != DamageCause.MAGIC) {
156            int armorPoints = 0;
157            int epf = 0;
158            int toughness = 0;
159
160            for (Item armor : inventory.getArmorContents()) {
161                armorPoints += armor.getArmorPoints();
162                epf += calculateEnchantmentProtectionFactor(armor, source);
163                //toughness += armor.getToughness();
164            }
165
166            if (source.canBeReducedByArmor()) {
167                source.setDamage(-source.getFinalDamage() * armorPoints * 0.04f, DamageModifier.ARMOR);
168            }
169
170            source.setDamage(-source.getFinalDamage() * Math.min(NukkitMath.ceilFloat(Math.min(epf, 25) * ((float) ThreadLocalRandom.current().nextInt(50, 100) / 100)), 20) * 0.04f,
171                    DamageModifier.ARMOR_ENCHANTMENTS);
172
173            source.setDamage(-Math.min(this.getAbsorption(), source.getFinalDamage()), DamageModifier.ABSORPTION);
174        }
175
176        if (super.attack(source)) {
177            Entity damager = null;
178
179            if (source instanceof EntityDamageByEntityEvent) {
180                damager = ((EntityDamageByEntityEvent) source).getDamager();
181            }
182
183            for (int slot = 0; slot < 4; slot++) {
184                Item armor = this.inventory.getArmorItem(slot);
185
186                if (armor.hasEnchantments()) {
187                    if (damager != null) {
188                        for (Enchantment enchantment : armor.getEnchantments()) {
189                            enchantment.doPostAttack(damager, this);
190                        }
191                    }
192
193                    Enchantment durability = armor.getEnchantment(Enchantment.ID_DURABILITY);
194                    if (durability != null && durability.getLevel() > 0 && (100 / (durability.getLevel() + 1)) <= ThreadLocalRandom.current().nextInt(100))
195                        continue;
196                }
197
198                if (armor.isUnbreakable()) {
199                    continue;
200                }
201
202                armor.setDamage(armor.getDamage() + 1);
203
204                if (armor.getDamage() >= armor.getMaxDurability()) {
205                    inventory.setArmorItem(slot, new ItemBlock(Block.get(BlockID.AIR)));
206                } else {
207                    inventory.setArmorItem(slot, armor, true);
208                }
209            }
210
211            return true;
212        } else {
213            return false;
214        }
215    }
216
217    protected double calculateEnchantmentProtectionFactor(Item item, EntityDamageEvent source) {
218        if (!item.hasEnchantments()) {
219            return 0;
220        }
221
222        double epf = 0;
223
224        for (Enchantment ench : item.getEnchantments()) {
225            epf += ench.getProtectionFactor(source);
226        }
227
228        return epf;
229    }
230
231    @Override
232    public void setOnFire(int seconds) {
233        int level = 0;
234
235        for (Item armor : this.inventory.getArmorContents()) {
236            Enchantment fireProtection = armor.getEnchantment(Enchantment.ID_PROTECTION_FIRE);
237
238            if (fireProtection != null && fireProtection.getLevel() > 0) {
239                level = Math.max(level, fireProtection.getLevel());
240            }
241        }
242
243        seconds = (int) (seconds * (1 - level * 0.15));
244
245        super.setOnFire(seconds);
246    }
