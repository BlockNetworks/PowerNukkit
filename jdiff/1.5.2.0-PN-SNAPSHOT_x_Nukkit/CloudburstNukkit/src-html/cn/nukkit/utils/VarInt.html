012/**
013 * Tool class for VarInt or VarLong operations.
014 * <p>
015 * Some code from http://wiki.vg/Protocol.
016 *
017 * @author MagicDroidX

020@API(usage = EXPERIMENTAL, definition = UNIVERSAL)

022
023    private VarInt() {
024        //no instance
025    }

027    /**
028     * @param v Signed int
029     * @return Unsigned encoded int
030     */
031    public static long encodeZigZag32(int v) {
032        // Note:  the right-shift must be arithmetic
033        return ((v << 1) ^ (v >> 31)) & 0xFFFFFFFFL;
034    }
035
036    /**
037     * @param v Unsigned encoded int
038     * @return Signed decoded int
039     */
040    public static int decodeZigZag32(long v) {
041        return (int) (v >> 1) ^ -(int) (v & 1);
042    }
043
044    /**
045     * @param v Signed long
046     * @return Unsigned encoded long
047     */
048    public static long encodeZigZag64(long v) {
049        return (v << 1) ^ (v >> 63);
050    }
051
052    /**
053     * @param v Signed encoded long
054     * @return Unsigned decoded long
055     */
056    public static long decodeZigZag64(long v) {
057        return (v >>> 1) ^ -(v & 1);
058    }
059
060    private static long read(BinaryStream stream, int maxSize) {
061        long value = 0;
062        int size = 0;
063        int b;
064        while (((b = stream.getByte()) & 0x80) == 0x80) {
065            value |= (long) (b & 0x7F) << (size++ * 7);
066            if (size >= maxSize) {
067                throw new IllegalArgumentException("VarLong too big");
068            }
069        }
070
071        return value | ((long) (b & 0x7F) << (size * 7));
072    }
073
074    private static long read(InputStream stream, int maxSize) throws IOException {
075        long value = 0;
076        int size = 0;
077        int b;
078        while (((b = stream.read()) & 0x80) == 0x80) {
079            value |= (long) (b & 0x7F) << (size++ * 7);
080            if (size >= maxSize) {
081                throw new IllegalArgumentException("VarLong too big");
082            }
083        }
084
085        return value | ((long) (b & 0x7F) << (size * 7));
086    }

152    private static void write(BinaryStream stream, long value) {
153        do {
154            byte temp = (byte) (value & 0b01111111);
155            // Note: >>> means that the sign bit is shifted with the rest of the number rather than being left alone
156            value >>>= 7;
157            if (value != 0) {
158                temp |= 0b10000000;
159            }
160            stream.putByte(temp);
161        } while (value != 0);
162    }
163
164    private static void write(OutputStream stream, long value) throws IOException {
165        do {
166            byte temp = (byte) (value & 0b01111111);
167            // Note: >>> means that the sign bit is shifted with the rest of the number rather than being left alone
168            value >>>= 7;
169            if (value != 0) {
170                temp |= 0b10000000;
171            }
172            stream.write(temp);
173        } while (value != 0);
174    }
