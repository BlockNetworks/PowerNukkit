020import java.util.List;
021import java.util.concurrent.ThreadLocalRandom;
022
023public class RepairItemTransaction extends InventoryTransaction {
024
025    private Item inputItem;
026    private Item materialItem;
027    private Item outputItem;
028
029    private int cost;
030
031    public RepairItemTransaction(Player source, List<InventoryAction> actions) {
032        super(source, actions);
033    }
034
035    @Override
036    public boolean canExecute() {
037        Inventory inventory = getSource().getWindowById(Player.ANVIL_WINDOW_ID);
038        if (inventory == null) {
039            return false;
040        }
041        AnvilInventory anvilInventory = (AnvilInventory) inventory;
042        return this.inputItem != null && this.outputItem != null && this.inputItem.equals(anvilInventory.getInputSlot(), true, true)
043                && (!this.hasMaterial() || this.materialItem.equals(anvilInventory.getMaterialSlot(), true, true))
044                && this.checkRecipeValid();
045    }
046
047    @Override
048    public boolean execute() {
049        if (this.hasExecuted() || !this.canExecute()) {
050            this.source.removeAllWindows(false);
051            this.sendInventories();
052            return false;
053        }
054        AnvilInventory inventory = (AnvilInventory) getSource().getWindowById(Player.ANVIL_WINDOW_ID);
055
056        if (inventory.getCost() != this.cost && !this.source.isCreative()) {
057            this.source.getServer().getLogger().debug("Got unexpected cost " + inventory.getCost() + " from " + this.source.getName() + "(expected " + this.cost + ")");
058        }
059
060        RepairItemEvent event = new RepairItemEvent(inventory, this.inputItem, this.outputItem, this.materialItem, this.cost, this.source);
061        this.source.getServer().getPluginManager().callEvent(event);
062        if (event.isCancelled()) {
063            this.source.removeAllWindows(false);
064            this.sendInventories();
065            return true;
066        }
067
068        for (InventoryAction action : this.actions) {
069            if (action.execute(this.source)) {
070                action.onExecuteSuccess(this.source);
071            } else {
072                action.onExecuteFail(this.source);
073            }
074        }
075
076        FakeBlockMenu holder = inventory.getHolder();
077        Block block = this.source.level.getBlock(holder.getFloorX(), holder.getFloorY(), holder.getFloorZ());
078        if (block.getId() == Block.ANVIL) {
079            int oldDamage = block.getDamage() >= 8 ? 2 : block.getDamage() >= 4 ? 1 : 0;
080            int newDamage = !this.source.isCreative() && ThreadLocalRandom.current().nextInt(100) < 12 ? oldDamage + 1 : oldDamage;
081
082            AnvilDamageEvent ev = new AnvilDamageEvent(block, oldDamage, newDamage, DamageCause.USE, this.source);
083            ev.setCancelled(oldDamage == newDamage);
084            this.source.getServer().getPluginManager().callEvent(ev);
085            if (!ev.isCancelled()) {
086                newDamage = ev.getNewDamage();
087                if (newDamage > 2) {
088                    this.source.level.setBlock(block, Block.get(Block.AIR), true);
089                    this.source.level.addLevelEvent(block, LevelEventPacket.EVENT_SOUND_ANVIL_BREAK);
090                } else {
091                    if (newDamage < 0) {
092                        newDamage = 0;
093                    }
094                    if (newDamage != oldDamage) {
095                        block.setDamage((newDamage << 2) | (block.getDamage() & 0x3));
096                        this.source.level.setBlock(block, block, true);
097                    }
098                    this.source.level.addLevelEvent(block, LevelEventPacket.EVENT_SOUND_ANVIL_USE);
099                }
100            } else {
101                this.source.level.addLevelEvent(block, LevelEventPacket.EVENT_SOUND_ANVIL_USE);
102            }
103        }
104
105        if (!this.source.isCreative()) {
106            this.source.setExperience(this.source.getExperience(), this.source.getExperienceLevel() - event.getCost());
107        }
108        return true;
109    }
110
111    @Override
112    public void addAction(InventoryAction action) {
113        super.addAction(action);
114        if (action instanceof RepairItemAction) {
115            switch (((RepairItemAction) action).getType()) {
116                case NetworkInventoryAction.SOURCE_TYPE_ANVIL_INPUT:
117                    this.inputItem = action.getTargetItem();
118                    break;
119                case NetworkInventoryAction.SOURCE_TYPE_ANVIL_RESULT:
120                    this.outputItem = action.getSourceItem();
121                    break;
122                case NetworkInventoryAction.SOURCE_TYPE_ANVIL_MATERIAL:
123                    this.materialItem = action.getTargetItem();
124                    break;
125            }
126        }
127    }

129    private boolean checkRecipeValid() {
130        int cost = 0;
131        int baseRepairCost = this.inputItem.getRepairCost();
132
133        if (this.isMapRecipe()) {
134            if (!this.matchMapRecipe()) {
135                return false;
136            }
137            baseRepairCost = 0;
138        } else if (this.hasMaterial()) {
139            baseRepairCost += this.materialItem.getRepairCost();
140
141            if (this.inputItem.getMaxDurability() != -1 && this.matchRepairItem()) {
142                int maxRepairDamage = this.inputItem.getMaxDurability() / 4;
143                int repairDamage = Math.min(this.inputItem.getDamage(), maxRepairDamage);
144                if (repairDamage <= 0) {
145                    return false;
146                }
147
148                int damage = this.inputItem.getDamage();
149                for (; repairDamage > 0 && cost < this.materialItem.getCount(); cost++) {
150                    damage = damage - repairDamage;
151                    repairDamage = Math.min(damage, maxRepairDamage);
152                }
153                if (this.outputItem.getDamage() != damage) {
154                    return false;
155                }
156            } else {
157                boolean consumeEnchantedBook = this.materialItem.getId() == Item.ENCHANTED_BOOK && this.materialItem.hasEnchantments();
158                if (!consumeEnchantedBook && (this.inputItem.getMaxDurability() == -1 || this.inputItem.getId() != this.materialItem.getId())) {
159                    return false;
160                }
161
162                if (!consumeEnchantedBook && this.inputItem.getMaxDurability() != -1) {
163                    int damage = this.inputItem.getDamage() - this.inputItem.getMaxDurability() + this.materialItem.getDamage() - this.inputItem.getMaxDurability() * 12 / 100 + 1;
164                    if (damage < 0) {
165                        damage = 0;
166                    }
167
168                    if (damage < this.inputItem.getDamage()) {
169                        if (this.outputItem.getDamage() != damage) {
170                            return false;
171                        }
172                        cost += 2;
173                    }
174                }
175
176                Int2IntMap enchantments = new Int2IntOpenHashMap();
177                enchantments.defaultReturnValue(-1);
178                for (Enchantment enchantment : this.inputItem.getEnchantments()) {
179                    enchantments.put(enchantment.getId(), enchantment.getLevel());
180                }
181
182                boolean hasCompatibleEnchantments = false;
183                boolean hasIncompatibleEnchantments = false;
184                for (Enchantment materialEnchantment : this.materialItem.getEnchantments()) {
185                    Enchantment enchantment = this.inputItem.getEnchantment(materialEnchantment.getId());
186                    int inputLevel = enchantment != null ? enchantment.getLevel() : 0;
187                    int materialLevel = materialEnchantment.getLevel();
188                    int outputLevel = inputLevel == materialLevel ? materialLevel + 1 : Math.max(materialLevel, inputLevel);
189
190                    boolean canEnchant = materialEnchantment.canEnchant(this.inputItem) || this.inputItem.getId() == Item.ENCHANTED_BOOK;
191                    for (Enchantment inputEnchantment : this.inputItem.getEnchantments()) {
192                        if (inputEnchantment.getId() != materialEnchantment.getId() && !materialEnchantment.isCompatibleWith(inputEnchantment)) {
193                            canEnchant = false;
194                            cost++;
195                        }
196                    }
197
198                    if (!canEnchant) {
199                        hasIncompatibleEnchantments = true;
200                    } else {
201                        hasCompatibleEnchantments = true;
202                        if (outputLevel > materialEnchantment.getMaxLevel()) {
203                            outputLevel = materialEnchantment.getMaxLevel();
204                        }
205
206                        enchantments.put(materialEnchantment.getId(), outputLevel);
207                        int rarityFactor;
208                        switch (materialEnchantment.getRarity()) {
209                            case COMMON:
210                                rarityFactor = 1;
211                                break;
212                            case UNCOMMON:
213                                rarityFactor = 2;
214                                break;
215                            case RARE:
216                                rarityFactor = 4;
217                                break;
218                            case VERY_RARE:
219                            default:
220                                rarityFactor = 8;
221                                break;
222                        }
223
224                        if (consumeEnchantedBook) {
225                            rarityFactor = Math.max(1, rarityFactor / 2);
226                        }
227
228                        cost += rarityFactor * Math.max(0, outputLevel - inputLevel);
229                        if (this.inputItem.getCount() > 1) {
230                            cost = 40;
231                        }
232                    }
233                }
234
235                Enchantment[] outputEnchantments = this.outputItem.getEnchantments();
236                if (hasIncompatibleEnchantments && !hasCompatibleEnchantments || enchantments.size() != outputEnchantments.length) {
237                    return false;
238                }
239
240                for (Enchantment enchantment : outputEnchantments) {
241                    if (enchantments.get(enchantment.getId()) != enchantment.getLevel()) {
242                        return false;
243                    }
244                }
245            }
246        }
247
248        int renameCost = 0;
249        if (!this.inputItem.getCustomName().equals(this.outputItem.getCustomName())) {
250            if (this.outputItem.getCustomName().length() > 30) {
251                return false;
252            }
253            renameCost = 1;
254            cost += renameCost;
255        }
256
257        this.cost = baseRepairCost + cost;
258        if (renameCost == cost && renameCost > 0 && this.cost >= 40) {
259            this.cost = 39;
260        }
261        if (baseRepairCost < 0 || cost < 0 || cost == 0 && !this.isMapRecipe() || this.cost > 39 && !this.source.isCreative()) {
262            return false;
263        }
264
265        int nextBaseRepairCost = this.inputItem.getRepairCost();
266        if (!this.isMapRecipe()) {
267            if (this.hasMaterial() && nextBaseRepairCost < this.materialItem.getRepairCost()) {
268                nextBaseRepairCost = this.materialItem.getRepairCost();
269            }
270            if (renameCost == 0 || renameCost != cost) {
271                nextBaseRepairCost = 2 * nextBaseRepairCost + 1;
272            }
273        }
274        if (this.outputItem.getRepairCost() != nextBaseRepairCost) {
275            this.source.getServer().getLogger().debug("Got unexpected base cost " + this.outputItem.getRepairCost() + " from " + this.source.getName() + "(expected " + nextBaseRepairCost + ")");
276            return false;
277        }
278
279        return true;
280    }
281
282    private boolean hasMaterial() {
283        return this.materialItem != null && !this.materialItem.isNull();
284    }
285
286    private boolean isMapRecipe() {
287        return this.hasMaterial() && (this.inputItem.getId() == Item.MAP || this.inputItem.getId() == Item.EMPTY_MAP)
288                && (this.materialItem.getId() == Item.EMPTY_MAP || this.materialItem.getId() == Item.PAPER || this.materialItem.getId() == Item.COMPASS);
289    }
290
291    private boolean matchMapRecipe() {
292        if (this.inputItem.getId() == Item.EMPTY_MAP) {
293            return this.inputItem.getDamage() != 2 && this.materialItem.getId() == Item.COMPASS // locator
294                    && this.outputItem.getId() == Item.EMPTY_MAP && this.outputItem.getDamage() == 2 && this.outputItem.getCount() == 1;
295        } else if (this.inputItem.getId() == Item.MAP && this.outputItem.getDamage() == this.inputItem.getDamage()) {
296            if (this.materialItem.getId() == Item.COMPASS) { // locator
297                return this.inputItem.getDamage() != 2 && this.outputItem.getId() == Item.MAP && this.outputItem.getCount() == 1;
298            } else if (this.materialItem.getId() == Item.EMPTY_MAP) { // clone
299                return this.outputItem.getId() == Item.MAP && this.outputItem.getCount() == 2;
300            } else if (this.materialItem.getId() == Item.PAPER && this.materialItem.getCount() >= 8) { // zoom out
301                return this.inputItem.getDamage() < 3 && this.outputItem.getId() == Item.MAP && this.outputItem.getCount() == 1;
302            }
303        }
304        return false;
305    }

307    private boolean matchRepairItem() {
308        switch (this.inputItem.getId()) {
309            case Item.LEATHER_CAP:
310            case Item.LEATHER_TUNIC:
311            case Item.LEATHER_PANTS:
312            case Item.LEATHER_BOOTS:
313                return this.materialItem.getId() == Item.LEATHER;
314            case Item.WOODEN_SWORD:
315            case Item.WOODEN_PICKAXE:
316            case Item.WOODEN_SHOVEL:
317            case Item.WOODEN_AXE:
318            case Item.WOODEN_HOE:
319                return this.materialItem.getId() == Item.PLANKS;
320            case Item.IRON_SWORD:
321            case Item.IRON_PICKAXE:
322            case Item.IRON_SHOVEL:
323            case Item.IRON_AXE:
324            case Item.IRON_HOE:
325            case Item.IRON_HELMET:
326            case Item.IRON_CHESTPLATE:
327            case Item.IRON_LEGGINGS:
328            case Item.IRON_BOOTS:
329            case Item.CHAIN_HELMET:
330            case Item.CHAIN_CHESTPLATE:
331            case Item.CHAIN_LEGGINGS:
332            case Item.CHAIN_BOOTS:
333                return this.materialItem.getId() == Item.IRON_INGOT;
334            case Item.GOLD_SWORD:
335            case Item.GOLD_PICKAXE:
336            case Item.GOLD_SHOVEL:
337            case Item.GOLD_AXE:
338            case Item.GOLD_HOE:
339            case Item.GOLD_HELMET:
340            case Item.GOLD_CHESTPLATE:
341            case Item.GOLD_LEGGINGS:
342            case Item.GOLD_BOOTS:
343                return this.materialItem.getId() == Item.GOLD_INGOT;
344            case Item.DIAMOND_SWORD:
345            case Item.DIAMOND_PICKAXE:
346            case Item.DIAMOND_SHOVEL:
347            case Item.DIAMOND_AXE:
348            case Item.DIAMOND_HOE:
349            case Item.DIAMOND_HELMET:
350            case Item.DIAMOND_CHESTPLATE:
351            case Item.DIAMOND_LEGGINGS:
352            case Item.DIAMOND_BOOTS:
353                return this.materialItem.getId() == Item.DIAMOND;
354            case Item.NETHERITE_SWORD:
355            case Item.NETHERITE_PICKAXE:
356            case Item.NETHERITE_SHOVEL:
357            case Item.NETHERITE_AXE:
358            case Item.NETHERITE_HOE:
359            case Item.NETHERITE_HELMET:
360            case Item.NETHERITE_CHESTPLATE:
361            case Item.NETHERITE_LEGGINGS:
362            case Item.NETHERITE_BOOTS:
363                return this.materialItem.getId() == Item.NETHERITE_INGOT;
364            case Item.TURTLE_SHELL:
365                return this.materialItem.getId() == Item.SCUTE;
366            case Item.ELYTRA:
367                return this.materialItem.getId() == Item.PHANTOM_MEMBRANE;
368        }
369        return false;
370    }

388    public static boolean checkForRepairItemPart(List<InventoryAction> actions) {
389        for (InventoryAction action : actions) {
390            if (action instanceof RepairItemAction) {
391                return true;
392            }
393        }
394        return false;
