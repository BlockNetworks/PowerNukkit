044    public static Item getFishingResult(Item rod) {
045        int fortuneLevel = 0;
046        int lureLevel = 0;
047        if (rod != null) {
048            if (rod.getEnchantment(Enchantment.ID_FORTUNE_FISHING) != null) {
049                fortuneLevel = rod.getEnchantment(Enchantment.ID_FORTUNE_FISHING).getLevel();
050            } else if (rod.getEnchantment(Enchantment.ID_LURE) != null) {
051                lureLevel = rod.getEnchantment(Enchantment.ID_LURE).getLevel();
052            }
053        }
054        return getFishingResult(fortuneLevel, lureLevel);
055    }
056
057    public static Item getFishingResult(int fortuneLevel, int lureLevel) {
058        float treasureChance = NukkitMath.clamp(0.05f + 0.01f * fortuneLevel - 0.01f * lureLevel, 0, 1);
059        float junkChance = NukkitMath.clamp(0.05f - 0.025f * fortuneLevel - 0.01f * lureLevel, 0, 1);
060        float fishChance = NukkitMath.clamp(1 - treasureChance - junkChance, 0, 1);
061        putSelector(FISHES, fishChance);
062        putSelector(TREASURES, treasureChance);
063        putSelector(JUNKS, junkChance);
064        Object result = selectFrom(ROOT_FISHING);
065        if (result instanceof Item) return (Item) result;
066        return null;
067    }
