001package cn.nukkit.item;
002
003import cn.nukkit.block.Block;
004import cn.nukkit.entity.Entity;
005import cn.nukkit.item.enchantment.Enchantment;
006import cn.nukkit.nbt.tag.ByteTag;
007import cn.nukkit.nbt.tag.Tag;
008
009import java.util.Random;
010
011/**
012 * author: MagicDroidX
013 * Nukkit Project
014 */
015public abstract class ItemTool extends Item implements ItemDurable {
016    public static final int TIER_WOODEN = 1;
017    public static final int TIER_GOLD = 2;
018    public static final int TIER_STONE = 3;
019    public static final int TIER_IRON = 4;
020    public static final int TIER_DIAMOND = 5;
021    public static final int TIER_NETHERITE = 6;
022
023    public static final int TYPE_NONE = 0;
024    public static final int TYPE_SWORD = 1;
025    public static final int TYPE_SHOVEL = 2;
026    public static final int TYPE_PICKAXE = 3;
027    public static final int TYPE_AXE = 4;
028    public static final int TYPE_SHEARS = 5;
029    public static final int TYPE_HOE = 6;
030
031    public static final int DURABILITY_WOODEN = 60;
032    public static final int DURABILITY_GOLD = 33;
033    public static final int DURABILITY_STONE = 132;
034    public static final int DURABILITY_IRON = 251;
035    public static final int DURABILITY_DIAMOND = 1562;
036    public static final int DURABILITY_NETHERITE = 2032;
037    public static final int DURABILITY_FLINT_STEEL = 65;
038    public static final int DURABILITY_SHEARS = 239;
039    public static final int DURABILITY_BOW = 385;
040    public static final int DURABILITY_TRIDENT = 251;
041    public static final int DURABILITY_FISHING_ROD = 65;
042    public static final int DURABILITY_CROSSBOW = 465;
043    public static final int DURABILITY_CARROT_ON_A_STICK = 25;
044    public static final int DURABILITY_WARPED_FUNGUS_ON_A_STICK = 100;
045
046    public ItemTool(int id) {
047        this(id, 0, 1, UNKNOWN_STR);
048    }
049
050    public ItemTool(int id, Integer meta) {
051        this(id, meta, 1, UNKNOWN_STR);
052    }
053
054    public ItemTool(int id, Integer meta, int count) {
055        this(id, meta, count, UNKNOWN_STR);
056    }
057
058    public ItemTool(int id, Integer meta, int count, String name) {
059        super(id, meta, count, name);
060    }
061
062    @Override
063    public int getMaxStackSize() {
064        return 1;
065    }
066
067    @Override
068    public boolean useOn(Block block) {
069        if (this.isUnbreakable() || isDurable() || noDamageOnBreak()) {
070            return true;
071        }
072
073        if (block.getToolType() == ItemTool.TYPE_PICKAXE && this.isPickaxe() ||
074                block.getToolType() == ItemTool.TYPE_SHOVEL && this.isShovel() ||
075                block.getToolType() == ItemTool.TYPE_AXE && this.isAxe() ||
076                block.getToolType() == ItemTool.TYPE_HOE && this.isHoe() ||
077                block.getToolType() == ItemTool.TYPE_SWORD && this.isSword() ||
078                block.getToolType() == ItemTool.TYPE_SHEARS && this.isShears()
079                ) {
080            this.meta++;
081        } else if (!this.isShears() && block.getBreakTime(this) > 0) {
082            this.meta += 2;
083        } else if (this.isHoe()) {
084            if (block.getId() == GRASS || block.getId() == DIRT) {
085                this.meta++;
086            }
087        } else {
088            this.meta++;
089        }
090        return true;
091    }
092
093    @Override
094    public boolean useOn(Entity entity) {
095        if (this.isUnbreakable() || isDurable() || noDamageOnAttack()) {
096            return true;
097        }
098
099        if ((entity != null) && !this.isSword()) {
100            this.meta += 2;
101        } else {
102            this.meta++;
103        }
104
105        return true;
106    }
107
108    private boolean isDurable() {
109        if (!hasEnchantments()) {
110            return false;
111        }
112
113        Enchantment durability = getEnchantment(Enchantment.ID_DURABILITY);
114        return durability != null && durability.getLevel() > 0 && (100 / (durability.getLevel() + 1)) <= new Random().nextInt(100);
115    }
116
117    @Override
118    public boolean isUnbreakable() {
119        Tag tag = this.getNamedTagEntry("Unbreakable");
120        return tag instanceof ByteTag && ((ByteTag) tag).data > 0;
121    }
122
123    @Override
124    public boolean isPickaxe() {
125        return false;
126    }
127
128    @Override
129    public boolean isAxe() {
130        return false;
131    }
132
133    @Override
134    public boolean isSword() {
135        return false;
136    }
137
138    @Override
139    public boolean isShovel() {
140        return false;
141    }
142
143    @Override
144    public boolean isHoe() {
145        return false;
146    }
147
148    @Override
149    public boolean isShears() {
150        return (this.id == SHEARS);
151    }
152
153    @Override
154    public boolean isTool() {
155        return true;
156    }
157
158    @Override
159    public int getEnchantAbility() {
160        switch (this.getTier()) {
161            case TIER_STONE:
162                return 5;
163            case TIER_WOODEN:
164                return 15;
165            case TIER_DIAMOND:
166                return 10;
167            case TIER_GOLD:
168                return 22;
169            case TIER_IRON:
170                return 14;
171        }
172
173        return 0;
174    }
175
176    /**
177     * No damage to item when it's used to attack entities
178     * @return whether the item should take damage when used to attack entities
179     */
180    public boolean noDamageOnAttack() {
181        return false;
182    }
183
184    /**
185     * No damage to item when it's used to break blocks
186     * @return whether the item should take damage when used to break blocks
187     */
188    public boolean noDamageOnBreak() {
189        return false;
190    }
191}
