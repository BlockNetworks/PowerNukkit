001package cn.nukkit.block;
002
003import cn.nukkit.Player;
004import cn.nukkit.Server;
005import cn.nukkit.event.block.BlockGrowEvent;
006import cn.nukkit.item.Item;
007import cn.nukkit.item.ItemSugarcane;
008import cn.nukkit.level.Level;
009import cn.nukkit.level.particle.BoneMealParticle;
010import cn.nukkit.math.BlockFace;
011import cn.nukkit.math.Vector3;
012import cn.nukkit.utils.BlockColor;
013
014/**
015 * Created by Pub4Game on 09.01.2016.
016 */
017public class BlockSugarcane extends BlockFlowable {
018
019    public BlockSugarcane() {
020        this(0);
021    }
022
023    public BlockSugarcane(int meta) {
024        super(meta);
025    }
026
027    @Override
028    public String getName() {
029        return "Sugarcane";
030    }
031
032    @Override
033    public int getId() {
034        return SUGARCANE_BLOCK;
035    }
036
037    @Override
038    public Item toItem() {
039        return new ItemSugarcane();
040    }
041
042    @Override
043    public boolean canBeActivated() {
044        return true;
045    }
046
047    @Override
048    public boolean onActivate(Item item, Player player) {
049        if (item.getId() == Item.DYE && item.getDamage() == 0x0F) { //Bonemeal
050            int count = 1;
051
052            for (int i = 1; i <= 2; i++) {
053                int id = this.level.getBlockIdAt(this.getFloorX(), this.getFloorY() - i, this.getFloorZ());
054
055                if (id == SUGARCANE_BLOCK) {
056                    count++;
057                }
058            }
059
060            if (count < 3) {
061                boolean success = false;
062                int toGrow = 3 - count;
063
064                for (int i = 1; i <= toGrow; i++) {
065                    Block block = this.up(i);
066                    if (block.getId() == 0) {
067                        BlockGrowEvent ev = new BlockGrowEvent(block, Block.get(BlockID.SUGARCANE_BLOCK));
068                        Server.getInstance().getPluginManager().callEvent(ev);
069
070                        if (!ev.isCancelled()) {
071                            this.getLevel().setBlock(block, ev.getNewState(), true);
072                            success = true;
073                        }
074                    } else if (block.getId() != SUGARCANE_BLOCK) {
075                        break;
076                    }
077                }
078
079                if (success) {
080                    if (player != null && (player.gamemode & 0x01) == 0) {
081                        item.count--;
082                    }
083
084                    this.level.addParticle(new BoneMealParticle(this));
085                }
086            }
087
088            return true;
089        }
090        return false;
091    }
092
093    @Override
094    public int onUpdate(int type) {
095        if (type == Level.BLOCK_UPDATE_NORMAL) {
096            Block down = this.down();
097            if (down.isTransparent() && down.getId() != SUGARCANE_BLOCK) {
098                this.getLevel().useBreakOn(this);
099                return Level.BLOCK_UPDATE_NORMAL;
100            }
101        } else if (type == Level.BLOCK_UPDATE_RANDOM) {
102            if (this.down().getId() != SUGARCANE_BLOCK) {
103                if (this.getDamage() == 0x0F) {
104                    for (int y = 1; y < 3; ++y) {
105                        Block b = this.getLevel().getBlock(new Vector3(this.x, this.y + y, this.z));
106                        if (b.getId() == AIR) {
107                            BlockGrowEvent ev = new BlockGrowEvent(b, Block.get(BlockID.SUGARCANE_BLOCK));
108                            Server.getInstance().getPluginManager().callEvent(ev);
109                            
110                            if (!ev.isCancelled()) {
111                                this.getLevel().setBlock(b, Block.get(BlockID.SUGARCANE_BLOCK), false);
112                            }
113                            break;
114                        }
115                    }
116                    this.setDamage(0);
117                    this.getLevel().setBlock(this, this, false);
118                } else {
119                    this.setDamage(this.getDamage() + 1);
120                    this.getLevel().setBlock(this, this, false);
121                }
122                return Level.BLOCK_UPDATE_RANDOM;
123            }
124        }
125        return 0;
126    }
127
128    @Override
129    public boolean place(Item item, Block block, Block target, BlockFace face, double fx, double fy, double fz, Player player) {
130        if (block.getId() != AIR) {
131            return false;
132        }
133        Block down = this.down();
134        if (down.getId() == SUGARCANE_BLOCK) {
135            this.getLevel().setBlock(block, Block.get(BlockID.SUGARCANE_BLOCK), true);
136            return true;
137        } else if (down.getId() == GRASS || down.getId() == DIRT || down.getId() == SAND || down.getId() == PODZOL) {
138            Block block0 = down.north();
139            Block block1 = down.south();
140            Block block2 = down.west();
141            Block block3 = down.east();
142            if ((block0 instanceof BlockWater) || (block1 instanceof BlockWater) || (block2 instanceof BlockWater) || (block3 instanceof BlockWater)) {
143                this.getLevel().setBlock(block, Block.get(BlockID.SUGARCANE_BLOCK), true);
144                return true;
145            }
146        }
147        return false;
148    }
149
150    @Override
151    public BlockColor getColor() {
152        return BlockColor.FOLIAGE_BLOCK_COLOR;
153    }
154}
