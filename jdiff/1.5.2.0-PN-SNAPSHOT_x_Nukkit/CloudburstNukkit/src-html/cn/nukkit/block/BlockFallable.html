001package cn.nukkit.block;
002
003import cn.nukkit.entity.Entity;
004import cn.nukkit.entity.item.EntityFallingBlock;
005import cn.nukkit.event.block.BlockFallEvent;
006import cn.nukkit.level.Level;
007import cn.nukkit.nbt.tag.CompoundTag;
008import cn.nukkit.nbt.tag.DoubleTag;
009import cn.nukkit.nbt.tag.FloatTag;
010import cn.nukkit.nbt.tag.ListTag;
011
012
013/**
014 * author: rcsuperman
015 * Nukkit Project
016 */
017public abstract class BlockFallable extends BlockSolid {
018
019    protected BlockFallable() {
020    }
021
022    public int onUpdate(int type) {
023        if (type == Level.BLOCK_UPDATE_NORMAL) {
024            Block down = this.down();
025            if (down.getId() == AIR || down instanceof BlockLiquid || down instanceof BlockFire) {
026                BlockFallEvent event = new BlockFallEvent(this);
027                this.level.getServer().getPluginManager().callEvent(event);
028                if (event.isCancelled()) {
029                    return type;
030                }
031
032                this.level.setBlock(this, Block.get(Block.AIR), true, true);
033                CompoundTag nbt = new CompoundTag()
034                        .putList(new ListTag<DoubleTag>("Pos")
035                                .add(new DoubleTag("", this.x + 0.5))
036                                .add(new DoubleTag("", this.y))
037                                .add(new DoubleTag("", this.z + 0.5)))
038                        .putList(new ListTag<DoubleTag>("Motion")
039                                .add(new DoubleTag("", 0))
040                                .add(new DoubleTag("", 0))
041                                .add(new DoubleTag("", 0)))
042
043                        .putList(new ListTag<FloatTag>("Rotation")
044                                .add(new FloatTag("", 0))
045                                .add(new FloatTag("", 0)))
046                        .putInt("TileID", this.getId())
047                        .putByte("Data", this.getDamage());
048
049                EntityFallingBlock fall = (EntityFallingBlock) Entity.createEntity("FallingSand", this.getLevel().getChunk((int) this.x >> 4, (int) this.z >> 4), nbt);
050
051                if (fall != null) {
052                    fall.spawnToAll();
053                }
054            }
055        }
056        return type;
057    }
058}
