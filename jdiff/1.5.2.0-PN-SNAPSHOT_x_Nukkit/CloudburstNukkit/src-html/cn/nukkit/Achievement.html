001package cn.nukkit;
002
003import cn.nukkit.utils.TextFormat;
004
005import java.util.HashMap;
006
007/**
008 * Created by CreeperFace on 9. 11. 2016.
009 */
010public class Achievement {
011
012    public static final HashMap<String, Achievement> achievements = new HashMap<String, Achievement>() {
013        {
014            put("mineWood", new Achievement("Getting Wood"));
015            put("buildWorkBench", new Achievement("Benchmarking", "mineWood"));
016            put("buildPickaxe", new Achievement("Time to Mine!", "buildWorkBench"));
017            put("buildFurnace", new Achievement("Hot Topic", "buildPickaxe"));
018            put("acquireIron", new Achievement("Acquire hardware", "buildFurnace"));
019            put("buildHoe", new Achievement("Time to Farm!", "buildWorkBench"));
020            put("makeBread", new Achievement("Bake Bread", "buildHoe"));
021            put("bakeCake", new Achievement("The Lie", "buildHoe"));
022            put("buildBetterPickaxe", new Achievement("Getting an Upgrade", "buildPickaxe"));
023            put("buildSword", new Achievement("Time to Strike!", "buildWorkBench"));
024            put("diamonds", new Achievement("DIAMONDS!", "acquireIron"));
025        }
026    };
027
028    public static boolean broadcast(Player player, String achievementId) {
029        if (!achievements.containsKey(achievementId)) {
030            return false;
031        }
032        String translation = Server.getInstance().getLanguage().translateString("chat.type.achievement", player.getDisplayName(), TextFormat.GREEN + achievements.get(achievementId).getMessage() + TextFormat.RESET);
033
034        if (Server.getInstance().getPropertyBoolean("announce-player-achievements", true)) {
035            Server.getInstance().broadcastMessage(translation);
036        } else {
037            player.sendMessage(translation);
038        }
039        return true;
040    }
041
042    public static boolean add(String name, Achievement achievement) {
043        if (achievements.containsKey(name)) {
044            return false;
045        }
046
047        achievements.put(name, achievement);
048        return true;
049    }
050
051    public final String message;
052    public final String[] requires;
053
054    public Achievement(String message, String... requires) {
055        this.message = message;
056        this.requires = requires;
057    }
058
059    public String getMessage() {
060        return message;
061    }
062
063    public void broadcast(Player player) {
064        String translation = Server.getInstance().getLanguage().translateString("chat.type.achievement", player.getDisplayName(), TextFormat.GREEN + this.getMessage(), null);
065
066        if (Server.getInstance().getPropertyBoolean("announce-player-achievements", true)) {
067            Server.getInstance().broadcastMessage(translation);
068        } else {
069            player.sendMessage(translation);
070        }
071    }
072}
