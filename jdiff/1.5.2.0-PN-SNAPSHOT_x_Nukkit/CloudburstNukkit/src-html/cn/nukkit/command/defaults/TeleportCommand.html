001package cn.nukkit.command.defaults;
002
003import cn.nukkit.Player;
004import cn.nukkit.command.Command;
005import cn.nukkit.command.CommandSender;
006import cn.nukkit.command.data.CommandParamType;
007import cn.nukkit.command.data.CommandParameter;
008import cn.nukkit.event.player.PlayerTeleportEvent;
009import cn.nukkit.lang.TranslationContainer;
010import cn.nukkit.level.Location;
011import cn.nukkit.math.NukkitMath;
012import cn.nukkit.utils.TextFormat;
013
014/**
015 * Created on 2015/11/12 by Pub4Game and milkice.
016 * Package cn.nukkit.command.defaults in project Nukkit .
017 */
018public class TeleportCommand extends VanillaCommand {
019    public TeleportCommand(String name) {
020        super(name, "%nukkit.command.tp.description", "%commands.tp.usage");
021        this.setPermission("nukkit.command.teleport");
022        this.commandParameters.clear();
023        this.commandParameters.put("->Player", new CommandParameter[]{
024                CommandParameter.newType("destination", CommandParamType.TARGET),
025        });
026        this.commandParameters.put("Player->Player", new CommandParameter[]{
027                CommandParameter.newType("victim", CommandParamType.TARGET),
028                CommandParameter.newType("destination", CommandParamType.TARGET)
029        });
030        this.commandParameters.put("Player->Pos", new CommandParameter[]{
031                CommandParameter.newType("victim", CommandParamType.TARGET),
032                CommandParameter.newType("destination", CommandParamType.POSITION),
033                CommandParameter.newType("yRot", true, CommandParamType.VALUE),
034                CommandParameter.newType("xRot", true, CommandParamType.VALUE)
035        });
036        this.commandParameters.put("->Pos", new CommandParameter[]{
037                CommandParameter.newType("destination", CommandParamType.POSITION),
038                CommandParameter.newType("yRot", true, CommandParamType.VALUE),
039                CommandParameter.newType("xRot", true, CommandParamType.VALUE)
040        });
041    }
042
043    @Override
044    public boolean execute(CommandSender sender, String commandLabel, String[] args) {
045        if (!this.testPermission(sender)) {
046            return true;
047        }
048        if (args.length < 1 || args.length > 6) {
049            sender.sendMessage(new TranslationContainer("commands.generic.usage", this.usageMessage));
050            return true;
051        }
052        CommandSender target;
053        CommandSender origin = sender;
054        if (args.length == 1 || args.length == 3) {
055            if (sender instanceof Player) {
056                target = sender;
057            } else {
058                sender.sendMessage(new TranslationContainer("commands.generic.ingame"));
059                return true;
060            }
061            if (args.length == 1) {
062                target = sender.getServer().getPlayer(args[0].replace("@s", sender.getName()));
063                if (target == null) {
064                    sender.sendMessage(TextFormat.RED + "Can't find player " + args[0]);
065                    return true;
066                }
067            }
068        } else {
069            target = sender.getServer().getPlayer(args[0].replace("@s", sender.getName()));
070            if (target == null) {
071                sender.sendMessage(TextFormat.RED + "Can't find player " + args[0]);
072                return true;
073            }
074            if (args.length == 2) {
075                origin = target;
076                target = sender.getServer().getPlayer(args[1].replace("@s", sender.getName()));
077                if (target == null) {
078                    sender.sendMessage(TextFormat.RED + "Can't find player " + args[1]);
079                    return true;
080                }
081            }
082        }
083        if (args.length < 3) {
084            ((Player) origin).teleport((Player) target, PlayerTeleportEvent.TeleportCause.COMMAND);
085            Command.broadcastCommandMessage(sender, new TranslationContainer("commands.tp.success", origin.getName(), target.getName()));
086            return true;
087        } else if (((Player) target).getLevel() != null) {
088            int pos;
089            if (args.length == 4 || args.length == 6) {
090                pos = 1;
091            } else {
092                pos = 0;
093            }
094            double x;
095            double y;
096            double z;
097            double yaw;
098            double pitch;
099            try {
100                x = Double.parseDouble(args[pos++].replace("~", "" + ((Player) target).x));
101                y = Double.parseDouble(args[pos++].replace("~", "" + ((Player) target).y));
102                z = Double.parseDouble(args[pos++].replace("~", "" + ((Player) target).z));
103                yaw = ((Player) target).getYaw();
104                pitch = ((Player) target).getPitch();
105            } catch (NumberFormatException e1) {
106                sender.sendMessage(new TranslationContainer("commands.generic.usage", this.usageMessage));
107                return true;
108            }
109            if (y < 0) y = 0;
110            if (y > 256) y = 256;
111            if (args.length == 6 || (args.length == 5 && pos == 3)) {
112                yaw = Integer.parseInt(args[pos++]);
113                pitch = Integer.parseInt(args[pos++]);
114            }
115            ((Player) target).teleport(new Location(x, y, z, yaw, pitch, ((Player) target).getLevel()), PlayerTeleportEvent.TeleportCause.COMMAND);
116            Command.broadcastCommandMessage(sender, new TranslationContainer("commands.tp.success.coordinates", target.getName(), String.valueOf(NukkitMath.round(x, 2)), String.valueOf(NukkitMath.round(y, 2)), String.valueOf(NukkitMath.round(z, 2))));
117            return true;
118        }
119        sender.sendMessage(new TranslationContainer("commands.generic.usage", this.usageMessage));
120        return true;
121    }
122}
