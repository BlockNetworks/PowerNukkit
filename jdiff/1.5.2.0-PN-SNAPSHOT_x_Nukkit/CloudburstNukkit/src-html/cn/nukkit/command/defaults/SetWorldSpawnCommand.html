001package cn.nukkit.command.defaults;
002
003import cn.nukkit.Player;
004import cn.nukkit.command.Command;
005import cn.nukkit.command.CommandSender;
006import cn.nukkit.command.data.CommandParamType;
007import cn.nukkit.command.data.CommandParameter;
008import cn.nukkit.lang.TranslationContainer;
009import cn.nukkit.level.Level;
010import cn.nukkit.math.Vector3;
011
012import java.text.DecimalFormat;
013
014/**
015 * Created on 2015/12/13 by xtypr.
016 * Package cn.nukkit.command.defaults in project Nukkit .
017 */
018public class SetWorldSpawnCommand extends VanillaCommand {
019    public SetWorldSpawnCommand(String name) {
020        super(name, "%nukkit.command.setworldspawn.description", "%commands.setworldspawn.usage");
021        this.setPermission("nukkit.command.setworldspawn");
022        this.commandParameters.clear();
023        this.commandParameters.put("default", new CommandParameter[]{
024                CommandParameter.newType("spawnPoint", true, CommandParamType.POSITION)
025        });
026    }
027
028    @Override
029    public boolean execute(CommandSender sender, String commandLabel, String[] args) {
030        if (!this.testPermission(sender)) {
031            return true;
032        }
033        Level level;
034        Vector3 pos;
035        if (args.length == 0) {
036            if (sender instanceof Player) {
037                level = ((Player) sender).getLevel();
038                pos = ((Player) sender).round();
039            } else {
040                sender.sendMessage(new TranslationContainer("commands.generic.ingame"));
041                return true;
042            }
043        } else if (args.length == 3) {
044            level = sender.getServer().getDefaultLevel();
045            try {
046                pos = new Vector3(Integer.parseInt(args[0]), Integer.parseInt(args[1]), Integer.parseInt(args[2]));
047            } catch (NumberFormatException e1) {
048                sender.sendMessage(new TranslationContainer("commands.generic.usage", this.usageMessage));
049                return true;
050            }
051        } else {
052            sender.sendMessage(new TranslationContainer("commands.generic.usage", this.usageMessage));
053            return true;
054        }
055        level.setSpawnLocation(pos);
056        DecimalFormat round2 = new DecimalFormat("##0.00");
057        Command.broadcastCommandMessage(sender, new TranslationContainer("commands.setworldspawn.success", round2.format(pos.x),
058                round2.format(pos.y),
059                round2.format(pos.z)));
060        return true;
061    }
062}
