023/**
024 * @author MagicDroidX
025 */
026public class BlockEntityFurnace extends BlockEntitySpawnable implements InventoryHolder, BlockEntityContainer, BlockEntityNameable {
027
028    protected FurnaceInventory inventory;
029
030    protected int burnTime = 0;
031    protected int burnDuration = 0;
032    protected int cookTime = 0;
033    protected int maxTime = 0;
034
035    public BlockEntityFurnace(FullChunk chunk, CompoundTag nbt) {
036        super(chunk, nbt);
037    }
038
039    @Override
040    protected void initBlockEntity() {
041        this.inventory = new FurnaceInventory(this);
042
043        if (!this.namedTag.contains("Items") || !(this.namedTag.get("Items") instanceof ListTag)) {
044            this.namedTag.putList(new ListTag<CompoundTag>("Items"));
045        }
046
047        for (int i = 0; i < this.getSize(); i++) {
048            this.inventory.setItem(i, this.getItem(i));
049        }
050
051        if (!this.namedTag.contains("BurnTime") || this.namedTag.getShort("BurnTime") < 0) {
052            burnTime = 0;
053        } else {
054            burnTime = this.namedTag.getShort("BurnTime");
055        }
056
057        if (!this.namedTag.contains("CookTime") || this.namedTag.getShort("CookTime") < 0 || (this.namedTag.getShort("BurnTime") == 0 && this.namedTag.getShort("CookTime") > 0)) {
058            cookTime = 0;
059        } else {
060            cookTime = this.namedTag.getShort("CookTime");
061        }
062
063        if (!this.namedTag.contains("MaxTime")) {
064            maxTime = burnTime;
065            burnDuration = 0;
066        } else {
067            maxTime = this.namedTag.getShort("MaxTime");
068        }
069
070        if (this.namedTag.contains("BurnTicks")) {
071            burnDuration = this.namedTag.getShort("BurnTicks");
072            this.namedTag.remove("BurnTicks");
073        }
074
075        if (burnTime > 0) {
076            this.scheduleUpdate();
077        }
078
079        super.initBlockEntity();
080    }
081
082    @Override
083    public String getName() {
084        return this.hasName() ? this.namedTag.getString("CustomName") : "Furnace";
085    }
086
087    @Override
088    public boolean hasName() {
089        return this.namedTag.contains("CustomName");
090    }
091
092    @Override
093    public void setName(String name) {
094        if (name == null || name.equals("")) {
095            this.namedTag.remove("CustomName");
096            return;
097        }
098
099        this.namedTag.putString("CustomName", name);
100    }
101
102    @Override
103    public void close() {
104        if (!closed) {
105            for (Player player : new HashSet<>(this.getInventory().getViewers())) {
106                player.removeWindow(this.getInventory());
107            }
108            super.close();
109        }
110    }
111
112    @Override
113    public void onBreak() {
114        for (Item content : inventory.getContents().values()) {
115            level.dropItem(this, content);
116        }
117    }
118
119    @Override
120    public void saveNBT() {
121        this.namedTag.putList(new ListTag<CompoundTag>("Items"));
122        for (int index = 0; index < this.getSize(); index++) {
123            this.setItem(index, this.inventory.getItem(index));
124        }
125
126        this.namedTag.putShort("CookTime", cookTime);
127        this.namedTag.putShort("BurnTime", burnTime);
128        this.namedTag.putShort("BurnDuration", burnDuration);
129        this.namedTag.putShort("MaxTime", maxTime);
130    }
131
132    @Override
133    public boolean isBlockEntityValid() {
134        int blockID = getBlock().getId();
135        return blockID == Block.FURNACE || blockID == Block.BURNING_FURNACE;
136    }
137
138    @Override
139    public int getSize() {
140        return 3;
141    }
142
143    protected int getSlotIndex(int index) {
144        ListTag<CompoundTag> list = this.namedTag.getList("Items", CompoundTag.class);
145        for (int i = 0; i < list.size(); i++) {
146            if (list.get(i).getByte("Slot") == index) {
147                return i;
148            }
149        }
150
151        return -1;
152    }
153
154    @Override
155    public Item getItem(int index) {
156        int i = this.getSlotIndex(index);
157        if (i < 0) {
158            return new ItemBlock(new BlockAir(), 0, 0);
159        } else {
160            CompoundTag data = (CompoundTag) this.namedTag.getList("Items").get(i);
161            return NBTIO.getItemHelper(data);
162        }
163    }
164
165    @Override
166    public void setItem(int index, Item item) {
167        int i = this.getSlotIndex(index);
168
169        CompoundTag d = NBTIO.putItemHelper(item, index);
170
171        if (item.getId() == Item.AIR || item.getCount() <= 0) {
172            if (i >= 0) {
173                this.namedTag.getList("Items").getAll().remove(i);
174            }
175        } else if (i < 0) {
176            (this.namedTag.getList("Items", CompoundTag.class)).add(d);
177        } else {
178            (this.namedTag.getList("Items", CompoundTag.class)).add(i, d);
179        }
180    }
181
182    @Override
183    public FurnaceInventory getInventory() {
184        return inventory;
185    }
186
187    protected void checkFuel(Item fuel) {
188        FurnaceBurnEvent ev = new FurnaceBurnEvent(this, fuel, fuel.getFuelTime() == null ? 0 : fuel.getFuelTime());
189        this.server.getPluginManager().callEvent(ev);
190        if (ev.isCancelled()) {
191            return;
192        }
193
194        maxTime = ev.getBurnTime();
195        burnTime = ev.getBurnTime();
196        burnDuration = 0;
197        if (this.getBlock().getId() == Item.FURNACE) {
198            this.getLevel().setBlock(this, new BlockFurnaceBurning(this.getBlock().getDamage()), true);
199        }
200
201        if (burnTime > 0 && ev.isBurning()) {
202            fuel.setCount(fuel.getCount() - 1);
203            if (fuel.getCount() == 0) {
204                if (fuel.getId() == Item.BUCKET && fuel.getDamage() == 10) {
205                    fuel.setDamage(0);
206                    fuel.setCount(1);
207                } else {
208                    fuel = new ItemBlock(new BlockAir(), 0, 0);
209                }
210            }
211            this.inventory.setFuel(fuel);
212        }
213    }
214
215    @Override
216    public boolean onUpdate() {
217        if (this.closed) {
218            return false;
219        }
220
221        this.timing.startTiming();
222
223        boolean ret = false;
224        Item fuel = this.inventory.getFuel();
225        Item raw = this.inventory.getSmelting();
226        Item product = this.inventory.getResult();
227        FurnaceRecipe smelt = this.server.getCraftingManager().matchFurnaceRecipe(raw);
228        boolean canSmelt = (smelt != null && raw.getCount() > 0 && ((smelt.getResult().equals(product, true) && product.getCount() < product.getMaxStackSize()) || product.getId() == Item.AIR));
229
230        if (burnTime <= 0 && canSmelt && fuel.getFuelTime() != null && fuel.getCount() > 0) {
231            this.checkFuel(fuel);
232        }
233
234        if (burnTime > 0) {
235            burnTime--;
236            burnDuration = (int) Math.ceil((float) burnTime / maxTime * 200);
237
238            if (smelt != null && canSmelt) {
239                cookTime++;
240                if (cookTime >= 200) {
241                    product = Item.get(smelt.getResult().getId(), smelt.getResult().getDamage(), product.getCount() + 1);
242
243                    FurnaceSmeltEvent ev = new FurnaceSmeltEvent(this, raw, product);
244                    this.server.getPluginManager().callEvent(ev);
245                    if (!ev.isCancelled()) {
246                        this.inventory.setResult(ev.getResult());
247                        raw.setCount(raw.getCount() - 1);
248                        if (raw.getCount() == 0) {
249                            raw = new ItemBlock(new BlockAir(), 0, 0);
250                        }
251                        this.inventory.setSmelting(raw);
252                    }
253
254                    cookTime -= 200;
255                }
256            } else if (burnTime <= 0) {
257                burnTime = 0;
258                cookTime = 0;
259                burnDuration = 0;
260            } else {
261                cookTime = 0;
262            }
263            ret = true;
264        } else {
265            if (this.getBlock().getId() == Item.BURNING_FURNACE) {
266                this.getLevel().setBlock(this, new BlockFurnace(this.getBlock().getDamage()), true);
267            }
268            burnTime = 0;
269            cookTime = 0;
270            burnDuration = 0;
271        }
272
273        for (Player player : this.getInventory().getViewers()) {
274            int windowId = player.getWindowId(this.getInventory());
275            if (windowId > 0) {
276                ContainerSetDataPacket pk = new ContainerSetDataPacket();
277                pk.windowId = windowId;
278                pk.property = ContainerSetDataPacket.PROPERTY_FURNACE_TICK_COUNT;
279
280                pk.value = cookTime;
281                player.dataPacket(pk);
282
283                pk = new ContainerSetDataPacket();
284                pk.windowId = windowId;
285                pk.property = ContainerSetDataPacket.PROPERTY_FURNACE_LIT_TIME;
286                pk.value = burnDuration;
287                player.dataPacket(pk);
288            }
289        }
290
291        this.lastUpdate = System.currentTimeMillis();
292
293        this.timing.stopTiming();
294
295        return ret;
296    }
297
298    @Override
299    public CompoundTag getSpawnCompound() {
300        CompoundTag c = new CompoundTag()
301                .putString("id", BlockEntity.FURNACE)
302                .putInt("x", (int) this.x)
303                .putInt("y", (int) this.y)
304                .putInt("z", (int) this.z)
305                .putShort("BurnDuration", burnDuration)
306                .putShort("BurnTime", burnTime)
307                .putShort("CookTime", cookTime);
308
309        if (this.hasName()) {
310            c.put("CustomName", this.namedTag.get("CustomName"));
311        }
312
313        return c;
314    }
