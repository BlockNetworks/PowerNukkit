001package cn.nukkit.blockentity;
002
003import cn.nukkit.Player;
004import cn.nukkit.block.Block;
005import cn.nukkit.block.BlockAir;
006import cn.nukkit.inventory.BaseInventory;
007import cn.nukkit.inventory.ChestInventory;
008import cn.nukkit.inventory.DoubleChestInventory;
009import cn.nukkit.inventory.InventoryHolder;
010import cn.nukkit.item.Item;
011import cn.nukkit.item.ItemBlock;
012import cn.nukkit.level.format.FullChunk;
013import cn.nukkit.math.Vector3;
014import cn.nukkit.nbt.NBTIO;
015import cn.nukkit.nbt.tag.CompoundTag;
016import cn.nukkit.nbt.tag.ListTag;
017
018import java.util.HashSet;
019
020/**
021 * author: MagicDroidX
022 * Nukkit Project
023 */
024public class BlockEntityChest extends BlockEntitySpawnable implements InventoryHolder, BlockEntityContainer, BlockEntityNameable {
025
026    protected ChestInventory inventory;
027
028    protected DoubleChestInventory doubleInventory = null;
029
030    public BlockEntityChest(FullChunk chunk, CompoundTag nbt) {
031        super(chunk, nbt);
032    }
033
034    @Override
035    protected void initBlockEntity() {
036        this.inventory = new ChestInventory(this);
037
038        if (!this.namedTag.contains("Items") || !(this.namedTag.get("Items") instanceof ListTag)) {
039            this.namedTag.putList(new ListTag<CompoundTag>("Items"));
040        }
041
042        /* for (int i = 0; i < this.getSize(); i++) {
043            this.inventory.setItem(i, this.getItem(i));
044        } */
045
046        ListTag<CompoundTag> list = (ListTag<CompoundTag>) this.namedTag.getList("Items");
047        for (CompoundTag compound : list.getAll()) {
048            Item item = NBTIO.getItemHelper(compound);
049            this.inventory.slots.put(compound.getByte("Slot"), item);
050        }
051
052        super.initBlockEntity();
053    }
054
055    @Override
056    public void close() {
057        if (!closed) {
058            unpair();
059
060            for (Player player : new HashSet<>(this.getInventory().getViewers())) {
061                player.removeWindow(this.getInventory());
062            }
063
064            for (Player player : new HashSet<>(this.getInventory().getViewers())) {
065                player.removeWindow(this.getRealInventory());
066            }
067            super.close();
068        }
069    }
070
071    @Override
072    public void onBreak() {
073        for (Item content : inventory.getContents().values()) {
074            level.dropItem(this, content);
075        }
076        inventory.clearAll(); // Stop items from being moved around by another player in the inventory
077    }
078
079    @Override
080    public void saveNBT() {
081        this.namedTag.putList(new ListTag<CompoundTag>("Items"));
082        for (int index = 0; index < this.getSize(); index++) {
083            this.setItem(index, this.inventory.getItem(index));
084        }
085    }
086
087    @Override
088    public boolean isBlockEntityValid() {
089        int blockID = this.getBlock().getId();
090        return blockID == Block.CHEST || blockID == Block.TRAPPED_CHEST;
091    }
092
093    @Override
094    public int getSize() {
095        return 27;
096    }
097
098    protected int getSlotIndex(int index) {
099        ListTag<CompoundTag> list = this.namedTag.getList("Items", CompoundTag.class);
100        for (int i = 0; i < list.size(); i++) {
101            if (list.get(i).getByte("Slot") == index) {
102                return i;
103            }
104        }
105
106        return -1;
107    }
108
109    @Override
110    public Item getItem(int index) {
111        int i = this.getSlotIndex(index);
112        if (i < 0) {
113            return new ItemBlock(new BlockAir(), 0, 0);
114        } else {
115            CompoundTag data = (CompoundTag) this.namedTag.getList("Items").get(i);
116            return NBTIO.getItemHelper(data);
117        }
118    }
119
120    @Override
121    public void setItem(int index, Item item) {
122        int i = this.getSlotIndex(index);
123
124        CompoundTag d = NBTIO.putItemHelper(item, index);
125
126        // If item is air or count less than 0, remove the item from the "Items" list
127        if (item.getId() == Item.AIR || item.getCount() <= 0) {
128            if (i >= 0) {
129                this.namedTag.getList("Items").remove(i);
130            }
131        } else if (i < 0) {
132            // If it is less than i, then it is a new item, so we are going to add it at the end of the list
133            (this.namedTag.getList("Items", CompoundTag.class)).add(d);
134        } else {
135            // If it is more than i, then it is an update on a inventorySlot, so we are going to overwrite the item in the list
136            (this.namedTag.getList("Items", CompoundTag.class)).add(i, d);
137        }
138    }
139
140    @Override
141    public BaseInventory getInventory() {
142        if (this.doubleInventory == null && this.isPaired()) {
143            this.checkPairing();
144        }
145
146        return this.doubleInventory != null ? this.doubleInventory : this.inventory;
147    }
148
149    public ChestInventory getRealInventory() {
150        return inventory;
151    }
152
153    protected void checkPairing() {
154        BlockEntityChest pair = this.getPair();
155
156        if (pair != null) {
157            if (!pair.isPaired()) {
158                pair.createPair(this);
159                pair.checkPairing();
160            }
161
162            if (pair.doubleInventory != null) {
163                this.doubleInventory = pair.doubleInventory;
164            } else if (this.doubleInventory == null) {
165                if ((pair.x + ((int) pair.z << 15)) > (this.x + ((int) this.z << 15))) { //Order them correctly
166                    this.doubleInventory = new DoubleChestInventory(pair, this);
167                } else {
168                    this.doubleInventory = new DoubleChestInventory(this, pair);
169                }
170            }
171        } else {
172            if (level.isChunkLoaded(this.namedTag.getInt("pairx") >> 4, this.namedTag.getInt("pairz") >> 4)) {
173                this.doubleInventory = null;
174                this.namedTag.remove("pairx");
175                this.namedTag.remove("pairz");
176            }
177        }
178    }
179
180    @Override
181    public String getName() {
182        return this.hasName() ? this.namedTag.getString("CustomName") : "Chest";
183    }
184
185    @Override
186    public boolean hasName() {
187        return this.namedTag.contains("CustomName");
188    }
189
190    @Override
191    public void setName(String name) {
192        if (name == null || name.equals("")) {
193            this.namedTag.remove("CustomName");
194            return;
195        }
196
197        this.namedTag.putString("CustomName", name);
198    }
199
200    public boolean isPaired() {
201        return this.namedTag.contains("pairx") && this.namedTag.contains("pairz");
202    }
203
204    public BlockEntityChest getPair() {
205        if (this.isPaired()) {
206            BlockEntity blockEntity = this.getLevel().getBlockEntityIfLoaded(new Vector3(this.namedTag.getInt("pairx"), this.y, this.namedTag.getInt("pairz")));
207            if (blockEntity instanceof BlockEntityChest) {
208                return (BlockEntityChest) blockEntity;
209            }
210        }
211
212        return null;
213    }
214
215    public boolean pairWith(BlockEntityChest chest) {
216        if (this.isPaired() || chest.isPaired() || this.getBlock().getId() != chest.getBlock().getId()) {
217            return false;
218        }
219
220        this.createPair(chest);
221
222        chest.spawnToAll();
223        this.spawnToAll();
224        this.checkPairing();
225
226        return true;
227    }
228
229    public void createPair(BlockEntityChest chest) {
230        this.namedTag.putInt("pairx", (int) chest.x);
231        this.namedTag.putInt("pairz", (int) chest.z);
232        chest.namedTag.putInt("pairx", (int) this.x);
233        chest.namedTag.putInt("pairz", (int) this.z);
234    }
235
236    public boolean unpair() {
237        if (!this.isPaired()) {
238            return false;
239        }
240
241        BlockEntityChest chest = this.getPair();
242
243        this.doubleInventory = null;
244        this.namedTag.remove("pairx");
245        this.namedTag.remove("pairz");
246
247        this.spawnToAll();
248
249        if (chest != null) {
250            chest.namedTag.remove("pairx");
251            chest.namedTag.remove("pairz");
252            chest.doubleInventory = null;
253            chest.checkPairing();
254            chest.spawnToAll();
255        }
256        this.checkPairing();
257
258        return true;
259    }
260
261    @Override
262    public CompoundTag getSpawnCompound() {
263        CompoundTag c;
264        if (this.isPaired()) {
265            c = new CompoundTag()
266                    .putString("id", BlockEntity.CHEST)
267                    .putInt("x", (int) this.x)
268                    .putInt("y", (int) this.y)
269                    .putInt("z", (int) this.z)
270                    .putInt("pairx", this.namedTag.getInt("pairx"))
271                    .putInt("pairz", this.namedTag.getInt("pairz"));
272        } else {
273            c = new CompoundTag()
274                    .putString("id", BlockEntity.CHEST)
275                    .putInt("x", (int) this.x)
276                    .putInt("y", (int) this.y)
277                    .putInt("z", (int) this.z);
278        }
279
280        if (this.hasName()) {
281            c.put("CustomName", this.namedTag.get("CustomName"));
282        }
283
284        return c;
285    }
