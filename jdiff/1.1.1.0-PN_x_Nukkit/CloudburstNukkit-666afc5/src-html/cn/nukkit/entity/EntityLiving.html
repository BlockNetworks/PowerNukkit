104    @Override
105    public boolean attack(EntityDamageEvent source) {
106        if (this.attackTime > 0 || this.noDamageTicks > 0) {
107            EntityDamageEvent lastCause = this.getLastDamageCause();
108            if (lastCause != null && lastCause.getDamage() >= source.getDamage()) {
109                return false;
110            }
111        }
112
113        if (super.attack(source)) {
114            if (source instanceof EntityDamageByEntityEvent) {
115                Entity damager = ((EntityDamageByEntityEvent) source).getDamager();
116                if (source instanceof EntityDamageByChildEntityEvent) {
117                    damager = ((EntityDamageByChildEntityEvent) source).getChild();
118                }
119
120                //Critical hit
121                if (damager instanceof Player && !damager.onGround) {
122                    AnimatePacket animate = new AnimatePacket();
123                    animate.action = AnimatePacket.Action.CRITICAL_HIT;
124                    animate.eid = getId();
125
126                    this.getLevel().addChunkPacket(damager.getChunkX(), damager.getChunkZ(), animate);
127                    this.getLevel().addLevelSoundEvent(this, LevelSoundEventPacket.SOUND_ATTACK_STRONG);
128
129                    source.setDamage(source.getDamage() * 1.5f);
130                }
131
132                if (damager.isOnFire() && !(damager instanceof Player)) {
133                    this.setOnFire(2 * this.server.getDifficulty());
134                }
135
136                double deltaX = this.x - damager.x;
137                double deltaZ = this.z - damager.z;
138                this.knockBack(damager, source.getDamage(), deltaX, deltaZ, ((EntityDamageByEntityEvent) source).getKnockBack());
139            }
140
141            EntityEventPacket pk = new EntityEventPacket();
142            pk.eid = this.getId();
143            pk.event = this.getHealth() <= 0 ? EntityEventPacket.DEATH_ANIMATION : EntityEventPacket.HURT_ANIMATION;
144            Server.broadcastPacket(this.hasSpawned.values(), pk);
145
146            this.attackTime = source.getAttackCooldown();
147
148            return true;
149        } else {
150            return false;
151        }
152    }

158    public void knockBack(Entity attacker, double damage, double x, double z, double base) {
159        double f = Math.sqrt(x * x + z * z);
160        if (f <= 0) {
161            return;
162        }
163
164        f = 1 / f;
165
166        Vector3 motion = new Vector3(this.motionX, this.motionY, this.motionZ);
167
168        motion.x /= 2d;
169        motion.y /= 2d;
170        motion.z /= 2d;
171        motion.x += x * f * base;
172        motion.y += base;
173        motion.z += z * f * base;
174
175        if (motion.y > base) {
176            motion.y = base;
177        }
178
179        this.setMotion(motion);
180    }

182    @Override
183    public void kill() {
184        if (!this.isAlive()) {
185            return;
186        }
187        super.kill();
188        EntityDeathEvent ev = new EntityDeathEvent(this, this.getDrops());
189        this.server.getPluginManager().callEvent(ev);
190
191        if (this.level.getGameRules().getBoolean(GameRule.DO_ENTITY_DROPS)) {
192            for (cn.nukkit.item.Item item : ev.getDrops()) {
193                this.getLevel().dropItem(this, item);
194            }
195        }
196    }

203    @Override
204    public boolean entityBaseTick(int tickDiff) {
205        Timings.livingEntityBaseTickTimer.startTiming();
206        boolean isBreathing = !this.isInsideOfWater();
207        if (this instanceof Player && (((Player) this).isCreative() || ((Player) this).isSpectator())) {
208            isBreathing = true;
209        }
210
211        if (this instanceof Player) {
212            if (!isBreathing && ((Player) this).getInventory().getHelmet() instanceof ItemTurtleShell) {
213                if (turtleTicks > 0) {
214                    isBreathing = true;
215                    turtleTicks--;
216                }
217            } else {
218                turtleTicks = 200;
219            }
220        }
221        
222        this.setDataFlag(DATA_FLAGS, DATA_FLAG_BREATHING, isBreathing);
223
224        boolean hasUpdate = super.entityBaseTick(tickDiff);
225
226        if (this.isAlive()) {
227
228            if (this.isInsideOfSolid()) {
229                hasUpdate = true;
230                this.attack(new EntityDamageEvent(this, DamageCause.SUFFOCATION, 1));
231            }
232
233            if (this.isOnLadder() || this.hasEffect(Effect.LEVITATION)) {
234                this.resetFallDistance();
235            }
236
237            if (!this.hasEffect(Effect.WATER_BREATHING) && this.isInsideOfWater()) {
238                if (this instanceof EntityWaterAnimal || (this instanceof Player && (((Player) this).isCreative() || ((Player) this).isSpectator()))) {
239                    this.setAirTicks(400);
240                } else {
241                    if (turtleTicks == 0 || turtleTicks == 200) {
242                        hasUpdate = true;
243                        int airTicks = this.getAirTicks() - tickDiff;
244
245                        if (airTicks <= -20) {
246                            airTicks = 0;
247                            this.attack(new EntityDamageEvent(this, DamageCause.DROWNING, 2));
248                        }
249
250                        setAirTicks(airTicks);
251                    }
252                }
253            } else {
254                if (this instanceof EntityWaterAnimal) {
255                    hasUpdate = true;
256                    int airTicks = getAirTicks() - tickDiff;
257
258                    if (airTicks <= -20) {
259                        airTicks = 0;
260                        this.attack(new EntityDamageEvent(this, DamageCause.SUFFOCATION, 2));
261                    }
262
263                    setAirTicks(airTicks);
264                } else {
265                    int airTicks = getAirTicks();
266
267                    if (airTicks < 400) {
268                        setAirTicks(Math.min(400, airTicks + tickDiff * 5));
269                    }
270                }
271            }
272        }
273
274        if (this.attackTime > 0) {
275            this.attackTime -= tickDiff;
276        }
277
278        if (this.riding == null) {
279            for (Entity entity : level.getNearbyEntities(this.boundingBox.grow(0.20000000298023224, 0.0D, 0.20000000298023224), this)) {
280                if (entity instanceof EntityRideable) {
281                    this.collidingWith(entity);
282                }
283            }
284        }
285
286        // Used to check collisions with magma blocks
287        Block block = this.level.getBlock((int) x, (int) y - 1, (int) z);
288        if (block instanceof BlockMagma) block.onEntityCollide(this);
289
290        Timings.livingEntityBaseTickTimer.stopTiming();
291
292        return hasUpdate;
293    }

312    public Block[] getLineOfSight(int maxDistance, int maxLength, Integer[] transparent) {
313        if (maxDistance > 120) {
314            maxDistance = 120;
315        }
316
317        if (transparent != null && transparent.length == 0) {
318            transparent = null;
319        }
320
321        List<Block> blocks = new ArrayList<>();
322
323        BlockIterator itr = new BlockIterator(this.level, this.getPosition(), this.getDirectionVector(), this.getEyeHeight(), maxDistance);
324
325        while (itr.hasNext()) {
326            Block block = itr.next();
327            blocks.add(block);
328
329            if (maxLength != 0 && blocks.size() > maxLength) {
330                blocks.remove(0);
331            }
332
333            int id = block.getId();
334
335            if (transparent == null) {
336                if (id != 0) {
337                    break;
338                }
339            } else {
340                if (Arrays.binarySearch(transparent, id) < 0) {
341                    break;
342                }
343            }
344        }
345
346        return blocks.toArray(new Block[0]);
347    }

358    public Block getTargetBlock(int maxDistance, Integer[] transparent) {
359        try {
360            Block[] blocks = this.getLineOfSight(maxDistance, 1, transparent);
361            Block block = blocks[0];
362            if (block != null) {
363                if (transparent != null && transparent.length != 0) {
364                    if (Arrays.binarySearch(transparent, block.getId()) < 0) {
365                        return block;
366                    }
367                } else {
368                    return block;
369                }
370            }
371        } catch (Exception ignored) {
372
373        }
374
375        return null;
376    }
