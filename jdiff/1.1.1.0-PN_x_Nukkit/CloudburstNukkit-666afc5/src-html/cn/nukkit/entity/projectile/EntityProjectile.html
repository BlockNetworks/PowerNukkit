001package cn.nukkit.entity.projectile;
002
003import cn.nukkit.entity.Entity;
004import cn.nukkit.entity.EntityLiving;
005import cn.nukkit.entity.data.LongEntityData;
006import cn.nukkit.entity.item.EntityEndCrystal;
007import cn.nukkit.event.entity.*;
008import cn.nukkit.event.entity.EntityDamageEvent.DamageCause;
009import cn.nukkit.level.MovingObjectPosition;
010import cn.nukkit.level.format.FullChunk;
011import cn.nukkit.math.AxisAlignedBB;
012import cn.nukkit.math.NukkitMath;
013import cn.nukkit.math.Vector3;
014import cn.nukkit.nbt.tag.CompoundTag;
015
016/**
017 * author: MagicDroidX
018 * Nukkit Project
019 */
020public abstract class EntityProjectile extends Entity {
021
022    public static final int DATA_SHOOTER_ID = 17;
023
024    public Entity shootingEntity = null;
025
026    protected double getDamage() {
027        return namedTag.contains("damage") ? namedTag.getDouble("damage") : getBaseDamage();
028    }
029
030    protected double getBaseDamage() {
031        return 0;
032    }
033
034    public boolean hadCollision = false;
035
036    public boolean closeOnCollide = true;
037
038    protected double damage = 0;
039
040    public EntityProjectile(FullChunk chunk, CompoundTag nbt) {
041        this(chunk, nbt, null);
042    }
043
044    public EntityProjectile(FullChunk chunk, CompoundTag nbt, Entity shootingEntity) {
045        super(chunk, nbt);
046        this.shootingEntity = shootingEntity;
047        if (shootingEntity != null) {
048            this.setDataProperty(new LongEntityData(DATA_SHOOTER_ID, shootingEntity.getId()));
049        }
050    }
051
052    public int getResultDamage() {
053        return NukkitMath.ceilDouble(Math.sqrt(this.motionX * this.motionX + this.motionY * this.motionY + this.motionZ * this.motionZ) * getDamage());
054    }
055
056    public boolean attack(EntityDamageEvent source) {
057        return source.getCause() == DamageCause.VOID && super.attack(source);
058    }
059
060    public void onCollideWithEntity(Entity entity) {
061        this.server.getPluginManager().callEvent(new ProjectileHitEvent(this, MovingObjectPosition.fromEntity(entity)));
062        float damage = this.getResultDamage();
063
064        EntityDamageEvent ev;
065        if (this.shootingEntity == null) {
066            ev = new EntityDamageByEntityEvent(this, entity, DamageCause.PROJECTILE, damage);
067        } else {
068            ev = new EntityDamageByChildEntityEvent(this.shootingEntity, this, entity, DamageCause.PROJECTILE, damage);
069        }
070        entity.attack(ev);
071        this.hadCollision = true;
072
073        if (this.fireTicks > 0) {
074            EntityCombustByEntityEvent event = new EntityCombustByEntityEvent(this, entity, 5);
075            this.server.getPluginManager().callEvent(ev);
076            if (!event.isCancelled()) {
077                entity.setOnFire(event.getDuration());
078            }
079        }
080        if (closeOnCollide) {
081            this.close();
082        }
083    }
084
085    @Override
086    protected void initEntity() {
087        super.initEntity();
088
089        this.setMaxHealth(1);
090        this.setHealth(1);
091        if (this.namedTag.contains("Age")) {
092            this.age = this.namedTag.getShort("Age");
093        }
094    }
095
096    @Override
097    public boolean canCollideWith(Entity entity) {
098        return (entity instanceof EntityLiving || entity instanceof EntityEndCrystal) && !this.onGround;
099    }
100
101    @Override
102    public void saveNBT() {
103        super.saveNBT();
104        this.namedTag.putShort("Age", this.age);
105    }
106
107    @Override
108    public boolean onUpdate(int currentTick) {
109        if (this.closed) {
110            return false;
111        }
112
113        int tickDiff = currentTick - this.lastUpdate;
114        if (tickDiff <= 0 && !this.justCreated) {
115            return true;
116        }
117        this.lastUpdate = currentTick;
118
119        boolean hasUpdate = this.entityBaseTick(tickDiff);
120
121        if (this.isAlive()) {
122
123            MovingObjectPosition movingObjectPosition = null;
124
125            if (!this.isCollided) {
126                this.motionY -= this.getGravity();
127            }
128
129            Vector3 moveVector = new Vector3(this.x + this.motionX, this.y + this.motionY, this.z + this.motionZ);
130
131            Entity[] list = this.getLevel().getCollidingEntities(this.boundingBox.addCoord(this.motionX, this.motionY, this.motionZ).expand(1, 1, 1), this);
132
133            double nearDistance = Integer.MAX_VALUE;
134            Entity nearEntity = null;
135
136            for (Entity entity : list) {
137                if (/*!entity.canCollideWith(this) or */
138                        (entity == this.shootingEntity && this.ticksLived < 5)
139                        ) {
140                    continue;
141                }
142
143                AxisAlignedBB axisalignedbb = entity.boundingBox.grow(0.3, 0.3, 0.3);
144                MovingObjectPosition ob = axisalignedbb.calculateIntercept(this, moveVector);
145
146                if (ob == null) {
147                    continue;
148                }
149
150                double distance = this.distanceSquared(ob.hitVector);
151
152                if (distance < nearDistance) {
153                    nearDistance = distance;
154                    nearEntity = entity;
155                }
156            }
157
158            if (nearEntity != null) {
159                movingObjectPosition = MovingObjectPosition.fromEntity(nearEntity);
160            }
161
162            if (movingObjectPosition != null) {
163                if (movingObjectPosition.entityHit != null) {
164                    onCollideWithEntity(movingObjectPosition.entityHit);
165                    return true;
166                }
167            }
168
169            this.move(this.motionX, this.motionY, this.motionZ);
170
171            if (this.isCollided && !this.hadCollision) { //collide with block
172                this.hadCollision = true;
173
174                this.motionX = 0;
175                this.motionY = 0;
176                this.motionZ = 0;
177
178                this.server.getPluginManager().callEvent(new ProjectileHitEvent(this, MovingObjectPosition.fromBlock(this.getFloorX(), this.getFloorY(), this.getFloorZ(), -1, this)));
179                return false;
180            } else if (!this.isCollided && this.hadCollision) {
181                this.hadCollision = false;
182            }
183
184            if (!this.hadCollision || Math.abs(this.motionX) > 0.00001 || Math.abs(this.motionY) > 0.00001 || Math.abs(this.motionZ) > 0.00001) {
185                double f = Math.sqrt((this.motionX * this.motionX) + (this.motionZ * this.motionZ));
186                this.yaw = Math.atan2(this.motionX, this.motionZ) * 180 / Math.PI;
187                this.pitch = Math.atan2(this.motionY, f) * 180 / Math.PI;
188                hasUpdate = true;
189            }
190
191            this.updateMovement();
192
193        }
194
195        return hasUpdate;
196    }
197}
