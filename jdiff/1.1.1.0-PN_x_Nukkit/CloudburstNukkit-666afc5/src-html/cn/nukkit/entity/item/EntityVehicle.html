001package cn.nukkit.entity.item;
002
003import cn.nukkit.Player;
004import cn.nukkit.entity.Entity;
005import cn.nukkit.entity.EntityInteractable;
006import cn.nukkit.entity.EntityRideable;
007import cn.nukkit.entity.data.IntEntityData;
008import cn.nukkit.event.entity.EntityDamageByEntityEvent;
009import cn.nukkit.event.entity.EntityDamageEvent;
010import cn.nukkit.event.vehicle.VehicleDamageEvent;
011import cn.nukkit.event.vehicle.VehicleDestroyEvent;
012import cn.nukkit.level.format.FullChunk;
013import cn.nukkit.nbt.tag.CompoundTag;
014
015/**
016 * author: MagicDroidX
017 * Nukkit Project
018 */
019public abstract class EntityVehicle extends Entity implements EntityRideable, EntityInteractable {
020
021    public EntityVehicle(FullChunk chunk, CompoundTag nbt) {
022        super(chunk, nbt);
023    }
024
025    public int getRollingAmplitude() {
026        return this.getDataPropertyInt(DATA_HURT_TIME);
027    }
028
029    public void setRollingAmplitude(int time) {
030        this.setDataProperty(new IntEntityData(DATA_HURT_TIME, time));
031    }
032
033    public int getRollingDirection() {
034        return this.getDataPropertyInt(DATA_HURT_DIRECTION);
035    }
036
037    public void setRollingDirection(int direction) {
038        this.setDataProperty(new IntEntityData(DATA_HURT_DIRECTION, direction));
039    }
040
041    public int getDamage() {
042        return this.getDataPropertyInt(DATA_HEALTH); // false data name (should be DATA_DAMAGE_TAKEN)
043    }
044
045    public void setDamage(int damage) {
046        this.setDataProperty(new IntEntityData(DATA_HEALTH, damage));
047    }
048
049    @Override
050    public String getInteractButtonText() {
051        return "Mount";
052    }
053
054    @Override
055    public boolean canDoInteraction() {
056        return passengers.isEmpty();
057    }
058
059    @Override
060    public boolean onUpdate(int currentTick) {
061        // The rolling amplitude
062        if (getRollingAmplitude() > 0) {
063            setRollingAmplitude(getRollingAmplitude() - 1);
064        }
065
066        // A killer task
067        if (y < -16) {
068            kill();
069        }
070        // Movement code
071        updateMovement();
072        return true;
073    }
074
075    protected boolean rollingDirection = true;
076
077    protected boolean performHurtAnimation() {
078        setRollingAmplitude(9);
079        setRollingDirection(rollingDirection ? 1 : -1);
080        rollingDirection = !rollingDirection;
081        return true;
082    }
083
084    @Override
085    public boolean attack(EntityDamageEvent source) {
086        VehicleDamageEvent event = new VehicleDamageEvent(this, source.getEntity(), source.getFinalDamage());
087        getServer().getPluginManager().callEvent(event);
088        if (event.isCancelled()) {
089            return false;
090        }
091
092        boolean instantKill = false;
093
094        if (source instanceof EntityDamageByEntityEvent) {
095            Entity damager = ((EntityDamageByEntityEvent) source).getDamager();
096            instantKill = damager instanceof Player && ((Player) damager).isCreative();
097        }
098
099        if (instantKill || getHealth() - source.getFinalDamage() < 1) {
100            VehicleDestroyEvent event2 = new VehicleDestroyEvent(this, source.getEntity());
101            getServer().getPluginManager().callEvent(event2);
102
103            if (event2.isCancelled()) {
104                return false;
105            }
106        }
107
108        if (instantKill) {
109            source.setDamage(1000);
110        }
111
112        return super.attack(source);
113    }
114}
