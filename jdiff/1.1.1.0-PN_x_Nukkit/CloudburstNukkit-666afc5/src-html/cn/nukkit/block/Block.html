303            for (int id = 0; id < 256; id++) {
304                Class c = list[id];
305                if (c != null) {
306                    Block block;
307                    try {
308                        block = (Block) c.newInstance();
309                        try {
310                            Constructor constructor = c.getDeclaredConstructor(int.class);
311                            constructor.setAccessible(true);
312                            for (int data = 0; data < 16; ++data) {
313                                fullList[(id << 4) | data] = (Block) constructor.newInstance(data);
314                            }
315                            hasMeta[id] = true;
316                        } catch (NoSuchMethodException ignore) {
317                            for (int data = 0; data < 16; ++data) {
318                                fullList[(id << 4) | data] = block;
319                            }
320                        }
321                    } catch (Exception e) {
322                        Server.getInstance().getLogger().error("Error while registering " + c.getName(), e);
323                        for (int data = 0; data < 16; ++data) {
324                            fullList[(id << 4) | data] = new BlockUnknown(id, data);
325                        }
326                        return;
327                    }
328
329                    solid[id] = block.isSolid();
330                    transparent[id] = block.isTransparent();
331                    hardness[id] = block.getHardness();
332                    light[id] = block.getLightLevel();
333
334                    if (block.isSolid()) {
335                        if (block.isTransparent()) {
336                            if (block instanceof BlockLiquid || block instanceof BlockIce) {
337                                lightFilter[id] = 2;
338                            } else {
339                                lightFilter[id] = 1;
340                            }
341                        } else {
342                            lightFilter[id] = 15;
343                        }
344                    } else {
345                        lightFilter[id] = 1;
346                    }
347                } else {
348                    lightFilter[id] = 1;
349                    for (int data = 0; data < 16; ++data) {
350                        fullList[(id << 4) | data] = new BlockUnknown(id, data);
351                    }
352                }
353            }
354        }

601    private static double breakTime0(double blockHardness, boolean correctTool, boolean canHarvestWithHand,
602                                     int blockId, int toolType, int toolTier, int efficiencyLoreLevel, int hasteEffectLevel,
603                                     boolean insideOfWaterWithoutAquaAffinity, boolean outOfWaterButNotOnGround) {
604        double baseTime = ((correctTool || canHarvestWithHand) ? 1.5 : 5.0) * blockHardness;
605        double speed = 1.0 / baseTime;
606        boolean isWoolBlock = blockId == Block.WOOL, isCobweb = blockId == Block.COBWEB;
607        if (correctTool) speed *= toolBreakTimeBonus0(toolType, toolTier, isWoolBlock, isCobweb);
608        speed += speedBonusByEfficiencyLore0(efficiencyLoreLevel);
609        speed *= speedRateByHasteLore0(hasteEffectLevel);
610        if (insideOfWaterWithoutAquaAffinity) speed *= 0.2;
611        if (outOfWaterButNotOnGround) speed *= 0.2;
612        return 1.0 / speed;
613    }

615    public double getBreakTime(Item item, Player player) {
616        Objects.requireNonNull(item, "getBreakTime: Item can not be null");
617        Objects.requireNonNull(player, "getBreakTime: Player can not be null");
618        double blockHardness = getHardness();
619        boolean correctTool = correctTool0(getToolType(), item);
620        boolean canHarvestWithHand = canHarvestWithHand();
621        int blockId = getId();
622        int itemToolType = toolType0(item);
623        int itemTier = item.getTier();
624        int efficiencyLoreLevel = Optional.ofNullable(item.getEnchantment(Enchantment.ID_EFFICIENCY))
625                .map(Enchantment::getLevel).orElse(0);
626        int hasteEffectLevel = Optional.ofNullable(player.getEffect(Effect.HASTE))
627                .map(Effect::getAmplifier).orElse(0);
628        boolean insideOfWaterWithoutAquaAffinity = player.isInsideOfWater() &&
629                Optional.ofNullable(player.getInventory().getHelmet().getEnchantment(Enchantment.ID_WATER_WORKER))
630                        .map(Enchantment::getLevel).map(l -> l >= 1).orElse(false);
631        boolean outOfWaterButNotOnGround = (!player.isInsideOfWater()) && (!player.isOnGround());
632        return breakTime0(blockHardness, correctTool, canHarvestWithHand, blockId, itemToolType, itemTier,
633                efficiencyLoreLevel, hasteEffectLevel, insideOfWaterWithoutAquaAffinity, outOfWaterButNotOnGround);

820    public MovingObjectPosition calculateIntercept(Vector3 pos1, Vector3 pos2) {
821        AxisAlignedBB bb = this.getBoundingBox();
822        if (bb == null) {
823            return null;
824        }
825
826        Vector3 v1 = pos1.getIntermediateWithXValue(pos2, bb.getMinX());
827        Vector3 v2 = pos1.getIntermediateWithXValue(pos2, bb.getMaxX());
828        Vector3 v3 = pos1.getIntermediateWithYValue(pos2, bb.getMinY());
829        Vector3 v4 = pos1.getIntermediateWithYValue(pos2, bb.getMaxY());
830        Vector3 v5 = pos1.getIntermediateWithZValue(pos2, bb.getMinZ());
831        Vector3 v6 = pos1.getIntermediateWithZValue(pos2, bb.getMaxZ());
832
833        if (v1 != null && !bb.isVectorInYZ(v1)) {
834            v1 = null;
835        }
836
837        if (v2 != null && !bb.isVectorInYZ(v2)) {
838            v2 = null;
839        }
840
841        if (v3 != null && !bb.isVectorInXZ(v3)) {
842            v3 = null;
843        }
844
845        if (v4 != null && !bb.isVectorInXZ(v4)) {
846            v4 = null;
847        }
848
849        if (v5 != null && !bb.isVectorInXY(v5)) {
850            v5 = null;
851        }
852
853        if (v6 != null && !bb.isVectorInXY(v6)) {
854            v6 = null;
855        }
856
857        Vector3 vector = v1;
858
859        if (v2 != null && (vector == null || pos1.distanceSquared(v2) < pos1.distanceSquared(vector))) {
860            vector = v2;
861        }
862
863        if (v3 != null && (vector == null || pos1.distanceSquared(v3) < pos1.distanceSquared(vector))) {
864            vector = v3;
865        }
866
867        if (v4 != null && (vector == null || pos1.distanceSquared(v4) < pos1.distanceSquared(vector))) {
868            vector = v4;
869        }
870
871        if (v5 != null && (vector == null || pos1.distanceSquared(v5) < pos1.distanceSquared(vector))) {
872            vector = v5;
873        }
874
875        if (v6 != null && (vector == null || pos1.distanceSquared(v6) < pos1.distanceSquared(vector))) {
876            vector = v6;
877        }
878
879        if (vector == null) {
880            return null;
881        }
882
883        int f = -1;
884
885        if (vector == v1) {
886            f = 4;
887        } else if (vector == v2) {
888            f = 5;
889        } else if (vector == v3) {
890            f = 0;
891        } else if (vector == v4) {
892            f = 1;
893        } else if (vector == v5) {
894            f = 2;
895        } else if (vector == v6) {
896            f = 3;
897        }
898
899        return MovingObjectPosition.fromBlock((int) this.x, (int) this.y, (int) this.z, f, vector.add(this.x, this.y, this.z));
900    }
