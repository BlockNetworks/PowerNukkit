001package cn.nukkit.timings;
002
003import cn.nukkit.level.Level;
004import co.aikar.timings.Timing;
005import co.aikar.timings.TimingsManager;
006
007/**
008 * @author Pub4Game
009 * @author Tee7even
010 */
011public class LevelTimings {
012    public final Timing doChunkUnload;
013    public final Timing doTickPending;
014    public final Timing doChunkGC;
015    public final Timing doTick;
016
017    public final Timing tickChunks;
018    public final Timing entityTick;
019    public final Timing blockEntityTick;
020
021    public final Timing syncChunkSendTimer;
022    public final Timing syncChunkSendPrepareTimer;
023    public final Timing syncChunkLoadTimer;
024    public final Timing syncChunkLoadDataTimer;
025    public final Timing syncChunkLoadEntitiesTimer;
026    public final Timing syncChunkLoadBlockEntitiesTimer;
027
028    public LevelTimings(Level level) {
029        String name = level.getFolderName() + " - ";
030
031        this.doChunkUnload = TimingsManager.getTiming(name + "doChunkUnload");
032        this.doTickPending = TimingsManager.getTiming(name + "doTickPending");
033        this.doChunkGC = TimingsManager.getTiming(name + "doChunkGC");
034        this.doTick = TimingsManager.getTiming(name + "doTick");
035
036        this.tickChunks = TimingsManager.getTiming(name + "tickChunks");
037        this.entityTick = TimingsManager.getTiming(name + "entityTick");
038        this.blockEntityTick = TimingsManager.getTiming(name + "blockEntityTick");
039
040        this.syncChunkSendTimer = TimingsManager.getTiming(name + "syncChunkSend");
041        this.syncChunkSendPrepareTimer = TimingsManager.getTiming(name + "syncChunkSendPrepare");
042        this.syncChunkLoadTimer = TimingsManager.getTiming(name + "syncChunkLoad");
043        this.syncChunkLoadDataTimer = TimingsManager.getTiming(name + "syncChunkLoad - Data");
044        this.syncChunkLoadEntitiesTimer = TimingsManager.getTiming(name + "syncChunkLoad - Entities");
045        this.syncChunkLoadBlockEntitiesTimer = TimingsManager.getTiming(name + "syncChunkLoad - BlockEntities");
046    }
047}
