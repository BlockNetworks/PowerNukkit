001package cn.nukkit.command.defaults;
002
003import cn.nukkit.command.Command;
004import cn.nukkit.command.CommandSender;
005import cn.nukkit.command.data.CommandParamType;
006import cn.nukkit.command.data.CommandParameter;
007import cn.nukkit.lang.TranslationContainer;
008
009/**
010 * author: MagicDroidX
011 * Nukkit Project
012 */
013public class PardonCommand extends VanillaCommand {
014
015    public PardonCommand(String name) {
016        super(name, "%nukkit.command.unban.player.description", "%commands.unban.usage");
017        this.setPermission("nukkit.command.unban.player");
018        this.setAliases(new String[]{"unban"});
019        this.commandParameters.clear();
020        this.commandParameters.put("default", new CommandParameter[]{
021                new CommandParameter("player", CommandParamType.TARGET, false)
022        });
023    }
024
025    @Override
026    public boolean execute(CommandSender sender, String commandLabel, String[] args) {
027        if (!this.testPermission(sender)) {
028            return true;
029        }
030
031        if (args.length != 1) {
032            sender.sendMessage(new TranslationContainer("commands.generic.usage", this.usageMessage));
033
034            return false;
035        }
036
037        sender.getServer().getNameBans().remove(args[0]);
038
039        Command.broadcastCommandMessage(sender, new TranslationContainer("%commands.unban.success", args[0]));
040
041        return true;
042    }
043}
