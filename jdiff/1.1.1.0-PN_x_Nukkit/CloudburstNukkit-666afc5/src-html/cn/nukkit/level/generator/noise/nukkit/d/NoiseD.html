007public abstract class NoiseD {
008    protected int[] perm;
009    protected double offsetX = 0;
010    protected double offsetY = 0;
011    protected double offsetZ = 0;
012    protected double octaves = 8;
013    protected double persistence;
014    protected double expansion;
015
016    public static int floor(double x) {
017        return x >= 0 ? (int) x : (int) (x - 1);
018    }
019
020    public static double fade(double x) {
021        return x * x * x * (x * (x * 6 - 15) + 10);
022    }
023
024    public static double lerp(double x, double y, double z) {
025        return y + x * (z - y);
026    }
027
028    public static double linearLerp(double x, double x1, double x2, double q0, double q1) {
029        return ((x2 - x) / (x2 - x1)) * q0 + ((x - x1) / (x2 - x1)) * q1;
030    }
031
032    public static double bilinearLerp(double x, double y, double q00, double q01, double q10, double q11, double x1, double x2, double y1, double y2) {
033        double dx1 = ((x2 - x) / (x2 - x1));
034        double dx2 = ((x - x1) / (x2 - x1));
035
036        return ((y2 - y) / (y2 - y1)) * (
037                dx1 * q00 + dx2 * q10
038        ) + ((y - y1) / (y2 - y1)) * (
039                dx1 * q01 + dx2 * q11
040        );
041    }
042
043    public static double trilinearLerp(double x, double y, double z, double q000, double q001, double q010, double q011, double q100, double q101, double q110, double q111, double x1, double x2, double y1, double y2, double z1, double z2) {
044        double dx1 = ((x2 - x) / (x2 - x1));
045        double dx2 = ((x - x1) / (x2 - x1));
046        double dy1 = ((y2 - y) / (y2 - y1));
047        double dy2 = ((y - y1) / (y2 - y1));
048
049        return ((z2 - z) / (z2 - z1)) * (
050                dy1 * (
051                        dx1 * q000 + dx2 * q100
052                ) + dy2 * (
053                        dx1 * q001 + dx2 * q101
054                )
055        ) + ((z - z1) / (z2 - z1)) * (
056                dy1 * (
057                        dx1 * q010 + dx2 * q110
058                ) + dy2 * (
059                        dx1 * q011 + dx2 * q111
060                )
061        );
062    }
063
064    public static double grad(int hash, double x, double y, double z) {
065        hash &= 15;
066        double u = hash < 8 ? x : y;
067        double v = hash < 4 ? y : ((hash == 12 || hash == 14) ? x :
068                z);
069
070        return ((hash & 1) == 0 ? u : -u) + ((hash & 2) == 0 ? v : -v);
071    }
072
073    abstract public double getNoise2D(double x, double z);
074
075    abstract public double getNoise3D(double x, double y, double z);
076
077    public double noise2D(double x, double z) {
078        return noise2D(x, z, false);
079    }
080
081    public double noise2D(double x, double z, boolean normalized) {
082        double result = 0;
083        double amp = 1;
084        double freq = 1;
085        double max = 0;
086
087        x *= this.expansion;
088        z *= this.expansion;
089
090        for (int i = 0; i < this.octaves; ++i) {
091            result += this.getNoise2D(x * freq, z * freq) * amp;
092            max += amp;
093            freq *= 2;
094            amp *= this.persistence;
095        }
096
097        if (normalized) {
098            result /= max;
099        }
100
101        return result;
102    }
103
104    public double noise3D(double x, double y, double z) {
105        return noise3D(x, y, z, false);
106    }
107
108    public double noise3D(double x, double y, double z, boolean normalized) {
109        double result = 0;
110        double amp = 1;
111        double freq = 1;
112        double max = 0;
113
114        x *= this.expansion;
115        y *= this.expansion;
116        z *= this.expansion;
117
118        for (int i = 0; i < this.octaves; ++i) {
119            result += this.getNoise3D(x * freq, y * freq, z * freq) * amp;
120            max += amp;
121            freq *= 2;
122            amp *= this.persistence;
123        }
124
125        if (normalized) {
126            result /= max;
127        }
128
129        return result;
130    }
131
132    public void setOffset(double x, double y, double z) {
133        this.offsetX = x;
134        this.offsetY = y;
135        this.offsetZ = z;
136    }
137}
