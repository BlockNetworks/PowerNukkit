001package cn.nukkit.level.generator.populator.impl;
002
003import cn.nukkit.level.format.FullChunk;
004import cn.nukkit.level.generator.populator.helper.EnsureBelow;
005import cn.nukkit.level.generator.populator.helper.EnsureCover;
006import cn.nukkit.level.generator.populator.helper.EnsureGrassBelow;
007import cn.nukkit.level.generator.populator.type.PopulatorSurfaceBlock;
008import cn.nukkit.math.NukkitRandom;
009
010/**
011 * @author DaPorkchop_
012 */
013public class PopulatorSmallMushroom extends PopulatorSurfaceBlock {
014
015    @Override
016    protected boolean canStay(int x, int y, int z, FullChunk chunk) {
017        return EnsureCover.ensureCover(x, y, z, chunk) && EnsureGrassBelow.ensureGrassBelow(x, y, z, chunk);
018    }
019
020    @Override
021    protected int getBlockId(int x, int z, NukkitRandom random, FullChunk chunk) {
022        return BROWN_MUSHROOM << 4;
023    }
024}
