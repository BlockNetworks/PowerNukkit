001package cn.nukkit.level.generator.object.tree;
002
003import cn.nukkit.block.Block;
004import cn.nukkit.block.BlockSapling;
005import cn.nukkit.level.ChunkManager;
006import cn.nukkit.math.NukkitRandom;
007
008/**
009 * author: MagicDroidX
010 * Nukkit Project
011 */
012public abstract class ObjectTree {
013    protected boolean overridable(int id) {
014        switch (id) {
015            case Block.AIR:
016            case Block.SAPLING:
017            case Block.LOG:
018            case Block.LEAVES:
019            case Block.SNOW_LAYER:
020            case Block.LOG2:
021            case Block.LEAVES2:
022                return true;
023            default:
024                return false;
025        }
026    }
027
028    public int getType() {
029        return 0;
030    }
031
032    public int getTrunkBlock() {
033        return Block.LOG;
034    }
035
036    public int getLeafBlock() {
037        return Block.LEAVES;
038    }
039
040    public int getTreeHeight() {
041        return 7;
042    }
043
044    public static void growTree(ChunkManager level, int x, int y, int z, NukkitRandom random) {
045        growTree(level, x, y, z, random, 0);
046    }
047
048    public static void growTree(ChunkManager level, int x, int y, int z, NukkitRandom random, int type) {
049        ObjectTree tree;
050        switch (type) {
051            case BlockSapling.SPRUCE:
052                tree = new ObjectSpruceTree();
053                break;
054            case BlockSapling.BIRCH:
055                tree = new ObjectBirchTree();
056                break;
057            case BlockSapling.JUNGLE:
058                tree = new ObjectJungleTree();
059                break;
060            case BlockSapling.BIRCH_TALL:
061                tree = new ObjectTallBirchTree();
062                break;
063            case BlockSapling.OAK:
064            default:
065                tree = new ObjectOakTree();
066                //todo: more complex treeeeeeeeeeeeeeeee
067                break;
068        }
069
070        if (tree.canPlaceObject(level, x, y, z, random)) {
071            tree.placeObject(level, x, y, z, random);
072        }
073    }
074
075
076    public boolean canPlaceObject(ChunkManager level, int x, int y, int z, NukkitRandom random) {
077        int radiusToCheck = 0;
078        for (int yy = 0; yy < this.getTreeHeight() + 3; ++yy) {
079            if (yy == 1 || yy == this.getTreeHeight()) {
080                ++radiusToCheck;
081            }
082            for (int xx = -radiusToCheck; xx < (radiusToCheck + 1); ++xx) {
083                for (int zz = -radiusToCheck; zz < (radiusToCheck + 1); ++zz) {
084                    if (!this.overridable(level.getBlockIdAt(x + xx, y + yy, z + zz))) {
085                        return false;
086                    }
087                }
088            }
089        }
090
091        return true;
092    }
093
094    public void placeObject(ChunkManager level, int x, int y, int z, NukkitRandom random) {
095
096        this.placeTrunk(level, x, y, z, random, this.getTreeHeight() - 1);
097
098        for (int yy = y - 3 + this.getTreeHeight(); yy <= y + this.getTreeHeight(); ++yy) {
099            double yOff = yy - (y + this.getTreeHeight());
100            int mid = (int) (1 - yOff / 2);
101            for (int xx = x - mid; xx <= x + mid; ++xx) {
102                int xOff = Math.abs(xx - x);
103                for (int zz = z - mid; zz <= z + mid; ++zz) {
104                    int zOff = Math.abs(zz - z);
105                    if (xOff == mid && zOff == mid && (yOff == 0 || random.nextBoundedInt(2) == 0)) {
106                        continue;
107                    }
108                    if (!Block.solid[level.getBlockIdAt(xx, yy, zz)]) {
109                        level.setBlockAt(xx, yy, zz, this.getLeafBlock(), this.getType());
110                    }
111                }
112            }
113        }
114    }
115
116    protected void placeTrunk(ChunkManager level, int x, int y, int z, NukkitRandom random, int trunkHeight) {
117        // The base dirt block
118        level.setBlockAt(x, y - 1, z, Block.DIRT);
119
120        for (int yy = 0; yy < trunkHeight; ++yy) {
121            int blockId = level.getBlockIdAt(x, y + yy, z);
122            if (this.overridable(blockId)) {
123                level.setBlockAt(x, y + yy, z, this.getTrunkBlock(), this.getType());
124            }
125        }
126    }
127}
