001package cn.nukkit.level;
002
003import cn.nukkit.Server;
004import cn.nukkit.nbt.NBTIO;
005import cn.nukkit.nbt.tag.CompoundTag;
006import cn.nukkit.nbt.tag.ListTag;
007import it.unimi.dsi.fastutil.ints.Int2IntMap;
008import it.unimi.dsi.fastutil.ints.Int2IntOpenHashMap;
009
010import java.io.IOException;
011import java.io.InputStream;
012import java.nio.ByteOrder;
013import java.util.Arrays;
014import java.util.NoSuchElementException;
015import java.util.concurrent.atomic.AtomicInteger;
016
017public class GlobalBlockPalette {
018    private static final Int2IntMap legacyToRuntimeId = new Int2IntOpenHashMap();
019    private static final Int2IntMap runtimeIdToLegacy = new Int2IntOpenHashMap();
020    private static final AtomicInteger runtimeIdAllocator = new AtomicInteger(0);
021    public static final byte[] BLOCK_PALETTE;
022
023    static {
024        legacyToRuntimeId.defaultReturnValue(-1);
025        runtimeIdToLegacy.defaultReturnValue(-1);
026
027        InputStream stream = Server.class.getClassLoader().getResourceAsStream("runtime_block_states.dat");
028        if (stream == null) {
029            throw new AssertionError("Unable to locate block state nbt");
030        }
031        ListTag<CompoundTag> tag;
032        try {
033            //noinspection unchecked
034            tag = (ListTag<CompoundTag>) NBTIO.readTag(stream, ByteOrder.LITTLE_ENDIAN, false);
035        } catch (IOException e) {
036            throw new AssertionError("Unable to load block palette", e);
037        }
038
039        for (CompoundTag state : tag.getAll()) {
040            int runtimeId = runtimeIdAllocator.getAndIncrement();
041            if (!state.contains("meta")) continue;
042
043            int id = state.getShort("id");
044            int[] meta = state.getIntArray("meta");
045
046            // Resolve to first legacy id
047            runtimeIdToLegacy.put(runtimeId, id << 6 | meta[0]);
048            for (int val : meta) {
049                int legacyId = id << 6 | val;
050                legacyToRuntimeId.put(legacyId, runtimeId);
051            }
052            state.remove("meta"); // No point in sending this since the client doesn't use it.
053        }
054
055        try {
056            BLOCK_PALETTE = NBTIO.write(tag, ByteOrder.LITTLE_ENDIAN, true);
057        } catch (IOException e) {
058            throw new AssertionError("Unable to write block palette", e);
059        }
060    }
061
062    public static int getOrCreateRuntimeId(int id, int meta) {
063        int legacyId = id << 6 | meta;
064        int runtimeId = legacyToRuntimeId.get(legacyId);
065        if (runtimeId == -1) {
066            //runtimeId = registerMapping(runtimeIdAllocator.incrementAndGet(), legacyId);
067            throw new NoSuchElementException("Unmapped block registered id:" + id + " meta:" + meta);
068        }
069        return runtimeId;
070    }
071
072    public static int getOrCreateRuntimeId(int legacyId) throws NoSuchElementException {
073        return getOrCreateRuntimeId(legacyId >> 4, legacyId & 0xf);
074    }
075}
