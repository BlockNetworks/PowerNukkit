001package cn.nukkit.level.biome;
002
003import cn.nukkit.block.BlockID;
004import cn.nukkit.level.ChunkManager;
005import cn.nukkit.level.format.FullChunk;
006import cn.nukkit.level.generator.populator.type.Populator;
007import cn.nukkit.math.NukkitRandom;
008
009import java.util.ArrayList;
010import java.util.List;
011
012/**
013 * author: MagicDroidX
014 * Nukkit Project
015 */
016public abstract class Biome implements BlockID {
017    public static final int MAX_BIOMES = 256;
018    public static final Biome[] biomes = new Biome[MAX_BIOMES];
019    public static final List<Biome> unorderedBiomes = new ArrayList<>();
020
021    private final ArrayList<Populator> populators = new ArrayList<>();
022    private int id;
023    private float baseHeight = 0.1f;
024    private float heightVariation = 0.3f;
025
026    protected static void register(int id, Biome biome) {
027        biome.setId(id);
028        biomes[id] = biome;
029        unorderedBiomes.add(biome);
030    }
031
032    public static Biome getBiome(int id) {
033        Biome biome = biomes[id];
034        return biome != null ? biome : EnumBiome.OCEAN.biome;
035    }
036
037    /**
038     * Get Biome by name.
039     *
040     * @param name Name of biome. Name could contain symbol "_" instead of space
041     * @return Biome. Null - when biome was not found
042     */
043    public static Biome getBiome(String name) {
044        for (Biome biome : biomes) {
045            if (biome != null) {
046                if (biome.getName().equalsIgnoreCase(name.replace("_", " "))) return biome;
047            }
048        }
049        return null;
050    }
051
052    public void clearPopulators() {
053        this.populators.clear();
054    }
055
056    public void addPopulator(Populator populator) {
057        this.populators.add(populator);
058    }
059
060    public void populateChunk(ChunkManager level, int chunkX, int chunkZ, NukkitRandom random) {
061        FullChunk chunk = level.getChunk(chunkX, chunkZ);
062        for (Populator populator : populators) {
063            populator.populate(level, chunkX, chunkZ, random, chunk);
064        }
065    }
066
067    public ArrayList<Populator> getPopulators() {
068        return populators;
069    }
070
071    public int getId() {
072        return id;
073    }
074
075    public void setId(int id) {
076        this.id = id;
077    }
078
079    public abstract String getName();
080
081    public void setBaseHeight(float baseHeight) {
082        this.baseHeight = baseHeight;
083    }
084
085    public void setHeightVariation(float heightVariation)   {
086        this.heightVariation = heightVariation;
087    }
088
089    public float getBaseHeight() {
090        return baseHeight;
091    }
092
093    public float getHeightVariation() {
094        return heightVariation;
095    }
096
097    @Override
098    public int hashCode() {
099        return getId();
100    }
101
102    @Override
103    public boolean equals(Object obj) {
104        return hashCode() == obj.hashCode();
105    }
106
107    //whether or not water should freeze into ice on generation
108    public boolean isFreezing() {
109        return false;
110    }
111
112    /**
113     * Whether or not overhangs should generate in this biome (places where solid blocks generate over air)
114     *
115     * This should probably be used with a custom max elevation or things can look stupid
116     *
117     * @return overhang
118     */
119    public boolean doesOverhang()   {
120        return false;
121    }
122
123    /**
124     * How much offset should be added to the min/max heights at this position
125     *
126     * @param x x
127     * @param z z
128     * @return height offset
129     */
130    public int getHeightOffset(int x, int z)    {
131        return 0;
132    }
133}
