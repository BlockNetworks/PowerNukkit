001package cn.nukkit.level.format.generic;
002
003import cn.nukkit.level.format.FullChunk;
004import cn.nukkit.level.format.LevelProvider;
005import cn.nukkit.level.format.anvil.Chunk;
006import cn.nukkit.level.format.anvil.ChunkSection;
007
008import java.util.ArrayList;
009
010public class ChunkConverter {
011
012    private BaseFullChunk chunk;
013    private Class<? extends FullChunk> toClass;
014    private LevelProvider provider;
015
016    public ChunkConverter(LevelProvider provider) {
017        this.provider = provider;
018    }
019
020    public ChunkConverter from(BaseFullChunk chunk) {
021        if (!(chunk instanceof cn.nukkit.level.format.mcregion.Chunk) && !(chunk instanceof cn.nukkit.level.format.leveldb.Chunk)) {
022            throw new IllegalArgumentException("From type can be only McRegion or LevelDB");
023        }
024        this.chunk = chunk;
025        return this;
026    }
027
028    public ChunkConverter to(Class<? extends FullChunk> toClass) {
029        if (toClass != Chunk.class) {
030            throw new IllegalArgumentException("To type can be only Anvil");
031        }
032        this.toClass = toClass;
033        return this;
034    }
035
036    public FullChunk perform() {
037        BaseFullChunk result;
038        try {
039            result = (BaseFullChunk) toClass.getMethod("getEmptyChunk", int.class, int.class, LevelProvider.class).invoke(null, chunk.getX(), chunk.getZ(), provider);
040        } catch (Exception e) {
041            throw new RuntimeException(e);
042        }
043        if (toClass == Chunk.class) {
044            for (int Y = 0; Y < 8; Y++) {
045                boolean empty = true;
046                for (int x = 0; x < 16; x++) {
047                    for (int y = 0; y < 16; y++) {
048                        for (int z = 0; z < 16; z++) {
049                            if (chunk.getBlockId(x, (Y << 4) | y, z) != 0) {
050                                empty = false;
051                                break;
052                            }
053                        }
054                        if (!empty) break;
055                    }
056                    if (!empty) break;
057                }
058                if (!empty) {
059                    ChunkSection section = new ChunkSection(Y);
060                    for (int x = 0; x < 16; x++) {
061                        for (int y = 0; y < 16; y++) {
062                            for (int z = 0; z < 16; z++) {
063                                section.setBlockId(x, y, z, chunk.getBlockId(x, (Y << 4) | y, z));
064                                section.setBlockData(x, y, z, chunk.getBlockData(x, (Y << 4) | y, z));
065                                section.setBlockLight(x, y, z, chunk.getBlockLight(x, (Y << 4) | y, z));
066                                section.setBlockSkyLight(x, y, z, chunk.getBlockSkyLight(x, (Y << 4) | y, z));
067                            }
068                        }
069                    }
070                    ((BaseChunk) result).sections[Y] = section;
071                }
072            }
073        }
074        System.arraycopy(chunk.biomes, 0, result.biomes, 0, 256);
075        System.arraycopy(chunk.getHeightMapArray(), 0, result.heightMap, 0, 256);
076        if (chunk.NBTentities != null && !chunk.NBTentities.isEmpty()) {
077            result.NBTentities = new ArrayList<>(chunk.NBTentities.size());
078            chunk.NBTentities.forEach((nbt) -> result.NBTentities.add(nbt.copy()));
079        }
080
081        if (chunk.NBTtiles != null && !chunk.NBTtiles.isEmpty()) {
082            result.NBTtiles = new ArrayList<>(chunk.NBTtiles.size());
083            chunk.NBTtiles.forEach((nbt) -> result.NBTtiles.add(nbt.copy()));
084        }
085        result.setGenerated();
086        result.setPopulated();
087        result.setLightPopulated();
088        result.initChunk();
089        return result;
090    }
091}
