001package cn.nukkit.level.format.anvil.palette;
002
003import java.util.Arrays;
004
005/**
006 * @author https://github.com/boy0001/
007 */
008public class IntPalette {
009    private static int[] INT0 = new int[0];
010    private int[] keys = INT0;
011    private int lastIndex = Integer.MIN_VALUE;
012
013    public void add(int key) {
014        keys = insert(key);
015        lastIndex = Integer.MIN_VALUE;
016    }
017
018    protected void set(int[] keys) {
019        this.keys = keys;
020        lastIndex = Integer.MIN_VALUE;
021    }
022
023    private int[] insert(int val) {
024        lastIndex = Integer.MIN_VALUE;
025        if (keys.length == 0) {
026            return new int[] { val };
027        }
028        else if (val < keys[0]) {
029            int[] s = new int[keys.length + 1];
030            System.arraycopy(keys, 0, s, 1, keys.length);
031            s[0] = val;
032            return s;
033        } else if (val > keys[keys.length - 1]) {
034            int[] s = Arrays.copyOf(keys, keys.length + 1);
035            s[keys.length] = val;
036            return s;
037        }
038        int[] s = Arrays.copyOf(keys, keys.length + 1);
039        for (int i = 0; i < s.length; i++) {
040            if (keys[i] < val) {
041                continue;
042            }
043            System.arraycopy(keys, i, s, i + 1, s.length - i - 1);
044            s[i] = val;
045            break;
046        }
047        return s;
048    }
049
050    public int getKey(int index) {
051        return keys[index];
052    }
053
054    public int getValue(int key) {
055        int lastTmp = lastIndex;
056        boolean hasLast = lastTmp != Integer.MIN_VALUE;
057        int index;
058        if (hasLast) {
059            int lastKey = keys[lastTmp];
060            if (lastKey == key) return lastTmp;
061            if (lastKey > key) {
062                index = binarySearch0(0, lastTmp, key);
063            } else {
064                index = binarySearch0(lastTmp + 1, keys.length, key);
065            }
066        } else {
067            index = binarySearch0(0, keys.length, key);
068        }
069        if (index >= keys.length || index < 0) {
070            return lastIndex = Integer.MIN_VALUE;
071        } else {
072            return lastIndex = index;
073        }
074    }
075
076    private int binarySearch0(int fromIndex, int toIndex, int key) {
077        int low = fromIndex;
078        int high = toIndex - 1;
079
080        while (low <= high) {
081            int mid = (low + high) >>> 1;
082            int midVal = keys[mid];
083
084            if (midVal < key)
085                low = mid + 1;
086            else if (midVal > key)
087                high = mid - 1;
088            else
089                return mid; // key found
090        }
091        return -(low + 1);  // key not found.
092    }
093
094    public int length() {
095        return keys.length;
096    }
097
098    public IntPalette clone() {
099        IntPalette p = new IntPalette();
100        p.keys = this.keys != INT0 ? this.keys.clone() : INT0;
101        p.lastIndex = this.lastIndex;
102        return p;
103    }
104}
