001package cn.nukkit.utils;
002
003import cn.nukkit.block.Block;
004import cn.nukkit.math.Vector3;
005
006/**
007 * author: MagicDroidX
008 * Nukkit Project
009 */
010public class BlockUpdateEntry implements Comparable<BlockUpdateEntry> {
011    private static long entryID = 0;
012
013    public int priority;
014    public long delay;
015
016    public final Vector3 pos;
017    public final Block block;
018
019    public final long id;
020
021    public BlockUpdateEntry(Vector3 pos, Block block) {
022        this.pos = pos;
023        this.block = block;
024        this.id = entryID++;
025    }
026
027    public BlockUpdateEntry(Vector3 pos, Block block, long delay, int priority) {
028        this.id = entryID++;
029        this.pos = pos;
030        this.priority = priority;
031        this.delay = delay;
032        this.block = block;
033    }
034
035    @Override
036    public int compareTo(BlockUpdateEntry entry) {
037        return this.delay < entry.delay ? -1 : (this.delay > entry.delay ? 1 : (this.priority != entry.priority ? this.priority - entry.priority : Long.compare(this.id, entry.id)));
038    }
039
040    @Override
041    public boolean equals(Object object) {
042        if (!(object instanceof BlockUpdateEntry)) {
043            if (object instanceof Vector3) {
044                return pos.equals(object);
045            }
046            return false;
047        } else {
048            BlockUpdateEntry entry = (BlockUpdateEntry) object;
049            return this.pos.equals(entry.pos) && Block.equals(this.block, entry.block, false);
050        }
051    }
052
053    @Override
054    public int hashCode() {
055        return this.pos.hashCode();
056    }
057}
