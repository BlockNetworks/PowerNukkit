001package cn.nukkit.utils;
002
003import java.io.ByteArrayInputStream;
004import java.io.IOException;
005import java.io.InputStream;
006import java.util.zip.Deflater;
007
008
009public abstract class Zlib {
010    private static ZlibProvider[] providers;
011    private static ZlibProvider provider;
012
013    static {
014        providers = new ZlibProvider[3];
015        providers[2] = new ZlibThreadLocal();
016        provider = providers[2];
017    }
018
019    public static void setProvider(int providerIndex) {
020        MainLogger.getLogger().info("Selected Zlib Provider: " + providerIndex + " (" + provider.getClass().getCanonicalName() + ")");
021        switch (providerIndex) {
022            case 0:
023                if (providers[providerIndex] == null)
024                    providers[providerIndex] = new ZlibOriginal();
025                break;
026            case 1:
027                if (providers[providerIndex] == null)
028                    providers[providerIndex] = new ZlibSingleThreadLowMem();
029                break;
030            case 2:
031                if (providers[providerIndex] == null)
032                    providers[providerIndex] = new ZlibThreadLocal();
033                break;
034            default:
035                throw new UnsupportedOperationException("Invalid provider: " + providerIndex);
036        }
037        if (providerIndex != 2) {
038            MainLogger.getLogger().warning(" - This Zlib will negatively affect performance");
039        }
040        provider = providers[providerIndex];
041    }
042
043    public static byte[] deflate(byte[] data) throws Exception {
044        return deflate(data, Deflater.DEFAULT_COMPRESSION);
045    }
046
047    public static byte[] deflate(byte[] data, int level) throws Exception {
048        return provider.deflate(data, level);
049    }
050
051    public static byte[] deflate(byte[][] data, int level) throws Exception {
052        return provider.deflate(data, level);
053    }
054
055    public static byte[] inflate(InputStream stream) throws IOException {
056        return provider.inflate(stream);
057    }
058
059    public static byte[] inflate(byte[] data) throws IOException {
060        return inflate(new ByteArrayInputStream(data));
061    }
062
063    public static byte[] inflate(byte[] data, int maxSize) throws IOException {
064        return inflate(new ByteArrayInputStream(data, 0, maxSize));
065    }
066}
