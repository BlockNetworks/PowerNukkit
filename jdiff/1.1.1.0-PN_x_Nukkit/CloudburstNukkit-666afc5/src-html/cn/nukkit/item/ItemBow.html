001package cn.nukkit.item;
002
003import cn.nukkit.Player;
004import cn.nukkit.Server;
005import cn.nukkit.entity.Entity;
006import cn.nukkit.entity.projectile.EntityArrow;
007import cn.nukkit.entity.projectile.EntityProjectile;
008import cn.nukkit.event.entity.EntityShootBowEvent;
009import cn.nukkit.event.entity.ProjectileLaunchEvent;
010import cn.nukkit.item.enchantment.Enchantment;
011import cn.nukkit.math.Vector3;
012import cn.nukkit.nbt.tag.CompoundTag;
013import cn.nukkit.nbt.tag.DoubleTag;
014import cn.nukkit.nbt.tag.FloatTag;
015import cn.nukkit.nbt.tag.ListTag;
016import cn.nukkit.network.protocol.LevelSoundEventPacket;
017
018import java.util.Random;
019
020/**
021 * author: MagicDroidX
022 * Nukkit Project
023 */
024public class ItemBow extends ItemTool {
025
026    public ItemBow() {
027        this(0, 1);
028    }
029
030    public ItemBow(Integer meta) {
031        this(meta, 1);
032    }
033
034    public ItemBow(Integer meta, int count) {
035        super(BOW, meta, count, "Bow");
036    }
037
038    @Override
039    public int getMaxDurability() {
040        return ItemTool.DURABILITY_BOW;
041    }
042
043    @Override
044    public int getEnchantAbility() {
045        return 1;
046    }
047
048    @Override
049    public boolean onClickAir(Player player, Vector3 directionVector) {
050        return player.getInventory().contains(Item.get(ItemID.ARROW)) || player.isCreative();
051    }
052
053    @Override
054    public boolean onRelease(Player player, int ticksUsed) {
055        Item itemArrow = Item.get(Item.ARROW, 0, 1);
056
057        if (player.isSurvival() && !player.getInventory().contains(itemArrow)) {
058            player.getInventory().sendContents(player);
059            return false;
060        }
061
062        double damage = 2;
063
064        Enchantment bowDamage = this.getEnchantment(Enchantment.ID_BOW_POWER);
065        if (bowDamage != null && bowDamage.getLevel() > 0) {
066            damage += 0.25 * (bowDamage.getLevel() + 1);
067        }
068
069        Enchantment flameEnchant = this.getEnchantment(Enchantment.ID_BOW_FLAME);
070        boolean flame = flameEnchant != null && flameEnchant.getLevel() > 0;
071
072        CompoundTag nbt = new CompoundTag()
073                .putList(new ListTag<DoubleTag>("Pos")
074                        .add(new DoubleTag("", player.x))
075                        .add(new DoubleTag("", player.y + player.getEyeHeight()))
076                        .add(new DoubleTag("", player.z)))
077                .putList(new ListTag<DoubleTag>("Motion")
078                        .add(new DoubleTag("", -Math.sin(player.yaw / 180 * Math.PI) * Math.cos(player.pitch / 180 * Math.PI)))
079                        .add(new DoubleTag("", -Math.sin(player.pitch / 180 * Math.PI)))
080                        .add(new DoubleTag("", Math.cos(player.yaw / 180 * Math.PI) * Math.cos(player.pitch / 180 * Math.PI))))
081                .putList(new ListTag<FloatTag>("Rotation")
082                        .add(new FloatTag("", (player.yaw > 180 ? 360 : 0) - (float) player.yaw))
083                        .add(new FloatTag("", (float) -player.pitch)))
084                .putShort("Fire", flame ? 45 * 60 : 0)
085                .putDouble("damage", damage);
086
087        double p = (double) ticksUsed / 20;
088        double f = Math.min((p * p + p * 2) / 3, 1) * 2;
089
090        EntityArrow arrow = (EntityArrow) Entity.createEntity("Arrow", player.chunk, nbt, player, f == 2);
091
092        if (arrow == null) {
093            return false;
094        }
095
096        EntityShootBowEvent entityShootBowEvent = new EntityShootBowEvent(player, this, arrow, f);
097
098        if (f < 0.1 || ticksUsed < 3) {
099            entityShootBowEvent.setCancelled();
100        }
101
102        Server.getInstance().getPluginManager().callEvent(entityShootBowEvent);
103        if (entityShootBowEvent.isCancelled()) {
104            entityShootBowEvent.getProjectile().kill();
105            player.getInventory().sendContents(player);
106        } else {
107            entityShootBowEvent.getProjectile().setMotion(entityShootBowEvent.getProjectile().getMotion().multiply(entityShootBowEvent.getForce()));
108            Enchantment infinityEnchant = this.getEnchantment(Enchantment.ID_BOW_INFINITY);
109            boolean infinity = infinityEnchant != null && infinityEnchant.getLevel() > 0;
110            EntityProjectile projectile;
111            if (infinity && (projectile = entityShootBowEvent.getProjectile()) instanceof EntityArrow) {
112                ((EntityArrow) projectile).setPickupMode(EntityArrow.PICKUP_CREATIVE);
113            }
114            if (player.isSurvival()) {
115                if (!infinity) {
116                    player.getInventory().removeItem(itemArrow);
117                }
118                if (!this.isUnbreakable()) {
119                    Enchantment durability = this.getEnchantment(Enchantment.ID_DURABILITY);
120                    if (!(durability != null && durability.getLevel() > 0 && (100 / (durability.getLevel() + 1)) <= new Random().nextInt(100))) {
121                        this.setDamage(this.getDamage() + 1);
122                        if (this.getDamage() >= getMaxDurability()) {
123                            this.count--;
124                        }
125                        player.getInventory().setItemInHand(this);
126                    }
127                }
128            }
129            if (entityShootBowEvent.getProjectile() != null) {
130                ProjectileLaunchEvent projectev = new ProjectileLaunchEvent(entityShootBowEvent.getProjectile());
131                Server.getInstance().getPluginManager().callEvent(projectev);
132                if (projectev.isCancelled()) {
133                    entityShootBowEvent.getProjectile().kill();
134                } else {
135                    entityShootBowEvent.getProjectile().spawnToAll();
136                    player.getLevel().addLevelSoundEvent(player, LevelSoundEventPacket.SOUND_BOW);
137                }
138            }
139        }
140
141        return true;
142    }
143}
