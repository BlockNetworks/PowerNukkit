024package co.aikar.timings;
025
026import cn.nukkit.Player;
027import cn.nukkit.Server;
028import cn.nukkit.blockentity.BlockEntity;
029import cn.nukkit.entity.Entity;
030import cn.nukkit.level.Level;
031import cn.nukkit.level.format.FullChunk;
032import cn.nukkit.timings.JsonUtil;
033import com.google.gson.Gson;
034import com.google.gson.JsonArray;
035import com.google.gson.JsonObject;
036
037import java.lang.management.ManagementFactory;
038import java.util.Collection;
039import java.util.HashMap;
040import java.util.Map;
041import java.util.concurrent.atomic.AtomicInteger;
042
043import static co.aikar.timings.Timings.fullServerTickTimer;
044import static co.aikar.timings.TimingsManager.MINUTE_REPORTS;
045
046public class TimingsHistory {
047    public static long lastMinuteTime;
048    public static long timedTicks;
049    public static long playerTicks;
050    public static long entityTicks;
051    public static long tileEntityTicks;
052    public static long activatedEntityTicks;
053
054    private static int levelIdPool = 1;
055    static Map<String, Integer> levelMap = new HashMap<>();
056    static Map<Integer, String> entityMap = new HashMap<>();
057    static Map<Integer, String> blockEntityMap = new HashMap<>();
058
059    private final long endTime;
060    private final long startTime;
061    private final long totalTicks;
062    // Represents all time spent running the server this history
063    private final long totalTime;
064    private final MinuteReport[] minuteReports;
065
066    private final TimingsHistoryEntry[] entries;
067    private final JsonObject levels = new JsonObject();
068
069    TimingsHistory() {
070        this.endTime = System.currentTimeMillis() / 1000;
071        this.startTime = TimingsManager.historyStart / 1000;
072
073        if (timedTicks % 1200 != 0 || MINUTE_REPORTS.isEmpty()) {
074            this.minuteReports = MINUTE_REPORTS.toArray(new MinuteReport[MINUTE_REPORTS.size() + 1]);
075            this.minuteReports[this.minuteReports.length - 1] = new MinuteReport();
076        } else {
077            this.minuteReports = MINUTE_REPORTS.toArray(new MinuteReport[0]);
078        }
079
080        long ticks = 0;
081        for (MinuteReport mr : this.minuteReports) {
082            ticks += mr.ticksRecord.timed;
083        }
084
085        this.totalTicks = ticks;
086        this.totalTime = fullServerTickTimer.record.totalTime;
087        this.entries = new TimingsHistoryEntry[TimingsManager.TIMINGS.size()];
088
089        int i = 0;
090        for (Timing timing : TimingsManager.TIMINGS) {
091            this.entries[i++] = new TimingsHistoryEntry(timing);
092        }
093
094        final Map<Integer, AtomicInteger> entityCounts = new HashMap<>();
095        final Map<Integer, AtomicInteger> blockEntityCounts = new HashMap<>();
096        final Gson GSON = new Gson();
097        // Information about all loaded entities/block entities
098        for (Level level : Server.getInstance().getLevels().values()) {
099            JsonArray jsonLevel = new JsonArray();
100            for (FullChunk chunk : level.getChunks().values()) {
101                entityCounts.clear();
102                blockEntityCounts.clear();
103
104                //count entities
105                for (Entity entity : chunk.getEntities().values()) {
106                    if (!entityCounts.containsKey(entity.getNetworkId()))
107                        entityCounts.put(entity.getNetworkId(), new AtomicInteger(0));
108                    entityCounts.get(entity.getNetworkId()).incrementAndGet();
109                    entityMap.put(entity.getNetworkId(), entity.getClass().getSimpleName());
110                }
111
112                //count block entities
113                for (BlockEntity blockEntity : chunk.getBlockEntities().values()) {
114                    if (!blockEntityCounts.containsKey(blockEntity.getBlock().getId()))
115                        blockEntityCounts.put(blockEntity.getBlock().getId(), new AtomicInteger(0));
116                    blockEntityCounts.get(blockEntity.getBlock().getId()).incrementAndGet();
117                    blockEntityMap.put(blockEntity.getBlock().getId(), blockEntity.getClass().getSimpleName());
118                }
119
120                if (blockEntityCounts.isEmpty() && entityCounts.isEmpty()) {
121                    continue;
122                }
123
124                JsonArray jsonChunk = new JsonArray();
125                jsonChunk.add(chunk.getX());
126                jsonChunk.add(chunk.getZ());
127                jsonChunk.add(GSON.toJsonTree(JsonUtil.mapToObject(entityCounts.entrySet(), (entry) -> new JsonUtil.JSONPair(entry.getKey(), entry.getValue().get()))).getAsJsonObject());
128                jsonChunk.add(GSON.toJsonTree(JsonUtil.mapToObject(blockEntityCounts.entrySet(), (entry) -> new JsonUtil.JSONPair(entry.getKey(), entry.getValue().get()))).getAsJsonObject());
129                jsonLevel.add(jsonChunk);
130            }
131
132            if (!levelMap.containsKey(level.getName())) levelMap.put(level.getName(), levelIdPool++);
133            levels.add(String.valueOf(levelMap.get(level.getName())), jsonLevel);
134        }
135    }
136
137    static void resetTicks(boolean fullReset) {
138        if (fullReset) {
139            timedTicks = 0;
140        }
141        lastMinuteTime = System.nanoTime();
142        playerTicks = 0;
143        tileEntityTicks = 0;
144        entityTicks = 0;
145        activatedEntityTicks = 0;
146    }
147
148    JsonObject export() {
149        JsonObject json = new JsonObject();
150        json.addProperty("s", this.startTime);
151        json.addProperty("e", this.endTime);
152        json.addProperty("tk", this.totalTicks);
153        json.addProperty("tm", this.totalTime);
154        json.add("w", this.levels);
155        json.add("h", JsonUtil.mapToArray(this.entries, (entry) -> {
156            if (entry.data.count == 0) {
157                return null;
158            }
159            return entry.export();
160        }));
161        json.add("mp", JsonUtil.mapToArray(this.minuteReports, MinuteReport::export));
162        return json;
163    }
164
165    static class MinuteReport {
166        final long time = System.currentTimeMillis() / 1000;
167
168        final TicksRecord ticksRecord = new TicksRecord();
169        final PingRecord pingRecord = new PingRecord();
170        final TimingData fst = Timings.fullServerTickTimer.minuteData.clone();
171        final double tps = 1E9 / (System.nanoTime() - lastMinuteTime) * this.ticksRecord.timed;
172        final double usedMemory = Timings.fullServerTickTimer.avgUsedMemory;
173        final double freeMemory = Timings.fullServerTickTimer.avgFreeMemory;
174        final double loadAvg = ManagementFactory.getOperatingSystemMXBean().getSystemLoadAverage();
175
176        JsonArray export() {
177            return JsonUtil.toArray(this.time,
178                    Math.round(this.tps * 100D) / 100D,
179                    Math.round(this.pingRecord.avg * 100D) / 100D,
180                    this.fst.export(),
181                    JsonUtil.toArray(this.ticksRecord.timed,
182                            this.ticksRecord.player,
183                            this.ticksRecord.entity,
184                            this.ticksRecord.activatedEntity,
185                            this.ticksRecord.tileEntity),
186                    this.usedMemory,
187                    this.freeMemory,
188                    this.loadAvg);
189        }
190    }
191
192    private static class TicksRecord {
193        final long timed;
194        final long player;
195        final long entity;
196        final long activatedEntity;
197        final long tileEntity;
198
199        TicksRecord() {
200            this.timed = timedTicks - (TimingsManager.MINUTE_REPORTS.size() * 1200);
201            this.player = playerTicks;
202            this.entity = entityTicks;
203            this.activatedEntity = activatedEntityTicks;
204            this.tileEntity = tileEntityTicks;
205        }
206    }
207
208    private static class PingRecord {
209        final double avg;
210
211        PingRecord() {
212            final Collection<Player> onlinePlayers = Server.getInstance().getOnlinePlayers().values();
213            int totalPing = 0;
214            for (Player player : onlinePlayers) {
215                totalPing += player.getPing();
216            }
217
218            this.avg = onlinePlayers.isEmpty() ? 0 : (float) totalPing / onlinePlayers.size();
219        }
220    }
221}
