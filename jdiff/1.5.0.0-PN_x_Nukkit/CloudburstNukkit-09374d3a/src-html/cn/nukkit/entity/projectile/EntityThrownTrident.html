001package cn.nukkit.entity.projectile;
002
003import cn.nukkit.Player;
004import cn.nukkit.entity.Entity;
005import cn.nukkit.event.entity.EntityDamageEvent;
006import cn.nukkit.event.entity.EntityDamageEvent.DamageCause;
007import cn.nukkit.event.entity.EntityDamageByChildEntityEvent;
008import cn.nukkit.event.entity.EntityDamageByEntityEvent;
009import cn.nukkit.event.entity.ProjectileHitEvent;
010import cn.nukkit.level.MovingObjectPosition;
011import cn.nukkit.level.Position;
012import cn.nukkit.item.Item;
013import cn.nukkit.level.format.FullChunk;
014import cn.nukkit.math.Vector3;
015import cn.nukkit.nbt.NBTIO;
016import cn.nukkit.nbt.tag.CompoundTag;
017import cn.nukkit.network.protocol.AddEntityPacket;
018import cn.nukkit.network.protocol.LevelSoundEventPacket;
019
020import java.util.Random;
021import java.util.concurrent.ThreadLocalRandom;
022
023/**
024 * Created by PetteriM1
025 */
026public class EntityThrownTrident extends EntityProjectile {
027
028    public static final int NETWORK_ID = 73;
029
030    public static final int DATA_SOURCE_ID = 17;
031
032    protected Item trident;
033
034    @Override
035    public int getNetworkId() {
036        return NETWORK_ID;
037    }
038
039    @Override
040    public float getWidth() {
041        return 0.05f;
042    }
043
044    @Override
045    public float getLength() {
046        return 0.5f;
047    }
048
049    @Override
050    public float getHeight() {
051        return 0.05f;
052    }
053
054    @Override
055    public float getGravity() {
056        return 0.04f;
057    }
058
059    @Override
060    public float getDrag() {
061        return 0.01f;
062    }
063
064    protected float gravity = 0.04f;
065    protected float drag = 0.01f;
066
067    public EntityThrownTrident(FullChunk chunk, CompoundTag nbt) {
068        this(chunk, nbt, null);
069    }
070
071    public EntityThrownTrident(FullChunk chunk, CompoundTag nbt, Entity shootingEntity) {
072        this(chunk, nbt, shootingEntity, false);
073    }
074
075    public EntityThrownTrident(FullChunk chunk, CompoundTag nbt, Entity shootingEntity, boolean critical) {
076        super(chunk, nbt, shootingEntity);
077    }
078
079    @Override
080    protected void initEntity() {
081        super.initEntity();
082
083        this.damage = namedTag.contains("damage") ? namedTag.getDouble("damage") : 8;
084        this.trident = namedTag.contains("Trident") ? NBTIO.getItemHelper(namedTag.getCompound("Trident")) : Item.get(0);
085
086        closeOnCollide = false;
087    }
088
089    @Override
090    public void saveNBT() {
091        super.saveNBT();
092
093        this.namedTag.put("Trident", NBTIO.putItemHelper(this.trident));
094    }
095
096    public Item getItem() {
097        return this.trident != null ? this.trident.clone() : Item.get(0);
098    }
099
100    public void setItem(Item item) {
101        this.trident = item.clone();
102    }
103
104    public void setCritical() {
105        this.setCritical(true);
106    }
107
108    public void setCritical(boolean value) {
109        this.setDataFlag(DATA_FLAGS, DATA_FLAG_CRITICAL, value);
110    }
111
112    public boolean isCritical() {
113        return this.getDataFlag(DATA_FLAGS, DATA_FLAG_CRITICAL);
114    }
115
116    @Override
117    public int getResultDamage() {
118        int base = super.getResultDamage();
119
120        if (this.isCritical()) {
121            base += ThreadLocalRandom.current().nextInt(base / 2 + 2);
122        }
123
124        return base;
125    }
126
127    @Override
128    protected double getBaseDamage() {
129        return 8;
130    }
131
132    @Override
133    public boolean onUpdate(int currentTick) {
134        if (this.closed) {
135            return false;
136        }
137
138        this.timing.startTiming();
139
140        if (this.isCollided && !this.hadCollision) {
141            this.getLevel().addLevelSoundEvent(this, LevelSoundEventPacket.SOUND_ITEM_TRIDENT_HIT_GROUND);
142        }
143
144        boolean hasUpdate = super.onUpdate(currentTick);
145
146        if (this.onGround || this.hadCollision) {
147            this.setCritical(false);
148        }
149
150        this.timing.stopTiming();
151
152        return hasUpdate;
153    }
154
155    @Override
156    public void spawnTo(Player player) {
157        AddEntityPacket pk = new AddEntityPacket();
158        pk.type = NETWORK_ID;
159        pk.entityUniqueId = this.getId();
160        pk.entityRuntimeId = this.getId();
161        pk.x = (float) this.x;
162        pk.y = (float) this.y;
163        pk.z = (float) this.z;
164        pk.speedX = (float) this.motionX;
165        pk.speedY = (float) this.motionY;
166        pk.speedZ = (float) this.motionZ;
167        pk.yaw = (float) this.yaw;
168        pk.pitch = (float) this.pitch;
169        pk.metadata = this.dataProperties;
170        player.dataPacket(pk);
171
172        super.spawnTo(player);
173    }
174
175    @Override
176    public void onCollideWithEntity(Entity entity) {
177        this.server.getPluginManager().callEvent(new ProjectileHitEvent(this, MovingObjectPosition.fromEntity(entity)));
178        float damage = this.getResultDamage();
179
180        EntityDamageEvent ev;
181        if (this.shootingEntity == null) {
182            ev = new EntityDamageByEntityEvent(this, entity, DamageCause.PROJECTILE, damage);
183        } else {
184            ev = new EntityDamageByChildEntityEvent(this.shootingEntity, this, entity, DamageCause.PROJECTILE, damage);
185        }
186        entity.attack(ev);
187        this.getLevel().addLevelSoundEvent(this, LevelSoundEventPacket.SOUND_ITEM_TRIDENT_HIT);
188        this.hadCollision = true;
189        this.close();
190        Entity newTrident = create("ThrownTrident", this);
191        ((EntityThrownTrident) newTrident).setItem(this.trident);
192        newTrident.spawnToAll();
193    }
194
195    public Entity create(Object type, Position source, Object... args) {
196        FullChunk chunk = source.getLevel().getChunk((int) source.x >> 4, (int) source.z >> 4);
197        if (chunk == null) return null;
198
199        CompoundTag nbt = Entity.getDefaultNBT(source.add(0.5, 0, 0.5), new Vector3(), new Random().nextFloat() * 360, 0);
200
201        return Entity.createEntity(type.toString(), chunk, nbt, args);
202    }
203}
