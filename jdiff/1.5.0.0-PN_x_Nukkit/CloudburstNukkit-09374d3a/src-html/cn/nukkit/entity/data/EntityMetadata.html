001package cn.nukkit.entity.data;
002
003import cn.nukkit.item.Item;
004import cn.nukkit.math.Vector3;
005import cn.nukkit.math.Vector3f;
006import cn.nukkit.nbt.tag.CompoundTag;
007
008import java.util.HashMap;
009import java.util.Map;
010
011/**
012 * author: MagicDroidX
013 * Nukkit Project
014 */
015public class EntityMetadata {
016
017    private final Map<Integer, EntityData> map = new HashMap<>();
018
019    public EntityData get(int id) {
020        return this.getOrDefault(id, null);
021    }
022
023    public EntityData getOrDefault(int id, EntityData defaultValue) {
024        try {
025            return this.map.getOrDefault(id, defaultValue).setId(id);
026        } catch (Exception e) {
027            if (defaultValue != null) {
028                return defaultValue.setId(id);
029            }
030            return null;
031        }
032    }
033
034    public boolean exists(int id) {
035        return this.map.containsKey(id);
036    }
037
038    public EntityMetadata put(EntityData data) {
039        this.map.put(data.getId(), data);
040        return this;
041    }
042
043    public int getByte(int id) {
044        return (int) this.getOrDefault(id, new ByteEntityData(id, 0)).getData() & 0xff;
045    }
046
047    public int getShort(int id) {
048        return (int) this.getOrDefault(id, new ShortEntityData(id, 0)).getData();
049    }
050
051    public int getInt(int id) {
052        return (int) this.getOrDefault(id, new IntEntityData(id, 0)).getData();
053    }
054
055    public long getLong(int id) {
056        return (Long) this.getOrDefault(id, new LongEntityData(id, 0)).getData();
057    }
058
059    public float getFloat(int id) {
060        return (float) this.getOrDefault(id, new FloatEntityData(id, 0)).getData();
061    }
062
063    public boolean getBoolean(int id) {
064        return this.getByte(id) == 1;
065    }
066
067    public CompoundTag getNBT(int id) {
068        return (CompoundTag) this.getOrDefault(id, new NBTEntityData(id, new CompoundTag())).getData();
069    }
070
071    public String getString(int id) {
072        return (String) this.getOrDefault(id, new StringEntityData(id, "")).getData();
073    }
074
075    public Vector3 getPosition(int id) {
076        return (Vector3) this.getOrDefault(id, new IntPositionEntityData(id, new Vector3())).getData();
077    }
078
079    public Vector3f getFloatPosition(int id) {
080        return (Vector3f) this.getOrDefault(id, new Vector3fEntityData(id, new Vector3f())).getData();
081    }
082
083    public EntityMetadata putByte(int id, int value) {
084        return this.put(new ByteEntityData(id, value));
085    }
086
087    public EntityMetadata putShort(int id, int value) {
088        return this.put(new ShortEntityData(id, value));
089    }
090
091    public EntityMetadata putInt(int id, int value) {
092        return this.put(new IntEntityData(id, value));
093    }
094
095    public EntityMetadata putLong(int id, long value) {
096        return this.put(new LongEntityData(id, value));
097    }
098
099    public EntityMetadata putFloat(int id, float value) {
100        return this.put(new FloatEntityData(id, value));
101    }
102
103    public EntityMetadata putBoolean(int id, boolean value) {
104        return this.putByte(id, value ? 1 : 0);
105    }
106
107    public EntityMetadata putNBT(int id, CompoundTag tag) {
108        return this.put(new NBTEntityData(id, tag));
109    }
110
111    @Deprecated
112    public EntityMetadata putSlot(int id, Item value) {
113        return this.put(new NBTEntityData(id, value.getNamedTag()));
114    }
115
116    public EntityMetadata putString(int id, String value) {
117        return this.put(new StringEntityData(id, value));
118    }
119
120    public Map<Integer, EntityData> getMap() {
121        return new HashMap<>(map);
122    }
123}
