001package cn.nukkit.block;
002
003import cn.nukkit.item.Item;
004import cn.nukkit.item.ItemTool;
005
006/**
007 * author: MagicDroidX
008 * Nukkit Project
009 */
010public class BlockOreGold extends BlockSolid {
011
012    public BlockOreGold() {
013    }
014
015    @Override
016    public int getId() {
017        return GOLD_ORE;
018    }
019
020    @Override
021    public double getHardness() {
022        return 3;
023    }
024
025    @Override
026    public double getResistance() {
027        return 15;
028    }
029
030    @Override
031    public int getToolType() {
032        return ItemTool.TYPE_PICKAXE;
033    }
034
035    @Override
036    public String getName() {
037        return "Gold Ore";
038    }
039
040    @Override
041    public Item[] getDrops(Item item) {
042        if (item.isPickaxe() && item.getTier() >= ItemTool.TIER_IRON) {
043            return new Item[]{
044                    Item.get(GOLD_ORE)
045            };
046        } else {
047            return new Item[0];
048        }
049    }
050
051    @Override
052    public boolean canHarvestWithHand() {
053        return false;
054    }
055}
