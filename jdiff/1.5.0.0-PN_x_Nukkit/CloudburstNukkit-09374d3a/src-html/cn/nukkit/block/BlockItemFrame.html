001package cn.nukkit.block;
002
003import cn.nukkit.Player;
004import cn.nukkit.blockentity.BlockEntity;
005import cn.nukkit.blockentity.BlockEntityItemFrame;
006import cn.nukkit.item.Item;
007import cn.nukkit.item.ItemItemFrame;
008import cn.nukkit.level.Level;
009import cn.nukkit.math.BlockFace;
010import cn.nukkit.nbt.tag.CompoundTag;
011import cn.nukkit.nbt.tag.Tag;
012import cn.nukkit.network.protocol.LevelEventPacket;
013
014import java.util.concurrent.ThreadLocalRandom;
015
016/**
017 * Created by Pub4Game on 03.07.2016.
018 */
019public class BlockItemFrame extends BlockTransparentMeta {
020    private final static int[] FACING = new int[]{4, 5, 3, 2, 1, 0}; // TODO when 1.13 support arrives, add UP/DOWN facings
021
022    private final static int FACING_BITMASK = 0b0111;
023    private final static int MAP_BIT = 0b1000;
024
025    public BlockItemFrame() {
026        this(0);
027    }
028
029    public BlockItemFrame(int meta) {
030        super(meta);
031    }
032
033    @Override
034    public int getId() {
035        return ITEM_FRAME_BLOCK;
036    }
037
038    @Override
039    public String getName() {
040        return "Item Frame";
041    }
042
043    @Override
044    public int onUpdate(int type) {
045        if (type == Level.BLOCK_UPDATE_NORMAL) {
046            if (!this.getSide(getFacing()).isSolid()) {
047                this.level.useBreakOn(this);
048                return type;
049            }
050        }
051
052        return 0;
053    }
054
055    @Override
056    public boolean canBeActivated() {
057        return true;
058    }
059
060    @Override
061    public boolean onActivate(Item item, Player player) {
062        BlockEntity blockEntity = this.getLevel().getBlockEntity(this);
063        BlockEntityItemFrame itemFrame = (BlockEntityItemFrame) blockEntity;
064        if (itemFrame.getItem().getId() == Item.AIR) {
065                Item itemOnFrame = item.clone();
066                if (player != null && player.isSurvival()) {
067                        itemOnFrame.setCount(itemOnFrame.getCount() - 1);
068                player.getInventory().setItemInHand(itemOnFrame);
069                }
070            itemOnFrame.setCount(1);
071            itemFrame.setItem(itemOnFrame);
072            this.getLevel().addLevelEvent(this, LevelEventPacket.EVENT_SOUND_ITEM_FRAME_ITEM_ADDED);
073        } else {
074            itemFrame.setItemRotation((itemFrame.getItemRotation() + 1) % 8);
075            this.getLevel().addLevelEvent(this, LevelEventPacket.EVENT_SOUND_ITEM_FRAME_ITEM_ROTATED);
076        }
077        return true;
078    }
079
080    @Override
081    public boolean place(Item item, Block block, Block target, BlockFace face, double fx, double fy, double fz, Player player) {
082        if (face.getIndex() > 1 && target.isSolid() && (!block.isSolid() || block.canBeReplaced())) {
083            this.setDamage(FACING[face.getIndex()]);
084            this.getLevel().setBlock(block, this, true, true);
085            CompoundTag nbt = new CompoundTag()
086                    .putString("id", BlockEntity.ITEM_FRAME)
087                    .putInt("x", (int) block.x)
088                    .putInt("y", (int) block.y)
089                    .putInt("z", (int) block.z)
090                    .putByte("ItemRotation", 0)
091                    .putFloat("ItemDropChance", 1.0f);
092            if (item.hasCustomBlockData()) {
093                for (Tag aTag : item.getCustomBlockData().getAllTags()) {
094                    nbt.put(aTag.getName(), aTag);
095                }
096            }
097            BlockEntityItemFrame frame = (BlockEntityItemFrame) BlockEntity.createBlockEntity(BlockEntity.ITEM_FRAME, this.getLevel().getChunk((int) this.x >> 4, (int) this.z >> 4), nbt);
098            if (frame == null) {
099                return false;
100            }
101            return true;
102        }
103        return false;
104    }
105
106    @Override
107    public boolean onBreak(Item item) {
108        this.getLevel().setBlock(this, Block.get(BlockID.AIR), true, true);
109        this.getLevel().addLevelEvent(this, LevelEventPacket.EVENT_SOUND_ITEM_FRAME_REMOVED);
110        return true;
111    }
112
113    @Override
114    public Item[] getDrops(Item item) {
115        BlockEntity blockEntity = this.getLevel().getBlockEntity(this);
116        BlockEntityItemFrame itemFrame = (BlockEntityItemFrame) blockEntity;
117        if (itemFrame != null && ThreadLocalRandom.current().nextFloat() <= itemFrame.getItemDropChance()) {
118            return new Item[]{
119                    toItem(), itemFrame.getItem().clone()
120            };
121        } else {
122            return new Item[]{
123                    toItem()
124            };
125        }
126    }
127
128    @Override
129    public Item toItem() {
130        return new ItemItemFrame();
131    }
132
133    @Override
134    public boolean canPassThrough() {
135        return true;
136    }
137
138    @Override
139    public boolean hasComparatorInputOverride() {
140        return true;
141    }
142
143    @Override
144    public int getComparatorInputOverride() {
145        BlockEntity blockEntity = this.level.getBlockEntity(this);
146
147        if (blockEntity instanceof BlockEntityItemFrame) {
148            return ((BlockEntityItemFrame) blockEntity).getAnalogOutput();
149        }
150
151        return super.getComparatorInputOverride();
152    }
153
154    public BlockFace getFacing() {
155        switch (this.getDamage() & FACING_BITMASK) {
156            case 0:
157                return BlockFace.WEST;
158            case 1:
159                return BlockFace.EAST;
160            case 2:
161                return BlockFace.NORTH;
162            case 3:
163                return BlockFace.SOUTH;
164        }
165
166        return null;
167    }
168
169    @Override
170    public double getHardness() {
171        return 0.25;
172    }
173}
