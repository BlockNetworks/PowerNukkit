003import cn.nukkit.block.Block;
004import cn.nukkit.level.ChunkManager;
005import cn.nukkit.level.format.FullChunk;

010import java.util.Random;
011
012public class PopulatorRavines extends Populator {
013
014    protected int checkAreaSize = 8;
015
016    private Random random;
017    private long worldLong1;
018    private long worldLong2;
019
020    private int ravineRarity = 1;//2
021    private int ravineMinAltitude = 20;
022    private int ravineMaxAltitude = 67;
023    private int ravineMinLength = 84;
024    private int ravineMaxLength = 111;
025
026    private double ravineDepth = 3;
027
028    private int worldHeightCap = 1 << 8;
029
030    private float[] a = new float[1024];
031
032    @Override
033    public void populate(ChunkManager level, int chunkX, int chunkZ, NukkitRandom random, FullChunk chunk) {
034        this.random = new Random();
035        this.random.setSeed(level.getSeed());
036        worldLong1 = this.random.nextLong();
037        worldLong2 = this.random.nextLong();
038
039        int i = this.checkAreaSize;
040
041        for (int x = chunkX - i; x <= chunkX + i; x++)
042            for (int z = chunkZ - i; z <= chunkZ + i; z++) {
043                long l3 = x * worldLong1;
044                long l4 = z * worldLong2;
045                this.random.setSeed(l3 ^ l4 ^ level.getSeed());
046                generateChunk(chunkX, chunkZ, level.getChunk(chunkX, chunkZ));
047            }
048    }
049
050    protected void generateChunk(int chunkX, int chunkZ, FullChunk generatingChunkBuffer) {
051        if (this.random.nextInt(300) >= this.ravineRarity)
052            return;
053        double d1 = (chunkX * 16) + this.random.nextInt(16);
054        double d2 = numberInRange(random, this.ravineMinAltitude, this.ravineMaxAltitude);
055        double d3 = (chunkZ * 16) + this.random.nextInt(16);
056
057        int i = 1;
058
059        for (int j = 0; j < i; j++) {
060            float f1 = this.random.nextFloat() * 3.141593F * 2.0F;
061            float f2 = (this.random.nextFloat() - 0.5F) * 2.0F / 8.0F;
062            float f3 = (this.random.nextFloat() * 2.0F + this.random.nextFloat()) * 2.0F;
063
064            int size = numberInRange(random, this.ravineMinLength, this.ravineMaxLength);
065
066            createRavine(this.random.nextLong(), generatingChunkBuffer, d1, d2, d3, f3, f1, f2, size, this.ravineDepth);
067        }
068    }
069
070    protected void createRavine(long paramLong, FullChunk generatingChunkBuffer, double paramDouble1, double paramDouble2, double paramDouble3,
071                                float paramFloat1, float paramFloat2, float paramFloat3, int size, double paramDouble4) {
072        Random localRandom = new Random(paramLong);
073
074        int chunkX = generatingChunkBuffer.getX();
075        int chunkZ = generatingChunkBuffer.getZ();
076
077        double d1 = chunkX * 16 + 8;
078        double d2 = chunkZ * 16 + 8;
079
080        float f1 = 0.0F;
081        float f2 = 0.0F;
082
083        int i = 0;
084
085        float f3 = 1.0F;
086        for (int j = 0; ; j++) {
087            if (j >= worldHeightCap)
088                break;
089            if ((j == 0) || (localRandom.nextInt(3) == 0)) {
090                f3 = 1.0F + localRandom.nextFloat() * localRandom.nextFloat() * 1.0F;
091            }
092            this.a[j] = (f3 * f3);
093        }
094
095        for (int stepCount = 0; stepCount < size; stepCount++) {
096            double d3 = 1.5D + MathHelper.sin(stepCount * 3.141593F / size) * paramFloat1 * 1.0F;
097            double d4 = d3 * paramDouble4;
098
099            d3 *= (localRandom.nextFloat() * 0.25D + 0.75D);
100            d4 *= (localRandom.nextFloat() * 0.25D + 0.75D);
101
102            float f4 = MathHelper.cos(paramFloat3);
103            float f5 = MathHelper.sin(paramFloat3);
104            paramDouble1 += MathHelper.cos(paramFloat2) * f4;
105            paramDouble2 += f5;
106            paramDouble3 += MathHelper.sin(paramFloat2) * f4;
107
108            paramFloat3 *= 0.7F;
109
110            paramFloat3 += f2 * 0.05F;
111            paramFloat2 += f1 * 0.05F;
112
113            f2 *= 0.8F;
114            f1 *= 0.5F;
115            f2 += (localRandom.nextFloat() - localRandom.nextFloat()) * localRandom.nextFloat() * 2.0F;
116            f1 += (localRandom.nextFloat() - localRandom.nextFloat()) * localRandom.nextFloat() * 4.0F;
117
118            if ((i == 0) && (localRandom.nextInt(4) == 0)) {
119                continue;
120            }
121            double d5 = paramDouble1 - d1;
122            double d6 = paramDouble3 - d2;
123            double d7 = size - stepCount;
124            double d8 = paramFloat1 + 2.0F + 16.0F;
125            if (d5 * d5 + d6 * d6 - d7 * d7 > d8 * d8) {
126                return;
127            }
128
129            if ((paramDouble1 < d1 - 16.0D - d3 * 2.0D) || (paramDouble3 < d2 - 16.0D - d3 * 2.0D) || (paramDouble1 > d1 + 16.0D + d3 * 2.0D) || (paramDouble3 > d2 + 16.0D + d3 * 2.0D))
130                continue;
131            int k = MathHelper.floor(paramDouble1 - d3) - (chunkX * 16) - 1;
132            int m = MathHelper.floor(paramDouble1 + d3) - (chunkZ * 16) + 1;
133
134            int maxY = MathHelper.floor(paramDouble2 - d4) - 1;
135            int minY = MathHelper.floor(paramDouble2 + d4) + 1;
136
137            int i2 = MathHelper.floor(paramDouble3 - d3) - (chunkX * 16) - 1;
138            int i3 = MathHelper.floor(paramDouble3 + d3) - (chunkZ * 16) + 1;
139
140            if (k < 0)
141                k = 0;
142            if (m > 16)
143                m = 16;
144
145            if (maxY < 1)
146                maxY = 1;
147            if (minY > this.worldHeightCap - 8)
148                minY = this.worldHeightCap - 8;
149
150            if (i2 < 0)
151                i2 = 0;
152            if (i3 > 16)
153                i3 = 16;
154
155            int i4 = 0;
156            for (int localX = k; (i4 == 0) && (localX < m); localX++) {
157                for (int localZ = i2; (i4 == 0) && (localZ < i3); localZ++) {
158                    for (int localY = minY + 1; (i4 == 0) && (localY >= maxY - 1); localY--) {
159                        if (localY < 0)
160                            continue;
161                        if (localY < this.worldHeightCap) {
162                            int materialAtPosition = generatingChunkBuffer.getBlockId(localX, localY, localZ);
163                            if (materialAtPosition == Block.WATER
164                                    || materialAtPosition == Block.STILL_WATER) {
165                                i4 = 1;
166                            }
167                            if ((localY != maxY - 1) && (localX != k) && (localX != m - 1) && (localZ != i2) && (localZ != i3 - 1))
168                                localY = maxY;
169                        }
170                    }
171                }
172            }
173            if (i4 != 0) {
174                continue;
175            }
176            for (int localX = k; localX < m; localX++) {
177                double d9 = (localX + (chunkX * 16) + 0.5D - paramDouble1) / d3;
178                for (int localZ = i2; localZ < i3; localZ++) {
179                    double d10 = (localZ + (chunkZ * 16) + 0.5D - paramDouble3) / d3;
180                    if (d9 * d9 + d10 * d10 < 1.0D) {
181                        for (int localY = minY; localY >= maxY; localY--) {
182                            double d11 = ((localY - 1) + 0.5D - paramDouble2) / d4;
183                            if ((d9 * d9 + d10 * d10) * this.a[localY - 1] + d11 * d11 / 6.0D < 1.0D) {
184                                int material = generatingChunkBuffer.getBlockId(localX, localY, localZ);
185                                if (material == Block.GRASS) {
186                                    if (localY - 1 < 10) {
187                                        generatingChunkBuffer.setBlock(localX, localY, localZ, Block.LAVA);
188                                    } else {
189                                        generatingChunkBuffer.setBlock(localX, localY, localZ, Block.AIR);
190                                    }
191                                }
192                            }
193                        }
194                    }
195                }
196            }
197            if (i != 0)
198                break;
199        }
200    }
201
202    public static int numberInRange(Random random, int min, int max) {
203        return min + random.nextInt(max - min + 1);
204    }
