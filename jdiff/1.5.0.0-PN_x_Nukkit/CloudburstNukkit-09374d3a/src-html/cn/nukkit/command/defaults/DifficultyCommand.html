001package cn.nukkit.command.defaults;
002
003import cn.nukkit.Server;
004import cn.nukkit.command.Command;
005import cn.nukkit.command.CommandSender;
006import cn.nukkit.command.data.CommandEnum;
007import cn.nukkit.command.data.CommandParamType;
008import cn.nukkit.command.data.CommandParameter;
009import cn.nukkit.lang.TranslationContainer;
010import cn.nukkit.network.protocol.SetDifficultyPacket;
011
012import java.util.ArrayList;
013
014/**
015 * Created on 2015/11/12 by xtypr.
016 * Package cn.nukkit.command.defaults in project Nukkit .
017 */
018public class DifficultyCommand extends VanillaCommand {
019
020    public DifficultyCommand(String name) {
021        super(name, "%nukkit.command.difficulty.description", "%commands.difficulty.usage");
022        this.setPermission("nukkit.command.difficulty");
023        this.commandParameters.clear();
024        this.commandParameters.put("default", new CommandParameter[]{
025                CommandParameter.newType("difficulty", CommandParamType.INT)
026        });
027        this.commandParameters.put("byString", new CommandParameter[]{
028                CommandParameter.newEnum("difficulty", new CommandEnum("Difficulty", "peaceful", "p", "easy", "e", "normal", "n", "hard", "h"))
029        });
030    }
031
032    @Override
033    public boolean execute(CommandSender sender, String commandLabel, String[] args) {
034        if (!this.testPermission(sender)) {
035            return true;
036        }
037
038        if (args.length != 1) {
039            sender.sendMessage(new TranslationContainer("commands.generic.usage", this.usageMessage));
040            return false;
041        }
042
043        int difficulty = Server.getDifficultyFromString(args[0]);
044
045        if (sender.getServer().isHardcore()) {
046            difficulty = 3;
047        }
048
049        if (difficulty != -1) {
050            sender.getServer().setPropertyInt("difficulty", difficulty);
051
052            SetDifficultyPacket pk = new SetDifficultyPacket();
053            pk.difficulty = sender.getServer().getDifficulty();
054            Server.broadcastPacket(new ArrayList<>(sender.getServer().getOnlinePlayers().values()), pk);
055
056            Command.broadcastCommandMessage(sender, new TranslationContainer("commands.difficulty.success", String.valueOf(difficulty)));
057        } else {
058            sender.sendMessage(new TranslationContainer("commands.generic.usage", this.usageMessage));
059
060            return false;
061        }
062
063        return true;
064    }
065}
