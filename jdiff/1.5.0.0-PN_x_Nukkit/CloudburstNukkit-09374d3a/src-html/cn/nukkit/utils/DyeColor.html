001package cn.nukkit.utils;
002
003public enum DyeColor {
004
005
006    BLACK(0, 15, "Black", "Ink Sack", BlockColor.BLACK_BLOCK_COLOR),
007    RED(1, 14, "Red", "Rose Red", BlockColor.RED_BLOCK_COLOR),
008    GREEN(2, 13, "Green", "Cactus Green", BlockColor.GREEN_BLOCK_COLOR),
009    BROWN(3, 12, "Brown", "Cocoa Beans", BlockColor.BROWN_BLOCK_COLOR),
010    BLUE(4, 11, "Blue", "Lapis Lazuli", BlockColor.BLUE_BLOCK_COLOR),
011    PURPLE(5, 10, "Purple", BlockColor.PURPLE_BLOCK_COLOR),
012    CYAN(6, 9, "Cyan", BlockColor.CYAN_BLOCK_COLOR),
013    LIGHT_GRAY(7, 8, "Light Gray", BlockColor.LIGHT_GRAY_BLOCK_COLOR),
014    GRAY(8, 7, "Gray", BlockColor.GRAY_BLOCK_COLOR),
015    PINK(9, 6, "Pink", BlockColor.PINK_BLOCK_COLOR),
016    LIME(10, 5, "Lime", BlockColor.LIME_BLOCK_COLOR),
017    YELLOW(11, 4, "Yellow", "Dandelion Yellow", BlockColor.YELLOW_BLOCK_COLOR),
018    LIGHT_BLUE(12, 3, "Light Blue", BlockColor.LIGHT_BLUE_BLOCK_COLOR),
019    MAGENTA(13, 2, "Magenta", BlockColor.MAGENTA_BLOCK_COLOR),
020    ORANGE(14, 1, "Orange", BlockColor.ORANGE_BLOCK_COLOR),
021    WHITE(15, 0, "White", "Bone Meal", BlockColor.WHITE_BLOCK_COLOR);
022
023
024    private int dyeColorMeta;
025    private int woolColorMeta;
026    private String colorName;
027    private String dyeName;
028    private BlockColor blockColor;
029
030
031    private final static DyeColor[] BY_WOOL_DATA;
032    private final static DyeColor[] BY_DYE_DATA;
033
034    DyeColor(int dyeColorMeta, int woolColorMeta, String colorName, BlockColor blockColor) {
035        this(dyeColorMeta, woolColorMeta, colorName, colorName + " Dye", blockColor);
036    }
037
038    DyeColor(int dyeColorMeta, int woolColorMeta, String colorName, String dyeName, BlockColor blockColor) {
039        this.dyeColorMeta = dyeColorMeta;
040        this.woolColorMeta = woolColorMeta;
041        this.colorName = colorName;
042        this.blockColor = blockColor;
043        this.dyeName = dyeName;
044    }
045
046    public BlockColor getColor() {
047        return this.blockColor;
048    }
049
050    public int getDyeData() {
051        return this.dyeColorMeta;
052    }
053
054    public int getWoolData() {
055        return this.woolColorMeta;
056    }
057
058    public String getName() {
059        return this.colorName;
060    }
061
062    public String getDyeName() {
063        return this.dyeName;
064    }
065
066    static {
067        BY_DYE_DATA = values();
068        BY_WOOL_DATA = values();
069
070        for (DyeColor color : values()) {
071            BY_WOOL_DATA[color.woolColorMeta & 0x0f] = color;
072            BY_DYE_DATA[color.dyeColorMeta & 0x0f] = color;
073        }
074    }
075
076    public static DyeColor getByDyeData(int dyeColorMeta) {
077        return BY_DYE_DATA[dyeColorMeta & 0x0f];
078    }
079
080    public static DyeColor getByWoolData(int woolColorMeta) {
081        return BY_WOOL_DATA[woolColorMeta & 0x0f];
082    }
083}
