001package cn.nukkit.raknet.protocol;
002
003import cn.nukkit.utils.Binary;
004import cn.nukkit.utils.BinaryStream;
005
006import java.util.TreeMap;
007
008/**
009 * author: MagicDroidX
010 * Nukkit Project
011 */
012public abstract class AcknowledgePacket extends Packet {
013
014    public TreeMap<Integer, Integer> packets;
015
016    @Override
017    public void encode() {
018        super.encode();
019        int count = this.packets.size();
020        int[] packets = new int[count];
021
022        int index = 0;
023        for (int i : this.packets.values()) {
024            packets[index++] = i;
025        }
026        short records = 0;
027        BinaryStream payload = new BinaryStream();
028
029        if (count > 0) {
030            int pointer = 1;
031            int start = packets[0];
032            int last = packets[0];
033
034            while (pointer < count) {
035                int current = packets[pointer++];
036                int diff = current - last;
037                if (diff == 1) {
038                    last = current;
039                } else if (diff > 1) {
040
041                    if (start == last) {
042                        payload.putByte((byte) 0x01);
043                        payload.put(Binary.writeLTriad(start));
044                        start = last = current;
045                    } else {
046                        payload.putByte((byte) 0x00);
047                        payload.put(Binary.writeLTriad(start));
048                        payload.put(Binary.writeLTriad(last));
049                        start = last = current;
050                    }
051                    ++records;
052                }
053            }
054
055            if (start == last) {
056                payload.putByte((byte) 0x01);
057                payload.put(Binary.writeLTriad(start));
058            } else {
059                payload.putByte((byte) 0x00);
060                payload.put(Binary.writeLTriad(start));
061                payload.put(Binary.writeLTriad(last));
062            }
063            ++records;
064        }
065
066        this.putShort(records);
067        this.buffer = Binary.appendBytes(
068                this.buffer,
069                payload.getBuffer()
070        );
071    }
072
073    @Override
074    public void decode() {
075        super.decode();
076        short count = this.getSignedShort();
077        this.packets = new TreeMap<>();
078        int cnt = 0;
079        for (int i = 0; i < count && !this.feof() && cnt < 4096; ++i) {
080            if (this.getByte() == 0) {
081                int start = this.getLTriad();
082                int end = this.getLTriad();
083                if ((end - start) > 512) {
084                    end = start + 512;
085                }
086                for (int c = start; c <= end; ++c) {
087                    packets.put(cnt++, c);
088                }
089            } else {
090                this.packets.put(cnt++, this.getLTriad());
091            }
092        }
093    }
094
095    @Override
096    public Packet clean() {
097        this.packets = new TreeMap<>();
098        return super.clean();
099    }
100
101    @Override
102    public AcknowledgePacket clone() throws CloneNotSupportedException {
103        AcknowledgePacket packet = (AcknowledgePacket) super.clone();
104        packet.packets = new TreeMap<>(this.packets);
105        return packet;
106    }
107}
