001package cn.nukkit.raknet.protocol.packet;
002
003import cn.nukkit.raknet.RakNet;
004import cn.nukkit.raknet.protocol.Packet;
005
006/**
007 * author: MagicDroidX
008 * Nukkit Project
009 */
010public class OPEN_CONNECTION_REPLY_1 extends Packet {
011    public static final byte ID = (byte) 0x06;
012
013    @Override
014    public byte getID() {
015        return ID;
016    }
017
018    public long serverID;
019    public short mtuSize;
020
021    @Override
022    public void encode() {
023        super.encode();
024        this.put(RakNet.MAGIC);
025        this.putLong(this.serverID);
026        this.putByte((byte) 0); //server security
027        this.putShort(this.mtuSize);
028    }
029
030    @Override
031    public void decode() {
032        super.decode();
033        this.offset += 16; //skip magic bytes
034        this.serverID = this.getLong();
035        this.getByte(); //skip security
036        this.mtuSize = this.getSignedShort();
037    }
038
039    public static final class Factory implements Packet.PacketFactory {
040
041        @Override
042        public Packet create() {
043            return new OPEN_CONNECTION_REPLY_1();
044        }
045
046    }
047}
