001package cn.nukkit.raknet.protocol;
002
003import cn.nukkit.utils.Binary;
004
005import java.io.ByteArrayOutputStream;
006import java.io.IOException;
007
008/**
009 * author: MagicDroidX
010 * Nukkit Project
011 */
012public class EncapsulatedPacket implements Cloneable {
013
014    public int reliability;
015    public boolean hasSplit = false;
016    public int length = 0;
017    public Integer messageIndex = null;
018    public Integer orderIndex = null;
019    public Integer orderChannel = null;
020    public Integer splitCount = null;
021    public Integer splitID = null;
022    public Integer splitIndex = null;
023    public byte[] buffer;
024    public boolean needACK = false;
025    public Integer identifierACK = null;
026
027    private int offset;
028
029    public int getOffset() {
030        return offset;
031    }
032
033    public static EncapsulatedPacket fromBinary(byte[] binary) {
034        return fromBinary(binary, false);
035    }
036
037    public static EncapsulatedPacket fromBinary(byte[] binary, boolean internal) {
038        EncapsulatedPacket packet = new EncapsulatedPacket();
039
040        int flags = binary[0] & 0xff;
041
042        packet.reliability = ((flags & 0b11100000) >> 5);
043        packet.hasSplit = (flags & 0b00010000) > 0;
044        int length, offset;
045        if (internal) {
046            length = Binary.readInt(Binary.subBytes(binary, 1, 4));
047            packet.identifierACK = Binary.readInt(Binary.subBytes(binary, 5, 4));
048            offset = 9;
049        } else {
050            length = (int) Math.ceil(((double) Binary.readShort(Binary.subBytes(binary, 1, 2)) / 8));
051            offset = 3;
052            packet.identifierACK = null;
053        }
054
055        if (packet.reliability > 0) {
056            if (packet.reliability >= 2 && packet.reliability != 5) {
057                packet.messageIndex = Binary.readLTriad(Binary.subBytes(binary, offset, 3));
058                offset += 3;
059            }
060
061            if (packet.reliability <= 4 && packet.reliability != 2) {
062                packet.orderIndex = Binary.readLTriad(Binary.subBytes(binary, offset, 3));
063                offset += 3;
064                packet.orderChannel = binary[offset++] & 0xff;
065            }
066        }
067
068        if (packet.hasSplit) {
069            packet.splitCount = Binary.readInt(Binary.subBytes(binary, offset, 4));
070            offset += 4;
071            packet.splitID = Binary.readShort(Binary.subBytes(binary, offset, 2));
072            offset += 2;
073            packet.splitIndex = Binary.readInt(Binary.subBytes(binary, offset, 4));
074            offset += 4;
075        }
076
077        packet.buffer = Binary.subBytes(binary, offset, length);
078        offset += length;
079        packet.offset = offset;
080
081        return packet;
082    }
083
084    public int getTotalLength() {
085        return 3 + this.buffer.length + (this.messageIndex != null ? 3 : 0) + (this.orderIndex != null ? 4 : 0) + (this.hasSplit ? 10 : 0);
086    }
087
088    public byte[] toBinary() {
089        return toBinary(false);
090    }
091
092    public byte[] toBinary(boolean internal) {
093        ByteArrayOutputStream stream = new ByteArrayOutputStream();
094        try {
095            stream.write((reliability << 5) | (hasSplit ? 0b00010000 : 0));
096            if (internal) {
097                stream.write(Binary.writeInt(buffer.length));
098                stream.write(Binary.writeInt(identifierACK == null ? 0 : identifierACK));
099            } else {
100                stream.write(Binary.writeShort(buffer.length << 3));
101            }
102
103            if (reliability > 0) {
104                if (reliability >= 2 && reliability != 5) {
105                    stream.write(Binary.writeLTriad(messageIndex == null ? 0 : messageIndex));
106                }
107                if (reliability <= 4 && reliability != 2) {
108                    stream.write(Binary.writeLTriad(orderIndex));
109                    stream.write((byte) (orderChannel & 0xff));
110                }
111            }
112
113            if (hasSplit) {
114                stream.write(Binary.writeInt(splitCount));
115                stream.write(Binary.writeShort(splitID));
116                stream.write(Binary.writeInt(splitIndex));
117            }
118
119            stream.write(buffer);
120        } catch (IOException e) {
121            throw new RuntimeException(e);
122        }
123
124        return stream.toByteArray();
125    }
126
127    @Override
128    public String toString() {
129        return Binary.bytesToHexString(this.toBinary());
130    }
131
132    @Override
133    public EncapsulatedPacket clone() throws CloneNotSupportedException {
134        EncapsulatedPacket packet = (EncapsulatedPacket) super.clone();
135        packet.buffer = this.buffer.clone();
136        return packet;
137    }
138}
