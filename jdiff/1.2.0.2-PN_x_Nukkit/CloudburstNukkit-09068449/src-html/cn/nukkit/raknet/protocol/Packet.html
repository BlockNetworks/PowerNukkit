001package cn.nukkit.raknet.protocol;
002
003import cn.nukkit.utils.Binary;
004
005import java.net.InetSocketAddress;
006import java.nio.charset.StandardCharsets;
007import java.util.Arrays;
008
009/**
010 * author: MagicDroidX
011 * Nukkit Project
012 */
013public abstract class Packet implements Cloneable {
014
015    protected int offset = 0;
016    public byte[] buffer;
017    public Long sendTime;
018
019    public abstract byte getID();
020
021    protected byte[] get(int len) {
022        if (len < 0) {
023            this.offset = this.buffer.length - 1;
024            return new byte[0];
025        }
026
027        byte[] buffer = new byte[len];
028        for (int i = 0; i < len; i++) {
029            buffer[i] = this.buffer[this.offset++];
030        }
031        return buffer;
032    }
033
034    protected byte[] getAll() {
035        return this.get();
036    }
037
038    protected byte[] get() {
039        try {
040            return Arrays.copyOfRange(this.buffer, this.offset, this.buffer.length - 1);
041        } catch (Exception e) {
042            return new byte[0];
043        }
044    }
045
046    protected long getLong() {
047        return Binary.readLong(this.get(8));
048    }
049
050    protected int getInt() {
051        return Binary.readInt(this.get(4));
052    }
053
054    protected short getSignedShort() {
055        return (short) this.getShort();
056    }
057
058    protected int getShort() {
059        return Binary.readShort(this.get(2));
060    }
061
062    protected int getTriad() {
063        return Binary.readTriad(this.get(3));
064    }
065
066    protected int getLTriad() {
067        return Binary.readLTriad(this.get(3));
068    }
069
070    protected byte getByte() {
071        return this.buffer[this.offset++];
072    }
073
074    protected String getString() {
075        return new String(this.get(this.getSignedShort()), StandardCharsets.UTF_8);
076    }
077
078    protected InetSocketAddress getAddress() {
079        byte version = this.getByte();
080        if (version == 4) {
081            String addr = ((~this.getByte()) & 0xff) + "." + ((~this.getByte()) & 0xff) + "." + ((~this.getByte()) & 0xff) + "." + ((~this.getByte()) & 0xff);
082            int port = this.getShort();
083            return new InetSocketAddress(addr, port);
084        } else {
085            //todo IPV6 SUPPORT
086            return null;
087        }
088    }
089
090    protected boolean feof() {
091        return !(this.offset >= 0 && this.offset + 1 <= this.buffer.length);
092    }
093
094    protected void put(byte[] b) {
095        this.buffer = Binary.appendBytes(this.buffer, b);
096    }
097
098    protected void putLong(long v) {
099        this.put(Binary.writeLong(v));
100    }
101
102    protected void putInt(int v) {
103        this.put(Binary.writeInt(v));
104    }
105
106    protected void putShort(int v) {
107        this.put(Binary.writeShort(v));
108    }
109
110    protected void putSignedShort(short v) {
111        this.put(Binary.writeShort(v & 0xffff));
112    }
113
114    protected void putTriad(int v) {
115        this.put(Binary.writeTriad(v));
116    }
117
118    protected void putLTriad(int v) {
119        this.put(Binary.writeLTriad(v));
120    }
121
122    protected void putByte(byte b) {
123        byte[] newBytes = new byte[this.buffer.length + 1];
124        System.arraycopy(this.buffer, 0, newBytes, 0, this.buffer.length);
125        newBytes[this.buffer.length] = b;
126        this.buffer = newBytes;
127    }
128
129    protected void putString(String str) {
130        byte[] b = str.getBytes(StandardCharsets.UTF_8);
131        this.putShort(b.length);
132        this.put(b);
133    }
134
135    protected void putAddress(String addr, int port) {
136        this.putAddress(addr, port, (byte) 4);
137    }
138
139    protected void putAddress(String addr, int port, byte version) {
140        this.putByte(version);
141        if (version == 4) {
142            for (String b : addr.split("\\.")) {
143                this.putByte((byte) ((~Integer.valueOf(b)) & 0xff));
144            }
145            this.putShort(port);
146        } else {
147            //todo ipv6
148        }
149    }
150
151    protected void putAddress(InetSocketAddress address) {
152        this.putAddress(address.getHostString(), address.getPort());
153    }
154
155    public void encode() {
156        this.buffer = new byte[]{getID()};
157    }
158
159    public void decode() {
160        this.offset = 1;
161    }
162
163    public Packet clean() {
164        this.buffer = null;
165        this.offset = 0;
166        this.sendTime = null;
167        return this;
168    }
169
170    @Override
171    public Packet clone() throws CloneNotSupportedException {
172        Packet packet = (Packet) super.clone();
173        packet.buffer = this.buffer.clone();
174        return packet;
175    }
176
177    /**
178     * A factory to create new packet instances
179     */
180    public interface PacketFactory {
181        /**
182         * Creates the packet
183         */
184        Packet create();
185    }
186}
