073    private void tickProcessor() throws Exception {
074        this.lastMeasure = System.currentTimeMillis();
075        while (!this.shutdown) {
076            long start = System.currentTimeMillis();
077            int max = 5000;
078            while (max > 0) {
079                try {
080                    if (!this.receivePacket()) {
081                        break;
082                    }
083                    --max;
084                } catch (Exception e) {
085                    if (!currentSource.isEmpty()) {
086                        this.blockAddress(currentSource);
087                    }
088                    // else ignore
089                }
090            }
091            while (this.receiveStream()) ;
092
093            long time = System.currentTimeMillis() - start;
094            if (time < 50) {
095                try {
096                    Thread.sleep(50 - time);
097                } catch (InterruptedException e) {
098                    //ignore
099                }
100            }
101            this.tick();
102        }
103    }

105    private void tick() throws Exception {
106        long time = System.currentTimeMillis();
107        for (Session session : new ArrayList<>(this.sessions.values())) {
108            session.update(time);
109        }
110
111        for (String address : this.ipSec.keySet()) {
112            int count = this.ipSec.get(address);
113            if (count >= this.packetLimit) {
114                this.blockAddress(address);
115            }
116        }
117        this.ipSec.clear();
118
119        if ((this.ticks & 0b1111) == 0) {
120            double diff = Math.max(5d, (double) time - this.lastMeasure);
121            this.streamOption("bandwidth", this.sendBytes / diff + ";" + this.receiveBytes / diff);
122            this.lastMeasure = time;
123            this.sendBytes = 0;
124            this.receiveBytes = 0;
125
126            if (!this.block.isEmpty()) {
127                long now = System.currentTimeMillis();
128                for (String address : new ArrayList<>(this.block.keySet())) {
129                    long timeout = this.block.get(address);
130                    if (timeout <= now) {
131                        this.block.remove(address);
132                        this.getLogger().notice("Unblocked " + address);
133                    } else {
134                        break;
135                    }
136                }
137            }
138        }
139
140        ++this.ticks;
141    }

143    private boolean receivePacket() throws Exception {
144        DatagramPacket datagramPacket = this.socket.readPacket();
145        if (datagramPacket != null) {
146            // Check this early
147            try {
148                String source = datagramPacket.sender().getHostString();
149                currentSource = source; //in order to block address
150                if (this.block.containsKey(source)) {
151                    return true;
152                }
153
154                if (this.ipSec.containsKey(source)) {
155                    this.ipSec.put(source, this.ipSec.get(source) + 1);
156                } else {
157                    this.ipSec.put(source, 1);
158                }
159
160                ByteBuf byteBuf = datagramPacket.content();
161                if (byteBuf.readableBytes() == 0) {
162                    // Exit early to process another packet
163                    return true;
164                }
165                byte[] buffer = new byte[byteBuf.readableBytes()];
166                byteBuf.readBytes(buffer);
167                int len = buffer.length;
168                int port = datagramPacket.sender().getPort();
169
170                this.receiveBytes += len;
171
172                byte pid = buffer[0];
173
174                if (pid == UNCONNECTED_PONG.ID) {
175                    return false;
176                }
177
178                Packet packet = this.getPacketFromPool(pid);
179                if (packet != null) {
180                    packet.buffer = buffer;
181                    this.getSession(source, port).handlePacket(packet);
182                    return true;
183                } else if (pid == UNCONNECTED_PING.ID) {
184                    packet = new UNCONNECTED_PING();
185                    packet.buffer = buffer;
186                    packet.decode();
187
188                    UNCONNECTED_PONG pk = new UNCONNECTED_PONG();
189                    pk.serverID = this.getID();
190                    pk.pingID = ((UNCONNECTED_PING) packet).pingID;
191                    pk.serverName = this.getName();
192                    this.sendPacket(pk, source, port);
193                } else if (buffer.length != 0) {
194                    this.streamRAW(source, port, buffer);
195                    return true;
196                } else {
197                    return false;
198                }
199            } finally {
200                datagramPacket.release();
201            }
202        }
203
204        return false;
205    }

221    public void streamEncapsulated(Session session, EncapsulatedPacket packet, int flags) {
222        String id = session.getAddress() + ":" + session.getPort();
223        byte[] buffer = Binary.appendBytes(
224                RakNet.PACKET_ENCAPSULATED,
225                new byte[]{(byte) (id.length() & 0xff)},
226                id.getBytes(StandardCharsets.UTF_8),
227                new byte[]{(byte) (flags & 0xff)},
228                packet.toBinary(true)
229        );
230        this.server.pushThreadToMainPacket(buffer);
231    }

298    private void checkSessions() {
299        int size = this.sessions.size();
300        if (size > 4096) {
301            List<String> keyToRemove = new ArrayList<>();
302            for (String i : this.sessions.keySet()) {
303                Session s = this.sessions.get(i);
304                if (s.isTemporal()) {
305                    keyToRemove.add(i);
306                    size--;
307                    if (size <= 4096) {
308                        break;
309                    }
310                }
311            }
312
313            for (String i : keyToRemove) {
314                this.sessions.remove(i);
315            }
316        }
317    }

319    public boolean receiveStream() throws Exception {
320        byte[] packet = this.server.readMainToThreadPacket();
321        if (packet != null && packet.length > 0) {
322            byte id = packet[0];
323            int offset = 1;
324            switch (id) {
325                case RakNet.PACKET_ENCAPSULATED:
326                    int len = packet[offset++];
327                    String identifier = new String(Binary.subBytes(packet, offset, len), StandardCharsets.UTF_8);
328                    offset += len;
329                    if (this.sessions.containsKey(identifier)) {
330                        byte flags = packet[offset++];
331                        byte[] buffer = Binary.subBytes(packet, offset);
332                        this.sessions.get(identifier).addEncapsulatedToQueue(EncapsulatedPacket.fromBinary(buffer, true), flags);
333                    } else {
334                        this.streamInvalid(identifier);
335                    }
336                    break;
337                case RakNet.PACKET_RAW:
338                    len = packet[offset++];
339                    String address = new String(Binary.subBytes(packet, offset, len), StandardCharsets.UTF_8);
340                    offset += len;
341                    int port = Binary.readShort(Binary.subBytes(packet, offset, 2));
342                    offset += 2;
343                    byte[] payload = Binary.subBytes(packet, offset);
344                    this.socket.writePacket(payload, address, port);
345                    break;
346                case RakNet.PACKET_CLOSE_SESSION:
347                    len = packet[offset++];
348                    identifier = new String(Binary.subBytes(packet, offset, len), StandardCharsets.UTF_8);
349                    if (this.sessions.containsKey(identifier)) {
350                        this.removeSession(this.sessions.get(identifier));
351                    } else {
352                        this.streamInvalid(identifier);
353                    }
354                    break;
355                case RakNet.PACKET_INVALID_SESSION:
356                    len = packet[offset++];
357                    identifier = new String(Binary.subBytes(packet, offset, len), StandardCharsets.UTF_8);
358                    if (this.sessions.containsKey(identifier)) {
359                        this.removeSession(this.sessions.get(identifier));
360                    }
361                    break;
362                case RakNet.PACKET_SET_OPTION:
363                    len = packet[offset++];
364                    String name = new String(Binary.subBytes(packet, offset, len), StandardCharsets.UTF_8);
365                    offset += len;
366                    String value = new String(Binary.subBytes(packet, offset), StandardCharsets.UTF_8);
367                    switch (name) {
368                        case "name":
369                            this.name = value;
370                            break;
371                        case "portChecking":
372                            this.portChecking = Boolean.valueOf(value);
373                            break;
374                        case "packetLimit":
375                            this.packetLimit = Integer.valueOf(value);
376                            break;
377                    }
378                    break;
379                case RakNet.PACKET_BLOCK_ADDRESS:
380                    len = packet[offset++];
381                    address = new String(Binary.subBytes(packet, offset, len), StandardCharsets.UTF_8);
382                    offset += len;
383                    int timeout = Binary.readInt(Binary.subBytes(packet, offset, 4));
384                    this.blockAddress(address, timeout);
385                    break;
386                case RakNet.PACKET_UNBLOCK_ADDRESS:
387                    len = packet[offset++];
388                    address = new String(Binary.subBytes(packet, offset, len), StandardCharsets.UTF_8);
389                    this.unblockAddress(address);
390                    break;
391                case RakNet.PACKET_SHUTDOWN:
392                    for (Session session : new ArrayList<>(this.sessions.values())) {
393                        this.removeSession(session);
394                    }
395
396                    this.socket.close();
397                    this.shutdown = true;
398                    break;
399                case RakNet.PACKET_EMERGENCY_SHUTDOWN:
400                    this.shutdown = true;
401                default:
402                    return false;
403            }
404            return true;
405        }
406
407        return false;
408    }

410    public void blockAddress(String address) {
411        this.blockAddress(address, 300);
412    }
413
414    public void blockAddress(String address, int timeout) {
415        long finalTime = System.currentTimeMillis() + timeout * 1000;
416        if (!this.block.containsKey(address) || timeout == -1) {
417            if (timeout == -1) {
418                finalTime = Long.MAX_VALUE;
419            } else {
420                this.getLogger().notice("Blocked " + address + " for " + timeout + " seconds");
421            }
422            this.block.put(address, finalTime);
423        } else if (this.block.get(address) < finalTime) {
424            this.block.put(address, finalTime);
425        }
426    }
427
428    public void unblockAddress(String address) {
429        this.block.remove(address);
430    }

432    public Session getSession(String ip, int port) {
433        String id = ip + ":" + port;
434        if (!this.sessions.containsKey(id)) {
435            this.checkSessions();
436            Session session = new Session(this, ip, port);
437            this.sessions.put(id, session);
438
439            return session;
440        }
441
442        return this.sessions.get(id);
443    }

445    public void removeSession(Session session) throws Exception {
446        this.removeSession(session, "unknown");
447    }

449    public void removeSession(Session session, String reason) throws Exception {
450        String id = session.getAddress() + ":" + session.getPort();
451        if (this.sessions.containsKey(id)) {
452            this.sessions.get(id).close();
453            this.sessions.remove(id);
454            this.streamClose(id, reason);
455        }
456    }
