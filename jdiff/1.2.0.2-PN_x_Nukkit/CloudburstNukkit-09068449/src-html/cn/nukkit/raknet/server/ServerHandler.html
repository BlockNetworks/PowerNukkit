001package cn.nukkit.raknet.server;
002
003import cn.nukkit.raknet.RakNet;
004import cn.nukkit.raknet.protocol.EncapsulatedPacket;
005import cn.nukkit.utils.Binary;
006
007import java.nio.charset.StandardCharsets;
008
009/**
010 * author: MagicDroidX
011 * Nukkit Project
012 */
013public class ServerHandler {
014
015    protected final RakNetServer server;
016
017    protected final ServerInstance instance;
018
019    public ServerHandler(RakNetServer server, ServerInstance instance) {
020        this.server = server;
021        this.instance = instance;
022    }
023
024    public void sendEncapsulated(String identifier, EncapsulatedPacket packet) {
025        this.sendEncapsulated(identifier, packet, RakNet.PRIORITY_NORMAL);
026    }
027
028    public void sendEncapsulated(String identifier, EncapsulatedPacket packet, int flags) {
029        byte[] buffer = Binary.appendBytes(
030                RakNet.PACKET_ENCAPSULATED,
031                new byte[]{(byte) (identifier.length() & 0xff)},
032                identifier.getBytes(StandardCharsets.UTF_8),
033                new byte[]{(byte) (flags & 0xff)},
034                packet.toBinary(true)
035        );
036        this.server.pushMainToThreadPacket(buffer);
037    }
038
039    public void sendRaw(String address, int port, byte[] payload) {
040        byte[] buffer = Binary.appendBytes(
041                RakNet.PACKET_RAW,
042                new byte[]{(byte) (address.length() & 0xff)},
043                address.getBytes(StandardCharsets.UTF_8),
044                Binary.writeShort(port),
045                payload
046        );
047        this.server.pushMainToThreadPacket(buffer);
048    }
049
050    public void closeSession(String identifier, String reason) {
051        byte[] buffer = Binary.appendBytes(
052                RakNet.PACKET_CLOSE_SESSION,
053                new byte[]{(byte) (identifier.length() & 0xff)},
054                identifier.getBytes(StandardCharsets.UTF_8),
055                new byte[]{(byte) (reason.length() & 0xff)},
056                reason.getBytes(StandardCharsets.UTF_8)
057        );
058        this.server.pushMainToThreadPacket(buffer);
059    }
060
061    public void sendOption(String name, String value) {
062        byte[] buffer = Binary.appendBytes(
063                RakNet.PACKET_SET_OPTION,
064                new byte[]{(byte) (name.length() & 0xff)},
065                name.getBytes(StandardCharsets.UTF_8),
066                value.getBytes(StandardCharsets.UTF_8)
067        );
068        this.server.pushMainToThreadPacket(buffer);
069    }
070
071    public void blockAddress(String address, int timeout) {
072        byte[] buffer = Binary.appendBytes(
073                RakNet.PACKET_BLOCK_ADDRESS,
074                new byte[]{(byte) (address.length() & 0xff)},
075                address.getBytes(StandardCharsets.UTF_8),
076                Binary.writeInt(timeout)
077        );
078        this.server.pushMainToThreadPacket(buffer);
079    }
080
081    public void unblockAddress(String address) {
082        byte[] buffer = Binary.appendBytes(
083                RakNet.PACKET_UNBLOCK_ADDRESS,
084                new byte[]{(byte) (address.length() & 0xff)},
085                address.getBytes(StandardCharsets.UTF_8)
086        );
087        this.server.pushMainToThreadPacket(buffer);
088    }
089
090    public void shutdown() {
091        this.server.pushMainToThreadPacket(new byte[]{RakNet.PACKET_SHUTDOWN});
092        this.server.shutdown();
093        synchronized (this) {
094            try {
095                this.wait(20);
096            } catch (InterruptedException e) {
097                //ignore
098            }
099        }
100        try {
101            this.server.join();
102        } catch (InterruptedException e) {
103            //ignore
104        }
105    }
106
107    public void emergencyShutdown() {
108        this.server.shutdown();
109        this.server.pushMainToThreadPacket(new byte[]{RakNet.PACKET_EMERGENCY_SHUTDOWN});
110    }
111
112    protected void invalidSession(String identifier) {
113        byte[] buffer = Binary.appendBytes(
114                RakNet.PACKET_INVALID_SESSION,
115                new byte[]{(byte) (identifier.length() & 0xff)},
116                identifier.getBytes(StandardCharsets.UTF_8)
117        );
118        this.server.pushMainToThreadPacket(buffer);
119    }
120
121    public boolean handlePacket() {
122        byte[] packet = this.server.readThreadToMainPacket();
123        if (packet != null && packet.length > 0) {
124            byte id = packet[0];
125            int offset = 1;
126            if (id == RakNet.PACKET_ENCAPSULATED) {
127                int len = packet[offset++];
128                String identifier = new String(Binary.subBytes(packet, offset, len), StandardCharsets.UTF_8);
129                offset += len;
130                int flags = packet[offset++];
131                byte[] buffer = Binary.subBytes(packet, offset);
132                this.instance.handleEncapsulated(identifier, EncapsulatedPacket.fromBinary(buffer, true), flags);
133            } else if (id == RakNet.PACKET_RAW) {
134                int len = packet[offset++];
135                String address = new String(Binary.subBytes(packet, offset, len), StandardCharsets.UTF_8);
136                offset += len;
137                int port = Binary.readShort(Binary.subBytes(packet, offset, 2)) & 0xffff;
138                offset += 2;
139                byte[] payload = Binary.subBytes(packet, offset);
140                this.instance.handleRaw(address, port, payload);
141            } else if (id == RakNet.PACKET_SET_OPTION) {
142                int len = packet[offset++];
143                String name = new String(Binary.subBytes(packet, offset, len), StandardCharsets.UTF_8);
144                offset += len;
145                String value = new String(Binary.subBytes(packet, offset), StandardCharsets.UTF_8);
146                this.instance.handleOption(name, value);
147            } else if (id == RakNet.PACKET_OPEN_SESSION) {
148                int len = packet[offset++];
149                String identifier = new String(Binary.subBytes(packet, offset, len), StandardCharsets.UTF_8);
150                offset += len;
151                len = packet[offset++];
152                String address = new String(Binary.subBytes(packet, offset, len), StandardCharsets.UTF_8);
153                offset += len;
154                int port = Binary.readShort(Binary.subBytes(packet, offset, 2)) & 0xffff;
155                offset += 2;
156                long clientID = Binary.readLong(Binary.subBytes(packet, offset, 8));
157                this.instance.openSession(identifier, address, port, clientID);
158            } else if (id == RakNet.PACKET_CLOSE_SESSION) {
159                int len = packet[offset++];
160                String identifier = new String(Binary.subBytes(packet, offset, len), StandardCharsets.UTF_8);
161                offset += len;
162                len = packet[offset++];
163                String reason = new String(Binary.subBytes(packet, offset, len), StandardCharsets.UTF_8);
164                this.instance.closeSession(identifier, reason);
165            } else if (id == RakNet.PACKET_INVALID_SESSION) {
166                int len = packet[offset++];
167                String identifier = new String(Binary.subBytes(packet, offset, len), StandardCharsets.UTF_8);
168                this.instance.closeSession(identifier, "Invalid session");
169            } else if (id == RakNet.PACKET_ACK_NOTIFICATION) {
170                int len = packet[offset++];
171                String identifier = new String(Binary.subBytes(packet, offset, len), StandardCharsets.UTF_8);
172                offset += len;
173                int identifierACK = Binary.readInt(Binary.subBytes(packet, offset, 4));
174                this.instance.notifyACK(identifier, identifierACK);
175            }
176            return true;
177        }
178
179        return false;
180    }
181
182}
