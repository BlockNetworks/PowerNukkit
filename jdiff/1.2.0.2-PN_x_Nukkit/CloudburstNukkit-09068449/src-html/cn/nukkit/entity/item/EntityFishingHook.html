090    @Override
091    public boolean onUpdate(int currentTick) {
092        boolean hasUpdate = super.onUpdate(currentTick);
093        if (hasUpdate) {
094            return false;
095        }
096
097        if (this.isInsideOfWater()) {
098            this.motionX = 0;
099            this.motionY -= getGravity() * -0.04;
100            this.motionZ = 0;
101            hasUpdate = true;
102        } else if (this.isCollided && this.keepMovement) {
103            this.motionX = 0;
104            this.motionY = 0;
105            this.motionZ = 0;
106            this.keepMovement = false;
107            hasUpdate = true;
108        }
109
110        Random random = new Random();
111
112        if (this.isInsideOfWater()) {
113            if (!this.attracted) {
114                if (this.waitChance > 0) {
115                    --this.waitChance;
116                }
117                if (this.waitChance == 0) {
118                    if (random.nextInt(100) < 90) {
119                        this.attractTimer = (random.nextInt(40) + 20);
120                        this.spawnFish();
121                        this.caught = false;
122                        this.attracted = true;
123                    } else {
124                        this.waitChance = WAIT_CHANCE;
125                    }
126                }
127            } else if (!this.caught) {
128                if (this.attractFish()) {
129                    this.coughtTimer = (random.nextInt(20) + 30);
130                    this.fishBites();
131                    this.caught = true;
132                }
133            } else {
134                if (this.coughtTimer > 0) {
135                    --this.coughtTimer;
136                }
137                if (this.coughtTimer == 0) {
138                    this.attracted = false;
139                    this.caught = false;
140                    this.waitChance = WAIT_CHANCE * 3;
141                }
142            }
143        }
144
145        return hasUpdate;
146    }

148    public int getWaterHeight() {
149        for (int y = this.getFloorY(); y < 256; y++) {
150            int id = this.level.getBlockIdAt(this.getFloorX(), y, this.getFloorZ());
151            if (id == Block.AIR) {
152                return y;
153            }
154        }
155        return this.getFloorY();
156    }

158    public void fishBites() {
159        EntityEventPacket pk = new EntityEventPacket();
160        pk.eid = this.getId();
161        pk.event = EntityEventPacket.FISH_HOOK_HOOK;
162        Server.broadcastPacket(this.level.getPlayers().values(), pk);
163
164        EntityEventPacket bubblePk = new EntityEventPacket();
165        bubblePk.eid = this.getId();
166        bubblePk.event = EntityEventPacket.FISH_HOOK_BUBBLE;
167        Server.broadcastPacket(this.level.getPlayers().values(), bubblePk);
168
169        EntityEventPacket teasePk = new EntityEventPacket();
170        teasePk.eid = this.getId();
171        teasePk.event = EntityEventPacket.FISH_HOOK_TEASE;
172        Server.broadcastPacket(this.level.getPlayers().values(), teasePk);
173
174        Random random = new Random();
175        for (int i = 0; i < 5; i++) {
176            this.level.addParticle(new BubbleParticle(this.setComponents(
177                    this.x + random.nextDouble() * 0.5 - 0.25,
178                    this.getWaterHeight(),
179                    this.z + random.nextDouble() * 0.5 - 0.25
180            )));
181        }
182    }

184    public void spawnFish() {
185        Random random = new Random();
186        this.fish = new Vector3(
187                this.x + (random.nextDouble() * 1.2 + 1) * (random.nextBoolean() ? -1 : 1),
188                this.getWaterHeight(),
189                this.z + (random.nextDouble() * 1.2 + 1) * (random.nextBoolean() ? -1 : 1)
190        );
191    }
192
193    public boolean attractFish() {
194        double multiply = 0.1;
195        this.fish.setComponents(
196                this.fish.x + (this.x - this.fish.x) * multiply,
197                this.fish.y,
198                this.fish.z + (this.z - this.fish.z) * multiply
199        );
200        if (new Random().nextInt(100) < 85) {
201            this.level.addParticle(new WaterParticle(this.fish));
202        }
203        double dist = Math.abs(Math.sqrt(this.x * this.x + this.z * this.z) - Math.sqrt(this.fish.x * this.fish.x + this.fish.z * this.fish.z));
204        if (dist < 0.15) {
205            return true;
206        }
207        return false;
208    }

210    public void reelLine() {
211        if (this.shootingEntity instanceof Player && this.caught) {
212            Item item = Fishing.getFishingResult(this.rod);
213            int experience = new Random().nextInt((3 - 1) + 1) + 1;
214            Vector3 motion;
215
216            if (this.shootingEntity != null) {
217                motion = this.shootingEntity.subtract(this).multiply(0.1);
218                motion.y += Math.sqrt(this.shootingEntity.distance(this)) * 0.08;
219            } else {
220                motion = new Vector3();
221            }
222
223            CompoundTag itemTag = NBTIO.putItemHelper(item);
224            itemTag.setName("Item");
225
226            EntityItem itemEntity = new EntityItem(
227                    this.level.getChunk((int) this.x >> 4, (int) this.z >> 4, true),
228                    new CompoundTag()
229                            .putList(new ListTag<DoubleTag>("Pos")
230                                    .add(new DoubleTag("", this.getX()))
231                                    .add(new DoubleTag("", this.getWaterHeight()))
232                                    .add(new DoubleTag("", this.getZ())))
233                            .putList(new ListTag<DoubleTag>("Motion")
234                                    .add(new DoubleTag("", motion.x))
235                                    .add(new DoubleTag("", motion.y))
236                                    .add(new DoubleTag("", motion.z)))
237                            .putList(new ListTag<FloatTag>("Rotation")
238                                    .add(new FloatTag("", new Random().nextFloat() * 360))
239                                    .add(new FloatTag("", 0)))
240                            .putShort("Health", 5).putCompound("Item", itemTag).putShort("PickupDelay", 1));
241
242            if (this.shootingEntity != null && this.shootingEntity instanceof Player) {
243                itemEntity.setOwner(this.shootingEntity.getName());
244            }
245            itemEntity.spawnToAll();
246
247            Player player = (Player) this.shootingEntity;
248            if (experience > 0) {
249                player.addExperience(experience);
250            }
251        }
252        if (this.shootingEntity instanceof Player) {
253            EntityEventPacket pk = new EntityEventPacket();
254            pk.eid = this.getId();
255            pk.event = EntityEventPacket.FISH_HOOK_TEASE;
256            Server.broadcastPacket(this.level.getPlayers().values(), pk);
257        }
258        if (!this.closed) {
259            this.kill();
260            this.close();
261        }
262    }

264    @Override
265    public void spawnTo(Player player) {
266        AddEntityPacket pk = new AddEntityPacket();
267        pk.entityRuntimeId = this.getId();
268        pk.entityUniqueId = this.getId();
269        pk.type = NETWORK_ID;
270        pk.x = (float) this.x;
271        pk.y = (float) this.y;
272        pk.z = (float) this.z;
273        pk.speedX = (float) this.motionX;
274        pk.speedY = (float) this.motionY;
275        pk.speedZ = (float) this.motionZ;
276        pk.yaw = (float) this.yaw;
277        pk.pitch = (float) this.pitch;
278
279        long ownerId = -1;
280        if (this.shootingEntity != null) {
281            ownerId = this.shootingEntity.getId();
282        }
283        pk.metadata = this.dataProperties.putLong(DATA_OWNER_EID, ownerId);
284        player.dataPacket(pk);
285        super.spawnTo(player);
286    }

288    @Override
289    public void onCollideWithEntity(Entity entity) {
290        this.server.getPluginManager().callEvent(new ProjectileHitEvent(this, MovingObjectPosition.fromEntity(entity)));
291        float damage = this.getResultDamage();
292
293        EntityDamageEvent ev;
294        if (this.shootingEntity == null) {
295            ev = new EntityDamageByEntityEvent(this, entity, DamageCause.PROJECTILE, damage);
296        } else {
297            ev = new EntityDamageByChildEntityEvent(this.shootingEntity, this, entity, DamageCause.PROJECTILE, damage);
298        }
299
300        entity.attack(ev);
301    }
