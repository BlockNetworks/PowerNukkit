001package cn.nukkit.level.generator.populator.impl;
002
003import cn.nukkit.block.Block;
004import cn.nukkit.level.ChunkManager;
005import cn.nukkit.level.format.FullChunk;
006import cn.nukkit.level.generator.populator.type.Populator;
007import cn.nukkit.math.NukkitRandom;
008
009public class PopulatorLava extends Populator {
010    private ChunkManager level;
011    private int randomAmount;
012    private int baseAmount;
013    private NukkitRandom random;
014
015    public void setRandomAmount(int amount) {
016        this.randomAmount = amount;
017    }
018
019    public void setBaseAmount(int amount) {
020        this.baseAmount = amount;
021    }
022
023    @Override
024    public void populate(ChunkManager level, int chunkX, int chunkZ, NukkitRandom random, FullChunk chunk) {
025        this.random = random;
026        if (random.nextRange(0, 100) < 5) {
027            this.level = level;
028            int amount = random.nextRange(0, this.randomAmount + 1) + this.baseAmount;
029            int bx = chunkX << 4;
030            int bz = chunkZ << 4;
031            int tx = bx + 15;
032            int tz = bz + 15;
033            for (int i = 0; i < amount; ++i) {
034                int x = random.nextRange(0, 15);
035                int z = random.nextRange(0, 15);
036                int y = this.getHighestWorkableBlock(chunk, x, z);
037                if (y != -1 && chunk.getBlockId(x, y, z) == Block.AIR) {
038                    chunk.setBlock(x, y, z, Block.LAVA);
039                    chunk.setBlockLight(x, y, z, Block.light[Block.LAVA]);
040                    this.lavaSpread(bx + x, y, bz + z);
041                }
042            }
043        }
044    }
045
046    private int getFlowDecay(int x1, int y1, int z1, int x2, int y2, int z2) {
047        if (this.level.getBlockIdAt(x1, y1, z1) != this.level.getBlockIdAt(x2, y2, z2)) {
048            return -1;
049        } else {
050            return this.level.getBlockDataAt(x2, y2, z2);
051        }
052    }
053
054    private void lavaSpread(int x, int y, int z) {
055        if (this.level.getChunk(x >> 4, z >> 4) == null) {
056            return;
057        }
058        int decay = this.getFlowDecay(x, y, z, x, y, z);
059        int multiplier = 2;
060        if (decay > 0) {
061            int smallestFlowDecay = -100;
062            smallestFlowDecay = this.getSmallestFlowDecay(x, y, z, x, y, z - 1, smallestFlowDecay);
063            smallestFlowDecay = this.getSmallestFlowDecay(x, y, z, x, y, z + 1, smallestFlowDecay);
064            smallestFlowDecay = this.getSmallestFlowDecay(x, y, z, x - 1, y, z, smallestFlowDecay);
065            smallestFlowDecay = this.getSmallestFlowDecay(x, y, z, x + 1, y, z, smallestFlowDecay);
066            int k = smallestFlowDecay + multiplier;
067            if (k >= 8 || smallestFlowDecay < 0) {
068                k = -1;
069            }
070            int topFlowDecay = this.getFlowDecay(x, y, z, x, y + 1, z);
071            if (topFlowDecay >= 0) {
072                if (topFlowDecay >= 8) {
073                    k = topFlowDecay;
074                } else {
075                    k = topFlowDecay | 0x08;
076                }
077            }
078            if (decay < 8 && k < 8 && k > 1 && random.nextRange(0, 4) != 0) {
079                k = decay;
080            }
081            if (k != decay) {
082                decay = k;
083                if (decay < 0) {
084                    this.level.setBlockAt(x, y, z, 0);
085                } else {
086                    this.level.setBlockAt(x, y, z, Block.LAVA, decay);
087                    this.lavaSpread(x, y, z);
088                    return;
089                }
090            }
091        }
092        if (this.canFlowInto(x, y - 1, z)) {
093            if (decay >= 8) {
094                this.flowIntoBlock(x, y - 1, z, decay);
095            } else {
096                this.flowIntoBlock(x, y - 1, z, decay | 0x08);
097            }
098        } else if (decay >= 0 && (decay == 0 || !this.canFlowInto(x, y - 1, z))) {
099            boolean[] flags = this.getOptimalFlowDirections(x, y, z);
100            int l = decay + multiplier;
101            if (decay >= 8) {
102                l = 1;
103            }
104            if (l >= 8) {
105                return;
106            }
107            if (flags[0]) {
108                this.flowIntoBlock(x - 1, y, z, l);
109            }
110            if (flags[1]) {
111                this.flowIntoBlock(x + 1, y, z, l);
112            }
113            if (flags[2]) {
114                this.flowIntoBlock(x, y, z - 1, l);
115            }
116            if (flags[3]) {
117                this.flowIntoBlock(x, y, z + 1, l);
118            }
119        }
120    }
121
122    private void flowIntoBlock(int x, int y, int z, int newFlowDecay) {
123        if (this.level.getBlockIdAt(x, y, z) == Block.AIR) {
124            this.level.setBlockAt(x, y, z, Block.LAVA, newFlowDecay);
125            this.lavaSpread(x, y, z);
126        }
127    }
128
129    private boolean canFlowInto(int x, int y, int z) {
130        int id = this.level.getBlockIdAt(x, y, z);
131        return id == Block.AIR || id == Block.LAVA || id == Block.STILL_LAVA;
132    }
133
134    private int calculateFlowCost(int xx, int yy, int zz, int accumulatedCost, int previousDirection) {
135        int cost = 1000;
136        for (int j = 0; j < 4; ++j) {
137            if (
138                    (j == 0 && previousDirection == 1) ||
139                            (j == 1 && previousDirection == 0) ||
140                            (j == 2 && previousDirection == 3) ||
141                            (j == 3 && previousDirection == 2)
142                    ) {
143                int x = xx;
144                int y = yy;
145                int z = zz;
146                if (j == 0) {
147                    --x;
148                } else if (j == 1) {
149                    ++x;
150                } else if (j == 2) {
151                    --z;
152                } else if (j == 3) {
153                    ++z;
154                }
155                if (!this.canFlowInto(x, y, z)) {
156                    continue;
157                } else if (this.canFlowInto(x, y, z) && this.level.getBlockDataAt(x, y, z) == 0) {
158                    continue;
159                } else if (this.canFlowInto(x, y - 1, z)) {
160                    return accumulatedCost;
161                }
162                if (accumulatedCost >= 4) {
163                    continue;
164                }
165                int realCost = this.calculateFlowCost(x, y, z, accumulatedCost + 1, j);
166                if (realCost < cost) {
167                    cost = realCost;
168                }
169            }
170        }
171        return cost;
172    }
173
174    private boolean[] getOptimalFlowDirections(int xx, int yy, int zz) {
175        int[] flowCost = {0, 0, 0, 0};
176        boolean[] isOptimalFlowDirection = {false, false, false, false};
177        for (int j = 0; j < 4; ++j) {
178            flowCost[j] = 1000;
179            int x = xx;
180            int y = yy;
181            int z = zz;
182            if (j == 0) {
183                --x;
184            } else if (j == 1) {
185                ++x;
186            } else if (j == 2) {
187                --z;
188            } else if (j == 3) {
189                ++z;
190            }
191            if (this.canFlowInto(x, y - 1, z)) {
192                flowCost[j] = 0;
193            } else {
194                flowCost[j] = this.calculateFlowCost(x, y, z, 1, j);
195            }
196        }
197        int minCost = flowCost[0];
198        for (int i = 1; i < 4; ++i) {
199            if (flowCost[i] < minCost) {
200                minCost = flowCost[i];
201            }
202        }
203        for (int i = 0; i < 4; ++i) {
204            isOptimalFlowDirection[i] = (flowCost[i] == minCost);
205        }
206        return isOptimalFlowDirection;
207    }
208
209    private int getSmallestFlowDecay(int x1, int y1, int z1, int x2, int y2, int z2, int decay) {
210        int blockDecay = this.getFlowDecay(x1, y1, z1, x2, y2, z2);
211        if (blockDecay < 0) {
212            return decay;
213        } else if (blockDecay >= 8) {
214            blockDecay = 0;
215        }
216        return (decay >= 0 && blockDecay >= decay) ? decay : blockDecay;
217    }
218
219
220    private int getHighestWorkableBlock(FullChunk chunk, int x, int z) {
221        int y;
222        for (y = 127; y >= 0; y--) {
223            int b = chunk.getBlockId(x, y, z);
224            if (b == Block.AIR) {
225                break;
226            }
227        }
228        return y == 0 ? -1 : y;
229    }
