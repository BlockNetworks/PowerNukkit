001package cn.nukkit.level.format.anvil.util;
002
003import cn.nukkit.level.GlobalBlockPalette;
004import cn.nukkit.level.util.PalettedBlockStorage;
005import cn.nukkit.utils.BinaryStream;
006import com.google.common.base.Preconditions;
007
008import java.util.Arrays;
009
010public class BlockStorage {
011    private static final int SECTION_SIZE = 4096;
012    private final byte[] blockIds;
013    private final NibbleArray blockData;
014
015    public BlockStorage() {
016        blockIds = new byte[SECTION_SIZE];
017        blockData = new NibbleArray(SECTION_SIZE);
018    }
019
020    private BlockStorage(byte[] blockIds, NibbleArray blockData) {
021        this.blockIds = blockIds;
022        this.blockData = blockData;
023    }
024
025    private static int getIndex(int x, int y, int z) {
026        int index = (x << 8) + (z << 4) + y; // XZY = Bedrock format
027        Preconditions.checkArgument(index >= 0 && index < SECTION_SIZE, "Invalid index");
028        return index;
029    }
030
031    public int getBlockData(int x, int y, int z) {
032        return blockData.get(getIndex(x, y, z)) & 0xf;
033    }
034
035    public int getBlockId(int x, int y, int z) {
036        return blockIds[getIndex(x, y, z)] & 0xFF;
037    }
038
039    public void setBlockId(int x, int y, int z, int id) {
040        blockIds[getIndex(x, y, z)] = (byte) (id & 0xff);
041    }
042
043    public void setBlockData(int x, int y, int z, int data) {
044        blockData.set(getIndex(x, y, z), (byte) data);
045    }
046
047    public int getFullBlock(int x, int y, int z) {
048        return getFullBlock(getIndex(x, y, z));
049    }
050
051    public void setFullBlock(int x, int y, int z, int value) {
052        this.setFullBlock(getIndex(x, y, z), (short) value);
053    }
054
055    public int getAndSetFullBlock(int x, int y, int z, int value) {
056        return getAndSetFullBlock(getIndex(x, y, z), (short) value);
057    }
058
059    private int getAndSetFullBlock(int index, short value) {
060        Preconditions.checkArgument(value < 0xfff, "Invalid full block");
061        byte oldBlock = blockIds[index];
062        byte oldData = blockData.get(index);
063        byte newBlock = (byte) ((value & 0xff0) >> 4);
064        byte newData = (byte) (value & 0xf);
065        if (oldBlock != newBlock) {
066            blockIds[index] = newBlock;
067        }
068        if (oldData != newData) {
069            blockData.set(index, newData);
070        }
071        return ((oldBlock & 0xff) << 4) | oldData;
072    }
073
074    private int getFullBlock(int index) {
075        byte block = blockIds[index];
076        byte data = blockData.get(index);
077        return ((block & 0xff) << 4) | data;
078    }
079
080    private void setFullBlock(int index, short value) {
081        Preconditions.checkArgument(value < 0xfff, "Invalid full block");
082        byte block = (byte) ((value & 0xff0) >> 4);
083        byte data = (byte) (value & 0xf);
084
085        blockIds[index] = block;
086        blockData.set(index, data);
087    }
088
089    public byte[] getBlockIds() {
090        return Arrays.copyOf(blockIds, blockIds.length);
091    }
092
093    public byte[] getBlockData() {
094        return blockData.getData();
095    }
096
097    public void writeTo(BinaryStream stream) {
098        PalettedBlockStorage storage = new PalettedBlockStorage();
099        for (int i = 0; i < SECTION_SIZE; i++) {
100            storage.setBlock(i, GlobalBlockPalette.getOrCreateRuntimeId(blockIds[i] & 0xff, blockData.get(i)));
101        }
102        storage.writeTo(stream);
103    }
104
105    public BlockStorage copy() {
106        return new BlockStorage(blockIds.clone(), blockData.copy());
107    }
108}
