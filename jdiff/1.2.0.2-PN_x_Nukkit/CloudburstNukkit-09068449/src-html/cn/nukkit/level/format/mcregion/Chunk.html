048    public Chunk(LevelProvider level, CompoundTag nbt) {
049        this.provider = level;
050        if (level != null) {
051            this.providerClass = level.getClass();
052        }
053
054        if (nbt == null) {
055            return;
056        }
057
058        this.nbt = nbt;
059
060        if (!(this.nbt.contains("Entities") && (this.nbt.get("Entities") instanceof ListTag))) {
061            this.nbt.putList(new ListTag<CompoundTag>("Entities"));
062        }
063
064        if (!(this.nbt.contains("TileEntities") && (this.nbt.get("TileEntities") instanceof ListTag))) {
065            this.nbt.putList(new ListTag<CompoundTag>("TileEntities"));
066        }
067
068        if (!(this.nbt.contains("TileTicks") && (this.nbt.get("TileTicks") instanceof ListTag))) {
069            this.nbt.putList(new ListTag<CompoundTag>("TileTicks"));
070        }
071
072        if (!(this.nbt.contains("Biomes") && (this.nbt.get("Biomes") instanceof ByteArrayTag))) {
073            this.nbt.putByteArray("Biomes", new byte[256]);
074        }
075
076        if (!(this.nbt.contains("HeightMap") && (this.nbt.get("HeightMap") instanceof IntArrayTag))) {
077            this.nbt.putIntArray("HeightMap", new int[256]);
078        }
079
080        if (!(this.nbt.contains("Blocks"))) {
081            this.nbt.putByteArray("Blocks", new byte[32768]);
082        }
083
084        if (!(this.nbt.contains("Data"))) {
085            this.nbt.putByteArray("Data", new byte[16384]);
086            this.nbt.putByteArray("SkyLight", new byte[16384]);
087            this.nbt.putByteArray("BlockLight", new byte[16384]);
088        }
089
090        Map<Integer, Integer> extraData = new HashMap<>();
091
092        if (!this.nbt.contains("ExtraData") || !(this.nbt.get("ExtraData") instanceof ByteArrayTag)) {
093            this.nbt.putByteArray("ExtraData", Binary.writeInt(0));
094        } else {
095            BinaryStream stream = new BinaryStream(this.nbt.getByteArray("ExtraData"));
096            for (int i = 0; i < stream.getInt(); i++) {
097                int key = stream.getInt();
098                extraData.put(key, stream.getShort());
099            }
100        }
101
102        this.setPosition(this.nbt.getInt("xPos"), this.nbt.getInt("zPos"));
103        this.blocks = this.nbt.getByteArray("Blocks");
104        this.data = this.nbt.getByteArray("Data");
105        this.skyLight = this.nbt.getByteArray("SkyLight");
106        this.blockLight = this.nbt.getByteArray("BlockLight");
107
108        if (this.nbt.contains("BiomeColors")) {
109            this.biomes = new byte[16 * 16];
110            int[] biomeColors = this.nbt.getIntArray("BiomeColors");
111            if (biomeColors.length == 256) {
112                BiomePalette palette = new BiomePalette(biomeColors);
113                for (int x = 0; x < 16; x++)    {
114                    for (int z = 0; z < 16; z++)    {
115                        this.biomes[(x << 4) | z] = (byte) (palette.get(x, z) >> 24);
116                    }
117                }
118            }
119        } else {
120            this.biomes = this.nbt.getByteArray("Biomes");
121        }
122
123        int[] heightMap = this.nbt.getIntArray("HeightMap");
124        this.heightMap = new byte[256];
125        if (heightMap.length != 256) {
126            Arrays.fill(this.heightMap, (byte) 255);
127        } else {
128            for (int i = 0; i < heightMap.length; i++) {
129                this.heightMap[i] = (byte) heightMap[i];
130            }
131        }
132
133        if (!extraData.isEmpty()) this.extraData = extraData;
134
135        this.NBTentities = ((ListTag<CompoundTag>) this.nbt.getList("Entities")).getAll();
136        this.NBTtiles = ((ListTag<CompoundTag>) this.nbt.getList("TileEntities")).getAll();
137
138        this.nbt.remove("Blocks");
139        this.nbt.remove("Data");
140        this.nbt.remove("SkyLight");
141        this.nbt.remove("BlockLight");
142        this.nbt.remove("BiomeColors");
143        this.nbt.remove("HeightMap");
144        this.nbt.remove("Biomes");
145    }

160        int b = this.data[(x << 10) | (z << 6) | (y >> 1)] & 0xff;
161        if ((y & 1) == 0) {
162            return b & 0x0f;
163        } else {
164            return b >> 4;
165        }

168    @Override
169    public void setBlockData(int x, int y, int z, int data) {
170        int i = (x << 10) | (z << 6) | (y >> 1);
171        int old = this.data[i] & 0xff;
172        if ((y & 1) == 0) {
173            this.data[i] = (byte) ((old & 0xf0) | (data & 0x0f));
174        } else {
175            this.data[i] = (byte) (((data & 0x0f) << 4) | (old & 0x0f));
176        }
177        setChanged();
178    }

180    @Override
181    public int getFullBlock(int x, int y, int z) {
182        int i = (x << 11) | (z << 7) | y;
183        int block = this.blocks[i] & 0xff;
184        int data = this.data[i >> 1] & 0xff;
185        if ((y & 1) == 0) {
186            return (block << 4) | (data & 0x0f);
187        } else {
188            return (block << 4) | (data >> 4);
189        }
190    }

197    @Override
198    public boolean setBlock(int x, int y, int z, int blockId, int meta) {
199        int i = (x << 11) | (z << 7) | y;
200        boolean changed = false;
201        byte id = (byte) blockId;
202        if (this.blocks[i] != id) {
203            this.blocks[i] = id;
204            changed = true;
205        }
206
207        if (Block.hasMeta[blockId]) {
208            i >>= 1;
209            int old = this.data[i] & 0xff;
210            if ((y & 1) == 0) {
211                this.data[i] = (byte) ((old & 0xf0) | (meta & 0x0f));
212                if ((old & 0x0f) != meta) {
213                    changed = true;
214                }
215            } else {
216                this.data[i] = (byte) (((meta & 0x0f) << 4) | (old & 0x0f));
217                if (meta != (old >> 4)) {
218                    changed = true;
219                }
220            }
221        }
222
223        if (changed) {
224            setChanged();
225        }
226        return changed;
227    }

229    @Override
230    public Block getAndSetBlock(int x, int y, int z, Block block) {
231        int i = (x << 11) | (z << 7) | y;
232        boolean changed = false;
233        byte id = (byte) block.getId();
234
235        byte previousId = this.blocks[i];
236
237        if (previousId != id) {
238            this.blocks[i] = id;
239            changed = true;
240        }
241
242        int previousData;
243        i >>= 1;
244        int old = this.data[i] & 0xff;
245        if ((y & 1) == 0) {
246            previousData = old & 0x0f;
247            if (Block.hasMeta[block.getId()]) {
248                this.data[i] = (byte) ((old & 0xf0) | (block.getDamage() & 0x0f));
249                if (block.getDamage() != previousData) {
250                    changed = true;
251                }
252            }
253        } else {
254            previousData = old >> 4;
255            if (Block.hasMeta[block.getId()]) {
256                this.data[i] = (byte) (((block.getDamage() & 0x0f) << 4) | (old & 0x0f));
257                if (block.getDamage() != previousData) {
258                    changed = true;
259                }
260            }
261        }
262
263        if (changed) {
264            setChanged();
265        }
266        return Block.get(previousId, previousData);
267    }

370    public static Chunk fromBinary(byte[] data, LevelProvider provider) {
371        try {
372            CompoundTag chunk = NBTIO.read(new ByteArrayInputStream(Zlib.inflate(data)), ByteOrder.BIG_ENDIAN);
373
374            if (!chunk.contains("Level") || !(chunk.get("Level") instanceof CompoundTag)) {
375                return null;
376            }
377            return new Chunk(provider != null ? provider : McRegion.class.newInstance(), chunk.getCompound("Level"));
378        } catch (Exception e) {
379            return null;
380        }
381    }

387    public static Chunk fromFastBinary(byte[] data, LevelProvider provider) {
388        try {
389            int offset = 0;
390            Chunk chunk = new Chunk(provider != null ? provider : McRegion.class.newInstance(), null);
391            chunk.provider = provider;
392            int chunkX = (Binary.readInt(Arrays.copyOfRange(data, offset, offset + 3)));
393            offset += 4;
394            int chunkZ = (Binary.readInt(Arrays.copyOfRange(data, offset, offset + 3)));
395            chunk.setPosition(chunkX, chunkZ);
396            offset += 4;
397            chunk.blocks = Arrays.copyOfRange(data, offset, offset + 32767);
398            offset += 32768;
399            chunk.data = Arrays.copyOfRange(data, offset, offset + 16383);
400            offset += 16384;
401            chunk.skyLight = Arrays.copyOfRange(data, offset, offset + 16383);
402            offset += 16384;
403            chunk.blockLight = Arrays.copyOfRange(data, offset, offset + 16383);
404            offset += 16384;
405            chunk.heightMap = Arrays.copyOfRange(data, offset, offset + 256);
406            offset += 256;
407            chunk.biomes = Arrays.copyOfRange(data, offset, offset + 256);
408            offset += 256;
409            byte flags = data[offset++];
410            chunk.nbt.putByte("TerrainGenerated", (flags & 0b1));
411            chunk.nbt.putByte("TerrainPopulated", ((flags >> 1) & 0b1));
412            chunk.nbt.putByte("LightPopulated", ((flags >> 2) & 0b1));
413            return chunk;
414        } catch (Exception e) {
415            throw new RuntimeException(e);
416        }
417    }

420    @Override
421    public byte[] toFastBinary() {
422        BinaryStream stream = new BinaryStream(new byte[65536]);
423        stream.put(Binary.writeInt(this.getX()));
424        stream.put(Binary.writeInt(this.getZ()));
425        stream.put(this.getBlockIdArray());
426        stream.put(this.getBlockDataArray());
427        stream.put(this.getBlockSkyLightArray());
428        stream.put(this.getBlockLightArray());
429        stream.put(this.getHeightMapArray());
430        stream.put(this.getBiomeIdArray());
431        stream.putByte((byte) ((this.isLightPopulated() ? 1 << 2 : 0) + (this.isPopulated() ? 1 << 2 : 0) + (this.isGenerated() ? 1 : 0)));
432        return stream.getBuffer();
433    }

435    @Override
436    public byte[] toBinary() {
437        CompoundTag nbt = this.getNBT().copy();
438        nbt.remove("BiomeColors");
439
440        nbt.putInt("xPos", this.getX());
441        nbt.putInt("zPos", this.getZ());
442
443        if (this.isGenerated()) {
444            nbt.putByteArray("Blocks", this.getBlockIdArray());
445            nbt.putByteArray("Data", this.getBlockDataArray());
446            nbt.putByteArray("SkyLight", this.getBlockSkyLightArray());
447            nbt.putByteArray("BlockLight", this.getBlockLightArray());
448            nbt.putByteArray("Biomes", this.getBiomeIdArray());
449
450            int[] heightInts = new int[256];
451            byte[] heightBytes = this.getHeightMapArray();
452            for (int i = 0; i < heightInts.length; i++) {
453                heightInts[i] = heightBytes[i] & 0xFF;
454            }
455            nbt.putIntArray("HeightMap", heightInts);
456        }
457
458
459        ArrayList<CompoundTag> entities = new ArrayList<>();
460        for (Entity entity : this.getEntities().values()) {
461            if (!(entity instanceof Player) && !entity.closed) {
462                entity.saveNBT();
463                entities.add(entity.namedTag);
464            }
465        }
466        ListTag<CompoundTag> entityListTag = new ListTag<>("Entities");
467        entityListTag.setAll(entities);
468        nbt.putList(entityListTag);
469
470        ArrayList<CompoundTag> tiles = new ArrayList<>();
471        for (BlockEntity blockEntity : this.getBlockEntities().values()) {
472            blockEntity.saveNBT();
473            tiles.add(blockEntity.namedTag);
474        }
475        ListTag<CompoundTag> tileListTag = new ListTag<>("TileEntities");
476        tileListTag.setAll(tiles);
477        nbt.putList(tileListTag);
478
479        BinaryStream extraData = new BinaryStream();
480        Map<Integer, Integer> extraDataArray = this.getBlockExtraDataArray();
481        extraData.putInt(extraDataArray.size());
482        for (Integer key : extraDataArray.keySet()) {
483            extraData.putInt(key);
484            extraData.putShort(extraDataArray.get(key));
485        }
486
487        nbt.putByteArray("ExtraData", extraData.getBuffer());
488
489        CompoundTag chunk = new CompoundTag("");
490        chunk.putCompound("Level", nbt);
491
492        try {
493            return Zlib.deflate(NBTIO.write(chunk, ByteOrder.BIG_ENDIAN), RegionLoader.COMPRESSION_LEVEL);
494        } catch (Exception e) {
495            throw new RuntimeException(e);
496        }
497    }

507    public static Chunk getEmptyChunk(int chunkX, int chunkZ, LevelProvider provider) {
508        try {
509            Chunk chunk;
510            if (provider != null) {
511                chunk = new Chunk(provider, null);
512            } else {
513                chunk = new Chunk(McRegion.class, null);
514            }
515
516            chunk.setPosition(chunkX, chunkZ);
517            chunk.data = new byte[16384];
518            chunk.blocks = new byte[32768];
519            byte[] skyLight = new byte[16384];
520            Arrays.fill(skyLight, (byte) 0xff);
521            chunk.skyLight = skyLight;
522            chunk.blockLight = chunk.data;
523
524            chunk.heightMap = new byte[256];
525            chunk.biomes = new byte[16 * 16];
526
527            chunk.nbt.putByte("V", 1);
528            chunk.nbt.putLong("InhabitedTime", 0);
529            chunk.nbt.putBoolean("TerrainGenerated", false);
530            chunk.nbt.putBoolean("TerrainPopulated", false);
531            chunk.nbt.putBoolean("LightPopulated", false);
532
533            return chunk;
534        } catch (Exception e) {
535            throw new RuntimeException(e);
536        }
