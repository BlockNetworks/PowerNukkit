001package cn.nukkit.block;
002
003import cn.nukkit.Player;
004import cn.nukkit.item.Item;
005import cn.nukkit.item.ItemTool;
006import cn.nukkit.math.BlockFace;
007import cn.nukkit.utils.Faceable;
008
009/**
010 * Created by CreeperFace on 2.6.2017.
011 */
012public abstract class BlockTerracottaGlazed extends BlockSolidMeta implements Faceable {
013
014    public BlockTerracottaGlazed() {
015        this(0);
016    }
017
018    public BlockTerracottaGlazed(int meta) {
019        super(meta);
020    }
021
022    @Override
023    public double getResistance() {
024        return 7;
025    }
026
027    @Override
028    public double getHardness() {
029        return 1.4;
030    }
031
032    @Override
033    public int getToolType() {
034        return ItemTool.TYPE_PICKAXE;
035    }
036
037    @Override
038    public Item[] getDrops(Item item) {
039        return item.getTier() >= ItemTool.TIER_WOODEN ? new Item[]{this.toItem()} : new Item[0];
040    }
041
042    @Override
043    public boolean place(Item item, Block block, Block target, BlockFace face, double fx, double fy, double fz, Player player) {
044        int[] faces = {2, 5, 3, 4};
045        this.setDamage(faces[player != null ? player.getDirection().getHorizontalIndex() : 0]);
046        return this.getLevel().setBlock(block, this, true, true);
047    }
048
049    @Override
050    public boolean canHarvestWithHand() {
051        return false;
052    }
053
054    @Override
055    public BlockFace getBlockFace() {
056        return BlockFace.fromHorizontalIndex(this.getDamage() & 0x07);
057    }
058}
