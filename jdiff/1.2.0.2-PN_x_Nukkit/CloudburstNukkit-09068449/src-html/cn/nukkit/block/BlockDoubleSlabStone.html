001package cn.nukkit.block;
002
003import cn.nukkit.item.Item;
004import cn.nukkit.item.ItemBlock;
005import cn.nukkit.item.ItemTool;
006import cn.nukkit.utils.BlockColor;
007
008/**
009 * author: MagicDroidX
010 * Nukkit Project
011 */
012public class BlockDoubleSlabStone extends BlockSolidMeta {
013    public static final int STONE = 0;
014    public static final int SANDSTONE = 1;
015    public static final int WOODEN = 2;
016    public static final int COBBLESTONE = 3;
017    public static final int BRICK = 4;
018    public static final int STONE_BRICK = 5;
019    public static final int QUARTZ = 6;
020    public static final int NETHER_BRICK = 7;
021
022    public BlockDoubleSlabStone() {
023        this(0);
024    }
025
026    public BlockDoubleSlabStone(int meta) {
027        super(meta);
028    }
029
030    @Override
031    public int getId() {
032        return DOUBLE_SLAB;
033    }
034
035    @Override
036    public double getResistance() {
037        return getToolType() > ItemTool.TIER_WOODEN ? 30 : 15;
038    }
039
040    @Override
041    public double getHardness() {
042        return 2;
043    }
044
045    @Override
046    public int getToolType() {
047        return ItemTool.TYPE_PICKAXE;
048    }
049
050    @Override
051    public String getName() {
052        String[] names = new String[]{
053                "Stone",
054                "Sandstone",
055                "Wooden",
056                "Cobblestone",
057                "Brick",
058                "Stone Brick",
059                "Quartz",
060                "Nether Brick"
061        };
062        return "Double " + names[this.getDamage() & 0x07] + " Slab";
063    }
064
065    @Override
066    public Item toItem() {
067        return new ItemBlock(Block.get(BlockID.STONE_SLAB), this.getDamage() & 0x07);
068    }
069
070    @Override
071    public Item[] getDrops(Item item) {
072        if (item.isPickaxe() && item.getTier() >= ItemTool.TIER_WOODEN) {
073            return new Item[]{
074                    Item.get(Item.SLAB, this.getDamage() & 0x07, 2)
075            };
076        } else {
077            return new Item[0];
078        }
079    }
080
081    @Override
082    public BlockColor getColor() {
083        switch (this.getDamage() & 0x07) {
084            default:
085                case BlockDoubleSlabStone.STONE:
086                case BlockDoubleSlabStone.COBBLESTONE:
087                case BlockDoubleSlabStone.BRICK:
088                case BlockDoubleSlabStone.STONE_BRICK:
089                    return BlockColor.STONE_BLOCK_COLOR;
090            case BlockDoubleSlabStone.SANDSTONE:
091                return BlockColor.SAND_BLOCK_COLOR;
092            case BlockDoubleSlabStone.WOODEN:
093                return BlockColor.WOOD_BLOCK_COLOR;
094            case BlockDoubleSlabStone.QUARTZ:
095                return BlockColor.QUARTZ_BLOCK_COLOR;
096            case BlockDoubleSlabStone.NETHER_BRICK:
097                return BlockColor.NETHERRACK_BLOCK_COLOR;
098        }
099    }
100
101    @Override
102    public boolean canHarvestWithHand() {
103        return false;
104    }
105}
