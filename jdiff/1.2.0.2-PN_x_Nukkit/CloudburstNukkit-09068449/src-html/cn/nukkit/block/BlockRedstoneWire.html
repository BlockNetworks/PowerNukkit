045    @Override
046    public boolean place(Item item, Block block, Block target, BlockFace face, double fx, double fy, double fz, Player player) {
047        if (face != BlockFace.UP || !canBePlacedOn(target)) {
048            return false;
049        }
050
051        this.getLevel().setBlock(block, this, true, false);
052        this.updateSurroundingRedstone(true);
053        Vector3 pos = getLocation();
054
055        for (BlockFace blockFace : Plane.VERTICAL) {
056            this.level.updateAroundRedstone(pos.getSide(blockFace), blockFace.getOpposite());
057        }
058
059        for (BlockFace blockFace : Plane.VERTICAL) {
060            this.updateAround(pos.getSide(blockFace), blockFace.getOpposite());
061        }
062
063        for (BlockFace blockFace : Plane.HORIZONTAL) {
064            Vector3 v = pos.getSide(blockFace);
065
066            if (this.level.getBlock(v).isNormalBlock()) {
067                this.updateAround(v.up(), BlockFace.DOWN);
068            } else {
069                this.updateAround(v.down(), BlockFace.UP);
070            }
071        }
072        return true;
073    }

075    private void updateAround(Vector3 pos, BlockFace face) {
076        if (this.level.getBlock(pos).getId() == Block.REDSTONE_WIRE) {
077            this.level.updateAroundRedstone(pos, face);
078
079            for (BlockFace side : BlockFace.values()) {
080                this.level.updateAroundRedstone(pos.getSide(side), side.getOpposite());
081            }
082        }
083    }

085    private void updateSurroundingRedstone(boolean force) {
086        this.calculateCurrentChanges(force);
087    }
088
089    private void calculateCurrentChanges(boolean force) {
090        Vector3 pos = this.getLocation();
091
092        int meta = this.getDamage();
093        int maxStrength = meta;
094        this.canProvidePower = false;
095        int power = this.getIndirectPower();
096
097        this.canProvidePower = true;
098
099        if (power > 0 && power > maxStrength - 1) {
100            maxStrength = power;
101        }
102
103        int strength = 0;
104
105        for (BlockFace face : Plane.HORIZONTAL) {
106            Vector3 v = pos.getSide(face);
107
108            if (v.getX() == this.getX() && v.getZ() == this.getZ()) {
109                continue;
110            }
111
112
113            strength = this.getMaxCurrentStrength(v, strength);
114
115            boolean vNormal = this.level.getBlock(v).isNormalBlock();
116
117            if (vNormal && !this.level.getBlock(pos.up()).isNormalBlock()) {
118                strength = this.getMaxCurrentStrength(v.up(), strength);
119            } else if (!vNormal) {
120                strength = this.getMaxCurrentStrength(v.down(), strength);
121            }
122        }
123
124        if (strength > maxStrength) {
125            maxStrength = strength - 1;
126        } else if (maxStrength > 0) {
127            --maxStrength;
128        } else {
129            maxStrength = 0;
130        }
131
132        if (power > maxStrength - 1) {
133            maxStrength = power;
134        } else if (power < maxStrength && strength <= maxStrength) {
135            maxStrength = Math.max(power, strength - 1);
136        }
137
138        if (meta != maxStrength) {
139            this.level.getServer().getPluginManager().callEvent(new BlockRedstoneEvent(this, meta, maxStrength));
140
141            this.setDamage(maxStrength);
142            this.level.setBlock(this, this, false, false);
143
144            this.level.updateAroundRedstone(this, null);
145            for (BlockFace face : BlockFace.values()) {
146                this.level.updateAroundRedstone(pos.getSide(face), face.getOpposite());
147            }
148        } else if (force) {
149            for (BlockFace face : BlockFace.values()) {
150                this.level.updateAroundRedstone(pos.getSide(face), face.getOpposite());
151            }
152        }
153    }
154
155    private int getMaxCurrentStrength(Vector3 pos, int maxStrength) {
156        if (this.level.getBlockIdAt(pos.getFloorX(), pos.getFloorY(), pos.getFloorZ()) != this.getId()) {
157            return maxStrength;
158        } else {
159            int strength = this.level.getBlockDataAt(pos.getFloorX(), pos.getFloorY(), pos.getFloorZ());
160            return strength > maxStrength ? strength : maxStrength;
161        }
162    }

164    @Override
165    public boolean onBreak(Item item) {
166        this.getLevel().setBlock(this, Block.get(BlockID.AIR), true, true);
167
168        Vector3 pos = getLocation();
169
170        this.updateSurroundingRedstone(false);
171
172        for (BlockFace blockFace : BlockFace.values()) {
173            this.level.updateAroundRedstone(pos.getSide(blockFace), null);
174        }
175
176        for (BlockFace blockFace : Plane.HORIZONTAL) {
177            Vector3 v = pos.getSide(blockFace);
178
179            if (this.level.getBlock(v).isNormalBlock()) {
180                this.updateAround(v.up(), BlockFace.DOWN);
181            } else {
182                this.updateAround(v.down(), BlockFace.UP);
183            }
184        }
185        return true;
186    }

198    @Override
199    public int onUpdate(int type) {
200        if (type != Level.BLOCK_UPDATE_NORMAL && type != Level.BLOCK_UPDATE_REDSTONE) {
201            return 0;
202        }
203        // Redstone event
204        RedstoneUpdateEvent ev = new RedstoneUpdateEvent(this);
205        getLevel().getServer().getPluginManager().callEvent(ev);
206        if (ev.isCancelled()) {
207            return 0;
208        }
209
210        if (type == Level.BLOCK_UPDATE_NORMAL && !this.canBePlacedOn(this.getLocation().down())) {
211            this.getLevel().useBreakOn(this);
212            return Level.BLOCK_UPDATE_NORMAL;
213        }
214
215        this.updateSurroundingRedstone(false);
216
217        return Level.BLOCK_UPDATE_NORMAL;
218    }

220    public boolean canBePlacedOn(Vector3 v) {
221        Block b = this.level.getBlock(v);
222
223        return b.isSolid() && !b.isTransparent() && b.getId() != Block.GLOWSTONE;
224    }

226    public int getStrongPower(BlockFace side) {
227        return !this.canProvidePower ? 0 : getWeakPower(side);
228    }
229
230    public int getWeakPower(BlockFace side) {
231        if (!this.canProvidePower) {
232            return 0;
233        } else {
234            int power = this.getDamage();
235
236            if (power == 0) {
237                return 0;
238            } else if (side == BlockFace.UP) {
239                return power;
240            } else {
241                EnumSet<BlockFace> enumset = EnumSet.noneOf(BlockFace.class);
242
243                for (BlockFace face : Plane.HORIZONTAL) {
244                    if (this.isPowerSourceAt(face)) {
245                        enumset.add(face);
246                    }
247                }
248
249                if (side.getAxis().isHorizontal() && enumset.isEmpty()) {
250                    return power;
251                } else if (enumset.contains(side) && !enumset.contains(side.rotateYCCW()) && !enumset.contains(side.rotateY())) {
252                    return power;
253                } else {
254                    return 0;
255                }
256            }
257        }
258    }

260    private boolean isPowerSourceAt(BlockFace side) {
261        Vector3 pos = getLocation();
262        Vector3 v = pos.getSide(side);
263        Block block = this.level.getBlock(v);
264        boolean flag = block.isNormalBlock();
265        boolean flag1 = this.level.getBlock(pos.up()).isNormalBlock();
266        return !flag1 && flag && canConnectUpwardsTo(this.level, v.up()) || (canConnectTo(block, side) || !flag && canConnectUpwardsTo(this.level, block.down()));
267    }
268
269    protected static boolean canConnectUpwardsTo(Level level, Vector3 pos) {
270        return canConnectUpwardsTo(level.getBlock(pos));
271    }
272
273    protected static boolean canConnectUpwardsTo(Block block) {
274        return canConnectTo(block, null);
275    }
276
277    protected static boolean canConnectTo(Block block, BlockFace side) {
278        if (block.getId() == Block.REDSTONE_WIRE) {
279            return true;
280        } else if (BlockRedstoneDiode.isDiode(block)) {
281            BlockFace face = ((BlockRedstoneDiode) block).getFacing();
282            return face == side || face.getOpposite() == side;
283        } else {
284            return block.isPowerSource() && side != null;
285        }
286    }

293    private int getIndirectPower() {
294        int power = 0;
295        Vector3 pos = getLocation();
296
297        for (BlockFace face : BlockFace.values()) {
298            int blockPower = this.getIndirectPower(pos.getSide(face), face);
299
300            if (blockPower >= 15) {
301                return 15;
302            }
303
304            if (blockPower > power) {
305                power = blockPower;
306            }
307        }
308
309        return power;
310    }
311
312    private int getIndirectPower(Vector3 pos, BlockFace face) {
313        Block block = this.level.getBlock(pos);
314        if (block.getId() == Block.REDSTONE_WIRE) {
315            return 0;
316        }
317        return block.isNormalBlock() ? getStrongPower(pos.getSide(face), face) : block.getWeakPower(face);
318    }
319
320    private int getStrongPower(Vector3 pos, BlockFace direction) {
321        Block block = this.level.getBlock(pos);
322
323        if (block.getId() == Block.REDSTONE_WIRE) {
324            return 0;
325        }
326
327        return block.getStrongPower(direction);
328    }
