001package cn.nukkit.block;
002
003import cn.nukkit.Player;
004import cn.nukkit.blockentity.BlockEntity;
005import cn.nukkit.blockentity.BlockEntitySign;
006import cn.nukkit.item.Item;
007import cn.nukkit.item.ItemSign;
008import cn.nukkit.item.ItemTool;
009import cn.nukkit.level.Level;
010import cn.nukkit.math.AxisAlignedBB;
011import cn.nukkit.math.BlockFace;
012import cn.nukkit.nbt.tag.CompoundTag;
013import cn.nukkit.nbt.tag.Tag;
014import cn.nukkit.utils.BlockColor;
015import cn.nukkit.utils.Faceable;
016
017/**
018 * @author Nukkit Project Team
019 */
020public class BlockSignPost extends BlockTransparentMeta implements Faceable {
021
022    public BlockSignPost() {
023        this(0);
024    }
025
026    public BlockSignPost(int meta) {
027        super(meta);
028    }
029
030    @Override
031    public int getId() {
032        return SIGN_POST;
033    }
034
035    @Override
036    public double getHardness() {
037        return 1;
038    }
039
040    @Override
041    public double getResistance() {
042        return 5;
043    }
044
045    @Override
046    public boolean isSolid() {
047        return false;
048    }
049
050    @Override
051    public String getName() {
052        return "Sign Post";
053    }
054
055    @Override
056    public AxisAlignedBB getBoundingBox() {
057        return null;
058    }
059
060    @Override
061    public boolean place(Item item, Block block, Block target, BlockFace face, double fx, double fy, double fz, Player player) {
062        if (face != BlockFace.DOWN) {
063            CompoundTag nbt = new CompoundTag()
064                    .putString("id", BlockEntity.SIGN)
065                    .putInt("x", (int) block.x)
066                    .putInt("y", (int) block.y)
067                    .putInt("z", (int) block.z)
068                    .putString("Text1", "")
069                    .putString("Text2", "")
070                    .putString("Text3", "")
071                    .putString("Text4", "");
072
073            if (face == BlockFace.UP) {
074                setDamage((int) Math.floor(((player.yaw + 180) * 16 / 360) + 0.5) & 0x0f);
075                getLevel().setBlock(block, Block.get(BlockID.SIGN_POST, getDamage()), true);
076            } else {
077                setDamage(face.getIndex());
078                getLevel().setBlock(block, Block.get(BlockID.WALL_SIGN, getDamage()), true);
079            }
080
081            if (player != null) {
082                nbt.putString("Creator", player.getUniqueId().toString());
083            }
084
085            if (item.hasCustomBlockData()) {
086                for (Tag aTag : item.getCustomBlockData().getAllTags()) {
087                    nbt.put(aTag.getName(), aTag);
088                }
089            }
090
091            BlockEntitySign sign = (BlockEntitySign) BlockEntity.createBlockEntity(BlockEntity.SIGN, getLevel().getChunk((int) block.x >> 4, (int) block.z >> 4), nbt);
092            return sign != null;
093        }
094
095        return false;
096    }
097
098    @Override
099    public int onUpdate(int type) {
100        if (type == Level.BLOCK_UPDATE_NORMAL) {
101            if (down().getId() == Block.AIR) {
102                getLevel().useBreakOn(this);
103
104                return Level.BLOCK_UPDATE_NORMAL;
105            }
106        }
107
108        return 0;
109    }
110
111    @Override
112    public Item toItem() {
113        return new ItemSign();
114    }
115
116    @Override
117    public int getToolType() {
118        return ItemTool.TYPE_AXE;
119    }
120
121    @Override
122    public BlockColor getColor() {
123        return BlockColor.AIR_BLOCK_COLOR;
124    }
125
126    @Override
127    public BlockFace getBlockFace() {
128        return BlockFace.fromIndex(this.getDamage() & 0x07);
129    }
130}
