001package cn.nukkit.inventory;
002
003import cn.nukkit.Player;
004import cn.nukkit.item.Item;
005import cn.nukkit.item.enchantment.Enchantment;
006import cn.nukkit.level.Position;
007import cn.nukkit.network.protocol.LevelSoundEventPacket;
008
009import java.util.ArrayList;
010import java.util.Arrays;
011
012/**
013 * author: MagicDroidX
014 * Nukkit Project
015 */
016public class AnvilInventory extends FakeBlockUIComponent {
017
018    public static final int TARGET = 0;
019    public static final int SACRIFICE = 1;
020    public static final int RESULT = 50;
021
022    public AnvilInventory(PlayerUIInventory playerUI, Position position) {
023        super(playerUI, InventoryType.ANVIL, 1, position);
024    }
025
026    public boolean onRename(Player player, Item resultItem) {
027        Item local = getItem(TARGET);
028        Item second = getItem(SACRIFICE);
029
030        if (!resultItem.equals(local, true, false) || resultItem.getCount() != local.getCount()) {
031            //Item does not match target item. Everything must match except the tags.
032            return false;
033        }
034
035        if (local.equals(resultItem)) {
036            //just item transaction
037            return true;
038        }
039
040        if (local.getId() != 0 && second.getId() == 0) { //only rename
041            local.setCustomName(resultItem.getCustomName());
042            setItem(RESULT, local);
043            player.getInventory().addItem(local);
044            clearAll();
045            player.getInventory().sendContents(player);
046            sendContents(player);
047
048            player.getLevel().addLevelSoundEvent(player, LevelSoundEventPacket.SOUND_RANDOM_ANVIL_USE);
049            return true;
050        } else if (local.getId() != 0 && second.getId() != 0) { //enchants combining
051            if (!local.equals(second, true, false)) {
052                return false;
053            }
054
055            if (local.getId() != 0 && second.getId() != 0) {
056                Item result = local.clone();
057                int enchants = 0;
058
059                ArrayList<Enchantment> enchantments = new ArrayList<>(Arrays.asList(second.getEnchantments()));
060
061                ArrayList<Enchantment> baseEnchants = new ArrayList<>();
062
063                for (Enchantment ench : local.getEnchantments()) {
064                    if (ench.isMajor()) {
065                        baseEnchants.add(ench);
066                    }
067                }
068
069                for (Enchantment enchantment : enchantments) {
070                    if (enchantment.getLevel() < 0 || enchantment.getId() < 0) {
071                        continue;
072                    }
073
074                    if (enchantment.isMajor()) {
075                        boolean same = false;
076                        boolean another = false;
077
078                        for (Enchantment baseEnchant : baseEnchants) {
079                            if (baseEnchant.getId() == enchantment.getId())
080                                same = true;
081                            else {
082                                another = true;
083                            }
084                        }
085
086                        if (!same && another) {
087                            continue;
088                        }
089                    }
090
091                    Enchantment localEnchantment = local.getEnchantment(enchantment.getId());
092
093                    if (localEnchantment != null) {
094                        int level = Math.max(localEnchantment.getLevel(), enchantment.getLevel());
095
096                        if (localEnchantment.getLevel() == enchantment.getLevel())
097                            level++;
098
099                        enchantment.setLevel(level);
100                        result.addEnchantment(enchantment);
101                        continue;
102                    }
103
104                    result.addEnchantment(enchantment);
105                    enchants++;
106                }
107
108                result.setCustomName(resultItem.getCustomName());
109
110                player.getInventory().addItem(result);
111                player.getInventory().sendContents(player);
112                clearAll();
113                sendContents(player);
114
115                player.getLevel().addLevelSoundEvent(player, LevelSoundEventPacket.SOUND_RANDOM_ANVIL_USE);
116                return true;
117            }
118        }
119
120        return false;
121    }
122
123    @Override
124    public void onClose(Player who) {
125        super.onClose(who);
126        who.craftingType = Player.CRAFTING_SMALL;
127        who.resetCraftingGridType();
128
129        for (int i = 0; i < 2; ++i) {
130            this.getHolder().getLevel().dropItem(this.getHolder().add(0.5, 0.5, 0.5), this.getItem(i));
131            this.clear(i);
132        }
133    }
134
135    @Override
136    public void onOpen(Player who) {
137        super.onOpen(who);
138        who.craftingType = Player.CRAFTING_ANVIL;
139    }
140}
