110        switch (this.sourceType) {

117            Inventory window = player.getWindowById(this.windowId);
118            if (window != null) {
119                return new SlotChangeAction(window, this.inventorySlot, this.oldItem, this.newItem);
120            }
121            player.getServer().getLogger().debug("Player " + player.getName() + " has no open container with window ID " + this.windowId);
122            return null;
123        case SOURCE_WORLD: 
124            if (this.inventorySlot != InventoryTransactionPacket.ACTION_MAGIC_SLOT_DROP_ITEM) {
125                player.getServer().getLogger().debug("Only expecting drop-item world actions from the client!");
126                return null;
127            }
128            return new DropItemAction(this.oldItem, this.newItem);
129        case SOURCE_CREATIVE: 
130            int type;
131            switch (this.inventorySlot) {
132            case InventoryTransactionPacket.ACTION_MAGIC_SLOT_CREATIVE_DELETE_ITEM: 
133                type = CreativeInventoryAction.TYPE_DELETE_ITEM;
134                break;
135            case InventoryTransactionPacket.ACTION_MAGIC_SLOT_CREATIVE_CREATE_ITEM: 
136                type = CreativeInventoryAction.TYPE_CREATE_ITEM;
137                break;
138            default: 
139                player.getServer().getLogger().debug("Unexpected creative action type " + this.inventorySlot);
140                return null;
141            }
142            return new CreativeInventoryAction(this.oldItem, this.newItem, type);
143        case SOURCE_CRAFT_SLOT: 
144        case SOURCE_TODO: 
145            //These types need special handling.
146            switch (this.windowId) {
147            case SOURCE_TYPE_CRAFTING_ADD_INGREDIENT: 
148            case SOURCE_TYPE_CRAFTING_REMOVE_INGREDIENT: 
149                return new SlotChangeAction(player.getCraftingGrid(), this.inventorySlot, this.oldItem, this.newItem);
150            case SOURCE_TYPE_CONTAINER_DROP_CONTENTS: 
151                Optional<Inventory> inventory = player.getTopWindow();
152                if (!inventory.isPresent()) {
153                    // No window open?
154                    return null;
155                }
156                return new SlotChangeAction(inventory.get(), this.inventorySlot, this.oldItem, this.newItem);
157            case SOURCE_TYPE_CRAFTING_RESULT: 
158                return new CraftingTakeResultAction(this.oldItem, this.newItem);
159            case SOURCE_TYPE_CRAFTING_USE_INGREDIENT: 
160                return new CraftingTransferMaterialAction(this.oldItem, this.newItem, this.inventorySlot);
161            }
162            if (this.windowId >= SOURCE_TYPE_ANVIL_OUTPUT && this.windowId <= SOURCE_TYPE_ANVIL_INPUT) {
163                //anvil actions
164                Inventory inv = player.getWindowById(Player.ANVIL_WINDOW_ID);
165                if (!(inv instanceof AnvilInventory)) {
166                    player.getServer().getLogger().debug("Player " + player.getName() + " has no open anvil inventory");
167                    return null;
168                }

170                switch (this.windowId) {
171                case SOURCE_TYPE_ANVIL_INPUT:

193                }
194            }

196                Inventory inv = player.getWindowById(Player.ENCHANT_WINDOW_ID);
