024package co.aikar.timings;
025
026import static co.aikar.timings.TimingIdentifier.DEFAULT_GROUP;
027import static co.aikar.timings.TimingsManager.*;
028
029public class FullServerTickTiming extends Timing {
030    private static final TimingIdentifier IDENTIFIER = new TimingIdentifier(DEFAULT_GROUP.name, "Full Server Tick", null);
031    final TimingData minuteData;
032    double avgFreeMemory = -1D;
033    double avgUsedMemory = -1D;
034
035    FullServerTickTiming() {
036        super(IDENTIFIER);
037        this.minuteData = new TimingData(this.id);
038
039        TIMING_MAP.put(IDENTIFIER, this);
040    }
041
042    @Override
043    public Timing startTiming() {
044        if (TimingsManager.needsFullReset) {
045            TimingsManager.resetTimings();
046        } else if (TimingsManager.needsRecheckEnabled) {
047            TimingsManager.recheckEnabled();
048        }
049        super.startTiming();
050        return this;
051    }
052
053    @Override
054    public void stopTiming() {
055        super.stopTiming();
056        if (!this.enabled) {
057            return;
058        }
059
060        if (TimingsHistory.timedTicks % 20 == 0) {
061            final Runtime runtime = Runtime.getRuntime();
062            double usedMemory = runtime.totalMemory() - runtime.freeMemory();
063            double freeMemory = runtime.maxMemory() - usedMemory;
064
065            if (this.avgFreeMemory == -1) {
066                this.avgFreeMemory = freeMemory;
067            } else {
068                this.avgFreeMemory = (this.avgFreeMemory * (59 / 60D)) + (freeMemory * (1 / 60D));
069            }
070
071            if (this.avgUsedMemory == -1) {
072                this.avgUsedMemory = usedMemory;
073            } else {
074                this.avgUsedMemory = (this.avgUsedMemory * (59 / 60D)) + (usedMemory * (1 / 60D));
075            }
076        }
077
078        long start = System.nanoTime();
079        TimingsManager.tick();
080        long diff = System.nanoTime() - start;
081
082        CURRENT = Timings.timingsTickTimer;
083        Timings.timingsTickTimer.addDiff(diff);
084        //addDiff for timingsTickTimer incremented this, bring it back down to 1 per tick.
085        this.record.curTickCount--;
086        this.minuteData.curTickTotal = this.record.curTickTotal;
087        this.minuteData.curTickCount = 1;
088        boolean violated = isViolated();
089        this.minuteData.tick(violated);
090        Timings.timingsTickTimer.tick(violated);
091        tick(violated);
092
093        if (TimingsHistory.timedTicks % 1200 == 0) {
094            MINUTE_REPORTS.add(new TimingsHistory.MinuteReport());
095            TimingsHistory.resetTicks(false);
096            this.minuteData.reset();
097        }
098
099        if (TimingsHistory.timedTicks % Timings.getHistoryInterval() == 0) {
100            TimingsManager.HISTORY.add(new TimingsHistory());
101            TimingsManager.resetTimings();
102        }
103    }
104
105    boolean isViolated() {
106        return this.record.curTickTotal > 50000000;
107    }
108}
