001package cn.nukkit.nbt.tag;
002
003import cn.nukkit.nbt.stream.NBTInputStream;
004import cn.nukkit.nbt.stream.NBTOutputStream;
005
006import java.io.IOException;
007import java.io.PrintStream;
008import java.util.Collection;
009import java.util.HashMap;
010import java.util.Map;
011import java.util.Map.Entry;
012import java.util.StringJoiner;
013
014public class CompoundTag extends Tag implements Cloneable {
015    private final Map<String, Tag> tags = new HashMap<>();
016
017    public CompoundTag() {
018        super("");
019    }
020
021    public CompoundTag(String name) {
022        super(name);
023    }
024
025    @Override
026    public void write(NBTOutputStream dos) throws IOException {
027
028        for (Map.Entry<String, Tag> entry : this.tags.entrySet()) {
029            Tag.writeNamedTag(entry.getValue(), entry.getKey(), dos);
030        }
031
032        dos.writeByte(Tag.TAG_End);
033    }
034
035    @Override
036    public void load(NBTInputStream dis) throws IOException {
037        tags.clear();
038        Tag tag;
039        while ((tag = Tag.readNamedTag(dis)).getId() != Tag.TAG_End) {
040            tags.put(tag.getName(), tag);
041        }
042    }
043
044    public Collection<Tag> getAllTags() {
045        return tags.values();
046    }
047
048    @Override
049    public byte getId() {
050        return TAG_Compound;
051    }
052
053    public CompoundTag put(String name, Tag tag) {
054        tags.put(name, tag.setName(name));
055        return this;
056    }
057
058    public CompoundTag putByte(String name, int value) {
059        tags.put(name, new ByteTag(name, value));
060        return this;
061    }
062
063    public CompoundTag putShort(String name, int value) {
064        tags.put(name, new ShortTag(name, value));
065        return this;
066    }
067
068    public CompoundTag putInt(String name, int value) {
069        tags.put(name, new IntTag(name, value));
070        return this;
071    }
072
073    public CompoundTag putLong(String name, long value) {
074        tags.put(name, new LongTag(name, value));
075        return this;
076    }
077
078    public CompoundTag putFloat(String name, float value) {
079        tags.put(name, new FloatTag(name, value));
080        return this;
081    }
082
083    public CompoundTag putDouble(String name, double value) {
084        tags.put(name, new DoubleTag(name, value));
085        return this;
086    }
087
088    public CompoundTag putString(String name, String value) {
089        tags.put(name, new StringTag(name, value));
090        return this;
091    }
092
093    public CompoundTag putByteArray(String name, byte[] value) {
094        tags.put(name, new ByteArrayTag(name, value));
095        return this;
096    }
097
098    public CompoundTag putIntArray(String name, int[] value) {
099        tags.put(name, new IntArrayTag(name, value));
100        return this;
101    }
102
103    public CompoundTag putList(ListTag<? extends Tag> listTag) {
104        tags.put(listTag.getName(), listTag);
105        return this;
106    }
107
108    public CompoundTag putCompound(String name, CompoundTag value) {
109        tags.put(name, value.setName(name));
110        return this;
111    }
112
113    public CompoundTag putBoolean(String string, boolean val) {
114        putByte(string, val ? 1 : 0);
115        return this;
116    }
117
118    public Tag get(String name) {
119        return tags.get(name);
120    }
121
122    public boolean contains(String name) {
123        return tags.containsKey(name);
124    }
125
126    public CompoundTag remove(String name) {
127        tags.remove(name);
128        return this;
129    }
130
131    public <T extends Tag> T removeAndGet(String name) {
132        return (T) tags.remove(name);
133    }
134
135
136    public int getByte(String name) {
137        if (!tags.containsKey(name)) return (byte) 0;
138        return ((NumberTag) tags.get(name)).getData().intValue();
139    }
140
141    public int getShort(String name) {
142        if (!tags.containsKey(name)) return 0;
143        return ((NumberTag) tags.get(name)).getData().intValue();
144    }
145
146    public int getInt(String name) {
147        if (!tags.containsKey(name)) return 0;
148        return ((NumberTag) tags.get(name)).getData().intValue();
149    }
150
151    public long getLong(String name) {
152        if (!tags.containsKey(name)) return 0;
153        return ((NumberTag) tags.get(name)).getData().longValue();
154    }
155
156    public float getFloat(String name) {
157        if (!tags.containsKey(name)) return (float) 0;
158        return ((NumberTag) tags.get(name)).getData().floatValue();
159    }
160
161    public double getDouble(String name) {
162        if (!tags.containsKey(name)) return 0;
163        return ((NumberTag) tags.get(name)).getData().doubleValue();
164    }
165
166    public String getString(String name) {
167        if (!tags.containsKey(name)) return "";
168        Tag tag = tags.get(name);
169        if (tag instanceof NumberTag) {
170            return String.valueOf(((NumberTag) tag).getData());
171        }
172        return ((StringTag) tag).data;
173    }
174
175    public byte[] getByteArray(String name) {
176        if (!tags.containsKey(name)) return new byte[0];
177        return ((ByteArrayTag) tags.get(name)).data;
178    }
179
180    public int[] getIntArray(String name) {
181        if (!tags.containsKey(name)) return new int[0];
182        return ((IntArrayTag) tags.get(name)).data;
183    }
184
185    public CompoundTag getCompound(String name) {
186        if (!tags.containsKey(name)) return new CompoundTag(name);
187        return (CompoundTag) tags.get(name);
188    }
189
190    public ListTag<? extends Tag> getList(String name) {
191        if (!tags.containsKey(name)) return new ListTag<>(name);
192        return (ListTag<? extends Tag>) tags.get(name);
193    }
194
195    @SuppressWarnings("unchecked")
196    public <T extends Tag> ListTag<T> getList(String name, Class<T> type) {
197        if (tags.containsKey(name)) {
198            return (ListTag<T>) tags.get(name);
199        }
200        return new ListTag<>(name);
201    }
202
203    public Map<String, Tag> getTags() {
204        return new HashMap<>(this.tags);
205    }
206
207    @Override
208    public Map<String, Object> parseValue() {
209        Map<String, Object> value = new HashMap<>(this.tags.size());
210
211        for (Entry<String, Tag> entry : this.tags.entrySet()) {
212            value.put(entry.getKey(), entry.getValue().parseValue());
213        }
214
215        return value;
216    }
217
218    public boolean getBoolean(String name) {
219        return getByte(name) != 0;
220    }
221
222    public String toString() {
223        StringJoiner joiner = new StringJoiner(",\n\t");
224        tags.forEach((key, tag) -> joiner.add('\'' + key + "' : " + tag.toString().replace("\n", "\n\t")));
225        return "CompoundTag '" + this.getName() + "' (" + tags.size() + " entries) {\n\t" + joiner.toString() + "\n}";
226    }
227
228    public void print(String prefix, PrintStream out) {
229        super.print(prefix, out);
230        out.println(prefix + "{");
231        String orgPrefix = prefix;
232        prefix += "   ";
233        for (Tag tag : tags.values()) {
234            tag.print(prefix, out);
235        }
236        out.println(orgPrefix + "}");
237    }
238
239    public boolean isEmpty() {
240        return tags.isEmpty();
241    }
242
243    public CompoundTag copy() {
244        CompoundTag tag = new CompoundTag(getName());
245        for (String key : tags.keySet()) {
246            tag.put(key, tags.get(key).copy());
247        }
248        return tag;
249    }
250
251    @Override
252    public boolean equals(Object obj) {
253        if (super.equals(obj)) {
254            CompoundTag o = (CompoundTag) obj;
255            return tags.entrySet().equals(o.tags.entrySet());
256        }
257        return false;
258    }
259
260    /**
261     * Check existence of NBT tag
262     *
263     * @param name - NBT tag Id.
264     * @return - true, if tag exists
265     */
266    public boolean exist(String name) {
267        return tags.containsKey(name);
268    }
269
270    @Override
271    public CompoundTag clone() {
272        CompoundTag nbt = new CompoundTag();
273        this.getTags().forEach((key, value) -> nbt.put(key, value.copy()));
274        return nbt;
275    }
276}
