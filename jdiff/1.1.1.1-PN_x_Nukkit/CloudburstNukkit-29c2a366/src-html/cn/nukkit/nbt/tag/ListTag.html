001package cn.nukkit.nbt.tag;
002
003import cn.nukkit.nbt.stream.NBTInputStream;
004import cn.nukkit.nbt.stream.NBTOutputStream;
005
006import java.io.IOException;
007import java.io.PrintStream;
008import java.util.ArrayList;
009import java.util.Collection;
010import java.util.List;
011import java.util.StringJoiner;
012
013public class ListTag<T extends Tag> extends Tag {
014
015    private List<T> list = new ArrayList<>();
016
017    public byte type;
018
019    public ListTag() {
020        super("");
021    }
022
023    public ListTag(String name) {
024        super(name);
025    }
026
027    @Override
028    void write(NBTOutputStream dos) throws IOException {
029        if (list.size() > 0) type = list.get(0).getId();
030        else type = 1;
031
032        dos.writeByte(type);
033        dos.writeInt(list.size());
034        for (T aList : list) aList.write(dos);
035    }
036
037    @Override
038    @SuppressWarnings("unchecked")
039    void load(NBTInputStream dis) throws IOException {
040        type = dis.readByte();
041        int size = dis.readInt();
042
043        list = new ArrayList<>(size);
044        for (int i = 0; i < size; i++) {
045            Tag tag = Tag.newTag(type, null);
046            tag.load(dis);
047            tag.setName("");
048            list.add((T) tag);
049        }
050    }
051
052    @Override
053    public byte getId() {
054        return TAG_List;
055    }
056
057    @Override
058    public String toString() {
059        StringJoiner joiner = new StringJoiner(",\n\t");
060        list.forEach(tag -> joiner.add(tag.toString().replace("\n", "\n\t")));
061        return "ListTag '" + this.getName() + "' (" + list.size() + " entries of type " + Tag.getTagName(type) + ") {\n\t" + joiner.toString() + "\n}";
062    }
063
064    public void print(String prefix, PrintStream out) {
065        super.print(prefix, out);
066
067        out.println(prefix + "{");
068        String orgPrefix = prefix;
069        prefix += "   ";
070        for (T aList : list) aList.print(prefix, out);
071        out.println(orgPrefix + "}");
072    }
073
074    public ListTag<T> add(T tag) {
075        type = tag.getId();
076        tag.setName("");
077        list.add(tag);
078        return this;
079    }
080
081    public ListTag<T> add(int index, T tag) {
082        type = tag.getId();
083        tag.setName("");
084
085        if (index >= list.size()) {
086            list.add(index, tag);
087        } else {
088            list.set(index, tag);
089        }
090        return this;
091    }
092
093    @Override
094    public List<Object> parseValue() {
095        List<Object> value = new ArrayList<>(this.list.size());
096
097        for (T t : this.list) {
098            value.add(t.parseValue());
099        }
100
101        return value;
102    }
103
104    public T get(int index) {
105        return list.get(index);
106    }
107
108    public List<T> getAll() {
109        return new ArrayList<>(list);
110    }
111
112    public void setAll(List<T> tags) {
113        this.list = new ArrayList<>(tags);
114    }
115
116    public void remove(T tag) {
117        list.remove(tag);
118    }
119
120    public void remove(int index) {
121        list.remove(index);
122    }
123
124    public void removeAll(Collection<T> tags) {
125        list.remove(tags);
126    }
127
128    public int size() {
129        return list.size();
130    }
131
132    @Override
133    public Tag copy() {
134        ListTag<T> res = new ListTag<>(getName());
135        res.type = type;
136        for (T t : list) {
137            @SuppressWarnings("unchecked")
138            T copy = (T) t.copy();
139            res.list.add(copy);
140        }
141        return res;
142    }
143
144    @SuppressWarnings("rawtypes")
145    @Override
146    public boolean equals(Object obj) {
147        if (super.equals(obj)) {
148            ListTag o = (ListTag) obj;
149            if (type == o.type) {
150                return list.equals(o.list);
151            }
152        }
153        return false;
154    }
155
156}
