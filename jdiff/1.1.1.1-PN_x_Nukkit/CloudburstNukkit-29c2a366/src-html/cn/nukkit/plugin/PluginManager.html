088    public Plugin loadPlugin(File file, Map<String, PluginLoader> loaders) {
089        for (PluginLoader loader : (loaders == null ? this.fileAssociations : loaders).values()) {
090            for (Pattern pattern : loader.getPluginFilters()) {
091                if (pattern.matcher(file.getName()).matches()) {
092                    PluginDescription description = loader.getPluginDescription(file);
093                    if (description != null) {
094                        try {
095                            Plugin plugin = loader.loadPlugin(file);
096                            if (plugin != null) {
097                                this.plugins.put(plugin.getDescription().getName(), plugin);
098
099                                List<PluginCommand> pluginCommands = this.parseYamlCommands(plugin);
100
101                                if (!pluginCommands.isEmpty()) {
102                                    this.commandMap.registerAll(plugin.getDescription().getName(), pluginCommands);
103                                }
104
105                                return plugin;
106                            }
107                        } catch (Exception e) {
108                            Server.getInstance().getLogger().critical("Could not load plugin", e);
109                            return null;
110                        }
111                    }
112                }
113            }
114        }
115
116        return null;
117    }

135    public Map<String, Plugin> loadPlugins(File dictionary, List<String> newLoaders, boolean includeDir) {
136        if (dictionary.isDirectory()) {
137            Map<String, File> plugins = new LinkedHashMap<>();
138            Map<String, Plugin> loadedPlugins = new LinkedHashMap<>();
139            Map<String, List<String>> dependencies = new LinkedHashMap<>();
140            Map<String, List<String>> softDependencies = new LinkedHashMap<>();
141            Map<String, PluginLoader> loaders = new LinkedHashMap<>();
142            if (newLoaders != null) {
143                for (String key : newLoaders) {
144                    if (this.fileAssociations.containsKey(key)) {
145                        loaders.put(key, this.fileAssociations.get(key));
146                    }
147                }
148            } else {
149                loaders = this.fileAssociations;
150            }
151
152            for (final PluginLoader loader : loaders.values()) {
153                for (File file : dictionary.listFiles((dir, name) -> {
154                    for (Pattern pattern : loader.getPluginFilters()) {
155                        if (pattern.matcher(name).matches()) {
156                            return true;
157                        }
158                    }
159                    return false;
160                })) {
161                    if (file.isDirectory() && !includeDir) {
162                        continue;
163                    }
164                    try {
165                        PluginDescription description = loader.getPluginDescription(file);
166                        if (description != null) {
167                            String name = description.getName();
168
169                            if (plugins.containsKey(name) || this.getPlugin(name) != null) {
170                                this.server.getLogger().error(this.server.getLanguage().translateString("nukkit.plugin.duplicateError", name));
171                                continue;
172                            }
173
174                            boolean compatible = false;
175
176                            for (String version : description.getCompatibleAPIs()) {
177
178                                try {
179                                    //Check the format: majorVersion.minorVersion.patch
180                                    if (!Pattern.matches("^[0-9]+\\.[0-9]+\\.[0-9]+$", version)) {
181                                        throw new IllegalArgumentException();
182                                    }
183                                } catch (NullPointerException | IllegalArgumentException e) {
184                                    this.server.getLogger().error(this.server.getLanguage().translateString("nukkit.plugin.loadError", new String[]{name, "Wrong API format"}));
185                                    continue;
186                                }
187
188                                String[] versionArray = version.split("\\.");
189                                String[] apiVersion = this.server.getApiVersion().split("\\.");
190
191                                //Completely different API version
192                                if (!Objects.equals(Integer.valueOf(versionArray[0]), Integer.valueOf(apiVersion[0]))) {
193                                    continue;
194                                }
195
196                                //If the plugin requires new API features, being backwards compatible
197                                if (Integer.parseInt(versionArray[1]) > Integer.parseInt(apiVersion[1])) {
198                                    continue;
199                                }
200
201                                compatible = true;
202                                break;
203                            }
204
205                            if (!compatible) {
206                                this.server.getLogger().error(this.server.getLanguage().translateString("nukkit.plugin.loadError", new String[]{name, "%nukkit.plugin.incompatibleAPI"}));
207                            }
208
209                            plugins.put(name, file);
210
211                            softDependencies.put(name, description.getSoftDepend());
212
213                            dependencies.put(name, description.getDepend());
214
215                            for (String before : description.getLoadBefore()) {
216                                if (softDependencies.containsKey(before)) {
217                                    softDependencies.get(before).add(name);
218                                } else {
219                                    List<String> list = new ArrayList<>();
220                                    list.add(name);
221                                    softDependencies.put(before, list);
222                                }
223                            }
224                        }
225                    } catch (Exception e) {
226                        this.server.getLogger().error(this.server.getLanguage().translateString("nukkit.plugin" +
227                                ".fileError", file.getName(), dictionary.toString(), Utils
228                                .getExceptionMessage(e)));
229                        MainLogger logger = this.server.getLogger();
230                        if (logger != null) {
231                            logger.logException(e);
232                        }
233                    }
234                }
235            }
236
237            while (!plugins.isEmpty()) {
238                boolean missingDependency = true;
239                for (String name : new ArrayList<>(plugins.keySet())) {
240                    File file = plugins.get(name);
241                    if (dependencies.containsKey(name)) {
242                        for (String dependency : new ArrayList<>(dependencies.get(name))) {
243                            if (loadedPlugins.containsKey(dependency) || this.getPlugin(dependency) != null) {
244                                dependencies.get(name).remove(dependency);
245                            } else if (!plugins.containsKey(dependency)) {
246                                this.server.getLogger().critical(this.server.getLanguage().translateString("nukkit" +
247                                        ".plugin.loadError", new String[]{name, "%nukkit.plugin.unknownDependency"}));
248                                break;
249                            }
250                        }
251
252                        if (dependencies.get(name).isEmpty()) {
253                            dependencies.remove(name);
254                        }
255                    }
256
257                    if (softDependencies.containsKey(name)) {
258                        softDependencies.get(name).removeIf(dependency ->
259                                loadedPlugins.containsKey(dependency) || this.getPlugin(dependency) != null);
260
261                        if (softDependencies.get(name).isEmpty()) {
262                            softDependencies.remove(name);
263                        }
264                    }
265
266                    if (!dependencies.containsKey(name) && !softDependencies.containsKey(name)) {
267                        plugins.remove(name);
268                        missingDependency = false;
269                        Plugin plugin = this.loadPlugin(file, loaders);
270                        if (plugin != null) {
271                            loadedPlugins.put(name, plugin);
272                        } else {
273                            this.server.getLogger().critical(this.server.getLanguage().translateString("nukkit.plugin.genericLoadError", name));
274                        }
275                    }
276                }
277
278                if (missingDependency) {
279                    for (String name : new ArrayList<>(plugins.keySet())) {
280                        File file = plugins.get(name);
281                        if (!dependencies.containsKey(name)) {
282                            softDependencies.remove(name);
283                            plugins.remove(name);
284                            missingDependency = false;
285                            Plugin plugin = this.loadPlugin(file, loaders);
286                            if (plugin != null) {
287                                loadedPlugins.put(name, plugin);
288                            } else {
289                                this.server.getLogger().critical(this.server.getLanguage().translateString("nukkit.plugin.genericLoadError", name));
290                            }
291                        }
292                    }
293
294                    if (missingDependency) {
295                        for (String name : plugins.keySet()) {
296                            this.server.getLogger().critical(this.server.getLanguage().translateString("nukkit.plugin.loadError", new String[]{name, "%nukkit.plugin.circularDependency"}));
297                        }
298                        plugins.clear();
299                    }
300                }
301            }
302
303            return loadedPlugins;
304        } else {
305            return new HashMap<>();
306        }
307    }

343    public void recalculatePermissionDefaults(Permission permission) {
344        if (this.permissions.containsKey(permission.getName())) {
345            this.defaultPermsOp.remove(permission.getName());
346            this.defaultPerms.remove(permission.getName());
347            this.calculatePermissionDefault(permission);
348        }
349    }

351    private void calculatePermissionDefault(Permission permission) {
352        Timings.permissionDefaultTimer.startTiming();
353        if (permission.getDefault().equals(Permission.DEFAULT_OP) || permission.getDefault().equals(Permission.DEFAULT_TRUE)) {
354            this.defaultPermsOp.put(permission.getName(), permission);
355            this.dirtyPermissibles(true);
356        }
357
358        if (permission.getDefault().equals(Permission.DEFAULT_NOT_OP) || permission.getDefault().equals(Permission.DEFAULT_TRUE)) {
359            this.defaultPerms.put(permission.getName(), permission);
360            this.dirtyPermissibles(false);
361        }
362        Timings.permissionDefaultTimer.startTiming();
363    }
364
365    private void dirtyPermissibles(boolean op) {
366        for (Permissible p : this.getDefaultPermSubscriptions(op)) {
367            p.recalculatePermissions();
368        }
369    }

430    public void enablePlugin(Plugin plugin) {
431        if (!plugin.isEnabled()) {
432            try {
433                for (Permission permission : plugin.getDescription().getPermissions()) {
434                    this.addPermission(permission);
435                }
436                plugin.getPluginLoader().enablePlugin(plugin);
437            } catch (Throwable e) {
438                MainLogger logger = this.server.getLogger();
439                if (logger != null) {
440                    logger.logException(new RuntimeException(e));
441                }
442                this.disablePlugin(plugin);
443            }
444        }
445    }

447    protected List<PluginCommand> parseYamlCommands(Plugin plugin) {
448        List<PluginCommand> pluginCmds = new ArrayList<>();
449
450        for (Map.Entry entry : plugin.getDescription().getCommands().entrySet()) {
451            String key = (String) entry.getKey();
452            Object data = entry.getValue();
453            if (key.contains(":")) {
454                this.server.getLogger().critical(this.server.getLanguage().translateString("nukkit.plugin.commandError", new String[]{key, plugin.getDescription().getFullName()}));
455                continue;
456            }
457            if (data instanceof Map) {
458                PluginCommand newCmd = new PluginCommand<>(key, plugin);
459
460                if (((Map) data).containsKey("description")) {
461                    newCmd.setDescription((String) ((Map) data).get("description"));
462                }
463
464                if (((Map) data).containsKey("usage")) {
465                    newCmd.setUsage((String) ((Map) data).get("usage"));
466                }
467
468                if (((Map) data).containsKey("aliases")) {
469                    Object aliases = ((Map) data).get("aliases");
470                    if (aliases instanceof List) {
471                        List<String> aliasList = new ArrayList<>();
472                        for (String alias : (List<String>) aliases) {
473                            if (alias.contains(":")) {
474                                this.server.getLogger().critical(this.server.getLanguage().translateString("nukkit.plugin.aliasError", new String[]{alias, plugin.getDescription().getFullName()}));
475                                continue;
476                            }
477                            aliasList.add(alias);
478                        }
479
480                        newCmd.setAliases(aliasList.toArray(new String[0]));
481                    }
482                }
483
484                if (((Map) data).containsKey("permission")) {
485                    newCmd.setPermission((String) ((Map) data).get("permission"));
486                }
487
488                if (((Map) data).containsKey("permission-message")) {
489                    newCmd.setPermissionMessage((String) ((Map) data).get("permission-message"));
490                }
491
492                pluginCmds.add(newCmd);
493            }
494        }
495
496        return pluginCmds;
497    }

499    public void disablePlugins() {
500        ListIterator<Plugin> plugins = new ArrayList<>(this.getPlugins().values()).listIterator(this.getPlugins().size());
501
502        while (plugins.hasPrevious()) {
503            this.disablePlugin(plugins.previous());
504        }
505    }

507    public void disablePlugin(Plugin plugin) {
508        if (plugin.isEnabled()) {
509            try {
510                plugin.getPluginLoader().disablePlugin(plugin);
511            } catch (Exception e) {
512                MainLogger logger = this.server.getLogger();
513                if (logger != null) {
514                    logger.logException(e);
515                }
516            }
517
518            this.server.getScheduler().cancelTask(plugin);
519            HandlerList.unregisterAll(plugin);
520            for (Permission permission : plugin.getDescription().getPermissions()) {
521                this.removePermission(permission);
522            }
523        }
524    }

535    public void callEvent(Event event) {
536        try {
537            for (RegisteredListener registration : getEventListeners(event.getClass()).getRegisteredListeners()) {
538                if (!registration.getPlugin().isEnabled()) {
539                    continue;
540                }
541
542                try {
543                    registration.callEvent(event);
544                } catch (Exception e) {
545                    this.server.getLogger().critical(this.server.getLanguage().translateString("nukkit.plugin.eventError", event.getEventName(), registration.getPlugin().getDescription().getFullName(), e.getMessage(), registration.getListener().getClass().getName()));
546                    this.server.getLogger().logException(e);
547                }
548            }
549        } catch (IllegalAccessException e) {
550            this.server.getLogger().logException(e);
551        }
552    }

554    public void registerEvents(Listener listener, Plugin plugin) {
555        if (!plugin.isEnabled()) {
556            throw new PluginException("Plugin attempted to register " + listener.getClass().getName() + " while not enabled");
557        }
558
559        Map<Class<? extends Event>, Set<RegisteredListener>> ret = new HashMap<>();
560        Set<Method> methods;
561        try {
562            Method[] publicMethods = listener.getClass().getMethods();
563            Method[] privateMethods = listener.getClass().getDeclaredMethods();
564            methods = new HashSet<>(publicMethods.length + privateMethods.length, 1.0f);
565            Collections.addAll(methods, publicMethods);
566            Collections.addAll(methods, privateMethods);
567        } catch (NoClassDefFoundError e) {
568            plugin.getLogger().error("Plugin " + plugin.getDescription().getFullName() + " has failed to register events for " + listener.getClass() + " because " + e.getMessage() + " does not exist.");
569            return;
570        }
571
572        for (final Method method : methods) {
573            final EventHandler eh = method.getAnnotation(EventHandler.class);
574            if (eh == null) continue;
575            if (method.isBridge() || method.isSynthetic()) {
576                continue;
577            }
578            final Class<?> checkClass;
579
580            if (method.getParameterTypes().length != 1 || !Event.class.isAssignableFrom(checkClass = method.getParameterTypes()[0])) {
581                plugin.getLogger().error(plugin.getDescription().getFullName() + " attempted to register an invalid EventHandler method signature \"" + method.toGenericString() + "\" in " + listener.getClass());
582                continue;
583            }
584
585            final Class<? extends Event> eventClass = checkClass.asSubclass(Event.class);
586            method.setAccessible(true);
587
588            for (Class<?> clazz = eventClass; Event.class.isAssignableFrom(clazz); clazz = clazz.getSuperclass()) {
589                // This loop checks for extending deprecated events
590                if (clazz.getAnnotation(Deprecated.class) != null) {
591                    if (Boolean.parseBoolean(String.valueOf(this.server.getConfig("settings.deprecated-verbpse", true)))) {
592                        this.server.getLogger().warning(this.server.getLanguage().translateString("nukkit.plugin.deprecatedEvent", plugin.getName(), clazz.getName(), listener.getClass().getName() + "." + method.getName() + "()"));
593                    }
594                    break;
595                }
596            }
597            this.registerEvent(eventClass, listener, eh.priority(), new MethodEventExecutor(method), plugin, eh.ignoreCancelled());
598        }
599    }

605    public void registerEvent(Class<? extends Event> event, Listener listener, EventPriority priority, EventExecutor executor, Plugin plugin, boolean ignoreCancelled) throws PluginException {
606        if (!plugin.isEnabled()) {
607            throw new PluginException("Plugin attempted to register " + event + " while not enabled");
608        }
609
610        try {
611            Timing timing = Timings.getPluginEventTiming(event, listener, executor, plugin);
612            this.getEventListeners(event).register(new RegisteredListener(listener, executor, priority, plugin, ignoreCancelled, timing));
613        } catch (IllegalAccessException e) {
614            Server.getInstance().getLogger().logException(e);
615        }
616    }

618    private HandlerList getEventListeners(Class<? extends Event> type) throws IllegalAccessException {
619        try {
620            Method method = getRegistrationClass(type).getDeclaredMethod("getHandlers");
621            method.setAccessible(true);
622            return (HandlerList) method.invoke(null);
623        } catch (NullPointerException e) {
624            throw new IllegalArgumentException("getHandlers method in " + type.getName() + " was not static!");
625        } catch (Exception e) {
626            throw new IllegalAccessException(Utils.getExceptionMessage(e));
627        }
628    }
629
630    private Class<? extends Event> getRegistrationClass(Class<? extends Event> clazz) throws IllegalAccessException {
631        try {
632            clazz.getDeclaredMethod("getHandlers");
633            return clazz;
634        } catch (NoSuchMethodException e) {
635            if (clazz.getSuperclass() != null
636                    && !clazz.getSuperclass().equals(Event.class)
637                    && Event.class.isAssignableFrom(clazz.getSuperclass())) {
638                return getRegistrationClass(clazz.getSuperclass().asSubclass(Event.class));
639            } else {
640                throw new IllegalAccessException("Unable to find handler list for event " + clazz.getName() + ". Static getHandlers method required!");
641            }
642        }
643    }
