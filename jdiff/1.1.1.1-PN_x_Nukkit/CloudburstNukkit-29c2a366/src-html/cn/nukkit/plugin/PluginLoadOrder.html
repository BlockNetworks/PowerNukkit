001package cn.nukkit.plugin;
002
003/**
004 * 描述一个Nukkit插件加载顺序的类。<br>
005 * Describes a Nukkit plugin load order.
006 * <p>
007 * <p>Nukkit插件的加载顺序有两个:{@link cn.nukkit.plugin.PluginLoadOrder#STARTUP}
008 * 和 {@link cn.nukkit.plugin.PluginLoadOrder#POSTWORLD}。<br>
009 * The load order of a Nukkit plugin can be {@link cn.nukkit.plugin.PluginLoadOrder#STARTUP}
010 * or {@link cn.nukkit.plugin.PluginLoadOrder#POSTWORLD}.</p>
011 *
012 * @author MagicDroidX(code) @ Nukkit Project
013 * @author iNevet(code) @ Nukkit Project
014 * @author 粉鞋大妈(javadoc) @ Nukkit Project
015 * @since Nukkit 1.0 | Nukkit API 1.0.0
016 */
017public enum PluginLoadOrder {
018    /**
019     * 表示这个插件在服务器启动时就开始加载。<br>
020     * Indicates that the plugin will be loaded at startup.
021     *
022     * @see cn.nukkit.plugin.PluginLoadOrder
023     * @since Nukkit 1.0 | Nukkit API 1.0.0
024     */
025    STARTUP,
026    /**
027     * 表示这个插件在第一个世界加载完成后开始加载。<br>
028     * Indicates that the plugin will be loaded after the first/default world was created.
029     *
030     * @see cn.nukkit.plugin.PluginLoadOrder
031     * @since Nukkit 1.0 | Nukkit API 1.0.0
032     */
033    POSTWORLD
034}
