001package cn.nukkit.entity.data;
002
003import cn.nukkit.entity.Entity;
004import cn.nukkit.math.BlockVector3;
005import cn.nukkit.math.Vector3;
006
007/**
008 * author: MagicDroidX
009 * Nukkit Project
010 */
011public class IntPositionEntityData extends EntityData<BlockVector3> {
012    public int x;
013    public int y;
014    public int z;
015
016    public IntPositionEntityData(int id, int x, int y, int z) {
017        super(id);
018        this.x = x;
019        this.y = y;
020        this.z = z;
021    }
022
023    public IntPositionEntityData(int id, Vector3 pos) {
024        this(id, (int) pos.x, (int) pos.y, (int) pos.z);
025    }
026
027    @Override
028    public BlockVector3 getData() {
029        return new BlockVector3(x, y, z);
030    }
031
032    @Override
033    public void setData(BlockVector3 data) {
034        if (data != null) {
035            this.x = data.x;
036            this.y = data.y;
037            this.z = data.z;
038        }
039    }
040
041    @Override
042    public int getType() {
043        return Entity.DATA_TYPE_POS;
044    }
045
046    @Override
047    public String toString() {
048        return "(" + x + ", " + y + ", " + z + ")";
049    }
050}
