001package cn.nukkit.entity.data;
002
003import cn.nukkit.entity.Entity;
004import cn.nukkit.math.Vector3f;
005
006/**
007 * author: MagicDroidX
008 * Nukkit Project
009 */
010public class Vector3fEntityData extends EntityData<Vector3f> {
011    public float x;
012    public float y;
013    public float z;
014
015    public Vector3fEntityData(int id, float x, float y, float z) {
016        super(id);
017        this.x = x;
018        this.y = y;
019        this.z = z;
020    }
021
022    public Vector3fEntityData(int id, Vector3f pos) {
023        this(id, pos.x, pos.y, pos.z);
024    }
025
026    @Override
027    public Vector3f getData() {
028        return new Vector3f(x, y, z);
029    }
030
031    @Override
032    public void setData(Vector3f data) {
033        if (data != null) {
034            this.x = data.x;
035            this.y = data.y;
036            this.z = data.z;
037        }
038    }
039
040    @Override
041    public int getType() {
042        return Entity.DATA_TYPE_VECTOR3F;
043    }
044
045    @Override
046    public String toString() {
047        return "(" + x + ", " + y + ", " + z + ")";
048    }
049}
