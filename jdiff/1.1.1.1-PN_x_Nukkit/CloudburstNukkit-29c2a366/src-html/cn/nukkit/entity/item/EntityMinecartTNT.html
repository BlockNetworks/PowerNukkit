001package cn.nukkit.entity.item;
002
003import cn.nukkit.Player;
004import cn.nukkit.block.Block;
005import cn.nukkit.block.BlockID;
006import cn.nukkit.entity.Entity;
007import cn.nukkit.entity.EntityExplosive;
008import cn.nukkit.entity.data.IntEntityData;
009import cn.nukkit.event.entity.EntityExplosionPrimeEvent;
010import cn.nukkit.item.Item;
011import cn.nukkit.item.ItemMinecartTNT;
012import cn.nukkit.level.Explosion;
013import cn.nukkit.level.GameRule;
014import cn.nukkit.level.format.FullChunk;
015import cn.nukkit.math.Vector3;
016import cn.nukkit.nbt.tag.CompoundTag;
017import cn.nukkit.network.protocol.LevelSoundEventPacket;
018import cn.nukkit.utils.MinecartType;
019
020import java.util.concurrent.ThreadLocalRandom;
021
022/**
023 * Author: Adam Matthew [larryTheCoder]
024 * <p>
025 * Nukkit Project.
026 */
027public class EntityMinecartTNT extends EntityMinecartAbstract implements EntityExplosive {
028
029    public static final int NETWORK_ID = 97;
030    private int fuse;
031    private boolean activated = false;
032
033    public EntityMinecartTNT(FullChunk chunk, CompoundTag nbt) {
034        super(chunk, nbt);
035        super.setDisplayBlock(Block.get(BlockID.TNT), false);
036    }
037
038    @Override
039    public boolean isRideable() {
040        return false;
041    }
042
043    @Override
044    public void initEntity() {
045        super.initEntity();
046
047        if (namedTag.contains("TNTFuse")) {
048            fuse = namedTag.getByte("TNTFuse");
049        } else {
050            fuse = 80;
051        }
052        this.setDataFlag(DATA_FLAGS, DATA_FLAG_CHARGED, false);
053    }
054
055    @Override
056    public boolean onUpdate(int currentTick) {
057        this.timing.startTiming();
058
059        if (fuse < 80) {
060            int tickDiff = currentTick - lastUpdate;
061
062            lastUpdate = currentTick;
063
064            if (fuse % 5 == 0) {
065                setDataProperty(new IntEntityData(DATA_FUSE_LENGTH, fuse));
066            }
067
068            fuse -= tickDiff;
069
070            if (isAlive() && fuse <= 0) {
071                if (this.level.getGameRules().getBoolean(GameRule.TNT_EXPLODES)) {
072                    this.explode(ThreadLocalRandom.current().nextInt(5));
073                }
074                this.close();
075                return false;
076            }
077        }
078
079        this.timing.stopTiming();
080
081        return super.onUpdate(currentTick);
082    }
083
084    @Override
085    public void activate(int x, int y, int z, boolean flag) {
086        level.addLevelSoundEvent(this, LevelSoundEventPacket.SOUND_IGNITE);
087        this.fuse = 79;
088    }
089
090    @Override
091    public void explode() {
092        explode(0);
093    }
094
095    public void explode(double square) {
096        double root = Math.sqrt(square);
097
098        if (root > 5.0D) {
099            root = 5.0D;
100        }
101
102        EntityExplosionPrimeEvent event = new EntityExplosionPrimeEvent(this, (4.0D + ThreadLocalRandom.current().nextDouble() * 1.5D * root));
103        server.getPluginManager().callEvent(event);
104        if (event.isCancelled()) {
105            return;
106        }
107        Explosion explosion = new Explosion(this, event.getForce(), this);
108        if (event.isBlockBreaking()) {
109            explosion.explodeA();
110        }
111        explosion.explodeB();
112        this.close();
113    }
114
115    @Override
116    public void dropItem() {
117        level.dropItem(this, new ItemMinecartTNT());
118    }
119
120    @Override
121    public String getName() {
122        return getType().getName();
123    }
124
125    @Override
126    public MinecartType getType() {
127        return MinecartType.valueOf(3);
128    }
129
130    @Override
131    public int getNetworkId() {
132        return EntityMinecartTNT.NETWORK_ID;
133    }
134
135    @Override
136    public void saveNBT() {
137        super.saveNBT();
138
139        super.namedTag.putInt("TNTFuse", this.fuse);
140    }
141
142    @Override
143    public boolean onInteract(Player player, Item item, Vector3 clickedPos) {
144        boolean interact = super.onInteract(player, item, clickedPos);
145        if (item.getId() == Item.FLINT_AND_STEEL || item.getId() == Item.FIRE_CHARGE) {
146            level.addLevelSoundEvent(this, LevelSoundEventPacket.SOUND_IGNITE);
147            this.fuse = 79;
148            return true;
149        }
150
151        return interact;
152    }
153
154    @Override
155    public boolean mountEntity(Entity entity, byte mode) {
156        return false;
157    }
158}
