076    @Override
077    protected void initEntity() {
078        super.initEntity();
079
080        this.setMaxHealth(5);
081        this.setHealth(this.namedTag.getShort("Health"));
082
083        if (this.namedTag.contains("Age")) {
084            this.age = this.namedTag.getShort("Age");
085        }
086
087        if (this.namedTag.contains("PickupDelay")) {
088            this.pickupDelay = this.namedTag.getShort("PickupDelay");
089        }
090
091        if (this.namedTag.contains("Owner")) {
092            this.owner = this.namedTag.getString("Owner");
093        }
094
095        if (this.namedTag.contains("Thrower")) {
096            this.thrower = this.namedTag.getString("Thrower");
097        }
098
099        if (!this.namedTag.contains("Item")) {
100            this.close();
101            return;
102        }
103
104        this.item = NBTIO.getItemHelper(this.namedTag.getCompound("Item"));
105        this.setDataFlag(DATA_FLAGS, DATA_FLAG_GRAVITY, true);
106
107        int id = this.item.getId();
108        if (id >= Item.NETHERITE_INGOT && id <= Item.NETHERITE_SCRAP) {
109            this.fireProof = true; // Netherite items are fireproof
110        }
111
112        this.server.getPluginManager().callEvent(new ItemSpawnEvent(this));
113    }

115    @Override
116    public boolean attack(EntityDamageEvent source) {
117        return (source.getCause() == DamageCause.VOID ||
118                source.getCause() == DamageCause.CONTACT ||
119                source.getCause() == DamageCause.FIRE_TICK ||
120                (source.getCause() == DamageCause.ENTITY_EXPLOSION ||
121                source.getCause() == DamageCause.BLOCK_EXPLOSION) &&
122                !this.isInsideOfWater() && (this.item == null ||
123                this.item.getId() != Item.NETHER_STAR)) && super.attack(source);
124    }

126    @Override
127    public boolean onUpdate(int currentTick) {
128        if (this.closed) {
129            return false;
130        }
131
132        int tickDiff = currentTick - this.lastUpdate;
133
134        if (tickDiff <= 0 && !this.justCreated) {
135            return true;
136        }
137
138        this.lastUpdate = currentTick;
139
140        this.timing.startTiming();
141        
142        if (this.age % 60 == 0 && this.onGround && this.getItem() != null && this.isAlive()) {
143            if (this.getItem().getCount() < this.getItem().getMaxStackSize()) {
144                for (Entity entity : this.getLevel().getNearbyEntities(getBoundingBox().grow(1, 1, 1), this, false)) {
145                    if (entity instanceof EntityItem) {
146                        if (!entity.isAlive()) {
147                            continue;
148                        }
149                        Item closeItem = ((EntityItem) entity).getItem();
150                        if (!closeItem.equals(getItem(), true, true)) {
151                            continue;
152                        }
153                        if (!entity.isOnGround()) {
154                            continue;
155                        }
156                        int newAmount = this.getItem().getCount() + closeItem.getCount();
157                        if (newAmount > this.getItem().getMaxStackSize()) {
158                            continue;
159                        }
160                        entity.close();
161                        this.getItem().setCount(newAmount);
162                        EntityEventPacket packet = new EntityEventPacket();
163                        packet.eid = getId();
164                        packet.data = newAmount;
165                        packet.event = EntityEventPacket.MERGE_ITEMS;
166                        Server.broadcastPacket(this.getViewers().values(), packet);
167                    }
168                }
169            }
170        }
171
172        boolean hasUpdate = this.entityBaseTick(tickDiff);
173
174        if (!this.fireProof && this.isInsideOfFire()) {
175            this.kill();
176        }
177
178        if (this.isAlive()) {
179            if (this.pickupDelay > 0 && this.pickupDelay < 32767) {
180                this.pickupDelay -= tickDiff;
181                if (this.pickupDelay < 0) {
182                    this.pickupDelay = 0;
183                }
184            }/* else { // Done in Player#checkNearEntities
185                for (Entity entity : this.level.getNearbyEntities(this.boundingBox.grow(1, 0.5, 1), this)) {
186                    if (entity instanceof Player) {
187                        if (((Player) entity).pickupEntity(this, true)) {
188                            return true;
189                        }
190                    }
191                }
192            }*/
193
194            int bid = level.getBlockIdAt(this.getFloorX(), this.getFloorY(), this.getFloorZ());
195            if (bid == BlockID.WATER || bid == BlockID.STILL_WATER) {
196                this.motionY = this.getGravity() - 0.06;
197            } else if (!this.isOnGround()) {
198                this.motionY -= this.getGravity();
199            }
200
201            if (this.checkObstruction(this.x, this.y, this.z)) {
202                hasUpdate = true;
203            }
204
205            this.move(this.motionX, this.motionY, this.motionZ);
206
207            double friction = 1 - this.getDrag();
208
209            if (this.onGround && (Math.abs(this.motionX) > 0.00001 || Math.abs(this.motionZ) > 0.00001)) {
210                friction *= this.getLevel().getBlock(this.temporalVector.setComponents((int) Math.floor(this.x), (int) Math.floor(this.y - 1), (int) Math.floor(this.z) - 1)).getFrictionFactor();
211            }
212
213            this.motionX *= friction;
214            this.motionY *= 1 - this.getDrag();
215            this.motionZ *= friction;
216
217            if (this.onGround) {
218                this.motionY *= -0.5;
219            }
220
221            this.updateMovement();
222
223            if (this.age > 6000) {
224                ItemDespawnEvent ev = new ItemDespawnEvent(this);
225                this.server.getPluginManager().callEvent(ev);
226                if (ev.isCancelled()) {
227                    this.age = 0;
228                } else {
229                    this.kill();
230                    hasUpdate = true;
231                }
232            }
233        }
234
235        this.timing.stopTiming();
236
237        return hasUpdate || !this.onGround || Math.abs(this.motionX) > 0.00001 || Math.abs(this.motionY) > 0.00001 || Math.abs(this.motionZ) > 0.00001;
238    }

240    @Override
241    public void saveNBT() {
242        super.saveNBT();
243        if (this.item != null) { // Yes, a item can be null... I don't know what causes this, but it can happen.
244            this.namedTag.putCompound("Item", NBTIO.putItemHelper(this.item, -1));
245            this.namedTag.putShort("Health", (int) this.getHealth());
246            this.namedTag.putShort("Age", this.age);
247            this.namedTag.putShort("PickupDelay", this.pickupDelay);
248            if (this.owner != null) {
249                this.namedTag.putString("Owner", this.owner);
250            }
251
252            if (this.thrower != null) {
253                this.namedTag.putString("Thrower", this.thrower);
254            }
255        }
256    }

296    @Override
297    public DataPacket createAddEntityPacket() {
298        AddItemEntityPacket addEntity = new AddItemEntityPacket();
299        addEntity.entityUniqueId = this.getId();
300        addEntity.entityRuntimeId = this.getId();
301        addEntity.x = (float) this.x;
302        addEntity.y = (float) this.y;
303        addEntity.z = (float) this.z;
304        addEntity.speedX = (float) this.motionX;
305        addEntity.speedY = (float) this.motionY;
306        addEntity.speedZ = (float) this.motionZ;
307        addEntity.metadata = this.dataProperties;
308        addEntity.item = this.getItem();
309        return addEntity;
310    }
