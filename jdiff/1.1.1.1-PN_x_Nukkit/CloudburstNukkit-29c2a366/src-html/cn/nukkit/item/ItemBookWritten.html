001package cn.nukkit.item;
002
003import cn.nukkit.nbt.tag.CompoundTag;
004import cn.nukkit.nbt.tag.ListTag;
005
006import java.util.concurrent.ThreadLocalRandom;
007
008public class ItemBookWritten extends Item {
009
010    protected boolean isWritten = false;
011
012    public ItemBookWritten() {
013        this(0, 1);
014    }
015
016    public ItemBookWritten(Integer meta, int count) {
017        super(Item.WRITTEN_BOOK, 0, count, "Book");
018    }
019
020    public Item writeBook(String author, String title, String[] pages) {
021        ListTag<CompoundTag> pageList = new ListTag<>("pages");
022        for (String page : pages) {
023            pageList.add(new CompoundTag().putString("photoname", "").putString("text", page));
024        }
025        return writeBook(author, title, pageList);
026    }
027
028    public Item writeBook(String author, String title, ListTag<CompoundTag> pages) {
029        if (pages.size() > 50 || pages.size() <= 0) return this; //Minecraft does not support more than 50 pages
030        if (this.isWritten) return this; //Book content can only be updated once
031        CompoundTag tag;
032        if (!this.hasCompoundTag()) {
033            tag = new CompoundTag();
034        } else {
035            tag = this.getNamedTag();
036        }
037
038        tag.putString("author", author);
039        tag.putString("title", title);
040        tag.putList(pages);
041
042        tag.putInt("generation", 0);
043        long randomId = 1095216660480L + ThreadLocalRandom.current().nextLong(0L, 2147483647L);
044        tag.putLong("id", randomId);
045
046        this.isWritten = true;
047        return this.setNamedTag(tag);
048    }
049
050    public String getAuthor() {
051        if (!this.isWritten) return "";
052        return this.getNamedTag().getString("author");
053    }
054
055    public String getTitle() {
056        if (!this.isWritten) return "Book";
057        return this.getNamedTag().getString("title");
058    }
059
060    public String[] getPages() {
061        if (!this.isWritten) return new String[0];
062        ListTag<CompoundTag> tag = (ListTag<CompoundTag>) this.getNamedTag().getList("pages");
063        String[] pages = new String[tag.size()];
064        int i = 0;
065        for (CompoundTag pageCompound : tag.getAll()) {
066            pages[i] = pageCompound.getString("text");
067            i++;
068        }
069        return pages;
070    }
071}
