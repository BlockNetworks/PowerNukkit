001package cn.nukkit.item;
002
003import cn.nukkit.Player;
004import cn.nukkit.Server;
005import cn.nukkit.entity.projectile.EntityThrownTrident;
006import cn.nukkit.event.entity.EntityShootBowEvent;
007import cn.nukkit.event.entity.ProjectileLaunchEvent;
008import cn.nukkit.item.enchantment.Enchantment;
009import cn.nukkit.math.Vector3;
010import cn.nukkit.nbt.tag.CompoundTag;
011import cn.nukkit.nbt.tag.DoubleTag;
012import cn.nukkit.nbt.tag.FloatTag;
013import cn.nukkit.nbt.tag.ListTag;
014import cn.nukkit.network.protocol.LevelSoundEventPacket;
015
016/**
017 * Created by PetteriM1
018 */
019public class ItemTrident extends ItemTool {
020
021    public ItemTrident() {
022        this(0, 1);
023    }
024
025    public ItemTrident(Integer meta) {
026        this(meta, 1);
027    }
028
029    public ItemTrident(Integer meta, int count) {
030        super(TRIDENT, meta, count, "Trident");
031    }
032
033    @Override
034    public int getMaxDurability() {
035        return ItemTool.DURABILITY_TRIDENT;
036    }
037
038    @Override
039    public boolean isSword() {
040        return true;
041    }
042
043    @Override
044    public int getAttackDamage() {
045        return 9;
046    }
047
048    @Override
049    public boolean onClickAir(Player player, Vector3 directionVector) {
050        return true;
051    }
052
053    @Override
054    public boolean onRelease(Player player, int ticksUsed) {
055        if (this.hasEnchantment(Enchantment.ID_TRIDENT_RIPTIDE)) {
056            return true;
057        }
058
059        this.useOn(player);
060
061        CompoundTag nbt = new CompoundTag()
062                .putList(new ListTag<DoubleTag>("Pos")
063                        .add(new DoubleTag("", player.x))
064                        .add(new DoubleTag("", player.y + player.getEyeHeight()))
065                        .add(new DoubleTag("", player.z)))
066                .putList(new ListTag<DoubleTag>("Motion")
067                        .add(new DoubleTag("", -Math.sin(player.yaw / 180 * Math.PI) * Math.cos(player.pitch / 180 * Math.PI)))
068                        .add(new DoubleTag("", -Math.sin(player.pitch / 180 * Math.PI)))
069                        .add(new DoubleTag("", Math.cos(player.yaw / 180 * Math.PI) * Math.cos(player.pitch / 180 * Math.PI))))
070                .putList(new ListTag<FloatTag>("Rotation")
071                        .add(new FloatTag("", (player.yaw > 180 ? 360 : 0) - (float) player.yaw))
072                        .add(new FloatTag("", (float) -player.pitch)));
073
074        EntityThrownTrident trident = new EntityThrownTrident(player.chunk, nbt, player);
075        trident.setItem(this);
076
077        double p = (double) ticksUsed / 20;
078        double f = Math.min((p * p + p * 2) / 3, 1) * 2.5;
079
080        EntityShootBowEvent entityShootBowEvent = new EntityShootBowEvent(player, this, trident, f);
081
082        if (f < 0.1 || ticksUsed < 5) {
083            entityShootBowEvent.setCancelled();
084        }
085
086        Server.getInstance().getPluginManager().callEvent(entityShootBowEvent);
087        if (entityShootBowEvent.isCancelled()) {
088            entityShootBowEvent.getProjectile().close();
089        } else {
090            entityShootBowEvent.getProjectile().setMotion(entityShootBowEvent.getProjectile().getMotion().multiply(entityShootBowEvent.getForce()));
091            ProjectileLaunchEvent ev = new ProjectileLaunchEvent(entityShootBowEvent.getProjectile());
092            Server.getInstance().getPluginManager().callEvent(ev);
093            if (ev.isCancelled()) {
094                entityShootBowEvent.getProjectile().close();
095            } else {
096                entityShootBowEvent.getProjectile().spawnToAll();
097                player.getLevel().addLevelSoundEvent(player, LevelSoundEventPacket.SOUND_ITEM_TRIDENT_THROW);
098                if (!player.isCreative()) {
099                    this.count--;
100                    player.getInventory().setItemInHand(this);
101                }
102            }
103        }
104
105        return true;
106    }
107}
