001package cn.nukkit.item;
002
003import cn.nukkit.Player;
004import cn.nukkit.block.Block;
005import cn.nukkit.block.BlockRail;
006import cn.nukkit.entity.Entity;
007import cn.nukkit.entity.item.EntityMinecartTNT;
008import cn.nukkit.level.Level;
009import cn.nukkit.math.BlockFace;
010import cn.nukkit.nbt.tag.CompoundTag;
011import cn.nukkit.nbt.tag.DoubleTag;
012import cn.nukkit.nbt.tag.FloatTag;
013import cn.nukkit.nbt.tag.ListTag;
014import cn.nukkit.utils.Rail;
015
016public class ItemMinecartTNT extends Item {
017
018    public ItemMinecartTNT() {
019        this(0, 1);
020    }
021
022    public ItemMinecartTNT(Integer meta) {
023        this(meta, 1);
024    }
025
026    public ItemMinecartTNT(Integer meta, int count) {
027        super(MINECART_WITH_TNT, meta, count, "Minecart with TNT");
028    }
029
030    @Override
031    public boolean canBeActivated() {
032        return true;
033    }
034
035    @Override
036    public boolean onActivate(Level level, Player player, Block block, Block target, BlockFace face, double fx, double fy, double fz) {
037        if (Rail.isRailBlock(target)) {
038            Rail.Orientation type = ((BlockRail) target).getOrientation();
039            double adjacent = 0.0D;
040            if (type.isAscending()) {
041                adjacent = 0.5D;
042            }
043            EntityMinecartTNT minecart = (EntityMinecartTNT) Entity.createEntity("MinecartTnt",
044                    level.getChunk(target.getFloorX() >> 4, target.getFloorZ() >> 4), new CompoundTag("")
045                    .putList(new ListTag<>("Pos")
046                            .add(new DoubleTag("", target.getX() + 0.5))
047                            .add(new DoubleTag("", target.getY() + 0.0625D + adjacent))
048                            .add(new DoubleTag("", target.getZ() + 0.5)))
049                    .putList(new ListTag<>("Motion")
050                            .add(new DoubleTag("", 0))
051                            .add(new DoubleTag("", 0))
052                            .add(new DoubleTag("", 0)))
053                    .putList(new ListTag<>("Rotation")
054                            .add(new FloatTag("", 0))
055                            .add(new FloatTag("", 0)))
056            );
057
058            if(minecart == null) {
059                return false;
060            }
061
062            if (player.isSurvival()) {
063                Item item = player.getInventory().getItemInHand();
064                item.setCount(item.getCount() - 1);
065                player.getInventory().setItemInHand(item);
066            }
067
068            minecart.spawnToAll();
069            return true;
070        }
071        return false;
072    }
073
074    @Override
075    public int getMaxStackSize() {
076        return 1;
077    }
078}
