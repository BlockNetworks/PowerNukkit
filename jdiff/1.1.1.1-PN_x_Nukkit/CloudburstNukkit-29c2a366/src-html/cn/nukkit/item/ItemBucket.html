001package cn.nukkit.item;
002
003import cn.nukkit.Player;
004import cn.nukkit.block.*;
005import cn.nukkit.event.player.PlayerBucketEmptyEvent;
006import cn.nukkit.event.player.PlayerBucketFillEvent;
007import cn.nukkit.event.player.PlayerItemConsumeEvent;
008import cn.nukkit.level.Level;
009import cn.nukkit.math.BlockFace;
010import cn.nukkit.math.BlockFace.Plane;
011import cn.nukkit.math.Vector3;
012import cn.nukkit.network.protocol.LevelSoundEventPacket;
013import cn.nukkit.network.protocol.UpdateBlockPacket;
014
015/**
016 * author: MagicDroidX
017 * Nukkit Project
018 */
019public class ItemBucket extends Item {
020
021    public ItemBucket() {
022        this(0, 1);
023    }
024
025    public ItemBucket(Integer meta) {
026        this(meta, 1);
027    }
028
029    public ItemBucket(Integer meta, int count) {
030        super(BUCKET, meta, count, getName(meta));
031    }
032
033    protected static String getName(int meta) {
034        switch (meta) {
035            case 1:
036                return "Milk";
037            case 2:
038                return "Bucket of Cod";
039            case 3:
040                return "Bucket of Salmon";
041            case 4:
042                return "Bucket of Tropical Fish";
043            case 5:
044                return "Bucket of Pufferfish";
045            case 8:
046                return "Water Bucket";
047            case 10:
048                return "Lava Bucket";
049            default:
050                return "Bucket";
051        }
052    }
053
054    public static int getDamageByTarget(int target) {
055        switch (target) {
056            case 2:
057            case 3:
058            case 4:
059            case 5:
060            case 8:
061            case 9:
062                return 8;
063            case 10:
064            case 11:
065                return 10;
066            default:
067                return 0;
068        }
069    }
070
071    @Override
072    public int getMaxStackSize() {
073        return this.meta == 0 ? 16 : 1;
074    }
075
076    @Override
077    public boolean canBeActivated() {
078        return true;
079    }
080
081    @Override
082    public boolean onActivate(Level level, Player player, Block block, Block target, BlockFace face, double fx, double fy, double fz) {
083        Block targetBlock = Block.get(getDamageByTarget(this.meta));
084
085        if (targetBlock instanceof BlockAir) {
086            if (target instanceof BlockLiquid && target.getDamage() == 0) {
087                Item result = Item.get(BUCKET, getDamageByTarget(target.getId()), 1);
088                PlayerBucketFillEvent ev;
089                player.getServer().getPluginManager().callEvent(ev = new PlayerBucketFillEvent(player, block, face, this, result));
090                if (!ev.isCancelled()) {
091                    player.getLevel().setBlock(target, new BlockAir(), true, true);
092
093                    // When water is removed ensure any adjacent still water is
094                    // replaced with water that can flow.
095                    for (BlockFace side : Plane.HORIZONTAL) {
096                        Block b = target.getSide(side);
097                        if (b.getId() == STILL_WATER) {
098                            level.setBlock(b, new BlockWater());
099                        }
100                    }
101
102                    if (player.isSurvival()) {
103                        Item clone = this.clone();
104                        clone.setCount(this.getCount() - 1);
105                        player.getInventory().setItemInHand(clone);
106                        player.getInventory().addItem(ev.getItem());
107                    }
108
109                    if (target instanceof BlockLava) {
110                        level.addLevelSoundEvent(block, LevelSoundEventPacket.SOUND_BUCKET_FILL_LAVA);
111                    } else {
112                        level.addLevelSoundEvent(block, LevelSoundEventPacket.SOUND_BUCKET_FILL_WATER);
113                    }
114
115                    return true;
116                } else {
117                    player.getInventory().sendContents(player);
118                }
119            }
120        } else if (targetBlock instanceof BlockLiquid) {
121            Item result = Item.get(BUCKET, 0, 1);
122            PlayerBucketEmptyEvent ev = new PlayerBucketEmptyEvent(player, block, face, this, result);
123            ev.setCancelled(!block.canBeFlowedInto());
124
125            if (player.getLevel().getDimension() == Level.DIMENSION_NETHER && this.getDamage() != 10) {
126                ev.setCancelled(true);
127            }
128
129            player.getServer().getPluginManager().callEvent(ev);
130
131            if (!ev.isCancelled()) {
132                player.getLevel().setBlock(block, targetBlock, true, true);
133                if (player.isSurvival()) {
134                    Item clone = this.clone();
135                    clone.setCount(this.getCount() - 1);
136                    player.getInventory().setItemInHand(clone);
137                    player.getInventory().addItem(ev.getItem());
138                }
139
140                if (this.getDamage() == 10) {
141                    level.addLevelSoundEvent(block, LevelSoundEventPacket.SOUND_BUCKET_EMPTY_LAVA);
142                } else {
143                    level.addLevelSoundEvent(block, LevelSoundEventPacket.SOUND_BUCKET_EMPTY_WATER);
144                }
145
146                return true;
147            } else {
148                player.getLevel().sendBlocks(new Player[]{player}, new Block[]{Block.get(Block.AIR, 0, block)}, UpdateBlockPacket.FLAG_ALL_PRIORITY, 1);
149                player.getInventory().sendContents(player);
150            }
151        }
152
153        return false;
154    }
155
156    @Override
157    public boolean onClickAir(Player player, Vector3 directionVector) {
158        return this.getDamage() == 1; // Milk
159    }
160
161    @Override
162    public boolean onUse(Player player, int ticksUsed) {
163        PlayerItemConsumeEvent consumeEvent = new PlayerItemConsumeEvent(player, this);
164
165        player.getServer().getPluginManager().callEvent(consumeEvent);
166        if (consumeEvent.isCancelled()) {
167            player.getInventory().sendContents(player);
168            return false;
169        }
170
171        if (player.isSurvival()) {
172            this.count--;
173            player.getInventory().setItemInHand(this);
174            player.getInventory().addItem(new ItemBucket());
175        }
176
177        player.removeAllEffects();
178        return true;
179    }
180}
