001package cn.nukkit.item;
002
003import cn.nukkit.Player;
004import cn.nukkit.block.Block;
005import cn.nukkit.entity.Entity;
006import cn.nukkit.entity.item.EntityPainting;
007import cn.nukkit.level.Level;
008import cn.nukkit.level.format.FullChunk;
009import cn.nukkit.math.BlockFace;
010import cn.nukkit.math.Vector3;
011import cn.nukkit.nbt.tag.CompoundTag;
012import cn.nukkit.nbt.tag.DoubleTag;
013import cn.nukkit.nbt.tag.FloatTag;
014import cn.nukkit.nbt.tag.ListTag;
015import cn.nukkit.utils.MainLogger;
016
017import java.util.ArrayList;
018import java.util.List;
019import java.util.concurrent.ThreadLocalRandom;
020
021/**
022 * author: MagicDroidX
023 * Nukkit Project
024 */
025public class ItemPainting extends Item {
026    private static final int[] DIRECTION = {2, 3, 4, 5};
027    private static final int[] RIGHT = {4, 5, 3, 2};
028    private static final double OFFSET = 0.53125;
029
030    public ItemPainting() {
031        this(0, 1);
032    }
033
034    public ItemPainting(Integer meta) {
035        this(meta, 1);
036    }
037
038    public ItemPainting(Integer meta, int count) {
039        super(PAINTING, 0, count, "Painting");
040    }
041
042    @Override
043    public boolean canBeActivated() {
044        return true;
045    }
046
047    @Override
048    public boolean onActivate(Level level, Player player, Block block, Block target, BlockFace face, double fx, double fy, double fz) {
049        FullChunk chunk = level.getChunk((int) block.getX() >> 4, (int) block.getZ() >> 4);
050
051        if (chunk == null || target.isTransparent() || face.getHorizontalIndex() == -1 || block.isSolid()) {
052            return false;
053        }
054
055        List<EntityPainting.Motive> validMotives = new ArrayList<>();
056        for (EntityPainting.Motive motive : EntityPainting.motives) {
057            boolean valid = true;
058            for (int x = 0; x < motive.width && valid; x++) {
059                for (int z = 0; z < motive.height && valid; z++) {
060                    if (target.getSide(BlockFace.fromIndex(RIGHT[face.getIndex() - 2]), x).isTransparent() ||
061                            target.up(z).isTransparent() ||
062                            block.getSide(BlockFace.fromIndex(RIGHT[face.getIndex() - 2]), x).isSolid() ||
063                            block.up(z).isSolid()) {
064                        valid = false;
065                    }
066                }
067            }
068
069            if (valid) {
070                validMotives.add(motive);
071            }
072        }
073        int direction = DIRECTION[face.getIndex() - 2];
074        EntityPainting.Motive motive = validMotives.get(ThreadLocalRandom.current().nextInt(validMotives.size()));
075
076        Vector3 position = new Vector3(target.x + 0.5, target.y + 0.5, target.z + 0.5);
077        double widthOffset = offset(motive.width);
078
079        switch (face.getHorizontalIndex()) {
080            case 0:
081                position.x += widthOffset;
082                position.z += OFFSET;
083                break;
084            case 1:
085                position.x -= OFFSET;
086                position.z += widthOffset;
087                break;
088            case 2:
089                position.x -= widthOffset;
090                position.z -= OFFSET;
091                break;
092            case 3:
093                position.x += OFFSET;
094                position.z -= widthOffset;
095                break;
096        }
097        position.y += offset(motive.height);
098
099        CompoundTag nbt = new CompoundTag()
100                .putByte("Direction", direction)
101                .putString("Motive", motive.title)
102                .putList(new ListTag<DoubleTag>("Pos")
103                        .add(new DoubleTag("0", position.x))
104                        .add(new DoubleTag("1", position.y))
105                        .add(new DoubleTag("2", position.z)))
106                .putList(new ListTag<DoubleTag>("Motion")
107                        .add(new DoubleTag("0", 0))
108                        .add(new DoubleTag("1", 0))
109                        .add(new DoubleTag("2", 0)))
110                .putList(new ListTag<FloatTag>("Rotation")
111                        .add(new FloatTag("0", direction * 90))
112                        .add(new FloatTag("1", 0)));
113
114        EntityPainting entity = (EntityPainting) Entity.createEntity("Painting", chunk, nbt);
115
116        if (entity == null) {
117            return false;
118        }
119
120        if (player.isSurvival()) {
121            Item item = player.getInventory().getItemInHand();
122            item.setCount(item.getCount() - 1);
123            player.getInventory().setItemInHand(item);
124        }
125
126        entity.spawnToAll();
127        return true;
128    }
129
130    private static double offset(int value) {
131        return value > 1 ? 0.5 : 0;
132    }
133}
