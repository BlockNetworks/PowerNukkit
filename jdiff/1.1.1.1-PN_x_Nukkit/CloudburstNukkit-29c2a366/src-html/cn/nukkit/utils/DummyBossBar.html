001package cn.nukkit.utils;
002
003import cn.nukkit.Player;
004import cn.nukkit.entity.Attribute;
005import cn.nukkit.entity.Entity;
006import cn.nukkit.entity.data.EntityMetadata;
007import cn.nukkit.entity.mob.EntityCreeper;
008import cn.nukkit.network.protocol.*;
009
010import java.util.concurrent.ThreadLocalRandom;
011
012/**
013 * DummyBossBar
014 * ===============
015 * author: boybook
016 * Nukkit Project
017 * ===============
018 */
019public class DummyBossBar {
020
021    private final Player player;
022    private final long bossBarId;
023
024    private String text;
025    private float length;
026    private BlockColor color;
027
028    private DummyBossBar(Builder builder) {
029        this.player = builder.player;
030        this.bossBarId = builder.bossBarId;
031        this.text = builder.text;
032        this.length = builder.length;
033        this.color = builder.color;
034    }
035
036    public static class Builder {
037        private final Player player;
038        private final long bossBarId;
039
040        private String text = "";
041        private float length = 100;
042        private BlockColor color = null;
043
044        public Builder(Player player) {
045            this.player = player;
046            this.bossBarId = 1095216660480L + ThreadLocalRandom.current().nextLong(0, 0x7fffffffL);
047        }
048
049        public Builder text(String text) {
050            this.text = text;
051            return this;
052        }
053
054        public Builder length(float length) {
055            if (length >= 0 && length <= 100) this.length = length;
056            return this;
057        }
058
059        public Builder color(BlockColor color) {
060            this.color = color;
061            return this;
062        }
063
064        public Builder color(int red, int green, int blue) {
065            return color(new BlockColor(red, green, blue));
066        }
067
068        public DummyBossBar build() {
069            return new DummyBossBar(this);
070        }
071    }
072
073    public Player getPlayer() {
074        return player;
075    }
076
077    public long getBossBarId() {
078        return bossBarId;
079    }
080
081    public String getText() {
082        return text;
083    }
084
085    public void setText(String text) {
086        if (!this.text.equals(text)) {
087            this.text = text;
088            this.updateBossEntityNameTag();
089            this.sendSetBossBarTitle();
090        }
091    }
092
093    public float getLength() {
094        return length;
095    }
096
097    public void setLength(float length) {
098        if (this.length != length) {
099            this.length = length;
100            this.sendAttributes();
101            this.sendSetBossBarLength();
102        }
103    }
104
105    /**
106     * Color is not working in the current version. We are keep waiting for client support.
107     * @param color the boss bar color
108     */
109    public void setColor(BlockColor color) {
110        if (this.color == null || !this.color.equals(color)) {
111            this.color = color;
112            this.sendSetBossBarTexture();
113        }
114    }
115
116    public void setColor(int red, int green, int blue) {
117        this.setColor(new BlockColor(red, green, blue));
118    }
119
120    public int getMixedColor() {
121        return this.color.getRGB();//(this.color.getRed() << 16 | this.color.getGreen() << 8 | this.color.getBlue()) & 0xffffff;
122    }
123
124    public BlockColor getColor() {
125        return this.color;
126    }
127
128    private void createBossEntity() {
129        AddEntityPacket pkAdd = new AddEntityPacket();
130        pkAdd.type = EntityCreeper.NETWORK_ID;
131        pkAdd.entityUniqueId = bossBarId;
132        pkAdd.entityRuntimeId = bossBarId;
133        pkAdd.x = (float) player.x;
134        pkAdd.y = (float) -10; // Below the bedrock
135        pkAdd.z = (float) player.z;
136        pkAdd.speedX = 0;
137        pkAdd.speedY = 0;
138        pkAdd.speedZ = 0;
139        pkAdd.metadata = new EntityMetadata()
140                // Default Metadata tags
141                .putLong(Entity.DATA_FLAGS, 0)
142                .putShort(Entity.DATA_AIR, 400)
143                .putShort(Entity.DATA_MAX_AIR, 400)
144                .putLong(Entity.DATA_LEAD_HOLDER_EID, -1)
145                .putString(Entity.DATA_NAMETAG, text) // Set the entity name
146                .putFloat(Entity.DATA_SCALE, 0); // And make it invisible
147
148        player.dataPacket(pkAdd);
149    }
150
151    private void sendAttributes() {
152        UpdateAttributesPacket pkAttributes = new UpdateAttributesPacket();
153        pkAttributes.entityId = bossBarId;
154        Attribute attr = Attribute.getAttribute(Attribute.MAX_HEALTH);
155        attr.setMaxValue(100); // Max value - We need to change the max value first, or else the "setValue" will return a IllegalArgumentException
156        attr.setValue(length); // Entity health
157        pkAttributes.entries = new Attribute[]{attr};
158        player.dataPacket(pkAttributes);
159    }
160
161    private void sendShowBossBar() {
162        BossEventPacket pkBoss = new BossEventPacket();
163        pkBoss.bossEid = bossBarId;
164        pkBoss.type = BossEventPacket.TYPE_SHOW;
165        pkBoss.title = text;
166        pkBoss.healthPercent = this.length / 100;
167        player.dataPacket(pkBoss);
168    }
169
170    private void sendHideBossBar() {
171        BossEventPacket pkBoss = new BossEventPacket();
172        pkBoss.bossEid = bossBarId;
173        pkBoss.type = BossEventPacket.TYPE_HIDE;
174        player.dataPacket(pkBoss);
175    }
176
177    private void sendSetBossBarTexture() {
178        BossEventPacket pk = new BossEventPacket();
179        pk.bossEid = this.bossBarId;
180        pk.type = BossEventPacket.TYPE_TEXTURE;
181        pk.color = this.getMixedColor();
182        player.dataPacket(pk);
183    }
184
185    private void sendSetBossBarTitle() {
186        BossEventPacket pkBoss = new BossEventPacket();
187        pkBoss.bossEid = bossBarId;
188        pkBoss.type = BossEventPacket.TYPE_TITLE;
189        pkBoss.title = text;
190        pkBoss.healthPercent = this.length / 100;
191        player.dataPacket(pkBoss);
192    }
193
194    private void sendSetBossBarLength() {
195        BossEventPacket pkBoss = new BossEventPacket();
196        pkBoss.bossEid = bossBarId;
197        pkBoss.type = BossEventPacket.TYPE_HEALTH_PERCENT;
198        pkBoss.healthPercent = this.length / 100;
199        player.dataPacket(pkBoss);
200    }
201
202    /**
203     * Don't let the entity go too far from the player, or the BossBar will disappear.
204     * Update boss entity's position when teleport and each 5s.
205     */
206    public void updateBossEntityPosition() {
207        MoveEntityAbsolutePacket pk = new MoveEntityAbsolutePacket();
208        pk.eid = this.bossBarId;
209        pk.x = this.player.x;
210        pk.y = -10;
211        pk.z = this.player.z;
212        pk.headYaw = 0;
213        pk.yaw = 0;
214        pk.pitch = 0;
215        player.dataPacket(pk);
216    }
217
218    private void updateBossEntityNameTag() {
219        SetEntityDataPacket pk = new SetEntityDataPacket();
220        pk.eid = this.bossBarId;
221        pk.metadata = new EntityMetadata().putString(Entity.DATA_NAMETAG, this.text);
222        player.dataPacket(pk);
223    }
224
225    private void removeBossEntity() {
226        RemoveEntityPacket pkRemove = new RemoveEntityPacket();
227        pkRemove.eid = bossBarId;
228        player.dataPacket(pkRemove);
229    }
230
231    public void create() {
232        createBossEntity();
233        sendAttributes();
234        sendShowBossBar();
235        sendSetBossBarLength();
236        if (color != null) this.sendSetBossBarTexture();
237    }
238
239    /**
240     * Once the player has teleported, resend Show BossBar
241     */
242    public void reshow() {
243        updateBossEntityPosition();
244        sendShowBossBar();
245        sendSetBossBarLength();
246    }
247
248    public void destroy() {
249        sendHideBossBar();
250        removeBossEntity();
251    }
252
253}
