052    /**
053     * Called when a client equips a hotbar inventorySlot. This method should not be used by plugins.
054     * This method will call PlayerItemHeldEvent.
055     *
056     * @param slot hotbar slot Number of the hotbar slot to equip.
057     * @return boolean if the equipment change was successful, false if not.
058     */
059    public boolean equipItem(int slot) {
060        if (!isHotbarSlot(slot)) {
061            this.sendContents((Player) this.getHolder());
062            return false;
063        }
064
065        if (this.getHolder() instanceof Player) {
066            Player player = (Player) this.getHolder();
067            PlayerItemHeldEvent ev = new PlayerItemHeldEvent(player, this.getItem(slot), slot);
068            this.getHolder().getLevel().getServer().getPluginManager().callEvent(ev);
069
070            if (ev.isCancelled()) {
071                this.sendContents(this.getViewers());
072                return false;
073            }
074
075            if (player.fishing != null) {
076                if (!(this.getItem(slot).equals(player.fishing.rod))) {
077                    player.stopFishing(false);
078                }
079            }
080        }
081
082        this.setHeldItemIndex(slot, false);
083        return true;
084    }

086    private boolean isHotbarSlot(int slot) {
087        return slot >= 0 && slot <= this.getHotbarSize();
088    }

152    public void sendHeldItem(Player... players) {
153        Item item = this.getItemInHand();
154
155        MobEquipmentPacket pk = new MobEquipmentPacket();
156        pk.item = item;
157        pk.inventorySlot = pk.hotbarSlot = this.getHeldItemIndex();
158
159        for (Player player : players) {
160            pk.eid = this.getHolder().getId();
161            if (player.equals(this.getHolder())) {
162                pk.eid = player.getId();
163                this.sendSlot(this.getHeldItemIndex(), player);
164            }
165
166            player.dataPacket(pk);
167        }
168    }

174    @Override
175    public void onSlotChange(int index, Item before, boolean send) {
176        EntityHuman holder = this.getHolder();
177        if (holder instanceof Player && !((Player) holder).spawned) {
178            return;
179        }
180
181        if (index >= this.getSize()) {
182            this.sendArmorSlot(index, this.getViewers());
183            this.sendArmorSlot(index, this.getHolder().getViewers().values());
184        } else {
185            super.onSlotChange(index, before, send);
186        }
187    }

242    private boolean setItem(int index, Item item, boolean send, boolean ignoreArmorEvents) {
243        if (index < 0 || index >= this.size) {
244            return false;
245        } else if (item.getId() == 0 || item.getCount() <= 0) {
246            return this.clear(index);
247        }
248
249        //Armor change
250        if (!ignoreArmorEvents && index >= this.getSize()) {
251            EntityArmorChangeEvent ev = new EntityArmorChangeEvent(this.getHolder(), this.getItem(index), item, index);
252            Server.getInstance().getPluginManager().callEvent(ev);
253            if (ev.isCancelled() && this.getHolder() != null) {
254                this.sendArmorSlot(index, this.getViewers());
255                return false;
256            }
257            item = ev.getNewItem();
258        } else {
259            EntityInventoryChangeEvent ev = new EntityInventoryChangeEvent(this.getHolder(), this.getItem(index), item, index);
260            Server.getInstance().getPluginManager().callEvent(ev);
261            if (ev.isCancelled()) {
262                this.sendSlot(index, this.getViewers());
263                return false;
264            }
265            item = ev.getNewItem();
266        }
267        Item old = this.getItem(index);
268        this.slots.put(index, item.clone());
269        this.onSlotChange(index, old, send);
270        return true;
271    }

273    @Override
274    public boolean clear(int index, boolean send) {
275        if (this.slots.containsKey(index)) {
276            Item item = new ItemBlock(new BlockAir(), null, 0);
277            Item old = this.slots.get(index);
278            if (index >= this.getSize() && index < this.size) {
279                EntityArmorChangeEvent ev = new EntityArmorChangeEvent(this.getHolder(), old, item, index);
280                Server.getInstance().getPluginManager().callEvent(ev);
281                if (ev.isCancelled()) {
282                    if (index >= this.size) {
283                        this.sendArmorSlot(index, this.getViewers());
284                    } else {
285                        this.sendSlot(index, this.getViewers());
286                    }
287                    return false;
288                }
289                item = ev.getNewItem();
290            } else {
291                EntityInventoryChangeEvent ev = new EntityInventoryChangeEvent(this.getHolder(), old, item, index);
292                Server.getInstance().getPluginManager().callEvent(ev);
293                if (ev.isCancelled()) {
294                    if (index >= this.size) {
295                        this.sendArmorSlot(index, this.getViewers());
296                    } else {
297                        this.sendSlot(index, this.getViewers());
298                    }
299                    return false;
300                }
301                item = ev.getNewItem();
302            }
303
304            if (item.getId() != Item.AIR) {
305                this.slots.put(index, item.clone());
306            } else {
307                this.slots.remove(index);
308            }
309
310            this.onSlotChange(index, old, send);
311        }
312
313        return true;
314    }

338    public void sendArmorContents(Player[] players) {
339        Item[] armor = this.getArmorContents();
340
341        MobArmorEquipmentPacket pk = new MobArmorEquipmentPacket();
342        pk.eid = this.getHolder().getId();
343        pk.slots = armor;
344        pk.encode();
345        pk.isEncoded = true;
346
347        for (Player player : players) {
348            if (player.equals(this.getHolder())) {
349                InventoryContentPacket pk2 = new InventoryContentPacket();
350                pk2.inventoryId = InventoryContentPacket.SPECIAL_ARMOR;
351                pk2.slots = armor;
352                player.dataPacket(pk2);
353            } else {
354                player.dataPacket(pk);
355            }
356        }
357    }

359    public void setArmorContents(Item[] items) {
360        if (items.length < 4) {
361            Item[] newItems = new Item[4];
362            System.arraycopy(items, 0, newItems, 0, items.length);
363            items = newItems;
364        }
365
366        for (int i = 0; i < 4; ++i) {
367            if (items[i] == null) {
368                items[i] = new ItemBlock(new BlockAir(), null, 0);
369            }
370
371            if (items[i].getId() == Item.AIR) {
372                this.clear(this.getSize() + i);
373            } else {
374                this.setItem(this.getSize() + i, items[i]);
375            }
376        }
377    }

423    @Override
424    public void sendContents(Player[] players) {
425        InventoryContentPacket pk = new InventoryContentPacket();
426        pk.slots = new Item[this.getSize()];
427        for (int i = 0; i < this.getSize(); ++i) {
428            pk.slots[i] = this.getItem(i);
429        }
430
431        /*//Because PE is stupid and shows 9 less slots than you send it, give it 9 dummy slots so it shows all the REAL slots.
432        for(int i = this.getSize(); i < this.getSize() + this.getHotbarSize(); ++i){
433            pk.slots[i] = new ItemBlock(new BlockAir());
434        }
435            pk.slots[i] = new ItemBlock(new BlockAir());
436        }*/
437
438        for (Player player : players) {
439            int id = player.getWindowId(this);
440            if (id == -1 || !player.spawned) {
441                if (this.getHolder() != player) this.close(player);
442                continue;
443            }
444            pk.inventoryId = id;
445            player.dataPacket(pk.clone());
446
447        }
448    }

460    @Override
461    public void sendSlot(int index, Player... players) {
462        InventorySlotPacket pk = new InventorySlotPacket();
463        pk.slot = index;
464        pk.item = this.getItem(index).clone();
465
466        for (Player player : players) {
467            if (player.equals(this.getHolder())) {
468                pk.inventoryId = ContainerIds.INVENTORY;
469                player.dataPacket(pk);
470            } else {
471                int id = player.getWindowId(this);
472                if (id == -1) {
473                    this.close(player);
474                    continue;
475                }
476                pk.inventoryId = id;
477                player.dataPacket(pk.clone());
478            }
479        }
480    }

482    public void sendCreativeContents() {
483        if (!(this.getHolder() instanceof Player)) {
484            return;
485        }
486        Player p = (Player) this.getHolder();
487
488        InventoryContentPacket pk = new InventoryContentPacket();
489        pk.inventoryId = ContainerIds.CREATIVE;
490
491        if (!p.isSpectator()) { //fill it for all gamemodes except spectator
492            pk.slots = Item.getCreativeItems().toArray(new Item[0]);
493        }
494
495        p.dataPacket(pk);
496    }
