001package cn.nukkit.inventory;
002
003import cn.nukkit.Player;
004import cn.nukkit.network.protocol.InventorySlotPacket;
005import cn.nukkit.network.protocol.types.ContainerIds;
006
007import java.util.HashMap;
008
009public class PlayerUIInventory extends BaseInventory {
010    private final Player player;
011
012    private final PlayerCursorInventory cursorInventory;
013    private final CraftingGrid craftingGrid;
014    private final BigCraftingGrid bigCraftingGrid;
015
016    public PlayerUIInventory(Player player) {
017        super(player, InventoryType.UI, new HashMap<>(), 51);
018        this.player = player;
019
020        this.cursorInventory = new PlayerCursorInventory(this);
021        this.craftingGrid = new CraftingGrid(this);
022        this.bigCraftingGrid = new BigCraftingGrid(this);
023    }
024
025    public PlayerCursorInventory getCursorInventory() {
026        return cursorInventory;
027    }
028
029    public CraftingGrid getCraftingGrid() {
030        return craftingGrid;
031    }
032
033    public BigCraftingGrid getBigCraftingGrid() {
034        return bigCraftingGrid;
035    }
036
037    @Override
038    public void setSize(int size) {
039        throw new UnsupportedOperationException("UI size is immutable");
040    }
041
042    @Override
043    public void sendSlot(int index, Player... target) {
044        InventorySlotPacket pk = new InventorySlotPacket();
045        pk.slot = index;
046        pk.item = this.getItem(index);
047
048        for (Player p : target) {
049            if (p == this.getHolder()) {
050                pk.inventoryId = ContainerIds.UI;
051            } else {
052                int id;
053
054                if ((id = p.getWindowId(this)) == ContainerIds.NONE) {
055                    this.close(p);
056                    continue;
057                }
058                pk.inventoryId = id;
059            }
060            p.dataPacket(pk);
061        }
062    }
063
064    @Override
065    public void sendContents(Player... target) {
066        //doesn't work here
067    }
068
069    @Override
070    public int getSize() {
071        return 51;
072    }
073
074    @Override
075    public Player getHolder() {
076        return player;
077    }
078}
