001package cn.nukkit.inventory;
002
003import cn.nukkit.Player;
004import cn.nukkit.item.Item;
005
006import java.util.Collections;
007import java.util.Map;
008import java.util.Set;
009
010public class PlayerUIComponent extends BaseInventory {
011
012    public static final int CREATED_ITEM_OUTPUT_UI_SLOT = 50;
013
014    private final PlayerUIInventory playerUI;
015    private final int offset;
016    private final int size;
017
018    PlayerUIComponent(PlayerUIInventory playerUI, int offset, int size) {
019        super(playerUI.holder, InventoryType.UI, Collections.emptyMap(), size);
020        this.playerUI = playerUI;
021        this.offset = offset;
022        this.size = size;
023    }
024
025    @Override
026    public int getSize() {
027        return size;
028    }
029
030    @Override
031    public int getMaxStackSize() {
032        return 64;
033    }
034
035    @Override
036    public void setMaxStackSize(int size) {
037        throw new UnsupportedOperationException();
038    }
039
040
041    @Override
042    public String getTitle() {
043        throw new UnsupportedOperationException();
044    }
045
046    @Override
047    public Item getItem(int index) {
048        return this.playerUI.getItem(index + this.offset);
049    }
050
051    @Override
052    public boolean setItem(int index, Item item, boolean send) {
053        return this.playerUI.setItem(index + this.offset, item, send);
054    }
055
056    @Override
057    public Map<Integer, Item> getContents() {
058        Map<Integer, Item> contents = playerUI.getContents();
059        contents.keySet().removeIf(slot -> slot < offset || slot > offset + size);
060        return contents;
061    }
062
063
064    @Override
065    public void sendContents(Player... players) {
066        this.playerUI.sendContents(players);
067    }
068
069    @Override
070    public void sendSlot(int index, Player... players) {
071        playerUI.sendSlot(index + this.offset, players);
072    }
073
074    @Override
075    public Set<Player> getViewers() {
076        return playerUI.viewers;
077    }
078
079    @Override
080    public InventoryType getType() {
081        return playerUI.type;
082    }
083
084    @Override
085    public void onOpen(Player who) {
086
087    }
088
089    @Override
090    public boolean open(Player who) {
091        return false;
092    }
093
094    @Override
095    public void close(Player who) {
096
097    }
098
099    @Override
100    public void onClose(Player who) {
101
102    }
103
104    @Override
105    public void onSlotChange(int index, Item before, boolean send) {
106        this.playerUI.onSlotChange(index + this.offset, before, send);
107    }
108}
