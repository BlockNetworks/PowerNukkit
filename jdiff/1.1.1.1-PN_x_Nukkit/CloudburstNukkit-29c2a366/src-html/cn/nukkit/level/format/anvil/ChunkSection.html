051    public ChunkSection(CompoundTag nbt) {
052        this.y = nbt.getByte("Y");
053
054        byte[] blocks = nbt.getByteArray("Blocks");
055        NibbleArray data = new NibbleArray(nbt.getByteArray("Data"));
056
057        storage = new BlockStorage();
058
059        // Convert YZX to XZY
060        for (int x = 0; x < 16; x++) {
061            for (int z = 0; z < 16; z++) {
062                for (int y = 0; y < 16; y++) {
063                    int index = getAnvilIndex(x, y, z);
064                    storage.setBlockId(x, y, z, blocks[index]);
065                    storage.setBlockData(x, y, z, data.get(index));
066                }
067            }
068        }
069
070        this.blockLight = nbt.getByteArray("BlockLight");
071        this.skyLight = nbt.getByteArray("SkyLight");
072    }
073
074    private static int getAnvilIndex(int x, int y, int z) {
075        return (y << 8) + (z << 4) + x;
076    }

140    @Override
141    public boolean setBlock(int x, int y, int z, int blockId, int meta) {
142        int newFullId = (blockId << 4) + meta;
143        synchronized (storage) {
144            int previousFullId = storage.getAndSetFullBlock(x, y, z, newFullId);

149    @Override
150    public int getBlockSkyLight(int x, int y, int z) {
151        if (this.skyLight == null) {
152            if (!hasSkyLight) {
153                return 0;
154            } else if (compressedLight == null) {
155                return 15;
156            }
157        }
158        this.skyLight = getSkyLightArray();
159        int sl = this.skyLight[(y << 7) | (z << 3) | (x >> 1)] & 0xff;
160        if ((x & 1) == 0) {
161            return sl & 0x0f;
162        }
163        return sl >> 4;
164    }
165
166    @Override
167    public void setBlockSkyLight(int x, int y, int z, int level) {
168        if (this.skyLight == null) {
169            if (hasSkyLight && compressedLight != null) {
170                this.skyLight = getSkyLightArray();
171            } else if (level == (hasSkyLight ? 15 : 0)) {
172                return;
173            } else {
174                this.skyLight = new byte[2048];
175                if (hasSkyLight) {
176                    Arrays.fill(this.skyLight, (byte) 0xFF);
177                }
178            }
179        }
180        int i = (y << 7) | (z << 3) | (x >> 1);
181        int old = this.skyLight[i] & 0xff;
182        if ((x & 1) == 0) {
183            this.skyLight[i] = (byte) ((old & 0xf0) | (level & 0x0f));
184        } else {
185            this.skyLight[i] = (byte) (((level & 0x0f) << 4) | (old & 0x0f));
186        }
187    }

189    @Override
190    public int getBlockLight(int x, int y, int z) {
191        if (blockLight == null && !hasBlockLight) return 0;
192        this.blockLight = getLightArray();
193        int l = blockLight[(y << 7) | (z << 3) | (x >> 1)] & 0xff;
194        if ((x & 1) == 0) {
195            return l & 0x0f;
196        }
197        return l >> 4;
198    }

200    @Override
201    public void setBlockLight(int x, int y, int z, int level) {
202        if (this.blockLight == null) {
203            if (hasBlockLight) {
204                this.blockLight = getLightArray();
205            } else if (level == 0) {
206                return;
207            } else {
208                this.blockLight = new byte[2048];
209            }
210        }
211        int i = (y << 7) | (z << 3) | (x >> 1);
212        int old = this.blockLight[i] & 0xff;
213        if ((x & 1) == 0) {
214            this.blockLight[i] = (byte) ((old & 0xf0) | (level & 0x0f));
215        } else {
216            this.blockLight[i] = (byte) (((level & 0x0f) << 4) | (old & 0x0f));
217        }
218    }

222        synchronized (storage) {
223            byte[] anvil = new byte[4096];
224            for (int x = 0; x < 16; x++) {
225                for (int z = 0; z < 16; z++) {
226                    for (int y = 0; y < 16; y++) {
227                        int index = getAnvilIndex(x, y, z);
228                        anvil[index] = (byte) storage.getBlockId(x, y, z);

230                }

232            return anvil;
233        }

241                for (int z = 0; z < 16; z++) {
242                    for (int y = 0; y < 16; y++) {
243                        int index = getAnvilIndex(x, y, z);

245                    }
246                }
247            }
248            return anvil.getData();

252    @Override
253    public byte[] getSkyLightArray() {
254        if (this.skyLight != null) return skyLight;
255        if (hasSkyLight) {
256            if (compressedLight != null) {
257                inflate();
258                return this.skyLight;
259            }
260            return EmptyChunkSection.EMPTY_SKY_LIGHT_ARR;
261        } else {
262            return EmptyChunkSection.EMPTY_LIGHT_ARR;
263        }
264    }

266    private void inflate() {
267        try {
268            if (compressedLight != null && compressedLight.length != 0) {
269                byte[] inflated = Zlib.inflate(compressedLight);
270                blockLight = Arrays.copyOfRange(inflated, 0, 2048);
271                if (inflated.length > 2048) {
272                    skyLight = Arrays.copyOfRange(inflated, 2048, 4096);
273                } else {
274                    skyLight = new byte[2048];
275                    if (hasSkyLight) {
276                        Arrays.fill(skyLight, (byte) 0xFF);
277                    }
278                }
279                compressedLight = null;
280            } else {
281                blockLight = new byte[2048];
282                skyLight = new byte[2048];
283                if (hasSkyLight) {
284                    Arrays.fill(skyLight, (byte) 0xFF);
285                }
286            }
287        } catch (IOException e) {
288            e.printStackTrace();
289        }
290    }

294        if (this.blockLight != null) return blockLight;
295        if (hasBlockLight) {
296            inflate();
297            return this.blockLight;
298        } else {
299            return EmptyChunkSection.EMPTY_LIGHT_ARR;

308    private byte[] toXZY(char[] raw) {
309        byte[] buffer = ThreadCache.byteCache6144.get();
310        for (int i = 0; i < 4096; i++) {
311            buffer[i] = (byte) (raw[i] >> 4);
312        }
313        for (int i = 0, j = 4096; i < 4096; i += 2, j++) {
314            buffer[j] = (byte) (((raw[i + 1] & 0xF) << 4) | (raw[i] & 0xF));
315        }
316        return buffer;
317    }

319    @Override
320    public byte[] getBytes() {
321        synchronized (storage) {
322
323            byte[] ids = storage.getBlockIds();
324            byte[] data = storage.getBlockData();
325            byte[] merged = new byte[ids.length + data.length];
326
327            System.arraycopy(ids, 0, merged, 0, ids.length);
328            System.arraycopy(data, 0, merged, ids.length, data.length);
329            return merged;
330        }
331    }

333    public boolean compress() {
334        if (blockLight != null) {
335            byte[] arr1 = blockLight;
336            hasBlockLight = !Utils.isByteArrayEmpty(arr1);
337            byte[] arr2;
338            if (skyLight != null) {
339                arr2 = skyLight;
340                hasSkyLight = !Utils.isByteArrayEmpty(arr2);
341            } else if (hasSkyLight) {
342                arr2 = EmptyChunkSection.EMPTY_SKY_LIGHT_ARR;
343            } else {
344                arr2 = EmptyChunkSection.EMPTY_LIGHT_ARR;
345                hasSkyLight = false;
346            }
347            blockLight = null;
348            skyLight = null;
349            byte[] toDeflate = null;
350            if (hasBlockLight && hasSkyLight && arr2 != EmptyChunkSection.EMPTY_SKY_LIGHT_ARR) {
351                toDeflate = Binary.appendBytes(arr1, arr2);
352            } else if (hasBlockLight) {
353                toDeflate = arr1;
354            }
355            if (toDeflate != null) {
356                try {
357                    compressedLight = Zlib.deflate(toDeflate, 1);
358                } catch (Exception e) {
359                    e.printStackTrace();
360                }
361            }
362            return true;
363        }
364        return false;
365    }

368        return new ChunkSection(

370                this.storage.copy(),
371                this.blockLight == null ? null : this.blockLight.clone(),
372                this.skyLight == null ? null : this.skyLight.clone(),
373                this.compressedLight == null ? null : this.compressedLight.clone(),
374                this.hasBlockLight,
375                this.hasSkyLight
376        );
