001package cn.nukkit.level.generator.populator.impl;
002
003import cn.nukkit.level.ChunkManager;
004import cn.nukkit.level.format.FullChunk;
005import cn.nukkit.level.generator.object.ore.OreType;
006import cn.nukkit.level.generator.populator.type.Populator;
007import cn.nukkit.math.NukkitMath;
008import cn.nukkit.math.NukkitRandom;
009
010/**
011 * @author DaPorkchop_
012 */
013public class PopulatorOre extends Populator {
014    private final int replaceId;
015    private final OreType[] oreTypes;
016
017    public PopulatorOre(int replaceId, OreType[] oreTypes) {
018        this.replaceId = replaceId;
019        this.oreTypes = oreTypes;
020    }
021
022    @Override
023    public void populate(ChunkManager level, int chunkX, int chunkZ, NukkitRandom random, FullChunk chunk) {
024        int sx = chunkX << 4;
025        int ex = sx + 15;
026        int sz = chunkZ << 4;
027        int ez = sz + 15;
028        for (OreType type : this.oreTypes) {
029            for (int i = 0; i < type.clusterCount; i++) {
030                int x = NukkitMath.randomRange(random, sx, ex);
031                int z = NukkitMath.randomRange(random, sz, ez);
032                int y = NukkitMath.randomRange(random, type.minHeight, type.maxHeight);
033                if (level.getBlockIdAt(x, y, z) != replaceId) {
034                    continue;
035                }
036                type.spawn(level, random, replaceId, x, y, z);
037            }
038        }
039    }
040}
