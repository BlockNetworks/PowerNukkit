001package cn.nukkit.block;
002
003import cn.nukkit.Player;
004import cn.nukkit.event.level.StructureGrowEvent;
005import cn.nukkit.item.Item;
006import cn.nukkit.level.Level;
007import cn.nukkit.level.ListChunkManager;
008import cn.nukkit.level.generator.object.mushroom.BigMushroom;
009import cn.nukkit.level.particle.BoneMealParticle;
010import cn.nukkit.math.BlockFace;
011import cn.nukkit.math.NukkitRandom;
012import cn.nukkit.utils.BlockColor;
013import cn.nukkit.utils.DyeColor;
014
015import java.util.concurrent.ThreadLocalRandom;
016
017public abstract class BlockMushroom extends BlockFlowable {
018
019    public BlockMushroom() {
020        this(0);
021    }
022
023    public BlockMushroom(int meta) {
024        super(0);
025    }
026
027    @Override
028    public int onUpdate(int type) {
029        if (type == Level.BLOCK_UPDATE_NORMAL) {
030            if (!canStay()) {
031                getLevel().useBreakOn(this);
032
033                return Level.BLOCK_UPDATE_NORMAL;
034            }
035        }
036        return 0;
037    }
038
039    @Override
040    public boolean place(Item item, Block block, Block target, BlockFace face, double fx, double fy, double fz, Player player) {
041        if (canStay()) {
042            getLevel().setBlock(block, this, true, true);
043            return true;
044        }
045        return false;
046    }
047
048    @Override
049    public boolean canBeActivated() {
050        return true;
051    }
052
053    @Override
054    public boolean onActivate(Item item, Player player) {
055        if (item.getId() == Item.DYE && item.getDamage() == DyeColor.WHITE.getDyeData()) {
056            if (player != null && (player.gamemode & 0x01) == 0) {
057                item.count--;
058            }
059
060            if (ThreadLocalRandom.current().nextFloat() < 0.4) {
061                this.grow();
062            }
063
064            this.level.addParticle(new BoneMealParticle(this));
065            return true;
066        }
067        return false;
068    }
069
070    public boolean grow() {
071        this.level.setBlock(this, Block.get(BlockID.AIR), true, false);
072
073        BigMushroom generator = new BigMushroom(getType());
074
075        ListChunkManager chunkManager = new ListChunkManager(this.level);
076        if (generator.generate(chunkManager, new NukkitRandom(), this)) {
077            StructureGrowEvent ev = new StructureGrowEvent(this, chunkManager.getBlocks());
078            this.level.getServer().getPluginManager().callEvent(ev);
079            if (ev.isCancelled()) {
080                return false;
081            }
082            for(Block block : ev.getBlockList()) {
083                this.level.setBlockAt(block.getFloorX(), block.getFloorY(), block.getFloorZ(), block.getId(), block.getDamage());
084            }
085            return true;
086        } else {
087            this.level.setBlock(this, this, true, false);
088            return false;
089        }
090    }
091
092    public boolean canStay() {
093        Block block = this.down();
094        return block.getId() == MYCELIUM || block.getId() == PODZOL || (!block.isTransparent() && this.level.getFullLight(this) < 13);
095    }
096
097    @Override
098    public BlockColor getColor() {
099        return BlockColor.FOLIAGE_BLOCK_COLOR;
100    }
101
102    @Override
103    public boolean canSilkTouch() {
104        return true;
105    }
106
107    protected abstract int getType();
108}
