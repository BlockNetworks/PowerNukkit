031

035    }

040    }
041

043    public boolean canPassThrough() {
044        return true;
045    }
046
047    @Override
048    public boolean isBreakable(Item item) {
049        return false;
050    }
051
052    @Override

054        return -1;

056

062    @Override
063    public Item toItem() {
064        return new ItemBlock(Block.get(BlockID.AIR));
065    }
066
067    @Override
068    public boolean onBreak(Item item) {
069        boolean result = super.onBreak(item);
070        for (BlockFace face : BlockFace.values()) {
071            Block b = this.getSide(face);
072            if (b != null) {
073                if (b instanceof BlockNetherPortal) {
074                    result &= b.onBreak(item);
075                }
076            }
077        }
078        return result;
079    }

095
096    @Override

098        return false;
099    }
100
101    @Override
102    protected AxisAlignedBB recalculateBoundingBox() {
103        return this;
104    }

106    public static void spawnPortal(Position pos)   {
107        Level lvl = pos.level;
108        int x = pos.getFloorX(), y = pos.getFloorY(), z = pos.getFloorZ();
109
110        for (int xx = -1; xx < 4; xx++) {
111            for (int yy = 1; yy < 4; yy++)  {
112                for (int zz = -1; zz < 3; zz++) {
113                    lvl.setBlockAt(x + xx, y + yy, z + zz, AIR);
114                }
115            }
116        }
117
118        lvl.setBlockAt(x + 1, y, z, OBSIDIAN);
119        lvl.setBlockAt(x + 2, y, z, OBSIDIAN);
120
121        z += 1;
122        lvl.setBlockAt(x, y, z, OBSIDIAN);
123        lvl.setBlockAt(x + 1, y, z, OBSIDIAN);
124        lvl.setBlockAt(x + 2, y, z, OBSIDIAN);
125        lvl.setBlockAt(x + 3, y, z, OBSIDIAN);
126
127        z += 1;
128        lvl.setBlockAt(x + 1, y, z, OBSIDIAN);
129        lvl.setBlockAt(x + 2, y, z, OBSIDIAN);
130
131        z -= 1;
132        y += 1;
133        lvl.setBlockAt(x, y, z, OBSIDIAN);
134        lvl.setBlockAt(x + 1, y, z, NETHER_PORTAL);
135        lvl.setBlockAt(x + 2, y, z, NETHER_PORTAL);
136        lvl.setBlockAt(x + 3, y, z, OBSIDIAN);
137
138        y += 1;
139        lvl.setBlockAt(x, y, z, OBSIDIAN);
140        lvl.setBlockAt(x + 1, y, z, NETHER_PORTAL);
141        lvl.setBlockAt(x + 2, y, z, NETHER_PORTAL);
142        lvl.setBlockAt(x + 3, y, z, OBSIDIAN);
143
144        y += 1;
145        lvl.setBlockAt(x, y, z, OBSIDIAN);
146        lvl.setBlockAt(x + 1, y, z, NETHER_PORTAL);
147        lvl.setBlockAt(x + 2, y, z, NETHER_PORTAL);
148        lvl.setBlockAt(x + 3, y, z, OBSIDIAN);
149
150        y += 1;
151        lvl.setBlockAt(x, y, z, OBSIDIAN);
152        lvl.setBlockAt(x + 1, y, z, OBSIDIAN);
153        lvl.setBlockAt(x + 2, y, z, OBSIDIAN);
154        lvl.setBlockAt(x + 3, y, z, OBSIDIAN);
155    }
