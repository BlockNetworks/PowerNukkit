017 */
018public class BlockEndPortalFrame extends BlockTransparentMeta implements Faceable {
019
020    private static final int[] FACES = {2, 3, 0, 1};
021
022    public BlockEndPortalFrame() {
023        this(0);
024    }
025
026    public BlockEndPortalFrame(int meta) {
027        super(meta);
028    }
029
030    @Override
031    public int getId() {
032        return END_PORTAL_FRAME;
033    }
034
035    @Override
036    public double getResistance() {
037        return 18000000;
038    }
039
040    @Override
041    public double getHardness() {
042        return -1;
043    }
044
045    @Override
046    public int getLightLevel() {
047        return 1;
048    }
049
050    @Override
051    public String getName() {
052        return "End Portal Frame";
053    }
054
055    @Override
056    public boolean isBreakable(Item item) {
057        return false;
058    }
059
060    @Override
061    public double getMaxY() {
062        return this.y + ((this.getDamage() & 0x04) > 0 ? 1 : 0.8125);
063    }
064
065    @Override
066    public boolean canBePushed() {
067        return false;
068    }
069
070    public boolean hasComparatorInputOverride() {
071        return true;
072    }
073
074    public int getComparatorInputOverride() {
075        return (getDamage() & 4) != 0 ? 15 : 0;
076    }
077
078    @Override
079    public boolean canBeActivated() {
080        return true;
081    }
082
083    @Override
084    public boolean onActivate(Item item, Player player) {
085        if((this.getDamage() & 0x04) == 0 && player != null && item.getId() == Item.ENDER_EYE) {
086            this.setDamage(this.getDamage() + 4);
087            this.getLevel().setBlock(this, this, true, true);
088            this.getLevel().addLevelSoundEvent(this, LevelSoundEventPacket.SOUND_BLOCK_END_PORTAL_FRAME_FILL);
089            this.createPortal();
090            return true;
091        }
092        return false;
093    }
094
095    public void createPortal() {
096        Vector3 centerSpot = this.searchCenter(new ArrayList<>());
097        if(centerSpot != null) {
098            for(int x = -2; x <= 2; x++) {
099                for(int z = -2; z <= 2; z++) {
100                    if((x == -2 || x == 2) && (z == -2 || z == 2))
101                        continue;
102                    if(x == -2 || x == 2 || z == -2 || z == 2) {
103                        if(!this.checkFrame(this.getLevel().getBlock(centerSpot.add(x, 0, z)), x, z)) {
104                            return;
105                        }
106                    }
107                }
108            }
109
110            for(int x = -1; x <= 1; x++) {
111                for(int z = -1; z <= 1; z++) {
112                    Vector3 vector3 = centerSpot.add(x, 0, z);
113                    if(this.getLevel().getBlock(vector3).getId() != Block.AIR) {
114                        this.getLevel().useBreakOn(vector3);
115                    }
116                    this.getLevel().setBlock(vector3, Block.get(Block.END_PORTAL));
117                }
118            }
119        }
120    }
121
122    private Vector3 searchCenter(List<Block> visited) {
123        for(int x = -2; x <= 2; x++) {
124            if(x == 0)
125                continue;
126            Block block = this.getLevel().getBlock(this.add(x, 0, 0));
127            Block iBlock = this.getLevel().getBlock(this.add(x * 2, 0, 0));
128            if(this.checkFrame(block) && !visited.contains(block)) {
129                visited.add(block);
130                if((x == -1 || x == 1) && this.checkFrame(iBlock))
131                    return ((BlockEndPortalFrame) block).searchCenter(visited);
132                for(int z = -4; z <= 4; z++) {
133                    if(z == 0)
134                        continue;
135                    block = this.getLevel().getBlock(this.add(x, 0, z));
136                    if(this.checkFrame(block)) {
137                        return this.add(x / 2, 0, z / 2);
138                    }
139                }
140            }
141        }
142        for(int z = -2; z <= 2; z++) {
143            if(z == 0)
144                continue;
145            Block block = this.getLevel().getBlock(this.add(0, 0, z));
146            Block iBlock = this.getLevel().getBlock(this.add(0, 0, z * 2));
147            if(this.checkFrame(block) && !visited.contains(block)) {
148                visited.add(block);
149                if((z == -1 || z == 1) && this.checkFrame(iBlock))
150                    return ((BlockEndPortalFrame) block).searchCenter(visited);
151                for(int x = -4; x <= 4; x++) {
152                    if(x == 0)
153                        continue;
154                    block = this.getLevel().getBlock(this.add(x, 0, z));
155                    if(this.checkFrame(block)) {
156                        return this.add(x / 2, 0, z / 2);
157                    }
158                }
159            }
160        }
161        return null;
162    }
163
164    private boolean checkFrame(Block block) {
165        return block.getId() == this.getId() && (block.getDamage() & 4) == 4;
166    }
167
168    private boolean checkFrame(Block block, int x, int z) {
169        return block.getId() == this.getId() && (block.getDamage() - 4) == (x == -2 ? 3 : x == 2 ? 1 : z == -2 ? 0 : z == 2 ? 2 : -1);
170    }
171

173    public boolean canHarvestWithHand() {
174        return false;
175    }
176
177    @Override
178    public Item toItem() {
179        return new ItemBlock(this, 0);
180    }
181
182    @Override
183    public BlockFace getBlockFace() {
184        return BlockFace.fromHorizontalIndex(this.getDamage() & 0x07);
185    }
186
187    @Override
188    public boolean place(Item item, Block block, Block target, BlockFace face, double fx, double fy, double fz, Player player) {
189        this.setDamage(FACES[player != null ? player.getDirection().getHorizontalIndex() : 0]);
190        this.getLevel().setBlock(block, this, true);
191        return true;
192    }
193
194    @Override
195    public BlockColor getColor() {
196        return BlockColor.GREEN_BLOCK_COLOR;
197    }
198}
