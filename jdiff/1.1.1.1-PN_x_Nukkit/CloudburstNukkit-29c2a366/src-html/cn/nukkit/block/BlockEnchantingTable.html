001package cn.nukkit.block;
002
003import cn.nukkit.Player;
004import cn.nukkit.blockentity.BlockEntity;
005import cn.nukkit.blockentity.BlockEntityEnchantTable;
006import cn.nukkit.inventory.EnchantInventory;
007import cn.nukkit.item.Item;
008import cn.nukkit.item.ItemTool;
009import cn.nukkit.math.BlockFace;
010import cn.nukkit.nbt.tag.CompoundTag;
011import cn.nukkit.nbt.tag.ListTag;
012import cn.nukkit.nbt.tag.StringTag;
013import cn.nukkit.nbt.tag.Tag;
014import cn.nukkit.utils.BlockColor;
015
016import java.util.Map;
017
018/**
019 * Created on 2015/11/22 by CreeperFace.
020 * Package cn.nukkit.block in project Nukkit .
021 */
022public class BlockEnchantingTable extends BlockTransparent {
023    public BlockEnchantingTable() {
024    }
025
026    @Override
027    public int getId() {
028        return ENCHANTING_TABLE;
029    }
030
031    @Override
032    public String getName() {
033        return "Enchanting Table";
034    }
035
036    @Override
037    public int getToolType() {
038        return ItemTool.TYPE_PICKAXE;
039    }
040
041    @Override
042    public double getHardness() {
043        return 5;
044    }
045
046    @Override
047    public double getResistance() {
048        return 6000;
049    }
050
051    @Override
052    public int getLightLevel() {
053        return 12;
054    }
055
056    @Override
057    public boolean canBeActivated() {
058        return true;
059    }
060
061    @Override
062    public Item[] getDrops(Item item) {
063        if (item.isPickaxe() && item.getTier() >= ItemTool.TIER_WOODEN) {
064            return new Item[]{
065                    toItem()
066            };
067        } else {
068            return new Item[0];
069        }
070    }
071
072    @Override
073    public boolean place(Item item, Block block, Block target, BlockFace face, double fx, double fy, double fz, Player player) {
074        this.getLevel().setBlock(block, this, true, true);
075
076        CompoundTag nbt = new CompoundTag()
077                .putString("id", BlockEntity.ENCHANT_TABLE)
078                .putInt("x", (int) this.x)
079                .putInt("y", (int) this.y)
080                .putInt("z", (int) this.z);
081
082        if (item.hasCustomName()) {
083            nbt.putString("CustomName", item.getCustomName());
084        }
085
086        if (item.hasCustomBlockData()) {
087            Map<String, Tag> customData = item.getCustomBlockData().getTags();
088            for (Map.Entry<String, Tag> tag : customData.entrySet()) {
089                nbt.put(tag.getKey(), tag.getValue());
090            }
091        }
092
093        BlockEntityEnchantTable enchantTable = (BlockEntityEnchantTable) BlockEntity.createBlockEntity(BlockEntity.ENCHANT_TABLE, getLevel().getChunk((int) this.x >> 4, (int) this.z >> 4), nbt);
094        return enchantTable != null;
095    }
096
097    @Override
098    public boolean onActivate(Item item, Player player) {
099        if (player != null) {
100            BlockEntity t = this.getLevel().getBlockEntity(this);
101            BlockEntityEnchantTable enchantTable;
102            if (t instanceof BlockEntityEnchantTable) {
103                enchantTable = (BlockEntityEnchantTable) t;
104            } else {
105                CompoundTag nbt = new CompoundTag()
106                        .putList(new ListTag<>("Items"))
107                        .putString("id", BlockEntity.ENCHANT_TABLE)
108                        .putInt("x", (int) this.x)
109                        .putInt("y", (int) this.y)
110                        .putInt("z", (int) this.z);
111                enchantTable = (BlockEntityEnchantTable) BlockEntity.createBlockEntity(BlockEntity.ENCHANT_TABLE, this.getLevel().getChunk((int) (this.x) >> 4, (int) (this.z) >> 4), nbt);
112                if (enchantTable == null) {
113                    return false;
114                }
115            }
116
117            if (enchantTable.namedTag.contains("Lock") && enchantTable.namedTag.get("Lock") instanceof StringTag) {
118                if (!enchantTable.namedTag.getString("Lock").equals(item.getCustomName())) {
119                    return true;
120                }
121            }
122
123            player.addWindow(new EnchantInventory(player.getUIInventory(), this.getLocation()), Player.ENCHANT_WINDOW_ID);
124        }
125
126        return true;
127    }
128
129    @Override
130    public boolean canHarvestWithHand() {
131        return false;
132    }
133
134    @Override
135    public BlockColor getColor() {
136        return BlockColor.RED_BLOCK_COLOR;
137    }
138}
