001package cn.nukkit.command.data;
002
003import static cn.nukkit.network.protocol.AvailableCommandsPacket.*;
004
005/**
006 * @author CreeperFace
007 */
008public enum CommandParamType {
009    INT(ARG_TYPE_INT),
010    FLOAT(ARG_TYPE_FLOAT),
011    VALUE(ARG_TYPE_VALUE),
012    WILDCARD_INT(ARG_TYPE_WILDCARD_INT),
013    TARGET(ARG_TYPE_TARGET),
014    WILDCARD_TARGET(ARG_TYPE_WILDCARD_TARGET),
015    STRING(ARG_TYPE_STRING),
016    POSITION(ARG_TYPE_POSITION),
017    MESSAGE(ARG_TYPE_MESSAGE),
018    RAWTEXT(ARG_TYPE_RAWTEXT),
019    JSON(ARG_TYPE_JSON),
020    TEXT(ARG_TYPE_RAWTEXT), // backwards compatibility
021    COMMAND(ARG_TYPE_COMMAND),
022    FILE_PATH(ARG_TYPE_FILE_PATH),
023    OPERATOR(ARG_TYPE_OPERATOR),
024    ;
025
026    private final int id;
027
028    CommandParamType(int id) {
029        this.id = id;
030    }
031
032    public int getId() {
033        return id;
034    }
035}
