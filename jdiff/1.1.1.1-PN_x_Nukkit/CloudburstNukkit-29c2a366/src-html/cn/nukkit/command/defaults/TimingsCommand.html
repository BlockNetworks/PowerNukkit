001package cn.nukkit.command.defaults;
002
003import cn.nukkit.command.CommandSender;
004import cn.nukkit.command.data.CommandEnum;
005import cn.nukkit.command.data.CommandParameter;
006import cn.nukkit.lang.TranslationContainer;
007import co.aikar.timings.Timings;
008import co.aikar.timings.TimingsExport;
009
010/**
011 * @author fromgate
012 * @author Pub4Game
013 */
014public class TimingsCommand extends VanillaCommand {
015
016    public TimingsCommand(String name) {
017        super(name, "%nukkit.command.timings.description", "%nukkit.command.timings.usage");
018        this.setPermission("nukkit.command.timings");
019        this.commandParameters.clear();
020        this.commandParameters.put("default", new CommandParameter[]{
021                CommandParameter.newEnum("action", new CommandEnum("TimingsAction", "on", "off", "paste", "verbon", "verboff", "reset", "report"))
022        });
023    }
024
025    @Override
026    public boolean execute(CommandSender sender, String commandLabel, String[] args) {
027        if (!this.testPermission(sender)) {
028            return true;
029        }
030
031        if (args.length != 1) {
032            sender.sendMessage(new TranslationContainer("commands.generic.usage", usageMessage));
033            return true;
034        }
035
036        String mode = args[0].toLowerCase();
037
038        if (mode.equals("on")) {
039            Timings.setTimingsEnabled(true);
040            Timings.reset();
041            sender.sendMessage(new TranslationContainer("nukkit.command.timings.enable"));
042            return true;
043        } else if (mode.equals("off")) {
044            Timings.setTimingsEnabled(false);
045            sender.sendMessage(new TranslationContainer("nukkit.command.timings.disable"));
046            return true;
047        }
048
049        if (!Timings.isTimingsEnabled()) {
050            sender.sendMessage(new TranslationContainer("nukkit.command.timings.timingsDisabled"));
051            return true;
052        }
053
054        switch (mode) {
055            case "verbon":
056                sender.sendMessage(new TranslationContainer("nukkit.command.timings.verboseEnable"));
057                Timings.setVerboseEnabled(true);
058                break;
059            case "verboff":
060                sender.sendMessage(new TranslationContainer("nukkit.command.timings.verboseDisable"));
061                Timings.setVerboseEnabled(true);
062                break;
063            case "reset":
064                Timings.reset();
065                sender.sendMessage(new TranslationContainer("nukkit.command.timings.reset"));
066                break;
067            case "report":
068            case "paste":
069                TimingsExport.reportTimings(sender);
070                break;
071        }
072        return true;
073    }
074}
