001package cn.nukkit.command.defaults;
002
003import cn.nukkit.Player;
004import cn.nukkit.Server;
005import cn.nukkit.command.Command;
006import cn.nukkit.command.CommandSender;
007import cn.nukkit.command.data.CommandParamType;
008import cn.nukkit.command.data.CommandParameter;
009import cn.nukkit.entity.Entity;
010import cn.nukkit.event.entity.EntityDamageEvent;
011import cn.nukkit.event.entity.EntityDamageEvent.DamageCause;
012import cn.nukkit.lang.TranslationContainer;
013import cn.nukkit.level.Level;
014import cn.nukkit.utils.TextFormat;
015
016import java.util.StringJoiner;
017
018/**
019 * Created on 2015/12/08 by Pub4Game.
020 * Package cn.nukkit.command.defaults in project Nukkit .
021 */
022public class KillCommand extends VanillaCommand {
023
024    public KillCommand(String name) {
025        super(name, "%nukkit.command.kill.description", "%nukkit.command.kill.usage", new String[]{"suicide"});
026        this.setPermission("nukkit.command.kill.self;"
027                + "nukkit.command.kill.other");
028        this.commandParameters.clear();
029        this.commandParameters.put("default", new CommandParameter[]{
030                CommandParameter.newType("player", true, CommandParamType.TARGET)
031        });
032    }
033
034    @Override
035    public boolean execute(CommandSender sender, String commandLabel, String[] args) {
036        if (!this.testPermission(sender)) {
037            return true;
038        }
039        if (args.length >= 2) {
040            sender.sendMessage(new TranslationContainer("commands.generic.usage", this.usageMessage));
041            return false;
042        }
043        if (args.length == 1) {
044            if (!sender.hasPermission("nukkit.command.kill.other")) {
045                sender.sendMessage(new TranslationContainer(TextFormat.RED + "%commands.generic.permission"));
046                return true;
047            }
048            Player player = sender.getServer().getPlayer(args[0]);
049            if (player != null) {
050                EntityDamageEvent ev = new EntityDamageEvent(player, DamageCause.SUICIDE, 1000);
051                sender.getServer().getPluginManager().callEvent(ev);
052                if (ev.isCancelled()) {
053                    return true;
054                }
055                player.setLastDamageCause(ev);
056                player.setHealth(0);
057                Command.broadcastCommandMessage(sender, new TranslationContainer("commands.kill.successful", player.getName()));
058            } else if (args[0].equals("@e")) {
059                StringJoiner joiner = new StringJoiner(", ");
060                for (Level level : Server.getInstance().getLevels().values()) {
061                    for (Entity entity : level.getEntities()) {
062                        if (!(entity instanceof Player)) {
063                            joiner.add(entity.getName());
064                            entity.close();
065                        }
066                    }
067                }
068                String entities = joiner.toString();
069                sender.sendMessage(new TranslationContainer("commands.kill.successful", entities.isEmpty() ? "0" : entities));
070            } else if (args[0].equals("@s")) {
071                if (!sender.hasPermission("nukkit.command.kill.self")) {
072                    sender.sendMessage(new TranslationContainer(TextFormat.RED + "%commands.generic.permission"));
073                    return true;
074                }
075                EntityDamageEvent ev = new EntityDamageEvent((Player) sender, DamageCause.SUICIDE, 1000);
076                sender.getServer().getPluginManager().callEvent(ev);
077                if (ev.isCancelled()) {
078                    return true;
079                }
080                ((Player) sender).setLastDamageCause(ev);
081                ((Player) sender).setHealth(0);
082                sender.sendMessage(new TranslationContainer("commands.kill.successful", sender.getName()));
083            } else if (args[0].equals("@a")) {
084                if (!sender.hasPermission("nukkit.command.kill.other")) {
085                    sender.sendMessage(new TranslationContainer(TextFormat.RED + "%commands.generic.permission"));
086                    return true;
087                }
088                for (Level level : Server.getInstance().getLevels().values()) {
089                    for (Entity entity : level.getEntities()) {
090                        if (entity instanceof Player) {
091                            entity.setHealth(0);
092                        }
093                    }
094                }
095                sender.sendMessage(new TranslationContainer(TextFormat.GOLD + "%commands.kill.all.successful"));
096            } else {
097                sender.sendMessage(new TranslationContainer(TextFormat.RED + "%commands.generic.player.notFound"));
098            }
099            return true;
100        }
101        if (sender instanceof Player) {
102            if (!sender.hasPermission("nukkit.command.kill.self")) {
103                sender.sendMessage(new TranslationContainer(TextFormat.RED + "%commands.generic.permission"));
104                return true;
105            }
106            EntityDamageEvent ev = new EntityDamageEvent((Player) sender, DamageCause.SUICIDE, 1000);
107            sender.getServer().getPluginManager().callEvent(ev);
108            if (ev.isCancelled()) {
109                return true;
110            }
111            ((Player) sender).setLastDamageCause(ev);
112            ((Player) sender).setHealth(0);
113            sender.sendMessage(new TranslationContainer("commands.kill.successful", sender.getName()));
114        } else {
115            sender.sendMessage(new TranslationContainer("commands.generic.usage", this.usageMessage));
116            return false;
117        }
118        return true;
119    }
120}
