001package cn.nukkit.event;
002
003import java.lang.annotation.ElementType;
004import java.lang.annotation.Retention;
005import java.lang.annotation.RetentionPolicy;
006import java.lang.annotation.Target;
007
008/**
009 * 定义一个事件的处理器的注解。<br>
010 * Annotation that defines a handler.
011 *
012 * <p>一个处理器的重要程度被称作处理器的<b>优先级</b>，优先级高的处理器有更多的决定权。参见：{@link #priority()}<br>
013 * The importance of a handler is called its <b>priority</b>, handlers with higher priority speaks louder then
014 * lower ones. See: {@link #priority()}</p>
015 *
016 * <p>处理器可以选择忽略或不忽略被取消的事件，这种特性可以在{@link #ignoreCancelled()}中定义。<br>
017 * A handler can choose to ignore a cancelled event or not, that can be defined in {@link #ignoreCancelled()}.</p>
018 *
019 * @author MagicDroidX(code) @ Nukkit Project
020 * @author null(javadoc) @ Nukkit Project
021 * @see cn.nukkit.event.Listener
022 * @see cn.nukkit.event.Event
023 * @since Nukkit 1.0 | Nukkit API 1.0.0
024 */
025
026@Target(ElementType.METHOD)
027@Retention(RetentionPolicy.RUNTIME)
028public @interface EventHandler {
029    /**
030     * 定义这个处理器的优先级。<br>
031     * Define the priority of the handler.
032     *
033     * <p>Nukkit调用处理器时会按照优先级从低到高的顺序调用，这样保证了高优先级的监听器能覆盖低优先级监听器做出的处理。
034     * 调用的先后顺序如下：<br> </p>
035     * When Nukkit calls all handlers, ones with lower priority is called earlier,
036     * that make handlers with higher priority can replace the decisions made by lower ones.
037     * The order that Nukkit call handlers is from the first to the last as:
038     * <ol>
039     * <li>EventPriority.LOWEST
040     * <li>EventPriority.LOW
041     * <li>EventPriority.NORMAL
042     * <li>EventPriority.HIGH
043     * <li>EventPriority.HIGHEST
044     * <li>EventPriority.MONITOR
045     * </ol>
046     *
047     * @return 这个处理器的优先级。<br>The priority of this handler.
048     * @see cn.nukkit.event.EventHandler
049     */
050    EventPriority priority() default EventPriority.NORMAL;
051
052    /**
053     * 定义这个处理器是否忽略被取消的事件。<br>
054     * Define if the handler ignores a cancelled event.
055     *
056     * <p>如果为{@code true}而且事件发生，这个处理器不会被调用，反之相反。<br>
057     * If ignoreCancelled is {@code true} and the event is cancelled, the method is
058     * not called. Otherwise, the method is always called.</p>
059     *
060     * @return 这个处理器是否忽略被取消的事件。<br>Whether cancelled events should be ignored.
061     * @see cn.nukkit.event.EventHandler
062     */
063    boolean ignoreCancelled() default false;
064}
