001package cn.nukkit.event.entity;
002
003import cn.nukkit.event.Cancellable;
004import cn.nukkit.event.Event;
005import cn.nukkit.event.HandlerList;
006import cn.nukkit.level.Position;
007import cn.nukkit.nbt.tag.CompoundTag;
008
009public class CreatureSpawnEvent extends Event implements Cancellable {
010
011    private static final HandlerList handlers = new HandlerList();
012
013    public static HandlerList getHandlers() {
014        return handlers;
015    }
016
017    private final SpawnReason reason;
018    private final int entityNetworkId;
019    private final Position position;
020    private final CompoundTag compoundTag;
021
022    public CreatureSpawnEvent(int networkId, Position position, CompoundTag nbt, SpawnReason reason) {
023        this.reason = reason;
024        this.entityNetworkId = networkId;
025        this.position = position;
026        this.compoundTag = nbt;
027    }
028
029    public SpawnReason getReason() {
030        return reason;
031    }
032
033    public int getEntityNetworkId() {
034        return entityNetworkId;
035    }
036
037    public CompoundTag getCompoundTag() {
038        return compoundTag;
039    }
040
041    public Position getPosition() {
042        return position;
043    }
044
045    /**
046     * An enum to specify the type of spawning
047     */
048    public enum SpawnReason {
049
050        /**
051         * When something spawns from natural means
052         */
053        NATURAL,
054        /**
055         * When an entity spawns as a jockey of another entity (mostly spider
056         * jockeys)
057         */
058        JOCKEY,
059        /**
060         * When a creature spawns from a spawner
061         */
062        SPAWNER,
063        /**
064         * When a creature spawns from an egg
065         */
066        EGG,
067        /**
068         * When a creature spawns from a Spawner Egg
069         */
070        SPAWN_EGG,
071        /**
072         * When a creature spawns because of a lightning strike
073         */
074        LIGHTNING,
075        /**
076         * When a snowman is spawned by being built
077         */
078        BUILD_SNOWMAN,
079        /**
080         * When an iron golem is spawned by being built
081         */
082        BUILD_IRONGOLEM,
083        /**
084         * When a wither boss is spawned by being built
085         */
086        BUILD_WITHER,
087        /**
088         * When an iron golem is spawned to defend a village
089         */
090        VILLAGE_DEFENSE,
091        /**
092         * When a zombie is spawned to invade a village
093         */
094        VILLAGE_INVASION,
095        /**
096         * When an animal breeds to create a child
097         */
098        BREEDING,
099        /**
100         * When a slime splits
101         */
102        SLIME_SPLIT,
103        /**
104         * When an entity calls for reinforcements
105         */
106        REINFORCEMENTS,
107        /**
108         * When a creature is spawned by nether portal
109         */
110        NETHER_PORTAL,
111        /**
112         * When a creature is spawned by a dispenser dispensing an egg
113         */
114        DISPENSE_EGG,
115        /**
116         * When a zombie infects a villager
117         */
118        INFECTION,
119        /**
120         * When a villager is cured from infection
121         */
122        CURED,
123        /**
124         * When an ocelot has a baby spawned along with them
125         */
126        OCELOT_BABY,
127        /**
128         * When a silverfish spawns from a block
129         */
130        SILVERFISH_BLOCK,
131        /**
132         * When an entity spawns as a mount of another entity (mostly chicken
133         * jockeys)
134         */
135        MOUNT,
136        /**
137         * When an entity spawns as a trap for players approaching
138         */
139        TRAP,
140        /**
141         * When an entity is spawned as a result of ender pearl usage
142         */
143        ENDER_PEARL,
144        /**
145         * When an entity is spawned as a result of the entity it is being
146         * perched on jumping or being damaged
147         */
148        SHOULDER_ENTITY,
149        /**
150         * When a creature is spawned by another entity drowning
151         */
152        DROWNED,
153        /**
154         * When an cow is spawned by shearing a mushroom cow
155         */
156        SHEARED,
157        /**
158         * When a creature is spawned by plugins
159         */
160        CUSTOM,
161        /**
162         * When an entity is missing a SpawnReason
163         */
164        DEFAULT
165    }
166}
