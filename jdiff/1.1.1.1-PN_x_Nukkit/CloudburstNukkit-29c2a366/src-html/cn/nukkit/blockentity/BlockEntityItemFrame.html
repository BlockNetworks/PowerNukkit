001package cn.nukkit.blockentity;
002
003import cn.nukkit.Player;
004import cn.nukkit.block.Block;
005import cn.nukkit.block.BlockID;
006import cn.nukkit.event.block.ItemFrameDropItemEvent;
007import cn.nukkit.item.Item;
008import cn.nukkit.item.ItemBlock;
009import cn.nukkit.level.format.FullChunk;
010import cn.nukkit.nbt.NBTIO;
011import cn.nukkit.nbt.tag.CompoundTag;
012import cn.nukkit.network.protocol.LevelEventPacket;
013
014import java.util.concurrent.ThreadLocalRandom;
015
016/**
017 * Created by Pub4Game on 03.07.2016.
018 */
019public class BlockEntityItemFrame extends BlockEntitySpawnable {
020
021    public BlockEntityItemFrame(FullChunk chunk, CompoundTag nbt) {
022        super(chunk, nbt);
023    }
024
025    @Override
026    protected void initBlockEntity() {
027        if (!namedTag.contains("Item")) {
028            namedTag.putCompound("Item", NBTIO.putItemHelper(new ItemBlock(Block.get(BlockID.AIR))));
029        }
030        if (!namedTag.contains("ItemRotation")) {
031            namedTag.putByte("ItemRotation", 0);
032        }
033        if (!namedTag.contains("ItemDropChance")) {
034            namedTag.putFloat("ItemDropChance", 1.0f);
035        }
036
037        this.level.updateComparatorOutputLevel(this);
038
039        super.initBlockEntity();
040    }
041
042    @Override
043    public String getName() {
044        return "Item Frame";
045    }
046
047    @Override
048    public boolean isBlockEntityValid() {
049        return this.getBlock().getId() == Block.ITEM_FRAME_BLOCK;
050    }
051
052    public int getItemRotation() {
053        return this.namedTag.getByte("ItemRotation");
054    }
055
056    public void setItemRotation(int itemRotation) {
057        this.namedTag.putByte("ItemRotation", itemRotation);
058        this.level.updateComparatorOutputLevel(this);
059        this.setDirty();
060    }
061
062    public Item getItem() {
063        CompoundTag NBTTag = this.namedTag.getCompound("Item");
064        return NBTIO.getItemHelper(NBTTag);
065    }
066
067    public void setItem(Item item) {
068        this.setItem(item, true);
069    }
070
071    public void setItem(Item item, boolean setChanged) {
072        this.namedTag.putCompound("Item", NBTIO.putItemHelper(item));
073        if (setChanged) {
074            this.setDirty();
075        }
076
077        this.level.updateComparatorOutputLevel(this);
078    }
079
080    public float getItemDropChance() {
081        return this.namedTag.getFloat("ItemDropChance");
082    }
083
084    public void setItemDropChance(float chance) {
085        this.namedTag.putFloat("ItemDropChance", chance);
086    }
087
088    public void setDirty() {
089        this.spawnToAll();
090        super.setDirty();
091    }
092
093    @Override
094    public CompoundTag getSpawnCompound() {
095        if (!this.namedTag.contains("Item")) {
096            this.setItem(new ItemBlock(Block.get(BlockID.AIR)), false);
097        }
098        CompoundTag item = namedTag.getCompound("Item").copy();
099        item.setName("Item");
100        CompoundTag tag = new CompoundTag()
101                .putString("id", BlockEntity.ITEM_FRAME)
102                .putInt("x", (int) this.x)
103                .putInt("y", (int) this.y)
104                .putInt("z", (int) this.z);
105
106        if (item.getShort("id") != Item.AIR) {
107            tag.putCompound("Item", item)
108                    .putByte("ItemRotation", this.getItemRotation());
109        }
110        return tag;
111    }
112
113    public int getAnalogOutput() {
114        return this.getItem() == null || this.getItem().getId() == 0 ? 0 : this.getItemRotation() % 8 + 1;
115    }
116
117    public boolean dropItem(Player player) {
118        Item item = this.getItem();
119        if (item != null && item.getId() != Item.AIR) {
120            if (player != null) {
121                ItemFrameDropItemEvent event = new ItemFrameDropItemEvent(player, this.getBlock(), this, item);
122                this.level.getServer().getPluginManager().callEvent(event);
123                if (event.isCancelled()) {
124                    this.spawnTo(player);
125                    return true;
126                }
127            }
128
129            if (this.getItemDropChance() > ThreadLocalRandom.current().nextFloat()) {
130                this.level.dropItem(this.add(0.5, 0, 0.5), item);
131            }
132            this.setItem(Item.get(Item.AIR));
133            this.setItemRotation(0);
134            this.level.addLevelEvent(this, LevelEventPacket.EVENT_SOUND_ITEM_FRAME_ITEM_REMOVED);
135            return true;
136        }
137        return false;
138    }
139}
