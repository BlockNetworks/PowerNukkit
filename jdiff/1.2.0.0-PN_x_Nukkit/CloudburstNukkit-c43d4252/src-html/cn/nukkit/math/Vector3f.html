138        return this.getSide(side, 1);
139    }

141    public Vector3f getSide(int side, int step) {
142        switch (side) {
143            case Vector3f.SIDE_DOWN:
144                return new Vector3f(this.x, this.y - step, this.z);
145            case Vector3f.SIDE_UP:
146                return new Vector3f(this.x, this.y + step, this.z);
147            case Vector3f.SIDE_NORTH:
148                return new Vector3f(this.x, this.y, this.z - step);
149            case Vector3f.SIDE_SOUTH:
150                return new Vector3f(this.x, this.y, this.z + step);
151            case Vector3f.SIDE_WEST:
152                return new Vector3f(this.x - step, this.y, this.z);
153            case Vector3f.SIDE_EAST:
154                return new Vector3f(this.x + step, this.y, this.z);
155            default:
156                return this;
157        }
158    }

160    public static int getOppositeSide(int side) {
161        switch (side) {
162            case Vector3f.SIDE_DOWN:
163                return Vector3f.SIDE_UP;
164            case Vector3f.SIDE_UP:
165                return Vector3f.SIDE_DOWN;
166            case Vector3f.SIDE_NORTH:
167                return Vector3f.SIDE_SOUTH;
168            case Vector3f.SIDE_SOUTH:
169                return Vector3f.SIDE_NORTH;
170            case Vector3f.SIDE_WEST:
171                return Vector3f.SIDE_EAST;
172            case Vector3f.SIDE_EAST:
173                return Vector3f.SIDE_WEST;
174            default:
175                return -1;
176        }
177    }

179    public double distance(Vector3f pos) {
180        return Math.sqrt(this.distanceSquared(pos));
181    }

183    public double distanceSquared(Vector3f pos) {
184        return Math.pow(this.x - pos.x, 2) + Math.pow(this.y - pos.y, 2) + Math.pow(this.z - pos.z, 2);
185    }

207    public double length() {
208        return Math.sqrt(this.lengthSquared());
209    }
210
211    public float lengthSquared() {
212        return this.x * this.x + this.y * this.y + this.z * this.z;
213    }
214
215    public Vector3f normalize() {
216        float len = this.lengthSquared();
217        if (len > 0) {
218            return this.divide((float) Math.sqrt(len));
219        }
220        return new Vector3f(0, 0, 0);
221    }

223    public float dot(Vector3f v) {
224        return this.x * v.x + this.y * v.y + this.z * v.z;
225    }
226
227    public Vector3f cross(Vector3f v) {
228        return new Vector3f(
229                this.y * v.z - this.z * v.y,
230                this.z * v.x - this.x * v.z,
231                this.x * v.y - this.y * v.x
232        );
233    }

235    /*
236     * Returns a new vector with x value equal to the second parameter, along the line between this vector and the
237     * passed in vector, or null if not possible.
238     */
239    public Vector3f getIntermediateWithXValue(Vector3f v, float x) {
240        float xDiff = v.x - this.x;
241        float yDiff = v.y - this.y;
242        float zDiff = v.z - this.z;
243        if (xDiff * xDiff < 0.0000001) {
244            return null;
245        }
246        float f = (x - this.x) / xDiff;
247        if (f < 0 || f > 1) {
248            return null;
249        } else {
250            return new Vector3f(this.x + xDiff * f, this.y + yDiff * f, this.z + zDiff * f);
251        }
252    }
253
254    /*
255     * Returns a new vector with y value equal to the second parameter, along the line between this vector and the
256     * passed in vector, or null if not possible.
257     */
258    public Vector3f getIntermediateWithYValue(Vector3f v, float y) {
259        float xDiff = v.x - this.x;
260        float yDiff = v.y - this.y;
261        float zDiff = v.z - this.z;
262        if (yDiff * yDiff < 0.0000001) {
263            return null;
264        }
265        float f = (y - this.y) / yDiff;
266        if (f < 0 || f > 1) {
267            return null;
268        } else {
269            return new Vector3f(this.x + xDiff * f, this.y + yDiff * f, this.z + zDiff * f);
270        }
271    }
272
273    /*
274     * Returns a new vector with z value equal to the second parameter, along the line between this vector and the
275     * passed in vector, or null if not possible.
276     */
277    public Vector3f getIntermediateWithZValue(Vector3f v, float z) {
278        float xDiff = v.x - this.x;
279        float yDiff = v.y - this.y;
280        float zDiff = v.z - this.z;
281        if (zDiff * zDiff < 0.0000001) {
282            return null;
283        }
284        float f = (z - this.z) / zDiff;
285        if (f < 0 || f > 1) {
286            return null;
287        } else {
288            return new Vector3f(this.x + xDiff * f, this.y + yDiff * f, this.z + zDiff * f);
289        }
290    }
