001package cn.nukkit.blockentity;
002
003/**
004 * 表达一个能被命名的事物的接口。<br>
005 * An interface describes an object that can be named.
006 *
007 * @author MagicDroidX(code) @ Nukkit Project
008 * @author 粉鞋大妈(javadoc) @ Nukkit Project
009 * @since Nukkit 1.0 | Nukkit API 1.0.0
010 */
011public interface BlockEntityNameable {
012
013    /**
014     * 返回这个事物的名字。<br>
015     * Gets the name of this object.
016     *
017     * @return 这个事物的名字。<br>The name of this object.
018     * @since Nukkit 1.0 | Nukkit API 1.0.0
019     */
020    String getName();
021
022    /**
023     * 设置或更改这个事物的名字。<br>
024     * Changes the name of this object, or names it.
025     *
026     * @param name 这个事物的新名字。<br>The new name of this object.
027     * @since Nukkit 1.0 | Nukkit API 1.0.0
028     */
029    void setName(String name);
030
031    /**
032     * 返回这个事物是否有名字。<br>
033     * Whether this object has a name.
034     *
035     * @return 如果有名字，返回 {@code true}。<br>{@code true} for this object has a name.
036     * @since Nukkit 1.0 | Nukkit API 1.0.0
037     */
038    boolean hasName();
039}
