001package cn.nukkit.item;
002
003import cn.nukkit.Player;
004import cn.nukkit.block.Block;
005import cn.nukkit.block.BlockWater;
006import cn.nukkit.entity.Entity;
007import cn.nukkit.entity.item.EntityBoat;
008import cn.nukkit.level.Level;
009import cn.nukkit.math.BlockFace;
010import cn.nukkit.nbt.tag.CompoundTag;
011import cn.nukkit.nbt.tag.DoubleTag;
012import cn.nukkit.nbt.tag.FloatTag;
013import cn.nukkit.nbt.tag.ListTag;
014
015/**
016 * Created by yescallop on 2016/2/13.
017 */
018public class ItemBoat extends Item {
019
020    public ItemBoat() {
021        this(0, 1);
022    }
023
024    public ItemBoat(Integer meta) {
025        this(meta, 1);
026    }
027
028    public ItemBoat(Integer meta, int count) {
029        super(BOAT, meta, count, "Boat");
030    }
031
032    @Override
033    public boolean canBeActivated() {
034        return true;
035    }
036
037    @Override
038    public boolean onActivate(Level level, Player player, Block block, Block target, BlockFace face, double fx, double fy, double fz) {
039        if (face != BlockFace.UP) return false;
040        EntityBoat boat = (EntityBoat) Entity.createEntity("Boat",
041                level.getChunk(block.getFloorX() >> 4, block.getFloorZ() >> 4), new CompoundTag("")
042                .putList(new ListTag<DoubleTag>("Pos")
043                        .add(new DoubleTag("", block.getX() + 0.5))
044                        .add(new DoubleTag("", block.getY() - (target instanceof BlockWater ? 0.0625 : 0)))
045                        .add(new DoubleTag("", block.getZ() + 0.5)))
046                .putList(new ListTag<DoubleTag>("Motion")
047                        .add(new DoubleTag("", 0))
048                        .add(new DoubleTag("", 0))
049                        .add(new DoubleTag("", 0)))
050                .putList(new ListTag<FloatTag>("Rotation")
051                        .add(new FloatTag("", (float) ((player.yaw + 90f) % 360)))
052                        .add(new FloatTag("", 0)))
053                .putByte("woodID", this.getDamage())
054        );
055
056        if (boat == null) {
057            return false;
058        }
059
060        if (player.isSurvival()) {
061            Item item = player.getInventory().getItemInHand();
062            item.setCount(item.getCount() - 1);
063            player.getInventory().setItemInHand(item);
064        }
065
066        boat.spawnToAll();
067        return true;
068    }
069
070    @Override
071    public int getMaxStackSize() {
072        return 1;
073    }
074}
