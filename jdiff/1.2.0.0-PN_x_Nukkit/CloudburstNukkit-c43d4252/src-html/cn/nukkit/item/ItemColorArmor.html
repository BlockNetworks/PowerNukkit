001package cn.nukkit.item;
002
003import cn.nukkit.nbt.tag.CompoundTag;
004import cn.nukkit.utils.BlockColor;
005import cn.nukkit.utils.DyeColor;
006
007/**
008 * Created by fromgate on 27.03.2016.
009 */
010abstract public class ItemColorArmor extends ItemArmor {
011
012    public ItemColorArmor(int id) {
013        super(id);
014    }
015
016    public ItemColorArmor(int id, Integer meta) {
017        super(id, meta);
018    }
019
020    public ItemColorArmor(int id, Integer meta, int count) {
021        super(id, meta, count);
022    }
023
024    public ItemColorArmor(int id, Integer meta, int count, String name) {
025        super(id, meta, count, name);
026    }
027
028    /**
029     * Set leather armor color
030     *
031     * @param dyeColor - Dye color data value
032     * @return - Return colored item
033     */
034    @Deprecated
035    public ItemColorArmor setColor(int dyeColor) {
036        BlockColor blockColor = DyeColor.getByDyeData(dyeColor).getColor();
037        return setColor(blockColor.getRed(), blockColor.getGreen(), blockColor.getBlue());
038    }
039
040    /**
041     * Set leather armor color
042     *
043     * @param dyeColor - DyeColor object
044     * @return - Return colored item
045     */
046    public ItemColorArmor setColor(DyeColor dyeColor) {
047        BlockColor blockColor = dyeColor.getColor();
048        return setColor(blockColor.getRed(), blockColor.getGreen(), blockColor.getBlue());
049    }
050
051    /**
052     * Set leather armor color
053     *
054     * @param color - BlockColor object
055     * @return - Return colored item
056     */
057    public ItemColorArmor setColor(BlockColor color) {
058        return setColor(color.getRed(), color.getGreen(), color.getBlue());
059    }
060
061    /**
062     * Set leather armor color
063     *
064     * @param r - red
065     * @param g - green
066     * @param b - blue
067     * @return - Return colored item
068     */
069    public ItemColorArmor setColor(int r, int g, int b) {
070        int rgb = r << 16 | g << 8 | b;
071        CompoundTag tag = this.hasCompoundTag() ? this.getNamedTag() : new CompoundTag();
072        tag.putInt("customColor", rgb);
073        this.setNamedTag(tag);
074        return this;
075    }
076
077    /**
078     * Get color of Leather Item
079     *
080     * @return - BlockColor, or null if item has no color
081     */
082    public BlockColor getColor() {
083        if (!this.hasCompoundTag()) return null;
084        CompoundTag tag = this.getNamedTag();
085        if (!tag.exist("customColor")) return null;
086        int rgb = tag.getInt("customColor");
087        return new BlockColor(rgb);
088    }
089}
