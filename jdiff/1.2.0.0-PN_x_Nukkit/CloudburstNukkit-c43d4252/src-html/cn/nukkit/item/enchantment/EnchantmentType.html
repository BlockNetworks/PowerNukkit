001package cn.nukkit.item.enchantment;
002
003import cn.nukkit.block.BlockPumpkin;
004import cn.nukkit.item.Item;
005import cn.nukkit.item.ItemArmor;
006import cn.nukkit.item.ItemBow;
007import cn.nukkit.item.ItemElytra;
008import cn.nukkit.item.ItemFishingRod;
009import cn.nukkit.item.ItemSkull;
010import cn.nukkit.item.ItemTrident;
011
012/**
013 * author: MagicDroidX
014 * Nukkit Project
015 */
016public enum EnchantmentType {
017    ALL,
018    ARMOR,
019    ARMOR_HEAD,
020    ARMOR_TORSO,
021    ARMOR_LEGS,
022    ARMOR_FEET,
023    SWORD,
024    DIGGER,
025    FISHING_ROD,
026    BREAKABLE,
027    BOW,
028    WEARABLE,
029    TRIDENT;
030
031    public boolean canEnchantItem(Item item) {
032        if (this == ALL) {
033            return true;
034
035        } else if (this == BREAKABLE && item.getMaxDurability() >= 0) {
036            return true;
037
038        } else if (item instanceof ItemArmor) {
039            if (this == ARMOR) {
040                return true;
041            }
042
043            switch (this) {
044                case ARMOR_HEAD:
045                    return item.isHelmet();
046                case ARMOR_TORSO:
047                    return item.isChestplate();
048                case ARMOR_LEGS:
049                    return item.isLeggings();
050                case ARMOR_FEET:
051                    return item.isBoots();
052                default:
053                    return false;
054            }
055
056        } else {
057            switch (this) {
058                case SWORD:
059                    return item.isSword();
060                case DIGGER:
061                    return item.isPickaxe() || item.isShovel() || item.isAxe();
062                case BOW:
063                    return item instanceof ItemBow;
064                case FISHING_ROD:
065                    return item instanceof ItemFishingRod;
066                case WEARABLE:
067                    return item instanceof ItemArmor || item instanceof ItemElytra || item instanceof ItemSkull || item.getBlock() instanceof BlockPumpkin;
068                case TRIDENT:
069                    return item instanceof ItemTrident;
070                default:
071                    return false;
072            }
073        }
074    }
075}
