001package cn.nukkit.level.generator.object.tree;
002
003import cn.nukkit.block.Block;
004import cn.nukkit.block.BlockWood;
005import cn.nukkit.level.ChunkManager;
006import cn.nukkit.math.NukkitRandom;
007
008/**
009 * author: MagicDroidX
010 * Nukkit Project
011 */
012public class ObjectSpruceTree extends ObjectTree {
013    protected int treeHeight;
014
015    @Override
016    public int getTrunkBlock() {
017        return Block.LOG;
018    }
019
020    @Override
021    public int getLeafBlock() {
022        return Block.LEAVES;
023    }
024
025    @Override
026    public int getType() {
027        return BlockWood.SPRUCE;
028    }
029
030    @Override
031    public int getTreeHeight() {
032        return this.treeHeight;
033    }
034
035    @Override
036    public void placeObject(ChunkManager level, int x, int y, int z, NukkitRandom random) {
037        this.treeHeight = random.nextBoundedInt(4) + 6;
038
039        int topSize = this.getTreeHeight() - (1 + random.nextBoundedInt(2));
040        int lRadius = 2 + random.nextBoundedInt(2);
041
042        this.placeTrunk(level, x, y, z, random, this.getTreeHeight() - random.nextBoundedInt(3));
043
044        this.placeLeaves(level, topSize, lRadius, x, y, z, random);
045    }
046
047    public void placeLeaves(ChunkManager level, int topSize, int lRadius, int x, int y, int z, NukkitRandom random)   {
048        int radius = random.nextBoundedInt(2);
049        int maxR = 1;
050        int minR = 0;
051
052        for (int yy = 0; yy <= topSize; ++yy) {
053            int yyy = y + this.treeHeight - yy;
054
055            for (int xx = x - radius; xx <= x + radius; ++xx) {
056                int xOff = Math.abs(xx - x);
057                for (int zz = z - radius; zz <= z + radius; ++zz) {
058                    int zOff = Math.abs(zz - z);
059                    if (xOff == radius && zOff == radius && radius > 0) {
060                        continue;
061                    }
062
063                    if (!Block.solid[level.getBlockIdAt(xx, yyy, zz)]) {
064                        level.setBlockAt(xx, yyy, zz, this.getLeafBlock(), this.getType());
065                    }
066                }
067            }
068
069            if (radius >= maxR) {
070                radius = minR;
071                minR = 1;
072                if (++maxR > lRadius) {
073                    maxR = lRadius;
074                }
075            } else {
076                ++radius;
077            }
078        }
079    }
080}
