001package cn.nukkit.level.biome.impl.extremehills;
002
003import cn.nukkit.level.generator.noise.nukkit.f.SimplexF;
004import cn.nukkit.math.NukkitRandom;
005
006/**
007 * author: DaPorkchop_
008 * Nukkit Project
009 * <p>
010 * make sure this is touching another extreme hills type or it'll look dumb
011 *
012 * very smooth hills with flat areas between
013 */
014public class ExtremeHillsMBiome extends ExtremeHillsPlusBiome {
015    private static final SimplexF gravelNoise = new SimplexF(new NukkitRandom(0), 1f, 1 / 4f, 1 / 64f);
016    private boolean isGravel = false;
017
018    public ExtremeHillsMBiome() {
019        this(true);
020    }
021
022    public ExtremeHillsMBiome(boolean tree) {
023        super(tree);
024
025        this.setBaseHeight(1f);
026        this.setHeightVariation(0.5f);
027    }
028
029    @Override
030    public String getName() {
031        return "Extreme Hills M";
032    }
033
034    @Override
035    public int getSurfaceBlock(int y) {
036        return isGravel ? GRAVEL : super.getSurfaceBlock(y);
037    }
038
039    @Override
040    public int getSurfaceDepth(int y) {
041        return isGravel ? 4 : super.getSurfaceDepth(y);
042    }
043
044    @Override
045    public int getGroundDepth(int y) {
046        return isGravel ? 0 : super.getGroundDepth(y);
047    }
048
049    @Override
050    public void preCover(int x, int z) {
051        //-0.75 is farily rare, so there'll be much more gravel than grass
052        isGravel = gravelNoise.noise2D(x, z, true) < -0.75f;
053    }
054
055    @Override
056    public boolean doesOverhang() {
057        return false;
058    }
059}
