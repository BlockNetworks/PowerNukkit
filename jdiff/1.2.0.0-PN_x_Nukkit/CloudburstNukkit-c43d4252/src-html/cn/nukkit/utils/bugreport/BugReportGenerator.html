001package cn.nukkit.utils.bugreport;
002
003import cn.nukkit.Nukkit;
004import cn.nukkit.Server;
005import cn.nukkit.lang.BaseLang;
006import cn.nukkit.utils.Utils;
007import com.sun.management.OperatingSystemMXBean;
008
009import java.io.File;
010import java.io.IOException;
011import java.io.PrintWriter;
012import java.io.StringWriter;
013import java.lang.management.ManagementFactory;
014import java.nio.file.FileStore;
015import java.nio.file.FileSystems;
016import java.nio.file.Files;
017import java.nio.file.Path;
018import java.text.SimpleDateFormat;
019import java.util.Date;
020
021/**
022 * Project nukkit
023 */
024public class BugReportGenerator extends Thread {
025
026    private Throwable throwable;
027
028    BugReportGenerator(Throwable throwable) {
029        this.throwable = throwable;
030    }
031
032    @Override
033    public void run() {
034        BaseLang baseLang = Server.getInstance().getLanguage();
035        try {
036            Server.getInstance().getLogger().info("[BugReport] " + baseLang.translateString("nukkit.bugreport.create"));
037            String path = generate();
038            Server.getInstance().getLogger().info("[BugReport] " + baseLang.translateString("nukkit.bugreport.archive", path));
039        } catch (Exception e) {
040            StringWriter stringWriter = new StringWriter();
041            e.printStackTrace(new PrintWriter(stringWriter));
042            Server.getInstance().getLogger().info("[BugReport] " + baseLang.translateString("nukkit.bugreport.error", stringWriter.toString()));
043        }
044    }
045
046    private String generate() throws IOException {
047        File reports = new File(Nukkit.DATA_PATH, "logs/bug_reports");
048        if (!reports.isDirectory()) {
049            reports.mkdirs();
050        }
051
052        SimpleDateFormat simpleDateFormat = new SimpleDateFormat("yyyyMMddHHmmSS");
053        String date = simpleDateFormat.format(new Date());
054
055        StringBuilder model = new StringBuilder();
056        long totalDiskSpace = 0;
057        int diskNum = 0;
058        for (Path root : FileSystems.getDefault().getRootDirectories()) {
059            try {
060                FileStore store = Files.getFileStore(root);
061                model.append("Disk ").append(diskNum++).append(":(avail=").append(getCount(store.getUsableSpace(), true))
062                        .append(", total=").append(getCount(store.getTotalSpace(), true))
063                        .append(") ");
064                totalDiskSpace += store.getTotalSpace();
065            } catch (IOException e) {
066                //
067            }
068        }
069
070        StringWriter stringWriter = new StringWriter();
071        throwable.printStackTrace(new PrintWriter(stringWriter));
072
073        StackTraceElement[] stackTrace = throwable.getStackTrace();
074        boolean pluginError = false;
075        if (stackTrace.length > 0) {
076            pluginError = !throwable.getStackTrace()[0].getClassName().startsWith("cn.nukkit");
077        }
078
079
080        File mdReport = new File(reports, date + "_" + throwable.getClass().getSimpleName() + ".md");
081        mdReport.createNewFile();
082        String content = Utils.readFile(this.getClass().getClassLoader().getResourceAsStream("report_template.md"));
083
084        String cpuType = System.getenv("PROCESSOR_IDENTIFIER");
085        OperatingSystemMXBean osMXBean = (OperatingSystemMXBean) ManagementFactory.getOperatingSystemMXBean();
086        content = content.replace("${NUKKIT_VERSION}", Nukkit.VERSION);
087        content = content.replace("${JAVA_VERSION}", System.getProperty("java.vm.name") + " (" + System.getProperty("java.runtime.version") + ")");
088        content = content.replace("${HOSTOS}", osMXBean.getName() + "-" + osMXBean.getArch() + " [" + osMXBean.getVersion() + "]");
089        content = content.replace("${MEMORY}", getCount(osMXBean.getTotalPhysicalMemorySize(), true));
090        content = content.replace("${STORAGE_SIZE}", getCount(totalDiskSpace, true));
091        content = content.replace("${CPU_TYPE}", cpuType == null ? "UNKNOWN" : cpuType);
092        content = content.replace("${AVAILABLE_CORE}", String.valueOf(osMXBean.getAvailableProcessors()));
093        content = content.replace("${STACKTRACE}", stringWriter.toString());
094        content = content.replace("${PLUGIN_ERROR}", Boolean.toString(pluginError).toUpperCase());
095        content = content.replace("${STORAGE_TYPE}", model.toString());
096
097        Utils.writeFile(mdReport, content);
098
099        return mdReport.getAbsolutePath();
100    }
101
102    //Code section from SOF
103    public static String getCount(long bytes, boolean si) {
104        int unit = si ? 1000 : 1024;
105        if (bytes < unit) return bytes + " B";
106        int exp = (int) (Math.log(bytes) / Math.log(unit));
107        String pre = (si ? "kMGTPE" : "KMGTPE").charAt(exp - 1) + (si ? "" : "i");
108        return String.format("%.1f %sB", bytes / Math.pow(unit, exp), pre);
109    }
