001package cn.nukkit.network.query;
002
003import cn.nukkit.Server;
004import cn.nukkit.event.server.QueryRegenerateEvent;
005import cn.nukkit.utils.Binary;
006
007import java.security.MessageDigest;
008import java.security.NoSuchAlgorithmException;
009import java.util.Random;
010
011/**
012 * author: MagicDroidX
013 * Nukkit Project
014 */
015public class QueryHandler {
016
017    public static final byte HANDSHAKE = 0x09;
018    public static final byte STATISTICS = 0x00;
019
020    private final Server server;
021    private byte[] lastToken;
022    private byte[] token;
023    private byte[] longData;
024    private byte[] shortData;
025    private long timeout;
026
027    public QueryHandler() {
028        this.server = Server.getInstance();
029        this.server.getLogger().info(this.server.getLanguage().translateString("nukkit.server.query.start"));
030        String ip = this.server.getIp();
031        String addr = (!ip.isEmpty()) ? ip : "0.0.0.0";
032        int port = this.server.getPort();
033        this.server.getLogger().info(this.server.getLanguage().translateString("nukkit.server.query.info", String.valueOf(port)));
034
035        this.regenerateToken();
036        this.lastToken = this.token;
037        this.regenerateInfo();
038        this.server.getLogger().info(this.server.getLanguage().translateString("nukkit.server.query.running", new String[]{addr, String.valueOf(port)}));
039    }
040
041    public void regenerateInfo() {
042        QueryRegenerateEvent ev = this.server.getQueryInformation();
043        this.longData = ev.getLongQuery(this.longData);
044        this.shortData = ev.getShortQuery(this.shortData);
045        this.timeout = System.currentTimeMillis() + ev.getTimeout();
046    }
047
048    public void regenerateToken() {
049        this.lastToken = this.token;
050        byte[] token = new byte[16];
051        for (int i = 0; i < 16; i++) {
052            token[i] = (byte) new Random().nextInt(255);
053        }
054        this.token = token;
055    }
056
057    public static String getTokenString(byte[] token, String salt) {
058        return getTokenString(new String(token), salt);
059    }
060
061
062    public static String getTokenString(String token, String salt) {
063        try {
064            return String.valueOf(Binary.readInt(Binary.subBytes(MessageDigest.getInstance("SHA-512").digest((salt + ":" + token).getBytes()), 7, 4)));
065        } catch (NoSuchAlgorithmException e) {
066            return String.valueOf(new Random().nextInt());
067        }
068    }
069
070    public void handle(String address, int port, byte[] packet) {
071        int offset = 2; //skip MAGIC
072        byte packetType = packet[offset++];
073        int sessionID = Binary.readInt(Binary.subBytes(packet, offset, 4));
074        offset += 4;
075        byte[] payload = Binary.subBytes(packet, offset);
076
077        switch (packetType) {
078            case HANDSHAKE:
079                byte[] reply = Binary.appendBytes(
080                        HANDSHAKE,
081                        Binary.writeInt(sessionID),
082                        getTokenString(this.token, address).getBytes(),
083                        new byte[]{0x00}
084                );
085
086                this.server.getNetwork().sendPacket(address, port, reply);
087                break;
088            case STATISTICS:
089                String token = String.valueOf(Binary.readInt(Binary.subBytes(payload, 0, 4)));
090                if (!token.equals(getTokenString(this.token, address)) && !token.equals(getTokenString(this.lastToken, address))) {
091                    break;
092                }
093
094                if (this.timeout < System.currentTimeMillis()) {
095                    this.regenerateInfo();
096                }
097                reply = Binary.appendBytes(
098                        STATISTICS,
099                        Binary.writeInt(sessionID),
100                        payload.length == 8 ? this.longData : this.shortData
101                );
102
103                this.server.getNetwork().sendPacket(address, port, reply);
104                break;
105        }
106    }
107}
