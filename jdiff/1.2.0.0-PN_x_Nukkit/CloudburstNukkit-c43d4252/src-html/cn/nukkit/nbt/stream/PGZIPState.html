001package cn.nukkit.nbt.stream;
002
003import java.io.ByteArrayOutputStream;
004import java.util.zip.Deflater;
005import java.util.zip.DeflaterOutputStream;
006
007public class PGZIPState {
008    protected final DeflaterOutputStream str;
009    protected final ByteArrayOutputStream buf;
010    protected final Deflater def;
011
012    public PGZIPState(PGZIPOutputStream parent) {
013        this.def = parent.newDeflater();
014        this.buf = new ByteArrayOutputStream(PGZIPBlock.SIZE);
015        this.str = PGZIPOutputStream.newDeflaterOutputStream(buf, def);
016    }
017
018
019}
