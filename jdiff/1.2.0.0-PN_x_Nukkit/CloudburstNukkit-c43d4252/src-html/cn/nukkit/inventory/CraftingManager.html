229    private Item[][] cloneItemMap(Item[][] map) {
230        Item[][] newMap = new Item[map.length][];
231        for (int i = 0; i < newMap.length; i++) {
232            Item[] old = map[i];
233            Item[] n = new Item[old.length];
234            System.arraycopy(old, 0, n, 0, n.length);
235            newMap[i] = n;
236        }
237        for (int y = 0; y < newMap.length; y++) {
238            Item[] row = newMap[y];
239            for (int x = 0; x < row.length; x++) {
240                Item item = newMap[y][x];
241                newMap[y][x] = item.clone();
242            }
243        }
244        return newMap;
245    }

297    public CraftingRecipe matchRecipe(Item[][] inputMap, Item primaryOutput, Item[][] extraOutputMap) {
298        //TODO: try to match special recipes before anything else (first they need to be implemented!)
299        int outputHash = getItemHash(primaryOutput);
300        if (this.shapedRecipes.containsKey(outputHash)) {
301            List<Item> itemCol = new ArrayList<>();
302            for (Item[] items : inputMap) itemCol.addAll(Arrays.asList(items));
303            UUID inputHash = getMultiItemHash(itemCol);
304            Map<UUID, ShapedRecipe> recipeMap = shapedRecipes.get(outputHash);
305            if (recipeMap != null) {
306                ShapedRecipe recipe = recipeMap.get(inputHash);
307                if (recipe != null && recipe.matchItems(this.cloneItemMap(inputMap), this.cloneItemMap(extraOutputMap))) {
308                    //matched a recipe by hash
309                    return recipe;
310                }
311                for (ShapedRecipe shapedRecipe : recipeMap.values()) {
312                    if (shapedRecipe.matchItems(this.cloneItemMap(inputMap), this.cloneItemMap(extraOutputMap))) {
313                        return shapedRecipe;
314                    }
315                }
316            }
317        }
318        if (shapelessRecipes.containsKey(outputHash)) {
319            List<Item> list = new ArrayList<>();
320            for (Item[] a : inputMap) {
321                list.addAll(Arrays.asList(a));
322            }
323            list.sort(recipeComparator);
324            UUID inputHash = getMultiItemHash(list);
325            Map<UUID, ShapelessRecipe> recipes = shapelessRecipes.get(outputHash);
326            if (recipes == null) {
327                return null;
328            }
329            ShapelessRecipe recipe = recipes.get(inputHash);
330            if (recipe != null && recipe.matchItems(this.cloneItemMap(inputMap), this.cloneItemMap(extraOutputMap))) {
331                return recipe;
332            }
333            for (ShapelessRecipe shapelessRecipe : recipes.values()) {
334                if (shapelessRecipe.matchItems(this.cloneItemMap(inputMap), this.cloneItemMap(extraOutputMap))) {
335                    return shapelessRecipe;
336                }
337            }
338        }
339        return null;
340    }
