043        EnchantItemEvent ev = new EnchantItemEvent(inv, inputItem, outputItem, cost, source);
044        source.getServer().getPluginManager().callEvent(ev);
045        if (ev.isCancelled()) {
046            source.removeAllWindows();
047            source.sendAllInventories();
048            // Cancelled by plugin, means handled OK
049            return true;
050        }
051        // This will process all the slot changes
052        for (InventoryAction a : this.actions) {
053            if (a.execute(source)) {
054                a.onExecuteSuccess(source);
055            } else {
056                a.onExecuteFail(source);
057            }
058        }
059        if (!ev.getNewItem().equals(this.outputItem, true, true)) {
060            // Plugin changed item, so the previous slot change is going to be invalid
061            // Send the replaced item to the enchant inventory manually
062            inv.setItem(0, ev.getNewItem(), true);
063        }
064        if (!source.isCreative()) {
065            source.setExperience(source.getExperience(), source.getExperienceLevel() - ev.getXpCost());
066        }
067        return true;
068    }
