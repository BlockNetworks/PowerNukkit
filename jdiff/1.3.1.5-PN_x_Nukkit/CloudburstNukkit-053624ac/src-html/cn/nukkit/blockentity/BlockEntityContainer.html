001package cn.nukkit.blockentity;
002
003import cn.nukkit.item.Item;
004
005/**
006 * 表达一个容器的接口。<br>
007 * An interface describes a container.
008 *
009 * <p>{@code BlockEntityContainer}容器必须包含物品的{@code Item}对象。<br>
010 * A {@code BlockEntityContainer} must contain items as {@code Item} objects.</p>
011 *
012 * @author MagicDroidX(code) @ Nukkit Project
013 * @author 粉鞋大妈(javadoc) @ Nukkit Project
014 * @see BlockEntityChest
015 * @see BlockEntityFurnace
016 * @since Nukkit 1.0 | Nukkit API 1.0.0
017 */
018public interface BlockEntityContainer {
019
020    /**
021     * 返回一个存储在容器里的物品的{@code Item}对象。<br>
022     * Returns an item that stores in this container, as an {@code Item} object.
023     *
024     * @param index 这个物品的索引序号。<br>The index number of this item.
025     * @return 这个物品的 {@code Item}对象。<br>An {@code Item} object for this item.
026     * @since Nukkit 1.0 | Nukkit API 1.0.0
027     */
028    Item getItem(int index);
029
030    /**
031     * 把一个物品存储进容器。<br>
032     * Sets or stores this item into this container.
033     *
034     * <p>注意：如果这个容器相应的索引序号已经有了物品，那么新存储的物品将会替换原有的物品。<br>
035     * Notice: If there is already an item for this index number, the new item being stored will REPLACE the old one.</p>
036     *
037     * @param index 这个物品的索引序号。<br>The index number of this item.
038     * @param item  描述这个物品的 {@code Item}对象。<br>The {@code Item} object that describes this item.
039     * @since Nukkit 1.0 | Nukkit API 1.0.0
040     */
041    void setItem(int index, Item item);
042
043    /**
044     * 返回这个容器最多能包含的物品数量。<br>
045     * Returns the max number of items that this container can contain.
046     *
047     * @return 最多能包含的物品数量。<br>The max number.
048     * @since Nukkit 1.0 | Nukkit API 1.0.0
049     */
050    int getSize();
051}
