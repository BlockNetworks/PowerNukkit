049    public BaseLevelProvider(Level level, String path) throws IOException {
050        this.level = level;
051        this.path = path;
052        File file_path = new File(this.path);
053        if (!file_path.exists()) {
054            file_path.mkdirs();
055        }
056        CompoundTag levelData = NBTIO.readCompressed(new FileInputStream(new File(this.getPath() + "level.dat")), ByteOrder.BIG_ENDIAN);
057        if (levelData.get("Data") instanceof CompoundTag) {
058            this.levelData = levelData.getCompound("Data");
059        } else {
060            throw new LevelException("Invalid level.dat");
061        }
062
063        if (!this.levelData.contains("generatorName")) {
064            this.levelData.putString("generatorName", Generator.getGenerator("DEFAULT").getSimpleName().toLowerCase());
065        }
066
067        if (!this.levelData.contains("generatorOptions")) {
068            this.levelData.putString("generatorOptions", "");
069        }
070
071        this.spawn = new Vector3(this.levelData.getInt("SpawnX"), this.levelData.getInt("SpawnY"), this.levelData.getInt("SpawnZ"));
072    }

262    @Override
263    public void doGarbageCollection() {
264        int limit = (int) (System.currentTimeMillis() - 50);
265        synchronized (regions) {
266            if (regions.isEmpty()) {
267                return;
268            }
269
270            ObjectIterator<BaseRegionLoader> iter = regions.values().iterator();
271            while (iter.hasNext()) {
272                BaseRegionLoader loader = iter.next();
273
274                if (loader.lastUsed <= limit) {
275                    try {
276                        loader.close();
277                    } catch (IOException e) {
278                        throw new RuntimeException("Unable to close RegionLoader", e);
279                    }
280                    lastRegion.set(null);
281                    iter.remove();
282                }
283
284            }
285        }
286    }

291            for (BaseFullChunk chunk : this.chunks.values()) {
292                if (chunk.getChanges() != 0) {
293                    chunk.setChanged(false);
294                    this.saveChunk(chunk.getX(), chunk.getZ());
295                }
296            }
297        }

324    @Override
325    public boolean loadChunk(int chunkX, int chunkZ, boolean create) {
326        long index = Level.chunkHash(chunkX, chunkZ);
327        synchronized (chunks) {
328            if (this.chunks.containsKey(index)) {
329                return true;
330            }
331        }
332        return loadChunk(index, chunkX, chunkZ, create) != null;
333    }

340    @Override
341    public boolean unloadChunk(int X, int Z, boolean safe) {
342        long index = Level.chunkHash(X, Z);
343        synchronized (chunks) {
344            BaseFullChunk chunk = this.chunks.get(index);
345            if (chunk != null && chunk.unload(false, safe)) {
346                lastChunk.set(null);
347                this.chunks.remove(index, chunk);
348                return true;
349            }
350        }
351        return false;
352    }

359    @Override
360    public BaseFullChunk getLoadedChunk(int chunkX, int chunkZ) {
361        BaseFullChunk tmp = lastChunk.get();
362        if (tmp != null && tmp.getX() == chunkX && tmp.getZ() == chunkZ) {
363            return tmp;
364        }
365        long index = Level.chunkHash(chunkX, chunkZ);
366        synchronized (chunks) {
367            lastChunk.set(tmp = chunks.get(index));
368        }
369        return tmp;
370    }

372    @Override
373    public BaseFullChunk getLoadedChunk(long hash) {
374        BaseFullChunk tmp = lastChunk.get();
375        if (tmp != null && tmp.getIndex() == hash) {
376            return tmp;
377        }
378        synchronized (chunks) {
379            lastChunk.set(tmp = chunks.get(hash));
380        }
381        return tmp;
382    }
383
384    @Override
385    public BaseFullChunk getChunk(int chunkX, int chunkZ, boolean create) {
386        BaseFullChunk tmp = lastChunk.get();
387        if (tmp != null && tmp.getX() == chunkX && tmp.getZ() == chunkZ) {
388            return tmp;
389        }
390        long index = Level.chunkHash(chunkX, chunkZ);
391        synchronized (chunks) {
392            lastChunk.set(tmp = chunks.get(index));
393        }
394        if (tmp != null) {
395            return tmp;
396        } else {
397            tmp = this.loadChunk(index, chunkX, chunkZ, create);
398            lastChunk.set(tmp);
399            return tmp;
400        }
401    }
402
403    @Override
404    public void setChunk(int chunkX, int chunkZ, FullChunk chunk) {
405        if (!(chunk instanceof BaseFullChunk)) {
406            throw new ChunkException("Invalid Chunk class");
407        }
408        chunk.setProvider(this);
409        chunk.setPosition(chunkX, chunkZ);
410        long index = Level.chunkHash(chunkX, chunkZ);
411        synchronized (chunks) {
412            if (this.chunks.containsKey(index) && !this.chunks.get(index).equals(chunk)) {
413                this.unloadChunk(chunkX, chunkZ, false);
414            }
415            this.chunks.put(index, (BaseFullChunk) chunk);
416        }
417    }

422        return chunk != null && chunk.isPopulated();
423    }

425    @Override
426    public synchronized void close() {
427        this.unloadChunks();
428        synchronized (regions) {
429            ObjectIterator<BaseRegionLoader> iter = this.regions.values().iterator();
430
431            while (iter.hasNext()) {
432                try {
433                    iter.next().close();
434                } catch (IOException e) {
435                    throw new RuntimeException("Unable to close RegionLoader", e);
436                }
437                lastRegion.set(null);
438                iter.remove();
439            }
440        }
441        this.level = null;
442    }

444    @Override
445    public boolean isChunkGenerated(int chunkX, int chunkZ) {
446        BaseRegionLoader region = this.getRegion(chunkX >> 5, chunkZ >> 5);
447        return region != null && region.chunkExists(chunkX - region.getX() * 32, chunkZ - region.getZ() * 32) && this.getChunk(chunkX - region.getX() * 32, chunkZ - region.getZ() * 32, true).isGenerated();
448    }
