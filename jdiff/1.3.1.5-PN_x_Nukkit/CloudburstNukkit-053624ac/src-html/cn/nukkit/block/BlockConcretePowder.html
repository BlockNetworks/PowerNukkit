001package cn.nukkit.block;
002
003import cn.nukkit.Player;
004import cn.nukkit.item.Item;
005import cn.nukkit.item.ItemTool;
006import cn.nukkit.level.Level;
007import cn.nukkit.math.BlockFace;
008
009/**
010 * Created by CreeperFace on 2.6.2017.
011 */
012public class BlockConcretePowder extends BlockFallable {
013    private int meta;
014
015    public BlockConcretePowder() {
016        this(0);
017    }
018
019    public BlockConcretePowder(int meta) {
020        this.meta = meta;
021    }
022
023    @Override
024    public int getFullId() {
025        return (getId() << 4) + getDamage();
026    }
027
028    @Override
029    public final int getDamage() {
030        return this.meta;
031    }
032
033    @Override
034    public final void setDamage(int meta) {
035        this.meta = meta;
036    }
037
038    @Override
039    public int getId() {
040        return CONCRETE_POWDER;
041    }
042
043    @Override
044    public String getName() {
045        return "Concrete Powder";
046    }
047
048    @Override
049    public double getResistance() {
050        return 2.5;
051    }
052
053    @Override
054    public double getHardness() {
055        return 0.5;
056    }
057
058    @Override
059    public int getToolType() {
060        return ItemTool.TYPE_SHOVEL;
061    }
062    
063    @Override
064    public int onUpdate(int type) {
065        if (type == Level.BLOCK_UPDATE_NORMAL) {
066            super.onUpdate(Level.BLOCK_UPDATE_NORMAL);
067
068            for (int side = 1; side <= 5; side++) {
069                Block block = this.getSide(BlockFace.fromIndex(side));
070                if (block.getId() == Block.WATER || block.getId() == Block.STILL_WATER || block.getId() == Block.LAVA || block.getId() == Block.STILL_LAVA) {
071                    this.level.setBlock(this, Block.get(Block.CONCRETE, this.meta), true, true);
072                }
073            }
074
075            return Level.BLOCK_UPDATE_NORMAL;
076        }
077        return 0;
078    }
079
080    @Override
081    public boolean place(Item item, Block b, Block target, BlockFace face, double fx, double fy, double fz, Player player) {
082        boolean concrete = false;
083
084        for (int side = 1; side <= 5; side++) {
085            Block block = this.getSide(BlockFace.fromIndex(side));
086            if (block.getId() == Block.WATER || block.getId() == Block.STILL_WATER || block.getId() == Block.LAVA || block.getId() == Block.STILL_LAVA) {
087                concrete = true;
088                break;
089            }
090        }
091
092        if (concrete) {
093            this.level.setBlock(this, Block.get(Block.CONCRETE, this.getDamage()), true, true);
094        } else {
095            this.level.setBlock(this, this, true, true);
096        }
097
098        return true;
099    }
100}
