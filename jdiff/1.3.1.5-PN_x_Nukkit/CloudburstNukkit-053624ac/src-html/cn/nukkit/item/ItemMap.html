001package cn.nukkit.item;
002
003import cn.nukkit.Player;
004import cn.nukkit.nbt.tag.CompoundTag;
005import cn.nukkit.network.protocol.ClientboundMapItemDataPacket;
006import cn.nukkit.utils.MainLogger;
007
008import javax.imageio.ImageIO;
009import java.awt.Graphics2D;
010import java.awt.image.BufferedImage;
011import java.io.ByteArrayInputStream;
012import java.io.ByteArrayOutputStream;
013import java.io.File;
014import java.io.IOException;
015
016/**
017 * Created by CreeperFace on 18.3.2017.
018 */
019public class ItemMap extends Item {
020
021    public static int mapCount = 0;
022
023    // not very pretty but definitely better than before.
024    private BufferedImage image;
025
026    public ItemMap() {
027        this(0, 1);
028    }
029
030    public ItemMap(Integer meta) {
031        this(meta, 1);
032    }
033
034    public ItemMap(Integer meta, int count) {
035        super(MAP, 0, count, "Map");
036
037        if (!hasCompoundTag() || !getNamedTag().contains("map_uuid")) {
038            CompoundTag tag = new CompoundTag();
039            tag.putLong("map_uuid", mapCount++);
040            this.setNamedTag(tag);
041        }
042    }
043
044    public void setImage(File file) throws IOException {
045        setImage(ImageIO.read(file));
046    }
047
048    public void setImage(BufferedImage image) {
049        try {
050            if (image.getHeight() != 128 || image.getWidth() != 128) { //resize
051                this.image = new BufferedImage(128, 128, image.getType());
052                Graphics2D g = this.image.createGraphics();
053                g.drawImage(image, 0, 0, 128, 128, null);
054                g.dispose();
055            } else {
056                this.image = image;
057            }
058
059            ByteArrayOutputStream baos = new ByteArrayOutputStream();
060            ImageIO.write(this.image, "png", baos);
061
062            this.getNamedTag().putByteArray("Colors", baos.toByteArray());
063        } catch (IOException e) {
064            MainLogger.getLogger().logException(e);
065        }
066    }
067
068    protected BufferedImage loadImageFromNBT() {
069        try {
070            byte[] data = getNamedTag().getByteArray("Colors");
071            image = ImageIO.read(new ByteArrayInputStream(data));
072            return image;
073        } catch (IOException e) {
074            MainLogger.getLogger().logException(e);
075        }
076
077        return null;
078    }
079
080    public long getMapId() {
081        return getNamedTag().getLong("map_uuid");
082    }
083
084    public void sendImage(Player p) {
085        // don't load the image from NBT if it has been done before.
086        BufferedImage image = this.image != null ? this.image : loadImageFromNBT();
087
088        ClientboundMapItemDataPacket pk = new ClientboundMapItemDataPacket();
089        pk.mapId = getMapId();
090        pk.update = 2;
091        pk.scale = 0;
092        pk.width = 128;
093        pk.height = 128;
094        pk.offsetX = 0;
095        pk.offsetZ = 0;
096        pk.image = image;
097
098        p.dataPacket(pk);
099    }
100
101    @Override
102    public boolean canBeActivated() {
103        return true;
104    }
105
106    @Override
107    public int getMaxStackSize() {
108        return 1;
109    }
110}
