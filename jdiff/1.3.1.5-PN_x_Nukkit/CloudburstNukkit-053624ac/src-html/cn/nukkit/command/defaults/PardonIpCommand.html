001package cn.nukkit.command.defaults;
002
003import cn.nukkit.command.Command;
004import cn.nukkit.command.CommandSender;
005import cn.nukkit.command.data.CommandParameter;
006import cn.nukkit.lang.TranslationContainer;
007
008import java.util.regex.Pattern;
009
010/**
011 * author: MagicDroidX
012 * Nukkit Project
013 */
014public class PardonIpCommand extends VanillaCommand {
015
016    public PardonIpCommand(String name) {
017        super(name, "%nukkit.command.unban.ip.description", "%commands.unbanip.usage");
018        this.setPermission("nukkit.command.unban.ip");
019        this.setAliases(new String[]{"unbanip", "unban-ip", "pardonip"});
020        this.commandParameters.clear();
021        this.commandParameters.put("default", new CommandParameter[]{
022                new CommandParameter("ip")
023        });
024    }
025
026    @Override
027    public boolean execute(CommandSender sender, String commandLabel, String[] args) {
028        if (!this.testPermission(sender)) {
029            return true;
030        }
031
032        if (args.length != 1) {
033            sender.sendMessage(new TranslationContainer("commands.generic.usage", this.usageMessage));
034
035            return false;
036        }
037
038        String value = args[0];
039
040        if (Pattern.matches("^(25[0-5]|2[0-4][0-9]|[0-1]{1}[0-9]{2}|[1-9]{1}[0-9]{1}|[1-9])\\.(25[0-5]|2[0-4][0-9]|[0-1]{1}[0-9]{2}|[1-9]{1}[0-9]{1}|[1-9]|0)\\.(25[0-5]|2[0-4][0-9]|[0-1]{1}[0-9]{2}|[1-9]{1}[0-9]{1}|[1-9]|0)\\.(25[0-5]|2[0-4][0-9]|[0-1]{1}[0-9]{2}|[1-9]{1}[0-9]{1}|[0-9])$", value)) {
041            sender.getServer().getIPBans().remove(value);
042            sender.getServer().getNetwork().unblockAddress(value);
043
044            Command.broadcastCommandMessage(sender, new TranslationContainer("commands.unbanip.success", value));
045        } else {
046
047            sender.sendMessage(new TranslationContainer("commands.unbanip.invalid"));
048        }
049
050        return true;
051    }
052}
