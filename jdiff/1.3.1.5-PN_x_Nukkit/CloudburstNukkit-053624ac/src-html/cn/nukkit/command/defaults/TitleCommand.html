001package cn.nukkit.command.defaults;
002
003import cn.nukkit.Player;
004import cn.nukkit.Server;
005import cn.nukkit.command.CommandSender;
006import cn.nukkit.command.data.CommandParamType;
007import cn.nukkit.command.data.CommandParameter;
008import cn.nukkit.lang.TranslationContainer;
009import cn.nukkit.utils.TextFormat;
010
011/**
012 * @author Tee7even
013 */
014public class TitleCommand extends VanillaCommand {
015    public TitleCommand(String name) {
016        super(name, "%nukkit.command.title.description", "%nukkit.command.title.usage");
017        this.setPermission("nukkit.command.title");
018
019        this.commandParameters.clear();
020        this.commandParameters.put("clear", new CommandParameter[]{
021                new CommandParameter("player", CommandParamType.TARGET, false),
022                new CommandParameter("clear", new String[]{"clear"})
023        });
024        this.commandParameters.put("reset", new CommandParameter[]{
025                new CommandParameter("player", CommandParamType.TARGET, false),
026                new CommandParameter("reset", new String[]{"reset"})
027        });
028        this.commandParameters.put("title", new CommandParameter[]{
029                new CommandParameter("player", CommandParamType.TARGET, false),
030                new CommandParameter("title", new String[]{"title"}),
031                new CommandParameter("titleText", CommandParamType.STRING, false)
032        });
033        this.commandParameters.put("subtitle", new CommandParameter[]{
034                new CommandParameter("player", CommandParamType.TARGET, false),
035                new CommandParameter("subtitle", new String[]{"subtitle"}),
036                new CommandParameter("titleText", CommandParamType.STRING, false)
037        });
038        this.commandParameters.put("actionbar", new CommandParameter[]{
039                new CommandParameter("player", CommandParamType.TARGET, false),
040                new CommandParameter("actionbar", new String[]{"actionbar"}),
041                new CommandParameter("titleText", CommandParamType.STRING, false)
042        });
043        this.commandParameters.put("times", new CommandParameter[]{
044                new CommandParameter("player", CommandParamType.TARGET, false),
045                new CommandParameter("times", new String[]{"times"}),
046                new CommandParameter("fadeIn", CommandParamType.INT, false),
047                new CommandParameter("stay", CommandParamType.INT, false),
048                new CommandParameter("fadeOut", CommandParamType.INT, false)
049        });
050    }
051
052    @Override
053    public boolean execute(CommandSender sender, String commandLabel, String[] args) {
054        if (!this.testPermission(sender)) {
055            return true;
056        }
057        if (args.length < 2) {
058            sender.sendMessage(new TranslationContainer("commands.generic.usage", this.usageMessage));
059            return false;
060        }
061
062        Player player = Server.getInstance().getPlayerExact(args[0]);
063        if (player == null) {
064            sender.sendMessage(new TranslationContainer(TextFormat.RED + "%commands.generic.player.notFound"));
065            return true;
066        }
067
068        if (args.length == 2) {
069            switch (args[1].toLowerCase()) {
070                case "clear":
071                    player.clearTitle();
072                    sender.sendMessage(new TranslationContainer("nukkit.command.title.clear", player.getName()));
073                    break;
074                case "reset":
075                    player.resetTitleSettings();
076                    sender.sendMessage(new TranslationContainer("nukkit.command.title.reset", player.getName()));
077                    break;
078                default:
079                    sender.sendMessage(new TranslationContainer("commands.generic.usage", this.usageMessage));
080                    return false;
081            }
082        } else if (args.length == 3) {
083            switch (args[1].toLowerCase()) {
084                case "title":
085                    player.sendTitle(args[2]);
086                    sender.sendMessage(new TranslationContainer("nukkit.command.title.title",
087                            TextFormat.clean(args[2]), player.getName()));
088                    break;
089                case "subtitle":
090                    player.setSubtitle(args[2]);
091                    sender.sendMessage(new TranslationContainer("nukkit.command.title.subtitle", TextFormat.clean(args[2]), player.getName()));
092                    break;
093                /*case "actionbar":
094                    player.sendActionBarTitle(args[2]);
095                    sender.sendMessage(new TranslationContainer("nukkit.command.title.actionbar", new String[]{TextFormat.clean(args[2]), player.getName()}));
096                    break;*/
097                default:
098                    sender.sendMessage(new TranslationContainer("commands.generic.usage", this.usageMessage));
099                    return false;
100            }
101        } else if (args.length == 5) {
102            if (args[1].toLowerCase().equals("times")) {
103                try {
104                    /*player.setTitleAnimationTimes(Integer.valueOf(args[2]), //fadeIn
105                            Integer.valueOf(args[3]), //stay
106                            Integer.valueOf(args[4])); //fadeOut*/
107                    sender.sendMessage(new TranslationContainer("nukkit.command.title.times.success",
108                            args[2], args[3], args[4], player.getName()));
109                } catch (NumberFormatException exception) {
110                    sender.sendMessage(new TranslationContainer(TextFormat.RED + "%nukkit.command.title.times.fail"));
111                }
112            } else {
113                sender.sendMessage(new TranslationContainer("commands.generic.usage", this.usageMessage));
114                return false;
115            }
116        } else {
117            sender.sendMessage(new TranslationContainer("commands.generic.usage", this.usageMessage));
118            return false;
119        }
120
121        return true;
122    }
123}
