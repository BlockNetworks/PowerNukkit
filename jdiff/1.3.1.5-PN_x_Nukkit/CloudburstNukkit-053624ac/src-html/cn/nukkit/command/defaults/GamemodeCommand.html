001package cn.nukkit.command.defaults;
002
003import cn.nukkit.Player;
004import cn.nukkit.Server;
005import cn.nukkit.command.Command;
006import cn.nukkit.command.CommandSender;
007import cn.nukkit.command.data.CommandParamType;
008import cn.nukkit.command.data.CommandParameter;
009import cn.nukkit.lang.TranslationContainer;
010import cn.nukkit.utils.TextFormat;
011
012/**
013 * Created on 2015/11/13 by xtypr.
014 * Package cn.nukkit.command.defaults in project Nukkit .
015 */
016public class GamemodeCommand extends VanillaCommand {
017
018    public GamemodeCommand(String name) {
019        super(name, "%nukkit.command.gamemode.description", "%commands.gamemode.usage",
020                new String[]{"gm"});
021        this.setPermission("nukkit.command.gamemode.survival;" +
022                "nukkit.command.gamemode.creative;" +
023                "nukkit.command.gamemode.adventure;" +
024                "nukkit.command.gamemode.spectator;" +
025                "nukkit.command.gamemode.other");
026        this.commandParameters.clear();
027        this.commandParameters.put("default", new CommandParameter[]{
028                new CommandParameter("mode", CommandParamType.INT, false),
029                new CommandParameter("player", CommandParamType.TARGET, true)
030        });
031        this.commandParameters.put("byString", new CommandParameter[]{
032                new CommandParameter("mode", new String[]{"survival", "s", "creative", "c",
033                        "adventure", "a", "spectator", "spc", "view", "v"}),
034                new CommandParameter("player", CommandParamType.TARGET, true)
035        });
036    }
037
038    @Override
039    public boolean execute(CommandSender sender, String commandLabel, String[] args) {
040        if (args.length == 0) {
041            sender.sendMessage(new TranslationContainer("commands.generic.usage", this.usageMessage));
042            return false;
043        }
044
045        int gameMode = Server.getGamemodeFromString(args[0]);
046        if (gameMode == -1) {
047            sender.sendMessage("Unknown game mode");
048            return true;
049        }
050
051        CommandSender target = sender;
052        if (args.length > 1) {
053            if (sender.hasPermission("nukkit.command.gamemode.other")) {
054                target = sender.getServer().getPlayer(args[1]);
055                if (target == null) {
056                    sender.sendMessage(new TranslationContainer(TextFormat.RED + "%commands.generic.player.notFound"));
057                    return true;
058                }
059            } else {
060                sender.sendMessage(new TranslationContainer(TextFormat.RED + "%commands.generic.permission"));
061                return true;
062            }
063        } else if (!(sender instanceof Player)) {
064            sender.sendMessage(new TranslationContainer("commands.generic.usage", this.usageMessage));
065            return true;
066        }
067
068        if ((gameMode == 0 && !sender.hasPermission("nukkit.command.gamemode.survival")) ||
069                (gameMode == 1 && !sender.hasPermission("nukkit.command.gamemode.creative")) ||
070                (gameMode == 2 && !sender.hasPermission("nukkit.command.gamemode.adventure")) ||
071                (gameMode == 3 && !sender.hasPermission("nukkit.command.gamemode.spectator"))) {
072            sender.sendMessage(new TranslationContainer(TextFormat.RED + "%commands.generic.permission"));
073            return true;
074        }
075
076        if (!((Player) target).setGamemode(gameMode)) {
077            sender.sendMessage("Game mode update for " + target.getName() + " failed");
078        } else {
079            if (target.equals(sender)) {
080                Command.broadcastCommandMessage(sender, new TranslationContainer("commands.gamemode.success.self", Server.getGamemodeString(gameMode)));
081            } else {
082                target.sendMessage(new TranslationContainer("gameMode.changed"));
083                Command.broadcastCommandMessage(sender, new TranslationContainer("commands.gamemode.success.other", Server.getGamemodeString(gameMode), target.getName()));
084            }
085        }
086
087        return true;
088    }
089}
