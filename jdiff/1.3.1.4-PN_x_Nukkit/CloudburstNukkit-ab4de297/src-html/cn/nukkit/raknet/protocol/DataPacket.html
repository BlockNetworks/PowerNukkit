001package cn.nukkit.raknet.protocol;
002
003import cn.nukkit.utils.Binary;
004
005import java.util.concurrent.ConcurrentLinkedQueue;
006
007/**
008 * author: MagicDroidX
009 * Nukkit Project
010 */
011public abstract class DataPacket extends Packet {
012
013    public ConcurrentLinkedQueue<Object> packets = new ConcurrentLinkedQueue<>();
014
015    public Integer seqNumber;
016
017    @Override
018    public void encode() {
019        super.encode();
020        this.putLTriad(this.seqNumber);
021        for (Object packet : this.packets) {
022            this.put(packet instanceof EncapsulatedPacket ? ((EncapsulatedPacket) packet).toBinary() : (byte[]) packet);
023        }
024    }
025
026    public int length() {
027        int length = 4;
028        for (Object packet : this.packets) {
029            length += packet instanceof EncapsulatedPacket ? ((EncapsulatedPacket) packet).getTotalLength() : ((byte[]) packet).length;
030        }
031
032        return length;
033    }
034
035    @Override
036    public void decode() {
037        super.decode();
038        this.seqNumber = this.getLTriad();
039
040        while (!this.feof()) {
041            byte[] data = Binary.subBytes(this.buffer, this.offset);
042            EncapsulatedPacket packet = EncapsulatedPacket.fromBinary(data, false);
043            this.offset += packet.getOffset();
044            if (packet.buffer.length == 0) {
045                break;
046            }
047            this.packets.add(packet);
048        }
049    }
050
051    @Override
052    public Packet clean() {
053        this.packets = new ConcurrentLinkedQueue<>();
054        this.seqNumber = null;
055        return super.clean();
056    }
057
058    @Override
059    public DataPacket clone() throws CloneNotSupportedException {
060        DataPacket packet = (DataPacket) super.clone();
061        packet.packets = new ConcurrentLinkedQueue<>(this.packets);
062        return packet;
063    }
064
065}
