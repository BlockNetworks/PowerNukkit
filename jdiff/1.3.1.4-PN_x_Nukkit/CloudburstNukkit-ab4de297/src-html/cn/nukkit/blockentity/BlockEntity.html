047    public static long count = 1;

049    private static final BiMap<String, Class<? extends BlockEntity>> knownBlockEntities = HashBiMap.create(21);
050
051    public FullChunk chunk;
052    public String name;
053    public long id;
054
055    public boolean movable = true;
056
057    public boolean closed = false;
058    public CompoundTag namedTag;
059    protected long lastUpdate;
060    protected Server server;
061    protected Timing timing;
062
063    public BlockEntity(FullChunk chunk, CompoundTag nbt) {
064        if (chunk == null || chunk.getProvider() == null) {
065            throw new ChunkException("Invalid garbage Chunk given to Block Entity");
066        }
067
068        this.timing = Timings.getBlockEntityTiming(this);
069        this.server = chunk.getProvider().getLevel().getServer();
070        this.chunk = chunk;
071        this.setLevel(chunk.getProvider().getLevel());
072        this.namedTag = nbt;
073        this.name = "";
074        this.lastUpdate = System.currentTimeMillis();
075        this.id = BlockEntity.count++;
076        this.x = this.namedTag.getInt("x");
077        this.y = this.namedTag.getInt("y");
078        this.z = this.namedTag.getInt("z");
079        this.movable = this.namedTag.getBoolean("isMovable");
080
081        this.initBlockEntity();
082
083        this.chunk.addBlockEntity(this);
084        this.getLevel().addBlockEntity(this);
085    }
086
087    protected void initBlockEntity() {
088
089    }
090
091    public static BlockEntity createBlockEntity(String type, FullChunk chunk, CompoundTag nbt, Object... args) {
092        type = type.replaceFirst("BlockEntity", ""); //TODO: Remove this after the first release
093        BlockEntity blockEntity = null;
094
095        if (knownBlockEntities.containsKey(type)) {
096            Class<? extends BlockEntity> clazz = knownBlockEntities.get(type);
097
098            if (clazz == null) {
099                return null;
100            }
101
102            for (Constructor constructor : clazz.getConstructors()) {
103                if (blockEntity != null) {
104                    break;
105                }
106
107                if (constructor.getParameterCount() != (args == null ? 2 : args.length + 2)) {
108                    continue;
109                }
110
111                try {
112                    if (args == null || args.length == 0) {
113                        blockEntity = (BlockEntity) constructor.newInstance(chunk, nbt);
114                    } else {
115                        Object[] objects = new Object[args.length + 2];
116
117                        objects[0] = chunk;
118                        objects[1] = nbt;
119                        System.arraycopy(args, 0, objects, 2, args.length);
120                        blockEntity = (BlockEntity) constructor.newInstance(objects);
121
122                    }
123                } catch (Exception e) {
124                    MainLogger.getLogger().logException(e);
125                }
126
127            }
128        }
129
130        return blockEntity;
131    }
132
133    public static boolean registerBlockEntity(String name, Class<? extends BlockEntity> c) {
134        if (c == null) {
135            return false;
136        }
137
138        knownBlockEntities.put(name, c);
139        return true;
140    }
141
142    public final String getSaveId() {
143        return knownBlockEntities.inverse().get(getClass());
144    }
145
146    public long getId() {
147        return id;
148    }
149
150    public void saveNBT() {
151        this.namedTag.putString("id", this.getSaveId());
152        this.namedTag.putInt("x", (int) this.getX());
153        this.namedTag.putInt("y", (int) this.getY());
154        this.namedTag.putInt("z", (int) this.getZ());
155        this.namedTag.putBoolean("isMovable", this.movable);
156    }
157
158    public CompoundTag getCleanedNBT() {
159        this.saveNBT();
160        CompoundTag tag = this.namedTag.clone();
161        tag.remove("x").remove("y").remove("z").remove("id");
162        if (tag.getTags().size() > 0) {
163            return tag;
164        } else {
165            return null;
166        }
167    }
168
169    public Block getBlock() {
170        return this.getLevelBlock();
171    }
172
173    public abstract boolean isBlockEntityValid();
174
175    public boolean onUpdate() {
176        return false;
177    }
178
179    public final void scheduleUpdate() {
180        this.level.scheduleBlockEntityUpdate(this);
181    }
182
183    public void close() {
184        if (!this.closed) {
185            this.closed = true;
186            if (this.chunk != null) {
187                this.chunk.removeBlockEntity(this);
188            }
189            if (this.level != null) {
190                this.level.removeBlockEntity(this);
191            }
192            this.level = null;
193        }
194    }
195
196    public void onBreak() {
197
198    }
199
200    public void setDirty() {
201        chunk.setChanged();
202    }
203
204    public String getName() {
205        return name;
206    }
207
208    public boolean isMovable() {
209        return movable;
210    }
211
212    public static CompoundTag getDefaultCompound(Vector3 pos, String id) {
213        return new CompoundTag("")
214                .putString("id", id)
215                .putInt("x", pos.getFloorX())
216                .putInt("y", pos.getFloorY())
217                .putInt("z", pos.getFloorZ());
218    }
