001package cn.nukkit.network.rcon;
002
003import cn.nukkit.Server;
004import cn.nukkit.command.RemoteConsoleCommandSender;
005import cn.nukkit.event.server.RemoteServerCommandEvent;
006import cn.nukkit.utils.TextFormat;
007
008import java.io.IOException;
009
010/**
011 * Implementation of Source RCON protocol.
012 * https://developer.valvesoftware.com/wiki/Source_RCON_Protocol
013 * <p>
014 * Wrapper for RCONServer. Handles data.
015 *
016 * @author Tee7even
017 */
018public class RCON {
019    private final Server server;
020    private final RCONServer serverThread;
021
022    public RCON(Server server, String password, String address, int port) {
023        if (password.isEmpty()) {
024            throw new IllegalArgumentException("nukkit.server.rcon.emptyPasswordError");
025        }
026
027        this.server = server;
028
029        try {
030            this.serverThread = new RCONServer(address, port, password);
031            this.serverThread.start();
032        } catch (IOException e) {
033            throw new IllegalArgumentException("nukkit.server.rcon.startupError", e);
034        }
035
036        this.server.getLogger().info(this.server.getLanguage().translateString("nukkit.server.rcon.running", new String[]{address, String.valueOf(port)}));
037    }
038
039    public void check() {
040        if (this.serverThread == null) {
041            return;
042        } else if (!this.serverThread.isAlive()) {
043            return;
044        }
045
046        RCONCommand command;
047        while ((command = serverThread.receive()) != null) {
048            RemoteConsoleCommandSender sender = new RemoteConsoleCommandSender();
049            RemoteServerCommandEvent event = new RemoteServerCommandEvent(sender, command.getCommand());
050            this.server.getPluginManager().callEvent(event);
051
052            if (!event.isCancelled()) {
053                this.server.dispatchCommand(sender, command.getCommand());
054            }
055
056            this.serverThread.respond(command.getSender(), command.getId(), TextFormat.clean(sender.getMessages()));
057        }
058    }
059
060    public void close() {
061        try {
062            synchronized (serverThread) {
063                serverThread.close();
064                serverThread.wait(5000);
065            }
066        } catch (InterruptedException exception) {
067            //
068        }
069    }
070}
