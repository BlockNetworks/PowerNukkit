010 */
011public class ShapelessRecipe implements CraftingRecipe {
012
013    private String recipeId;
014
015    private final Item output;
016
017    private long least,most;
018
019    private final List<Item> ingredients;
020    private final List<Item> ingredientsAggregate;
021
022    private final int priority;
023
024    public ShapelessRecipe(Item result, Collection<Item> ingredients) {
025        this(null, 10, result, ingredients);
026    }
027
028    public ShapelessRecipe(String recipeId, int priority, Item result, Collection<Item> ingredients) {
029        this.recipeId = recipeId;
030        this.priority = priority;
031        this.output = result.clone();
032        if (ingredients.size() > 9) {
033            throw new IllegalArgumentException("Shapeless recipes cannot have more than 9 ingredients");
034        }
035
036        this.ingredients = new ArrayList<>();
037        this.ingredientsAggregate = new ArrayList<>();
038
039        for (Item item : ingredients) {
040            if (item.getCount() < 1) {
041                throw new IllegalArgumentException("Recipe '" + recipeId + "' Ingredient amount was not 1 (value: " + item.getCount() + ")");
042            }
043            boolean found = false;
044            for (Item existingIngredient : this.ingredientsAggregate) {
045                if (existingIngredient.equals(item, item.hasMeta(), item.hasCompoundTag())) {
046                    existingIngredient.setCount(existingIngredient.getCount() + item.getCount());
047                    found = true;
048                    break;
049                }
050            }
051            if (!found)
052                this.ingredientsAggregate.add(item.clone());
053            this.ingredients.add(item.clone());
054        }
055
056        this.ingredientsAggregate.sort(CraftingManager.recipeComparator);
057    }
058
059    @Override

061        return this.output.clone();
062    }
063

070    public UUID getId() {
071        return new UUID(least, most);
072    }
073
074    @Override
075    public void setId(UUID uuid) {
076        this.least = uuid.getLeastSignificantBits();
077        this.most = uuid.getMostSignificantBits();
078
079        if (this.recipeId == null) {
080            this.recipeId = this.getId().toString();
081        }
082    }
083
084    public List<Item> getIngredientList() {
085        List<Item> ingredients = new ArrayList<>();
086        for (Item ingredient : this.ingredients) {
087            ingredients.add(ingredient.clone());
088        }
089
090        return ingredients;
091    }
092

094        return ingredients.size();
095    }

114        return new ArrayList<>();
115    }
116

122    @Override
123    public int getPriority() {
124        return this.priority;

127    public boolean matchItems(List<Item> inputList, List<Item> extraOutputList, int multiplier) {
128        List<Item> haveInputs = new ArrayList<>();
129        for (Item item : inputList) {
130            if (item.isNull())
131                continue;
132            haveInputs.add(item.clone());
133        }
134        List<Item> needInputs = new ArrayList<>();
135        if(multiplier != 1){
136            for (Item item : ingredientsAggregate) {
137                if (item.isNull())
138                    continue;
139                Item itemClone = item.clone();
140                itemClone.setCount(itemClone.getCount() * multiplier);
141                needInputs.add(itemClone);
142            }
143        } else {
144            for (Item item : ingredientsAggregate) {
145                if (item.isNull())
146                    continue;
147                needInputs.add(item.clone());
148            }
149        }
150
151        if (!matchItemList(haveInputs, needInputs)) {
152            return false;
153        }
154
155        List<Item> haveOutputs = new ArrayList<>();
156        for (Item item : extraOutputList) {
157            if (item.isNull())
158                continue;
159            haveOutputs.add(item.clone());
160        }
161        haveOutputs.sort(CraftingManager.recipeComparator);
162        List<Item> needOutputs = new ArrayList<>();
163        if(multiplier != 1){
164            for (Item item : getExtraResults()) {
165                if (item.isNull())
166                    continue;
167                Item itemClone = item.clone();
168                itemClone.setCount(itemClone.getCount() * multiplier);
169                needOutputs.add(itemClone);
170            }
171        } else {
172            for (Item item : getExtraResults()) {
173                if (item.isNull())
174                    continue;
175                needOutputs.add(item.clone());
176            }
177        }
178        needOutputs.sort(CraftingManager.recipeComparator);
179
180        return this.matchItemList(haveOutputs, needOutputs);
181    }
182
183    /**
184     * Returns whether the specified list of crafting grid inputs and outputs matches this recipe. Outputs DO NOT
185     * include the primary result item.
186     *
187     * @param inputList  list of items taken from the crafting grid
188     * @param extraOutputList list of items put back into the crafting grid (secondary results)
189     * @return bool
190     */
191    @Override
192    public boolean matchItems(List<Item> inputList, List<Item> extraOutputList) {
193        return matchItems(inputList, extraOutputList, 1);
194    }
195
196    private boolean matchItemList(List<Item> haveItems, List<Item> needItems) {
197        for (Item needItem : new ArrayList<>(needItems)) {
198            for (Item haveItem : new ArrayList<>(haveItems)) {
199                if (needItem.equals(haveItem, needItem.hasMeta(), needItem.hasCompoundTag())) {
200                    int amount = Math.min(haveItem.getCount(), needItem.getCount());
201                    needItem.setCount(needItem.getCount() - amount);
202                    haveItem.setCount(haveItem.getCount() - amount);
203                    if (haveItem.getCount() == 0) {
204                        haveItems.remove(haveItem);
205                    }
206                    if (needItem.getCount() == 0) {
207                        needItems.remove(needItem);
208                        break;
209                    }
210                }
211            }
212        }
213        return haveItems.isEmpty() && needItems.isEmpty();
214    }
215
216    @Override
217    public List<Item> getIngredientsAggregate() {
218        return ingredientsAggregate;
219    }
