003import cn.nukkit.Player;
004import cn.nukkit.block.Block;
005import cn.nukkit.command.CommandSender;
006import cn.nukkit.command.data.CommandParamType;
007import cn.nukkit.command.data.CommandParameter;
008import cn.nukkit.item.Item;
009import cn.nukkit.lang.TranslationContainer;

013
014import java.util.Random;
015

017 * Created on 2015/11/12 by xtypr.
018 * Package cn.nukkit.command.defaults in project Nukkit .
019 */
020public class ParticleCommand extends VanillaCommand {
021    private static final String[] ENUM_VALUES = new String[]{"explode", "hugeexplosion", "hugeexplosionseed", "bubble"
022            , "splash", "wake", "water", "crit", "smoke", "spell", "instantspell", "dripwater", "driplava", "townaura"
023            , "spore", "portal", "flame", "lava", "reddust", "snowballpoof", "slime", "itembreak", "terrain", "heart"
024            , "ink", "droplet", "enchantmenttable", "happyvillager", "angryvillager", "forcefield"};
025    public ParticleCommand(String name) {
026        super(name, "%nukkit.command.particle.description", "%nukkit.command.particle.usage");
027        this.setPermission("nukkit.command.particle");
028        this.commandParameters.clear();
029        this.commandParameters.put("default", new CommandParameter[]{
030                new CommandParameter("name", false, ENUM_VALUES),
031                new CommandParameter("position", CommandParamType.POSITION, false),
032                new CommandParameter("count", CommandParamType.INT, true),
033                new CommandParameter("data", true)
034        });
035    }
036
037    @Override
038    public boolean execute(CommandSender sender, String commandLabel, String[] args) {
039        if (!this.testPermission(sender)) {
040            return true;
041        }
042
043        if (args.length < 4) {
044            sender.sendMessage(new TranslationContainer("commands.generic.usage", this.usageMessage));
045
046            return true;
047        }
048
049        Position defaultPosition;
050        if (sender instanceof Player) {
051            defaultPosition = ((Player) sender).getPosition();
052        } else {
053            defaultPosition = new Position(0, 0, 0, sender.getServer().getDefaultLevel());
054        }
055
056        String name = args[0].toLowerCase();
057
058        double x;
059        double y;
060        double z;
061
062        try {
063            x = getDouble(args[1], defaultPosition.getX());
064            y = getDouble(args[2], defaultPosition.getY());
065            z = getDouble(args[3], defaultPosition.getZ());
066        } catch (Exception e) {
067            return false;
068        }
069        Position position = new Position(x, y, z, defaultPosition.getLevel());
070
071        int count = 1;
072        if (args.length > 4) {
073            try {
074                double c = Double.parseDouble(args[4]);
075                count = (int) c;
076            } catch (Exception e) {
077                //ignore
078            }
079        }
080        count = Math.max(1, count);
081
082        int data = -1;
083        if (args.length > 5) {
084            try {
085                double d = Double.parseDouble(args[5]);
086                data = (int) d;
087            } catch (Exception e) {
088                //ignore
089            }
090        }
091
092        Particle particle = this.getParticle(name, position, data);
093
094        if (particle == null) {
095            position.level.addParticleEffect(position.asVector3f(), args[0], -1, position.level.getDimension());
096            return true;
097        }
098
099        sender.sendMessage(new TranslationContainer("commands.particle.success", name, String.valueOf(count)));
100
101        Random random = new Random(System.currentTimeMillis());
102
103        for (int i = 0; i < count; i++) {
104            particle.setComponents(
105                    position.x + (random.nextFloat() * 2 - 1),
106                    position.y + (random.nextFloat() * 2 - 1),
107                    position.z + (random.nextFloat() * 2 - 1)
108            );
109            position.getLevel().addParticle(particle);
110        }
111
112        return true;
113    }
114
115    private Particle getParticle(String name, Vector3 pos, int data) {
116        switch (name) {
117            case "explode":
118                return new ExplodeParticle(pos);
119            case "hugeexplosion":
120                return new HugeExplodeParticle(pos);
121            case "hugeexplosionseed":
122                return new HugeExplodeSeedParticle(pos);
123            case "bubble":
124                return new BubbleParticle(pos);
125            case "splash":
126                return new SplashParticle(pos);
127            case "wake":
128            case "water":
129                return new WaterParticle(pos);
130            case "crit":
131                return new CriticalParticle(pos);
132            case "smoke":
133                return new SmokeParticle(pos, data != -1 ? data : 0);
134            case "spell":
135                return new EnchantParticle(pos);
136            case "instantspell":
137                return new InstantEnchantParticle(pos);
138            case "dripwater":
139                return new WaterDripParticle(pos);
140            case "driplava":
141                return new LavaDripParticle(pos);
142            case "townaura":
143            case "spore":
144                return new SporeParticle(pos);
145            case "portal":
146                return new PortalParticle(pos);
147            case "flame":
148                return new FlameParticle(pos);
149            case "lava":
150                return new LavaParticle(pos);
151            case "reddust":
152                return new RedstoneParticle(pos, data != -1 ? data : 1);
153            case "snowballpoof":
154                return new ItemBreakParticle(pos, Item.get(Item.SNOWBALL));
155            case "slime":
156                return new ItemBreakParticle(pos, Item.get(Item.SLIMEBALL));
157            case "itembreak":
158                if (data != -1 && data != 0) {
159                    return new ItemBreakParticle(pos, Item.get(data));
160                }
161                break;
162            case "terrain":
163                if (data != -1 && data != 0) {
164                    return new TerrainParticle(pos, Block.get(data));
165                }
166                break;
167            case "heart":
168                return new HeartParticle(pos, data != -1 ? data : 0);
169            case "ink":
170                return new InkParticle(pos, data != -1 ? data : 0);
171            case "droplet":
172                return new RainSplashParticle(pos);
173            case "enchantmenttable":
174                return new EnchantmentTableParticle(pos);
175            case "happyvillager":
176                return new HappyVillagerParticle(pos);
177            case "angryvillager":
178                return new AngryVillagerParticle(pos);
179            case "forcefield":
180                return new BlockForceFieldParticle(pos);
181        }
182
183        if (name.startsWith("iconcrack_")) {
184            String[] d = name.split("_");
185            if (d.length == 3) {
186                return new ItemBreakParticle(pos, Item.get(Integer.parseInt(d[1]), Integer.valueOf(d[2])));
187            }
188        } else if (name.startsWith("blockcrack_")) {
189            String[] d = name.split("_");
190            if (d.length == 2) {
191                return new TerrainParticle(pos, Block.get(Integer.parseInt(d[1]) & 0xff, Integer.parseInt(d[1]) >> 12));
192            }
193        } else if (name.startsWith("blockdust_")) {
194            String[] d = name.split("_");
195            if (d.length >= 4) {
196                return new DustParticle(pos, Integer.parseInt(d[1]) & 0xff, Integer.parseInt(d[2]) & 0xff, Integer.parseInt(d[3]) & 0xff, d.length >= 5 ? Integer.parseInt(d[4]) & 0xff : 255);
197            }
198        }
199
200        return null;
201    }
202
203    private static double getDouble(String arg, double defaultValue) throws Exception {
204        if (arg.startsWith("~")) {
205            String relativePos = arg.substring(1);
206            if (relativePos.isEmpty()) {
207                return defaultValue;
208            }
209            return defaultValue + Double.parseDouble(relativePos);
210        }
211        return Double.parseDouble(arg);
212    }
213}
