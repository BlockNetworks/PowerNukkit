001package cn.nukkit.form.window;
002
003import cn.nukkit.form.element.ElementButton;
004import cn.nukkit.form.response.FormResponseSimple;
005
006import java.util.ArrayList;
007import java.util.List;
008
009public class FormWindowSimple extends FormWindow {
010
011    private final String type = "form"; //This variable is used for JSON import operations. Do NOT delete :) -- @Snake1999
012    private String title = "";
013    private String content = "";
014    private List<ElementButton> buttons;
015
016    private FormResponseSimple response = null;
017
018    public FormWindowSimple(String title, String content) {
019        this(title, content, new ArrayList<>());
020    }
021
022    public FormWindowSimple(String title, String content, List<ElementButton> buttons) {
023        this.title = title;
024        this.content = content;
025        this.buttons = buttons;
026    }
027
028    public String getTitle() {
029        return title;
030    }
031
032    public void setTitle(String title) {
033        this.title = title;
034    }
035
036    public String getContent() {
037        return content;
038    }
039
040    public void setContent(String content) {
041        this.content = content;
042    }
043
044    public List<ElementButton> getButtons() {
045        return buttons;
046    }
047
048    public void addButton(ElementButton button) {
049        this.buttons.add(button);
050    }
051
052    public FormResponseSimple getResponse() {
053        return response;
054    }
055
056    public void setResponse(String data) {
057        if (data.equals("null")) {
058            this.closed = true;
059            return;
060        }
061        int buttonID;
062        try {
063            buttonID = Integer.parseInt(data);
064        } catch (Exception e) {
065            return;
066        }
067        if (buttonID >= this.buttons.size()) {
068            this.response = new FormResponseSimple(buttonID, null);
069            return;
070        }
071        this.response = new FormResponseSimple(buttonID, buttons.get(buttonID));
072    }
073
074}
