020public class BlockSapling extends BlockFlowable {
021    public static final int OAK = 0;
022    public static final int SPRUCE = 1;
023    public static final int BIRCH = 2;
024    /**
025     * placeholder
026     */
027    public static final int BIRCH_TALL = 8 | BIRCH;
028    public static final int JUNGLE = 3;
029    public static final int ACACIA = 4;
030    public static final int DARK_OAK = 5;
031
032    public BlockSapling() {
033        this(0);
034    }
035
036    public BlockSapling(int meta) {
037        super(meta);
038    }
039
040    @Override
041    public int getId() {
042        return SAPLING;
043    }
044
045    @Override
046    public String getName() {
047        String[] names = new String[]{
048                "Oak Sapling",
049                "Spruce Sapling",
050                "Birch Sapling",
051                "Jungle Sapling",
052                "Acacia Sapling",
053                "Dark Oak Sapling",
054                "",
055                ""
056        };
057        return names[this.getDamage() & 0x07];
058    }
059
060    @Override
061    public boolean place(Item item, Block block, Block target, BlockFace face, double fx, double fy, double fz, Player player) {
062        Block down = this.down();
063        if (down.getId() == Block.GRASS || down.getId() == Block.DIRT || down.getId() == Block.FARMLAND || down.getId() == Block.PODZOL) {
064            this.getLevel().setBlock(block, this, true, true);
065            return true;
066        }
067
068        return false;
069    }
070
071    @Override
072    public boolean canBeActivated() {
073        return true;
074    }
075
076    public boolean onActivate(Item item, Player player) {
077        if (item.getId() == Item.DYE && item.getDamage() == 0x0F) { //BoneMeal
078            if (player != null && (player.gamemode & 0x01) == 0) {
079                item.count--;
080            }
081
082            this.level.addParticle(new BoneMealParticle(this));
083            if (ThreadLocalRandom.current().nextFloat() >= 0.45) {
084                return true;
085            }
086
087            this.grow();
088
089            return true;
090        }
091        return false;
092    }
093
094    public int onUpdate(int type) {
095        if (type == Level.BLOCK_UPDATE_NORMAL) {
096            if (this.down().isTransparent()) {
097                this.getLevel().useBreakOn(this);
098                return Level.BLOCK_UPDATE_NORMAL;
099            }
100        } else if (type == Level.BLOCK_UPDATE_RANDOM) { //Growth
101            if (ThreadLocalRandom.current().nextInt(1, 8) == 1) {
102                if ((this.getDamage() & 0x08) == 0x08) {
103                    this.grow();
104                } else {
105                    this.setDamage(this.getDamage() | 0x08);
106                    this.getLevel().setBlock(this, this, true);
107                    return Level.BLOCK_UPDATE_RANDOM;
108                }
109            } else {
110                return Level.BLOCK_UPDATE_RANDOM;
111            }
112        }
113        return Level.BLOCK_UPDATE_NORMAL;
114    }
115
116    private void grow() {
117        BasicGenerator generator = null;
118        boolean bigTree = false;
119
120        int x = 0;
121        int z = 0;
122
123        switch (this.getDamage() & 0x07) {
124            case JUNGLE:
125                loop:
126                for (; x >= -1; --x) {
127                    for (; z >= -1; --z) {
128                        if (this.findSaplings(x, z, JUNGLE)) {
129                            generator = new ObjectJungleBigTree(10, 20, Block.get(BlockID.WOOD, BlockWood.JUNGLE), Block.get(BlockID.LEAVES, BlockLeaves.JUNGLE));
130                            bigTree = true;
131                            break loop;
132                        }
133                    }
134                }
135
136                if (!bigTree) {
137                    x = 0;
138                    z = 0;
139                    generator = new NewJungleTree(4, 7);
140                }
141                break;
142            case ACACIA:
143                generator = new ObjectSavannaTree();
144                break;
145            case DARK_OAK:
146                loop:
147                for (; x >= -1; --x) {
148                    for (; z >= -1; --z) {
149                        if (this.findSaplings(x, z, DARK_OAK)) {
150                            generator = new ObjectDarkOakTree();
151                            bigTree = true;
152                            break loop;
153                        }
154                    }
155                }
156
157                if (!bigTree) {
158                    return;
159                }
160                break;
161            //TODO: big spruce
162            default:
163                ObjectTree.growTree(this.level, this.getFloorX(), this.getFloorY(), this.getFloorZ(), new NukkitRandom(), this.getDamage() & 0x07);
164                return;
165        }
166
167        if (bigTree) {
168            this.level.setBlock(this.add(x, 0, z), get(AIR), true, false);
169            this.level.setBlock(this.add(x + 1, 0, z), get(AIR), true, false);
170            this.level.setBlock(this.add(x, 0, z + 1), get(AIR), true, false);
171            this.level.setBlock(this.add(x + 1, 0, z + 1), get(AIR), true, false);
172        } else {
173            this.level.setBlock(this, get(AIR), true, false);
174        }
175
176        if (!generator.generate(this.level, new NukkitRandom(), this.add(x, 0, z))) {
177            if (bigTree) {
178                this.level.setBlock(this.add(x, 0, z), this, true, false);
179                this.level.setBlock(this.add(x + 1, 0, z), this, true, false);
180                this.level.setBlock(this.add(x, 0, z + 1), this, true, false);
181                this.level.setBlock(this.add(x + 1, 0, z + 1), this, true, false);
182            } else {
183                this.level.setBlock(this, this, true, false);
184            }
185        }
186    }
187
188    private boolean findSaplings(int x, int z, int type) {
189        return this.isSameType(this.add(x, 0, z), type) && this.isSameType(this.add(x + 1, 0, z), type) && this.isSameType(this.add(x, 0, z + 1), type) && this.isSameType(this.add(x + 1, 0, z + 1), type);
190    }
191
192    public boolean isSameType(Vector3 pos, int type) {
193        Block block = this.level.getBlock(pos);
194        return block.getId() == this.getId() && (block.getDamage() & 0x07) == (type & 0x07);
195    }
196
197    @Override
198    public Item toItem() {
199        return Item.get(BlockID.SAPLING, this.getDamage() & 0x7);
200    }
201
202    @Override
203    public BlockColor getColor() {
204        return BlockColor.FOLIAGE_BLOCK_COLOR;
205    }
