001package cn.nukkit.item;
002
003import cn.nukkit.Player;
004import cn.nukkit.block.Block;
005import cn.nukkit.entity.Entity;
006import cn.nukkit.event.entity.CreatureSpawnEvent;
007import cn.nukkit.event.entity.CreatureSpawnEvent.SpawnReason;
008import cn.nukkit.level.Level;
009import cn.nukkit.level.format.FullChunk;
010import cn.nukkit.math.BlockFace;
011import cn.nukkit.nbt.tag.CompoundTag;
012import cn.nukkit.nbt.tag.DoubleTag;
013import cn.nukkit.nbt.tag.FloatTag;
014import cn.nukkit.nbt.tag.ListTag;
015
016import java.util.Random;
017
018/**
019 * author: MagicDroidX
020 * Nukkit Project
021 */
022public class ItemSpawnEgg extends Item {
023
024    public ItemSpawnEgg() {
025        this(0, 1);
026    }
027
028    public ItemSpawnEgg(Integer meta) {
029        this(meta, 1);
030    }
031
032    public ItemSpawnEgg(Integer meta, int count) {
033        super(SPAWN_EGG, meta, count, "Spawn EntityEgg");
034    }
035
036    @Override
037    public boolean canBeActivated() {
038        return true;
039    }
040
041    @Override
042    public boolean onActivate(Level level, Player player, Block block, Block target, BlockFace face, double fx, double fy, double fz) {
043        FullChunk chunk = level.getChunk((int) block.getX() >> 4, (int) block.getZ() >> 4);
044
045        if (chunk == null) {
046            return false;
047        }
048
049        CompoundTag nbt = new CompoundTag()
050                .putList(new ListTag<DoubleTag>("Pos")
051                        .add(new DoubleTag("", block.getX() + 0.5))
052                        .add(new DoubleTag("", target.getBoundingBox() == null ? block.getY() : target.getBoundingBox().getMaxY() + 0.0001f))
053                        .add(new DoubleTag("", block.getZ() + 0.5)))
054                .putList(new ListTag<DoubleTag>("Motion")
055                        .add(new DoubleTag("", 0))
056                        .add(new DoubleTag("", 0))
057                        .add(new DoubleTag("", 0)))
058                .putList(new ListTag<FloatTag>("Rotation")
059                        .add(new FloatTag("", new Random().nextFloat() * 360))
060                        .add(new FloatTag("", 0)));
061
062        if (this.hasCustomName()) {
063            nbt.putString("CustomName", this.getCustomName());
064        }
065
066        CreatureSpawnEvent ev = new CreatureSpawnEvent(this.meta, block, nbt, SpawnReason.SPAWN_EGG);
067        level.getServer().getPluginManager().callEvent(ev);
068
069        if (ev.isCancelled()) {
070            return false;
071        }
072
073        Entity entity = Entity.createEntity(this.meta, chunk, nbt);
074
075        if (entity != null) {
076            if (player.isSurvival()) {
077                player.getInventory().decreaseCount(player.getInventory().getHeldItemIndex());
078            }
079            entity.spawnToAll();
080            return true;
081        }
082
083        return false;
084    }
085}
