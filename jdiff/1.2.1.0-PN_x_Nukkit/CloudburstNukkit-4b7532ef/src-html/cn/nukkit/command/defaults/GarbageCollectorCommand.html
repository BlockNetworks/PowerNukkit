001package cn.nukkit.command.defaults;
002
003import cn.nukkit.command.CommandSender;
004import cn.nukkit.level.Level;
005import cn.nukkit.math.NukkitMath;
006import cn.nukkit.utils.TextFormat;
007import cn.nukkit.utils.ThreadCache;
008
009/**
010 * Created on 2015/11/11 by xtypr.
011 * Package cn.nukkit.command.defaults in project Nukkit .
012 */
013public class GarbageCollectorCommand extends VanillaCommand {
014
015    public GarbageCollectorCommand(String name) {
016        super(name, "%nukkit.command.gc.description", "%nukkit.command.gc.usage");
017        this.setPermission("nukkit.command.gc");
018        this.commandParameters.clear();
019    }
020
021    @Override
022    public boolean execute(CommandSender sender, String commandLabel, String[] args) {
023        if (!this.testPermission(sender)) {
024            return true;
025        }
026
027        int chunksCollected = 0;
028        int entitiesCollected = 0;
029        int tilesCollected = 0;
030        long memory = Runtime.getRuntime().freeMemory();
031
032        for (Level level : sender.getServer().getLevels().values()) {
033            int chunksCount = level.getChunks().size();
034            int entitiesCount = level.getEntities().length;
035            int tilesCount = level.getBlockEntities().size();
036            level.doChunkGarbageCollection();
037            level.unloadChunks(true);
038            chunksCollected += chunksCount - level.getChunks().size();
039            entitiesCollected += entitiesCount - level.getEntities().length;
040            tilesCollected += tilesCount - level.getBlockEntities().size();
041        }
042
043        ThreadCache.clean();
044        System.gc();
045
046        long freedMemory = Runtime.getRuntime().freeMemory() - memory;
047
048        sender.sendMessage(TextFormat.GREEN + "---- " + TextFormat.WHITE + "Garbage collection result" + TextFormat.GREEN + " ----");
049        sender.sendMessage(TextFormat.GOLD + "Chunks: " + TextFormat.RED + chunksCollected);
050        sender.sendMessage(TextFormat.GOLD + "Entities: " + TextFormat.RED + entitiesCollected);
051        sender.sendMessage(TextFormat.GOLD + "Block Entities: " + TextFormat.RED + tilesCollected);
052        sender.sendMessage(TextFormat.GOLD + "Memory freed: " + TextFormat.RED + NukkitMath.round((freedMemory / 1024d / 1024d), 2) + " MB");
053        return true;
054    }
055}
