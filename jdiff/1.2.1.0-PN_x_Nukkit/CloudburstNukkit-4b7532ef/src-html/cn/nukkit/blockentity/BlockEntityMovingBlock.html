001package cn.nukkit.blockentity;
002
003import cn.nukkit.block.Block;
004import cn.nukkit.level.format.FullChunk;
005import cn.nukkit.math.BlockVector3;
006import cn.nukkit.nbt.tag.CompoundTag;
007
008/**
009 * Created by CreeperFace on 11.4.2017.
010 */
011public class BlockEntityMovingBlock extends BlockEntitySpawnable {
012
013    public Block block;
014
015    public BlockVector3 piston;
016    public int progress;
017
018    public BlockEntityMovingBlock(FullChunk chunk, CompoundTag nbt) {
019        super(chunk, nbt);
020    }
021
022    @Override
023    protected void initBlockEntity() {
024        if (namedTag.contains("movingBlockData") && namedTag.contains("movingBlockId")) {
025            this.block = Block.get(namedTag.getInt("movingBlockId"), namedTag.getInt("movingBlockData"));
026        } else {
027            this.close();
028        }
029
030        if (namedTag.contains("pistonPosX") && namedTag.contains("pistonPosY") && namedTag.contains("pistonPosZ")) {
031            this.piston = new BlockVector3(namedTag.getInt("pistonPosX"), namedTag.getInt("pistonPosY"), namedTag.getInt("pistonPosZ"));
032        } else {
033            this.close();
034        }
035
036        super.initBlockEntity();
037    }
038
039    public Block getBlock() {
040        return this.block;
041    }
042
043    @Override
044    public boolean isBlockEntityValid() {
045        return true;
046    }
047
048    @Override
049    public CompoundTag getSpawnCompound() {
050        return getDefaultCompound(this, MOVING_BLOCK)
051                .putFloat("movingBlockId", this.block.getId())
052                .putFloat("movingBlockData", this.block.getDamage())
053                .putInt("pistonPosX", this.piston.x)
054                .putInt("pistonPosY", this.piston.y)
055                .putInt("pistonPosZ", this.piston.z);
056    }
057}
