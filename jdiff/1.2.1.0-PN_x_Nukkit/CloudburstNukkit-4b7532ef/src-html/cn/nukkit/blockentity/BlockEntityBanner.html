001package cn.nukkit.blockentity;
002
003import cn.nukkit.block.Block;
004import cn.nukkit.level.format.FullChunk;
005import cn.nukkit.nbt.tag.CompoundTag;
006import cn.nukkit.nbt.tag.ListTag;
007import cn.nukkit.utils.BannerPattern;
008import cn.nukkit.utils.DyeColor;
009
010public class BlockEntityBanner extends BlockEntitySpawnable {
011
012    public int color;
013
014    public BlockEntityBanner(FullChunk chunk, CompoundTag nbt) {
015        super(chunk, nbt);
016    }
017
018    @Override
019    protected void initBlockEntity() {
020        if (!this.namedTag.contains("color")) {
021            this.namedTag.putByte("color", 0);
022        }
023
024        this.color = this.namedTag.getByte("color");
025
026        super.initBlockEntity();
027    }
028
029    @Override
030    public boolean isBlockEntityValid() {
031        return this.getBlock().getId() == Block.WALL_BANNER || this.getBlock().getId() == Block.STANDING_BANNER;
032    }
033
034    @Override
035    public void saveNBT() {
036        super.saveNBT();
037        this.namedTag.putByte("color", this.color);
038    }
039
040    @Override
041    public String getName() {
042        return "Banner";
043    }
044
045    public int getBaseColor() {
046        return this.namedTag.getInt("Base");
047    }
048
049    public void setBaseColor(DyeColor color) {
050        this.namedTag.putInt("Base", color.getDyeData() & 0x0f);
051    }
052
053    public int getType() {
054        return this.namedTag.getInt("Type");
055    }
056
057    public void setType(int type) {
058        this.namedTag.putInt("Type", type);
059    }
060
061    public void addPattern(BannerPattern pattern) {
062        ListTag<CompoundTag> patterns = this.namedTag.getList("Patterns", CompoundTag.class);
063        patterns.add(new CompoundTag("").
064                putInt("Color", pattern.getColor().getDyeData() & 0x0f).
065                putString("Pattern", pattern.getType().getName()));
066        this.namedTag.putList(patterns);
067    }
068
069    public BannerPattern getPattern(int index) {
070        return BannerPattern.fromCompoundTag(this.namedTag.getList("Patterns").size() > index && index >= 0 ? this.namedTag.getList("Patterns", CompoundTag.class).get(index) : new CompoundTag());
071    }
072
073    public void removePattern(int index) {
074        ListTag<CompoundTag> patterns = this.namedTag.getList("Patterns", CompoundTag.class);
075        if(patterns.size() > index && index >= 0) {
076            patterns.remove(index);
077        }
078    }
079
080    public int getPatternsSize() {
081        return this.namedTag.getList("Patterns").size();
082    }
083
084    @Override
085    public CompoundTag getSpawnCompound() {
086        return getDefaultCompound(this, BANNER)
087                .putInt("Base", getBaseColor())
088                .putList(this.namedTag.getList("Patterns"))
089                .putInt("Type", getType())
090                .putByte("color", this.color);
091    }
092
093    public DyeColor getDyeColor() {
094        return DyeColor.getByWoolData(color);
095    }
096}
