031    public static void writeFile(File file, InputStream content) throws IOException {
032        if (content == null) {
033            throw new IllegalArgumentException("content must not be null");
034        }
035        if (!file.exists()) {
036            file.createNewFile();
037        }
038        try (FileOutputStream stream = new FileOutputStream(file)) {
039            byte[] buffer = new byte[1024];
040            int length;
041            while ((length = content.read(buffer)) != -1) {
042                stream.write(buffer, 0, length);
043            }
044        }
045        content.close();
046    }

067    private static String readFile(Reader reader) throws IOException {
068        try (BufferedReader br = new BufferedReader(reader)) {
069            String temp;
070            StringBuilder stringBuilder = new StringBuilder();
071            temp = br.readLine();
072            while (temp != null) {
073                if (stringBuilder.length() != 0) {
074                    stringBuilder.append("\n");
075                }
076                stringBuilder.append(temp);
077                temp = br.readLine();
078            }
079            return stringBuilder.toString();
080        }
081    }

083    public static void copyFile(File from, File to) throws IOException {
084        if (!from.exists()) {
085            throw new FileNotFoundException();
086        }
087        if (from.isDirectory() || to.isDirectory()) {
088            throw new FileNotFoundException();
089        }
090        FileInputStream fi = null;
091        FileChannel in = null;
092        FileOutputStream fo = null;
093        FileChannel out = null;
094        try {
095            if (!to.exists()) {
096                to.createNewFile();
097            }
098            fi = new FileInputStream(from);
099            in = fi.getChannel();
100            fo = new FileOutputStream(to);
101            out = fo.getChannel();
102            in.transferTo(0, in.size(), out);
103        } finally {
104            if (fi != null) fi.close();
105            if (in != null) in.close();
106            if (fo != null) fo.close();
107            if (out != null) out.close();
108        }
109    }

167    public static long toRGB(byte r, byte g, byte b, byte a) {
168        long result = (int) r & 0xff;
169        result |= ((int) g & 0xff) << 8;
170        result |= ((int) b & 0xff) << 16;
171        result |= ((int) a & 0xff) << 24;
172        return result & 0xFFFFFFFFL;
173    }

175    public static long toABGR(int argb) {
176        long result = argb & 0xFF00FF00L;
177        result |= (argb << 16) & 0x00FF0000L; // B to R
178        result |= (argb >>> 16) & 0xFFL; // R to B
179        return result & 0xFFFFFFFFL;
180    }

182    public static Object[][] splitArray(Object[] arrayToSplit, int chunkSize) {
183        if (chunkSize <= 0) {
184            return null;
185        }
186
187        int rest = arrayToSplit.length % chunkSize;
188        int chunks = arrayToSplit.length / chunkSize + (rest > 0 ? 1 : 0);
189
190        Object[][] arrays = new Object[chunks][];
191        for (int i = 0; i < (rest > 0 ? chunks - 1 : chunks); i++) {
192            arrays[i] = Arrays.copyOfRange(arrayToSplit, i * chunkSize, i * chunkSize + chunkSize);
193        }
194        if (rest > 0) {
195            arrays[chunks - 1] = Arrays.copyOfRange(arrayToSplit, (chunks - 1) * chunkSize, (chunks - 1) * chunkSize + rest);
196        }
197        return arrays;
198    }

204    public static <T> T[] reverseArray(T[] array, boolean copy) {
205        T[] data = array;
206
207        if (copy) {
208            data = Arrays.copyOf(array, array.length);
209        }
210
211        for (int left = 0, right = data.length - 1; left < right; left++, right--) {
212            // swap the values at the left and right indices
213            T temp = data[left];
214            data[left] = data[right];
215            data[right] = temp;
216        }
217
218        return data;
219    }

231    public static <T,U,V> Map<U,V> getOrCreate(Map<T, Map<U, V>> map, T key) {
232        Map<U, V> existing = map.get(key);
233        if (existing == null) {
234            ConcurrentHashMap<U, V> toPut = new ConcurrentHashMap<>();
235            existing = map.putIfAbsent(key, toPut);
236            if (existing == null) {
237                existing = toPut;
238            }
239        }
240        return existing;
241    }

243    public static <T, U, V extends U> U getOrCreate(Map<T, U> map, Class<V> clazz, T key) {
244        U existing = map.get(key);
245        if (existing != null) {
246            return existing;
247        }
248        try {
249            U toPut = clazz.newInstance();
250            existing = map.putIfAbsent(key, toPut);
251            if (existing == null) {
252                return toPut;
253            }
254            return existing;
255        } catch (InstantiationException | IllegalAccessException e) {
256            throw new RuntimeException(e);
257        }
258    }

268    public static byte[] parseHexBinary(String s) {
269        final int len = s.length();
270
271        // "111" is not a valid hex encoding.
272        if(len % 2 != 0)
273            throw new IllegalArgumentException("hexBinary needs to be even-length: " + s);
274
275        byte[] out = new byte[len / 2];
276
277        for(int i = 0; i < len; i += 2) {
278            int h = hexToBin(s.charAt(i));
279            int l = hexToBin(s.charAt(i + 1));
280            if(h == -1 || l == -1)
281                throw new IllegalArgumentException("contains illegal character for hexBinary: " + s);
282
283            out[i / 2] = (byte)(h * 16 + l);
284        }
285
286        return out;
287    }

289    private static int hexToBin( char ch ) {
290        if('0' <= ch && ch <= '9')    return ch - '0';
291        if('A' <= ch && ch <= 'F')    return ch - 'A' + 10;
292        if('a' <= ch && ch <= 'f')    return ch - 'a' + 10;
293        return -1;
294    }
