001package cn.nukkit.utils;
002
003import java.io.*;
004import java.net.HttpURLConnection;
005import java.net.URL;
006import java.util.ArrayList;
007import java.util.List;
008import java.util.regex.Matcher;
009import java.util.regex.Pattern;
010
011public class HastebinUtility {
012
013    public static final String BIN_URL = "https://hastebin.com/documents", USER_AGENT = "Mozilla/5.0";
014    public static final Pattern PATTERN = Pattern.compile("\\{\"key\":\"([\\S\\s]*)\"}");
015
016    public static String upload(final String string) throws IOException {
017        final URL url = new URL(BIN_URL);
018        final HttpURLConnection connection = (HttpURLConnection) url.openConnection();
019
020        connection.setRequestMethod("POST");
021        connection.setRequestProperty("User-Agent", USER_AGENT);
022        connection.setDoOutput(true);
023
024        try (DataOutputStream outputStream = new DataOutputStream(connection.getOutputStream())) {
025            outputStream.write(string.getBytes());
026            outputStream.flush();
027        }
028
029        StringBuilder response;
030        try (BufferedReader in = new BufferedReader(new InputStreamReader(connection.getInputStream()))) {
031            response = new StringBuilder();
032
033            String inputLine;
034            while ((inputLine = in.readLine()) != null) {
035                response.append(inputLine);
036            }
037        }
038
039        Matcher matcher = PATTERN.matcher(response.toString());
040        if (matcher.matches()) {
041            return "https://hastebin.com/" + matcher.group(1);
042        } else {
043            throw new RuntimeException("Couldn't read response!");
044        }
045    }
046
047    public static String upload(final File file) throws IOException {
048        final StringBuilder content = new StringBuilder();
049        List<String> lines = new ArrayList<>();
050        try (BufferedReader reader = new BufferedReader(new FileReader(file))) {
051            String line;
052            int i = 0;
053            while ((line = reader.readLine()) != null) {
054                if (!line.contains("rcon.password=")) {
055                    lines.add(line);
056                }
057            }
058        }
059        for (int i = Math.max(0, lines.size() - 1000); i < lines.size(); i++) {
060            content.append(lines.get(i)).append("\n");
061        }
062        return upload(content.toString());
063    }
064
065}
