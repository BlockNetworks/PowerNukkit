001package cn.nukkit.level;
002
003import java.util.Optional;
004
005public enum GameRule {
006    COMMAND_BLOCK_OUTPUT("commandBlockOutput"),
007    DO_DAYLIGHT_CYCLE("doDaylightCycle"),
008    DO_ENTITY_DROPS("doEntityDrops"),
009    DO_FIRE_TICK("doFireTick"),
010    DO_IMMEDIATE_RESPAWN("doImmediateRespawn"),
011    DO_MOB_LOOT("doMobLoot"),
012    DO_MOB_SPAWNING("doMobSpawning"),
013    DO_TILE_DROPS("doTileDrops"),
014    DO_WEATHER_CYCLE("doWeatherCycle"),
015    DROWNING_DAMAGE("drowningDamage"),
016    FALL_DAMAGE("fallDamage"),
017    FIRE_DAMAGE("fireDamage"),
018    KEEP_INVENTORY("keepInventory"),
019    MOB_GRIEFING("mobGriefing"),
020    NATURAL_REGENERATION("naturalRegeneration"),
021    PVP("pvp"),
022    RANDOM_TICK_SPEED("randomTickSpeed"),
023    SEND_COMMAND_FEEDBACK("sendCommandFeedback"),
024    SHOW_COORDINATES("showCoordinates"),
025    TNT_EXPLODES("tntExplodes"),
026    SHOW_DEATH_MESSAGE("showDeathMessage");
027
028    private final String name;
029
030    GameRule(String name) {
031        this.name = name;
032    }
033
034    public static Optional<GameRule> parseString(String gameRuleString) {
035        for (GameRule gameRule: values()) {
036            if (gameRule.getName().equalsIgnoreCase(gameRuleString)) {
037                return Optional.of(gameRule);
038            }
039        }
040        return Optional.empty();
041    }
042
043    public static String[] getNames() {
044        String[] stringValues = new String[values().length];
045
046        for (int i = 0; i < values().length; i++) {
047            stringValues[i] = values()[i].getName();
048        }
049        return stringValues;
050    }
051
052    public String getName() {
053        return name;
054    }
055}
