053    public static boolean isValid(String path) {
054        boolean isValid = (new File(path + "/level.dat").exists()) && new File(path + "/region/").isDirectory();
055        if (isValid) {
056            for (File file : new File(path + "/region/").listFiles((dir, name) -> Pattern.matches("^.+\\.mc[r|a]$", name))) {
057                if (!file.getName().endsWith(".mca")) {
058                    isValid = false;
059                    break;
060                }
061            }
062        }
063        return isValid;
064    }

070    public static void generate(String path, String name, long seed, Class<? extends Generator> generator, Map<String, String> options) throws IOException {
071        if (!new File(path + "/region").exists()) {
072            new File(path + "/region").mkdirs();
073        }
074
075        CompoundTag levelData = new CompoundTag("Data")
076                .putCompound("GameRules", new CompoundTag())
077
078                .putLong("DayTime", 0)
079                .putInt("GameType", 0)
080                .putString("generatorName", Generator.getGeneratorName(generator))
081                .putString("generatorOptions", options.getOrDefault("preset", ""))
082                .putInt("generatorVersion", 1)
083                .putBoolean("hardcore", false)
084                .putBoolean("initialized", true)
085                .putLong("LastPlayed", System.currentTimeMillis() / 1000)
086                .putString("LevelName", name)
087                .putBoolean("raining", false)
088                .putInt("rainTime", 0)
089                .putLong("RandomSeed", seed)
090                .putInt("SpawnX", 128)
091                .putInt("SpawnY", 70)
092                .putInt("SpawnZ", 128)
093                .putBoolean("thundering", false)
094                .putInt("thunderTime", 0)
095                .putInt("version", VERSION)
096                .putLong("Time", 0)
097                .putLong("SizeOnDisk", 0);
098
099        NBTIO.writeGZIPCompressed(new CompoundTag().putCompound("Data", levelData), new FileOutputStream(path + "level.dat"), ByteOrder.BIG_ENDIAN);
100    }

107    @Override
108    public AsyncTask requestChunkTask(int x, int z) throws ChunkException {
109        Chunk chunk = (Chunk) this.getChunk(x, z, false);
110        if (chunk == null) {
111            throw new ChunkException("Invalid Chunk Set");
112        }
113
114        long timestamp = chunk.getChanges();
115
116        byte[] blockEntities = new byte[0];
117
118        if (!chunk.getBlockEntities().isEmpty()) {
119            List<CompoundTag> tagList = new ArrayList<>();
120
121            for (BlockEntity blockEntity : chunk.getBlockEntities().values()) {
122                if (blockEntity instanceof BlockEntitySpawnable) {
123                    tagList.add(((BlockEntitySpawnable) blockEntity).getSpawnCompound());
124                }
125            }
126
127            try {
128                blockEntities = NBTIO.write(tagList, ByteOrder.LITTLE_ENDIAN, true);
129            } catch (IOException e) {
130                throw new RuntimeException(e);
131            }
132        }
133
134        Map<Integer, Integer> extra = chunk.getBlockExtraDataArray();
135        BinaryStream extraData;
136        if (!extra.isEmpty()) {
137            extraData = new BinaryStream();
138            extraData.putVarInt(extra.size());
139            for (Map.Entry<Integer, Integer> entry : extra.entrySet()) {
140                extraData.putVarInt(entry.getKey());
141                extraData.putLShort(entry.getValue());
142            }
143        } else {
144            extraData = null;
145        }
146
147        BinaryStream stream = ThreadCache.binaryStream.get().reset();
148        int count = 0;
149        cn.nukkit.level.format.ChunkSection[] sections = chunk.getSections();
150        for (int i = sections.length - 1; i >= 0; i--) {
151            if (!sections[i].isEmpty()) {
152                count = i + 1;
153                break;
154            }
155        }
156//        stream.putByte((byte) count);  count is now sent in packet
157        for (int i = 0; i < count; i++) {
158            sections[i].writeTo(stream);
159        }
160//        for (byte height : chunk.getHeightMapArray()) {
161//            stream.putByte(height);
162//        } computed client side?
163        stream.put(chunk.getBiomeIdArray());
164        stream.putByte((byte) 0);
165        if (extraData != null) {
166            stream.put(extraData.getBuffer());
167        } else {
168            stream.putVarInt(0);
169        }
170        stream.put(blockEntities);
171
172        this.getLevel().chunkRequestCallback(timestamp, x, z, count, stream.getBuffer());
173
174        return null;
175    }

179    @Override
180    public void doGarbageCollection(long time) {
181        long start = System.currentTimeMillis();
182        int maxIterations = size();
183        if (lastPosition > maxIterations) lastPosition = 0;
184        int i;
185        synchronized (chunks) {
186            ObjectIterator<BaseFullChunk> iter = chunks.values().iterator();
187            if (lastPosition != 0) iter.skip(lastPosition);
188            for (i = 0; i < maxIterations; i++) {
189                if (!iter.hasNext()) {
190                    iter = chunks.values().iterator();
191                }
192                if (!iter.hasNext()) break;
193                BaseFullChunk chunk = iter.next();
194                if (chunk == null) continue;
195                if (chunk.isGenerated() && chunk.isPopulated() && chunk instanceof Chunk) {
196                    Chunk anvilChunk = (Chunk) chunk;
197                    chunk.compress();
198                    if (System.currentTimeMillis() - start >= time) break;
199                }
200            }
201        }
202        lastPosition += i;
203    }

205    @Override
206    public synchronized BaseFullChunk loadChunk(long index, int chunkX, int chunkZ, boolean create) {
207        int regionX = getRegionIndexX(chunkX);
208        int regionZ = getRegionIndexZ(chunkZ);
209        BaseRegionLoader region = this.loadRegion(regionX, regionZ);
210        this.level.timings.syncChunkLoadDataTimer.startTiming();
211        BaseFullChunk chunk;
212        try {
213            chunk = region.readChunk(chunkX - regionX * 32, chunkZ - regionZ * 32);
214        } catch (IOException e) {
215            throw new RuntimeException(e);
216        }
217        if (chunk == null) {
218            if (create) {
219                chunk = this.getEmptyChunk(chunkX, chunkZ);
220                putChunk(index, chunk);
221            }
222        } else {
223            putChunk(index, chunk);
224        }
225        this.level.timings.syncChunkLoadDataTimer.stopTiming();
226        return chunk;
227    }

229    @Override
230    public synchronized void saveChunk(int X, int Z) {
231        BaseFullChunk chunk = this.getChunk(X, Z);
232        if (chunk != null) {
233            try {
234                this.loadRegion(X >> 5, Z >> 5).writeChunk(chunk);
235            } catch (Exception e) {
236                throw new ChunkException("Error saving chunk (" + X + ", " + Z + ")", e);
237            }
238        }
239    }

242    @Override
243    public synchronized void saveChunk(int x, int z, FullChunk chunk) {
244        if (!(chunk instanceof Chunk)) {
245            throw new ChunkException("Invalid Chunk class");
246        }
247        int regionX = x >> 5;
248        int regionZ = z >> 5;
249        this.loadRegion(regionX, regionZ);
250        chunk.setX(x);
251        chunk.setZ(z);
252        try {
253            this.getRegion(regionX, regionZ).writeChunk(chunk);
254        } catch (Exception e) {
255            throw new RuntimeException(e);
256        }
257    }

265    protected synchronized BaseRegionLoader loadRegion(int x, int z) {
266        BaseRegionLoader tmp = lastRegion.get();
267        if (tmp != null && x == tmp.getX() && z == tmp.getZ()) {
268            return tmp;
269        }
270        long index = Level.chunkHash(x, z);
271        synchronized (regions) {
272            BaseRegionLoader region = this.regions.get(index);
273            if (region == null) {
274                try {
275                    region = new RegionLoader(this, x, z);
276                } catch (IOException e) {
277                    throw new RuntimeException(e);
278                }
279                this.regions.put(index, region);
280            }
281            lastRegion.set(region);
282            return region;
283        }
284    }
