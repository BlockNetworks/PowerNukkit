001package cn.nukkit.math;
002
003import java.util.Random;
004
005public class MathHelper {
006    private static float[] a = new float[65536];
007
008    static {
009        for (int i = 0; i < 65536; i++)
010            a[i] = (float) Math.sin(i * 3.141592653589793D * 2.0D / 65536.0D);
011    }
012
013    private MathHelper() {
014    }
015
016    public static float sqrt(float paramFloat) {
017        return (float) Math.sqrt(paramFloat);
018    }
019
020    public static float sin(float paramFloat) {
021        return a[((int) (paramFloat * 10430.378F) & 0xFFFF)];
022    }
023
024    public static float cos(float paramFloat) {
025        return a[((int) (paramFloat * 10430.378F + 16384.0F) & 0xFFFF)];
026    }
027
028    public static float sin(double paramFloat) {
029        return a[((int) (paramFloat * 10430.378F) & 0xFFFF)];
030    }
031
032    public static float cos(double paramFloat) {
033        return a[((int) (paramFloat * 10430.378F + 16384.0F) & 0xFFFF)];
034    }
035
036    public static int floor(double d0) {
037        int i = (int) d0;
038
039        return d0 < (double) i ? i - 1 : i;
040    }
041
042    public static long floor_double_long(double d) {
043        long l = (long) d;
044        return d >= (double) l ? l : l - 1L;
045    }
046
047    public static int floor_float_int(float f) {
048        int i = (int) f;
049        return f >= i ? i : i - 1;
050    }
051
052    public static int abs(int number) {
053        if (number > 0) {
054            return number;
055        } else {
056            return -number;
057        }
058    }
059
060    public static int log2(int bits) {
061        return Integer.SIZE - Integer.numberOfLeadingZeros(bits);
062    }
063
064    /**
065     * Returns a random number between min and max, inclusive.
066     *
067     * @param random The random number generator.
068     * @param min    The minimum value.
069     * @param max    The maximum value.
070     * @return A random number between min and max, inclusive.
071     */
072    public static int getRandomNumberInRange(Random random, int min, int max) {
073        return min + random.nextInt(max - min + 1);
074    }
075
076    public static double max(double first, double second, double third, double fourth) {
077        if (first > second && first > third && first > fourth) {
078            return first;
079        }
080        if (second > third && second > fourth) {
081            return second;
082        }
083        if (third > fourth) {
084            return third;
085        }
086        return fourth;
087    }
088
089    public static int ceil(float floatNumber) {
090        int truncated = (int) floatNumber;
091        return floatNumber > truncated ? truncated + 1 : truncated;
092    }
093
094    public static int clamp(int check, int min, int max) {
095        return check > max ? max : (check < min ? min : check);
096    }
097
098    public static double denormalizeClamp(double lowerBnd, double upperBnd, double slide) {
099        return slide < 0.0D ? lowerBnd : (slide > 1.0D ? upperBnd : lowerBnd + (upperBnd - lowerBnd) * slide);
100    }
101
102    public static float denormalizeClamp(float lowerBnd, float upperBnd, float slide) {
103        return slide < 0.0f ? lowerBnd : (slide > 1.0f ? upperBnd : lowerBnd + (upperBnd - lowerBnd) * slide);
104    }
105}
