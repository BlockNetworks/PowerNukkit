001package cn.nukkit.block;
002
003import cn.nukkit.event.redstone.RedstoneUpdateEvent;
004import cn.nukkit.item.Item;
005import cn.nukkit.item.ItemBlock;
006import cn.nukkit.level.Level;
007
008/**
009 * @author Pub4Game
010 */
011public class BlockRedstoneLampLit extends BlockRedstoneLamp {
012
013    public BlockRedstoneLampLit() {
014    }
015
016    @Override
017    public String getName() {
018        return "Lit Redstone Lamp";
019    }
020
021    @Override
022    public int getId() {
023        return LIT_REDSTONE_LAMP;
024    }
025
026    @Override
027    public int getLightLevel() {
028        return 15;
029    }
030
031    @Override
032    public Item toItem() {
033        return new ItemBlock(Block.get(BlockID.REDSTONE_LAMP));
034    }
035
036    @Override
037    public int onUpdate(int type) {
038        if ((type == Level.BLOCK_UPDATE_NORMAL || type == Level.BLOCK_UPDATE_REDSTONE) && !this.level.isBlockPowered(this.getLocation())) {
039            // Redstone event
040            RedstoneUpdateEvent ev = new RedstoneUpdateEvent(this);
041            getLevel().getServer().getPluginManager().callEvent(ev);
042            if (ev.isCancelled()) {
043                return 0;
044            }
045            this.level.scheduleUpdate(this, 4);
046            return 1;
047        }
048
049        if (type == Level.BLOCK_UPDATE_SCHEDULED && !this.level.isBlockPowered(this.getLocation())) {
050            this.level.setBlock(this, Block.get(BlockID.REDSTONE_LAMP), false, false);
051        }
052        return 0;
053    }
054}
