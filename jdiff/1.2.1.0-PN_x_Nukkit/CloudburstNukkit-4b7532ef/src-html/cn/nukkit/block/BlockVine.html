001package cn.nukkit.block;
002
003import cn.nukkit.Player;
004import cn.nukkit.entity.Entity;
005import cn.nukkit.item.Item;
006import cn.nukkit.item.ItemBlock;
007import cn.nukkit.item.ItemTool;
008import cn.nukkit.level.Level;
009import cn.nukkit.math.AxisAlignedBB;
010import cn.nukkit.math.BlockFace;
011import cn.nukkit.math.SimpleAxisAlignedBB;
012import cn.nukkit.utils.BlockColor;
013
014/**
015 * Created by Pub4Game on 15.01.2016.
016 */
017public class BlockVine extends BlockTransparentMeta {
018
019    public BlockVine(int meta) {
020        super(meta);
021    }
022
023    public BlockVine() {
024        this(0);
025    }
026
027    @Override
028    public String getName() {
029        return "Vines";
030    }
031
032    @Override
033    public int getId() {
034        return VINE;
035    }
036
037    @Override
038    public double getHardness() {
039        return 0.2;
040    }
041
042    @Override
043    public double getResistance() {
044        return 1;
045    }
046
047    @Override
048    public boolean canPassThrough() {
049        return true;
050    }
051
052    @Override
053    public boolean hasEntityCollision() {
054        return true;
055    }
056
057    @Override
058    public boolean canBeReplaced() {
059        return true;
060    }
061
062    @Override
063    public boolean canBeClimbed() {
064        return true;
065    }
066
067    @Override
068    public void onEntityCollide(Entity entity) {
069        entity.resetFallDistance();
070        entity.onGround = true;
071    }
072
073    @Override
074    public boolean isSolid() {
075        return false;
076    }
077
078    @Override
079    protected AxisAlignedBB recalculateBoundingBox() {
080        double f1 = 1;
081        double f2 = 1;
082        double f3 = 1;
083        double f4 = 0;
084        double f5 = 0;
085        double f6 = 0;
086        boolean flag = this.getDamage() > 0;
087        if ((this.getDamage() & 0x02) > 0) {
088            f4 = Math.max(f4, 0.0625);
089            f1 = 0;
090            f2 = 0;
091            f5 = 1;
092            f3 = 0;
093            f6 = 1;
094            flag = true;
095        }
096        if ((this.getDamage() & 0x08) > 0) {
097            f1 = Math.min(f1, 0.9375);
098            f4 = 1;
099            f2 = 0;
100            f5 = 1;
101            f3 = 0;
102            f6 = 1;
103            flag = true;
104        }
105        if ((this.getDamage() & 0x01) > 0) {
106            f3 = Math.min(f3, 0.9375);
107            f6 = 1;
108            f1 = 0;
109            f4 = 1;
110            f2 = 0;
111            f5 = 1;
112            flag = true;
113        }
114        if (!flag && this.up().isSolid()) {
115            f2 = Math.min(f2, 0.9375);
116            f5 = 1;
117            f1 = 0;
118            f4 = 1;
119            f3 = 0;
120            f6 = 1;
121        }
122        return new SimpleAxisAlignedBB(
123                this.x + f1,
124                this.y + f2,
125                this.z + f3,
126                this.x + f4,
127                this.y + f5,
128                this.z + f6
129        );
130    }
131
132    @Override
133    public boolean place(Item item, Block block, Block target, BlockFace face, double fx, double fy, double fz, Player player) {
134        if (target.isSolid() && face.getHorizontalIndex() != -1) {
135            this.setDamage(getMetaFromFace(face.getOpposite()));
136            this.getLevel().setBlock(block, this, true, true);
137            return true;
138        }
139
140        return false;
141    }
142
143    @Override
144    public Item[] getDrops(Item item) {
145        if (item.isShears()) {
146            return new Item[]{
147                    toItem()
148            };
149        } else {
150            return new Item[0];
151        }
152    }
153
154    @Override
155    public Item toItem() {
156        return new ItemBlock(this, 0);
157    }
158
159    @Override
160    public int onUpdate(int type) {
161        if (type == Level.BLOCK_UPDATE_NORMAL) {
162            if (!this.getSide(getFace()).isSolid()) {
163                Block up = this.up();
164                if (up.getId() != this.getId() || up.getDamage() != this.getDamage()) {
165                    this.getLevel().useBreakOn(this, null, null, true);
166                    return Level.BLOCK_UPDATE_NORMAL;
167                }
168            }
169        }
170        return 0;
171    }
172
173    private BlockFace getFace() {
174        int meta = this.getDamage();
175        if ((meta & 1) > 0) {
176            return BlockFace.SOUTH;
177        } else if ((meta & 2) > 0) {
178            return BlockFace.WEST;
179        } else if ((meta & 4) > 0) {
180            return BlockFace.NORTH;
181        } else if ((meta & 8) > 0) {
182            return BlockFace.EAST;
183        }
184
185        return BlockFace.SOUTH;
186    }
187
188    private int getMetaFromFace(BlockFace face) {
189        switch (face) {
190            case SOUTH:
191            default:
192                return 0x01;
193            case WEST:
194                return 0x02;
195            case NORTH:
196                return 0x04;
197            case EAST:
198                return 0x08;
199        }
200    }
201
202    @Override
203    public int getToolType() {
204        return ItemTool.TYPE_SHEARS;
205    }
206
207    @Override
208    public BlockColor getColor() {
209        return BlockColor.FOLIAGE_BLOCK_COLOR;
210    }
211}
