001package cn.nukkit.item;
002
003import cn.nukkit.block.Block;
004import cn.nukkit.entity.Entity;
005import cn.nukkit.item.enchantment.Enchantment;
006import cn.nukkit.nbt.tag.ByteTag;
007import cn.nukkit.nbt.tag.Tag;
008
009import java.util.Random;
010
011/**
012 * author: MagicDroidX
013 * Nukkit Project
014 */
015public abstract class ItemTool extends Item implements ItemDurable {
016    public static final int TIER_WOODEN = 1;
017    public static final int TIER_GOLD = 2;
018    public static final int TIER_STONE = 3;
019    public static final int TIER_IRON = 4;
020    public static final int TIER_DIAMOND = 5;
021    public static final int TIER_NETHERITE = 6;
022
023    public static final int TYPE_NONE = 0;
024    public static final int TYPE_SWORD = 1;
025    public static final int TYPE_SHOVEL = 2;
026    public static final int TYPE_PICKAXE = 3;
027    public static final int TYPE_AXE = 4;
028    public static final int TYPE_SHEARS = 5;
029    public static final int TYPE_HOE = 6;
030
031    public static final int DURABILITY_WOODEN = 60;
032    public static final int DURABILITY_GOLD = 33;
033    public static final int DURABILITY_STONE = 132;
034    public static final int DURABILITY_IRON = 251;
035    public static final int DURABILITY_DIAMOND = 1562;
036    public static final int DURABILITY_NETHERITE = 2032;
037    public static final int DURABILITY_FLINT_STEEL = 65;
038    public static final int DURABILITY_SHEARS = 239;
039    public static final int DURABILITY_BOW = 385;
040    public static final int DURABILITY_TRIDENT = 251;
041    public static final int DURABILITY_FISHING_ROD = 65;
042    public static final int DURABILITY_CROSSBOW = 465;
043
044    public ItemTool(int id) {
045        this(id, 0, 1, UNKNOWN_STR);
046    }
047
048    public ItemTool(int id, Integer meta) {
049        this(id, meta, 1, UNKNOWN_STR);
050    }
051
052    public ItemTool(int id, Integer meta, int count) {
053        this(id, meta, count, UNKNOWN_STR);
054    }
055
056    public ItemTool(int id, Integer meta, int count, String name) {
057        super(id, meta, count, name);
058    }
059
060    @Override
061    public int getMaxStackSize() {
062        return 1;
063    }
064
065    @Override
066    public boolean useOn(Block block) {
067        if (this.isUnbreakable() || isDurable()) {
068            return true;
069        }
070
071        if (block.getToolType() == ItemTool.TYPE_PICKAXE && this.isPickaxe() ||
072                block.getToolType() == ItemTool.TYPE_SHOVEL && this.isShovel() ||
073                block.getToolType() == ItemTool.TYPE_AXE && this.isAxe() ||
074                block.getToolType() == ItemTool.TYPE_HOE && this.isHoe() ||
075                block.getToolType() == ItemTool.TYPE_SWORD && this.isSword() ||
076                block.getToolType() == ItemTool.TYPE_SHEARS && this.isShears()
077                ) {
078            this.meta++;
079        } else if (!this.isShears() && block.getBreakTime(this) > 0) {
080            this.meta += 2;
081        } else if (this.isHoe()) {
082            if (block.getId() == GRASS || block.getId() == DIRT) {
083                this.meta++;
084            }
085        } else {
086            this.meta++;
087        }
088        return true;
089    }
090
091    @Override
092    public boolean useOn(Entity entity) {
093        if (this.isUnbreakable() || isDurable()) {
094            return true;
095        }
096
097        if ((entity != null) && !this.isSword()) {
098            this.meta += 2;
099        } else {
100            this.meta++;
101        }
102
103        return true;
104    }
105
106    private boolean isDurable() {
107        if (!hasEnchantments()) {
108            return false;
109        }
110
111        Enchantment durability = getEnchantment(Enchantment.ID_DURABILITY);
112        return durability != null && durability.getLevel() > 0 && (100 / (durability.getLevel() + 1)) <= new Random().nextInt(100);
113    }
114
115    @Override
116    public boolean isUnbreakable() {
117        Tag tag = this.getNamedTagEntry("Unbreakable");
118        return tag instanceof ByteTag && ((ByteTag) tag).data > 0;
119    }
120
121    @Override
122    public boolean isPickaxe() {
123        return false;
124    }
125
126    @Override
127    public boolean isAxe() {
128        return false;
129    }
130
131    @Override
132    public boolean isSword() {
133        return false;
134    }
135
136    @Override
137    public boolean isShovel() {
138        return false;
139    }
140
141    @Override
142    public boolean isHoe() {
143        return false;
144    }
145
146    @Override
147    public boolean isShears() {
148        return (this.id == SHEARS);
149    }
150
151    @Override
152    public boolean isTool() {
153        return (this.id == FLINT_STEEL || this.id == SHEARS || this.id == BOW || this.id == CROSSBOW || this.isPickaxe() || this.isAxe() || this.isShovel() || this.isSword() || this.isHoe());
154    }
155
156    @Override
157    public int getEnchantAbility() {
158        switch (this.getTier()) {
159            case TIER_STONE:
160                return 5;
161            case TIER_WOODEN:
162                return 15;
163            case TIER_DIAMOND:
164                return 10;
165            case TIER_GOLD:
166                return 22;
167            case TIER_IRON:
168                return 14;
169        }
170
171        return 0;
172    }
173}
