001package cn.nukkit.item;
002
003import cn.nukkit.Player;
004import cn.nukkit.block.*;
005import cn.nukkit.entity.Entity;
006import cn.nukkit.event.player.PlayerBucketEmptyEvent;
007import cn.nukkit.event.player.PlayerBucketFillEvent;
008import cn.nukkit.event.player.PlayerItemConsumeEvent;
009import cn.nukkit.level.Level;
010import cn.nukkit.level.particle.ExplodeParticle;
011import cn.nukkit.math.BlockFace;
012import cn.nukkit.math.BlockFace.Plane;
013import cn.nukkit.math.Vector3;
014import cn.nukkit.network.protocol.LevelSoundEventPacket;
015import cn.nukkit.network.protocol.UpdateBlockPacket;
016
017/**
018 * author: MagicDroidX
019 * Nukkit Project
020 */
021public class ItemBucket extends Item {
022
023    public ItemBucket() {
024        this(0, 1);
025    }
026
027    public ItemBucket(Integer meta) {
028        this(meta, 1);
029    }
030
031    public ItemBucket(Integer meta, int count) {
032        super(BUCKET, meta, count, getName(meta));
033    }
034
035    protected static String getName(int meta) {
036        switch (meta) {
037            case 1:
038                return "Milk";
039            case 2:
040                return "Bucket of Cod";
041            case 3:
042                return "Bucket of Salmon";
043            case 4:
044                return "Bucket of Tropical Fish";
045            case 5:
046                return "Bucket of Pufferfish";
047            case 8:
048                return "Water Bucket";
049            case 10:
050                return "Lava Bucket";
051            default:
052                return "Bucket";
053        }
054    }
055
056    public static int getDamageByTarget(int target) {
057        switch (target) {
058            case 2:
059            case 3:
060            case 4:
061            case 5:
062            case 8:
063            case 9:
064                return 8;
065            case 10:
066            case 11:
067                return 10;
068            default:
069                return 0;
070        }
071    }
072
073    @Override
074    public int getMaxStackSize() {
075        return this.meta == 0 ? 16 : 1;
076    }
077
078    @Override
079    public boolean canBeActivated() {
080        return true;
081    }
082
083    @Override
084    public boolean onActivate(Level level, Player player, Block block, Block target, BlockFace face, double fx, double fy, double fz) {
085        if (player.isAdventure()) {
086            return false;
087        }
088
089        Block targetBlock = Block.get(getDamageByTarget(this.meta));
090
091        if (targetBlock instanceof BlockAir) {
092            if (target instanceof BlockLiquid && target.getDamage() == 0) {
093                Item result = Item.get(BUCKET, getDamageByTarget(target.getId()), 1);
094                PlayerBucketFillEvent ev;
095                player.getServer().getPluginManager().callEvent(ev = new PlayerBucketFillEvent(player, block, face, this, result));
096                if (!ev.isCancelled()) {
097                    player.getLevel().setBlock(target, Block.get(BlockID.AIR), true, true);
098
099                    // When water is removed ensure any adjacent still water is
100                    // replaced with water that can flow.
101                    for (BlockFace side : Plane.HORIZONTAL) {
102                        Block b = target.getSide(side);
103                        if (b.getId() == STILL_WATER) {
104                            level.setBlock(b, Block.get(BlockID.WATER));
105                        }
106                    }
107
108                    if (player.isSurvival()) {
109                        if (this.getCount() - 1 <= 0) {
110                            player.getInventory().setItemInHand(ev.getItem());
111                        } else {
112                            Item clone = this.clone();
113                            clone.setCount(this.getCount() - 1);
114                            player.getInventory().setItemInHand(clone);
115                            if (player.getInventory().canAddItem(ev.getItem())) {
116                                player.getInventory().addItem(ev.getItem());
117                            } else {
118                                player.dropItem(ev.getItem());
119                            }
120                        }
121                    }
122
123                    if (target instanceof BlockLava) {
124                        level.addLevelSoundEvent(block, LevelSoundEventPacket.SOUND_BUCKET_FILL_LAVA);
125                    } else {
126                        level.addLevelSoundEvent(block, LevelSoundEventPacket.SOUND_BUCKET_FILL_WATER);
127                    }
128
129                    return true;
130                } else {
131                    player.getInventory().sendContents(player);
132                }
133            }
134        } else if (targetBlock instanceof BlockLiquid) {
135            Item result = Item.get(BUCKET, 0, 1);
136            PlayerBucketEmptyEvent ev = new PlayerBucketEmptyEvent(player, block, face, this, result);
137            if (!block.canBeFlowedInto()) {
138                ev.setCancelled(true);
139            }
140
141            boolean nether = false;
142            if (player.getLevel().getDimension() == Level.DIMENSION_NETHER && this.getDamage() != 10) {
143                ev.setCancelled(true);
144                nether = true;
145            }
146
147            player.getServer().getPluginManager().callEvent(ev);
148
149            if (!ev.isCancelled()) {
150                player.getLevel().setBlock(block, targetBlock, true, true);
151                if (player.isSurvival()) {
152                    if (this.getCount() - 1 <= 0) {
153                        player.getInventory().setItemInHand(ev.getItem());
154                    } else {
155                        Item clone = this.clone();
156                        clone.setCount(this.getCount() - 1);
157                        player.getInventory().setItemInHand(clone);
158                        if (player.getInventory().canAddItem(ev.getItem())) {
159                            player.getInventory().addItem(ev.getItem());
160                        } else {
161                            player.dropItem(ev.getItem());
162                        }
163                    }
164                }
165
166                if (this.getDamage() == 10) {
167                    level.addLevelSoundEvent(block, LevelSoundEventPacket.SOUND_BUCKET_EMPTY_LAVA);
168                } else {
169                    level.addLevelSoundEvent(block, LevelSoundEventPacket.SOUND_BUCKET_EMPTY_WATER);
170                }
171
172                switch (this.getDamage()) {
173                    case 2:
174                        Entity e2 = Entity.createEntity("Cod", block);
175                        if (e2 != null) e2.spawnToAll();
176                        break;
177                    case 3:
178                        Entity e3 = Entity.createEntity("Salmon", block);
179                        if (e3 != null) e3.spawnToAll();
180                        break;
181                    case 4:
182                        Entity e4 = Entity.createEntity("TropicalFish", block);
183                        if (e4 != null) e4.spawnToAll();
184                        break;
185                    case 5:
186                        Entity e5 = Entity.createEntity("Pufferfish", block);
187                        if (e5 != null) e5.spawnToAll();
188                        break;
189                }
190
191                return true;
192            } else if (nether) {
193                if (!player.isCreative()) {
194                    this.setDamage(0); // Empty bucket
195                    player.getInventory().setItemInHand(this);
196                }
197                player.getLevel().addLevelSoundEvent(target, LevelSoundEventPacket.SOUND_FIZZ);
198                player.getLevel().addParticle(new ExplodeParticle(target.add(0.5, 1, 0.5)));
199            } else {
200                player.getLevel().sendBlocks(new Player[]{player}, new Block[]{Block.get(Block.AIR, 0, block)}, UpdateBlockPacket.FLAG_ALL_PRIORITY, 1);
201                player.getInventory().sendContents(player);
202            }
203        }
204
205        return false;
206    }
207
208    @Override
209    public boolean onClickAir(Player player, Vector3 directionVector) {
210        return this.getDamage() == 1; // Milk
211    }
212
213    @Override
214    public boolean onUse(Player player, int ticksUsed) {
215        if (player.isSpectator() || this.getDamage() != 1) {
216            return false;
217        }
218
219        PlayerItemConsumeEvent consumeEvent = new PlayerItemConsumeEvent(player, this);
220
221        player.getServer().getPluginManager().callEvent(consumeEvent);
222        if (consumeEvent.isCancelled()) {
223            player.getInventory().sendContents(player);
224            return false;
225        }
226
227        if (!player.isCreative()) {
228            player.getInventory().setItemInHand(new ItemBucket());
229        }
230
231        player.removeAllEffects();
232        return true;
233    }
234}
