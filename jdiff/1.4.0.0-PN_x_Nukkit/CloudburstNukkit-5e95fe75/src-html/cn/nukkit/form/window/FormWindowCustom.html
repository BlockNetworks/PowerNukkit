001package cn.nukkit.form.window;
002
003import cn.nukkit.form.element.*;
004import cn.nukkit.form.response.FormResponseCustom;
005import cn.nukkit.form.response.FormResponseData;
006import com.google.gson.Gson;
007import com.google.gson.reflect.TypeToken;
008
009import java.util.ArrayList;
010import java.util.HashMap;
011import java.util.List;
012
013public class FormWindowCustom extends FormWindow {
014
015    private final String type = "custom_form"; //This variable is used for JSON import operations. Do NOT delete :) -- @Snake1999
016    private String title = "";
017    private ElementButtonImageData icon;
018    private List<Element> content;
019
020    private FormResponseCustom response;
021
022    public FormWindowCustom(String title) {
023        this(title, new ArrayList<>());
024    }
025
026    public FormWindowCustom(String title, List<Element> contents) {
027        this(title, contents, (ElementButtonImageData) null);
028    }
029
030    public FormWindowCustom(String title, List<Element> contents, String icon) {
031        this(title, contents, icon.isEmpty() ? null : new ElementButtonImageData(ElementButtonImageData.IMAGE_DATA_TYPE_URL, icon));
032    }
033
034    public FormWindowCustom(String title, List<Element> contents, ElementButtonImageData icon) {
035        this.title = title;
036        this.content = contents;
037        this.icon = icon;
038    }
039
040    public String getTitle() {
041        return title;
042    }
043
044    public void setTitle(String title) {
045        this.title = title;
046    }
047
048    public List<Element> getElements() {
049        return content;
050    }
051
052    public void addElement(Element element) {
053        content.add(element);
054    }
055
056    public ElementButtonImageData getIcon() {
057        return icon;
058    }
059
060    public void setIcon(String icon) {
061        if (!icon.isEmpty()) this.icon = new ElementButtonImageData(ElementButtonImageData.IMAGE_DATA_TYPE_URL, icon);
062    }
063
064    public void setIcon(ElementButtonImageData icon) {
065        this.icon = icon;
066    }
067
068    public FormResponseCustom getResponse() {
069        return response;
070    }
071
072    public void setResponse(String data) {
073        if (data.equals("null")) {
074            this.closed = true;
075            return;
076        }
077
078        List<String> elementResponses = new Gson().fromJson(data, new TypeToken<List<String>>() {
079        }.getType());
080        //elementResponses.remove(elementResponses.size() - 1); //submit button //maybe mojang removed that?
081
082        int i = 0;
083
084        HashMap<Integer, FormResponseData> dropdownResponses = new HashMap<>();
085        HashMap<Integer, String> inputResponses = new HashMap<>();
086        HashMap<Integer, Float> sliderResponses = new HashMap<>();
087        HashMap<Integer, FormResponseData> stepSliderResponses = new HashMap<>();
088        HashMap<Integer, Boolean> toggleResponses = new HashMap<>();
089        HashMap<Integer, Object> responses = new HashMap<>();
090        HashMap<Integer, String> labelResponses = new HashMap<>();
091
092        for (String elementData : elementResponses) {
093            if (i >= content.size()) {
094                break;
095            }
096
097            Element e = content.get(i);
098            if (e == null) break;
099            if (e instanceof ElementLabel) {
100                labelResponses.put(i, ((ElementLabel) e).getText());
101                responses.put(i, ((ElementLabel) e).getText());
102            } else if (e instanceof ElementDropdown) {
103                String answer = ((ElementDropdown) e).getOptions().get(Integer.parseInt(elementData));
104                dropdownResponses.put(i, new FormResponseData(Integer.parseInt(elementData), answer));
105                responses.put(i, answer);
106            } else if (e instanceof ElementInput) {
107                inputResponses.put(i, elementData);
108                responses.put(i, elementData);
109            } else if (e instanceof ElementSlider) {
110                Float answer = Float.parseFloat(elementData);
111                sliderResponses.put(i, answer);
112                responses.put(i, answer);
113            } else if (e instanceof ElementStepSlider) {
114                String answer = ((ElementStepSlider) e).getSteps().get(Integer.parseInt(elementData));
115                stepSliderResponses.put(i, new FormResponseData(Integer.parseInt(elementData), answer));
116                responses.put(i, answer);
117            } else if (e instanceof ElementToggle) {
118                Boolean answer = Boolean.parseBoolean(elementData);
119                toggleResponses.put(i, answer);
120                responses.put(i, answer);
121            }
122            i++;
123        }
124
125        this.response = new FormResponseCustom(responses, dropdownResponses, inputResponses,
126                sliderResponses, stepSliderResponses, toggleResponses, labelResponses);
127    }
128
129    /**
130     * Set Elements from Response
131     * Used on ServerSettings Form Response. After players set settings, we need to sync these settings to the server.
132     */
133    public void setElementsFromResponse() {
134        if (this.response != null) {
135            this.response.getResponses().forEach((i, response) -> {
136                Element e = content.get(i);
137                if (e != null) {
138                    if (e instanceof ElementDropdown) {
139                        ((ElementDropdown) e).setDefaultOptionIndex(((ElementDropdown) e).getOptions().indexOf(response));
140                    } else if (e instanceof ElementInput) {
141                        ((ElementInput) e).setDefaultText((String)response);
142                    } else if (e instanceof ElementSlider) {
143                        ((ElementSlider) e).setDefaultValue((Float)response);
144                    } else if (e instanceof ElementStepSlider) {
145                        ((ElementStepSlider) e).setDefaultOptionIndex(((ElementStepSlider) e).getSteps().indexOf(response));
146                    } else if (e instanceof ElementToggle) {
147                        ((ElementToggle) e).setDefaultValue((Boolean)response);
148                    }
149                }
150            });
151        }
152    }
153
154}
