001package cn.nukkit.utils;
002
003import org.apache.logging.log4j.util.TriConsumer;
004
005import java.util.function.BiConsumer;
006
007/**
008 * author: MagicDroidX
009 * Nukkit Project
010 */
011public enum LogLevel implements Comparable<LogLevel> {
012    NONE((logger, message) -> {}, (mainLogger, s, throwable) -> {}),
013    EMERGENCY(MainLogger::emergency, MainLogger::emergency),
014    ALERT(MainLogger::alert, MainLogger::alert),
015    CRITICAL(MainLogger::critical, MainLogger::critical),
016    ERROR(MainLogger::error, MainLogger::error),
017    WARNING(MainLogger::warning, MainLogger::warning),
018    NOTICE(MainLogger::notice, MainLogger::notice),
019    INFO(MainLogger::info, MainLogger::info),
020    DEBUG(MainLogger::debug, MainLogger::debug);
021
022    public static final LogLevel DEFAULT_LEVEL = INFO;
023
024    private final BiConsumer<MainLogger, String> logTo;
025    private final TriConsumer<MainLogger, String, Throwable> logThrowableTo;
026
027    LogLevel(BiConsumer<MainLogger, String> logTo, TriConsumer<MainLogger, String, Throwable> logThrowableTo) {
028        this.logTo = logTo;
029        this.logThrowableTo = logThrowableTo;
030    }
031
032    public void log(MainLogger logger, String message) {
033        logTo.accept(logger, message);
034    }
035
036    public void log(MainLogger logger, String message, Throwable throwable) {
037        logThrowableTo.accept(logger, message, throwable);
038    }
039
040    public int getLevel() {
041        return ordinal();
042    }
043}
