001package cn.nukkit.utils;

003import cn.nukkit.api.API;
004import cn.nukkit.block.Block;
005import cn.nukkit.math.BlockFace;
006
007import java.util.Arrays;
008import java.util.List;
009import java.util.Objects;
010import java.util.Optional;
011import java.util.stream.Stream;
012
013import static cn.nukkit.math.BlockFace.*;
014import static cn.nukkit.utils.Rail.Orientation.State.*;
015
016/**

022@API(usage = API.Usage.BLEEDING, definition = API.Definition.INTERNAL)
023public final class Rail {
024
025    public static boolean isRailBlock(Block block) {
026        Objects.requireNonNull(block, "Rail block predicate can not accept null block");
027        return isRailBlock(block.getId());
028    }
029
030    public enum Orientation {
031        STRAIGHT_NORTH_SOUTH(0, STRAIGHT, NORTH, SOUTH, null),
032        STRAIGHT_EAST_WEST(1, STRAIGHT, EAST, WEST, null),
033        ASCENDING_EAST(2, ASCENDING, EAST, WEST, EAST),
034        ASCENDING_WEST(3, ASCENDING, EAST, WEST, WEST),
035        ASCENDING_NORTH(4, ASCENDING, NORTH, SOUTH, NORTH),
036        ASCENDING_SOUTH(5, ASCENDING, NORTH, SOUTH, SOUTH),
037        CURVED_SOUTH_EAST(6, CURVED, SOUTH, EAST, null),
038        CURVED_SOUTH_WEST(7, CURVED, SOUTH, WEST, null),
039        CURVED_NORTH_WEST(8, CURVED, NORTH, WEST, null),
040        CURVED_NORTH_EAST(9, CURVED, NORTH, EAST, null);
041
042        private static final Orientation[] META_LOOKUP = new Orientation[values().length];
043        private final int meta;
044        private final State state;
045        private final List<BlockFace> connectingDirections;
046        private final BlockFace ascendingDirection;
047
048        Orientation(int meta, State state, BlockFace from, BlockFace to, BlockFace ascendingDirection) {
049            this.meta = meta;
050            this.state = state;
051            this.connectingDirections = Arrays.asList(from, to);
052            this.ascendingDirection = ascendingDirection;
053        }
054
055        public static Orientation byMetadata(int meta) {
056            if (meta < 0 || meta >= META_LOOKUP.length) {
057                meta = 0;
058            }
059
060            return META_LOOKUP[meta];
061        }
062
063        public static Orientation straight(BlockFace face) {
064            switch (face) {
065                case NORTH:
066                case SOUTH:
067                    return STRAIGHT_NORTH_SOUTH;
068                case EAST:
069                case WEST:
070                    return STRAIGHT_EAST_WEST;
071            }
072            return STRAIGHT_NORTH_SOUTH;
073        }
074
075        public static Orientation ascending(BlockFace face) {
076            switch (face) {
077                case NORTH:
078                    return ASCENDING_NORTH;
079                case SOUTH:
080                    return ASCENDING_SOUTH;
081                case EAST:
082                    return ASCENDING_EAST;
083                case WEST:
084                    return ASCENDING_WEST;
085            }
086            return ASCENDING_EAST;
087        }
088
089        public static Orientation curved(BlockFace f1, BlockFace f2) {
090            for (Orientation o : new Orientation[]{CURVED_SOUTH_EAST, CURVED_SOUTH_WEST, CURVED_NORTH_WEST, CURVED_NORTH_EAST}) {
091                if (o.connectingDirections.contains(f1) && o.connectingDirections.contains(f2)) {
092                    return o;
093                }
094            }
095            return CURVED_SOUTH_EAST;
096        }
097
098        public static Orientation straightOrCurved(BlockFace f1, BlockFace f2) {
099            for (Orientation o : new Orientation[]{STRAIGHT_NORTH_SOUTH, STRAIGHT_EAST_WEST, CURVED_SOUTH_EAST, CURVED_SOUTH_WEST, CURVED_NORTH_WEST, CURVED_NORTH_EAST}) {
100                if (o.connectingDirections.contains(f1) && o.connectingDirections.contains(f2)) {
101                    return o;
102                }
103            }
104            return STRAIGHT_NORTH_SOUTH;
105        }
106
107        public int metadata() {
108            return meta;
109        }
110
111        public boolean hasConnectingDirections(BlockFace... faces) {
112            return Stream.of(faces).allMatch(connectingDirections::contains);
113        }
114
115        public List<BlockFace> connectingDirections() {
116            return connectingDirections;
117        }
118
119        public Optional<BlockFace> ascendingDirection() {
120            return Optional.ofNullable(ascendingDirection);
121        }
122
123        public enum State {
124            STRAIGHT, ASCENDING, CURVED
125        }
126
127        public boolean isStraight() {
128            return state == STRAIGHT;
129        }
130
131        public boolean isAscending() {
132            return state == ASCENDING;
133        }
134
135        public boolean isCurved() {
136            return state == CURVED;
137        }
138
139        static {
140            for (Orientation o : values()) {
141                META_LOOKUP[o.meta] = o;
142            }
143        }
144    }
145
146    public static boolean isRailBlock(int blockId) {
147        switch (blockId) {
148            case Block.RAIL:
149            case Block.POWERED_RAIL:
150            case Block.ACTIVATOR_RAIL:
151            case Block.DETECTOR_RAIL:
152                return true;
153            default:
154                return false;
155        }
156    }
157
158    private Rail() {
159        //no instance
160    }
161}
