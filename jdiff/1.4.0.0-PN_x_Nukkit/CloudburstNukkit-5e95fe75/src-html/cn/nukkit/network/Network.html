107            FastByteArrayOutputStream bos = ThreadCache.fbaos.get();

222    public void processBatch(byte[] payload, Collection<DataPacket> packets) throws ProtocolException {
223        byte[] data;
224        try {
225            data = Network.inflateRaw(payload);
226        } catch (
227        //data = Zlib.inflate(packet.payload, 2 * 1024 * 1024); // Max 2MB
228        Exception e) {
229            log.debug("Exception while inflating batch packet", e);
230            return;
231        }
232        BinaryStream stream = new BinaryStream(data);
233        try {
234            int count = 0;
235            while (!stream.feof()) {
236                count++;
237                if (count >= 1000) {
238                    throw new ProtocolException("Illegal batch with " + count + " packets");
239                }
240                byte[] buf = stream.getByteArray();
241                ByteArrayInputStream bais = new ByteArrayInputStream(buf);
242                int header = (int) VarInt.readUnsignedVarInt(bais);
243                // | Client ID | Sender ID | Packet ID |
244                // |   2 bits  |   2 bits  |  10 bits  |
245                int packetId = header & 1023;
246                DataPacket pk = this.getPacket(packetId);
247                if (pk != null) {
248                    pk.setBuffer(buf, buf.length - bais.available());
249                    try {
250                        pk.decode();
251                    } catch (Exception e) {
252                        if (log.isTraceEnabled()) {
253                            log.trace("Dumping Packet\n{}", ByteBufUtil.prettyHexDump(Unpooled.wrappedBuffer(buf)));
254                        }
255                        log.error("Unable to decode packet", e);
256                        throw new IllegalStateException("Unable to decode " + pk.getClass().getSimpleName());
257                    }
258                    packets.add(pk);
259                } else {
260                    log.debug("Received unknown packet with ID: {}", Integer.toHexString(packetId));
261                }
262            }
263        } catch (Exception e) {
