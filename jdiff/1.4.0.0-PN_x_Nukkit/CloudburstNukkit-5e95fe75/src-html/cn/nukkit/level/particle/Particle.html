095    public static final Integer getParticleIdByName(String name) {
096        name = name.toUpperCase();
097
098        try {
099            Field field = Particle.class.getField((name.startsWith("TYPE_") == true ? name : ("TYPE_" + name)));
100
101            Class<?> type = field.getType();
102
103            if(type==int.class) {
104                return field.getInt(null);
105            }
106        } catch(NoSuchFieldException | IllegalAccessException e) {
107            // ignore
108        }
109        return null;
110    }
111
112    public static final boolean particleExists(String name) {
113        return getParticleIdByName(name) != null;   
114    }
