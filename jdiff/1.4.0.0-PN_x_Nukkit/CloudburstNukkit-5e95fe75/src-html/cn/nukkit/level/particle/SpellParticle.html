001package cn.nukkit.level.particle;
002
003import cn.nukkit.math.Vector3;
004import cn.nukkit.network.protocol.DataPacket;
005import cn.nukkit.network.protocol.LevelEventPacket;
006import cn.nukkit.utils.BlockColor;
007
008/**
009 * Created on 2015/12/27 by xtypr.
010 * Package cn.nukkit.level.particle in project Nukkit .
011 * The name "spell" comes from minecraft wiki.
012 */
013public class SpellParticle extends Particle {
014
015    protected final int data;
016
017    public SpellParticle(Vector3 pos) {
018        this(pos, 0);
019    }
020
021    public SpellParticle(Vector3 pos, int data) {
022        super(pos.x, pos.y, pos.z);
023        this.data = data;
024    }
025
026    public SpellParticle(Vector3 pos, BlockColor blockColor) {
027        //alpha is ignored
028        this(pos, blockColor.getRed(), blockColor.getGreen(), blockColor.getBlue());
029    }
030
031    public SpellParticle(Vector3 pos, int r, int g, int b) {
032        this(pos, r, g, b, 0x00);
033    }
034
035    protected SpellParticle(Vector3 pos, int r, int g, int b, int a) {
036        this(pos, ((a & 0xff) << 24) | ((r & 0xff) << 16) | ((g & 0xff) << 8) | (b & 0xff));
037    }
038
039    @Override
040    public DataPacket[] encode() {
041        LevelEventPacket pk = new LevelEventPacket();
042        pk.evid = LevelEventPacket.EVENT_PARTICLE_SPLASH;
043        pk.x = (float) this.x;
044        pk.y = (float) this.y;
045        pk.z = (float) this.z;
046        pk.data = this.data;
047
048        return new DataPacket[]{pk};
049    }
050}
