001package cn.nukkit.level.generator.populator.impl;
002
003import cn.nukkit.block.Block;
004import cn.nukkit.block.BlockSapling;
005import cn.nukkit.level.ChunkManager;
006import cn.nukkit.level.format.FullChunk;
007import cn.nukkit.level.generator.object.tree.ObjectTree;
008import cn.nukkit.level.generator.populator.type.PopulatorCount;
009import cn.nukkit.math.NukkitMath;
010import cn.nukkit.math.NukkitRandom;
011
012/**
013 * author: DaPorkchop_
014 * Nukkit Project
015 */
016public class PopulatorTree extends PopulatorCount {
017
018    private final int type;
019    private ChunkManager level;
020
021    public PopulatorTree() {
022        this(BlockSapling.OAK);
023    }
024
025    public PopulatorTree(int type) {
026        this.type = type;
027    }
028
029    @Override
030    public void populateCount(ChunkManager level, int chunkX, int chunkZ, NukkitRandom random, FullChunk chunk) {
031        this.level = level;
032        int x = NukkitMath.randomRange(random, chunkX << 4, (chunkX << 4) + 15);
033        int z = NukkitMath.randomRange(random, chunkZ << 4, (chunkZ << 4) + 15);
034        int y = this.getHighestWorkableBlock(x, z);
035        if (y < 3) {
036            return;
037        }
038        ObjectTree.growTree(this.level, x, y, z, random, this.type);
039    }
040
041    private int getHighestWorkableBlock(int x, int z) {
042        int y;
043        for (y = 254; y > 0; --y) {
044            int b = this.level.getBlockIdAt(x, y, z);
045            if (b == Block.DIRT || b == Block.GRASS) {
046                break;
047            } else if (b != Block.AIR && b != Block.SNOW_LAYER) {
048                return -1;
049            }
050        }
051
052        return ++y;
053    }
054}
