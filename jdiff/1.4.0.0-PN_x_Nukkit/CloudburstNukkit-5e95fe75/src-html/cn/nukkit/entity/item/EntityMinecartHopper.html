001package cn.nukkit.entity.item;
002
003import cn.nukkit.Player;
004import cn.nukkit.block.Block;
005import cn.nukkit.entity.Entity;
006import cn.nukkit.inventory.InventoryHolder;
007import cn.nukkit.inventory.MinecartHopperInventory;
008import cn.nukkit.item.Item;
009import cn.nukkit.level.format.FullChunk;
010import cn.nukkit.math.Vector3;
011import cn.nukkit.nbt.NBTIO;
012import cn.nukkit.nbt.tag.CompoundTag;
013import cn.nukkit.nbt.tag.ListTag;
014import cn.nukkit.utils.MinecartType;
015
016public class EntityMinecartHopper extends EntityMinecartAbstract implements InventoryHolder {
017
018    public static final int NETWORK_ID = 96;
019
020    protected MinecartHopperInventory inventory;
021
022    public EntityMinecartHopper(FullChunk chunk, CompoundTag nbt) {
023        super(chunk, nbt);
024        setDisplayBlock(Block.get(Block.HOPPER_BLOCK), false);
025    }
026
027    @Override
028    public String getName() {
029        return getType().getName();
030    }
031
032    @Override
033    public MinecartType getType() {
034        return MinecartType.valueOf(5);
035    }
036
037    @Override
038    public boolean isRideable(){
039        return false;
040    }
041
042    @Override
043    public int getNetworkId() {
044        return NETWORK_ID;
045    }
046
047    @Override
048    public void dropItem() {
049        super.dropItem();
050
051        this.level.dropItem(this, Item.get(Item.HOPPER));
052        for (Item item : this.inventory.getContents().values()) {
053            this.level.dropItem(this, item);
054        }
055        this.inventory.clearAll();
056    }
057
058    @Override
059    public boolean mountEntity(Entity entity, byte mode) {
060        return false;
061    }
062
063    @Override
064    public boolean onInteract(Player player, Item item, Vector3 clickedPos) {
065        player.addWindow(this.inventory);
066        return false; // If true, the count of items player has in hand decreases
067    }
068
069    @Override
070    public MinecartHopperInventory getInventory() {
071        return inventory;
072    }
073
074    @Override
075    public void initEntity() {
076        super.initEntity();
077
078        this.inventory = new MinecartHopperInventory(this);
079        if (this.namedTag.contains("Items") && this.namedTag.get("Items") instanceof ListTag) {
080            ListTag<CompoundTag> inventoryList = this.namedTag.getList("Items", CompoundTag.class);
081            for (CompoundTag item : inventoryList.getAll()) {
082                this.inventory.setItem(item.getByte("Slot"), NBTIO.getItemHelper(item));
083            }
084        }
085
086        this.dataProperties
087                .putByte(DATA_CONTAINER_TYPE, 11)
088                .putInt(DATA_CONTAINER_BASE_SIZE, this.inventory.getSize())
089                .putInt(DATA_CONTAINER_EXTRA_SLOTS_PER_STRENGTH, 0);
090    }
091
092    @Override
093    public void saveNBT() {
094        super.saveNBT();
095
096        this.namedTag.putList(new ListTag<CompoundTag>("Items"));
097        if (this.inventory != null) {
098            for (int slot = 0; slot < 5; ++slot) {
099                Item item = this.inventory.getItem(slot);
100                if (item != null && item.getId() != Item.AIR) {
101                    this.namedTag.getList("Items", CompoundTag.class)
102                            .add(NBTIO.putItemHelper(item, slot));
103                }
104            }
105        }
106    }
107}
