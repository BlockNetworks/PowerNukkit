001package cn.nukkit.entity.item;
002
003import cn.nukkit.Server;
004import cn.nukkit.entity.Entity;
005import cn.nukkit.entity.data.ByteEntityData;
006import cn.nukkit.entity.data.IntEntityData;
007import cn.nukkit.entity.data.NBTEntityData;
008import cn.nukkit.event.entity.EntityDamageEvent;
009import cn.nukkit.event.entity.EntityDamageEvent.DamageCause;
010import cn.nukkit.item.Item;
011import cn.nukkit.item.ItemFirework;
012import cn.nukkit.level.format.FullChunk;
013import cn.nukkit.nbt.NBTIO;
014import cn.nukkit.nbt.tag.CompoundTag;
015import cn.nukkit.network.protocol.EntityEventPacket;
016import cn.nukkit.network.protocol.LevelSoundEventPacket;
017
018import java.util.Random;
019
020/**
021 * @author CreeperFace
022 */
023public class EntityFirework extends Entity {
024
025    public static final int NETWORK_ID = 72;
026
027    private int fireworkAge;
028    private int lifetime;
029    private Item firework;
030
031    public EntityFirework(FullChunk chunk, CompoundTag nbt) {
032        super(chunk, nbt);
033
034        this.fireworkAge = 0;
035        Random rand = new Random();
036        this.lifetime = 30 + rand.nextInt(6) + rand.nextInt(7);
037
038        this.motionX = rand.nextGaussian() * 0.001D;
039        this.motionZ = rand.nextGaussian() * 0.001D;
040        this.motionY = 0.05D;
041
042        if (nbt.contains("FireworkItem")) {
043            firework = NBTIO.getItemHelper(nbt.getCompound("FireworkItem"));
044        } else {
045            firework = new ItemFirework();
046        }
047
048        this.setDataProperty(new NBTEntityData(Entity.DATA_DISPLAY_ITEM, firework.getNamedTag()));
049        this.setDataProperty(new IntEntityData(Entity.DATA_DISPLAY_OFFSET, 1));
050        this.setDataProperty(new ByteEntityData(Entity.DATA_HAS_DISPLAY, 1));
051    }
052
053    @Override
054    public int getNetworkId() {
055        return NETWORK_ID;
056    }
057
058    @Override
059    public boolean onUpdate(int currentTick) {
060        if (this.closed) {
061            return false;
062        }
063
064        int tickDiff = currentTick - this.lastUpdate;
065
066        if (tickDiff <= 0 && !this.justCreated) {
067            return true;
068        }
069
070        this.lastUpdate = currentTick;
071
072        this.timing.startTiming();
073
074
075        boolean hasUpdate = this.entityBaseTick(tickDiff);
076
077        if (this.isAlive()) {
078
079            this.motionX *= 1.15D;
080            this.motionZ *= 1.15D;
081            this.motionY += 0.04D;
082            this.move(this.motionX, this.motionY, this.motionZ);
083
084            this.updateMovement();
085
086
087            float f = (float) Math.sqrt(this.motionX * this.motionX + this.motionZ * this.motionZ);
088            this.yaw = (float) (Math.atan2(this.motionX, this.motionZ) * (180D / Math.PI));
089
090            this.pitch = (float) (Math.atan2(this.motionY, f) * (180D / Math.PI));
091
092
093            if (this.fireworkAge == 0) {
094                this.getLevel().addLevelSoundEvent(this, LevelSoundEventPacket.SOUND_LAUNCH);
095            }
096
097            this.fireworkAge++;
098
099            hasUpdate = true;
100            if (this.fireworkAge >= this.lifetime) {
101                EntityEventPacket pk = new EntityEventPacket();
102                pk.data = 0;
103                pk.event = EntityEventPacket.FIREWORK_EXPLOSION;
104                pk.eid = this.getId();
105
106                level.addLevelSoundEvent(this, LevelSoundEventPacket.SOUND_LARGE_BLAST, -1, NETWORK_ID);
107
108                Server.broadcastPacket(getViewers().values(), pk);
109
110                this.kill();
111                hasUpdate = true;
112            }
113        }
114
115        this.timing.stopTiming();
116
117        return hasUpdate || !this.onGround || Math.abs(this.motionX) > 0.00001 || Math.abs(this.motionY) > 0.00001 || Math.abs(this.motionZ) > 0.00001;
118    }
119
120    @Override
121    public boolean attack(EntityDamageEvent source) {
122        return (source.getCause() == DamageCause.VOID ||
123                source.getCause() == DamageCause.FIRE_TICK ||
124                source.getCause() == DamageCause.ENTITY_EXPLOSION ||
125                source.getCause() == DamageCause.BLOCK_EXPLOSION)
126                && super.attack(source);
127    }
128
129    public void setFirework(Item item) {
130        this.firework = item;
131        this.setDataProperty(new NBTEntityData(Entity.DATA_DISPLAY_ITEM, item.getNamedTag()));
132    }
133
134    @Override
135    public float getWidth() {
136        return 0.25f;
137    }
138
139    @Override
140    public float getHeight() {
141        return 0.25f;
142    }
143}
