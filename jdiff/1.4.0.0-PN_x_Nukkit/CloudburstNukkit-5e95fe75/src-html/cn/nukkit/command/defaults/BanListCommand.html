001package cn.nukkit.command.defaults;
002
003import cn.nukkit.command.CommandSender;
004import cn.nukkit.command.data.CommandEnum;
005import cn.nukkit.command.data.CommandParameter;
006import cn.nukkit.lang.TranslationContainer;
007import cn.nukkit.permission.BanEntry;
008import cn.nukkit.permission.BanList;
009
010import java.util.Iterator;
011
012/**
013 * Created on 2015/11/11 by xtypr.
014 * Package cn.nukkit.command.defaults in project Nukkit .
015 */
016public class BanListCommand extends VanillaCommand {
017    public BanListCommand(String name) {
018        super(name, "%nukkit.command.banlist.description", "%commands.banlist.usage");
019        this.setPermission("nukkit.command.ban.list");
020        this.commandParameters.clear();
021        this.commandParameters.put("default", new CommandParameter[]{
022                CommandParameter.newEnum("type", true, new CommandEnum("BanListType", "ips", "players"))
023        });
024    }
025
026    @Override
027    public boolean execute(CommandSender sender, String commandLabel, String[] args) {
028        if (!this.testPermission(sender)) {
029            return true;
030        }
031
032        BanList list;
033        boolean ips = false;
034        if (args.length > 0) {
035            switch (args[0].toLowerCase()) {
036                case "ips":
037                    list = sender.getServer().getIPBans();
038                    ips = true;
039                    break;
040                case "players":
041                    list = sender.getServer().getNameBans();
042                    break;
043                default:
044                    sender.sendMessage(new TranslationContainer("commands.generic.usage", this.usageMessage));
045                    return false;
046            }
047        } else {
048            list = sender.getServer().getNameBans();
049        }
050
051        StringBuilder builder = new StringBuilder();
052        Iterator<BanEntry> itr = list.getEntires().values().iterator();
053        while (itr.hasNext()) {
054            builder.append(itr.next().getName());
055            if (itr.hasNext()) {
056                builder.append(", ");
057            }
058        }
059
060        if (ips) {
061            sender.sendMessage(new TranslationContainer("commands.banlist.ips", String.valueOf(list.getEntires().size())));
062        } else {
063            sender.sendMessage(new TranslationContainer("commands.banlist.players", String.valueOf(list.getEntires().size())));
064        }
065        sender.sendMessage(builder.toString());
066        return true;
067    }
068}
