001package cn.nukkit.command.defaults;
002
003import cn.nukkit.Player;
004import cn.nukkit.command.Command;
005import cn.nukkit.command.CommandSender;
006import cn.nukkit.command.data.CommandEnum;
007import cn.nukkit.command.data.CommandParamType;
008import cn.nukkit.command.data.CommandParameter;
009import cn.nukkit.lang.TranslationContainer;
010import cn.nukkit.level.Level;
011import cn.nukkit.utils.TextFormat;
012
013/**
014 * Created on 2015/11/11 by xtypr.
015 * Package cn.nukkit.command.defaults in project Nukkit .
016 */
017public class TimeCommand extends VanillaCommand {
018
019    public TimeCommand(String name) {
020        super(name, "%nukkit.command.time.description", "%nukkit.command.time.usage");
021        this.setPermission("nukkit.command.time.add;" +
022                "nukkit.command.time.set;" +
023                "nukkit.command.time.start;" +
024                "nukkit.command.time.stop");
025        this.commandParameters.clear();
026        this.commandParameters.put("1arg", new CommandParameter[]{
027                CommandParameter.newEnum("mode", new CommandEnum("TimeMode", "query", "start", "stop"))
028        });
029        this.commandParameters.put("add", new CommandParameter[]{
030                CommandParameter.newEnum("mode", new CommandEnum("TimeModeAdd", "add")),
031                CommandParameter.newType("amount", CommandParamType.INT)
032        });
033        this.commandParameters.put("setAmount", new CommandParameter[]{
034                CommandParameter.newEnum("mode", false, new CommandEnum("TimeModeSet", "set")),
035                CommandParameter.newType("amount", CommandParamType.INT)
036        });
037        this.commandParameters.put("setTime", new CommandParameter[]{
038                CommandParameter.newEnum("mode", new CommandEnum("TimeModeSet", "set")),
039                CommandParameter.newEnum("time", new CommandEnum("TimeSpec", "day", "night", "midnight", "noon", "sunrise", "sunset"))
040        });
041    }
042
043    @Override
044    public boolean execute(CommandSender sender, String commandLabel, String[] args) {
045        if (args.length < 1) {
046            sender.sendMessage(new TranslationContainer("commands.generic.usage", this.usageMessage));
047
048            return false;
049        }
050
051        if ("start".equals(args[0])) {
052            if (!sender.hasPermission("nukkit.command.time.start")) {
053                sender.sendMessage(new TranslationContainer(TextFormat.RED + "%commands.generic.permission"));
054
055                return true;
056            }
057            for (Level level : sender.getServer().getLevels().values()) {
058                level.checkTime();
059                level.startTime();
060                level.checkTime();
061            }
062            Command.broadcastCommandMessage(sender, "Restarted the time");
063            return true;
064        } else if ("stop".equals(args[0])) {
065            if (!sender.hasPermission("nukkit.command.time.stop")) {
066                sender.sendMessage(new TranslationContainer(TextFormat.RED + "%commands.generic.permission"));
067
068                return true;
069            }
070            for (Level level : sender.getServer().getLevels().values()) {
071                level.checkTime();
072                level.stopTime();
073                level.checkTime();
074            }
075            Command.broadcastCommandMessage(sender, "Stopped the time");
076            return true;
077        } else if ("query".equals(args[0])) {
078            if (!sender.hasPermission("nukkit.command.time.query")) {
079                sender.sendMessage(new TranslationContainer(TextFormat.RED + "%commands.generic.permission"));
080
081                return true;
082            }
083            Level level;
084            if (sender instanceof Player) {
085                level = ((Player) sender).getLevel();
086            } else {
087                level = sender.getServer().getDefaultLevel();
088            }
089            sender.sendMessage(new TranslationContainer("commands.time.query.gametime", String.valueOf(level.getTime())));
090            return true;
091        }
092
093
094        if (args.length < 2) {
095            sender.sendMessage(new TranslationContainer("commands.generic.usage", this.usageMessage));
096
097            return false;
098        }
099
100        if ("set".equals(args[0])) {
101            if (!sender.hasPermission("nukkit.command.time.set")) {
102                sender.sendMessage(new TranslationContainer(TextFormat.RED + "%commands.generic.permission"));
103
104                return true;
105            }
106
107            int value;
108            if ("day".equals(args[1])) {
109                value = Level.TIME_DAY;
110            } else if ("night".equals(args[1])) {
111                value = Level.TIME_NIGHT;
112            } else if ("midnight".equals(args[1])) {
113                value = Level.TIME_MIDNIGHT;
114            } else if ("noon".equals(args[1])) {
115                value = Level.TIME_NOON;
116            } else if ("sunrise".equals(args[1])) {
117                value = Level.TIME_SUNRISE;
118            } else if ("sunset".equals(args[1])) {
119                value = Level.TIME_SUNSET;
120            } else {
121                try {
122                    value = Math.max(0, Integer.parseInt(args[1]));
123                } catch (Exception e) {
124                    sender.sendMessage(new TranslationContainer("commands.generic.usage", this.usageMessage));
125                    return true;
126                }
127            }
128
129            for (Level level : sender.getServer().getLevels().values()) {
130                level.checkTime();
131                level.setTime(value);
132                level.checkTime();
133            }
134            Command.broadcastCommandMessage(sender, new TranslationContainer("commands.time.set", String.valueOf(value)));
135        } else if ("add".equals(args[0])) {
136            if (!sender.hasPermission("nukkit.command.time.add")) {
137                sender.sendMessage(new TranslationContainer(TextFormat.RED + "%commands.generic.permission"));
138
139                return true;
140            }
141
142            int value;
143            try {
144                value = Math.max(0, Integer.parseInt(args[1]));
145            } catch (Exception e) {
146                sender.sendMessage(new TranslationContainer("commands.generic.usage", this.usageMessage));
147                return true;
148            }
149
150            for (Level level : sender.getServer().getLevels().values()) {
151                level.checkTime();
152                level.setTime(level.getTime() + value);
153                level.checkTime();
154            }
155            Command.broadcastCommandMessage(sender, new TranslationContainer("commands.time.added", String.valueOf(value)));
156        } else {
157            sender.sendMessage(new TranslationContainer("commands.generic.usage", this.usageMessage));
158        }
159
160        return true;
161    }
162}
