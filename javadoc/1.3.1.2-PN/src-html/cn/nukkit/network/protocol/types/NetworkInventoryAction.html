140                } else

145                        return null;
146                    }
147                    this.windowId = Player.ENCHANT_WINDOW_ID;
148                    this.inventorySlot = 1;
149                }
150            }
151            Inventory window = player.getWindowById(this.windowId);
152            if (window != null) {
153                return new SlotChangeAction(window, this.inventorySlot, this.oldItem, this.newItem);
154            }
155            player.getServer().getLogger().debug("Player " + player.getName() + " has no open container with window ID " + this.windowId);
156            return null;
157        case SOURCE_WORLD: 
158            if (this.inventorySlot != InventoryTransactionPacket.ACTION_MAGIC_SLOT_DROP_ITEM) {

170                type = CreativeInventoryAction.TYPE_CREATE_ITEM;
171                break;
172            default: 
173                player.getServer().getLogger().debug("Unexpected creative action type " + this.inventorySlot);
174                return null;
175            }
176            return new CreativeInventoryAction(this.oldItem, this.newItem, type);
177        case SOURCE_CRAFT_SLOT: 
178        case SOURCE_TODO: 
179            //These types need special handling.
180            switch (this.windowId) {
181            case SOURCE_TYPE_CRAFTING_ADD_INGREDIENT:
