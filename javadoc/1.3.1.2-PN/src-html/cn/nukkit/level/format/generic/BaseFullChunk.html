075        BaseFullChunk chunk;
076        try {
077            chunk = (BaseFullChunk) super.clone();
078        } catch (CloneNotSupportedException e) {
079            return null;
080        }

082            chunk.biomes = this.biomes.clone();
083        }
084
085        if (this.blocks != null) {
086            chunk.blocks = this.blocks.clone();
087        }
088
089        if (this.data != null) {
090            chunk.data = this.data.clone();
091        }
092
093        if (this.skyLight != null) {
094            chunk.skyLight = this.skyLight.clone();
095        }
096
097        if (this.blockLight != null) {
098            chunk.blockLight = this.blockLight.clone();
099        }
100
101        if (this.heightMap != null) {
102            chunk.heightMap = this.getHeightMapArray().clone();
103        }
104        return chunk;

128    public void initChunk() {
129        if (this.getProvider() != null && !this.isInit) {
130            boolean changed = false;
131            if (this.NBTentities != null) {
132                this.getProvider().getLevel().timings.syncChunkLoadEntitiesTimer.startTiming();
133                for (CompoundTag nbt : NBTentities) {
134                    if (!nbt.contains("id")) {
135                        this.setChanged();
136                        continue;
137                    }
138                    ListTag pos = nbt.getList("Pos");
139                    if ((((NumberTag) pos.get(0)).getData().intValue() >> 4) != this.getX() || ((((NumberTag) pos.get(2)).getData().intValue() >> 4) != this.getZ())) {
140                        changed = true;
141                        continue;
142                    }
143                    Entity entity = Entity.createEntity(nbt.getString("id"), this, nbt);
144                    if (entity != null) {
145                        changed = true;
146                    }
147                }
148                this.getProvider().getLevel().timings.syncChunkLoadEntitiesTimer.stopTiming();
149                this.NBTentities = null;
150            }
151
152            if (this.NBTtiles != null) {
153                this.getProvider().getLevel().timings.syncChunkLoadBlockEntitiesTimer.startTiming();
154                for (CompoundTag nbt : NBTtiles) {
155                    if (nbt != null) {
156                        if (!nbt.contains("id")) {
157                            changed = true;
158                            continue;
159                        }
160                        if ((nbt.getInt("x") >> 4) != this.getX() || ((nbt.getInt("z") >> 4) != this.getZ())) {
161                            changed = true;
162                            continue;
163                        }
164                        BlockEntity blockEntity = BlockEntity.createBlockEntity(nbt.getString("id"), this, nbt);
165                        if (blockEntity == null) {
166                            changed = true;
167                        }
168                    }
169                }
170                this.getProvider().getLevel().timings.syncChunkLoadBlockEntitiesTimer.stopTiming();
171                this.NBTtiles = null;
172            }
173            
174            if (changed) {
175                this.setChanged();
176            }
177
178            this.isInit = true;
179        }
180    }

245    @Override
246    public void recalculateHeightMap() {
247        for (int z = 0; z < 16; ++z) {
248            for (int x = 0; x < 16; ++x) {
249                recalculateHeightMapColumn(x, z);
250            }
251        }
252    }

254    @Override
255    public int recalculateHeightMapColumn(int x, int z) {
256        int max = getHighestBlockAt(x, z, false);
257        int y;
258        for (y = max; y >= 0; --y) {
259            if (Block.lightFilter[getBlockIdAt(x, y, z)] > 1 || Block.diffusesSkyLight[getBlockIdAt(x, y, z)]) {
260                break;
261            }
262        }
263
264        setHeightMap(x, z, y + 1);
265        return y + 1;
266    }

292    @Override
293    public void populateSkyLight() {
294        // basic light calculation
295        for (int z = 0; z < 16; ++z) {
296            for (int x = 0; x < 16; ++x) { // iterating over all columns in chunk
297                int top = this.getHeightMap(x, z) - 1; // top-most block
298
299                int y;
300
301                for (y = 255; y > top; --y) {
302                    // all the blocks above & including the top-most block in a column are exposed to sun and
303                    // thus have a skylight value of 15
304                    this.setBlockSkyLight(x, y, z, 15);
305                }
306
307                int nextLight = 15; // light value that will be applied starting with the next block
308                int nextDecrease = 0; // decrease that that will be applied starting with the next block
309
310                // TODO: remove nextLight & nextDecrease, use only light & decrease variables
311                for (y = top; y >= 0; --y) { // going under the top-most block
312                    nextLight -= nextDecrease;
313                    int light = nextLight; // this light value will be applied for this block. The following checks are all about the next blocks
314
315                    if (light < 0) {
316                        light = 0;
317                    }
318
319                    this.setBlockSkyLight(x, y, z, light);
320
321                    if (light == 0) { // skipping block checks, because everything under a block that has a skylight value
322                                      // of 0 also has a skylight value of 0
323                        continue;
324                    }
325
326                    // START of checks for the next block
327                    int id = this.getBlockId(x, y, z);
328
329                    if (!Block.transparent[id]) { // if we encounter an opaque block, all the blocks under it will
330                                           // have a skylight value of 0 (the block itself has a value of 15, if it's a top-most block)
331                        nextLight = 0;
332                    } else if (Block.diffusesSkyLight[id]) {
333                        nextDecrease += 1; // skylight value decreases by one for each block under a block
334                                           // that diffuses skylight. The block itself has a value of 15 (if it's a top-most block)
335                    } else {
336                        nextDecrease -= Block.lightFilter[id]; // blocks under a light filtering block will have a skylight value
337                                                            // decreased by the lightFilter value of that block. The block itself
338                                                            // has a value of 15 (if it's a top-most block)
339                    }
340                    // END of checks for the next block
341                }
342            }
343        }
344    }

351    @Override
352    public int getHighestBlockAt(int x, int z, boolean cache) {
353        if (cache) {
354            int h = this.getHeightMap(x, z);
355            if (h != 0 && h != 255) {
356                return h;
357            }
358        }
359        for (int y = 255; y >= 0; --y) {
360            if (getBlockId(x, y, z) != 0x00) {
361                this.setHeightMap(x, z, y);
362                return y;
363            }
364        }
365        return 0;
366    }

389    @Override
390    public void addBlockEntity(BlockEntity blockEntity) {
391        if (this.tiles == null) {
392            this.tiles = new Long2ObjectOpenHashMap<>();
393            this.tileList = new Int2ObjectOpenHashMap<>();
394        }
395        this.tiles.put(blockEntity.getId(), blockEntity);
396        int index = ((blockEntity.getFloorZ() & 0x0f) << 12) | ((blockEntity.getFloorX() & 0x0f) << 8) | (blockEntity.getFloorY() & 0xff);
397        if (this.tileList.containsKey(index) && !this.tileList.get(index).equals(blockEntity)) {
398            BlockEntity entity = this.tileList.get(index);
399            this.tiles.remove(entity.getId());
400            entity.close();
401        }
402        this.tileList.put(index, blockEntity);
403        if (this.isInit) {
404            this.setChanged();
405        }
406    }

409    public void removeBlockEntity(BlockEntity blockEntity) {
410        if (this.tiles != null) {
411            this.tiles.remove(blockEntity.getId());
412            int index = ((blockEntity.getFloorZ() & 0x0f) << 12) | ((blockEntity.getFloorX() & 0x0f) << 8) | (blockEntity.getFloorY() & 0xff);
413            this.tileList.remove(index);
414            if (this.isInit) {
415                this.setChanged();
416            }
417        }
418    }

465    @Override
466    public boolean unload(boolean save, boolean safe) {
467        LevelProvider provider = this.getProvider();
468        if (provider == null) {
469            return true;
470        }
471        if (save && this.changes != 0) {
472            provider.saveChunk(this.getX(), this.getZ());
473        }
474        if (safe) {
475            for (Entity entity : this.getEntities().values()) {
476                if (entity instanceof Player) {
477                    return false;
478                }
479            }
480        }
481        for (Entity entity : new ArrayList<>(this.getEntities().values())) {
482            if (entity instanceof Player) {
483                continue;
484            }
485            entity.close();
486        }
487
488        for (BlockEntity blockEntity : new ArrayList<>(this.getBlockEntities().values())) {
489            blockEntity.close();
490        }
491        this.provider = null;
492        return true;
493    }
