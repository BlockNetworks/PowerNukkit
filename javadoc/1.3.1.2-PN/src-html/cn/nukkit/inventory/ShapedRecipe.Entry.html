033    /**
034     * Constructs a ShapedRecipe instance.
035     *
036     * @param primaryResult    Primary result of the recipe
037     * @param shape<br>        Array of 1, 2, or 3 strings representing the rows of the recipe.
038     *                         This accepts an array of 1, 2 or 3 strings. Each string should be of the same length and must be at most 3
039     *                         characters long. Each character represents a unique type of ingredient. Spaces are interpreted as air.
040     * @param ingredients<br>  Char =&gt; Item map of items to be set into the shape.
041     *                         This accepts an array of Items, indexed by character. Every unique character (except space) in the shape
042     *                         array MUST have a corresponding item in this list. Space character is automatically treated as air.
043     * @param extraResults<br> List of additional result items to leave in the crafting grid afterwards. Used for things like cake recipe
044     *                         empty buckets.
045     *                         <p>
046     *                         Note: Recipes **do not** need to be square. Do NOT add padding for empty rows/columns.
047     */
048    public ShapedRecipe(String recipeId, int priority, Item primaryResult, String[] shape, Map<Character, Item> ingredients, List<Item> extraResults) {
049        this.recipeId = recipeId;
050        this.priority = priority;
051        int rowCount = shape.length;
052        if (rowCount > 3 || rowCount <= 0) {
053            throw new RuntimeException("Shaped recipes may only have 1, 2 or 3 rows, not " + rowCount);
054        }
055
056        int columnCount = shape[0].length();
057        if (columnCount > 3 || columnCount <= 0) {
058            throw new RuntimeException("Shaped recipes may only have 1, 2 or 3 columns, not " + columnCount);
059        }
060
061
062        //for($shape as $y => $row) {
063        for (String row : shape) {
064            if (row.length() != columnCount) {
065                throw new RuntimeException("Shaped recipe rows must all have the same length (expected " + columnCount + ", got " + row.length() + ")");
066            }
067
068            for (int x = 0; x < columnCount; ++x) {
069                char c = row.charAt(x);
070
071                if (c != ' ' && !ingredients.containsKey(c)) {
072                    throw new RuntimeException("No item specified for symbol '" + c + "'");
073                }
074            }
075        }
076
077        this.primaryResult = primaryResult.clone();
078        this.extraResults.addAll(extraResults);
079
080        this.shape = shape;
081
082        for (Map.Entry<Character, Item> entry : ingredients.entrySet()) {
083            this.setIngredient(entry.getKey(), entry.getValue());
084        }
085
086        this.ingredientsAggregate = new ArrayList<>();
087        for (char c : String.join("", this.shape).toCharArray()) {
088            if (c == ' ')
089                continue;
090            Item ingredient = this.ingredients.get(c).clone();
091            for (Item existingIngredient : this.ingredientsAggregate) {
092                if (existingIngredient.equals(ingredient, ingredient.hasMeta(), ingredient.hasCompoundTag())) {
093                    existingIngredient.setCount(existingIngredient.getCount() + ingredient.getCount());
094                    ingredient = null;
095                    break;
096                }
097            }
098            if (ingredient != null)
099                this.ingredientsAggregate.add(ingredient);
100        }
101        this.ingredientsAggregate.sort(CraftingManager.recipeComparator);
102    }

140    public ShapedRecipe setIngredient(char key, Item item) {
141        if (String.join("", this.shape).indexOf(key) < 0) {
142            throw new RuntimeException("Symbol does not appear in the shape: " + key);
143        }
144
145        this.ingredients.put(key, item);
146        return this;
147    }

149    public List<Item> getIngredientList() {
150        List<Item> items = new ArrayList<>();
151        for (int y = 0, y2 = getHeight(); y < y2; ++y) {
152            for (int x = 0, x2 = getWidth(); x < x2; ++x) {
153                items.add(getIngredient(x, y));
154            }
155        }
156        return items;
157    }
158
159    public Map<Integer, Map<Integer, Item>> getIngredientMap() {
160        Map<Integer, Map<Integer, Item>> ingredients = new LinkedHashMap<>();
161
162        for (int y = 0, y2 = getHeight(); y < y2; ++y) {
163            Map<Integer, Item> m = new LinkedHashMap<>();
164
165            for (int x = 0, x2 = getWidth(); x < x2; ++x) {
166                m.put(x, getIngredient(x, y));
167            }
168
169            ingredients.put(y, m);
170        }
171
172        return ingredients;
173    }

175    public Item getIngredient(int x, int y) {
176        Item item = this.ingredients.get(this.shape[y].charAt(x));
177
178        return item != null ? item.clone() : Item.get(Item.AIR);
179    }

214    public boolean matchItems(List<Item> inputList, List<Item> extraOutputList, int multiplier) {
215        List<Item> haveInputs = new ArrayList<>();
216        for (Item item : inputList) {
217            if (item.isNull())
218                continue;
219            haveInputs.add(item.clone());
220        }
221        List<Item> needInputs = new ArrayList<>();
222        if(multiplier != 1){
223            for (Item item : ingredientsAggregate) {
224                if (item.isNull())
225                    continue;
226                Item itemClone = item.clone();
227                itemClone.setCount(itemClone.getCount() * multiplier);
228                needInputs.add(itemClone);
229            }
230        } else {
231            for (Item item : ingredientsAggregate) {
232                if (item.isNull())
233                    continue;
234                needInputs.add(item.clone());
235            }
236        }
237
238        if (!matchItemList(haveInputs, needInputs)) {
239            return false;
240        }
241
242        List<Item> haveOutputs = new ArrayList<>();
243        for (Item item : extraOutputList) {
244            if (item.isNull())
245                continue;
246            haveOutputs.add(item.clone());
247        }
248        haveOutputs.sort(CraftingManager.recipeComparator);
249        List<Item> needOutputs = new ArrayList<>();
250        if(multiplier != 1){
251            for (Item item : getExtraResults()) {
252                if (item.isNull())
253                    continue;
254                Item itemClone = item.clone();
255                itemClone.setCount(itemClone.getCount() * multiplier);
256                needOutputs.add(itemClone);
257            }
258        } else {
259            for (Item item : getExtraResults()) {
260                if (item.isNull())
261                    continue;
262                needOutputs.add(item.clone());
263            }
264        }
265        needOutputs.sort(CraftingManager.recipeComparator);
266
267        return this.matchItemList(haveOutputs, needOutputs);
268    }

270    /**
271     * Returns whether the specified list of crafting grid inputs and outputs matches this recipe. Outputs DO NOT
272     * include the primary result item.
273     *
274     * @param inputList  list of items taken from the crafting grid
275     * @param extraOutputList list of items put back into the crafting grid (secondary results)
276     * @return bool
277     */
278    @Override
279    public boolean matchItems(List<Item> inputList, List<Item> extraOutputList) {
280        return matchItems(inputList, extraOutputList, 1);
281    }
282
283    private boolean matchItemList(List<Item> haveItems, List<Item> needItems) {
284        for (Item needItem : new ArrayList<>(needItems)) {
285            for (Item haveItem : new ArrayList<>(haveItems)) {
286                if (needItem.equals(haveItem, needItem.hasMeta(), needItem.hasCompoundTag())) {
287                    int amount = Math.min(haveItem.getCount(), needItem.getCount());
288                    needItem.setCount(needItem.getCount() - amount);
289                    haveItem.setCount(haveItem.getCount() - amount);
290                    if (haveItem.getCount() == 0) {
291                        haveItems.remove(haveItem);
292                    }
293                    if (needItem.getCount() == 0) {
294                        needItems.remove(needItem);
295                        break;
296                    }
297                }
298            }
299        }
300        return haveItems.isEmpty() && needItems.isEmpty();
301    }
