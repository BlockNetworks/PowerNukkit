073        if (type == Level.BLOCK_UPDATE_NORMAL) {
074            Optional<BlockFace> ascendingDirection = this.getOrientation().ascendingDirection();
075            if (this.down().isTransparent() || (ascendingDirection.isPresent() && this.getSide(ascendingDirection.get()).isTransparent())) {
076                this.getLevel().useBreakOn(this);
077                return Level.BLOCK_UPDATE_NORMAL;
078            }
079        }
080        return 0;
081    }

099    @Override
100    public boolean place(Item item, Block block, Block target, BlockFace face, double fx, double fy, double fz, Player player) {
101        Block down = this.down();
102        if (down == null || down.isTransparent()) {
103            return false;
104        }
105        Map<BlockRail, BlockFace> railsAround = this.checkRailsAroundAffected();
106        List<BlockRail> rails = new ArrayList<>(railsAround.keySet());
107        List<BlockFace> faces = new ArrayList<>(railsAround.values());
108        if (railsAround.size() == 1) {
109            BlockRail other = rails.get(0);
110            this.setDamage(this.connect(other, railsAround.get(other)).metadata());
111        } else if (railsAround.size() == 4) {
112            if (this.isAbstract()) {
113                this.setDamage(this.connect(rails.get(faces.indexOf(SOUTH)), SOUTH, rails.get(faces.indexOf(EAST)), EAST).metadata());
114            } else {
115                this.setDamage(this.connect(rails.get(faces.indexOf(EAST)), EAST, rails.get(faces.indexOf(WEST)), WEST).metadata());
116            }
117        } else if (!railsAround.isEmpty()) {
118            if (this.isAbstract()) {
119                if (railsAround.size() == 2) {
120                    BlockRail rail1 = rails.get(0);
121                    BlockRail rail2 = rails.get(1);
122                    this.setDamage(this.connect(rail1, railsAround.get(rail1), rail2, railsAround.get(rail2)).metadata());
123                } else {
124                    List<BlockFace> cd = Stream.of(CURVED_SOUTH_EAST, CURVED_NORTH_EAST, CURVED_SOUTH_WEST)
125                            .filter(o -> faces.containsAll(o.connectingDirections()))
126                            .findFirst().get().connectingDirections();
127                    BlockFace f1 = cd.get(0);
128                    BlockFace f2 = cd.get(1);
129                    this.setDamage(this.connect(rails.get(faces.indexOf(f1)), f1, rails.get(faces.indexOf(f2)), f2).metadata());
130                }
131            } else {
132                BlockFace f = faces.stream().min((f1, f2) -> (f1.getIndex() < f2.getIndex()) ? 1 : ((x == y) ? 0 : -1)).get();
133                BlockFace fo = f.getOpposite();
134                if (faces.contains(fo)) { //Opposite connectable
135                    this.setDamage(this.connect(rails.get(faces.indexOf(f)), f, rails.get(faces.indexOf(fo)), fo).metadata());
136                } else {
137                    this.setDamage(this.connect(rails.get(faces.indexOf(f)), f).metadata());
138                }
139            }
140        }
141        this.level.setBlock(this, this, true, true);
142        if (!isAbstract()) {
143            level.scheduleUpdate(this, this, 0);
144        }
145        return true;
146    }

148    private Orientation connect(BlockRail rail1, BlockFace face1, BlockRail rail2, BlockFace face2) {
149        this.connect(rail1, face1);
150        this.connect(rail2, face2);
151
152        if (face1.getOpposite() == face2) {
153            int delta1 = (int) (this.y - rail1.y);
154            int delta2 = (int) (this.y - rail2.y);
155
156            if (delta1 == -1) {
157                return Orientation.ascending(face1);
158            } else if (delta2 == -1) {
159                return Orientation.ascending(face2);
160            }
161        }
162        return straightOrCurved(face1, face2);
163    }
164
165    private Orientation connect(BlockRail other, BlockFace face) {
166        int delta = (int) (this.y - other.y);
167        Map<BlockRail, BlockFace> rails = other.checkRailsConnected();
168        if (rails.isEmpty()) { //Only one
169            other.setOrientation(delta == 1 ? ascending(face.getOpposite()) : straight(face));
170            return delta == -1 ? ascending(face) : straight(face);
171        } else if (rails.size() == 1) { //Already connected
172            BlockFace faceConnected = rails.values().iterator().next();
173
174            if (other.isAbstract() && faceConnected != face) { //Curve!
175                other.setOrientation(curved(face.getOpposite(), faceConnected));
176                return delta == -1 ? ascending(face) : straight(face);
177            } else if (faceConnected == face) { //Turn!
178                if (!other.getOrientation().isAscending()) {
179                    other.setOrientation(delta == 1 ? ascending(face.getOpposite()) : straight(face));
180                }
181                return delta == -1 ? ascending(face) : straight(face);
182            } else if (other.getOrientation().hasConnectingDirections(NORTH, SOUTH)) { //North-south
183                other.setOrientation(delta == 1 ? ascending(face.getOpposite()) : straight(face));
184                return delta == -1 ? ascending(face) : straight(face);
185            }
186        }
187        return STRAIGHT_NORTH_SOUTH;
188    }
189
190    private Map<BlockRail, BlockFace> checkRailsAroundAffected() {
191        Map<BlockRail, BlockFace> railsAround = this.checkRailsAround(Arrays.asList(SOUTH, EAST, WEST, NORTH));
192        return railsAround.keySet().stream()
193                .filter(r -> r.checkRailsConnected().size() != 2)
194                .collect(Collectors.toMap(r -> r, railsAround::get));
195    }
196
197    private Map<BlockRail, BlockFace> checkRailsAround(Collection<BlockFace> faces) {
198        Map<BlockRail, BlockFace> result = new HashMap<>();
199        faces.forEach(f -> {
200            Block b = this.getSide(f);
201            Stream.of(b, b.up(), b.down())
202                    .filter(Rail::isRailBlock)
203                    .forEach(block -> result.put((BlockRail) block, f));
204        });
205        return result;
206    }
207
208    protected Map<BlockRail, BlockFace> checkRailsConnected() {
209        Map<BlockRail, BlockFace> railsAround = this.checkRailsAround(this.getOrientation().connectingDirections());
210        return railsAround.keySet().stream()
211                .filter(r -> r.getOrientation().hasConnectingDirections(railsAround.get(r).getOpposite()))
212                .collect(Collectors.toMap(r -> r, railsAround::get));
213    }

227    public void setOrientation(Orientation o) {
228        if (o.metadata() != this.getRealMeta()) {
229            this.setDamage(o.metadata());
230            this.level.setBlock(this, this, false, true);
231        }
232    }

234    public int getRealMeta() {
235        // Check if this can be powered
236        // Avoid modifying the value from meta (The rail orientation may be false)
237        // Reason: When the rail is curved, the meta will return STRAIGHT_NORTH_SOUTH.
238        // OR Null Pointer Exception
239        if (!isAbstract()) {
240            return getDamage() & 0x7;
241        }
242        // Return the default: This meta
243        return getDamage();
244    }
245
246    public boolean isActive() {
247        return (getDamage() & 0x8) != 0;
248    }
249
250    public void setActive(boolean active) {
251        if (active) {
252            setDamage(getDamage() | 0x8);
253        } else {
254            setDamage(getDamage() & 0x7);
255        }
256        level.setBlock(this, this, true, true);
257    }
