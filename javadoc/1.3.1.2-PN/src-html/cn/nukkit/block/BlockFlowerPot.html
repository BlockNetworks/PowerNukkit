001package cn.nukkit.block;
002
003import cn.nukkit.Player;
004import cn.nukkit.blockentity.BlockEntity;
005import cn.nukkit.blockentity.BlockEntityFlowerPot;
006import cn.nukkit.item.Item;
007import cn.nukkit.item.ItemFlowerPot;
008import cn.nukkit.level.Level;
009import cn.nukkit.math.AxisAlignedBB;
010import cn.nukkit.math.BlockFace;
011import cn.nukkit.nbt.tag.CompoundTag;
012import cn.nukkit.nbt.tag.Tag;
013
014/**
015 * @author Nukkit Project Team
016 */
017public class BlockFlowerPot extends BlockFlowable {
018
019    public BlockFlowerPot() {
020        this(0);
021    }
022
023    public BlockFlowerPot(int meta) {
024        super(meta);
025    }
026
027    protected static boolean canPlaceIntoFlowerPot(int id) {
028        switch (id) {
029            case SAPLING:
030            case DEAD_BUSH:
031            case DANDELION:
032            case ROSE:
033            case RED_MUSHROOM:
034            case BROWN_MUSHROOM:
035            case CACTUS:
036            case WITHER_ROSE:
037            case BAMBOO:
038                // TODO: 2016/2/4 case NETHER_WART:
039                return true;
040            default:
041                return false;
042        }
043    }
044
045    @Override
046    public int getWaterloggingLevel() {
047        return 1;
048    }
049
050    @Override
051    public String getName() {
052        return "Flower Pot";
053    }
054
055    @Override
056    public int getId() {
057        return FLOWER_POT_BLOCK;
058    }
059
060    @Override
061    public double getHardness() {
062        return 0;
063    }
064
065    @Override
066    public double getResistance() {
067        return 0;
068    }
069    
070    private boolean isSupportValid(Block block) {
071        return block.isSolid(BlockFace.UP) || block instanceof BlockFence || block instanceof BlockWall || block instanceof BlockHopper;
072    }
073
074    @Override
075    public int onUpdate(int type) {
076        if (type == Level.BLOCK_UPDATE_NORMAL) {
077            if (!isSupportValid(down())) {
078                level.useBreakOn(this);
079                return type;
080            }
081        }
082        return 0;
083    }
084
085    @Override
086    public boolean place(Item item, Block block, Block target, BlockFace face, double fx, double fy, double fz, Player player) {
087        if (!isSupportValid(down())) {
088            return false;
089        }
090        
091        CompoundTag nbt = new CompoundTag()
092                .putString("id", BlockEntity.FLOWER_POT)
093                .putInt("x", (int) this.x)
094                .putInt("y", (int) this.y)
095                .putInt("z", (int) this.z)
096                .putShort("item", 0)
097                .putInt("data", 0);
098        if (item.hasCustomBlockData()) {
099            for (Tag aTag : item.getCustomBlockData().getAllTags()) {
100                nbt.put(aTag.getName(), aTag);
101            }
102        }
103        BlockEntityFlowerPot flowerPot = (BlockEntityFlowerPot) BlockEntity.createBlockEntity(BlockEntity.FLOWER_POT, getLevel().getChunk((int) block.x >> 4, (int) block.z >> 4), nbt);
104        if (flowerPot == null) return false;
105
106        this.getLevel().setBlock(block, this, true, true);
107        return true;
108    }
109
110    @Override
111    public boolean canBeActivated() {
112        return true;
113    }
114
115    @Override
116    public boolean onActivate(Item item) {
117        return this.onActivate(item, null);
118    }
119
120    @Override
121    public boolean onActivate(Item item, Player player) {
122        BlockEntity blockEntity = getLevel().getBlockEntity(this);
123        if (!(blockEntity instanceof BlockEntityFlowerPot)) return false;
124        if (blockEntity.namedTag.getShort("item") != 0 || blockEntity.namedTag.getInt("mData") != 0) return false;
125        int blockId;
126        int itemMeta;
127        if (!canPlaceIntoFlowerPot(item.getId())) {
128            if (!canPlaceIntoFlowerPot(item.getBlock().getId())) {
129                return true;
130            }
131            blockId = item.getBlock().getId();
132            itemMeta = item.getDamage();
133        } else {
134            blockId = item.getId();
135            itemMeta = item.getDamage();
136        }
137        blockEntity.namedTag.putShort("item", blockId);
138        blockEntity.namedTag.putInt("data", itemMeta);
139
140        this.setDamage(1);
141        this.getLevel().setBlock(this, this, true);
142        ((BlockEntityFlowerPot) blockEntity).spawnToAll();
143
144        if (player.isSurvival()) {
145            item.setCount(item.getCount() - 1);
146            player.getInventory().setItemInHand(item.getCount() > 0 ? item : Item.get(Item.AIR));
147        }
148        return true;
149    }
150
151    @Override
152    public Item[] getDrops(Item item) {
153        boolean dropInside = false;
154        int insideID = 0;
155        int insideMeta = 0;
156        BlockEntity blockEntity = getLevel().getBlockEntity(this);
157        if (blockEntity instanceof BlockEntityFlowerPot) {
158            dropInside = true;
159            insideID = blockEntity.namedTag.getShort("item");
160            insideMeta = blockEntity.namedTag.getInt("data");
161        }
162
163        if (dropInside) {
164            return new Item[]{
165                    new ItemFlowerPot(),
166                    Block.get(insideID, insideMeta).toItem()
167            };
168        } else {
169            return new Item[]{
170                    new ItemFlowerPot()
171            };
172        }
173    }
174
175    @Override
176    protected AxisAlignedBB recalculateBoundingBox() {
177        return this;
178    }
179
180    @Override
181    public double getMinX() {
182        return this.x + 0.3125;
183    }
184
185    @Override
186    public double getMinZ() {
187        return this.z + 0.3125;
188    }
189
190    @Override
191    public double getMaxX() {
192        return this.x + 0.6875;
193    }
194
195    @Override
196    public double getMaxY() {
197        return this.y + 0.375;
198    }
199
200    @Override
201    public double getMaxZ() {
202        return this.z + 0.6875;
203    }
204
205    @Override
206    public boolean canPassThrough() {
207        return false;
208    }
209
210    @Override
211    public Item toItem() {
212        return new ItemFlowerPot();
213    }
214}
