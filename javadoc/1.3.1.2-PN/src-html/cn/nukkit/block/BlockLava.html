022/**
023 * author: MagicDroidX

025 */
026public class BlockLava extends BlockLiquid {
027
028    public BlockLava() {
029        this(0);
030    }
031
032    public BlockLava(int meta) {
033        super(meta);
034    }
035
036    @Override
037    public int getId() {
038        return LAVA;
039    }
040
041    @Override
042    public int getLightLevel() {
043        return 15;
044    }
045
046    @Override
047    public String getName() {
048        return "Lava";
049    }
050
051    @Override
052    public void onEntityCollide(Entity entity) {
053        entity.highestPosition -= (entity.highestPosition - entity.y) * 0.5;
054
055        // Always setting the duration to 15 seconds? TODO
056        EntityCombustByBlockEvent ev = new EntityCombustByBlockEvent(this, entity, 15);
057        Server.getInstance().getPluginManager().callEvent(ev);
058        if (!ev.isCancelled()
059                // Making sure the entity is actually alive and not invulnerable.
060                && entity.isAlive()
061                && entity.noDamageTicks == 0) {
062            entity.setOnFire(ev.getDuration());
063        }
064
065        if (!entity.hasEffect(Effect.FIRE_RESISTANCE)) {
066            entity.attack(new EntityDamageByBlockEvent(this, entity, DamageCause.LAVA, 4));
067        }
068
069        super.onEntityCollide(entity);
070    }
071
072    @Override
073    public boolean place(Item item, Block block, Block target, BlockFace face, double fx, double fy, double fz, Player player) {
074        boolean ret = this.getLevel().setBlock(this, this, true, false);
075        this.getLevel().scheduleUpdate(this, this.tickRate());
076
077        return ret;
078    }
079
080    @Override
081    public int onUpdate(int type) {
082        int result = super.onUpdate(type);
083
084        if (type == Level.BLOCK_UPDATE_RANDOM && this.level.gameRules.getBoolean(GameRule.DO_FIRE_TICK)) {
085            Random random = ThreadLocalRandom.current();
086
087            int i = random.nextInt(3);
088
089            if (i > 0) {
090                for (int k = 0; k < i; ++k) {
091                    Vector3 v = this.add(random.nextInt(3) - 1, 1, random.nextInt(3) - 1);
092                    Block block = this.getLevel().getBlock(v);
093
094                    if (block.getId() == AIR) {
095                        if (this.isSurroundingBlockFlammable(block)) {
096                            BlockIgniteEvent e = new BlockIgniteEvent(block, this, null, BlockIgniteEvent.BlockIgniteCause.LAVA);
097                            this.level.getServer().getPluginManager().callEvent(e);
098
099                            if (!e.isCancelled()) {
100                                Block fire = Block.get(BlockID.FIRE);
101                                this.getLevel().setBlock(v, fire, true);
102                                this.getLevel().scheduleUpdate(fire, fire.tickRate());
103                                return Level.BLOCK_UPDATE_RANDOM;
104                            }
105
106                            return 0;
107                        }
108                    } else if (block.isSolid()) {
109                        return Level.BLOCK_UPDATE_RANDOM;
110                    }
111                }
112            } else {
113                for (int k = 0; k < 3; ++k) {
114                    Vector3 v = this.add(random.nextInt(3) - 1, 0, random.nextInt(3) - 1);
115                    Block block = this.getLevel().getBlock(v);
116
117                    if (block.up().getId() == AIR && block.getBurnChance() > 0) {
118                        BlockIgniteEvent e = new BlockIgniteEvent(block, this, null, BlockIgniteEvent.BlockIgniteCause.LAVA);
119                        this.level.getServer().getPluginManager().callEvent(e);
120
121                        if (!e.isCancelled()) {
122                            Block fire = Block.get(BlockID.FIRE);
123                            this.getLevel().setBlock(v, fire, true);
124                            this.getLevel().scheduleUpdate(fire, fire.tickRate());
125                        }
126                    }
127                }
128            }
129        }
130
131        return result;
132    }
133
134    protected boolean isSurroundingBlockFlammable(Block block) {
135        for (BlockFace face : BlockFace.values()) {
136            if (block.getSide(face).getBurnChance() > 0) {
137                return true;
138            }
139        }
140
141        return false;
142    }
143
144    @Override
145    public BlockColor getColor() {
146        return BlockColor.LAVA_BLOCK_COLOR;
147    }
148
149    @Override
150    public BlockLiquid getBlock(int meta) {
151        return (BlockLiquid) Block.get(BlockID.LAVA, meta);
152    }
153
154    @Override
155    public int tickRate() {
156        return 30;
157    }
158
159    @Override
160    public int getFlowDecayPerBlock() {
161        if (this.level.getDimension() == Level.DIMENSION_NETHER) {
162            return 1;
163        }
164        return 2;
165    }
166
167    @Override
168    protected void checkForHarden(){ 
169        Block colliding = null;
170        for(int side = 1; side < 6; ++side){ //don't check downwards side
171            Block blockSide = this.getSide(BlockFace.fromIndex(side));
172            if(blockSide instanceof BlockWater || blockSide.getLevelBlockAtLayer(1) instanceof BlockWater){
173                colliding = blockSide;
174                break;
175            }
176        }
177        if(colliding != null){
178            if(this.getDamage() == 0){
179                this.liquidCollide(colliding, Block.get(BlockID.OBSIDIAN));
180            }else if(this.getDamage() <= 4){
181                this.liquidCollide(colliding, Block.get(BlockID.COBBLESTONE));
182            }
183        }
184    }
185
186    @Override
187    protected void flowIntoBlock(Block block, int newFlowDecay){
188        if(block instanceof BlockWater){
189            ((BlockLiquid) block).liquidCollide(this, Block.get(BlockID.STONE));
190        }else{
191            super.flowIntoBlock(block, newFlowDecay);
192        }
193    }
194
195    @Override
196    public void addVelocityToEntity(Entity entity, Vector3 vector) {
197        if (!(entity instanceof EntityPrimedTNT)) {
198            super.addVelocityToEntity(entity, vector);
199        }
200    }
201}
