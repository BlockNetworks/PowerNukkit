001package cn.nukkit;
002
003import cn.nukkit.api.PowerNukkitDifference;
004import cn.nukkit.network.protocol.AdventureSettingsPacket;
005
006import java.util.EnumMap;
007import java.util.Map;
008
009/**
010 * Nukkit Project
011 * Author: MagicDroidX
012 */
013public class AdventureSettings implements Cloneable {
014
015    public static final int PERMISSION_NORMAL = 0;
016    public static final int PERMISSION_OPERATOR = 1;
017    public static final int PERMISSION_HOST = 2;
018    public static final int PERMISSION_AUTOMATION = 3;
019    public static final int PERMISSION_ADMIN = 4;
020
021    private Map<Type, Boolean> values = new EnumMap<>(Type.class);
022
023    private Player player;
024
025    public AdventureSettings(Player player) {
026        this.player = player;
027    }
028
029    public AdventureSettings clone(Player newPlayer) {
030        try {
031            AdventureSettings settings = (AdventureSettings) super.clone();
032            settings.player = newPlayer;
033            return settings;
034        } catch (CloneNotSupportedException e) {
035            return null;
036        }
037    }
038
039    public AdventureSettings set(Type type, boolean value) {
040        this.values.put(type, value);
041        return this;
042    }
043
044    public boolean get(Type type) {
045        Boolean value = this.values.get(type);
046
047        return value == null ? type.getDefaultValue() : value;
048    }
049
050    @PowerNukkitDifference(
051            info = "Players in spectator mode will be flagged as member even if they are OP due to a client-side limitation",
052            since = "1.3.1.2-PN")
053    public void update() {
054        AdventureSettingsPacket pk = new AdventureSettingsPacket();
055        for (Type t : Type.values()) {
056            pk.setFlag(t.getId(), get(t));
057        }
058
059        pk.commandPermission = (player.isOp() ? AdventureSettingsPacket.PERMISSION_OPERATOR : AdventureSettingsPacket.PERMISSION_NORMAL);
060        pk.playerPermission = (player.isOp() && !player.isSpectator() ? Player.PERMISSION_OPERATOR : Player.PERMISSION_MEMBER);
061        pk.entityUniqueId = player.getId();
062
063        Server.broadcastPacket(player.getViewers().values(), pk);
064        player.dataPacket(pk);
065
066        player.resetInAirTicks();
067    }
068
069    public enum Type {
070        WORLD_IMMUTABLE(AdventureSettingsPacket.WORLD_IMMUTABLE, false),
071        AUTO_JUMP(AdventureSettingsPacket.AUTO_JUMP, true),
072        ALLOW_FLIGHT(AdventureSettingsPacket.ALLOW_FLIGHT, false),
073        NO_CLIP(AdventureSettingsPacket.NO_CLIP, false),
074        WORLD_BUILDER(AdventureSettingsPacket.WORLD_BUILDER, true),
075        FLYING(AdventureSettingsPacket.FLYING, false),
076        MUTED(AdventureSettingsPacket.MUTED, false),
077        BUILD_AND_MINE(AdventureSettingsPacket.BUILD_AND_MINE, true),
078        DOORS_AND_SWITCHED(AdventureSettingsPacket.DOORS_AND_SWITCHES, true),
079        OPEN_CONTAINERS(AdventureSettingsPacket.OPEN_CONTAINERS, true),
080        ATTACK_PLAYERS(AdventureSettingsPacket.ATTACK_PLAYERS, true),
081        ATTACK_MOBS(AdventureSettingsPacket.ATTACK_MOBS, true),
082        OPERATOR(AdventureSettingsPacket.OPERATOR, false),
083        TELEPORT(AdventureSettingsPacket.TELEPORT, false);
084
085        private final int id;
086        private final boolean defaultValue;
087
088        Type(int id, boolean defaultValue) {
089            this.id = id;
090            this.defaultValue = defaultValue;
091        }
092
093        public int getId() {
094            return id;
095        }
096
097        public boolean getDefaultValue() {
098            return this.defaultValue;
099        }
100    }
101}
