001package cn.nukkit.item;
002
003import cn.nukkit.nbt.tag.CompoundTag;
004import cn.nukkit.nbt.tag.ListTag;
005import com.google.common.base.Preconditions;
006
007import java.util.List;
008
009public abstract class ItemBookWritable extends Item {
010
011    protected ItemBookWritable(int id) {
012        super(id);
013    }
014
015    protected ItemBookWritable(int id, Integer meta) {
016        super(id, meta);
017    }
018
019    protected ItemBookWritable(int id, Integer meta, int count) {
020        super(id, meta, count);
021    }
022
023    protected ItemBookWritable(int id, Integer meta, int count, String name) {
024        super(id, meta, count, name);
025    }
026
027    /**
028     * Returns whether the given page exists in this book.
029     */
030    public boolean pageExists(int pageId) {
031        Preconditions.checkArgument(pageId >= 0 && pageId < 50, "Page number " + pageId + " is out of range");
032        if (this.hasCompoundTag()) {
033            CompoundTag tag = this.getNamedTag();
034            if (tag.contains("pages") && tag.get("pages") instanceof ListTag) {
035                return tag.getList("pages", CompoundTag.class).size() > pageId;
036            }
037        }
038        return false;
039    }
040
041    /**
042     * Returns a string containing the content of a page (which could be empty), or null if the page doesn't exist.
043     */
044    public String getPageText(int pageId) {
045        Preconditions.checkArgument(pageId >= 0 && pageId < 50, "Page number " + pageId + " is out of range");
046        if (this.hasCompoundTag()) {
047            CompoundTag tag = this.getNamedTag();
048            if (tag.contains("pages") && tag.get("pages") instanceof ListTag) {
049                ListTag<CompoundTag> pages = tag.getList("pages", CompoundTag.class);
050                if (pages.size() > pageId) {
051                    return pages.get(pageId).getString("text");
052                }
053            }
054        }
055        return null;
056    }
057
058    /**
059     * Sets the text of a page in the book. Adds the page if the page does not yet exist.
060     * @return boolean indicating success
061     */
062    public boolean setPageText(int pageId, String pageText) {
063        Preconditions.checkArgument(pageId >= 0 && pageId < 50, "Page number " + pageId + " is out of range");
064        CompoundTag tag;
065        if (this.hasCompoundTag()) {
066            tag = this.getNamedTag();
067        } else if (pageText.isEmpty()) {
068            return false;
069        } else {
070            tag = new CompoundTag();
071        }
072        ListTag<CompoundTag> pages;
073        if (!tag.contains("pages") || !(tag.get("pages") instanceof ListTag)) {
074            pages = new ListTag<>("pages");
075            tag.putList(pages);
076        } else {
077            pages = tag.getList("pages", CompoundTag.class);
078        }
079        if (pages.size() <= pageId) {
080            for (int current = pages.size(); current <= pageId; current++) {
081                pages.add(createPageTag());
082            }
083        }
084
085        pages.get(pageId).putString("text", pageText);
086        this.setCompoundTag(tag);
087        return true;
088    }
089
090    /**
091     * Adds a new page with the given page ID.
092     * Creates a new page for every page between the given ID and existing pages that doesn't yet exist.
093     * @return boolean indicating success
094     */
095    public boolean addPage(int pageId) {
096        Preconditions.checkArgument(pageId >= 0 && pageId < 50, "Page number " + pageId + " is out of range");
097        CompoundTag tag = this.hasCompoundTag() ? this.getNamedTag() : new CompoundTag();
098        ListTag<CompoundTag> pages;
099        if (!tag.contains("pages") || !(tag.get("pages") instanceof ListTag)) {
100            pages = new ListTag<>("pages");
101            tag.putList(pages);
102        } else {
103            pages = tag.getList("pages", CompoundTag.class);
104        }
105
106        for (int current = pages.size(); current <= pageId; current++) {
107            pages.add(createPageTag());
108        }
109        this.setCompoundTag(tag);
110        return true;
111    }
112
113    /**
114     * Deletes an existing page with the given page ID.
115     * @return boolean indicating success
116     */
117    public boolean deletePage(int pageId) {
118        Preconditions.checkArgument(pageId >= 0 && pageId < 50, "Page number " + pageId + " is out of range");
119        if (this.hasCompoundTag()) {
120            CompoundTag tag = this.getNamedTag();
121            if (tag.contains("pages") && tag.get("pages") instanceof ListTag) {
122                ListTag<CompoundTag> pages = tag.getList("pages", CompoundTag.class);
123                if (pages.size() > pageId) {
124                    pages.remove(pageId);
125                    this.setCompoundTag(tag);
126                }
127            }
128        }
129        return true;
130    }
131
132    /**
133     * Inserts a new page with the given text and moves other pages upwards.
134     * @return boolean indicating success
135     */
136    public boolean insertPage(int pageId) {
137        return this.insertPage(pageId, "");
138    }
139
140    /**
141     * Inserts a new page with the given text and moves other pages upwards.
142     * @return boolean indicating success
143     */
144    public boolean insertPage(int pageId, String pageText) {
145        Preconditions.checkArgument(pageId >= 0 && pageId < 50, "Page number " + pageId + " is out of range");
146        CompoundTag tag = this.hasCompoundTag() ? this.getNamedTag() : new CompoundTag();
147        ListTag<CompoundTag> pages;
148        if (!tag.contains("pages") || !(tag.get("pages") instanceof ListTag)) {
149            pages = new ListTag<>("pages");
150            tag.putList(pages);
151        } else {
152            pages = tag.getList("pages", CompoundTag.class);
153        }
154
155        if (pages.size() <= pageId) {
156            for (int current = pages.size(); current <= pageId; current++) {
157                pages.add(createPageTag());
158            }
159            pages.get(pageId).putString("text", pageText);
160        } else {
161            pages.add(pageId, createPageTag(pageText));
162        }
163        this.setCompoundTag(tag);
164        return true;
165    }
166
167     /**
168     * Switches the text of two pages with each other.
169     * @return boolean indicating success
170     */
171    public boolean swapPages(int pageId1, int pageId2) {
172        Preconditions.checkArgument(pageId1 >= 0 && pageId1 < 50, "Page number " + pageId1 + " is out of range");
173        Preconditions.checkArgument(pageId2 >= 0 && pageId2 < 50, "Page number " + pageId2 + " is out of range");
174        if (this.hasCompoundTag()) {
175            CompoundTag tag = this.getNamedTag();
176            if (tag.contains("pages") && tag.get("pages") instanceof ListTag) {
177                ListTag<CompoundTag> pages = tag.getList("pages", CompoundTag.class);
178                if (pages.size() > pageId1 && pages.size() > pageId2) {
179                    String pageContents1 = pages.get(pageId1).getString("text");
180                    String pageContents2 = pages.get(pageId2).getString("text");
181                    pages.get(pageId1).putString("text", pageContents2);
182                    pages.get(pageId2).putString("text", pageContents1);
183                    return true;
184                }
185            }
186        }
187        return false;
188    }
189
190    /**
191     * Returns an list containing all pages of this book.
192     */
193    public List getPages() {
194        CompoundTag tag = this.hasCompoundTag() ? this.getNamedTag() : new CompoundTag();
195        ListTag<CompoundTag> pages;
196        if (!tag.contains("pages") || !(tag.get("pages") instanceof ListTag)) {
197            pages = new ListTag<>("pages");
198            tag.putList(pages);
199        } else {
200            pages = tag.getList("pages", CompoundTag.class);
201        }
202        return pages.parseValue();
203    }
204
205    protected static CompoundTag createPageTag() {
206        return createPageTag("");
207    }
208
209    protected static CompoundTag createPageTag(String pageText) {
210        return new CompoundTag()
211                .putString("text", pageText)
212                .putString("photoname", "");
213    }
214}
