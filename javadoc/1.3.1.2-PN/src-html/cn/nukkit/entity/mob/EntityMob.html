001package cn.nukkit.entity.mob;
002
003import cn.nukkit.Player;
004import cn.nukkit.entity.EntityCreature;
005import cn.nukkit.entity.EntityNameable;
006import cn.nukkit.item.Item;
007import cn.nukkit.level.format.FullChunk;
008import cn.nukkit.math.Vector3;
009import cn.nukkit.nbt.tag.CompoundTag;
010
011/**
012 * author: MagicDroidX
013 * Nukkit Project
014 */
015public abstract class EntityMob extends EntityCreature implements EntityNameable {
016    public EntityMob(FullChunk chunk, CompoundTag nbt) {
017        super(chunk, nbt);
018    }
019
020    @Override
021    public boolean onInteract(Player player, Item item, Vector3 clickedPos) {
022        return EntityNameable.super.onInteract(player, item, clickedPos)
023                || EntityMob.super.onInteract(player, item, clickedPos);
024    }
025}
