001package cn.nukkit.metadata;
002
003import cn.nukkit.plugin.Plugin;
004import cn.nukkit.utils.PluginException;
005import cn.nukkit.utils.ServerException;
006
007import java.util.*;
008
009/**
010 * author: MagicDroidX
011 * Nukkit Project
012 */
013public abstract class MetadataStore {
014
015    private final Map<String, Map<Plugin, MetadataValue>> metadataMap = new HashMap<>();
016
017    public void setMetadata(Object subject, String metadataKey, MetadataValue newMetadataValue) {
018        if (newMetadataValue == null) {
019            throw new ServerException("Value cannot be null");
020        }
021        Plugin owningPlugin = newMetadataValue.getOwningPlugin();
022        if (owningPlugin == null) {
023            throw new PluginException("Plugin cannot be null");
024        }
025        String key = this.disambiguate((Metadatable) subject, metadataKey);
026        Map<Plugin, MetadataValue> entry = this.metadataMap.computeIfAbsent(key, k -> new WeakHashMap<>(1));
027        entry.put(owningPlugin, newMetadataValue);
028    }
029
030    public List<MetadataValue> getMetadata(Object subject, String metadataKey) {
031        String key = this.disambiguate((Metadatable) subject, metadataKey);
032        if (this.metadataMap.containsKey(key)) {
033            Collection values = ((Map) this.metadataMap.get(key)).values();
034            return Collections.unmodifiableList(new ArrayList<>(values));
035        }
036        return Collections.emptyList();
037    }
038
039    public boolean hasMetadata(Object subject, String metadataKey) {
040        return this.metadataMap.containsKey(this.disambiguate((Metadatable) subject, metadataKey));
041    }
042
043    public void removeMetadata(Object subject, String metadataKey, Plugin owningPlugin) {
044        if (owningPlugin == null) {
045            throw new PluginException("Plugin cannot be null");
046        }
047        String key = this.disambiguate((Metadatable) subject, metadataKey);
048        Map entry = this.metadataMap.get(key);
049        if (entry == null) {
050            return;
051        }
052        entry.remove(owningPlugin);
053        if (entry.isEmpty()) {
054            this.metadataMap.remove(key);
055        }
056    }
057
058    public void invalidateAll(Plugin owningPlugin) {
059        if (owningPlugin == null) {
060            throw new PluginException("Plugin cannot be null");
061        }
062        for (Map value : this.metadataMap.values()) {
063            if (value.containsKey(owningPlugin)) {
064                ((MetadataValue) value.get(owningPlugin)).invalidate();
065            }
066        }
067    }
068
069    protected abstract String disambiguate(Metadatable subject, String metadataKey);
070}
