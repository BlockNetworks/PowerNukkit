001package cn.nukkit.blockentity;
002
003import cn.nukkit.Player;
004import cn.nukkit.block.Block;
005import cn.nukkit.level.GlobalBlockPalette;
006import cn.nukkit.level.format.FullChunk;
007import cn.nukkit.nbt.tag.CompoundTag;
008
009import cn.nukkit.network.protocol.UpdateBlockPacket;
010import cn.nukkit.utils.BlockColor;
011
012/**
013 * author: CreeperFace
014 * Nukkit Project
015 */
016public class BlockEntityCauldron extends BlockEntitySpawnable {
017    
018    public static final int POTION_TYPE_EMPTY = 0xFFFF;
019    public static final int POTION_TYPE_NORMAL = 0;
020    public static final int POTION_TYPE_SPLASH = 1;
021    public static final int POTION_TYPE_LINGERING = 2;
022    public static final int POTION_TYPE_LAVA = 0xF19B;
023
024    public BlockEntityCauldron(FullChunk chunk, CompoundTag nbt) {
025        super(chunk, nbt);
026    }
027
028    @Override
029    protected void initBlockEntity() {
030        int potionId;
031        if (!namedTag.contains("PotionId")) {
032            namedTag.putShort("PotionId", 0xffff);
033        }
034        potionId = namedTag.getShort("PotionId");
035        
036        int potionType = (potionId & 0xFFFF) == 0xFFFF? POTION_TYPE_EMPTY : POTION_TYPE_NORMAL;
037        if (namedTag.getBoolean("SplashPotion")) {
038            potionType = POTION_TYPE_SPLASH;
039            namedTag.remove("SplashPotion");
040        }
041        
042        if (!namedTag.contains("PotionType")) {
043            namedTag.putShort("PotionType", potionType);
044        }
045
046
047        super.initBlockEntity();
048    }
049
050    public int getPotionId() {
051        return namedTag.getShort("PotionId");
052    }
053
054    public void setPotionId(int potionId) {
055        namedTag.putShort("PotionId", potionId);
056        this.spawnToAll();
057    }
058
059    public boolean hasPotion() {
060        return getPotionId() != 0xffff;
061    }
062    
063    public void setPotionType(int potionType) {
064        this.namedTag.putShort("PotionType", potionType & 0xFFFF);
065    }
066    
067    public int getPotionType() {
068        return this.namedTag.getShort("PotionType") & 0xFFFF;
069    }
070
071    public boolean isSplashPotion() {
072        return namedTag.getShort("PotionType") == POTION_TYPE_SPLASH;
073    }
074    
075    /**
076     * @deprecated Use {@link #setPotionType(int)} instead.
077     */
078    @Deprecated
079    public void setSplashPotion(boolean value) {
080        namedTag.putShort("PotionType", value ? 1 : 0);
081    }
082
083    public BlockColor getCustomColor() {
084        if (isCustomColor()) {
085            int color = namedTag.getInt("CustomColor");
086
087            int red = (color >> 16) & 0xff;
088            int green = (color >> 8) & 0xff;
089            int blue = (color) & 0xff;
090
091            return new BlockColor(red, green, blue);
092        }
093
094        return null;
095    }
096
097    public boolean isCustomColor() {
098        return namedTag.contains("CustomColor");
099    }
100
101    public void setCustomColor(BlockColor color) {
102        setCustomColor(color.getRed(), color.getGreen(), color.getBlue());
103    }
104
105    public void setCustomColor(int r, int g, int b) {
106        int color = (r << 16 | g << 8 | b) & 0xffffff;
107
108        namedTag.putInt("CustomColor", color);
109    
110        Block block = getBlock();
111        Player[] viewers = this.level.getChunkPlayers(getChunkX(), getChunkZ()).values().toArray(new Player[0]);
112        UpdateBlockPacket air = new UpdateBlockPacket();
113        air.blockRuntimeId = GlobalBlockPalette.getOrCreateRuntimeId(0, 0);
114        air.flags = UpdateBlockPacket.FLAG_ALL_PRIORITY;
115        air.x = (int) x;
116        air.y = (int) y;
117        air.z = (int) z;
118        UpdateBlockPacket self = (UpdateBlockPacket) air.clone();
119        self.blockRuntimeId = GlobalBlockPalette.getOrCreateRuntimeId(block.getId(), block.getDamage());
120        for (Player viewer : viewers) {
121            viewer.dataPacket(air);
122            viewer.dataPacket(self);
123        }
124        
125        spawnToAll();
126    }
127
128    public void clearCustomColor() {
129        namedTag.remove("CustomColor");
130        spawnToAll();
131    }
132
133    @Override
134    public boolean isBlockEntityValid() {
135        int id = getBlock().getId();
136        return id == Block.CAULDRON_BLOCK || id == Block.LAVA_CAULDRON;
137    }
138
139    @Override
140    public CompoundTag getSpawnCompound() {
141        CompoundTag compoundTag = new CompoundTag()
142                .putString("id", BlockEntity.CAULDRON)
143                .putInt("x", (int) this.x)
144                .putInt("y", (int) this.y)
145                .putInt("z", (int) this.z)
146                .putShort("PotionId", namedTag.getShort("PotionId"))
147                .putByte("PotionType", namedTag.getShort("PotionType"));
148        if (namedTag.contains("CustomColor")) {
149            compoundTag.putInt("CustomColor", namedTag.getInt("CustomColor"));
150        }
151        return compoundTag;
152    }
153}
