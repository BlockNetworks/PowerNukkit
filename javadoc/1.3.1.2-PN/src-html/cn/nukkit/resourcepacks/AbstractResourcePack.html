001package cn.nukkit.resourcepacks;
002
003import com.google.gson.JsonArray;
004import com.google.gson.JsonObject;
005
006import java.util.UUID;
007
008public abstract class AbstractResourcePack implements ResourcePack {
009    protected JsonObject manifest;
010    private UUID id = null;
011
012    protected boolean verifyManifest() {
013        if (this.manifest.has("format_version") && this.manifest.has("header") && this.manifest.has("modules")) {
014            JsonObject header = this.manifest.getAsJsonObject("header");
015            return header.has("description") &&
016                    header.has("name") &&
017                    header.has("uuid") &&
018                    header.has("version") &&
019                    header.getAsJsonArray("version").size() == 3;
020        } else {
021            return false;
022        }
023    }
024
025    @Override
026    public String getPackName() {
027        return this.manifest.getAsJsonObject("header")
028                .get("name").getAsString();
029    }
030
031    @Override
032    public UUID getPackId() {
033        if (id == null) {
034            id = UUID.fromString(this.manifest.getAsJsonObject("header").get("uuid").getAsString());
035        }
036        return id;
037    }
038
039    @Override
040    public String getPackVersion() {
041        JsonArray version = this.manifest.getAsJsonObject("header")
042                .get("version").getAsJsonArray();
043
044        return String.join(".", version.get(0).getAsString(),
045                version.get(1).getAsString(),
046                version.get(2).getAsString());
047    }
048}
