001package cn.nukkit.block;
002
003import cn.nukkit.Player;
004import cn.nukkit.Server;
005import cn.nukkit.event.block.BlockGrowEvent;
006import cn.nukkit.item.Item;
007import cn.nukkit.item.ItemSugarcane;
008import cn.nukkit.level.Level;
009import cn.nukkit.level.particle.BoneMealParticle;
010import cn.nukkit.math.BlockFace;
011import cn.nukkit.utils.BlockColor;
012
013/**
014 * Created by Pub4Game on 09.01.2016.
015 */
016public class BlockSugarcane extends BlockFlowable {
017
018    public BlockSugarcane() {
019        this(0);
020    }
021
022    public BlockSugarcane(int meta) {
023        super(meta);
024    }
025
026    @Override
027    public String getName() {
028        return "Sugarcane";
029    }
030
031    @Override
032    public int getId() {
033        return SUGARCANE_BLOCK;
034    }
035
036    @Override
037    public Item toItem() {
038        return new ItemSugarcane();
039    }
040
041    @Override
042    public boolean canBeActivated() {
043        return true;
044    }
045
046    @Override
047    public boolean onActivate(Item item, Player player) {
048        if (item.getId() == Item.DYE && item.getDamage() == 0x0F) { //Bonemeal
049            int count = 1;
050
051            for (int i = 1; i <= 2; i++) {
052                int id = this.level.getBlockIdAt(this.getFloorX(), this.getFloorY() - i, this.getFloorZ());
053
054                if (id == SUGARCANE_BLOCK) {
055                    count++;
056                }
057            }
058
059            if (count < 3) {
060                boolean success = false;
061                int toGrow = 3 - count;
062
063                for (int i = 1; i <= toGrow; i++) {
064                    Block block = this.up(i);
065                    if (block.getId() == 0) {
066                        BlockGrowEvent ev = new BlockGrowEvent(block, Block.get(BlockID.SUGARCANE_BLOCK));
067                        Server.getInstance().getPluginManager().callEvent(ev);
068
069                        if (!ev.isCancelled()) {
070                            this.getLevel().setBlock(block, ev.getNewState(), true);
071                            success = true;
072                        }
073                    } else if (block.getId() != SUGARCANE_BLOCK) {
074                        break;
075                    }
076                }
077
078                if (success) {
079                    if (player != null && (player.gamemode & 0x01) == 0) {
080                        item.count--;
081                    }
082
083                    this.level.addParticle(new BoneMealParticle(this));
084                }
085            }
086
087            return true;
088        }
089        return false;
090    }
091
092    @Override
093    public int onUpdate(int type) {
094        Level level = getLevel();
095        if (type == Level.BLOCK_UPDATE_NORMAL) {
096            level.scheduleUpdate(this, 0);
097            return type;
098        }
099        
100        if (type == Level.BLOCK_UPDATE_SCHEDULED) {
101            if (!isSupportValid()) {
102                level.useBreakOn(this);
103            }
104            return type;
105        }
106        
107        if (type == Level.BLOCK_UPDATE_RANDOM) {
108            if (!isSupportValid()) {
109                level.scheduleUpdate(this, 0);
110                return type;
111            }
112            if (getDamage() < 15) {
113                setDamage(this.getDamage() + 1);
114                level.setBlock(this, this, false);
115                return type;
116            }
117            Block up = up();
118            if (up.getId() != AIR) {
119                return type;
120            }
121            
122            int height = 0;
123            for (Block current = this; height < 3 && current.getId() == SUGARCANE_BLOCK; height++) {
124                current = current.down();
125            }
126            if (height >= 3) {
127                return type;
128            }
129
130            BlockGrowEvent ev = new BlockGrowEvent(up, Block.get(BlockID.SUGARCANE_BLOCK));
131            Server.getInstance().getPluginManager().callEvent(ev);
132
133            if (ev.isCancelled()) {
134                return type;
135            }
136            
137            if (!level.setBlock(up, Block.get(BlockID.SUGARCANE_BLOCK), false)) {
138                return type;
139            }
140            
141            setDamage(0);
142            level.setBlock(this, this, false);
143            return type;
144        }
145        return 0;
146    }
147
148    @Override
149    public boolean place(Item item, Block block, Block target, BlockFace face, double fx, double fy, double fz, Player player) {
150        if (block.getId() != AIR) {
151            return false;
152        }
153        if (isSupportValid()) {
154            this.getLevel().setBlock(block, this, true);
155            return true;
156        }
157        return false;
158    }
159
160    /**
161     * @since 1.2.0.2-PN
162     */
163    private boolean isSupportValid() {
164        Block down = this.down();
165        int downId = down.getId();
166        if (downId == SUGARCANE_BLOCK) {
167            return true;
168        }
169        if (downId != GRASS && downId != DIRT && downId != SAND) {
170            return false;
171        }
172        for (BlockFace face : BlockFace.Plane.HORIZONTAL) {
173            Block possibleWater = down.getSide(face);
174            if (possibleWater instanceof BlockWater
175                    || possibleWater instanceof BlockIceFrosted
176                    || possibleWater.getLevelBlockAtLayer(1) instanceof BlockWater) {
177                return true;
178            }
179        }
180        return false;
181    }
182
183    @Override
184    public BlockColor getColor() {
185        return BlockColor.FOLIAGE_BLOCK_COLOR;
186    }
187}
