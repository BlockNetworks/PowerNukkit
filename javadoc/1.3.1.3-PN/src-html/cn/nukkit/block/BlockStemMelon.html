001package cn.nukkit.block;
002
003import cn.nukkit.Server;
004import cn.nukkit.api.PowerNukkitDifference;
005import cn.nukkit.api.PowerNukkitOnly;
006import cn.nukkit.api.Since;
007import cn.nukkit.event.block.BlockGrowEvent;
008import cn.nukkit.item.Item;
009import cn.nukkit.item.ItemSeedsMelon;
010import cn.nukkit.level.Level;
011import cn.nukkit.math.BlockFace;
012import cn.nukkit.math.BlockFace.Plane;
013import cn.nukkit.math.NukkitRandom;
014import cn.nukkit.utils.Faceable;
015
016/**
017 * Created by Pub4Game on 15.01.2016.
018 * 
019 * @apiNote Implements {@link Faceable} only on PowerNukkit since 1.3.0.0-PN
020 */
021@PowerNukkitOnly("Implements Faceable only on PowerNukkit since 1.3.0.0-PN")
022public class BlockStemMelon extends BlockCrops implements Faceable {
023
024    public BlockStemMelon() {
025        this(0);
026    }
027
028    public BlockStemMelon(int meta) {
029        super(meta);
030    }
031
032    @Override
033    public int getId() {
034        return MELON_STEM;
035    }
036
037    @Override
038    public String getName() {
039        return "Melon Stem";
040    }
041
042    @PowerNukkitOnly("Implements Faceable only on PowerNukkit since 1.3.0.0-PN")
043    @Since("1.3.0.0-PN")
044    @Override
045    public BlockFace getBlockFace() {
046        return BlockFace.fromIndex(getDamage() >> 3 & 0b111);
047    }
048    
049    @PowerNukkitOnly
050    @Since("1.3.0.0-PN")
051    public void setBlockFace(BlockFace face) {
052        setDamage(getDamage() & ~(0b111 << 3) | (face.getIndex() << 3));
053    }
054
055    @PowerNukkitDifference(info = "Will bind to the melon by the server-side", since = "1.3.0.0-PN")
056    @Override
057    public int onUpdate(int type) {
058        if (type == Level.BLOCK_UPDATE_NORMAL) {
059            if (this.down().getId() != FARMLAND) {
060                this.getLevel().useBreakOn(this);
061                return Level.BLOCK_UPDATE_NORMAL;
062            }
063        } else if (type == Level.BLOCK_UPDATE_RANDOM) {
064            NukkitRandom random = new NukkitRandom();
065            if (random.nextRange(1, 2) == 1 && getLevel().getFullLight(this) >= MINIMUM_LIGHT_LEVEL) {
066                if (this.getDamage() < 0x07) {
067                    Block block = this.clone();
068                    block.setDamage(block.getDamage() + 1);
069                    BlockGrowEvent ev = new BlockGrowEvent(this, block);
070                    Server.getInstance().getPluginManager().callEvent(ev);
071                    if (!ev.isCancelled()) {
072                        this.getLevel().setBlock(this, ev.getNewState(), true);
073                    }
074                    return Level.BLOCK_UPDATE_RANDOM;
075                } else {
076                    for (BlockFace face : Plane.HORIZONTAL) {
077                        Block b = this.getSide(face);
078                        if (b.getId() == MELON_BLOCK) {
079                            return Level.BLOCK_UPDATE_RANDOM;
080                        }
081                    }
082                    BlockFace sideFace = Plane.HORIZONTAL.random(random);
083                    Block side = this.getSide(sideFace);
084                    Block d = side.down();
085                    if (side.getId() == AIR && (d.getId() == FARMLAND || d.getId() == GRASS || d.getId() == DIRT)) {
086                        BlockGrowEvent ev = new BlockGrowEvent(side, Block.get(BlockID.MELON_BLOCK));
087                        Server.getInstance().getPluginManager().callEvent(ev);
088                        if (!ev.isCancelled()) {
089                            this.getLevel().setBlock(side, ev.getNewState(), true);
090                            setBlockFace(sideFace);
091                            this.getLevel().setBlock(this, this, true);
092                        }
093                    }
094                }
095            }
096            return Level.BLOCK_UPDATE_RANDOM;
097        }
098        return 0;
099    }
100
101    @Override
102    public Item toItem() {
103        return new ItemSeedsMelon();
104    }
105
106    @Override
107    public Item[] getDrops(Item item) {
108        NukkitRandom random = new NukkitRandom();
109        return new Item[]{
110                new ItemSeedsMelon(0, random.nextRange(0, 3))
111        };
112    }
113}
