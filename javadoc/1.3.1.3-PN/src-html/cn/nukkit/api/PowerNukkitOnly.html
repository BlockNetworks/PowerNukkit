001package cn.nukkit.api;
002
003import java.lang.annotation.*;
004
005/**
006 * Indicates that the annotated element is only available in PowerNukkit environment
007 * and will cause issues when used in a normal NukkitX server without PowerNukkit's patches and features. 
008 */
009@Retention(RetentionPolicy.CLASS)
010@Target({ElementType.CONSTRUCTOR, ElementType.METHOD, ElementType.ANNOTATION_TYPE, ElementType.TYPE,
011        ElementType.FIELD, ElementType.PACKAGE})
012@PowerNukkitOnly @Since("1.2.1.0-PN")
013@Inherited
014@Documented
015public @interface PowerNukkitOnly {
016    String value() default "";
017}
