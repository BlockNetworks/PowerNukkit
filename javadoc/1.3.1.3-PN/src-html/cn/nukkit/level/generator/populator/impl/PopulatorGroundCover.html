001package cn.nukkit.level.generator.populator.impl;
002
003import cn.nukkit.block.Block;
004import cn.nukkit.block.BlockID;
005import cn.nukkit.level.ChunkManager;
006import cn.nukkit.level.biome.EnumBiome;
007import cn.nukkit.level.format.FullChunk;
008import cn.nukkit.level.biome.Biome;
009import cn.nukkit.level.biome.type.CoveredBiome;
010import cn.nukkit.level.generator.Normal;
011import cn.nukkit.level.generator.populator.type.Populator;
012import cn.nukkit.math.NukkitRandom;
013
014/**
015 * @author DaPorkchop_
016 */
017public class PopulatorGroundCover extends Populator {
018    public static final int STONE = BlockID.STONE << Block.DATA_BITS;
019
020    @Override
021    public void populate(ChunkManager level, int chunkX, int chunkZ, NukkitRandom random, FullChunk chunk) {
022        //reverse iteration to 0 is faster
023        for (int x = 15; x >= 0; x--) {
024            for (int z = 15; z >= 0; z--) {
025                Biome realBiome = EnumBiome.getBiome(chunk.getBiomeId(x, z));
026                if (realBiome instanceof CoveredBiome) {
027                    ((CoveredBiome) realBiome).doCover(x, z, chunk);
028                }
029            }
030        }
031    }
032}
