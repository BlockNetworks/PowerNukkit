009public class SimplexD extends PerlinD {
010    protected static double SQRT_3;
011    protected static double SQRT_5;
012    protected static double F2;
013    protected static double G2;
014    protected static double G22;
015    protected static double F3;
016    protected static double G3;
017    protected static double F4;
018    protected static double G4;
019    protected static double G42;
020    protected static double G43;
021    protected static double G44;
022    public static final int[][] grad3 = {
023            {1, 1, 0}, {-1, 1, 0}, {1, -1, 0}, {-1, -1, 0},
024            {1, 0, 1}, {-1, 0, 1}, {1, 0, -1}, {-1, 0, -1},
025            {0, 1, 1}, {0, -1, 1}, {0, 1, -1}, {0, -1, -1}
026    };
027    protected final double offsetW;
028
029    public SimplexD(NukkitRandom random, double octaves, double persistence) {
030        super(random, octaves, persistence);
031        this.offsetW = random.nextDouble() * 256;
032        SQRT_3 = Math.sqrt(3);
033        SQRT_5 = Math.sqrt(5);
034        F2 = 0.5 * (SQRT_3 - 1);
035        G2 = (3 - SQRT_3) / 6;
036        G22 = G2 * 2.0 - 1;
037        F3 = 1.0 / 3.0;
038        G3 = 1.0 / 6.0;
039        F4 = (SQRT_5 - 1.0) / 4.0;
040        G4 = (5.0 - SQRT_5) / 20.0;
041        G42 = G4 * 2.0;
042        G43 = G4 * 3.0;
043        G44 = G4 * 4.0 - 1.0;
044    }
045
046    public SimplexD(NukkitRandom random, double octaves, double persistence, double expansion) {
047        super(random, octaves, persistence, expansion);
048        this.offsetW = random.nextDouble() * 256;
049        SQRT_3 = Math.sqrt(3);
050        SQRT_5 = Math.sqrt(5);
051        F2 = 0.5 * (SQRT_3 - 1);
052        G2 = (3 - SQRT_3) / 6;
053        G22 = G2 * 2.0 - 1;
054        F3 = 1.0 / 3.0;
055        G3 = 1.0 / 6.0;
056        F4 = (SQRT_5 - 1.0) / 4.0;
057        G4 = (5.0 - SQRT_5) / 20.0;
058        G42 = G4 * 2.0;
059        G43 = G4 * 3.0;
060        G44 = G4 * 4.0 - 1.0;
061    }

064    protected static double dot2D(int[] g, double x, double y) {
065        return g[0] * x + g[1] * y;
066    }
067
068    protected static double dot3D(int[] g, double x, double y, double z) {
069        return g[0] * x + g[1] * y + g[2] * z;
070    }
071
072    protected static double dot4D(int[] g, double x, double y, double z, double w) {
073        return g[0] * x + g[1] * y + g[2] * z + g[3] * w;
074    }

076    @Override
077    public double getNoise3D(double x, double y, double z) {
078        x += this.offsetX;
079        y += this.offsetY;
080        z += this.offsetZ;
081
082        // Skew the input space to determine which simplex cell we're in
083        double s = (x + y + z) * F3; // Very nice and simple skew factor for 3D
084        int i = (int) (x + s);
085        int j = (int) (y + s);
086        int k = (int) (z + s);
087        double t = (i + j + k) * G3;
088        // Unskew the cell origin back to (x,y,z) space
089        double x0 = x - (i - t); // The x,y,z distances from the cell origin
090        double y0 = y - (j - t);
091        double z0 = z - (k - t);
092
093        // For the 3D case, the simplex shape is a slightly irregular tetrahedron.
094        int i1 = 0;
095        int j1 = 0;
096        int k1 = 0;
097        int i2 = 0;
098        int j2 = 0;
099        int k2 = 0;
100
101        // Determine which simplex we are in.
102        if (x0 >= y0) {
103            if (y0 >= z0) {
104                i1 = 1;
105                j1 = 0;
106                k1 = 0;
107                i2 = 1;
108                j2 = 1;
109                k2 = 0;
110            } // X Y Z order
111            else if (x0 >= z0) {
112                i1 = 1;
113                j1 = 0;
114                k1 = 0;
115                i2 = 1;
116                j2 = 0;
117                k2 = 1;
118            } // X Z Y order
119            else {
120                i1 = 0;
121                j1 = 0;
122                k1 = 1;
123                i2 = 1;
124                j2 = 0;
125                k2 = 1;
126            }
127            // Z X Y order
128        } else { // x0<y0
129            if (y0 < z0) {
130                i1 = 0;
131                j1 = 0;
132                k1 = 1;
133                i2 = 0;
134                j2 = 1;
135                k2 = 1;
136            } // Z Y X order
137            else if (x0 < z0) {
138                i1 = 0;
139                j1 = 1;
140                k1 = 0;
141                i2 = 0;
142                j2 = 1;
143                k2 = 1;
144            } // Y Z X order
145            else {
146                i1 = 0;
147                j1 = 1;
148                k1 = 0;
149                i2 = 1;
150                j2 = 1;
151                k2 = 0;
152            }
153            // Y X Z order
154        }
155
156        // A step of (1,0,0) in (i,j,k) means a step of (1-c,-c,-c) in (x,y,z),
157        // a step of (0,1,0) in (i,j,k) means a step of (-c,1-c,-c) in (x,y,z), and
158        // a step of (0,0,1) in (i,j,k) means a step of (-c,-c,1-c) in (x,y,z), where
159        // c = 1/6.
160        double x1 = x0 - i1 + G3; // Offsets for second corner in (x,y,z) coords
161        double y1 = y0 - j1 + G3;
162        double z1 = z0 - k1 + G3;
163        double x2 = x0 - i2 + 2.0 * G3; // Offsets for third corner in (x,y,z) coords
164        double y2 = y0 - j2 + 2.0 * G3;
165        double z2 = z0 - k2 + 2.0 * G3;
166        double x3 = x0 - 1.0 + 3.0 * G3; // Offsets for last corner in (x,y,z) coords
167        double y3 = y0 - 1.0 + 3.0 * G3;
168        double z3 = z0 - 1.0 + 3.0 * G3;
169
170        // Work out the hashed gradient indices of the four simplex corners
171        int ii = i & 255;
172        int jj = j & 255;
173        int kk = k & 255;
174
175        double n = 0;
176
177        // Calculate the contribution from the four corners
178        double t0 = 0.6 - x0 * x0 - y0 * y0 - z0 * z0;
179        if (t0 > 0) {
180            int[] gi0 = grad3[this.perm[ii + this.perm[jj + this.perm[kk]]] % 12];
181            n += t0 * t0 * t0 * t0 * (gi0[0] * x0 + gi0[1] * y0 + gi0[2] * z0);
182        }
183
184        double t1 = 0.6 - x1 * x1 - y1 * y1 - z1 * z1;
185        if (t1 > 0) {
186            int[] gi1 = grad3[this.perm[ii + i1 + this.perm[jj + j1 + this.perm[kk + k1]]] % 12];
187            n += t1 * t1 * t1 * t1 * (gi1[0] * x1 + gi1[1] * y1 + gi1[2] * z1);
188        }
189
190        double t2 = 0.6 - x2 * x2 - y2 * y2 - z2 * z2;
191        if (t2 > 0) {
192            int[] gi2 = grad3[this.perm[ii + i2 + this.perm[jj + j2 + this.perm[kk + k2]]] % 12];
193            n += t2 * t2 * t2 * t2 * (gi2[0] * x2 + gi2[1] * y2 + gi2[2] * z2);
194        }
195
196        double t3 = 0.6 - x3 * x3 - y3 * y3 - z3 * z3;
197        if (t3 > 0) {
198            int[] gi3 = grad3[this.perm[ii + 1 + this.perm[jj + 1 + this.perm[kk + 1]]] % 12];
199            n += t3 * t3 * t3 * t3 * (gi3[0] * x3 + gi3[1] * y3 + gi3[2] * z3);
200        }
201
202        // Add contributions from each corner to get the noise value.
203        // The result is scaled to stay just inside [-1,1]
204        return 32.0 * n;
205    }

207    @Override
208    public double getNoise2D(double x, double y) {
209        x += this.offsetX;
210        y += this.offsetY;
211
212        // Skew the input space to determine which simplex cell we're in
213        double s = (x + y) * F2; // Hairy factor for 2D
214        int i = (int) (x + s);
215        int j = (int) (y + s);
216        double t = (i + j) * G2;
217        // Unskew the cell origin back to (x,y) space
218        double x0 = x - (i - t); // The x,y distances from the cell origin
219        double y0 = y - (j - t);
220
221        // For the 2D case, the simplex shape is an equilateral triangle.
222        int i1 = 0;
223        int j1 = 0;
224        // Determine which simplex we are in.
225        if (x0 > y0) {
226            i1 = 1;
227            j1 = 0;
228        } // lower triangle, XY order: (0,0).(1,0).(1,1)
229        else {
230            i1 = 0;
231            j1 = 1;
232        }
233        // upper triangle, YX order: (0,0).(0,1).(1,1)
234
235        // A step of (1,0) in (i,j) means a step of (1-c,-c) in (x,y), and
236        // a step of (0,1) in (i,j) means a step of (-c,1-c) in (x,y), where
237        // c = (3-sqrt(3))/6
238
239        double x1 = x0 - i1 + G2; // Offsets for middle corner in (x,y) unskewed coords
240        double y1 = y0 - j1 + G2;
241        double x2 = x0 + G22; // Offsets for last corner in (x,y) unskewed coords
242        double y2 = y0 + G22;
243
244        // Work out the hashed gradient indices of the three simplex corners
245        int ii = i & 255;
246        int jj = j & 255;
247
248        double n = 0;
249
250        // Calculate the contribution from the three corners
251        double t0 = 0.5 - x0 * x0 - y0 * y0;
252        if (t0 > 0) {
253            int[] gi0 = grad3[this.perm[ii + this.perm[jj]] % 12];
254            n += t0 * t0 * t0 * t0 * (gi0[0] * x0 + gi0[1] * y0); // (x,y) of grad3 used for 2D gradient
255        }
256
257        double t1 = 0.5 - x1 * x1 - y1 * y1;
258        if (t1 > 0) {
259            int[] gi1 = grad3[this.perm[ii + i1 + this.perm[jj + j1]] % 12];
260            n += t1 * t1 * t1 * t1 * (gi1[0] * x1 + gi1[1] * y1);
261        }
262
263        double t2 = 0.5 - x2 * x2 - y2 * y2;
264        if (t2 > 0) {
265            int[] gi2 = grad3[this.perm[ii + 1 + this.perm[jj + 1]] % 12];
266            n += t2 * t2 * t2 * t2 * (gi2[0] * x2 + gi2[1] * y2);
267        }
268
269        // Add contributions from each corner to get the noise value.
270        // The result is scaled to return values in the interval [-1,1].
271        return 70.0 * n;
272    }
