001package cn.nukkit.level.generator.noise.vanilla.d;
002
003import cn.nukkit.math.MathHelper;
004import cn.nukkit.math.NukkitRandom;
005
006public class NoiseGeneratorOctavesD {
007    /**
008     * Collection of noise generation functions.  Output is combined to produce different octaves of noise.
009     */
010    private final NoiseGeneratorImprovedD[] generatorCollection;
011    private final int octaves;
012
013    public NoiseGeneratorOctavesD(NukkitRandom seed, int octavesIn) {
014        this.octaves = octavesIn;
015        this.generatorCollection = new NoiseGeneratorImprovedD[octavesIn];
016
017        for (int i = 0; i < octavesIn; ++i) {
018            this.generatorCollection[i] = new NoiseGeneratorImprovedD(seed);
019        }
020    }
021
022    /*
023     * pars:(par2,3,4=noiseOffset ; so that adjacent noise segments connect) (pars5,6,7=x,y,zArraySize),(pars8,10,12 =
024     * x,y,z noiseScale)
025     */
026    public double[] generateNoiseOctaves(double[] noiseArray, int xOffset, int yOffset, int zOffset, int xSize, int ySize, int zSize, double xScale, double yScale, double zScale) {
027        if (noiseArray == null) {
028            noiseArray = new double[xSize * ySize * zSize];
029        } else {
030            for (int i = 0; i < noiseArray.length; ++i) {
031                noiseArray[i] = 0.0D;
032            }
033        }
034
035        double d3 = 1.0D;
036
037        for (int j = 0; j < this.octaves; ++j) {
038            double d0 = (double) xOffset * d3 * xScale;
039            double d1 = (double) yOffset * d3 * yScale;
040            double d2 = (double) zOffset * d3 * zScale;
041            long k = MathHelper.floor_double_long(d0);
042            long l = MathHelper.floor_double_long(d2);
043            d0 = d0 - (double) k;
044            d2 = d2 - (double) l;
045            k = k % 16777216L;
046            l = l % 16777216L;
047            d0 = d0 + (double) k;
048            d2 = d2 + (double) l;
049            this.generatorCollection[j].populateNoiseArray(noiseArray, d0, d1, d2, xSize, ySize, zSize, xScale * d3, yScale * d3, zScale * d3, d3);
050            d3 /= 2.0D;
051        }
052
053        return noiseArray;
054    }
055
056    /*
057     * Bouncer function to the main one with some default arguments.
058     */
059    public double[] generateNoiseOctaves(double[] noiseArray, int xOffset, int zOffset, int xSize, int zSize, double xScale, double zScale, double p_76305_10_) {
060        return this.generateNoiseOctaves(noiseArray, xOffset, 10, zOffset, xSize, 1, zSize, xScale, 1.0D, zScale);
061    }
062}
