025    @Override
026    protected boolean isChunkGenerated(int index) {
027        Integer[] array = this.locationTable.get(index);
028        return !(array[0] == 0 || array[1] == 0);
029    }

031    @Override
032    public Chunk readChunk(int x, int z) throws IOException {
033        int index = getChunkOffset(x, z);
034        if (index < 0 || index >= 4096) {
035            return null;
036        }
037
038        this.lastUsed = System.currentTimeMillis();
039
040        if (!this.isChunkGenerated(index)) {
041            return null;
042        }
043
044        try {
045        Integer[] table = this.locationTable.get(index);
046        RandomAccessFile raf = this.getRandomAccessFile();
047        raf.seek(table[0] << 12);
048        int length = raf.readInt();
049        byte compression = raf.readByte();
050        if (length <= 0 || length >= MAX_SECTOR_LENGTH) {
051            if (length >= MAX_SECTOR_LENGTH) {
052                table[0] = ++this.lastSector;
053                table[1] = 1;
054                this.locationTable.put(index, table);
055                MainLogger.getLogger().error("Corrupted chunk header detected");
056            }
057            return null;
058        }
059
060        if (length > (table[1] << 12)) {
061            MainLogger.getLogger().error("Corrupted bigger chunk detected");
062            table[1] = length >> 12;
063            this.locationTable.put(index, table);
064            this.writeLocationIndex(index);
065        } else if (compression != COMPRESSION_ZLIB && compression != COMPRESSION_GZIP) {
066            MainLogger.getLogger().error("Invalid compression type");
067            return null;
068        }
069
070        byte[] data = new byte[length - 1];
071        raf.readFully(data);
072        Chunk chunk = this.unserializeChunk(data);
073        if (chunk != null) {
074            return chunk;
075        } else {
076            MainLogger.getLogger().error("Corrupted chunk detected at (" + x + ", " + z + ") in " + levelProvider.getName());
077            return null;
078        }
079        } catch (EOFException e) {
080            MainLogger.getLogger().error("Your world is corrupt, because some code is bad and corrupted it. oops. ");
081            return null;
082        }
083    }
084
085    @Override
086    protected Chunk unserializeChunk(byte[] data) {
087        return Chunk.fromBinary(data, this.levelProvider);
088    }

090    @Override
091    public boolean chunkExists(int x, int z) {
092        return this.isChunkGenerated(getChunkOffset(x, z));

095    @Override
096    protected void saveChunk(int x, int z, byte[] chunkData) throws IOException {
097        int length = chunkData.length + 1;
098        if (length + 4 > MAX_SECTOR_LENGTH) {
099            throw new ChunkException("Chunk is too big! " + (length + 4) + " > " + MAX_SECTOR_LENGTH);
100        }
101        int sectors = (int) Math.ceil((length + 4) / 4096d);
102        int index = getChunkOffset(x, z);
103        boolean indexChanged = false;
104        Integer[] table = this.locationTable.get(index);
105
106        if (table[1] < sectors) {
107            table[0] = this.lastSector + 1;
108            this.locationTable.put(index, table);
109            this.lastSector += sectors;
110            indexChanged = true;
111        } else if (table[1] != sectors) {
112            indexChanged = true;
113        }
114
115        table[1] = sectors;
116        table[2] = (int) (System.currentTimeMillis() / 1000d);
117
118        this.locationTable.put(index, table);
119        RandomAccessFile raf = this.getRandomAccessFile();
120        raf.seek(table[0] << 12);
121
122        BinaryStream stream = new BinaryStream();
123        stream.put(Binary.writeInt(length));
124        stream.putByte(COMPRESSION_ZLIB);
125        stream.put(chunkData);
126        byte[] data = stream.getBuffer();
127        if (data.length < sectors << 12) {
128            byte[] newData = new byte[sectors << 12];
129            System.arraycopy(data, 0, newData, 0, data.length);
130            data = newData;
131        }
132
133        raf.write(data);
134
135        if (indexChanged) {
136            this.writeLocationIndex(index);
137        }
138    }

140    @Override
141    public void removeChunk(int x, int z) {
142        int index = getChunkOffset(x, z);
143        Integer[] table = this.locationTable.get(0);
144        table[0] = 0;
145        table[1] = 0;
146        this.locationTable.put(index, table);
147    }

149    @Override
150    public void writeChunk(FullChunk chunk) throws Exception {
151        this.lastUsed = System.currentTimeMillis();
152        byte[] chunkData = chunk.toBinary();
153        this.saveChunk(chunk.getX() & 0x1f, chunk.getZ() & 0x1f, chunkData);
154    }
155
156    protected static int getChunkOffset(int x, int z) {
157        return x | (z << 5);
158    }
159
160    @Override
161    public void close() throws IOException {
162        this.writeLocationTable();
163        this.levelProvider = null;
164        super.close();
165    }

167    @Override
168    public int doSlowCleanUp() throws Exception {
169        RandomAccessFile raf = this.getRandomAccessFile();
170        for (int i = 0; i < 1024; i++) {
171            Integer[] table = this.locationTable.get(i);
172            if (table[0] == 0 || table[1] == 0) {
173                continue;
174            }
175            raf.seek(table[0] << 12);
176            byte[] chunk = new byte[table[1] << 12];
177            raf.readFully(chunk);
178            int length = Binary.readInt(Arrays.copyOfRange(chunk, 0, 3));
179            if (length <= 1) {
180                this.locationTable.put(i, (table = new Integer[]{0, 0, 0}));
181            }
182            try {
183                chunk = Zlib.inflate(Arrays.copyOf(chunk, 5));
184            } catch (Exception e) {
185                this.locationTable.put(i, new Integer[]{0, 0, 0});
186                continue;
187            }
188            chunk = Zlib.deflate(chunk, 9);
189            ByteBuffer buffer = ByteBuffer.allocate(4 + 1 + chunk.length);
190            buffer.put(Binary.writeInt(chunk.length + 1));
191            buffer.put(COMPRESSION_ZLIB);
192            buffer.put(chunk);
193            chunk = buffer.array();
194            int sectors = (int) Math.ceil(chunk.length / 4096d);
195            if (sectors > table[1]) {
196                table[0] = this.lastSector + 1;
197                this.lastSector += sectors;
198                this.locationTable.put(i, table);
199            }
200            raf.seek(table[0] << 12);
201            byte[] bytes = new byte[sectors << 12];
202            ByteBuffer buffer1 = ByteBuffer.wrap(bytes);
203            buffer1.put(chunk);
204            raf.write(buffer1.array());
205        }
206        this.writeLocationTable();
207        int n = this.cleanGarbage();
208        this.writeLocationTable();
209        return n;
210    }

212    @Override
213    protected void loadLocationTable() throws IOException {
214        RandomAccessFile raf = this.getRandomAccessFile();
215        raf.seek(0);
216        this.lastSector = 1;
217        int[] data = new int[1024 * 2]; //1024 records * 2 times
218        for (int i = 0; i < 1024 * 2; i++) {
219            data[i] = raf.readInt();
220        }
221        for (int i = 0; i < 1024; ++i) {
222            int index = data[i];
223            this.locationTable.put(i, new Integer[]{index >> 8, index & 0xff, data[1024 + i]});
224            int value = this.locationTable.get(i)[0] + this.locationTable.get(i)[1] - 1;
225            if (value > this.lastSector) {
226                this.lastSector = value;
227            }
228        }
229    }
230
231    private void writeLocationTable() throws IOException {
232        RandomAccessFile raf = this.getRandomAccessFile();
233        raf.seek(0);
234        for (int i = 0; i < 1024; ++i) {
235            Integer[] array = this.locationTable.get(i);
236            raf.writeInt((array[0] << 8) | array[1]);
237        }
238        for (int i = 0; i < 1024; ++i) {
239            Integer[] array = this.locationTable.get(i);
240            raf.writeInt(array[2]);
241        }
242    }
243
244    private int cleanGarbage() throws IOException {
245        RandomAccessFile raf = this.getRandomAccessFile();
246        Map<Integer, Integer> sectors = new TreeMap<>();
247        for (Map.Entry entry : this.locationTable.entrySet()) {
248            Integer index = (Integer) entry.getKey();
249            Integer[] data = (Integer[]) entry.getValue();
250            if (data[0] == 0 || data[1] == 0) {
251                this.locationTable.put(index, new Integer[]{0, 0, 0});
252                continue;
253            }
254            sectors.put(data[0], index);
255        }
256
257        if (sectors.size() == (this.lastSector - 2)) {
258            return 0;
259        }
260        int shift = 0;
261        int lastSector = 1;
262
263        raf.seek(8192);
264        int s = 2;
265        for (int sector : sectors.keySet()) {
266            s = sector;
267            int index = sectors.get(sector);
268            if ((sector - lastSector) > 1) {
269                shift += sector - lastSector - 1;
270            }
271            if (shift > 0) {
272                raf.seek(sector << 12);
273                byte[] old = new byte[4096];
274                raf.readFully(old);
275                raf.seek((sector - shift) << 12);
276                raf.write(old);
277            }
278            Integer[] v = this.locationTable.get(index);
279            v[0] -= shift;
280            this.locationTable.put(index, v);
281            this.lastSector = sector;
282        }
283        raf.setLength((s + 1) << 12);
284        return shift;
285    }

287    @Override
288    protected void writeLocationIndex(int index) throws IOException {
289        RandomAccessFile raf = this.getRandomAccessFile();
290        Integer[] array = this.locationTable.get(index);
291        raf.seek(index << 2);
292        raf.writeInt((array[0] << 8) | array[1]);
293        raf.seek(4096 + (index << 2));
294        raf.writeInt(array[2]);
295    }

297    @Override
298    protected void createBlank() throws IOException {
299        RandomAccessFile raf = this.getRandomAccessFile();
300        raf.seek(0);
301        raf.setLength(0);
302        this.lastSector = 1;
303        int time = (int) (System.currentTimeMillis() / 1000d);
304        for (int i = 0; i < 1024; ++i) {
305            this.locationTable.put(i, new Integer[]{0, 0, time});
306            raf.writeInt(0);
307        }
308        for (int i = 0; i < 1024; ++i) {
309            raf.writeInt(time);
310        }
311    }
