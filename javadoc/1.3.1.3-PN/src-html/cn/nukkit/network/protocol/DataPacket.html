001package cn.nukkit.network.protocol;
002
003import cn.nukkit.Server;
004import cn.nukkit.network.Network;
005import cn.nukkit.utils.Binary;
006import cn.nukkit.utils.BinaryStream;
007import cn.nukkit.utils.Zlib;
008import com.nukkitx.network.raknet.RakNetReliability;
009
010/**
011 * author: MagicDroidX
012 * Nukkit Project
013 */
014public abstract class DataPacket extends BinaryStream implements Cloneable {
015
016    public boolean isEncoded = false;
017    private int channel = 0;
018
019    public RakNetReliability reliability = RakNetReliability.RELIABLE_ORDERED;
020
021    public abstract byte pid();
022
023    public abstract void decode();
024
025    public abstract void encode();
026
027    @Override
028    public DataPacket reset() {
029        super.reset();
030        this.putUnsignedVarInt(this.pid() & 0xff);
031        return this;
032    }
033
034    public void setChannel(int channel) {
035        this.channel = channel;
036    }
037
038    public int getChannel() {
039        return channel;
040    }
041
042    public DataPacket clean() {
043        this.setBuffer(null);
044        this.setOffset(0);
045        this.isEncoded = false;
046        return this;
047    }
048
049    @Override
050    public DataPacket clone() {
051        try {
052            return (DataPacket) super.clone();
053        } catch (CloneNotSupportedException e) {
054            return null;
055        }
056    }
057
058    public BatchPacket compress() {
059        return compress(Server.getInstance().networkCompressionLevel);
060    }
061
062    public BatchPacket compress(int level) {
063        BatchPacket batch = new BatchPacket();
064        byte[][] batchPayload = new byte[2][];
065        byte[] buf = getBuffer();
066        batchPayload[0] = Binary.writeUnsignedVarInt(buf.length);
067        batchPayload[1] = buf;
068        byte[] data = Binary.appendBytes(batchPayload);
069        try {
070            batch.payload = Network.deflateRaw(data, level);
071        } catch (Exception e) {
072            throw new RuntimeException(e);
073        }
074        return batch;
075    }
076}
