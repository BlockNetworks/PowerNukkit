001package cn.nukkit.network.protocol.types;
002
003import cn.nukkit.api.DeprecationDetails;
004import cn.nukkit.api.PowerNukkitOnly;
005import cn.nukkit.api.Since;
006
007public class EntityLink {
008
009    public static final byte TYPE_REMOVE = 0;
010    public static final byte TYPE_RIDER = 1;
011    public static final byte TYPE_PASSENGER = 2;
012
013    public long fromEntityUniquieId;
014    public long toEntityUniquieId;
015    public byte type;
016    public boolean immediate;
017    
018    @Since("1.3.0.0-PN")
019    public boolean riderInitiated;
020
021    @Since("1.3.0.0-PN")
022    public EntityLink(long fromEntityUniquieId, long toEntityUniquieId, byte type, boolean immediate, boolean riderInitiated) {
023        this.fromEntityUniquieId = fromEntityUniquieId;
024        this.toEntityUniquieId = toEntityUniquieId;
025        this.type = type;
026        this.immediate = immediate;
027        this.riderInitiated = riderInitiated;
028    }
029
030    @Since("1.3.0.0-PN")
031    @PowerNukkitOnly("Backward compatibility")
032    @Deprecated
033    @DeprecationDetails(
034            since = "1.3.0.0-PN", toBeRemovedAt = "1.4.0.0-PN",
035            reason = "NukkitX added the immediate riderInitiated", 
036            replaceWith = "EntityLink(long fromEntityUniquieId, long toEntityUniquieId, byte type, boolean immediate, boolean riderInitiated)")
037    public EntityLink(long fromEntityUniquieId, long toEntityUniquieId, byte type, boolean immediate) {
038        this(fromEntityUniquieId, toEntityUniquieId, type, immediate, false);
039    }
040}
