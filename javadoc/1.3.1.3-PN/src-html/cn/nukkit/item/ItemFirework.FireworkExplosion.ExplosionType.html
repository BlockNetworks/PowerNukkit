021/**
022 * @author CreeperFace
023 */
024public class ItemFirework extends Item {
025
026    public ItemFirework() {
027        this(0);
028    }
029
030    public ItemFirework(Integer meta) {
031        this(meta, 1);
032    }
033    
034    @PowerNukkitDifference(info = "Will not add compound tag automatically")
035    public ItemFirework(Integer meta, int count) {
036        super(FIREWORKS, meta, count, "Fireworks");
037    }
038
039    @Override
040    public boolean canBeActivated() {
041        return true;
042    }
043
044    @Override
045    public boolean onActivate(Level level, Player player, Block block, Block target, BlockFace face, double fx, double fy, double fz) {
046        if (block.canPassThrough()) {
047            this.spawnFirework(level, block);
048
049            if (!player.isCreative()) {
050                player.getInventory().decreaseCount(player.getInventory().getHeldItemIndex());
051            }
052
053            return true;
054        }
055
056        return false;
057    }
058
059    @Override
060    public boolean onClickAir(Player player, Vector3 directionVector) {
061        if (player.getInventory().getChestplate() instanceof ItemElytra && player.isGliding()) {
062            this.spawnFirework(player.getLevel(), player);
063
064            player.setMotion(new Vector3(
065                    -Math.sin(Math.toRadians(player.yaw)) * Math.cos(Math.toRadians(player.pitch)) * 2,
066                    -Math.sin(Math.toRadians(player.pitch)) * 2,
067                    Math.cos(Math.toRadians(player.yaw)) * Math.cos(Math.toRadians(player.pitch)) * 2));
068
069            if (!player.isCreative()) {
070                this.count--;
071            }
072
073            return true;
074        }
075
076        return false;
077    }
078
079    public void addExplosion(FireworkExplosion explosion) {
080        List<DyeColor> colors = explosion.getColors();
081        List<DyeColor> fades = explosion.getFades();
082
083        if (colors.isEmpty()) {
084            return;
085        }
086        byte[] clrs = new byte[colors.size()];
087        for (int i = 0; i < clrs.length; i++) {
088            clrs[i] = (byte) colors.get(i).getDyeData();
089        }
090
091        byte[] fds = new byte[fades.size()];
092        for (int i = 0; i < fds.length; i++) {
093            fds[i] = (byte) fades.get(i).getDyeData();
094        }
095
096        ListTag<CompoundTag> explosions = this.getNamedTag().getCompound("Fireworks").getList("Explosions", CompoundTag.class);
097        CompoundTag tag = new CompoundTag()
098                .putByteArray("FireworkColor", clrs)
099                .putByteArray("FireworkFade", fds)
100                .putBoolean("FireworkFlicker", explosion.flicker)
101                .putBoolean("FireworkTrail", explosion.trail)
102                .putByte("FireworkType", explosion.type.ordinal());
103
104        explosions.add(tag);
105    }
106
107    public void clearExplosions() {
108        this.getNamedTag().getCompound("Fireworks").putList(new ListTag<CompoundTag>("Explosions"));
109    }
110
111    private void spawnFirework(Level level, Vector3 pos) {
112        CompoundTag nbt = new CompoundTag()
113                .putList(new ListTag<DoubleTag>("Pos")
114                        .add(new DoubleTag("", pos.x + 0.5))
115                        .add(new DoubleTag("", pos.y + 0.5))
116                        .add(new DoubleTag("", pos.z + 0.5)))
117                .putList(new ListTag<DoubleTag>("Motion")
118                        .add(new DoubleTag("", 0))
119                        .add(new DoubleTag("", 0))
120                        .add(new DoubleTag("", 0)))
121                .putList(new ListTag<FloatTag>("Rotation")
122                        .add(new FloatTag("", 0))
123                        .add(new FloatTag("", 0)))
124                .putCompound("FireworkItem", NBTIO.putItemHelper(this));
125
126        EntityFirework entity = (EntityFirework) Entity.createEntity("Firework", level.getChunk(pos.getFloorX() >> 4, pos.getFloorZ() >> 4), nbt);
127        if (entity != null) {
128            entity.spawnToAll();
129        }
130    }
131
132    public static class FireworkExplosion {
133
134        private List<DyeColor> colors = new ArrayList<>();
135        private List<DyeColor> fades = new ArrayList<>();
136        private boolean flicker = false;
137        private boolean trail = false;
138        private ExplosionType type = ExplosionType.CREEPER_SHAPED;
139
140        public List<DyeColor> getColors() {
141            return this.colors;
142        }
143
144        public List<DyeColor> getFades() {
145            return this.fades;
146        }
147
148        public boolean hasFlicker() {
149            return this.flicker;
150        }
151
152        public boolean hasTrail() {
153            return this.trail;
154        }
155
156        public ExplosionType getType() {
157            return this.type;
158        }
159
160        public FireworkExplosion setFlicker(boolean flicker) {
161            this.flicker = flicker;
162            return this;
163        }
164
165        public FireworkExplosion setTrail(boolean trail) {
166            this.trail = trail;
167            return this;
168        }
169
170        public FireworkExplosion type(ExplosionType type) {
171            this.type = type;
172            return this;
173        }
174
175        public FireworkExplosion addColor(DyeColor color) {
176            colors.add(color);
177            return this;
178        }
179
180        public FireworkExplosion addFade(DyeColor fade) {
181            fades.add(fade);
182            return this;
183        }
184
185        public enum ExplosionType {
186            SMALL_BALL,
187            LARGE_BALL,
188            STAR_SHAPED,
189            CREEPER_SHAPED,
190            BURST
191        }
192    }
193}
