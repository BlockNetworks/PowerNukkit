001package cn.nukkit.event;
002
003import cn.nukkit.utils.EventException;
004
005/**
006 * 描述服务器中可能发生的事情的类。<br>
007 * Describes things that happens in the server.
008 *
009 * <p>服务器中可能发生的事情称作<b>事件</b>。定义一个需要它在一个事件发生时被运行的过程，这个过程称作<b>监听器</b>。<br>
010 * Things that happens in the server is called a <b>event</b>. Define a procedure that should be executed
011 * when a event happens, this procedure is called a <b>listener</b>.</p>
012 *
013 * <p>Nukkit调用事件的处理器时，会通过参数的类型判断需要被监听的事件。<br>
014 * When Nukkit is calling a handler, the event needed to listen is judged by the type of the parameter. </p>
015 *
016 * <p>关于监听器的实现，参阅：{@link Listener} <br>
017 * For the way to implement a listener, see: {@link cn.nukkit.event.Listener}</p>
018 *
019 * @author Unknown(code) @ Nukkit Project
020 * @author 粉鞋大妈(javadoc) @ Nukkit Project
021 * @see cn.nukkit.event.EventHandler
022 * @since Nukkit 1.0 | Nukkit API 1.0.0
023 */
024public abstract class Event {
025
026    protected String eventName = null;
027    private boolean isCancelled = false;
028
029    final public String getEventName() {
030        return eventName == null ? getClass().getName() : eventName;
031    }
032
033    public boolean isCancelled() {
034        if (!(this instanceof Cancellable)) {
035            throw new EventException("Event is not Cancellable");
036        }
037        return isCancelled;
038    }
039
040    public void setCancelled() {
041        setCancelled(true);
042    }
043
044    public void setCancelled(boolean value) {
045        if (!(this instanceof Cancellable)) {
046            throw new EventException("Event is not Cancellable");
047        }
048        isCancelled = value;
049    }
050}
