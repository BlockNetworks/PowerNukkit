001package cn.nukkit.block;
002
003import cn.nukkit.Player;
004import cn.nukkit.item.Item;
005import cn.nukkit.item.ItemSeedsWheat;
006import cn.nukkit.level.Level;
007import cn.nukkit.level.particle.BoneMealParticle;
008import cn.nukkit.math.BlockFace;
009import cn.nukkit.utils.BlockColor;
010
011import java.util.concurrent.ThreadLocalRandom;
012
013/**
014 * Created on 2015/11/23 by xtypr.
015 * Package cn.nukkit.block in project Nukkit .
016 */
017public class BlockDoublePlant extends BlockFlowable {
018    public static final int SUNFLOWER = 0;
019    public static final int LILAC = 1;
020    public static final int TALL_GRASS = 2;
021    public static final int LARGE_FERN = 3;
022    public static final int ROSE_BUSH = 4;
023    public static final int PEONY = 5;
024    public static final int TOP_HALF_BITMASK = 0x8;
025
026    private static final String[] NAMES = new String[]{
027            "Sunflower",
028            "Lilac",
029            "Double Tallgrass",
030            "Large Fern",
031            "Rose Bush",
032            "Peony"
033    };
034
035    public BlockDoublePlant() {
036        this(0);
037    }
038
039    public BlockDoublePlant(int meta) {
040        super(meta);
041    }
042
043    @Override
044    public int getId() {
045        return DOUBLE_PLANT;
046    }
047
048    @Override
049    public boolean canBeReplaced() {
050        return this.getDamage() == TALL_GRASS || this.getDamage() == LARGE_FERN;
051    }
052
053    @Override
054    public String getName() {
055        return NAMES[this.getDamage() > 5 ? 0 : this.getDamage()];
056    }
057
058    @Override
059    public int onUpdate(int type) {
060        if (type == Level.BLOCK_UPDATE_NORMAL) {
061            if ((this.getDamage() & TOP_HALF_BITMASK) == TOP_HALF_BITMASK) {
062                // Top
063                if (!(this.down().getId() == DOUBLE_PLANT)) {
064                    this.getLevel().setBlock(this, new BlockAir(), false, true);
065                    return Level.BLOCK_UPDATE_NORMAL;
066                }
067            } else {
068                // Bottom
069                if (this.down().isTransparent() || !(this.up().getId() == DOUBLE_PLANT)) {
070                    this.getLevel().useBreakOn(this);
071                    return Level.BLOCK_UPDATE_NORMAL;
072                }
073            }
074        }
075        return 0;
076    }
077
078    @Override
079    public boolean place(Item item, Block block, Block target, BlockFace face, double fx, double fy, double fz, Player player) {
080        Block down = down();
081        Block up = up();
082
083        if (up.getId() == AIR && (down.getId() == GRASS || down.getId() == DIRT)) {
084            this.getLevel().setBlock(block, this, true, false); // If we update the bottom half, it will drop the item because there isn't a flower block above
085            this.getLevel().setBlock(up, new BlockDoublePlant(getDamage() ^ TOP_HALF_BITMASK), true, true);
086            return true;
087        }
088
089        return false;
090    }
091
092    @Override
093    public boolean onBreak(Item item) {
094        Block down = down();
095
096        if ((this.getDamage() & TOP_HALF_BITMASK) == TOP_HALF_BITMASK) { // Top half
097            this.getLevel().useBreakOn(down);
098        } else {
099            this.getLevel().setBlock(this, new BlockAir(), true, true);
100        }
101
102        return true;
103    }
104
105    @Override
106    public Item[] getDrops(Item item) {
107        if ((this.getDamage() & TOP_HALF_BITMASK) != TOP_HALF_BITMASK) {
108            switch (this.getDamage() & 0x07) {
109                case TALL_GRASS:
110                case LARGE_FERN:
111                    boolean dropSeeds = ThreadLocalRandom.current().nextInt(10) == 0;
112                    if (item.isShears()) {
113                        //todo enchantment
114                        if (dropSeeds) {
115                            return new Item[]{
116                                    new ItemSeedsWheat(0, 1),
117                                    toItem()
118                            };
119                        } else {
120                            return new Item[]{
121                                    toItem()
122                            };
123                        }
124                    }
125
126                    if (dropSeeds) {
127                        return new Item[]{
128                                new ItemSeedsWheat()
129                        };
130                    } else {
131                        return new Item[0];
132                    }
133            }
134
135            return new Item[]{toItem()};
136        }
137
138        return new Item[0];
139    }
140
141    @Override
142    public BlockColor getColor() {
143        return BlockColor.FOLIAGE_BLOCK_COLOR;
144    }
145
146    @Override
147    public boolean canBeActivated() {
148        return true;
149    }
150
151    @Override
152    public boolean onActivate(Item item, Player player) {
153        if (item.getId() == Item.DYE && item.getDamage() == 0x0f) { //Bone meal
154            switch (this.getDamage() & 0x07) {
155                case SUNFLOWER:
156                case LILAC:
157                case ROSE_BUSH:
158                case PEONY:
159                    if (player != null && (player.gamemode & 0x01) == 0) {
160                        item.count--;
161                    }
162                    this.level.addParticle(new BoneMealParticle(this));
163                    this.level.dropItem(this, this.toItem());
164            }
165
166            return true;
167        }
168
169        return false;
170    }
171}
