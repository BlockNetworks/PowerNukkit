001package cn.nukkit.block;
002
003import cn.nukkit.item.Item;
004import cn.nukkit.item.ItemBlock;
005import cn.nukkit.item.ItemTool;
006import cn.nukkit.level.Level;
007import cn.nukkit.math.Vector3;
008import cn.nukkit.utils.BlockColor;
009
010/**
011 * Created on 2015/12/2 by xtypr.
012 * Package cn.nukkit.block in project Nukkit .
013 */
014public class BlockFarmland extends BlockTransparentMeta {
015
016    public BlockFarmland() {
017        this(0);
018    }
019
020    public BlockFarmland(int meta) {
021        super(meta);
022    }
023
024    @Override
025    public String getName() {
026        return "Farmland";
027    }
028
029    @Override
030    public int getId() {
031        return FARMLAND;
032    }
033
034    @Override
035    public double getResistance() {
036        return 3;
037    }
038
039    @Override
040    public double getHardness() {
041        return 0.6;
042    }
043
044    @Override
045    public int getToolType() {
046        return ItemTool.TYPE_SHOVEL;
047    }
048
049    @Override
050    public double getMaxY() {
051        return this.y + 0.9375;
052    }
053
054    @Override
055    public int onUpdate(int type) {
056        if (type == Level.BLOCK_UPDATE_RANDOM) {
057            Vector3 v = new Vector3();
058
059            if (this.level.getBlock(v.setComponents(x, this.y + 1, z)) instanceof BlockCrops) {
060                return 0;
061            }
062
063            if (this.level.getBlock(v.setComponents(x, this.y + 1, z)).isSolid()) {
064                this.level.setBlock(this, new BlockDirt(), false, true);
065
066                return Level.BLOCK_UPDATE_RANDOM;
067            }
068
069            boolean found = false;
070
071            if (this.level.isRaining()) {
072                found = true;
073            } else {
074                for (int x = (int) this.x - 4; x <= this.x + 4; x++) {
075                    for (int z = (int) this.z - 4; z <= this.z + 4; z++) {
076                        for (int y = (int) this.y; y <= this.y + 1; y++) {
077                            if (z == this.z && x == this.x && y == this.y) {
078                                continue;
079                            }
080
081                            v.setComponents(x, y, z);
082                            int block = this.level.getBlockIdAt(v.getFloorX(), v.getFloorY(), v.getFloorZ());
083
084                            if (block == WATER || block == STILL_WATER || block == ICE_FROSTED) {
085                                found = true;
086                                break;
087                            } else {
088                                block = this.level.getBlockIdAt(v.getFloorX(), v.getFloorY(), v.getFloorZ(), 1);
089                                if (block == WATER || block == STILL_WATER || block == ICE_FROSTED) {
090                                    found = true;
091                                    break;
092                                }
093                            }
094                        }
095                    }
096                }
097            }
098
099            Block block = this.level.getBlock(v.setComponents(x, y - 1, z));
100            if (found || block instanceof BlockWater || block instanceof BlockIceFrosted) {
101                if (this.getDamage() < 7) {
102                    this.setDamage(7);
103                    this.level.setBlock(this, this, false, false);
104                }
105                return Level.BLOCK_UPDATE_RANDOM;
106            }
107
108            if (this.getDamage() > 0) {
109                this.setDamage(this.getDamage() - 1);
110                this.level.setBlock(this, this, false, false);
111            } else {
112                this.level.setBlock(this, Block.get(Block.DIRT), false, true);
113            }
114
115            return Level.BLOCK_UPDATE_RANDOM;
116        }
117
118        return 0;
119    }
120
121    @Override
122    public Item toItem() {
123        return new ItemBlock(new BlockDirt());
124    }
125
126    @Override
127    public BlockColor getColor() {
128        return BlockColor.DIRT_BLOCK_COLOR;
129    }
130}
