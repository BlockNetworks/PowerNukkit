006package cn.nukkit.block;
007
008import cn.nukkit.Player;
009import cn.nukkit.blockentity.BlockEntity;
010import cn.nukkit.blockentity.BlockEntityShulkerBox;
011import cn.nukkit.inventory.ContainerInventory;
012import cn.nukkit.inventory.InventoryHolder;
013import cn.nukkit.inventory.ShulkerBoxInventory;
014import cn.nukkit.item.Item;
015import cn.nukkit.item.ItemBlock;
016import cn.nukkit.item.ItemTool;
017import cn.nukkit.math.BlockFace;
018import cn.nukkit.nbt.NBTIO;
019import cn.nukkit.nbt.tag.CompoundTag;
020import cn.nukkit.nbt.tag.ListTag;
021import cn.nukkit.utils.BlockColor;
022
023/**
024 *
025 * @author Reece Mackie
026 */
027public class BlockUndyedShulkerBox extends BlockTransparent {
028
029    public BlockUndyedShulkerBox() {
030        super();
031    }
032
033    @Override
034    public int getId() {
035        return UNDYED_SHULKER_BOX;
036    }
037
038    @Override
039    public String getName() {
040        return "Shulker Box";
041    }
042
043    @Override
044    public double getHardness() {
045        return 2;
046    }
047
048    @Override
049    public double getResistance() {
050        return 10;
051    }
052
053    @Override
054    public boolean canBeActivated() {
055        return true;
056    }
057
058    @Override
059    public int getToolType() {
060        return ItemTool.TYPE_PICKAXE;
061    }
062
063    @Override
064    public int getWaterloggingLevel() {
065        return 1;
066    }
067
068    @Override
069    public Item toItem() {
070        ItemBlock item = new ItemBlock(this, this.getDamage(), 1);
071
072        BlockEntityShulkerBox t = (BlockEntityShulkerBox) this.getLevel().getBlockEntity(this);
073
074        if (t != null) {
075            ShulkerBoxInventory i = t.getRealInventory();
076
077            if (!i.isEmpty()) {
078                CompoundTag nbt = item.getNamedTag();
079                if (nbt == null)
080                    nbt = new CompoundTag("");
081
082                ListTag<CompoundTag> items = new ListTag<>();
083
084                for (int it = 0; it < i.getSize(); it++) {
085                    if (i.getItem(it).getId() != Item.AIR) {
086                        CompoundTag d = NBTIO.putItemHelper(i.getItem(it), it);
087                        items.add(d);
088                    }
089                }
090
091                nbt.put("Items", items);
092
093                item.setCompoundTag(nbt);
094            }
095
096            if (t.hasName()) {
097                item.setCustomName(t.getName());
098            }
099        }
100
101        return item;
102    }
103
104    @Override
105    public boolean place(Item item, Block block, Block target, BlockFace face, double fx, double fy, double fz, Player player) {
106        this.getLevel().setBlock(block, this, true);
107        CompoundTag nbt = BlockEntity.getDefaultCompound(this, BlockEntity.SHULKER_BOX)
108                .putByte("facing", face.getIndex());
109
110        if (item.hasCustomName()) {
111            nbt.putString("CustomName", item.getCustomName());
112        }
113
114        CompoundTag t = item.getNamedTag();
115
116        if (t != null) {
117            if (t.contains("Items")) {
118                nbt.putList(t.getList("Items"));
119            }
120        }
121
122        BlockEntityShulkerBox box = (BlockEntityShulkerBox) BlockEntity.createBlockEntity(BlockEntity.SHULKER_BOX, this.getLevel().getChunk(this.getFloorX() >> 4, this.getFloorZ() >> 4), nbt);
123        return box != null;
124    }
125
126    @Override
127    public boolean canHarvestWithHand() {
128        return false;
129    }
130
131    @Override
132    public boolean onActivate(Item item, Player player) {
133        if (player != null) {
134            BlockEntity t = this.getLevel().getBlockEntity(this);
135            BlockEntityShulkerBox box;
136            if (t instanceof BlockEntityShulkerBox) {
137                box = (BlockEntityShulkerBox) t;
138            } else {
139                CompoundTag nbt = BlockEntity.getDefaultCompound(this, BlockEntity.SHULKER_BOX);
140                box = (BlockEntityShulkerBox) BlockEntity.createBlockEntity(BlockEntity.SHULKER_BOX, this.getLevel().getChunk(this.getFloorX() >> 4, this.getFloorZ() >> 4), nbt);
141                if (box == null) {
142                    return false;
143                }
144            }
145
146            Block block = this.getSide(BlockFace.fromIndex(box.namedTag.getByte("facing")));
147            if (!(block instanceof BlockAir) && !(block instanceof BlockLiquid) && !(block instanceof BlockFlowable)) {
148                return true;
149            }
150
151            player.addWindow(box.getInventory());
152        }
153
154        return true;
155    }
156
157    @Override
158    public boolean hasComparatorInputOverride() {
159        return true;
160    }
161
162    @Override
163    public int getComparatorInputOverride() {
164        BlockEntity be = this.getLevel().getBlockEntity(this);
165
166        if (!(be instanceof InventoryHolder)) {
167            return 0;
168        }
169
170        return ContainerInventory.calculateRedstone(((InventoryHolder) be).getInventory());
171    }
172
173    @Override
174    public BlockColor getColor() {
175        return BlockColor.PURPLE_BLOCK_COLOR;
176    }
177
178    @Override
179    public boolean breaksWhenMoved() {
180        return true;
181    }
182
183    @Override
184    public boolean sticksToPiston() {
185        return false;
186    }
187}
