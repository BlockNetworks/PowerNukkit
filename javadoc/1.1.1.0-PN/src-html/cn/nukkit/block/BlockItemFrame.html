001package cn.nukkit.block;
002
003import cn.nukkit.Player;
004import cn.nukkit.blockentity.BlockEntity;
005import cn.nukkit.blockentity.BlockEntityItemFrame;
006import cn.nukkit.item.Item;
007import cn.nukkit.item.ItemItemFrame;
008import cn.nukkit.level.Level;
009import cn.nukkit.level.Sound;
010import cn.nukkit.math.BlockFace;
011import cn.nukkit.nbt.tag.CompoundTag;
012import cn.nukkit.nbt.tag.Tag;
013
014import java.util.Random;
015
016/**
017 * Created by Pub4Game on 03.07.2016.
018 */
019public class BlockItemFrame extends BlockTransparentMeta {
020
021    public BlockItemFrame() {
022        this(0);
023    }
024
025    public BlockItemFrame(int meta) {
026        super(meta);
027    }
028
029    @Override
030    public int getId() {
031        return ITEM_FRAME_BLOCK;
032    }
033
034    @Override
035    public String getName() {
036        return "Item Frame";
037    }
038
039    @Override
040    public int onUpdate(int type) {
041        if (type == Level.BLOCK_UPDATE_NORMAL) {
042            if (this.getSideAtLayer(0, getFacing()).isTransparent()) {
043                this.level.useBreakOn(this);
044                return type;
045            }
046        }
047
048        return 0;
049    }
050
051    @Override
052    public boolean canBeActivated() {
053        return true;
054    }
055
056    @Override
057    public int getWaterloggingLevel() {
058        return 1;
059    }
060
061    @Override
062    public boolean onActivate(Item item, Player player) {
063        BlockEntity blockEntity = this.getLevel().getBlockEntity(this);
064        BlockEntityItemFrame itemFrame = (BlockEntityItemFrame) blockEntity;
065        if (itemFrame == null) {
066            itemFrame = (BlockEntityItemFrame) BlockEntity.createBlockEntity(BlockEntity.ITEM_FRAME, this,
067                    BlockEntity.getDefaultCompound(this, BlockEntity.ITEM_FRAME)
068                            .putByte("ItemRotation", 0)
069                            .putFloat("ItemDropChance", 1.0f));
070        }
071        if (itemFrame == null) {
072            return false;
073        }
074        if (itemFrame.getItem().getId() == Item.AIR) {
075                Item itemOnFrame = item.clone();
076                if (player != null && player.isSurvival()) {
077                        itemOnFrame.setCount(itemOnFrame.getCount() - 1);
078                player.getInventory().setItemInHand(itemOnFrame);
079                }
080            itemOnFrame.setCount(1);
081            itemFrame.setItem(itemOnFrame);
082            this.getLevel().addSound(this, Sound.BLOCK_ITEMFRAME_ADD_ITEM);
083        } else {
084            itemFrame.setItemRotation((itemFrame.getItemRotation() + 1) % 8);
085            this.getLevel().addSound(this, Sound.BLOCK_ITEMFRAME_ROTATE_ITEM);
086        }
087        return true;
088    }
089
090    @Override
091    public boolean place(Item item, Block block, Block target, BlockFace face, double fx, double fy, double fz, Player player) {
092        if (!target.isTransparent() && face.getIndex() > 1 && !block.isSolid()) {
093            switch (face) {
094                case NORTH:
095                    this.setDamage(3);
096                    break;
097                case SOUTH:
098                    this.setDamage(2);
099                    break;
100                case WEST:
101                    this.setDamage(1);
102                    break;
103                case EAST:
104                    this.setDamage(0);
105                    break;
106                default:
107                    return false;
108            }
109            this.getLevel().setBlock(block, this, true, true);
110            CompoundTag nbt = new CompoundTag()
111                    .putString("id", BlockEntity.ITEM_FRAME)
112                    .putInt("x", (int) block.x)
113                    .putInt("y", (int) block.y)
114                    .putInt("z", (int) block.z)
115                    .putByte("ItemRotation", 0)
116                    .putFloat("ItemDropChance", 1.0f);
117            if (item.hasCustomBlockData()) {
118                for (Tag aTag : item.getCustomBlockData().getAllTags()) {
119                    nbt.put(aTag.getName(), aTag);
120                }
121            }
122            BlockEntityItemFrame frame = (BlockEntityItemFrame) BlockEntity.createBlockEntity(BlockEntity.ITEM_FRAME, this.getLevel().getChunk((int) this.x >> 4, (int) this.z >> 4), nbt);
123            if (frame == null) {
124                return false;
125            }
126            this.getLevel().addSound(this, Sound.BLOCK_ITEMFRAME_PLACE);
127            return true;
128        }
129        return false;
130    }
131
132    @Override
133    public boolean onBreak(Item item) {
134        this.getLevel().setBlock(this, layer, new BlockAir(), true, true);
135        this.getLevel().addSound(this, Sound.BLOCK_ITEMFRAME_REMOVE_ITEM);
136        return true;
137    }
138
139    @Override
140    public Item[] getDrops(Item item) {
141        BlockEntity blockEntity = this.getLevel().getBlockEntity(this);
142        BlockEntityItemFrame itemFrame = (BlockEntityItemFrame) blockEntity;
143        int chance = new Random().nextInt(100) + 1;
144        if (itemFrame != null && chance <= (itemFrame.getItemDropChance() * 100)) {
145            return new Item[]{
146                    toItem(), itemFrame.getItem().clone()
147            };
148        } else {
149            return new Item[]{
150                    toItem()
151            };
152        }
153    }
154
155    @Override
156    public Item toItem() {
157        return new ItemItemFrame();
158    }
159
160    @Override
161    public boolean canPassThrough() {
162        return true;
163    }
164
165    @Override
166    public boolean hasComparatorInputOverride() {
167        return true;
168    }
169
170    @Override
171    public int getComparatorInputOverride() {
172        BlockEntity blockEntity = this.level.getBlockEntity(this);
173
174        if (blockEntity instanceof BlockEntityItemFrame) {
175            return ((BlockEntityItemFrame) blockEntity).getAnalogOutput();
176        }
177
178        return super.getComparatorInputOverride();
179    }
180
181    public BlockFace getFacing() {
182        switch (this.getDamage() & 3) {
183            case 0:
184                return BlockFace.WEST;
185            case 1:
186                return BlockFace.EAST;
187            case 2:
188                return BlockFace.NORTH;
189            case 3:
190                return BlockFace.SOUTH;
191        }
192
193        return null;
194    }
195
196    @Override
197    public double getHardness() {
198        return 0.25;
199    }
200
201    @Override
202    public boolean breaksWhenMoved() {
203        return true;
204    }
205
206    @Override
207    public boolean sticksToPiston() {
208        return false;
209    }
210}
