001package cn.nukkit.block;
002
003import cn.nukkit.Player;
004import cn.nukkit.inventory.AnvilInventory;
005import cn.nukkit.item.Item;
006import cn.nukkit.item.ItemBlock;
007import cn.nukkit.item.ItemTool;
008import cn.nukkit.level.Sound;
009import cn.nukkit.math.BlockFace;
010import cn.nukkit.utils.BlockColor;
011import cn.nukkit.utils.Faceable;
012
013/**
014 * Created by Pub4Game on 27.12.2015.
015 */
016public class BlockAnvil extends BlockFallable implements Faceable {
017
018    private static final String[] NAMES = new String[]{
019            "Anvil",
020            "Anvil",
021            "Anvil",
022            "Anvil",
023            "Slighty Damaged Anvil",
024            "Slighty Damaged Anvil",
025            "Slighty Damaged Anvil",
026            "Slighty Damaged Anvil",
027            "Very Damaged Anvil",
028            "Very Damaged Anvil",
029            "Very Damaged Anvil",
030            "Very Damaged Anvil"
031    };
032
033    private int meta;
034
035    public BlockAnvil() {
036        this(0);
037    }
038
039    public BlockAnvil(int meta) {
040        this.meta = meta;
041    }
042
043    @Override
044    public int getFullId() {
045        return (getId() << DATA_BITS) + getDamage();
046    }
047
048    @Override
049    public final int getDamage() {
050        return this.meta;
051    }
052
053    @Override
054    public final void setDamage(int meta) {
055        this.meta = meta;
056    }
057
058    @Override
059    public int getId() {
060        return ANVIL;
061    }
062
063    @Override
064    public boolean canBeActivated() {
065        return true;
066    }
067
068    @Override
069    public int getWaterloggingLevel() {
070        return 1;
071    }
072
073    @Override
074    public boolean isTransparent() {
075        return true;
076    }
077
078    @Override
079    public double getHardness() {
080        return 5;
081    }
082
083    @Override
084    public double getResistance() {
085        return 6000;
086    }
087
088    @Override
089    public int getToolType() {
090        return ItemTool.TYPE_PICKAXE;
091    }
092
093    @Override
094    public String getName() {
095        return NAMES[this.getDamage() > 11 ? 0 : this.getDamage()];
096    }
097
098    @Override
099    public boolean place(Item item, Block block, Block target, BlockFace face, double fx, double fy, double fz, Player player) {
100        if (!target.isTransparent() || target.getId() == Block.SNOW_LAYER) {
101            int damage = this.getDamage();
102            int[] faces = {1, 2, 3, 0};
103            this.setDamage(faces[player != null ? player.getDirection().getHorizontalIndex() : 0]);
104            if (damage >= 4 && damage <= 7) {
105                this.setDamage(this.getDamage() | 0x04);
106            } else if (damage >= 8 && damage <= 11) {
107                this.setDamage(this.getDamage() | 0x08);
108            }
109            this.getLevel().setBlock(block, this, true);
110            this.getLevel().addSound(this, Sound.RANDOM_ANVIL_LAND, 1, 0.8F);
111            return true;
112        }
113        return false;
114    }
115
116    @Override
117    public boolean onActivate(Item item, Player player) {
118        if (player != null) {
119            player.addWindow(new AnvilInventory(player.getUIInventory(), this), Player.ANVIL_WINDOW_ID);
120        }
121        return true;
122    }
123
124    @Override
125    public Item toItem() {
126        int damage = this.getDamage();
127        if (damage >= 4 && damage <= 7) {
128            return new ItemBlock(this, this.getDamage() & 0x04);
129        } else if (damage >= 8 && damage <= 11) {
130            return new ItemBlock(this, this.getDamage() & 0x08);
131        } else {
132            return new ItemBlock(this);
133        }
134    }
135
136    @Override
137    public Item[] getDrops(Item item) {
138        if (item.isPickaxe() && item.getTier() >= ItemTool.TIER_WOODEN) {
139            return new Item[]{
140                    this.toItem()
141            };
142        }
143        return new Item[0];
144    }
145
146    @Override
147    public BlockColor getColor() {
148        return BlockColor.IRON_BLOCK_COLOR;
149    }
150
151    @Override
152    public boolean canHarvestWithHand() {
153        return false;
154    }
155
156    @Override
157    public BlockFace getBlockFace() {
158        return BlockFace.fromHorizontalIndex(this.getDamage() & 0x7);
159    }
160}
