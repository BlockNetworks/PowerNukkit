001package cn.nukkit.block;
002
003import cn.nukkit.Player;
004import cn.nukkit.item.Item;
005import cn.nukkit.item.ItemBlock;
006import cn.nukkit.item.ItemTool;
007import cn.nukkit.math.BlockFace;
008import cn.nukkit.utils.BlockColor;
009
010public class BlockPurpur extends BlockSolidMeta {
011
012    public static final int PURPUR_NORMAL = 0;
013    public static final int PURPUR_PILLAR = 2;
014
015    public BlockPurpur() {
016        this(0);
017    }
018
019    public BlockPurpur(int meta) {
020        super(meta);
021    }
022
023    @Override
024    public String getName() {
025        String[] names = new String[]{
026                "Purpur Block",
027                "",
028                "Purpur Pillar",
029                ""
030        };
031
032        return names[this.getDamage() & 0x03];
033    }
034
035    @Override
036    public int getId() {
037        return PURPUR_BLOCK;
038    }
039
040    @Override
041    public double getHardness() {
042        return 1.5;
043    }
044
045    @Override
046    public double getResistance() {
047        return 30;
048    }
049
050    @Override
051    public int getToolType() {
052        return ItemTool.TYPE_PICKAXE;
053    }
054
055    @Override
056    public boolean place(Item item, Block block, Block target, BlockFace face, double fx, double fy, double fz, Player player) {
057        if (this.getDamage() != PURPUR_NORMAL) {
058            short[] faces = new short[]{
059                    0,
060                    0,
061                    0b1000,
062                    0b1000,
063                    0b0100,
064                    0b0100
065            };
066
067            this.setDamage(((this.getDamage() & 0x03) | faces[face.getIndex()]));
068        }
069        this.getLevel().setBlock(block, this, true, true);
070
071        return true;
072    }
073
074    @Override
075    public Item[] getDrops(Item item) {
076        if (item.isPickaxe() && item.getTier() >= ItemTool.TIER_WOODEN) {
077            return new Item[]{
078                    toItem()
079            };
080        } else {
081            return new Item[0];
082        }
083    }
084
085    @Override
086    public Item toItem() {
087        return new ItemBlock(new BlockPurpur(), this.getDamage() & 0x03, 1);
088    }
089
090    @Override
091    public BlockColor getColor() {
092        return BlockColor.MAGENTA_BLOCK_COLOR;
093    }
094}
