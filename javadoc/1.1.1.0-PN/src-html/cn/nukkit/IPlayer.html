001package cn.nukkit;
002
003import cn.nukkit.metadata.Metadatable;
004import cn.nukkit.permission.ServerOperator;
005
006import java.util.UUID;
007
008/**
009 * 用来描述一个玩家和获得这个玩家相应信息的接口。<br>
010 * An interface to describe a player and get its information.
011 *
012 * <p>这个玩家可以在线，也可以是不在线。<br>
013 * This player can be online or offline.</p>
014 *
015 * @author MagicDroidX(code) @ Nukkit Project
016 * @author 粉鞋大妈(javadoc) @ Nukkit Project
017 * @see cn.nukkit.Player
018 * @see cn.nukkit.OfflinePlayer
019 * @since Nukkit 1.0 | Nukkit API 1.0.0
020 */
021public interface IPlayer extends ServerOperator, Metadatable {
022
023    /**
024     * 返回这个玩家是否在线。<br>
025     * Returns if this player is online.
026     *
027     * @return 这个玩家是否在线。<br>If this player is online.
028     * @since Nukkit 1.0 | Nukkit API 1.0.0
029     */
030    boolean isOnline();
031
032    /**
033     * 返回这个玩家的名称。<br>
034     * Returns the name of this player.
035     *
036     * <p>如果是在线的玩家，这个函数只会返回登录名字。如果要返回显示的名字，参见{@link cn.nukkit.Player#getDisplayName}<br>
037     * Notice that this will only return its login name. If you need its display name, turn to
038     * {@link cn.nukkit.Player#getDisplayName}</p>
039     *
040     * @return 这个玩家的名称。<br>The name of this player.
041     * @since Nukkit 1.0 | Nukkit API 1.0.0
042     */
043    String getName();
044
045    UUID getUniqueId();
046
047    /**
048     * 返回这个玩家是否被封禁(ban)。<br>
049     * Returns if this player is banned.
050     *
051     * @return 这个玩家的名称。<br>The name of this player.
052     * @see #setBanned
053     * @since Nukkit 1.0 | Nukkit API 1.0.0
054     */
055    boolean isBanned();
056
057    /**
058     * 设置这个玩家是否被封禁(ban)。<br>
059     * Sets this player to be banned or to be pardoned.
060     *
061     * @param value 如果为{@code true}，封禁这个玩家。如果为{@code false}，解封这个玩家。<br>
062     *              {@code true} for ban and {@code false} for pardon.
063     * @see #isBanned
064     * @since Nukkit 1.0 | Nukkit API 1.0.0
065     */
066    void setBanned(boolean value);
067
068    /**
069     * 返回这个玩家是否已加入白名单。<br>
070     * Returns if this player is pardoned by whitelist.
071     *
072     * @return 这个玩家是否已加入白名单。<br>If this player is pardoned by whitelist.
073     * @see cn.nukkit.Server#isWhitelisted
074     * @since Nukkit 1.0 | Nukkit API 1.0.0
075     */
076    boolean isWhitelisted();
077
078    /**
079     * 把这个玩家加入白名单，或者取消这个玩家的白名单。<br>
080     * Adds this player to the white list, or removes it from the whitelist.
081     *
082     * @param value 如果为{@code true}，把玩家加入白名单。如果为{@code false}，取消这个玩家的白名单。<br>
083     *              {@code true} for add and {@code false} for remove.
084     * @see #isWhitelisted
085     * @see cn.nukkit.Server#addWhitelist
086     * @see cn.nukkit.Server#removeWhitelist
087     * @since Nukkit 1.0 | Nukkit API 1.0.0
088     */
089    void setWhitelisted(boolean value);
090
091    /**
092     * 得到这个接口的{@code Player}对象。<br>
093     * Returns a {@code Player} object for this interface.
094     *
095     * @return 这个接口的 {@code Player}对象。<br>a {@code Player} object for this interface.
096     * @see cn.nukkit.Server#getPlayerExact
097     * @since Nukkit 1.0 | Nukkit API 1.0.0
098     */
099    Player getPlayer();
100
101    /**
102     * 返回玩家所在的服务器。<br>
103     * Returns the server carrying this player.
104     *
105     * @return 玩家所在的服务器。<br>the server carrying this player.
106     * @since Nukkit 1.0 | Nukkit API 1.0.0
107     */
108    Server getServer();
109
110    /**
111     * 得到这个玩家第一次游戏的时间。<br>
112     * Returns the time this player first played in this server.
113     *
114     * @return 这个玩家第一次游戏的时间。<br>The time this player first played in this server.
115     * @since Nukkit 1.0 | Nukkit API 1.0.0
116     */
117    Long getFirstPlayed();
118
119    /**
120     * 得到这个玩家上次加入游戏的时间。<br>
121     * Returns the time this player last joined in this server.
122     *
123     * @return 这个玩家上次游戏的时间。<br>The time this player last joined in this server.
124     * @since Nukkit 1.0 | Nukkit API 1.0.0
125     */
126    Long getLastPlayed();
127
128    /**
129     * 返回这个玩家以前是否来过服务器。<br>
130     * Returns if this player has played in this server before.
131     *
132     * <p>如果想得到这个玩家是不是第一次玩，可以使用：<br>
133     * If you want to know if this player is the first time playing in this server, you can use:<br>
134     * </p>
135     * <pre>if(!player.hasPlayerBefore()) {...}</pre>
136     *
137     * @return 这个玩家以前是不是玩过游戏。<br>If this player has played in this server before.
138     * @since Nukkit 1.0 | Nukkit API 1.0.0
139     */
140    boolean hasPlayedBefore();
141
142}
