001package cn.nukkit.level.biome.impl.iceplains;
002
003import cn.nukkit.block.Block;
004import cn.nukkit.level.ChunkManager;
005import cn.nukkit.level.format.FullChunk;
006import cn.nukkit.level.generator.populator.type.Populator;
007import cn.nukkit.math.NukkitRandom;
008
009/**
010 * author: DaPorkchop_
011 * Nukkit Project
012 */
013public class IcePlainsSpikesBiome extends IcePlainsBiome {
014
015    public IcePlainsSpikesBiome() {
016        super();
017
018        PopulatorIceSpikes iceSpikes = new PopulatorIceSpikes();
019        this.addPopulator(iceSpikes);
020    }
021
022    @Override
023    public int getSurfaceId(int x, int y, int z) {
024        return SNOW_BLOCK << Block.DATA_BITS;
025    }
026
027    public String getName() {
028        return "Ice Plains Spikes";
029    }
030
031    @Override
032    public boolean isFreezing() {
033        return true;
034    }
035
036    /**
037     * @author DaPorkchop_
038     * <p>
039     * Please excuse this mess, but it runs way faster than the correct method
040     */
041    private static class PopulatorIceSpikes extends Populator {
042
043        @Override
044        public void populate(ChunkManager level, int chunkX, int chunkZ, NukkitRandom random, FullChunk chunk) {
045            for (int i = 0; i < 8; i++) {
046                int x = (chunkX << 4) + random.nextBoundedInt(16);
047                int z = (chunkZ << 4) + random.nextBoundedInt(16);
048                boolean isTall = random.nextBoundedInt(16) == 0;
049                int height = 10 + random.nextBoundedInt(16) + (isTall ? random.nextBoundedInt(31) : 0);
050                int startY = getHighestWorkableBlock(x, z, chunk);
051                int maxY = startY + height;
052                if (isTall) {
053                    for (int y = startY; y < maxY; y++) {
054                        //center column
055                        level.setBlockAt(x, y, z, PACKED_ICE);
056                        //t shape
057                        level.setBlockAt(x + 1, y, z, PACKED_ICE);
058                        level.setBlockAt(x - 1, y, z, PACKED_ICE);
059                        level.setBlockAt(x, y, z + 1, PACKED_ICE);
060                        level.setBlockAt(x, y, z - 1, PACKED_ICE);
061                        //additional blocks on the side
062                        if (random.nextBoolean()) {
063                            level.setBlockAt(x + 1, y, z + 1, PACKED_ICE);
064                        }
065                        if (random.nextBoolean()) {
066                            level.setBlockAt(x + 1, y, z - 1, PACKED_ICE);
067                        }
068                        if (random.nextBoolean()) {
069                            level.setBlockAt(x - 1, y, z + 1, PACKED_ICE);
070                        }
071                        if (random.nextBoolean()) {
072                            level.setBlockAt(x - 1, y, z - 1, PACKED_ICE);
073                        }
074                    }
075                    //finish with a point
076                    level.setBlockAt(x + 1, maxY, z, PACKED_ICE);
077                    level.setBlockAt(x - 1, maxY, z, PACKED_ICE);
078                    level.setBlockAt(x, maxY, z + 1, PACKED_ICE);
079                    level.setBlockAt(x, maxY, z - 1, PACKED_ICE);
080                    for (int y = maxY; y < maxY + 3; maxY++) {
081                        level.setBlockAt(x, y, z, PACKED_ICE);
082                    }
083                } else {
084                    //the maximum possible radius in blocks
085                    int baseWidth = random.nextBoundedInt(1) + 4;
086                    float shrinkFactor = baseWidth / (float) height;
087                    float currWidth = baseWidth;
088                    for (int y = startY; y < maxY; y++) {
089                        for (int xx = (int) -currWidth; xx < currWidth; xx++) {
090                            for (int zz = (int) -currWidth; zz < currWidth; zz++) {
091                                int currDist = (int) Math.sqrt(xx * xx + zz * zz);
092                                if ((int) currWidth != currDist && random.nextBoolean()) {
093                                    level.setBlockAt(x + xx, y, z + zz, PACKED_ICE);
094                                }
095                            }
096                        }
097                        currWidth -= shrinkFactor;
098                    }
099                }
100            }
101        }
102
103        public int getHighestWorkableBlock(int x, int z, FullChunk chunk) {
104            return chunk.getHighestBlockAt(x & 0xF, z & 0xF) - 5;
105        }
106    }
107}
