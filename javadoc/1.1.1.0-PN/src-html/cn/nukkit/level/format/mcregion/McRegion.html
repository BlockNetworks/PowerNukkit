050    public static boolean isValid(String path) {
051        boolean isValid = (new File(path + "/level.dat").exists()) && new File(path + "/region/").isDirectory();
052        if (isValid) {
053            for (File file : new File(path + "/region/").listFiles((dir, name) -> Pattern.matches("^.+\\.mc[r|a]$", name))) {
054                if (!file.getName().endsWith(".mcr")) {
055                    isValid = false;
056                    break;
057                }
058            }
059        }
060        return isValid;
061    }

067    public static void generate(String path, String name, long seed, Class<? extends Generator> generator, Map<String, String> options) throws IOException {
068        if (!new File(path + "/region").exists()) {
069            new File(path + "/region").mkdirs();
070        }
071
072        CompoundTag levelData = new CompoundTag("Data")
073                .putCompound("GameRules", new CompoundTag())
074
075                .putLong("DayTime", 0)
076                .putInt("GameType", 0)
077                .putString("generatorName", Generator.getGeneratorName(generator))
078                .putString("generatorOptions", options.getOrDefault("preset", ""))
079                .putInt("generatorVersion", 1)
080                .putBoolean("hardcore", false)
081                .putBoolean("initialized", true)
082                .putLong("LastPlayed", System.currentTimeMillis() / 1000)
083                .putString("LevelName", name)
084                .putBoolean("raining", false)
085                .putInt("rainTime", 0)
086                .putLong("RandomSeed", seed)
087                .putInt("SpawnX", 128)
088                .putInt("SpawnY", 70)
089                .putInt("SpawnZ", 128)
090                .putBoolean("thundering", false)
091                .putInt("thunderTime", 0)
092                .putInt("version", 19133)
093                .putLong("Time", 0)
094                .putLong("SizeOnDisk", 0);
095
096        NBTIO.writeGZIPCompressed(new CompoundTag().putCompound("Data", levelData), new FileOutputStream(path + "level.dat"), ByteOrder.BIG_ENDIAN);
097    }

099    @Override
100    public AsyncTask requestChunkTask(int x, int z) throws ChunkException {
101        BaseFullChunk chunk = this.getChunk(x, z, false);
102        if (chunk == null) {
103            throw new ChunkException("Invalid Chunk Sent");
104        }
105
106        long timestamp = chunk.getChanges();
107
108        byte[] tiles = new byte[0];
109
110        if (!chunk.getBlockEntities().isEmpty()) {
111            List<CompoundTag> tagList = new ArrayList<>();
112
113            for (BlockEntity blockEntity : chunk.getBlockEntities().values()) {
114                if (blockEntity instanceof BlockEntitySpawnable) {
115                    tagList.add(((BlockEntitySpawnable) blockEntity).getSpawnCompound());
116                }
117            }
118
119            try {
120                tiles = NBTIO.write(tagList, ByteOrder.LITTLE_ENDIAN, true);
121            } catch (IOException e) {
122                throw new RuntimeException(e);
123            }
124        }
125
126        Map<Integer, Integer> extra = chunk.getBlockExtraDataArray();
127        BinaryStream extraData;
128        if (!extra.isEmpty()) {
129            extraData = new BinaryStream();
130            extraData.putLInt(extra.size());
131            for (Map.Entry<Integer, Integer> entry : extra.entrySet()) {
132                extraData.putLInt(entry.getKey());
133                extraData.putLShort(entry.getValue());
134            }
135        } else {
136            extraData = null;
137        }
138
139        BinaryStream stream = new BinaryStream();
140        stream.put(chunk.getBlockIdArray());
141        stream.put(chunk.getBlockDataArray());
142        stream.put(chunk.getBlockSkyLightArray());
143        stream.put(chunk.getBlockLightArray());
144        stream.put(chunk.getHeightMapArray());
145        stream.put(chunk.getBiomeIdArray());
146        if (extraData != null) {
147            stream.put(extraData.getBuffer());
148        } else {
149            stream.putLInt(0);
150        }
151        stream.put(tiles);
152
153        this.getLevel().chunkRequestCallback(timestamp, x, z, 16, stream.getBuffer());
154
155        return null;
156    }

158    @Override
159    public BaseFullChunk loadChunk(long index, int chunkX, int chunkZ, boolean create) {
160        int regionX = getRegionIndexX(chunkX);
161        int regionZ = getRegionIndexZ(chunkZ);
162        BaseRegionLoader region = this.loadRegion(regionX, regionZ);
163        this.level.timings.syncChunkLoadDataTimer.startTiming();
164        BaseFullChunk chunk;
165        try {
166            chunk = region.readChunk(chunkX - regionX * 32, chunkZ - regionZ * 32);
167        } catch (IOException e) {
168            throw new RuntimeException(e);
169        }
170        if (chunk == null) {
171            if (create) {
172                chunk = this.getEmptyChunk(chunkX, chunkZ);
173                putChunk(index, chunk);
174            }
175        } else {
176            putChunk(index, chunk);
177        }
178        this.level.timings.syncChunkLoadDataTimer.stopTiming();
179        return chunk;
180    }

187    @Override
188    public void saveChunk(int x, int z) {
189        if (this.isChunkLoaded(x, z)) {
190            try {
191                this.getRegion(x >> 5, z >> 5).writeChunk(this.getChunk(x, z));
192            } catch (Exception e) {
193                throw new RuntimeException(e);
194            }
195        }
196    }

198    @Override
199    public void saveChunk(int x, int z, FullChunk chunk) {
200        if (!(chunk instanceof Chunk)) {
201            throw new ChunkException("Invalid Chunk class");
202        }
203        this.loadRegion(x >> 5, z >> 5);
204        chunk.setPosition(x, z);
205        try {
206            this.getRegion(x >> 5, z >> 5).writeChunk(chunk);
207        } catch (Exception e) {
208            throw new RuntimeException(e);
209        }
210    }

216    protected BaseRegionLoader loadRegion(int x, int z) {
217        BaseRegionLoader tmp = lastRegion.get();
218        if (tmp != null && x == tmp.getX() && z == tmp.getZ()) {
219            return tmp;
220        }
221        long index = Level.chunkHash(x, z);
222        synchronized (regions) {
223            BaseRegionLoader region = this.regions.get(index);
224            if (region == null) {
225                region = new RegionLoader(this, x, z);
226                this.regions.put(index, region);
227            }
228            lastRegion.set(region);
229            return region;
230        }
231    }
