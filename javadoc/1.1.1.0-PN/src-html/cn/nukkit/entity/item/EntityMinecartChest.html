001package cn.nukkit.entity.item;
002
003import cn.nukkit.Player;
004import cn.nukkit.block.Block;
005import cn.nukkit.entity.Entity;
006import cn.nukkit.inventory.InventoryHolder;
007import cn.nukkit.inventory.MinecartChestInventory;
008import cn.nukkit.item.Item;
009import cn.nukkit.level.format.FullChunk;
010import cn.nukkit.math.Vector3;
011import cn.nukkit.nbt.NBTIO;
012import cn.nukkit.nbt.tag.CompoundTag;
013import cn.nukkit.nbt.tag.ListTag;
014import cn.nukkit.utils.MinecartType;
015
016/**
017 * Created by Snake1999 on 2016/1/30.
018 * Package cn.nukkit.entity.item in project Nukkit.
019 */
020public class EntityMinecartChest extends EntityMinecartAbstract implements InventoryHolder {
021
022    public static final int NETWORK_ID = 98;
023
024    protected MinecartChestInventory inventory;
025
026    public EntityMinecartChest(FullChunk chunk, CompoundTag nbt) {
027        super(chunk, nbt);
028        setDisplayBlock(Block.get(Block.CHEST), false);
029    }
030
031    @Override
032    public MinecartType getType() {
033        return MinecartType.valueOf(1);
034    }
035
036    @Override
037    public boolean isRideable(){
038        return false;
039    }
040
041    @Override
042    public int getNetworkId() {
043        return NETWORK_ID;
044    }
045
046    @Override
047    public void dropItem() {
048        super.dropItem();
049
050        this.level.dropItem(this, Item.get(Item.CHEST));
051        for (Item item : this.inventory.getContents().values()) {
052            this.level.dropItem(this, item);
053        }
054    }
055
056    @Override
057    public boolean mountEntity(Entity entity, byte mode) {
058        return false;
059    }
060
061    @Override
062    public boolean onInteract(Player player, Item item, Vector3 clickedPos) {
063        player.addWindow(this.inventory);
064        return true;
065    }
066
067    @Override
068    public MinecartChestInventory getInventory() {
069        return inventory;
070    }
071
072    @Override
073    public void initEntity() {
074        super.initEntity();
075
076        this.inventory = new MinecartChestInventory(this);
077        if (this.namedTag.contains("Items") && this.namedTag.get("Items") instanceof ListTag) {
078            ListTag<CompoundTag> inventoryList = this.namedTag.getList("Items", CompoundTag.class);
079            for (CompoundTag item : inventoryList.getAll()) {
080                this.inventory.setItem(item.getByte("Slot"), NBTIO.getItemHelper(item));
081            }
082        }
083
084        this.dataProperties
085                .putByte(DATA_CONTAINER_TYPE, 10)
086                .putInt(DATA_CONTAINER_BASE_SIZE, this.inventory.getSize())
087                .putInt(DATA_CONTAINER_EXTRA_SLOTS_PER_STRENGTH, 0);
088    }
089
090    @Override
091    public void saveNBT() {
092        super.saveNBT();
093
094        this.namedTag.putList(new ListTag<CompoundTag>("Items"));
095        if (this.inventory != null) {
096            for (int slot = 0; slot < 27; ++slot) {
097                Item item = this.inventory.getItem(slot);
098                if (item != null && item.getId() != Item.AIR) {
099                    this.namedTag.getList("Items", CompoundTag.class)
100                            .add(NBTIO.putItemHelper(item, slot));
101                }
102            }
103        }
104    }
105}
