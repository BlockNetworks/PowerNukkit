045    private final int size;
046    private final String title;
047    private final int typeId;
048
049    InventoryType(int defaultSize, String defaultBlockEntity, int typeId) {
050        this.size = defaultSize;
051        this.title = defaultBlockEntity;
052        this.typeId = typeId;
053    }
054
055    public int getDefaultSize() {
056        return size;
057    }
058
059    public String getDefaultTitle() {
060        return title;
061    }
062
063    public int getNetworkType() {
064        return typeId;
065    }
