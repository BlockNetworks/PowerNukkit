001package cn.nukkit.utils;
002
003import cn.nukkit.nbt.tag.CompoundTag;
004
005import java.util.EnumMap;
006import java.util.HashMap;
007import java.util.Map;
008
009public class BannerPattern {
010
011    private Type type;
012    private DyeColor color;
013
014    public BannerPattern(Type type, DyeColor color) {
015        this.type = type;
016        this.color = color;
017    }
018
019    public DyeColor getColor() {
020        return this.color;
021    }
022
023    public Type getType() {
024        return this.type;
025    }
026
027    public static BannerPattern fromCompoundTag(CompoundTag compoundTag) {
028        return new BannerPattern(Type.getByName(compoundTag.contains("Pattern") ? compoundTag.getString("Pattern") : ""), compoundTag.contains("Color") ? DyeColor.getByDyeData(compoundTag.getInt("Color")) : DyeColor.BLACK);
029    }
030
031    public enum Type {
032
033        PATTERN_BOTTOM_STRIPE("bs"),
034        PATTERN_TOP_STRIPE("ts"),
035        PATTERN_LEFT_STRIPE("ls"),
036        PATTERN_RIGHT_STRIPE("rs"),
037        PATTERN_CENTER_STRIPE("cs"),
038        PATTERN_MIDDLE_STRIPE("ms"),
039        PATTERN_DOWN_RIGHT_STRIPE("drs"),
040        PATTERN_DOWN_LEFT_STRIPE("dls"),
041        PATTERN_SMALL_STRIPES("ss"),
042        PATTERN_DIAGONAL_CROSS("cr"),
043        PATTERN_SQUARE_CROSS("sc"),
044        PATTERN_LEFT_OF_DIAGONAL("ld"),
045        PATTERN_RIGHT_OF_UPSIDE_DOWN_DIAGONAL("rud"),
046        PATTERN_LEFT_OF_UPSIDE_DOWN_DIAGONAL("lud"),
047        PATTERN_RIGHT_OF_DIAGONAL("rd"),
048        PATTERN_VERTICAL_HALF_LEFT("vh"),
049        PATTERN_VERTICAL_HALF_RIGHT("vhr"),
050        PATTERN_HORIZONTAL_HALF_TOP("hh"),
051        PATTERN_HORIZONTAL_HALF_BOTTOM("hhb"),
052        PATTERN_BOTTOM_LEFT_CORNER("bl"),
053        PATTERN_BOTTOM_RIGHT_CORNER("br"),
054        PATTERN_TOP_LEFT_CORNER("tl"),
055        PATTERN_TOP_RIGHT_CORNER("tr"),
056        PATTERN_BOTTOM_TRIANGLE("bt"),
057        PATTERN_TOP_TRIANGLE("tt"),
058        PATTERN_BOTTOM_TRIANGLE_SAWTOOTH("bts"),
059        PATTERN_TOP_TRIANGLE_SAWTOOTH("tts"),
060        PATTERN_MIDDLE_CIRCLE("mc"),
061        PATTERN_MIDDLE_RHOMBUS("mr"),
062        PATTERN_BORDER("bo"),
063        PATTERN_CURLY_BORDER("cbo"),
064        PATTERN_BRICK("bri"),
065        PATTERN_GRADIENT("gra"),
066        PATTERN_GRADIENT_UPSIDE_DOWN("gru"),
067        PATTERN_CREEPER("cre"),
068        PATTERN_SKULL("sku"),
069        PATTERN_FLOWER("flo"),
070        PATTERN_MOJANG("moj");
071
072        private final static Map<String, Type> BY_NAME = new HashMap<>();
073
074        private String name;
075
076        Type(String name) {
077            this.name = name;
078        }
079
080        public String getName() {
081            return this.name;
082        }
083
084        static {
085            for (Type type : values()) {
086                BY_NAME.put(type.getName(), type);
087            }
088        }
089
090        public static Type getByName(String name) {
091            return BY_NAME.get(name);
092        }
093
094    }
095
096}
