074    private int red;
075    private int green;
076    private int blue;
077    private int alpha;
078
079    public BlockColor(int red, int green, int blue, int alpha) {
080        this.red = red;
081        this.green = green;
082        this.blue = blue;
083        this.alpha = alpha;
084    }
085
086    public BlockColor(int red, int green, int blue) {
087        this(red, green, blue, 0xff);
088    }
089
090    public BlockColor(int rgb) {
091        this.red = (rgb >> 16) & 0xff;
092        this.green = (rgb >> 8) & 0xff;
093        this.blue = rgb & 0xff;
094        this.alpha = 0xff;
095    }
096
097    @Override
098    public boolean equals(Object obj) {
099        if (!(obj instanceof BlockColor)) {
100            return false;
101        }
102        BlockColor other = (BlockColor) obj;
103        return this.red == other.red && this.green == other.green &&
104                this.blue == other.blue && this.alpha == other.alpha;
105    }
106
107    @Override
108    public String toString() {
109        return "BlockColor[r=" + this.red + ",g=" + this.green + ",b=" + this.blue + ",a=" + this.alpha + "]";
110    }
111
112    public int getRed() {
113        return this.red;
114    }
115
116    public int getGreen() {
117        return this.green;
118    }
119
120    public int getBlue() {
121        return this.blue;
122    }
123
124    public int getAlpha() {
125        return this.alpha;
126    }
127
128    public int getRGB() {
129        return (this.red << 16 | this.green << 8 | this.blue) & 0xffffff;
130    }
