001package cn.nukkit.command.defaults;
002
003import cn.nukkit.Player;
004import cn.nukkit.command.Command;
005import cn.nukkit.command.CommandSender;
006import cn.nukkit.command.data.CommandParamType;
007import cn.nukkit.command.data.CommandParameter;
008import cn.nukkit.lang.TranslationContainer;
009import cn.nukkit.level.Level;
010
011/**
012 * author: Angelic47
013 * Nukkit Project
014 */
015public class WeatherCommand extends VanillaCommand {
016
017    private java.util.Random rand = new java.util.Random();
018
019    public WeatherCommand(String name) {
020        super(name, "%nukkit.command.weather.description", "%commands.weather.usage");
021        this.setPermission("nukkit.command.weather");
022        this.commandParameters.clear();
023        this.commandParameters.put("default", new CommandParameter[]{
024                new CommandParameter("clear|rain|thunder", CommandParamType.STRING, false),
025                new CommandParameter("duration in seconds", CommandParamType.INT, true)
026        });
027    }
028
029    @Override
030    public boolean execute(CommandSender sender, String commandLabel, String[] args) {
031        if (!this.testPermission(sender)) {
032            return true;
033        }
034        if (args.length == 0 || args.length > 2) {
035            sender.sendMessage(new TranslationContainer("commands.weather.usage", this.usageMessage));
036            return false;
037        }
038
039        String weather = args[0];
040        Level level;
041        int seconds;
042        if (args.length > 1) {
043            try {
044                seconds = Integer.parseInt(args[1]);
045            } catch (Exception e) {
046                sender.sendMessage(new TranslationContainer("commands.generic.usage", this.usageMessage));
047                return true;
048            }
049        } else {
050            seconds = 600 * 20;
051        }
052
053        if (sender instanceof Player) {
054            level = ((Player) sender).getLevel();
055        } else {
056            level = sender.getServer().getDefaultLevel();
057        }
058
059        switch (weather) {
060            case "clear":
061                level.setRaining(false);
062                level.setThundering(false);
063                level.setRainTime(seconds * 20);
064                level.setThunderTime(seconds * 20);
065                Command.broadcastCommandMessage(sender,
066                        new TranslationContainer("commands.weather.clear"));
067                return true;
068            case "rain":
069                level.setRaining(true);
070                level.setRainTime(seconds * 20);
071                Command.broadcastCommandMessage(sender,
072                        new TranslationContainer("commands.weather.rain"));
073                return true;
074            case "thunder":
075                level.setThundering(true);
076                level.setRainTime(seconds * 20);
077                level.setThunderTime(seconds * 20);
078                Command.broadcastCommandMessage(sender,
079                        new TranslationContainer("commands.weather.thunder"));
080                return true;
081            default:
082                sender.sendMessage(new TranslationContainer("commands.weather.usage", this.usageMessage));
083                return false;
084        }
085
086    }
087}
