001package cn.nukkit.entity.item;

005import cn.nukkit.entity.Entity;

013import cn.nukkit.math.BlockFace.Axis;
014import cn.nukkit.math.SimpleAxisAlignedBB;
015import cn.nukkit.math.Vector3;

023/**
024 * author: MagicDroidX
025 * Nukkit Project
026 */
027public class EntityPainting extends EntityHanging {
028
029    public static final int NETWORK_ID = 83;
030
031    public final static Motive[] motives = Motive.values();
032    private Motive motive;
033
034    private float width;
035    private float length;
036    private float height;
037
038    public EntityPainting(FullChunk chunk, CompoundTag nbt) {
039        super(chunk, nbt);
040    }
041
042    public static Motive getMotive(String name) {
043        return Motive.BY_NAME.getOrDefault(name, Motive.KEBAB);
044    }
045
046    @Override
047    public float getWidth() {
048        return width;
049    }
050
051    @Override
052    public float getLength() {
053        return length;
054    }
055
056    @Override
057    public float getHeight() {
058        return height;
059    }
060
061    @Override
062    public int getNetworkId() {
063        return NETWORK_ID;
064    }
065
066    @Override
067    protected void initEntity() {
068        this.motive = getMotive(this.namedTag.getString("Motive"));
069
070        if (this.motive != null) {
071            BlockFace face = getHorizontalFacing();
072
073            Vector3 size = new Vector3(this.motive.width, this.motive.height, this.motive.width).multiply(0.5);
074
075            if (face.getAxis() == Axis.Z) {
076                size.z = 0.5;
077            } else {
078                size.x = 0.5;
079            }
080
081            this.width = (float) size.x;
082            this.length = (float) size.z;
083            this.height = (float) size.y;
084
085            this.boundingBox = new SimpleAxisAlignedBB(
086                    this.x - size.x,
087                    this.y - size.y,
088                    this.z - size.z,
089                    this.x + size.x,
090                    this.y + size.y,
091                    this.z + size.z
092            );
093        } else {
094            this.width = 0;
095            this.height = 0;
096            this.length = 0;
097        }
098
099        super.initEntity();
100    }
101
102    @Override
103    public DataPacket createAddEntityPacket() {
104        AddPaintingPacket addPainting = new AddPaintingPacket();
105        addPainting.entityUniqueId = this.getId();
106        addPainting.entityRuntimeId = this.getId();
107        addPainting.x = (float) this.x;
108        addPainting.y = (float) this.y;
109        addPainting.z = (float) this.z;
110        addPainting.direction = this.getDirection().getHorizontalIndex();
111        addPainting.title = this.namedTag.getString("Motive");
112        return addPainting;
113    }
114
115    @Override
116    public boolean attack(EntityDamageEvent source) {
117        if (super.attack(source)) {
118            if (source instanceof EntityDamageByEntityEvent) {
119                Entity damager = ((EntityDamageByEntityEvent) source).getDamager();
120                if (damager instanceof Player && ((Player) damager).isSurvival() && this.level.getGameRules().getBoolean(GameRule.DO_ENTITY_DROPS)) {
121                    this.level.dropItem(this, new ItemPainting());
122                }
123            }
124            this.close();
125            return true;
126        } else {
127            return false;
128        }
129    }
130
131    @Override
132    public void saveNBT() {
133        super.saveNBT();
134        this.namedTag.putString("Motive", this.motive.title);
135    }
136
137    @Override
138    public void onPushByPiston(BlockEntityPistonArm piston) {
139        if (this.level.getGameRules().getBoolean(GameRule.DO_ENTITY_DROPS)) {
140            this.level.dropItem(this, new ItemPainting());
141        }
142
143        this.close();
144    }
145
146    public Motive getArt() {
147        return getMotive();
148    }
149
150    public Motive getMotive() {
151        return Motive.BY_NAME.get(namedTag.getString("Motive"));
152    }
153
154    public enum Motive {
155        KEBAB("Kebab", 1, 1),
156        AZTEC("Aztec", 1, 1),
157        ALBAN("Alban", 1, 1),
158        AZTEC2("Aztec2", 1, 1),
159        BOMB("Bomb", 1, 1),
160        PLANT("Plant", 1, 1),
161        WASTELAND("Wasteland", 1, 1),
162        WANDERER("Wanderer", 1, 2),
163        GRAHAM("Graham", 1, 2),
164        POOL("Pool", 2, 1),
165        COURBET("Courbet", 2, 1),
166        SUNSET("Sunset", 2, 1),
167        SEA("Sea", 2, 1),
168        CREEBET("Creebet", 2, 1),
169        MATCH("Match", 2, 2),
170        BUST("Bust", 2, 2),
171        STAGE("Stage", 2, 2),
172        VOID("Void", 2, 2),
173        SKULL_AND_ROSES("SkullAndRoses", 2, 2),
174        WITHER("Wither", 2, 2),
175        FIGHTERS("Fighters", 4, 2),
176        SKELETON("Skeleton", 4, 3),
177        DONKEY_KONG("DonkeyKong", 4, 3),
178        POINTER("Pointer", 4, 4),
179        PIG_SCENE("Pigscene", 4, 4),
180        FLAMING_SKULL("Flaming Skull", 4, 4);
181
182        public final String title;
183        public final int width;
184        public final int height;
185
186        private static final Map<String, Motive> BY_NAME = new HashMap<>();
187
188        static {
189            for (Motive motive : values()) {
190                BY_NAME.put(motive.title, motive);
191            }
192        }
193
194        Motive(String title, int width, int height) {
195            this.title = title;
196            this.width = width;
197            this.height = height;
198        }
199    }
200}
