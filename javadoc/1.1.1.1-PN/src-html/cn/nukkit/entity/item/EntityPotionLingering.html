001package cn.nukkit.entity.item;
002
003import cn.nukkit.entity.Entity;
004import cn.nukkit.level.format.FullChunk;
005import cn.nukkit.nbt.tag.CompoundTag;
006import cn.nukkit.nbt.tag.DoubleTag;
007import cn.nukkit.nbt.tag.FloatTag;
008import cn.nukkit.nbt.tag.ListTag;
009import cn.nukkit.potion.Effect;
010import cn.nukkit.potion.Potion;
011
012public class EntityPotionLingering extends EntityPotion {
013    
014    public static final int NETWORK_ID = 101;
015    
016    public EntityPotionLingering(FullChunk chunk, CompoundTag nbt) {
017        super(chunk, nbt);
018    }
019    
020    public EntityPotionLingering(FullChunk chunk, CompoundTag nbt, Entity shootingEntity) {
021        super(chunk, nbt, shootingEntity);
022    }
023
024    @Override
025    protected void initEntity() {
026        super.initEntity();
027        setDataFlag(DATA_FLAGS, DATA_FLAG_LINGER, true);
028    }
029
030    @Override
031    protected void splash(Entity collidedWith) {
032        super.splash(collidedWith);
033        saveNBT();
034        ListTag<?> pos = (ListTag<?>) namedTag.getList("Pos", CompoundTag.class).copy();
035        EntityAreaEffectCloud entity = (EntityAreaEffectCloud) Entity.createEntity("AreaEffectCloud", getChunk(),
036                new CompoundTag().putList(pos)
037                        .putList(new ListTag<>("Rotation")
038                                .add(new FloatTag("", 0))
039                                .add(new FloatTag("", 0))
040                        )
041                        .putList(new ListTag<>("Motion")
042                                .add(new DoubleTag("", 0))
043                                .add(new DoubleTag("", 0))
044                                .add(new DoubleTag("", 0))
045                        )
046                        .putShort("PotionId", potionId)
047        );
048    
049        Effect effect = Potion.getEffect(potionId, true);
050    
051        if (effect != null && entity != null) {
052            entity.cloudEffects.add(effect.setDuration(1).setVisible(false).setAmbient(false));
053            entity.spawnToAll();
054        }
055    }
056}
