001package cn.nukkit.entity;
002
003import cn.nukkit.level.format.FullChunk;
004import cn.nukkit.math.BlockFace;
005import cn.nukkit.nbt.tag.CompoundTag;
006
007/**
008 * author: MagicDroidX
009 * Nukkit Project
010 */
011public abstract class EntityHanging extends Entity {
012    protected int direction;
013
014    public EntityHanging(FullChunk chunk, CompoundTag nbt) {
015        super(chunk, nbt);
016    }
017
018    @Override
019    protected void initEntity() {
020        super.initEntity();
021
022        this.setMaxHealth(1);
023        this.setHealth(1);
024
025        if (this.namedTag.contains("Direction")) {
026            this.direction = this.namedTag.getByte("Direction");
027        } else if (this.namedTag.contains("Dir")) {
028            int d = this.namedTag.getByte("Dir");
029            if (d == 2) {
030                this.direction = 0;
031            } else if (d == 0) {
032                this.direction = 2;
033            }
034        }
035
036    }
037
038    @Override
039    public void saveNBT() {
040        super.saveNBT();
041
042        this.namedTag.putByte("Direction", this.getDirection().getHorizontalIndex());
043        this.namedTag.putInt("TileX", (int) this.x);
044        this.namedTag.putInt("TileY", (int) this.y);
045        this.namedTag.putInt("TileZ", (int) this.z);
046    }
047
048    @Override
049    public BlockFace getDirection() {
050        return BlockFace.fromIndex(this.direction);
051    }
052
053    @Override
054    public boolean onUpdate(int currentTick) {
055        if (this.closed) {
056            return false;
057        }
058
059        if (!this.isAlive()) {
060
061            this.despawnFromAll();
062            if (!this.isPlayer) {
063                this.close();
064            }
065
066            return true;
067        }
068
069        if (this.lastYaw != this.yaw || this.lastX != this.x || this.lastY != this.y || this.lastZ != this.z) {
070            this.despawnFromAll();
071
072            this.direction = (int) (this.yaw / 90);
073
074            this.lastYaw = this.yaw;
075            this.lastX = this.x;
076            this.lastY = this.y;
077            this.lastZ = this.z;
078
079            this.spawnToAll();
080
081            return true;
082        }
083
084        return false;
085    }
086
087    protected boolean isSurfaceValid() {
088        return true;
089    }
090
091}
