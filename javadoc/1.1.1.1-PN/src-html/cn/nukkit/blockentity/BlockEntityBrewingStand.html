045        super(chunk, nbt);

048    @Override
049    protected void initBlockEntity() {
050        inventory = new BrewingInventory(this);
051
052        if (!namedTag.contains("Items") || !(namedTag.get("Items") instanceof ListTag)) {
053            namedTag.putList(new ListTag<CompoundTag>("Items"));
054        }
055
056        for (int i = 0; i < getSize(); i++) {
057            inventory.setItem(i, this.getItem(i));
058        }
059
060        if (!namedTag.contains("CookTime") || namedTag.getShort("CookTime") > MAX_BREW_TIME) {
061            this.brewTime = MAX_BREW_TIME;
062        } else {
063            this.brewTime = namedTag.getShort("CookTime");
064        }
065
066        this.fuelAmount = namedTag.getShort("FuelAmount");
067        this.fuelTotal = namedTag.getShort("FuelTotal");
068
069        if (brewTime < MAX_BREW_TIME) {
070            this.scheduleUpdate();
071        }
072
073        super.initBlockEntity();
074    }

097    public void close() {
098        if (!closed) {
099            for (Player player : new HashSet<>(getInventory().getViewers())) {
100                player.removeWindow(getInventory());
101            }
102            super.close();
103        }

115        namedTag.putList(new ListTag<CompoundTag>("Items"));
116        for (int index = 0; index < getSize(); index++) {
117            this.setItem(index, inventory.getItem(index));

135    protected int getSlotIndex(int index) {
136        ListTag<CompoundTag> list = this.namedTag.getList("Items", CompoundTag.class);
137        for (int i = 0; i < list.size(); i++) {
138            if (list.get(i).getByte("Slot") == index) {
139                return i;
140            }
141        }
142
143        return -1;
144    }

146    @Override
147    public Item getItem(int index) {
148        int i = this.getSlotIndex(index);
149        if (i < 0) {
150            return new ItemBlock(new BlockAir(), 0, 0);
151        } else {
152            CompoundTag data = (CompoundTag) this.namedTag.getList("Items").get(i);
153            return NBTIO.getItemHelper(data);
154        }
155    }
156
157    @Override
158    public void setItem(int index, Item item) {
159        int i = this.getSlotIndex(index);
160
161        CompoundTag d = NBTIO.putItemHelper(item, index);
162
163        if (item.getId() == Item.AIR || item.getCount() <= 0) {
164            if (i >= 0) {
165                this.namedTag.getList("Items").getAll().remove(i);
166            }
167        } else if (i < 0) {
168            (this.namedTag.getList("Items", CompoundTag.class)).add(d);
169        } else {
170            (this.namedTag.getList("Items", CompoundTag.class)).add(i, d);
171        }
172    }

183    @Override
184    public boolean onUpdate() {
185        if (closed) {
186            return false;
187        }
188
189        boolean ret = false;
190
191        Item ingredient = this.inventory.getIngredient();
192        boolean canBrew = false;
193
194        Item fuel = this.getInventory().getFuel();
195        if (this.fuelAmount <= 0 && fuel.getId() == Item.BLAZE_POWDER && fuel.getCount() > 0) {
196            fuel.count--;
197            this.fuelAmount = 20;
198            this.fuelTotal = 20;
199
200            this.inventory.setFuel(fuel);
201            this.sendFuel();
202        }
203
204        if (this.fuelAmount > 0) {
205            for (int i = 1; i <= 3; i++) {
206                if (this.inventory.getItem(i).getId() == Item.POTION) {
207                    canBrew = true;
208                }
209            }
210
211            if (this.brewTime <= MAX_BREW_TIME && canBrew && ingredient.getCount() > 0) {
212                if (!this.checkIngredient(ingredient)) {
213                    canBrew = false;
214                }
215            } else {
216                canBrew = false;
217            }
218        }
219
220        if (canBrew) {
221            if (this.brewTime == MAX_BREW_TIME) {
222                this.sendBrewTime();
223                StartBrewEvent e = new StartBrewEvent(this);
224                this.server.getPluginManager().callEvent(e);
225
226                if (e.isCancelled()) {
227                    return false;
228                }
229            }
230
231            this.brewTime--;
232
233            if (this.brewTime <= 0) { //20 seconds
234                BrewEvent e = new BrewEvent(this);
235                this.server.getPluginManager().callEvent(e);
236
237                if (!e.isCancelled()) {
238                    for (int i = 1; i <= 3; i++) {
239                        Item potion = this.inventory.getItem(i);
240
241                        ContainerRecipe containerRecipe = Server.getInstance().getCraftingManager().matchContainerRecipe(ingredient, potion);
242                        if (containerRecipe != null) {
243                            Item result = containerRecipe.getResult();
244                            result.setDamage(potion.getDamage());
245                            this.inventory.setItem(i, result);
246                        } else {
247                            BrewingRecipe recipe = Server.getInstance().getCraftingManager().matchBrewingRecipe(ingredient, potion);
248                            if (recipe != null) {
249                                this.inventory.setItem(i, recipe.getResult());
250                            }
251                        }
252                    }
253                    this.getLevel().addLevelSoundEvent(this, LevelSoundEventPacket.SOUND_POTION_BREWED);
254
255                    ingredient.count--;
256                    this.inventory.setIngredient(ingredient);
257
258                    this.fuelAmount--;
259                    this.sendFuel();
260                }
261
262                this.brewTime = MAX_BREW_TIME;
263            }
264
265            ret = true;
266        } else {
267            this.brewTime = MAX_BREW_TIME;
268        }
269
270        //this.sendBrewTime();
271        lastUpdate = System.currentTimeMillis();
272
273        return ret;
274    }
275
276    protected void sendFuel() {
277        ContainerSetDataPacket pk = new ContainerSetDataPacket();
278
279        for (Player p : this.inventory.getViewers()) {
280            int windowId = p.getWindowId(this.inventory);
281            if (windowId > 0) {
282                pk.windowId = windowId;
283
284                pk.property = ContainerSetDataPacket.PROPERTY_BREWING_STAND_FUEL_AMOUNT;
285                pk.value = this.fuelAmount;
286                p.dataPacket(pk);
287
288                pk.property = ContainerSetDataPacket.PROPERTY_BREWING_STAND_FUEL_TOTAL;
289                pk.value = this.fuelTotal;
290                p.dataPacket(pk);
291            }
292        }
293    }
294
295    protected void sendBrewTime() {
296        ContainerSetDataPacket pk = new ContainerSetDataPacket();
297        pk.property = ContainerSetDataPacket.PROPERTY_BREWING_STAND_BREW_TIME;
298        pk.value = this.brewTime;
299
300        for (Player p : this.inventory.getViewers()) {
301            int windowId = p.getWindowId(this.inventory);
302            if (windowId > 0) {
303                pk.windowId = windowId;
304
305                p.dataPacket(pk);
306            }
307        }
308    }

310    public void updateBlock() {
311        Block block = this.getLevelBlock();
312
313        if (!(block instanceof BlockBrewingStand)) {
314            return;
315        }
316
317        int meta = 0;
318
319        for (int i = 1; i <= 3; ++i) {
320            Item potion = this.inventory.getItem(i);
321
322            int id = potion.getId();
323            if ((id == Item.POTION || id == Item.SPLASH_POTION || id == Item.LINGERING_POTION) && potion.getCount() > 0) {
324                meta |= 1 << (i - 1);
325            }
326        }
327
328        block.setDamage(meta);
329        this.level.setBlock(block, block, false, false);
330    }

340    @Override
341    public CompoundTag getSpawnCompound() {
342        CompoundTag nbt = new CompoundTag()
343                .putString("id", BlockEntity.BREWING_STAND)
344                .putInt("x", (int) this.x)
345                .putInt("y", (int) this.y)
346                .putInt("z", (int) this.z)
347                .putShort("FuelTotal", this.fuelTotal)
348                .putShort("FuelAmount", this.fuelAmount);
349
350        if (this.brewTime < MAX_BREW_TIME) {
351            nbt.putShort("CookTime", this.brewTime);
352        }
353
354        if (this.hasName()) {
355            nbt.put("CustomName", namedTag.get("CustomName"));
356        }
357
358        return nbt;
359    }
