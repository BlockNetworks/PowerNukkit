001package cn.nukkit.level.format.anvil.palette;
002
003import cn.nukkit.utils.ThreadCache;
004
005/**
006 * @author https://github.com/boy0001/
007 */
008public final class BitArray4096 {
009
010    private final int bitsPerEntry;
011    private final int maxSeqLocIndex;
012    private final int maxEntryValue;
013    private final long[] data;
014
015    public BitArray4096(int bitsPerEntry) {
016        this.bitsPerEntry = bitsPerEntry;
017        this.maxSeqLocIndex = 64 - bitsPerEntry;
018        maxEntryValue = (1 << bitsPerEntry) - 1;
019        int longLen = (this.bitsPerEntry * 4096) >> 6;
020        this.data = new long[longLen];
021    }
022
023    public final void setAt(int index, int value) {
024        if (data.length == 0) return;
025        int bitIndexStart = index * bitsPerEntry;
026        int longIndexStart = bitIndexStart >> 6;
027        int localBitIndexStart = bitIndexStart & 63;
028        this.data[longIndexStart] = this.data[longIndexStart] & ~((long) maxEntryValue << localBitIndexStart) | ((long) value) << localBitIndexStart;
029
030        if(localBitIndexStart > maxSeqLocIndex) {
031            int longIndexEnd = longIndexStart + 1;
032            int localShiftStart = 64 - localBitIndexStart;
033            int localShiftEnd = bitsPerEntry - localShiftStart;
034            this.data[longIndexEnd] = this.data[longIndexEnd] >>> localShiftEnd << localShiftEnd | (((long) value) >> localShiftStart);
035        }
036    }
037
038    public final int getAt(int index) {
039        if (data.length == 0) return 0;
040        int bitIndexStart = index * bitsPerEntry;
041
042        int longIndexStart = bitIndexStart >> 6;
043
044        int localBitIndexStart = bitIndexStart & 63;
045        if(localBitIndexStart <= maxSeqLocIndex) {
046            return (int)(this.data[longIndexStart] >>> localBitIndexStart & maxEntryValue);
047        } else {
048            int localShift = 64 - localBitIndexStart;
049            return (int) ((this.data[longIndexStart] >>> localBitIndexStart | this.data[longIndexStart + 1] << localShift) & maxEntryValue);
050        }
051    }
052
053    public final void fromRawSlow(char[] arr) {
054        for (int i = 0; i < arr.length; i++) {
055            setAt(i, arr[i]);
056        }
057    }
058
059    public final void fromRaw(char[] arr) {
060        final long[] data = this.data;
061        final int dataLength = data.length;
062        final int bitsPerEntry = this.bitsPerEntry;
063        final int maxEntryValue = this.maxEntryValue;
064        final int maxSeqLocIndex = this.maxSeqLocIndex;
065
066        int localStart = 0;
067        char lastVal;
068        int arrI = 0;
069        long l = 0;
070        long nextVal;
071        for (int i = 0; i < dataLength; i++) {
072            for (; localStart <= maxSeqLocIndex; localStart += bitsPerEntry) {
073                lastVal = arr[arrI++];
074                l |= ((long) lastVal << localStart);
075            }
076            if (localStart < 64) {
077                if (i != dataLength - 1) {
078                    lastVal = arr[arrI++];
079                    int shift = 64 - localStart;
080
081                    nextVal = lastVal >> shift;
082
083                    l |= ((lastVal - (nextVal << shift)) << localStart);
084
085                    data[i] = l;
086                    data[i + 1] = l = nextVal;
087
088                    localStart -= maxSeqLocIndex;
089                }
090            } else {
091                localStart = 0;
092                data[i] = l;
093                l = 0;
094            }
095        }
096    }
097
098    public BitArray4096 grow(int newBitsPerEntry) {
099        int amtGrow = newBitsPerEntry - this.bitsPerEntry;
100        if (amtGrow <= 0) return this;
101        BitArray4096 newBitArray = new BitArray4096(newBitsPerEntry);
102
103        char[] buffer = ThreadCache.charCache4096.get();
104        toRaw(buffer);
105        newBitArray.fromRaw(buffer);
106
107        return newBitArray;
108    }
109
110    public BitArray4096 growSlow(int bitsPerEntry) {
111        BitArray4096 newBitArray = new BitArray4096(bitsPerEntry);
112        for (int i = 0; i < 4096; i++) {
113            newBitArray.setAt(i, getAt(i));
114        }
115        return newBitArray;
116    }
117
118    public final char[] toRawSlow() {
119        char[] arr = new char[4096];
120        for (int i = 0; i < arr.length; i++) {
121            arr[i] = (char) getAt(i);
122        }
123        return arr;
124    }
125
126    public final char[] toRaw() {
127        return toRaw(new char[4096]);
128    }
129
130    protected final char[] toRaw(char[] buffer) {
131        final long[] data = this.data;
132        final int dataLength = data.length;
133        final int bitsPerEntry = this.bitsPerEntry;
134        final int maxEntryValue = this.maxEntryValue;
135        final int maxSeqLocIndex = this.maxSeqLocIndex;
136
137        int localStart = 0;
138        char lastVal;
139        int arrI = 0;
140        long l;
141        for (int i = 0; i < dataLength; i++) {
142            l = data[i];
143            for (; localStart <= maxSeqLocIndex; localStart += bitsPerEntry) {
144                lastVal = (char) (l >>> localStart & maxEntryValue);
145                buffer[arrI++] = lastVal;
146            }
147            if (localStart < 64) {
148                if (i != dataLength - 1) {
149                    lastVal = (char) (l >>> localStart);
150                    localStart -= maxSeqLocIndex;
151                    l = data[i + 1];
152                    int localShift = bitsPerEntry - localStart;
153                    lastVal |= l << localShift;
154                    lastVal &= maxEntryValue;
155                    buffer[arrI++] = lastVal;
156                }
157            } else {
158                localStart = 0;
159            }
160        }
161        return buffer;
162    }
163}
