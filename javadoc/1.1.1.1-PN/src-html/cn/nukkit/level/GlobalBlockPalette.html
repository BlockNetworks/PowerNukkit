001package cn.nukkit.level;
002
003import cn.nukkit.Server;
004import cn.nukkit.block.Block;
005import cn.nukkit.nbt.NBTIO;
006import cn.nukkit.nbt.tag.CompoundTag;
007import cn.nukkit.nbt.tag.ListTag;
008import it.unimi.dsi.fastutil.ints.Int2IntMap;
009import it.unimi.dsi.fastutil.ints.Int2IntOpenHashMap;
010import it.unimi.dsi.fastutil.ints.Int2ObjectMap;
011import it.unimi.dsi.fastutil.ints.Int2ObjectOpenHashMap;
012
013import java.io.BufferedInputStream;
014import java.io.IOException;
015import java.io.InputStream;
016import java.util.HashMap;
017import java.util.Map;
018import java.nio.ByteOrder;
019import java.util.LinkedHashMap;
020import java.util.Map;
021import java.util.NoSuchElementException;
022import java.util.concurrent.atomic.AtomicInteger;
023
024public class GlobalBlockPalette {
025    private static final Int2IntMap legacyToRuntimeId = new Int2IntOpenHashMap();
026    private static final Int2IntMap runtimeIdToLegacy = new Int2IntOpenHashMap();
027    private static final Int2ObjectMap<String> legacyIdToString = new Int2ObjectOpenHashMap<>();
028    private static final Map<String, Integer> stringToLegacyId = new HashMap<>();
029    private static final AtomicInteger runtimeIdAllocator = new AtomicInteger(0);
030    public static final byte[] BLOCK_PALETTE;
031
032    static {
033        legacyToRuntimeId.defaultReturnValue(-1);
034        runtimeIdToLegacy.defaultReturnValue(-1);
035
036        Map<CompoundTag, int[]> metaOverrides = new LinkedHashMap<>();
037        try (InputStream stream = Server.class.getClassLoader().getResourceAsStream("runtime_block_states_overrides.dat")) {
038            if (stream == null) {
039                throw new AssertionError("Unable to locate block state nbt");
040            }
041
042            ListTag<CompoundTag> states;
043            try (BufferedInputStream buffered = new BufferedInputStream(stream)) {
044                states = NBTIO.read(buffered).getList("Overrides", CompoundTag.class);
045            }
046
047            for (CompoundTag override : states.getAll()) {
048                if (override.contains("block") && override.contains("meta")) {
049                    metaOverrides.put(override.getCompound("block").remove("version"), override.getIntArray("meta"));
050                }
051            }
052
053        } catch (IOException e) {
054            throw new AssertionError(e);
055        }
056
057        ListTag<CompoundTag> tag;
058        try (InputStream stream = Server.class.getClassLoader().getResourceAsStream("runtime_block_states.dat")) {
059            if (stream == null) {
060                throw new AssertionError("Unable to locate block state nbt");
061            }
062
063            try(BufferedInputStream buffered = new BufferedInputStream(stream)) {
064                //noinspection unchecked
065                tag = (ListTag<CompoundTag>) NBTIO.readTag(buffered, ByteOrder.LITTLE_ENDIAN, false);
066            }
067
068        } catch (IOException e) {
069            throw new AssertionError(e);
070        }
071
072        for (CompoundTag state : tag.getAll()) {
073            int runtimeId = runtimeIdAllocator.getAndIncrement();
074            int[] meta = metaOverrides.get(state.getCompound("block").copy().remove("version"));
075            if (meta == null) {
076                if (!state.contains("meta")) {
077                    continue;
078                } else {
079                    meta = state.getIntArray("meta");
080                }
081            }
082
083            int id = state.getShort("id");
084            String name = state.getCompound("block").getString("name");
085
086            // Resolve to first legacy id
087            runtimeIdToLegacy.put(runtimeId, id << 6 | meta[0]);
088            stringToLegacyId.put(name, id);
089            legacyIdToString.put(id, name);
090
091            for (int val : meta) {
092                int legacyId = id << 6 | val;
093                legacyToRuntimeId.put(legacyId, runtimeId);
094            }
095            state.remove("meta"); // No point in sending this since the client doesn't use it.
096        }
097
098        try {
099            BLOCK_PALETTE = NBTIO.write(tag, ByteOrder.LITTLE_ENDIAN, true);
100        } catch (IOException e) {
101            throw new ExceptionInInitializerError(e);
102        }
103    }
104
105    public static int getOrCreateRuntimeId(int id, int meta) {
106        int legacyId = id << 6 | meta;
107        int runtimeId = legacyToRuntimeId.get(legacyId);
108        if (runtimeId == -1) {
109            //runtimeId = registerMapping(runtimeIdAllocator.incrementAndGet(), legacyId);
110            runtimeId = legacyToRuntimeId.get(248 << 6);
111            //throw new NoSuchElementException("Unmapped block registered id:" + id + " meta:" + meta);
112        }
113        return runtimeId;
114    }
115
116    public static int getOrCreateRuntimeId(int legacyId) throws NoSuchElementException {
117        return getOrCreateRuntimeId(legacyId >> Block.DATA_BITS, legacyId & Block.DATA_MASK);
118    }
119
120    public static int getLegacyId(int runtimeId) {
121        return runtimeIdToLegacy.get(runtimeId);
122    }
123
124    public static String getName(int id) {
125        return legacyIdToString.get(id);
126    }
127}
