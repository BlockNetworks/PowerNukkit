107    private static final float[] biomeWeights = new float[25];
108
109    static {
110        for (int i = -2; i <= 2; ++i) {
111            for (int j = -2; j <= 2; ++j) {
112                biomeWeights[i + 2 + (j + 2) * 5] = (float) (10.0F / Math.sqrt((float) (i * i + j * j) + 0.2F));
113            }
114        }
115    }

171    @Override
172    public void init(ChunkManager level, NukkitRandom random) {
173        this.level = level;
174        this.nukkitRandom = random;
175        this.random = new Random();
176        this.nukkitRandom.setSeed(this.level.getSeed());
177        this.localSeed1 = this.random.nextLong();
178        this.localSeed2 = this.random.nextLong();
179        this.nukkitRandom.setSeed(this.level.getSeed());
180        this.selector = new BiomeSelector(this.nukkitRandom);
181
182        this.minLimitPerlinNoise = new NoiseGeneratorOctavesF(random, 16);
183        this.maxLimitPerlinNoise = new NoiseGeneratorOctavesF(random, 16);
184        this.mainPerlinNoise = new NoiseGeneratorOctavesF(random, 8);
185        this.surfaceNoise = new NoiseGeneratorPerlinF(random, 4);
186        this.scaleNoise = new NoiseGeneratorOctavesF(random, 10);
187        this.depthNoise = new NoiseGeneratorOctavesF(random, 16);
188
189        //this should run before all other populators so that we don't do things like generate ground cover on bedrock or something
190        this.generationPopulators = ImmutableList.of(
191                new PopulatorBedrock(),
192                new PopulatorGroundCover()
193        );
194
195        this.populators = ImmutableList.of(
196                new PopulatorOre(STONE, new OreType[]{
197                        new OreType(new BlockOreCoal(), 20, 17, 0, 128),
198                        new OreType(new BlockOreIron(), 20, 9, 0, 64),
199                        new OreType(new BlockOreRedstone(), 8, 8, 0, 16),
200                        new OreType(new BlockOreLapis(), 1, 7, 0, 16),
201                        new OreType(new BlockOreGold(), 2, 9, 0, 32),
202                        new OreType(new BlockOreDiamond(), 1, 8, 0, 16),
203                        new OreType(new BlockDirt(), 10, 33, 0, 128),
204                        new OreType(new BlockGravel(), 8, 33, 0, 128),
205                        new OreType(new BlockStone(BlockStone.GRANITE), 10, 33, 0, 80),
206                        new OreType(new BlockStone(BlockStone.DIORITE), 10, 33, 0, 80),
207                        new OreType(new BlockStone(BlockStone.ANDESITE), 10, 33, 0, 80)
208                }),
209                new PopulatorCaves()//,
210                //new PopulatorRavines()
211        );
212    }

214    @Override
215    public void generateChunk(final int chunkX, final int chunkZ) {
216        int baseX = chunkX << 4;
217        int baseZ = chunkZ << 4;
218        this.nukkitRandom.setSeed(chunkX * this.localSeed1 ^ chunkZ * this.localSeed2 ^ this.level.getSeed());
219
220        BaseFullChunk chunk = this.level.getChunk(chunkX, chunkZ);
221
222        //generate base noise values
223        float[] depthRegion = this.depthNoise.generateNoiseOctaves(this.depthRegion.get(), chunkX * 4, chunkZ * 4, 5, 5, 200f, 200f, 0.5f);
224        this.depthRegion.set(depthRegion);
225        float[] mainNoiseRegion = this.mainPerlinNoise.generateNoiseOctaves(this.mainNoiseRegion.get(), chunkX * 4, 0, chunkZ * 4, 5, 33, 5, 684.412f / 60f, 684.412f / 160f, 684.412f / 60f);
226        this.mainNoiseRegion.set(mainNoiseRegion);
227        float[] minLimitRegion = this.minLimitPerlinNoise.generateNoiseOctaves(this.minLimitRegion.get(), chunkX * 4, 0, chunkZ * 4, 5, 33, 5, 684.412f, 684.412f, 684.412f);
228        this.minLimitRegion.set(minLimitRegion);
229        float[] maxLimitRegion = this.maxLimitPerlinNoise.generateNoiseOctaves(this.maxLimitRegion.get(), chunkX * 4, 0, chunkZ * 4, 5, 33, 5, 684.412f, 684.412f, 684.412f);
230        this.maxLimitRegion.set(maxLimitRegion);
231        float[] heightMap = this.heightMap.get();
232
233        //generate heightmap and smooth biome heights
234        int horizCounter = 0;
235        int vertCounter = 0;
236        for (int xSeg = 0; xSeg < 5; ++xSeg) {
237            for (int zSeg = 0; zSeg < 5; ++zSeg) {
238                float heightVariationSum = 0.0F;
239                float baseHeightSum = 0.0F;
240                float biomeWeightSum = 0.0F;
241                Biome biome = this.pickBiome(baseX + (xSeg * 4), baseZ + (zSeg * 4));
242
243                for (int xSmooth = -2; xSmooth <= 2; ++xSmooth) {
244                    for (int zSmooth = -2; zSmooth <= 2; ++zSmooth) {
245                        Biome biome1 = this.pickBiome(baseX + (xSeg * 4) + xSmooth, baseZ + (zSeg * 4) + zSmooth);
246                        float baseHeight = biome1.getBaseHeight();
247                        float heightVariation = biome1.getHeightVariation();
248
249                        float scaledWeight = biomeWeights[xSmooth + 2 + (zSmooth + 2) * 5] / (baseHeight + 2.0F);
250
251                        if (biome1.getBaseHeight() > biome.getBaseHeight()) {
252                            scaledWeight /= 2.0F;
253                        }
254
255                        heightVariationSum += heightVariation * scaledWeight;
256                        baseHeightSum += baseHeight * scaledWeight;
257                        biomeWeightSum += scaledWeight;
258                    }
259                }
260
261                heightVariationSum = heightVariationSum / biomeWeightSum;
262                baseHeightSum = baseHeightSum / biomeWeightSum;
263                heightVariationSum = heightVariationSum * 0.9F + 0.1F;
264                baseHeightSum = (baseHeightSum * 4.0F - 1.0F) / 8.0F;
265                float depthNoise = depthRegion[vertCounter] / 8000.0f;
266
267                if (depthNoise < 0.0f) {
268                    depthNoise = -depthNoise * 0.3f;
269                }
270
271                depthNoise = depthNoise * 3.0f - 2.0f;
272
273                if (depthNoise < 0.0f) {
274                    depthNoise = depthNoise / 2.0f;
275
276                    if (depthNoise < -1.0f) {
277                        depthNoise = -1.0f;
278                    }
279
280                    depthNoise = depthNoise / 1.4f;
281                    depthNoise = depthNoise / 2.0f;
282                } else {
283                    if (depthNoise > 1.0f) {
284                        depthNoise = 1.0f;
285                    }
286
287                    depthNoise = depthNoise / 8.0f;
288                }
289
290                ++vertCounter;
291                float baseHeightClone = baseHeightSum;
292                float heightVariationClone = heightVariationSum;
293                baseHeightClone = baseHeightClone + depthNoise * 0.2f;
294                baseHeightClone = baseHeightClone * 8.5f / 8.0f;
295                float baseHeightFactor = 8.5f + baseHeightClone * 4.0f;
296
297                for (int ySeg = 0; ySeg < 33; ++ySeg) {
298                    float baseScale = ((float) ySeg - baseHeightFactor) * 12f * 128.0f / 256.0f / heightVariationClone;
299
300                    if (baseScale < 0.0f) {
301                        baseScale *= 4.0f;
302                    }
303
304                    float minScaled = minLimitRegion[horizCounter] / 512f;
305                    float maxScaled = maxLimitRegion[horizCounter] / 512f;
306                    float noiseScaled = (mainNoiseRegion[horizCounter] / 10.0f + 1.0f) / 2.0f;
307                    float clamp = MathHelper.denormalizeClamp(minScaled, maxScaled, noiseScaled) - baseScale;
308
309                    if (ySeg > 29) {
310                        float yScaled = ((float) (ySeg - 29) / 3.0F);
311                        clamp = clamp * (1.0f - yScaled) + -10.0f * yScaled;
312                    }
313
314                    heightMap[horizCounter] = clamp;
315                    ++horizCounter;
316                }
317            }
318        }
319
320        //place blocks
321        for (int xSeg = 0; xSeg < 4; ++xSeg) {
322            int xScale = xSeg * 5;
323            int xScaleEnd = (xSeg + 1) * 5;
324
325            for (int zSeg = 0; zSeg < 4; ++zSeg) {
326                int zScale1 = (xScale + zSeg) * 33;
327                int zScaleEnd1 = (xScale + zSeg + 1) * 33;
328                int zScale2 = (xScaleEnd + zSeg) * 33;
329                int zScaleEnd2 = (xScaleEnd + zSeg + 1) * 33;
330
331                for (int ySeg = 0; ySeg < 32; ++ySeg) {
332                    double height1 = heightMap[zScale1 + ySeg];
333                    double height2 = heightMap[zScaleEnd1 + ySeg];
334                    double height3 = heightMap[zScale2 + ySeg];
335                    double height4 = heightMap[zScaleEnd2 + ySeg];
336                    double height5 = (heightMap[zScale1 + ySeg + 1] - height1) * 0.125f;
337                    double height6 = (heightMap[zScaleEnd1 + ySeg + 1] - height2) * 0.125f;
338                    double height7 = (heightMap[zScale2 + ySeg + 1] - height3) * 0.125f;
339                    double height8 = (heightMap[zScaleEnd2 + ySeg + 1] - height4) * 0.125f;
340
341                    for (int yIn = 0; yIn < 8; ++yIn) {
342                        double baseIncr = height1;
343                        double baseIncr2 = height2;
344                        double scaleY = (height3 - height1) * 0.25f;
345                        double scaleY2 = (height4 - height2) * 0.25f;
346
347                        for (int zIn = 0; zIn < 4; ++zIn) {
348                            double scaleZ = (baseIncr2 - baseIncr) * 0.25f;
349                            double scaleZ2 = baseIncr - scaleZ;
350
351                            for (int xIn = 0; xIn < 4; ++xIn) {
352                                if ((scaleZ2 += scaleZ) > 0.0f) {
353                                    chunk.setBlockId(xSeg * 4 + zIn, ySeg * 8 + yIn, zSeg * 4 + xIn, STONE);
354                                } else if (ySeg * 8 + yIn <= seaHeight) {
355                                    chunk.setBlockId(xSeg * 4 + zIn, ySeg * 8 + yIn, zSeg * 4 + xIn, STILL_WATER);
356                                }
357                            }
358
359                            baseIncr += scaleY;
360                            baseIncr2 += scaleY2;
361                        }
362
363                        height1 += height5;
364                        height2 += height6;
365                        height3 += height7;
366                        height4 += height8;
367                    }
368                }
369            }
370        }
371
372        for (int x = 0; x < 16; x++) {
373            for (int z = 0; z < 16; z++) {
374                Biome biome = this.selector.pickBiome(baseX | x, baseZ | z);
375
376                chunk.setBiome(x, z, biome);
377            }
378        }
379
380        //populate chunk
381        for (Populator populator : this.generationPopulators) {
382            populator.populate(this.level, chunkX, chunkZ, this.nukkitRandom, chunk);
383        }
384    }

386    @Override
387    public void populateChunk(int chunkX, int chunkZ) {
388        BaseFullChunk chunk = this.level.getChunk(chunkX, chunkZ);
389        this.nukkitRandom.setSeed(0xdeadbeef ^ (chunkX << 8) ^ chunkZ ^ this.level.getSeed());
390        for (Populator populator : this.populators) {
391            populator.populate(this.level, chunkX, chunkZ, this.nukkitRandom, chunk);
392        }
393
394        @SuppressWarnings("deprecation")
395        Biome biome = EnumBiome.getBiome(chunk.getBiomeId(7, 7));
396        biome.populateChunk(this.level, chunkX, chunkZ, this.nukkitRandom);
397    }
