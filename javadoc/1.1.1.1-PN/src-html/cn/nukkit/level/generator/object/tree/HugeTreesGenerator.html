001package cn.nukkit.level.generator.object.tree;
002
003import cn.nukkit.block.Block;
004import cn.nukkit.level.ChunkManager;
005import cn.nukkit.math.NukkitRandom;
006import cn.nukkit.math.Vector3;
007
008public abstract class HugeTreesGenerator extends TreeGenerator {
009    /**
010     * The base height of the tree
011     */
012    protected final int baseHeight;
013
014    /**
015     * Sets the metadata for the wood blocks used
016     */
017    protected final Block woodMetadata;
018
019    /**
020     * Sets the metadata for the leaves used in huge trees
021     */
022    protected final Block leavesMetadata;
023    protected int extraRandomHeight;
024
025    public HugeTreesGenerator(int baseHeightIn, int extraRandomHeightIn, Block woodMetadataIn, Block leavesMetadataIn) {
026        this.baseHeight = baseHeightIn;
027        this.extraRandomHeight = extraRandomHeightIn;
028        this.woodMetadata = woodMetadataIn;
029        this.leavesMetadata = leavesMetadataIn;
030    }
031
032    /*
033     * Calculates the height based on this trees base height and its extra random height
034     */
035    protected int getHeight(NukkitRandom rand) {
036        int i = rand.nextBoundedInt(3) + this.baseHeight;
037
038        if (this.extraRandomHeight > 1) {
039            i += rand.nextBoundedInt(this.extraRandomHeight);
040        }
041
042        return i;
043    }
044
045    /*
046     * returns whether or not there is space for a tree to grow at a certain position
047     */
048    private boolean isSpaceAt(ChunkManager worldIn, Vector3 leavesPos, int height) {
049        boolean flag = true;
050
051        if (leavesPos.getY() >= 1 && leavesPos.getY() + height + 1 <= 256) {
052            for (int i = 0; i <= 1 + height; ++i) {
053                int j = 2;
054
055                if (i == 0) {
056                    j = 1;
057                } else if (i >= 1 + height - 2) {
058                    j = 2;
059                }
060
061                for (int k = -j; k <= j && flag; ++k) {
062                    for (int l = -j; l <= j && flag; ++l) {
063                        Vector3 blockPos = leavesPos.add(k, i, l);
064                        if (leavesPos.getY() + i < 0 || leavesPos.getY() + i >= 256 || !this.canGrowInto(worldIn.getBlockIdAt((int) blockPos.x, (int) blockPos.y, (int) blockPos.z))) {
065                            flag = false;
066                        }
067                    }
068                }
069            }
070
071            return flag;
072        } else {
073            return false;
074        }
075    }
076
077    /*
078     * returns whether or not there is dirt underneath the block where the tree will be grown.
079     * It also generates dirt around the block in a 2x2 square if there is dirt underneath the blockpos.
080     */
081    private boolean ensureDirtsUnderneath(Vector3 pos, ChunkManager worldIn) {
082        Vector3 blockpos = pos.down();
083        int block = worldIn.getBlockIdAt((int) blockpos.x, (int) blockpos.y, (int) blockpos.z);
084
085        if ((block == Block.GRASS || block == Block.DIRT) && pos.getY() >= 2) {
086            this.setDirtAt(worldIn, blockpos);
087            this.setDirtAt(worldIn, blockpos.east());
088            this.setDirtAt(worldIn, blockpos.south());
089            this.setDirtAt(worldIn, blockpos.south().east());
090            return true;
091        } else {
092            return false;
093        }
094    }
095
096    /*
097     * returns whether or not a tree can grow at a specific position.
098     * If it can, it generates surrounding dirt underneath.
099     */
100    protected boolean ensureGrowable(ChunkManager worldIn, NukkitRandom rand, Vector3 treePos, int p_175929_4_) {
101        return this.isSpaceAt(worldIn, treePos, p_175929_4_) && this.ensureDirtsUnderneath(treePos, worldIn);
102    }
103
104    /*
105     * grow leaves in a circle with the outsides being within the circle
106     */
107    protected void growLeavesLayerStrict(ChunkManager worldIn, Vector3 layerCenter, int width) {
108        int i = width * width;
109
110        for (int j = -width; j <= width + 1; ++j) {
111            for (int k = -width; k <= width + 1; ++k) {
112                int l = j - 1;
113                int i1 = k - 1;
114
115                if (j * j + k * k <= i || l * l + i1 * i1 <= i || j * j + i1 * i1 <= i || l * l + k * k <= i) {
116                    Vector3 blockpos = layerCenter.add(j, 0, k);
117                    int id = worldIn.getBlockIdAt((int) blockpos.x, (int) blockpos.y, (int) blockpos.z);
118
119                    if (id == Block.AIR || id == Block.LEAVES) {
120                        this.setBlockAndNotifyAdequately(worldIn, blockpos, this.leavesMetadata);
121                    }
122                }
123            }
124        }
125    }
126
127    /*
128     * grow leaves in a circle
129     */
130    protected void growLeavesLayer(ChunkManager worldIn, Vector3 layerCenter, int width) {
131        int i = width * width;
132
133        for (int j = -width; j <= width; ++j) {
134            for (int k = -width; k <= width; ++k) {
135                if (j * j + k * k <= i) {
136                    Vector3 blockpos = layerCenter.add(j, 0, k);
137                    int id = worldIn.getBlockIdAt((int) blockpos.x, (int) blockpos.y, (int) blockpos.z);
138
139                    if (id == Block.AIR || id == Block.LEAVES) {
140                        this.setBlockAndNotifyAdequately(worldIn, blockpos, this.leavesMetadata);
141                    }
142                }
143            }
144        }
145    }
146}
