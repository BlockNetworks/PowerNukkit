001package cn.nukkit.level.generator.object.tree;
002
003import cn.nukkit.block.Block;
004import cn.nukkit.block.BlockLeaves;
005import cn.nukkit.block.BlockVine;
006import cn.nukkit.block.BlockWood;
007import cn.nukkit.level.ChunkManager;
008import cn.nukkit.math.BlockVector3;
009import cn.nukkit.math.NukkitRandom;
010import cn.nukkit.math.Vector3;
011
012public class ObjectSwampTree extends TreeGenerator {
013
014    /**
015     * The metadata value of the wood to use in tree generation.
016     */
017    private final Block metaWood = new BlockWood(BlockWood.OAK);
018
019    /**
020     * The metadata value of the leaves to use in tree generation.
021     */
022    private final Block metaLeaves = new BlockLeaves(BlockLeaves.OAK);
023
024    @Override
025    public boolean generate(ChunkManager worldIn, NukkitRandom rand, Vector3 vectorPosition) {
026        BlockVector3 position = new BlockVector3(vectorPosition.getFloorX(), vectorPosition.getFloorY(), vectorPosition.getFloorZ());
027
028        int i = rand.nextBoundedInt(4) + 5;
029        boolean flag = true;
030
031        if (position.getY() >= 1 && position.getY() + i + 1 <= 256) {
032            for (int j = position.getY(); j <= position.getY() + 1 + i; ++j) {
033                int k = 1;
034
035                if (j == position.getY()) {
036                    k = 0;
037                }
038
039                if (j >= position.getY() + 1 + i - 2) {
040                    k = 3;
041                }
042
043                BlockVector3 pos2 = new BlockVector3();
044
045                for (int l = position.getX() - k; l <= position.getX() + k && flag; ++l) {
046                    for (int i1 = position.getZ() - k; i1 <= position.getZ() + k && flag; ++i1) {
047                        if (j >= 0 && j < 256) {
048                            pos2.setComponents(l, j, i1);
049                            if (!this.canGrowInto(worldIn.getBlockIdAt(pos2.x, pos2.y, pos2.z))) {
050                                flag = false;
051                            }
052                        } else {
053                            flag = false;
054                        }
055                    }
056                }
057            }
058
059            if (!flag) {
060                return false;
061            } else {
062                BlockVector3 down = position.down();
063                int block = worldIn.getBlockIdAt(down.x, down.y, down.z);
064
065                if ((block == Block.GRASS || block == Block.DIRT) && position.getY() < 256 - i - 1) {
066                    this.setDirtAt(worldIn, down);
067
068                    for (int k1 = position.getY() - 3 + i; k1 <= position.getY() + i; ++k1) {
069                        int j2 = k1 - (position.getY() + i);
070                        int l2 = 2 - j2 / 2;
071
072                        for (int j3 = position.getX() - l2; j3 <= position.getX() + l2; ++j3) {
073                            int k3 = j3 - position.getX();
074
075                            for (int i4 = position.getZ() - l2; i4 <= position.getZ() + l2; ++i4) {
076                                int j1 = i4 - position.getZ();
077
078                                if (Math.abs(k3) != l2 || Math.abs(j1) != l2 || rand.nextBoundedInt(2) != 0 && j2 != 0) {
079                                    BlockVector3 blockpos = new BlockVector3(j3, k1, i4);
080                                    int id = worldIn.getBlockIdAt(blockpos.x, blockpos.y, blockpos.z);
081
082                                    if (id == Block.AIR || id == Block.LEAVES || id == Block.VINE) {
083                                        this.setBlockAndNotifyAdequately(worldIn, blockpos, this.metaLeaves);
084                                    }
085                                }
086                            }
087                        }
088                    }
089
090                    for (int l1 = 0; l1 < i; ++l1) {
091                        BlockVector3 up = position.up(l1);
092                        int id = worldIn.getBlockIdAt(up.x, up.y, up.z);
093
094                        if (id == Block.AIR || id == Block.LEAVES || id == Block.WATER || id == Block.STILL_WATER) {
095                            this.setBlockAndNotifyAdequately(worldIn, up, this.metaWood);
096                        }
097                    }
098
099                    for (int i2 = position.getY() - 3 + i; i2 <= position.getY() + i; ++i2) {
100                        int k2 = i2 - (position.getY() + i);
101                        int i3 = 2 - k2 / 2;
102                        BlockVector3 pos2 = new BlockVector3();
103
104                        for (int l3 = position.getX() - i3; l3 <= position.getX() + i3; ++l3) {
105                            for (int j4 = position.getZ() - i3; j4 <= position.getZ() + i3; ++j4) {
106                                pos2.setComponents(l3, i2, j4);
107
108                                if (worldIn.getBlockIdAt(pos2.x, pos2.y, pos2.z) == Block.LEAVES) {
109                                    BlockVector3 blockpos2 = pos2.west();
110                                    BlockVector3 blockpos3 = pos2.east();
111                                    BlockVector3 blockpos4 = pos2.north();
112                                    BlockVector3 blockpos1 = pos2.south();
113
114                                    if (rand.nextBoundedInt(4) == 0 && worldIn.getBlockIdAt(blockpos2.x, blockpos2.y, blockpos2.z) == Block.AIR) {
115                                        this.addHangingVine(worldIn, blockpos2, 8);
116                                    }
117
118                                    if (rand.nextBoundedInt(4) == 0 && worldIn.getBlockIdAt(blockpos3.x, blockpos3.y, blockpos3.z) == Block.AIR) {
119                                        this.addHangingVine(worldIn, blockpos3, 2);
120                                    }
121
122                                    if (rand.nextBoundedInt(4) == 0 && worldIn.getBlockIdAt(blockpos4.x, blockpos4.y, blockpos4.z) == Block.AIR) {
123                                        this.addHangingVine(worldIn, blockpos4, 1);
124                                    }
125
126                                    if (rand.nextBoundedInt(4) == 0 && worldIn.getBlockIdAt(blockpos1.x, blockpos1.y, blockpos1.z) == Block.AIR) {
127                                        this.addHangingVine(worldIn, blockpos1, 4);
128                                    }
129                                }
130                            }
131                        }
132                    }
133                    return true;
134                } else {
135                    return false;
136                }
137            }
138        } else {
139            return false;
140        }
141    }
142
143    private void addVine(ChunkManager worldIn, BlockVector3 pos, int meta) {
144        this.setBlockAndNotifyAdequately(worldIn, pos, new BlockVine(meta));
145    }
146
147    private void addHangingVine(ChunkManager worldIn, BlockVector3 pos, int meta) {
148        this.addVine(worldIn, pos, meta);
149        int i = 4;
150
151        for (pos = pos.down(); i > 0 && worldIn.getBlockIdAt(pos.x, pos.y, pos.z) == Block.AIR; --i) {
152            this.addVine(worldIn, pos, meta);
153            pos = pos.down();
154        }
155    }
156}
