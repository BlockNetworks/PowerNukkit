001package cn.nukkit.level.generator.object;
002
003import cn.nukkit.block.Block;
004import cn.nukkit.level.ChunkManager;
005import cn.nukkit.math.BlockVector3;
006import cn.nukkit.math.NukkitRandom;
007import cn.nukkit.math.Vector3;
008
009public abstract class BasicGenerator {
010
011    //also autism, see below
012    public abstract boolean generate(ChunkManager level, NukkitRandom rand, Vector3 position);
013
014    public void setDecorationDefaults() {
015    }
016
017    protected void setBlockAndNotifyAdequately(ChunkManager level, BlockVector3 pos, Block state) {
018        setBlock(level, new Vector3(pos.x, pos.y, pos.z), state);
019    }
020
021    protected void setBlockAndNotifyAdequately(ChunkManager level, Vector3 pos, Block state) {
022        setBlock(level, pos, state);
023    }
024
025    //what autism is this? why are we using floating-point vectors for setting block IDs?
026    protected void setBlock(ChunkManager level, Vector3 v, Block b) {
027        level.setBlockAt((int) v.x, (int) v.y, (int) v.z, b.getId(), b.getDamage());
028    }
029}
