001package cn.nukkit.level;
002
003import cn.nukkit.math.Vector3;
004import cn.nukkit.utils.LevelException;
005
006/**
007 * author: MagicDroidX
008 * Nukkit Project
009 */
010public class Location extends Position {
011
012    public double yaw;
013    public double pitch;
014
015    public Location() {
016        this(0);
017    }
018
019    public Location(double x) {
020        this(x, 0);
021    }
022
023    public Location(double x, double y) {
024        this(x, y, 0);
025    }
026
027    public Location(double x, double y, double z, Level level) {
028        this(x, y, z, 0, 0, level);
029    }
030
031    public Location(double x, double y, double z) {
032        this(x, y, z, 0);
033    }
034
035    public Location(double x, double y, double z, double yaw) {
036        this(x, y, z, yaw, 0);
037    }
038
039    public Location(double x, double y, double z, double yaw, double pitch) {
040        this(x, y, z, yaw, pitch, null);
041    }
042
043    public Location(double x, double y, double z, double yaw, double pitch, Level level) {
044        this.x = x;
045        this.y = y;
046        this.z = z;
047        this.yaw = yaw;
048        this.pitch = pitch;
049        this.level = level;
050    }
051
052    public static Location fromObject(Vector3 pos) {
053        return fromObject(pos, null, 0.0f, 0.0f);
054    }
055
056    public static Location fromObject(Vector3 pos, Level level) {
057        return fromObject(pos, level, 0.0f, 0.0f);
058    }
059
060    public static Location fromObject(Vector3 pos, Level level, double yaw) {
061        return fromObject(pos, level, yaw, 0.0f);
062    }
063
064    public static Location fromObject(Vector3 pos, Level level, double yaw, double pitch) {
065        return new Location(pos.x, pos.y, pos.z, yaw, pitch, (level == null) ? ((pos instanceof Position) ? ((Position) pos).level : null) : level);
066    }
067
068    public double getYaw() {
069        return this.yaw;
070    }
071
072    public double getPitch() {
073        return this.pitch;
074    }
075
076    public Location setYaw(double yaw) {
077        this.yaw = yaw;
078        return this;
079    }
080
081    public Location setPitch(double pitch) {
082        this.pitch = pitch;
083        return this;
084    }
085
086    @Override
087    public String toString() {
088        return "Location (level=" + (this.isValid() ? this.getLevel().getName() : "null") + ", x=" + this.x + ", y=" + this.y + ", z=" + this.z + ", yaw=" + this.yaw + ", pitch=" + this.pitch + ")";
089    }
090
091    @Override
092    public Location getLocation() {
093        if (this.isValid()) return new Location(this.x, this.y, this.z, this.yaw, this.pitch, this.level);
094        else throw new LevelException("Undefined Level reference");
095    }
096
097    @Override
098    public Location add(double x) {
099        return this.add(x, 0, 0);
100    }
101
102    @Override
103    public Location add(double x, double y) {
104        return this.add(x, y, 0);
105    }
106
107    @Override
108    public Location add(double x, double y, double z) {
109        return new Location(this.x + x, this.y + y, this.z + z, this.yaw, this.pitch, this.level);
110    }
111
112    @Override
113    public Location add(Vector3 x) {
114        return new Location(this.x + x.getX(), this.y + x.getY(), this.z + x.getZ(), this.yaw, this.pitch, this.level);
115    }
116
117    @Override
118    public Location subtract() {
119        return this.subtract(0, 0, 0);
120    }
121
122    @Override
123    public Location subtract(double x) {
124        return this.subtract(x, 0, 0);
125    }
126
127    @Override
128    public Location subtract(double x, double y) {
129        return this.subtract(x, y, 0);
130    }
131
132    @Override
133    public Location subtract(double x, double y, double z) {
134        return this.add(-x, -y, -z);
135    }
136
137    @Override
138    public Location subtract(Vector3 x) {
139        return this.add(-x.getX(), -x.getY(), -x.getZ());
140    }
141
142    @Override
143    public Location multiply(double number) {
144        return new Location(this.x * number, this.y * number, this.z * number, this.yaw, this.pitch, this.level);
145    }
146
147    @Override
148    public Location divide(double number) {
149        return new Location(this.x / number, this.y / number, this.z / number, this.yaw, this.pitch, this.level);
150    }
151
152    @Override
153    public Location ceil() {
154        return new Location((int) Math.ceil(this.x), (int) Math.ceil(this.y), (int) Math.ceil(this.z), this.yaw, this.pitch, this.level);
155    }
156
157    @Override
158    public Location floor() {
159        return new Location(this.getFloorX(), this.getFloorY(), this.getFloorZ(), this.yaw, this.pitch, this.level);
160    }
161
162    @Override
163    public Location round() {
164        return new Location(Math.round(this.x), Math.round(this.y), Math.round(this.z), this.yaw, this.pitch, this.level);
165    }
166
167    @Override
168    public Location abs() {
169        return new Location((int) Math.abs(this.x), (int) Math.abs(this.y), (int) Math.abs(this.z), this.yaw, this.pitch, this.level);
170    }
171
172    public Vector3 getDirectionVector() {
173        double pitch = ((getPitch() + 90) * Math.PI) / 180;
174        double yaw = ((getYaw() + 90) * Math.PI) / 180;
175        double x = Math.sin(pitch) * Math.cos(yaw);
176        double z = Math.sin(pitch) * Math.sin(yaw);
177        double y = Math.cos(pitch);
178        return new Vector3(x, y, z).normalize();
179    }
180
181    @Override
182    public Location clone() {
183        return (Location) super.clone();
184    }
185}
