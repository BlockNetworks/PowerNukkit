088    @Override
089    public Item[] getDrops(Item item) {
090        if (item.isShears()) {
091            return new Item[]{
092                    toItem()
093            };
094        } else {
095            if (this.canDropApple() && ThreadLocalRandom.current().nextInt(200) == 0) {
096                return new Item[]{
097                        Item.get(Item.APPLE)
098                };
099            }
100            if (ThreadLocalRandom.current().nextInt(20) == 0) {
101                if (ThreadLocalRandom.current().nextBoolean()) {
102                    return new Item[]{
103                            Item.get(Item.STICK, 0, ThreadLocalRandom.current().nextInt(1, 2))
104                    };
105                } else if ((this.getDamage() & 0x03) != JUNGLE || ThreadLocalRandom.current().nextInt(20) == 0) {
106                    return new Item[]{
107                            this.getSapling()
108                    };
109                }
110            }
111        }
112        return new Item[0];
113    }

115    @Override
116    public int onUpdate(int type) {
117        if (type == Level.BLOCK_UPDATE_RANDOM && !isPersistent() && !isCheckDecay()) {
118            setCheckDecay(true);
119            getLevel().setBlock(this, this, false, false);
120        } else if (type == Level.BLOCK_UPDATE_RANDOM && isCheckDecay() && !isPersistent()) {
121            setDamage(getDamage() & 0x03);
122            int check = 0;
123
124            LeavesDecayEvent ev = new LeavesDecayEvent(this);
125
126            Server.getInstance().getPluginManager().callEvent(ev);
127            if (ev.isCancelled() || findLog(this, new LongArraySet(), 0, check)) {
128                getLevel().setBlock(this, this, false, false);
129            } else {
130                getLevel().useBreakOn(this);
131                return Level.BLOCK_UPDATE_NORMAL;
132            }
133        }
134        return 0;
135    }
136
137    private Boolean findLog(Block pos, LongSet visited, Integer distance, Integer check) {
138        return findLog(pos, visited, distance, check, null);
139    }
140
141    private Boolean findLog(Block pos, LongSet visited, Integer distance, Integer check, BlockFace fromSide) {
142        ++check;
143        long index = Hash.hashBlock((int) pos.x, (int) pos.y, (int) pos.z);
144        if (visited.contains(index)) return false;
145        if (pos.getId() == WOOD || pos.getId() == WOOD2) return true;
146        if ((pos.getId() == LEAVES || pos.getId() == LEAVES2) && distance <= 4) {
147            visited.add(index);
148            int down = pos.down().getId();
149            if (down == WOOD || down == WOOD2) {
150                return true;
151            }
152            if (fromSide == null) {
153                //North, East, South, West
154                for (int side = 2; side <= 5; ++side) {
155                    if (this.findLog(pos.getSide(BlockFace.fromIndex(side)), visited, distance + 1, check, BlockFace.fromIndex(side)))
156                        return true;
157                }
158            } else { //No more loops
159                switch (fromSide) {
160                    case NORTH:
161                        if (this.findLog(pos.getSide(BlockFace.NORTH), visited, distance + 1, check, fromSide))
162                            return true;
163                        if (this.findLog(pos.getSide(BlockFace.WEST), visited, distance + 1, check, fromSide))
164                            return true;
165                        if (this.findLog(pos.getSide(BlockFace.EAST), visited, distance + 1, check, fromSide))
166                            return true;
167                        break;
168                    case SOUTH:
169                        if (this.findLog(pos.getSide(BlockFace.SOUTH), visited, distance + 1, check, fromSide))
170                            return true;
171                        if (this.findLog(pos.getSide(BlockFace.WEST), visited, distance + 1, check, fromSide))
172                            return true;
173                        if (this.findLog(pos.getSide(BlockFace.EAST), visited, distance + 1, check, fromSide))
174                            return true;
175                        break;
176                    case WEST:
177                        if (this.findLog(pos.getSide(BlockFace.NORTH), visited, distance + 1, check, fromSide))
178                            return true;
179                        if (this.findLog(pos.getSide(BlockFace.SOUTH), visited, distance + 1, check, fromSide))
180                            return true;
181                        if (this.findLog(pos.getSide(BlockFace.WEST), visited, distance + 1, check, fromSide))
182                            return true;
183                    case EAST:
184                        if (this.findLog(pos.getSide(BlockFace.NORTH), visited, distance + 1, check, fromSide))
185                            return true;
186                        if (this.findLog(pos.getSide(BlockFace.SOUTH), visited, distance + 1, check, fromSide))
187                            return true;
188                        if (this.findLog(pos.getSide(BlockFace.EAST), visited, distance + 1, check, fromSide))
189                            return true;
190                        break;
191                }
192            }
193        }
194        return false;
195    }

197    public boolean isCheckDecay() {
198        return (this.getDamage() & 0x08) != 0;
199    }
200
201    public void setCheckDecay(boolean checkDecay) {
202        if (checkDecay) {
203            this.setDamage(this.getDamage() | 0x08);
204        } else {
205            this.setDamage(this.getDamage() & ~0x08);
206        }
207    }
208
209    public boolean isPersistent() {
210        return (this.getDamage() & 0x04) != 0;
211    }
212
213    public void setPersistent(boolean persistent) {
214        if (persistent) {
215            this.setDamage(this.getDamage() | 0x04);
216        } else {
217            this.setDamage(this.getDamage() & ~0x04);
218        }
219    }
