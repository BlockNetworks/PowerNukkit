001package cn.nukkit.block;
002
003import cn.nukkit.math.AxisAlignedBB;
004import cn.nukkit.math.SimpleAxisAlignedBB;
005import cn.nukkit.utils.LevelException;
006
007/**
008 * Created on 2015/12/6 by xtypr.
009 * Package cn.nukkit.block in project Nukkit .
010 */
011public abstract class BlockThin extends BlockTransparent {
012
013    protected BlockThin() {
014    }
015
016    @Override
017    public boolean isSolid() {
018        return false;
019    }
020
021    protected AxisAlignedBB recalculateBoundingBox() {
022        double f = 0.4375;
023        double f1 = 0.5625;
024        double f2 = 0.4375;
025        double f3 = 0.5625;
026        try {
027            boolean flag = this.canConnect(this.north());
028            boolean flag1 = this.canConnect(this.south());
029            boolean flag2 = this.canConnect(this.west());
030            boolean flag3 = this.canConnect(this.east());
031            if ((!flag2 || !flag3) && (flag2 || flag3 || flag || flag1)) {
032                if (flag2) {
033                    f = 0;
034                } else if (flag3) {
035                    f1 = 1;
036                }
037            } else {
038                f = 0;
039                f1 = 1;
040            }
041            if ((!flag || !flag1) && (flag2 || flag3 || flag || flag1)) {
042                if (flag) {
043                    f2 = 0;
044                } else if (flag1) {
045                    f3 = 1;
046                }
047            } else {
048                f2 = 0;
049                f3 = 1;
050            }
051        } catch (LevelException ignore) {
052            //null sucks
053        }
054        return new SimpleAxisAlignedBB(
055                this.x + f,
056                this.y,
057                this.z + f2,
058                this.x + f1,
059                this.y + 1,
060                this.z + f3
061        );
062    }
063
064    public boolean canConnect(Block block) {
065        return block.isSolid() || block.getId() == this.getId() || block.getId() == GLASS_PANE || block.getId() == GLASS;
066    }
067
068}
