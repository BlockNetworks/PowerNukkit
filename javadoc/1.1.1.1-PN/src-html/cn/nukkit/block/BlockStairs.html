001package cn.nukkit.block;
002
003import cn.nukkit.Player;
004import cn.nukkit.item.Item;
005import cn.nukkit.item.ItemTool;
006import cn.nukkit.math.AxisAlignedBB;
007import cn.nukkit.math.BlockFace;
008import cn.nukkit.math.SimpleAxisAlignedBB;
009import cn.nukkit.utils.Faceable;
010
011/**
012 * author: MagicDroidX
013 * Nukkit Project
014 */
015public abstract class BlockStairs extends BlockTransparentMeta implements Faceable {
016
017    protected BlockStairs(int meta) {
018        super(meta);
019    }
020
021    @Override
022    public double getMinY() {
023        // TODO: this seems wrong
024        return this.y + ((getDamage() & 0x04) > 0 ? 0.5 : 0);
025    }
026
027    @Override
028    public double getMaxY() {
029        // TODO: this seems wrong
030        return this.y + ((getDamage() & 0x04) > 0 ? 1 : 0.5);
031    }
032
033    @Override
034    public boolean place(Item item, Block block, Block target, BlockFace face, double fx, double fy, double fz, Player player) {
035        int[] faces = new int[]{2, 1, 3, 0};
036        this.setDamage(faces[player != null ? player.getDirection().getHorizontalIndex() : 0]);
037        if ((fy > 0.5 && face != BlockFace.UP) || face == BlockFace.DOWN) {
038            this.setDamage(this.getDamage() | 0x04); //Upside-down stairs
039        }
040        this.getLevel().setBlock(block, this, true, true);
041
042        return true;
043    }
044
045    @Override
046    public Item[] getDrops(Item item) {
047        if (item.isPickaxe() && item.getTier() >= ItemTool.TIER_WOODEN) {
048            return new Item[]{
049                    toItem()
050            };
051        } else {
052            return new Item[0];
053        }
054    }
055
056    @Override
057    public Item toItem() {
058        Item item = super.toItem();
059        item.setDamage(0);
060        return item;
061    }
062
063    @Override
064    public boolean collidesWithBB(AxisAlignedBB bb) {
065        int damage = this.getDamage();
066        int side = damage & 0x03;
067        double f = 0;
068        double f1 = 0.5;
069        double f2 = 0.5;
070        double f3 = 1;
071        if ((damage & 0x04) > 0) {
072            f = 0.5;
073            f1 = 1;
074            f2 = 0;
075            f3 = 0.5;
076        }
077
078        if (bb.intersectsWith(new SimpleAxisAlignedBB(
079                this.x,
080                this.y + f,
081                this.z,
082                this.x + 1,
083                this.y + f1,
084                this.z + 1
085        ))) {
086            return true;
087        }
088
089
090        if (side == 0) {
091            if (bb.intersectsWith(new SimpleAxisAlignedBB(
092                    this.x + 0.5,
093                    this.y + f2,
094                    this.z,
095                    this.x + 1,
096                    this.y + f3,
097                    this.z + 1
098            ))) {
099                return true;
100            }
101        } else if (side == 1) {
102            if (bb.intersectsWith(new SimpleAxisAlignedBB(
103                    this.x,
104                    this.y + f2,
105                    this.z,
106                    this.x + 0.5,
107                    this.y + f3,
108                    this.z + 1
109            ))) {
110                return true;
111            }
112        } else if (side == 2) {
113            if (bb.intersectsWith(new SimpleAxisAlignedBB(
114                    this.x,
115                    this.y + f2,
116                    this.z + 0.5,
117                    this.x + 1,
118                    this.y + f3,
119                    this.z + 1
120            ))) {
121                return true;
122            }
123        } else if (side == 3) {
124            if (bb.intersectsWith(new SimpleAxisAlignedBB(
125                    this.x,
126                    this.y + f2,
127                    this.z,
128                    this.x + 1,
129                    this.y + f3,
130                    this.z + 0.5
131            ))) {
132                return true;
133            }
134        }
135
136        return false;
137    }
138
139    @Override
140    public int getWaterloggingLevel() {
141        return 1;
142    }
143
144    @Override
145    public BlockFace getBlockFace() {
146        return BlockFace.fromHorizontalIndex(this.getDamage() & 0x7);
147    }
148}
