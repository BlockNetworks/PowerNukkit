791    private static double breakTime0(double blockHardness, boolean correctTool, boolean canHarvestWithHand,
792                                     int blockId, int toolType, int toolTier, int efficiencyLoreLevel, int hasteEffectLevel,
793                                     boolean insideOfWaterWithoutAquaAffinity, boolean outOfWaterButNotOnGround) {
794        double baseTime = ((correctTool || canHarvestWithHand) ? 1.5 : 5.0) * blockHardness;
795        double speed = 1.0 / baseTime;
796        boolean isWoolBlock = blockId == Block.WOOL, isCobweb = blockId == Block.COBWEB;
797        if (correctTool) speed *= toolBreakTimeBonus0(toolType, toolTier, isWoolBlock, isCobweb);
798        speed += speedBonusByEfficiencyLore0(efficiencyLoreLevel);
799        speed *= speedRateByHasteLore0(hasteEffectLevel);
800        if (insideOfWaterWithoutAquaAffinity) speed *= 0.2;
801        if (outOfWaterButNotOnGround) speed *= 0.2;
802        return 1.0 / speed;
803    }

1048    public MovingObjectPosition calculateIntercept(Vector3 pos1, Vector3 pos2) {
1049        AxisAlignedBB bb = this.getBoundingBox();
1050        if (bb == null) {
1051            return null;
1052        }
1053
1054        Vector3 v1 = pos1.getIntermediateWithXValue(pos2, bb.getMinX());
1055        Vector3 v2 = pos1.getIntermediateWithXValue(pos2, bb.getMaxX());
1056        Vector3 v3 = pos1.getIntermediateWithYValue(pos2, bb.getMinY());
1057        Vector3 v4 = pos1.getIntermediateWithYValue(pos2, bb.getMaxY());
1058        Vector3 v5 = pos1.getIntermediateWithZValue(pos2, bb.getMinZ());
1059        Vector3 v6 = pos1.getIntermediateWithZValue(pos2, bb.getMaxZ());
1060
1061        if (v1 != null && !bb.isVectorInYZ(v1)) {
1062            v1 = null;
1063        }
1064
1065        if (v2 != null && !bb.isVectorInYZ(v2)) {
1066            v2 = null;
1067        }
1068
1069        if (v3 != null && !bb.isVectorInXZ(v3)) {
1070            v3 = null;
1071        }
1072
1073        if (v4 != null && !bb.isVectorInXZ(v4)) {
1074            v4 = null;
1075        }
1076
