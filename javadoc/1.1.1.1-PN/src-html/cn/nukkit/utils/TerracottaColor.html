001package cn.nukkit.utils;
002
003public enum TerracottaColor {
004
005    BLACK(0, 15, "Black", "Ink Sack", BlockColor.BLACK_TERRACOTA_BLOCK_COLOR),
006    RED(1, 14, "Red", "Rose Red", BlockColor.RED_TERRACOTA_BLOCK_COLOR),
007    GREEN(2, 13, "Green", "Cactus Green", BlockColor.GREEN_TERRACOTA_BLOCK_COLOR),
008    BROWN(3, 12, "Brown", "Cocoa Beans", BlockColor.BROWN_TERRACOTA_BLOCK_COLOR),
009    BLUE(4, 11, "Blue", "Lapis Lazuli", BlockColor.BLUE_TERRACOTA_BLOCK_COLOR),
010    PURPLE(5, 10, "Purple", BlockColor.PURPLE_TERRACOTA_BLOCK_COLOR),
011    CYAN(6, 9, "Cyan", BlockColor.CYAN_TERRACOTA_BLOCK_COLOR),
012    LIGHT_GRAY(7, 8, "Light Gray", BlockColor.LIGHT_GRAY_TERRACOTA_BLOCK_COLOR),
013    GRAY(8, 7, "Gray", BlockColor.GRAY_TERRACOTA_BLOCK_COLOR),
014    PINK(9, 6, "Pink", BlockColor.PINK_TERRACOTA_BLOCK_COLOR),
015    LIME(10, 5, "Lime", BlockColor.LIME_TERRACOTA_BLOCK_COLOR),
016    YELLOW(11, 4, "Yellow", "Dandelion Yellow", BlockColor.YELLOW_TERRACOTA_BLOCK_COLOR),
017    LIGHT_BLUE(12, 3, "Light Blue", BlockColor.LIGHT_BLUE_TERRACOTA_BLOCK_COLOR),
018    MAGENTA(13, 2, "Magenta", BlockColor.MAGENTA_TERRACOTA_BLOCK_COLOR),
019    ORANGE(14, 1, "Orange", BlockColor.ORANGE_TERRACOTA_BLOCK_COLOR),
020    WHITE(15, 0, "White", "Bone Meal", BlockColor.WHITE_TERRACOTA_BLOCK_COLOR);
021
022
023    private int dyeColorMeta;
024    private int terracottaColorMeta;
025    private String colorName;
026    private String dyeName;
027    private BlockColor blockColor;
028
029
030    private final static TerracottaColor[] BY_TERRACOTA_DATA;
031    private final static TerracottaColor[] BY_DYE_DATA;
032
033    TerracottaColor(int dyeColorMeta, int terracottaColorMeta, String colorName, BlockColor blockColor) {
034        this(dyeColorMeta, terracottaColorMeta, colorName, colorName + " Dye", blockColor);
035    }
036
037    TerracottaColor(int dyeColorMeta, int terracottaColorMeta, String colorName, String dyeName, BlockColor blockColor) {
038        this.dyeColorMeta = dyeColorMeta;
039        this.terracottaColorMeta = terracottaColorMeta;
040        this.colorName = colorName;
041        this.blockColor = blockColor;
042        this.dyeName = dyeName;
043    }
044
045    public BlockColor getColor() {
046        return this.blockColor;
047    }
048
049    public int getDyeData() {
050        return this.dyeColorMeta;
051    }
052
053    public int getTerracottaData() {
054        return this.terracottaColorMeta;
055    }
056
057    public String getName() {
058        return this.colorName;
059    }
060
061    public String getDyeName() {
062        return this.dyeName;
063    }
064
065    static {
066        BY_DYE_DATA = values();
067        BY_TERRACOTA_DATA = values();
068
069        for (TerracottaColor color : values()) {
070            BY_TERRACOTA_DATA[color.terracottaColorMeta & 0x0f] = color;
071            BY_DYE_DATA[color.dyeColorMeta & 0x0f] = color;
072        }
073    }
074
075    public static TerracottaColor getByDyeData(int dyeColorMeta) {
076        return BY_DYE_DATA[dyeColorMeta & 0x0f];
077    }
078
079    public static TerracottaColor getByTerracottaData(int terracottaColorMeta) {
080        return BY_TERRACOTA_DATA[terracottaColorMeta & 0x0f];
081    }
082
083}
