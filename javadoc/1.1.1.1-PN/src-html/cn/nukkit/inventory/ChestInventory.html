001package cn.nukkit.inventory;
002
003import cn.nukkit.Player;
004import cn.nukkit.blockentity.BlockEntityChest;
005import cn.nukkit.level.Level;
006import cn.nukkit.network.protocol.BlockEventPacket;
007import cn.nukkit.network.protocol.LevelSoundEventPacket;
008
009/**
010 * author: MagicDroidX
011 * Nukkit Project
012 */
013public class ChestInventory extends ContainerInventory {
014
015    protected DoubleChestInventory doubleInventory;
016
017    public ChestInventory(BlockEntityChest chest) {
018        super(chest, InventoryType.CHEST);
019    }
020
021    @Override
022    public BlockEntityChest getHolder() {
023        return (BlockEntityChest) this.holder;
024    }
025
026    @Override
027    public void onOpen(Player who) {
028        super.onOpen(who);
029
030        if (this.getViewers().size() == 1) {
031            BlockEventPacket pk = new BlockEventPacket();
032            pk.x = (int) this.getHolder().getX();
033            pk.y = (int) this.getHolder().getY();
034            pk.z = (int) this.getHolder().getZ();
035            pk.case1 = 1;
036            pk.case2 = 2;
037
038            Level level = this.getHolder().getLevel();
039            if (level != null) {
040                level.addLevelSoundEvent(this.getHolder().add(0.5, 0.5, 0.5), LevelSoundEventPacket.SOUND_CHEST_OPEN);
041                level.addChunkPacket((int) this.getHolder().getX() >> 4, (int) this.getHolder().getZ() >> 4, pk);
042            }
043        }
044    }
045
046    @Override
047    public void onClose(Player who) {
048        if (this.getViewers().size() == 1) {
049            BlockEventPacket pk = new BlockEventPacket();
050            pk.x = (int) this.getHolder().getX();
051            pk.y = (int) this.getHolder().getY();
052            pk.z = (int) this.getHolder().getZ();
053            pk.case1 = 1;
054            pk.case2 = 0;
055
056            Level level = this.getHolder().getLevel();
057            if (level != null) {
058                level.addLevelSoundEvent(this.getHolder().add(0.5, 0.5, 0.5), LevelSoundEventPacket.SOUND_CHEST_CLOSED);
059                level.addChunkPacket((int) this.getHolder().getX() >> 4, (int) this.getHolder().getZ() >> 4, pk);
060            }
061        }
062
063        super.onClose(who);
064    }
065
066    public void setDoubleInventory(DoubleChestInventory doubleInventory) {
067        this.doubleInventory = doubleInventory;
068    }
069
070    public DoubleChestInventory getDoubleInventory() {
071        return doubleInventory;
072    }
073
074    @Override
075    public void sendSlot(int index, Player... players) {
076        if (this.doubleInventory != null) {
077            this.doubleInventory.sendSlot(this, index, players);
078        } else {
079            super.sendSlot(index, players);
080        }
081    }
082}
