001package cn.nukkit.scheduler;
002
003import cn.nukkit.Server;
004import cn.nukkit.utils.Utils;
005
006import java.io.ByteArrayInputStream;
007import java.io.File;
008import java.io.IOException;
009import java.io.InputStream;
010import java.nio.charset.StandardCharsets;
011
012/**
013 * author: MagicDroidX
014 * Nukkit Project
015 */
016public class FileWriteTask extends AsyncTask {
017    private final File file;
018    private final InputStream contents;
019
020    public FileWriteTask(String path, String contents) {
021        this(new File(path), contents);
022    }
023
024    public FileWriteTask(String path, byte[] contents) {
025        this(new File(path), contents);
026    }
027
028    public FileWriteTask(String path, InputStream contents) {
029        this.file = new File(path);
030        this.contents = contents;
031    }
032
033    public FileWriteTask(File file, String contents) {
034        this.file = file;
035        this.contents = new ByteArrayInputStream(contents.getBytes(StandardCharsets.UTF_8));
036    }
037
038    public FileWriteTask(File file, byte[] contents) {
039        this.file = file;
040        this.contents = new ByteArrayInputStream(contents);
041    }
042
043    public FileWriteTask(File file, InputStream contents) {
044        this.file = file;
045        this.contents = contents;
046    }
047
048    @Override
049    public void onRun() {
050        try {
051            Utils.writeFile(file, contents);
052        } catch (IOException e) {
053            Server.getInstance().getLogger().logException(e);
054        }
055    }
056
057}
