025/**
026 * Stores a sequential range of {@link PositionTracking} objects in a file. The read operation is cached.
027 * <p>This object holds a file handler and must be closed when it is no longer needed.</p>
028 * <p>Once closed the instance cannot be reused.</p>
029 * @author joserobjr
030 */
031@PowerNukkitOnly
032@Since("1.4.0.0-PN")
033@ParametersAreNonnullByDefault
034public class PositionTrackingStorage implements Closeable {
035    @PowerNukkitOnly
036    @Since("1.4.0.0-PN")
037    public static final int DEFAULT_MAX_STORAGE = 500;
038    private static final byte[] HEADER = new byte[]{ 12,32,32, 'P', 'N', 'P', 'T', 'D', 'B', '1' };
039    private final int startIndex;
040    private final int maxStorage;
041    private final long garbagePos;
042    private final long stringHeapPos;
043    private final RandomAccessFile persistence;
044    private final Cache<Integer, Optional<PositionTracking>> cache = CacheBuilder.newBuilder().expireAfterAccess(5,TimeUnit.MINUTES).concurrencyLevel(1).build();
045    private int nextIndex;

047    /**
048     * Opens or create the file and all directories in the path automatically. The given start index will be used
049     * in new files and will be checked when opening files. If the file being opened don't matches this value
050     * internally than an <code>IllegalArgumentException</code> will be thrown.
051     * @param startIndex The number of the first handler. Must be higher than 0 and must match the number of the existing file.
052     * @param persistenceFile The file being opened or created. Parent directories will also be created if necessary.
053     * @throws IOException If an error has occurred while reading, parsing or creating the file
054     * @throws IllegalArgumentException If opening an existing file and the internal startIndex don't match the given startIndex
055     */
056    @PowerNukkitOnly
057    @Since("1.4.0.0-PN")
058    public PositionTrackingStorage(int startIndex, File persistenceFile) throws IOException {
059        this(startIndex, persistenceFile, 0);
060    }

062    /**
063     * Opens or create the file and all directories in the path automatically. The given start index will be used
064     * in new files and will be checked when opening files. If the file being opened don't matches this value
065     * internally than an <code>IllegalArgumentException</code> will be thrown.
066     * @param startIndex The number of the first handler. Must be higher than 0 and must match the number of the existing file.
067     * @param persistenceFile The file being opened or created. Parent directories will also be created if necessary.
068     * @param maxStorage The maximum amount of positions that this storage may hold. It cannot be changed after creation. 
069     *                   Ignored when loading an existing file. When zero or negative, a default value will be used.
070     * @throws IOException If an error has occurred while reading, parsing or creating the file
071     * @throws IllegalArgumentException If opening an existing file and the internal startIndex don't match the given startIndex
072     */
073    @PowerNukkitOnly
074    @Since("1.4.0.0-PN")
075    public PositionTrackingStorage(int startIndex, File persistenceFile, int maxStorage) throws IOException {
076        Preconditions.checkArgument(startIndex > 0, "Start index must be positive. Got {}", startIndex);
077        this.startIndex = startIndex;
078        if (maxStorage <= 0) {
079            maxStorage = DEFAULT_MAX_STORAGE;
080        }
081        
082        boolean created = false;
083        if (!persistenceFile.isFile()) {
084            if (!persistenceFile.getParentFile().isDirectory() && !persistenceFile.getParentFile().mkdirs()) {
085                throw new FileNotFoundException("Could not create the directory "+persistenceFile.getParent());
086            }
087            if (!persistenceFile.createNewFile()) {
088                throw new FileNotFoundException("Could not create the file "+persistenceFile);
089            }
090            created = true;
091        } else if(persistenceFile.length() == 0) {
092            created = true;
093        }
094        
095        this.persistence = new RandomAccessFile(persistenceFile, "rwd");
096        try {
097            if (created) {
098                persistence.write(ByteBuffer.allocate(HEADER.length + 4 + 4 + 4)
099                        .put(HEADER)
100                        .putInt(maxStorage)
101                        .putInt(startIndex)
102                        .putInt(startIndex)
103                        .array());
104                this.maxStorage = maxStorage;
105                nextIndex = startIndex;
106            } else {
107                byte[] check = new byte[HEADER.length];
108                EOFException eof = null;
109                int max;
110                int next;
111                int start;
112                try {
113                    persistence.readFully(check);
114                    byte[] buf = new byte[4 + 4 + 4];
115                    persistence.readFully(buf);
116                    ByteBuffer buffer = ByteBuffer.wrap(buf);
117                    max = buffer.getInt();
118                    next = buffer.getInt();
119                    start = buffer.getInt();
120                } catch (EOFException e) {
121                    eof = e;
122                    max = 0;
123                    next = 0;
124                    start = 0;
125                }
126                if (eof != null || max <= 0 || next <= 0 || start <= 0 || !Arrays.equals(check, HEADER)) {
127                    throw new IOException("The file " + persistenceFile + " is not a valid PowerNukkit TrackingPositionDB persistence file.", eof);
128                }
129                if (start != startIndex) {
130                    throw new IllegalArgumentException("The start index "+startIndex+" was given but the file "+persistenceFile+" has start index "+start);
131                }
132                this.maxStorage = maxStorage = max;
133                this.nextIndex = next;
134            }
135            garbagePos = getAxisPos(startIndex + maxStorage);
136
137            //                          cnt  off len  max
138            stringHeapPos = garbagePos + 4 + (8 + 4) * 15;
139
140            if (created) {
141                persistence.seek(stringHeapPos - 1);
142                persistence.writeByte(0);
143            }
144        } catch (Throwable e) {
145            try {
146                persistence.close();
147            } catch (Throwable e2) {
148                e.addSuppressed(e2);
149            }
150            throw e;
151        }
152    }
153
154    private long getAxisPos(int trackingHandler) {
155        //                    max str cur  on  nam len  x   y   z 
156        return HEADER.length + 4 + 4 + 4 + (1 + 8 + 4 + 8 + 8 + 8) * (long)(trackingHandler - startIndex);
157    }
158
159    private void validateHandler(int trackingHandler) {
160        Preconditions.checkArgument(trackingHandler >= startIndex, "The trackingHandler {} is too low for this storage (starts at {})", trackingHandler, startIndex);
161        int limit = startIndex + maxStorage;
162        Preconditions.checkArgument(trackingHandler <= limit, "The trackingHandler {} is too high for this storage (ends at {})", trackingHandler, limit);
163    }

165    /**
166     * Retrieves the {@link PositionTracking} object that is assigned to the given trackingHandler. 
167     * The handler must be valid for this storage.
168     * <p>This call may return a cached result but the returned object can be modified freely.</p>
169     * @param trackingHandler A valid handler for this storage
170     * @return A clone of the cached result.
171     * @throws IOException If an error has occurred while accessing the file
172     * @throws IllegalArgumentException If the trackingHandler is not valid for this storage
173     */
174    @PowerNukkitOnly
175    @Since("1.4.0.0-PN")
176    @Nullable
177    public PositionTracking getPosition(int trackingHandler) throws IOException {
178        validateHandler(trackingHandler);
179        try {
180            return cache.get(trackingHandler, ()-> loadPosition(trackingHandler, true))
181                    .map(PositionTracking::clone)
182                    .orElse(null);
183        } catch (ExecutionException e) {
184            throw handleExecutionException(e);
185        }
186    }

189     * Retrieves the {@link PositionTracking} object that is assigned to the given trackingHandler. 
190     * The handler must be valid for this storage.
191     * <p>This call may return a cached result but the returned object can be modified freely.</p>
192     * @param trackingHandler A valid handler for this storage
193     * @param onlyEnabled When false, disabled positions that wasn't invalidated may be returned. 
194     *                    Caching only works when this is set to true 
195     * @return A clone of the cached result.
196     * @throws IOException If an error has occurred while accessing the file
197     * @throws IllegalArgumentException If the trackingHandler is not valid for this storage
198     */
199    @PowerNukkitOnly
200    @Since("1.4.0.0-PN")

202    public PositionTracking getPosition(int trackingHandler, boolean onlyEnabled) throws IOException {
203        if (onlyEnabled) {
204            return getPosition(trackingHandler);
205        }
206        validateHandler(trackingHandler);
207        return loadPosition(trackingHandler, false).orElse(null);
208    }

239    /**
240     * Adds the given position as a new entry in this storage, even if the position is already registered and enabled.
241     * @param position The position that needs a handler 
242     * @param enabled If the position will be added as enabled or disabled
243     * @return The trackingHandler assigned to the position or an empty OptionalInt if none was found and this storage is full
244     * @throws IOException If an error occurred while reading or writing the file
245     */
246    @PowerNukkitOnly
247    @Since("1.4.0.0-PN")
248    public synchronized OptionalInt addNewPosition(NamedPosition position, boolean enabled) throws IOException {
249        OptionalInt handler = addNewPos(position, enabled);
250        if (!handler.isPresent()) {
251            return handler;
252        }
253        if (enabled) {
254            cache.put(handler.getAsInt(), Optional.of(new PositionTracking(position)));
255        }
256        return handler;
257    }

259    @PowerNukkitOnly
260    @Since("1.4.0.0-PN")
261    @Nonnull
262    public OptionalInt findTrackingHandler(NamedPosition position) throws IOException {
263        OptionalInt cached = cache.asMap().entrySet().stream()
264                .filter(e-> e.getValue().filter(position::matchesNamedPosition).isPresent())
265                .mapToInt(Map.Entry::getKey)
266                .findFirst();
267        if (cached.isPresent()) {
268            return cached;
269        }
270        IntList handlers = findTrackingHandlers(position, true, 1);
271        if (handlers.isEmpty()) {
272            return OptionalInt.empty();
273        }
274        int found = handlers.getInt(0);
275        cache.put(found, Optional.of(new PositionTracking(position)));
276        return OptionalInt.of(found);
277    }

302    @PowerNukkitOnly

304    public synchronized boolean setEnabled(int trackingHandler, boolean enabled) throws IOException {
305        validateHandler(trackingHandler);
306        long pos = getAxisPos(trackingHandler);
307        persistence.seek(pos);
308        if (persistence.readBoolean() == enabled) {
309            return false;
310        }
311        if (persistence.readLong() == 0 && enabled) {
312            return false;
313        }
314        persistence.seek(pos);
315        persistence.writeBoolean(enabled);
316        cache.invalidate(trackingHandler);
317        return true;
318    }

338    private synchronized void invalidatePos(int trackingHandler) throws IOException {
339        long pos = getAxisPos(trackingHandler);
340        persistence.seek(pos);
341        persistence.writeBoolean(false);
342        byte[] buf = new byte[8 + 4];
343        persistence.readFully(buf);
344        ByteBuffer buffer = ByteBuffer.wrap(buf);
345        long namePos = buffer.getLong();
346        int nameLen = buffer.getInt();
347        persistence.seek(pos + 1);
348        persistence.write(new byte[8 + 4]);
349        cache.put(trackingHandler, Optional.empty());
350        addGarbage(namePos, nameLen);
351    }

353    private synchronized void addGarbage(long pos, int len) throws IOException {
354        persistence.seek(garbagePos);
355        int count = persistence.readInt();
356        if (count >= 15) {
357            return;
358        }
359        byte[] buf = new byte[4 + 8];
360        ByteBuffer buffer = ByteBuffer.wrap(buf);
361        if (count > 0) {
362            for (int attempt = 0; attempt < 15; attempt++) {
363                persistence.readFully(buf);
364                buffer.rewind();
365                long garbage = buffer.getLong();
366                int garbageLen = buffer.getInt();
367                if (garbage != 0) {
368                    if (garbage + garbageLen == pos) {
369                        persistence.seek(persistence.getFilePointer() - 4 - 8);
370                        buffer.rewind();
371                        buffer.putLong(garbage)
372                                .putInt(garbageLen + len);
373                        persistence.write(buf);
374                        return;
375                    } else if (pos + len == garbage) {
376                        persistence.seek(persistence.getFilePointer() - 4 - 8);
377                        buffer.rewind();
378                        buffer.putLong(pos)
379                                .putInt(garbageLen + len);
380                        persistence.write(buf);
381                        return;
382                    }
383                }
384            }
385            
386            persistence.seek(garbagePos + 4);
387        }
388        
389        for (int attempt = 0; attempt < 15; attempt++) {
390            persistence.readFully(buf);
391            buffer.rewind();
392            long garbage = buffer.getLong();
393            if (garbage == 0) {
394                persistence.seek(persistence.getFilePointer() - 4 - 8);
395                buffer.rewind();
396                buffer.putLong(pos).putInt(len);
397                persistence.write(buf);
398                persistence.seek(garbagePos);
399                persistence.writeInt(count + 1);
400                return;
401            }
402        }
403    }

405    private synchronized long findSpaceInStringHeap(int len) throws IOException {
406        persistence.seek(garbagePos);
407        int remaining = persistence.readInt();
408        if (remaining <= 0) {
409            return persistence.length();
410        }
411        
412        byte[] buf = new byte[4 + 8];
413        ByteBuffer buffer = ByteBuffer.wrap(buf);
414        for (int attempt = 0; attempt < 15; attempt++) {
415            persistence.readFully(buf);
416            buffer.rewind();
417            long garbage = buffer.getLong();
418            int garbageLen = buffer.getInt();
419            if (garbage >= stringHeapPos && len <= garbageLen) {
420                persistence.seek(persistence.getFilePointer() - 4 - 8);
421                if (garbageLen == len) {
422                    persistence.write(new byte[8 + 4]);
423                    persistence.seek(garbagePos);
424                    persistence.writeInt(remaining - 1);
425                } else {
426                    buffer.rewind();
427                    buffer.putLong(garbage + len).putInt(garbageLen - len);
428                    persistence.write(buf);
429                }
430                return garbage;
431            }
432        }
433        return persistence.length();
434    }

436    private synchronized OptionalInt addNewPos(NamedPosition pos, boolean enabled) throws IOException {
437        if (nextIndex - startIndex >= maxStorage) {
438            return OptionalInt.empty();
439        }
440        int handler = nextIndex++;
441        writePos(handler, pos, enabled);
442        persistence.seek(HEADER.length + 4);
443        persistence.writeInt(nextIndex);
444        return OptionalInt.of(handler);
445    }
446    
447    private synchronized void writePos(int trackingHandler, NamedPosition pos, boolean enabled) throws IOException {
448        byte[] name = pos.getLevelName().getBytes(StandardCharsets.UTF_8);
449        long namePos = addLevelName(name);
450        persistence.seek(getAxisPos(trackingHandler));
451        persistence.write(ByteBuffer.allocate(1 + 8 + 4 + 8 + 8 + 8)
452                .put(enabled? (byte)1 : 0)
453                .putLong(namePos)
454                .putInt(name.length)
455                .putDouble(pos.x)
456                .putDouble(pos.y)
457                .putDouble(pos.z)
458                .array());
459    }
460    
461    private synchronized long addLevelName(byte[] name) throws IOException {
462        long pos = findSpaceInStringHeap(name.length);
463        persistence.seek(pos);
464        persistence.write(name);
465        return pos;
466    }

482    @PowerNukkitOnly
483    @Since("1.4.0.0-PN")
484    @Nonnull
485    public synchronized IntList findTrackingHandlers(NamedPosition pos, boolean onlyEnabled, int limit) throws IOException {
486        persistence.seek(HEADER.length + 4 + 4 + 4);
487        int handler = startIndex - 1;
488        final double lookingX = pos.x;
489        final double lookingY = pos.y;
490        final double lookingZ = pos.z;
491        final byte[] lookingName = pos.getLevelName().getBytes(StandardCharsets.UTF_8);
492        IntList results = new IntArrayList(NukkitMath.clamp(limit, 1, 16));
493        byte[] buf = new byte[8 + 4 + 8 + 8 + 8];
494        ByteBuffer buffer = ByteBuffer.wrap(buf);
495        while (true) {
496            handler++;
497            if (handler >= nextIndex) {
498                return results;
499            }
500            boolean enabled = persistence.readBoolean();
501            if (onlyEnabled && !enabled) {
502                if (persistence.skipBytes(36) != 36) throw new EOFException();
503                continue;
504            }
505            
506            persistence.readFully(buf);
507            
508            buffer.rewind();
509            long namePos = buffer.getLong();
510            int nameLen = buffer.getInt();
511            double x = buffer.getDouble();
512            double y = buffer.getDouble();
513            double z = buffer.getDouble();
514            if (namePos > 0 && nameLen > 0 && x == lookingX && y == lookingY && z == lookingZ) {
515                long fp = persistence.getFilePointer();
516                byte[] nameBytes = new byte[nameLen];
517                persistence.seek(namePos);
518                persistence.readFully(nameBytes);
519                if (Arrays.equals(lookingName, nameBytes)) {
520                    results.add(handler);
521                    if (results.size() >= limit) {
522                        return results;
523                    }
524                }
525                persistence.seek(fp);
526            }
527        }
528    }

530    private synchronized Optional<PositionTracking> loadPosition(int trackingHandler, boolean onlyEnabled) throws IOException {
531        if (trackingHandler >= nextIndex) {
532            return Optional.empty();
533        }
534        
535        persistence.seek(getAxisPos(trackingHandler));
536        byte[] buf = new byte[1 + 8 + 4 + 8 + 8 + 8];
537        persistence.readFully(buf);
538        boolean enabled = buf[0] == 1;
539        if (!enabled && onlyEnabled) {
540            return Optional.empty();
541        }
542
543        ByteBuffer buffer = ByteBuffer.wrap(buf, 1, buf.length - 1);
544
545        long namePos = buffer.getLong();
546        if (namePos == 0) {
547            return Optional.empty();
548        }
549        int nameLen = buffer.getInt();
550
551        double x = buffer.getDouble();
552        double y = buffer.getDouble();
553        double z = buffer.getDouble();
554
555        byte[] nameBytes = new byte[nameLen];
556        persistence.seek(namePos);
557        persistence.readFully(nameBytes);
558        String name = new String(nameBytes, StandardCharsets.UTF_8);
559        return Optional.of(new PositionTracking(name, x, y, z));
560    }
