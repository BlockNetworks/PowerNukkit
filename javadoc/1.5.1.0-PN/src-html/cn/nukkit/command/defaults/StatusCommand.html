001package cn.nukkit.command.defaults;
002
003import cn.nukkit.Nukkit;
004import cn.nukkit.Server;
005import cn.nukkit.command.CommandSender;
006import cn.nukkit.level.Level;
007import cn.nukkit.math.NukkitMath;
008import cn.nukkit.utils.TextFormat;
009
010import java.util.Objects;
011import java.util.concurrent.TimeUnit;
012
013/**
014 * @author xtypr
015 * @since 2015/11/11
016 */
017public class StatusCommand extends VanillaCommand {
018    private static final String UPTIME_FORMAT = TextFormat.RED + "%d" + TextFormat.GOLD + " days " +
019            TextFormat.RED + "%d" + TextFormat.GOLD + " hours " +
020            TextFormat.RED + "%d" + TextFormat.GOLD + " minutes " +
021            TextFormat.RED + "%d" + TextFormat.GOLD + " seconds";
022
023    public StatusCommand(String name) {
024        super(name, "%nukkit.command.status.description", "%nukkit.command.status.usage");
025        this.setPermission("nukkit.command.status");
026        this.commandParameters.clear();
027    }
028
029    @Override
030    public boolean execute(CommandSender sender, String commandLabel, String[] args) {
031        if (!this.testPermission(sender)) {
032            return true;
033        }
034
035        Server server = sender.getServer();
036        sender.sendMessage(TextFormat.GREEN + "---- " + TextFormat.WHITE + "Server status" + TextFormat.GREEN + " ----");
037
038        long time = System.currentTimeMillis() - Nukkit.START_TIME;
039
040        sender.sendMessage(TextFormat.GOLD + "Uptime: " + formatUptime(time));
041
042        TextFormat tpsColor = TextFormat.GREEN;
043        float tps = server.getTicksPerSecond();
044        if (tps < 17) {
045            tpsColor = TextFormat.GOLD;
046        } else if (tps < 12) {
047            tpsColor = TextFormat.RED;
048        }
049
050        sender.sendMessage(TextFormat.GOLD + "Current TPS: " + tpsColor + NukkitMath.round(tps, 2));
051
052        sender.sendMessage(TextFormat.GOLD + "Load: " + tpsColor + server.getTickUsage() + "%");
053
054        sender.sendMessage(TextFormat.GOLD + "Network upload: " + TextFormat.GREEN + NukkitMath.round((server.getNetwork().getUpload() / 1024 * 1000), 2) + " kB/s");
055
056        sender.sendMessage(TextFormat.GOLD + "Network download: " + TextFormat.GREEN + NukkitMath.round((server.getNetwork().getDownload() / 1024 * 1000), 2) + " kB/s");
057
058        sender.sendMessage(TextFormat.GOLD + "Thread count: " + TextFormat.GREEN + Thread.getAllStackTraces().size());
059
060
061        Runtime runtime = Runtime.getRuntime();
062        double totalMB = NukkitMath.round(((double) runtime.totalMemory()) / 1024 / 1024, 2);
063        double usedMB = NukkitMath.round((double) (runtime.totalMemory() - runtime.freeMemory()) / 1024 / 1024, 2);
064        double maxMB = NukkitMath.round(((double) runtime.maxMemory()) / 1024 / 1024, 2);
065        double usage = usedMB / maxMB * 100;
066        TextFormat usageColor = TextFormat.GREEN;
067
068        if (usage > 85) {
069            usageColor = TextFormat.GOLD;
070        }
071
072        sender.sendMessage(TextFormat.GOLD + "Used memory: " + usageColor + usedMB + " MB. (" + NukkitMath.round(usage, 2) + "%)");
073
074        sender.sendMessage(TextFormat.GOLD + "Total memory: " + TextFormat.RED + totalMB + " MB.");
075
076        sender.sendMessage(TextFormat.GOLD + "Maximum VM memory: " + TextFormat.RED + maxMB + " MB.");
077
078        sender.sendMessage(TextFormat.GOLD + "Available processors: " + TextFormat.GREEN + runtime.availableProcessors());
079
080
081        TextFormat playerColor = TextFormat.GREEN;
082        if (((float) server.getOnlinePlayers().size() / (float) server.getMaxPlayers()) > 0.85) {
083            playerColor = TextFormat.GOLD;
084        }
085
086        sender.sendMessage(TextFormat.GOLD + "Players: " + playerColor + server.getOnlinePlayers().size() + TextFormat.GREEN + " online, " +
087                TextFormat.RED + server.getMaxPlayers() + TextFormat.GREEN + " max. ");
088
089        for (Level level : server.getLevels().values()) {
090            sender.sendMessage(
091                    TextFormat.GOLD + "World \"" + level.getFolderName() + "\"" + (!Objects.equals(level.getFolderName(), level.getName()) ? " (" + level.getName() + ")" : "") + ": " +
092                            TextFormat.RED + level.getChunks().size() + TextFormat.GREEN + " chunks, " +
093                            TextFormat.RED + level.getEntities().length + TextFormat.GREEN + " entities, " +
094                            TextFormat.RED + level.getBlockEntities().size() + TextFormat.GREEN + " blockEntities." +
095                            " Time " + ((level.getTickRate() > 1 || level.getTickRateTime() > 40) ? TextFormat.RED : TextFormat.YELLOW) + NukkitMath.round(level.getTickRateTime(), 2) + "ms" +
096                            (level.getTickRate() > 1 ? " (tick rate " + (19 - level.getTickRate()) + ")" : "")
097            );
098        }
099
100        return true;
101    }
102
103    private static String formatUptime(long uptime) {
104        long days = TimeUnit.MILLISECONDS.toDays(uptime);
105        uptime -= TimeUnit.DAYS.toMillis(days);
106        long hours = TimeUnit.MILLISECONDS.toHours(uptime);
107        uptime -= TimeUnit.HOURS.toMillis(hours);
108        long minutes = TimeUnit.MILLISECONDS.toMinutes(uptime);
109        uptime -= TimeUnit.MINUTES.toMillis(minutes);
110        long seconds = TimeUnit.MILLISECONDS.toSeconds(uptime);
111        return String.format(UPTIME_FORMAT, days, hours, minutes, seconds);
112    }
113}
