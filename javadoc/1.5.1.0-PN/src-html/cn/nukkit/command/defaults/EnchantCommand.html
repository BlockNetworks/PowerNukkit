022    public EnchantCommand(String name) {
023        super(name, "%nukkit.command.enchant.description", "%commands.enchant.usage");
024        this.setPermission("nukkit.command.enchant");
025        this.commandParameters.clear();
026        this.commandParameters.put("default", new CommandParameter[]{
027                CommandParameter.newType("player", CommandParamType.TARGET),
028                CommandParameter.newType("enchantmentId", CommandParamType.INT),
029                CommandParameter.newType("level", true, CommandParamType.INT)
030        });
031        this.commandParameters.put("byName", new CommandParameter[]{
032                CommandParameter.newType("player", CommandParamType.TARGET),
033                CommandParameter.newEnum("enchantmentName", new CommandEnum("Enchant",
034                        "protection", "fire_protection", "feather_falling", "blast_protection", "projectile_projection", "thorns", "respiration",
035                        "aqua_affinity", "depth_strider", "sharpness", "smite", "bane_of_arthropods", "knockback", "fire_aspect", "looting", "efficiency",
036                        "silk_touch", "durability", "fortune", "power", "punch", "flame", "infinity", "luck_of_the_sea", "lure", "frost_walker", "mending",
037                        "binding_curse", "vanishing_curse", "impaling", "loyality", "riptide", "channeling", "multishot", "piercing", "quick_charge",
038                        "soul_speed")),
039                CommandParameter.newType("level", true, CommandParamType.INT)
040        });
041    }
042
043    @Override
044    public boolean execute(CommandSender sender, String commandLabel, String[] args) {
045        if (!this.testPermission(sender)) {
046            return true;
047        }
048        if (args.length < 2) {
049            sender.sendMessage(new TranslationContainer("commands.generic.usage", this.usageMessage));
050            return true;
051        }
052        Player player = sender.getServer().getPlayer(args[0]);
053        if (player == null) {
054            sender.sendMessage(new TranslationContainer(TextFormat.RED + "%commands.generic.player.notFound"));
055            return true;
056        }
057        int enchantId;
058        int enchantLevel;
059        try {
060            enchantId = getIdByName(args[1]);
061            enchantLevel = args.length == 3 ? Integer.parseInt(args[2]) : 1;
062        } catch (NumberFormatException e) {
063            sender.sendMessage(new TranslationContainer("commands.generic.usage", this.usageMessage));
064            return true;
065        }
066        Enchantment enchantment = Enchantment.getEnchantment(enchantId);
067        if (enchantment == null) {
068            sender.sendMessage(new TranslationContainer("commands.enchant.notFound", String.valueOf(enchantId)));
069            return true;
070        }
071        enchantment.setLevel(enchantLevel);
072        Item item = player.getInventory().getItemInHand();
073        if (item.getId() == 0) {
074            sender.sendMessage(new TranslationContainer("commands.enchant.noItem"));
075            return true;
076        }
077        if (item.getId() != ItemID.BOOK) {
078            item.addEnchantment(enchantment);
079            player.getInventory().setItemInHand(item);
080        } else {
081            Item enchanted = Item.get(ItemID.ENCHANTED_BOOK, 0, 1, item.getCompoundTag());
082            enchanted.addEnchantment(enchantment);
083            Item clone = item.clone();
084            clone.count--;
085            PlayerInventory inventory = player.getInventory();
086            inventory.setItemInHand(clone);
087            player.giveItem(enchanted);
088        }
089        Command.broadcastCommandMessage(sender, new TranslationContainer("%commands.enchant.success", args[1]));
090        return true;
091    }
