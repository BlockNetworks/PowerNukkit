110    private int red;
111    private int green;
112    private int blue;
113    private int alpha;
114
115    public BlockColor(int red, int green, int blue, int alpha) {
116        this.red = red;
117        this.green = green;
118        this.blue = blue;
119        this.alpha = alpha;
120    }
121
122    public BlockColor(int red, int green, int blue) {
123        this(red, green, blue, 0xff);
124    }
125
126    public BlockColor(int rgb) {
127        this.red = (rgb >> 16) & 0xff;
128        this.green = (rgb >> 8) & 0xff;
129        this.blue = rgb & 0xff;
130        this.alpha = 0xff;
131    }
132
133    @Override
134    public boolean equals(Object obj) {
135        if (!(obj instanceof BlockColor)) {
136            return false;
137        }
138        BlockColor other = (BlockColor) obj;
139        return this.red == other.red && this.green == other.green &&
140                this.blue == other.blue && this.alpha == other.alpha;
141    }
142
143    @Override
144    public String toString() {
145        return "BlockColor[r=" + this.red + ",g=" + this.green + ",b=" + this.blue + ",a=" + this.alpha + "]";
146    }
147
148    public int getRed() {
149        return this.red;
150    }
151
152    public int getGreen() {
153        return this.green;
154    }
155
156    public int getBlue() {
157        return this.blue;
158    }
159
160    public int getAlpha() {
161        return this.alpha;
162    }
163
164    public int getRGB() {
165        return (this.red << 16 | this.green << 8 | this.blue) & 0xffffff;
166    }
