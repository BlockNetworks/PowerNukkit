019package cn.nukkit.utils.collection;
020
021import cn.nukkit.api.PowerNukkitOnly;
022import cn.nukkit.api.Since;
023
024import java.util.AbstractMap;
025import java.util.Map;
026import java.util.Objects;
027import java.util.Set;
028import java.util.function.Function;
029
030/**
031 * @author joserobjr
032 * @since 2020-10-05
033 */
034@PowerNukkitOnly
035@Since("1.4.0.0-PN")
036public class ConvertingMapWrapper<K, V1, V2> extends AbstractMap<K, V1> {
037    private final Function<V1, V2> converter;
038    private final Function<V2, V1> reverseConverter;
039    private final Map<K, V2> proxied;
040    private final ConvertingSetWrapper<Entry<K, V1>, Entry<K, V2>> entrySet;
041    private final boolean convertReturnedNulls;
042
043    @PowerNukkitOnly
044    @Since("1.4.0.0-PN")
045    public ConvertingMapWrapper(Map<K, V2> proxied, Function<V1, V2> converter, Function<V2, V1> reverseConverter, boolean convertReturnedNulls) {
046        this.proxied = proxied;
047        this.converter = converter;
048        this.reverseConverter = reverseConverter;
049        this.convertReturnedNulls = convertReturnedNulls;
050        entrySet = new ConvertingSetWrapper<>(
051                proxied.entrySet(),
052                entry -> new EntryWrapper<>(entry, reverseConverter, converter),
053                entry -> new EntryWrapper<>(entry, converter, reverseConverter)
054        );
055    }
056
057    @PowerNukkitOnly
058    @Since("1.4.0.0-PN")
059    public ConvertingMapWrapper(Map<K, V2> proxied, Function<V1, V2> converter, Function<V2, V1> reverseConverter) {
060        this(proxied, converter, reverseConverter, false);
061    }
062    
063
064    @Override
065    public Set<Entry<K, V1>> entrySet() {
066        return entrySet;
067    }
068
069    @Override
070    public int size() {
071        return proxied.size();
072    }
073
074    @Override
075    public boolean isEmpty() {
076        return proxied.isEmpty();
077    }
078
079    @Override
080    public boolean containsValue(Object value) {
081        Function uncheckedConverter = converter;
082        Object converted = uncheckedConverter.apply(value);
083        return proxied.containsValue(converted);
084    }
085
086    @Override
087    public boolean containsKey(Object key) {
088        return proxied.containsKey(key);
089    }
090
091    @Override
092    public V1 get(Object key) {
093        V2 found = proxied.get(key);
094        if (found == null && !convertReturnedNulls) {
095            return null;
096        }
097        return reverseConverter.apply(found);
098    }
099
100    @Override
101    public V1 put(K key, V1 value) {
102        V2 removed = proxied.put(key, converter.apply(value));
103        if (removed == null && !convertReturnedNulls) {
104            return null;
105        }
106        return reverseConverter.apply(removed);
107    }
108
109    @Override
110    public V1 remove(Object key) {
111        V2 removed = proxied.remove(key);
112        if (removed == null && !convertReturnedNulls) {
113            return null;
114        }
115        return reverseConverter.apply(removed);
116    }
117
118    @Override
119    public boolean remove(Object key, Object value) {
120        Function uncheckedConverter = converter;
121        Object converted = uncheckedConverter.apply(value);
122        return proxied.remove(key, converted);
123    }
124
125    @Override
126    public void clear() {
127        proxied.clear();
128    }
129
130    @Override
131    public Set<K> keySet() {
132        return proxied.keySet();
133    }
134
135    private class EntryWrapper<E1, E2> implements Map.Entry<K, E1> {
136        private final Function<E1, E2> entryConverter;
137        private final Function<E2, E1> entryReverseConverter;
138        private final Map.Entry<K, E2> entryProxied;
139
140        public EntryWrapper(Entry<K, E2> entryProxied, Function<E1, E2> entryConverter, Function<E2, E1> entryReverseConverter) {
141            this.entryConverter = entryConverter;
142            this.entryReverseConverter = entryReverseConverter;
143            this.entryProxied = entryProxied;
144        }
145
146        @Override
147        public K getKey() {
148            return entryProxied.getKey();
149        }
150
151        @Override
152        public E1 getValue() {
153            E2 value = entryProxied.getValue();
154            if (value == null && !convertReturnedNulls) {
155                return null;
156            }
157            return entryReverseConverter.apply(value);
158        }
159
160        @Override
161        public E1 setValue(E1 value) {
162            E2 newValue = entryConverter.apply(value);
163            E2 oldValue = entryProxied.setValue(newValue);
164            if (oldValue == null && !convertReturnedNulls) {
165                return null;
166            }
167            return entryReverseConverter.apply(oldValue);
168        }
169
170        @Override
171        public String toString() {
172            return entryProxied.getKey()+"="+getValue();
173        }
174
175        @Override
176        public boolean equals(Object o) {
177            if (o == this) {
178                return true;
179            }
180            if (o instanceof Map.Entry) {
181                Map.Entry<?,?> e = (Map.Entry<?,?>)o;
182                return Objects.equals(entryProxied.getKey(), e.getKey()) && Objects.equals(getValue(), e.getValue());
183            }
184            return false;
185        }
186
187        @Override
188        public int hashCode() {
189            return Objects.hashCode(entryProxied.getKey()) ^ Objects.hashCode(getValue());
190        }
191    }
192}
