019package cn.nukkit.utils.collection;
020
021import cn.nukkit.api.PowerNukkitOnly;
022import cn.nukkit.api.Since;
023
024import javax.annotation.Nonnull;
025import java.util.AbstractSet;
026import java.util.Iterator;
027import java.util.Set;
028import java.util.function.Function;
029
030/**
031 * @author joserobjr
032 * @since 2020-10-05
033 */
034@PowerNukkitOnly
035@Since("1.4.0.0-PN")
036public class ConvertingSetWrapper<V1, V2> extends AbstractSet<V1> {
037    private final Function<V1, V2> converter;
038    private final Function<V2, V1> reverseConverter;
039    private final Set<V2> proxied;
040
041    @PowerNukkitOnly
042    @Since("1.4.0.0-PN")
043    public ConvertingSetWrapper(Set<V2> proxied, Function<V1, V2> converter, Function<V2, V1> reverseConverter) {
044        this.proxied = proxied;
045        this.converter = converter;
046        this.reverseConverter = reverseConverter;
047    }
048
049    @Override
050    @Nonnull
051    public Iterator<V1> iterator() {
052        return new ConvertingIterator();
053    }
054
055    @Override
056    public int size() {
057        return proxied.size();
058    }
059
060    @Override
061    public boolean isEmpty() {
062        return proxied.isEmpty();
063    }
064
065    @Override
066    public boolean contains(Object o) {
067        Function uncheckedConverter = converter;
068        Object converted = uncheckedConverter.apply(o);
069        return proxied.contains(converted);
070    }
071
072    @Override
073    public boolean add(V1 v1) {
074        return proxied.add(converter.apply(v1));
075    }
076
077    @Override
078    public boolean remove(Object o) {
079        Function uncheckedConverter = converter;
080        Object converted = uncheckedConverter.apply(o);
081        return proxied.remove(converted);
082    }
083
084    @Override
085    public void clear() {
086        proxied.clear();
087    }
088
089    private class ConvertingIterator implements Iterator<V1> {
090        private final Iterator<V2> proxiedIterator = proxied.iterator();
091
092        @Override
093        public void remove() {
094            proxiedIterator.remove();
095        }
096
097        @Override
098        public boolean hasNext() {
099            return proxiedIterator.hasNext();
100        }
101
102        @Override
103        public V1 next() {
104            return reverseConverter.apply(proxiedIterator.next());
105        }
106    }
107}
