001package cn.nukkit.utils;
002
003import java.lang.ref.Reference;
004import java.lang.reflect.Array;
005import java.lang.reflect.Field;
006import java.lang.reflect.Method;
007import java.util.Collection;
008import java.util.Collections;
009import java.util.Iterator;
010import java.util.concurrent.ConcurrentLinkedDeque;
011
012public abstract class IterableThreadLocal<T> extends ThreadLocal<T> implements Iterable<T> {
013    private ThreadLocal<T> flag;
014    private ConcurrentLinkedDeque<T> allValues = new ConcurrentLinkedDeque<>();
015
016    public IterableThreadLocal() {
017    }
018
019    @Override
020    protected final T initialValue() {
021        T value = init();
022        if (value != null) {
023            allValues.add(value);
024        }
025        return value;
026    }
027
028    @Override
029    public final Iterator<T> iterator() {
030        return getAll().iterator();
031    }
032
033    public T init() {
034        return null;
035    }
036
037    public void clean() {
038        IterableThreadLocal.clean(this);
039    }
040
041    public static void clean(ThreadLocal instance) {
042        try {
043            ThreadGroup rootGroup = Thread.currentThread( ).getThreadGroup( );
044            ThreadGroup parentGroup;
045            while ( ( parentGroup = rootGroup.getParent() ) != null ) {
046                rootGroup = parentGroup;
047            }
048            Thread[] threads = new Thread[ rootGroup.activeCount() ];
049            if (threads.length != 0) {
050                while (rootGroup.enumerate(threads, true) == threads.length) {
051                    threads = new Thread[threads.length * 2];
052                }
053            }
054            Field tl = Thread.class.getDeclaredField("threadLocals");
055            tl.setAccessible(true);
056            Method methodRemove = null;
057            for (Thread thread : threads) {
058                if (thread != null) {
059                    Object tlm = tl.get(thread);
060                    if (tlm != null) {
061                        if (methodRemove == null) {
062                            methodRemove = tlm.getClass().getDeclaredMethod("remove", ThreadLocal.class);
063                            methodRemove.setAccessible(true);
064                        }
065                        if (methodRemove != null) {
066                            try {
067                                methodRemove.invoke(tlm, instance);
068                            } catch (Throwable ignore) {}
069                        }
070                    }
071                }
072            }
073        } catch (Exception e) {
074            e.printStackTrace();
075        }
076    }
077
078    public static void cleanAll() {
079        try {
080            // Get a reference to the thread locals table of the current thread
081            Thread thread = Thread.currentThread();
082            Field threadLocalsField = Thread.class.getDeclaredField("threadLocals");
083            threadLocalsField.setAccessible(true);
084            Object threadLocalTable = threadLocalsField.get(thread);
085
086            // Get a reference to the array holding the thread local variables inside the
087            // ThreadLocalMap of the current thread
088            Class threadLocalMapClass = Class.forName("java.lang.ThreadLocal$ThreadLocalMap");
089            Field tableField = threadLocalMapClass.getDeclaredField("table");
090            tableField.setAccessible(true);
091            Object table = tableField.get(threadLocalTable);
092
093            // The key to the ThreadLocalMap is a WeakReference object. The referent field of this object
094            // is a reference to the actual ThreadLocal variable
095            Field referentField = Reference.class.getDeclaredField("referent");
096            referentField.setAccessible(true);
097
098            for (int i = 0; i < Array.getLength(table); i++) {
099                // Each entry in the table array of ThreadLocalMap is an Entry object
100                // representing the thread local reference and its value
101                Object entry = Array.get(table, i);
102                if (entry != null) {
103                    // Get a reference to the thread local object and remove it from the table
104                    ThreadLocal threadLocal = (ThreadLocal)referentField.get(entry);
105                    clean(threadLocal);
106                }
107            }
108        } catch(Exception e) {
109            // We will tolerate an exception here and just log it
110            throw new IllegalStateException(e);
111        }
112    }
113
114    public final Collection<T> getAll() {
115        return Collections.unmodifiableCollection(allValues);
116    }
117
118    @Override
119    protected void finalize() throws Throwable {
120        clean(this);
121        super.finalize();
122    }
123}
