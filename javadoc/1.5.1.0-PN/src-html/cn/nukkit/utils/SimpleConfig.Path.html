042    public boolean save(boolean async) {
043        if (configFile.exists()) try {
044            configFile.createNewFile();
045        } catch (Exception e) {
046            return false;
047        }
048        Config cfg = new Config(configFile, Config.YAML);
049        for (Field field : this.getClass().getDeclaredFields()) {
050            if (skipSave(field)) continue;
051            String path = getPath(field);
052            try {
053                if (path != null) cfg.set(path, field.get(this));
054            } catch (Exception e) {
055                return false;
056            }
057        }
058        cfg.save(async);
059        return true;
060    }

062    public boolean load() {
063        if (!this.configFile.exists()) return false;
064        Config cfg = new Config(configFile, Config.YAML);
065        for (Field field : this.getClass().getDeclaredFields()) {
066            if (field.getName().equals("configFile")) continue;
067            if (skipSave(field)) continue;
068            String path = getPath(field);
069            if (path == null) continue;
070            if (path.isEmpty()) continue;
071            field.setAccessible(true);
072            try {
073                if (field.getType() == int.class || field.getType() == Integer.class) field.set(this, cfg.getInt(path, field.getInt(this)));
074                 else if (field.getType() == boolean.class || field.getType() == Boolean.class) field.set(this, cfg.getBoolean(path, field.getBoolean(this)));
075                 else if (field.getType() == long.class || field.getType() == Long.class) field.set(this, cfg.getLong(path, field.getLong(this)));
076                 else if (field.getType() == double.class || field.getType() == Double.class) field.set(this, cfg.getDouble(path, field.getDouble(this)));
077                 else if (field.getType() == String.class) field.set(this, cfg.getString(path, (String) field.get(this)));
078                 else if (field.getType() == ConfigSection.class) field.set(this, cfg.getSection(path));
079                 else if (field.getType() == List.class) {
080                    Type genericFieldType = field.getGenericType();
081                    if (genericFieldType instanceof ParameterizedType) {
082                        ParameterizedType aType = (ParameterizedType) genericFieldType;
083                        Class fieldArgClass = (Class) aType.getActualTypeArguments()[0];
084                        if (fieldArgClass == Integer.class) field.set(this, cfg.getIntegerList(path));
085                         else if (fieldArgClass == Boolean.class) field.set(this, cfg.getBooleanList(path));
086                         else if (fieldArgClass == Double.class) field.set(this, cfg.getDoubleList(path));
087                         else if (fieldArgClass == Character.class) field.set(this, cfg.getCharacterList(path));
088                         else if (fieldArgClass == Byte.class) field.set(this, cfg.getByteList(path));
089                         else if (fieldArgClass == Float.class) field.set(this, cfg.getFloatList(path));
090                         else if (fieldArgClass == Short.class) field.set(this, cfg.getFloatList(path));
091                         else if (fieldArgClass == String.class) field.set(this, cfg.getStringList(path));
092                    } else field.set(this, cfg.getList(path)); // Hell knows what's kind of List was found :)
093                } else throw new IllegalStateException("SimpleConfig did not supports class: " + field.getType().getName() + " for config field " + configFile.getName());
094            } catch (Exception e) {
095                log.error("An error occurred while loading the config {}", configFile, e);
096                return false;
097            }
098        }
099        return true;
100    }

102    private String getPath(Field field) {
103        String path = null;
104        if (field.isAnnotationPresent(Path.class)) {
105            Path pathDefine = field.getAnnotation(Path.class);
106            path = pathDefine.value();
107        }
108        if (path == null || path.isEmpty()) path = field.getName().replaceAll("_", ".");
109        if (Modifier.isFinal(field.getModifiers())) return null;
110        if (Modifier.isPrivate(field.getModifiers())) field.setAccessible(true);
111        return path;
112    }
113
114    private boolean skipSave(Field field) {
115        if (!field.isAnnotationPresent(Skip.class)) return false;
116        return field.getAnnotation(Skip.class).skipSave();
117    }
118
119    private boolean skipLoad(Field field) {
120        if (!field.isAnnotationPresent(Skip.class)) return false;
121        return field.getAnnotation(Skip.class).skipLoad();
122    }
