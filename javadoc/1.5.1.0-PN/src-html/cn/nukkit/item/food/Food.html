134    //Opened API for plugins
135    public static Food registerFood(Food food, Plugin plugin) {
136        Objects.requireNonNull(food);
137        Objects.requireNonNull(plugin);
138        food.relativeIDs.forEach(n -> registryCustom.put(new NodeIDMetaPlugin(n.id, n.meta, plugin), food));
139        return food;
140    }
141
142    private static Food registerDefaultFood(Food food) {
143        food.relativeIDs.forEach(n -> registryDefault.put(n, food));
144        return food;
145    }
146
147    public static Food getByRelative(Item item) {
148        Objects.requireNonNull(item);
149        return getByRelative(item.getId(), item.getDamage());
150    }
151
152    public static Food getByRelative(Block block) {
153        Objects.requireNonNull(block);
154        return getByRelative(block.getId(), block.getDamage());
155    }
156
157    public static Food getByRelative(int relativeID, int meta) {
158        final Food[] result = {null};
159        registryCustom.forEach((n, f) -> {
160            if (n.id == relativeID && n.meta == meta && n.plugin.isEnabled()) result[0] = f;
161        });
162        if (result[0] == null) {
163            registryDefault.forEach((n, f) -> {
164                if (n.id == relativeID && n.meta == meta) result[0] = f;
165            });
166        }
167        return result[0];
168    }

170    protected int restoreFood = 0;
171    protected float restoreSaturation = 0;
172    protected final List<NodeIDMeta> relativeIDs = new ArrayList<>();
173
174    public final boolean eatenBy(Player player) {
175        PlayerEatFoodEvent event = new PlayerEatFoodEvent(player, this);
176        player.getServer().getPluginManager().callEvent(event);
177        if (event.isCancelled()) return false;
178        return event.getFood().onEatenBy(player);
179    }
180
181    protected boolean onEatenBy(Player player) {
182        player.getFoodData().addFoodLevel(this);
183        return true;
184    }
185
186    public Food addRelative(int relativeID) {
187        return addRelative(relativeID, 0);
188    }
189
190    public Food addRelative(int relativeID, int meta) {
191        NodeIDMeta node = new NodeIDMeta(relativeID, meta);
192        return addRelative(node);
193    }
194
195    private Food addRelative(NodeIDMeta node) {
196        if (!relativeIDs.contains(node)) relativeIDs.add(node);
197        return this;
198    }

235    static class NodeIDMeta {
236        final int id;
237        final int meta;
238
239        NodeIDMeta(int id, int meta) {
240            this.id = id;
241            this.meta = meta;
242        }
243    }

245    static class NodeIDMetaPlugin extends NodeIDMeta {
246        final Plugin plugin;
