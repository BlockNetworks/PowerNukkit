404    @SuppressWarnings("unchecked")
405    private static void initCreativeItems() {
406        try {
407            clearCreativeItems();
408            Config config = new Config(Config.JSON);
409            try (InputStream resourceAsStream = Server.class.getClassLoader().getResourceAsStream("creativeitems.json")) {
410                config.load(resourceAsStream);
411            }
412            List<Map> list = config.getMapList("items");
413            for (Map map : list) {
414                try {
415                    Item item = loadCreativeItemEntry(map);
416                    if (item != null) {
417                        addCreativeItem(item);
418                    }
419                } catch (Exception e) {
420                    log.error("Error while registering a creative item", e);
421                }
422            }
423        } catch (final IOException $ex) {
424            throw lombok.Lombok.sneakyThrow($ex);
425        }
426    }

428    private static Item loadCreativeItemEntry(Map<String, Object> data) {
429        String nbt = (String) data.get("nbt_b64");
430        byte[] nbtBytes = nbt != null ? Base64.getDecoder().decode(nbt) : EmptyArrays.EMPTY_BYTES;
431        String id = data.get("id").toString();
432        Item item = null;
433        if (data.containsKey("damage")) {
434            int meta = Utils.toInt(data.get("damage"));
435            item = fromString(id + ":" + meta);
436        } else if (data.containsKey("blockRuntimeId")) {
437            Integer blockId = BlockStateRegistry.getBlockId(id);
438            if (blockId == null || blockId > BlockID.QUARTZ_BRICKS) {
439                //TODO Remove this after the support is added
440                return null;
441            }
442            int blockRuntimeId = -1;
443            try {
444                blockRuntimeId = ((Number) data.get("blockRuntimeId")).intValue();
445                BlockState blockState = BlockStateRegistry.getBlockStateByRuntimeId(blockRuntimeId);
446                if (blockState != null) {
447                    item = blockState.asItemBlock();
448                } else {
449                    log.warn("Block state not found for the creative item {} with runtimeId {}", id, blockRuntimeId);
450                }
451            } catch (Throwable e) {
452                log.error("Error loading the creative item {} with runtimeId {}", id, blockRuntimeId, e);
453                return null;
454            }
455        }
456        if (item == null) {
457            item = fromString(id);
458        }
459        item.setCompoundTag(nbtBytes);
460        return item;
461    }

535    @PowerNukkitDifference(info = "Prevents players from getting invalid items by limiting the return to the maximum damage defined in Block.getMaxItemDamage()", since = "1.4.0.0-PN")
536    public static Item get(int id, Integer meta, int count, byte[] tags) {
537        try {
538            Class c = null;
539            if (id < 0) {
540                int blockId = 255 - id;
541                c = Block.list[blockId];
542            } else {
543                c = list[id];
544            }
545            Item item;
546            if (id < 256) {
547                int blockId = id < 0 ? 255 - id : id;
548                if (meta == 0) {
549                    item = new ItemBlock(Block.get(blockId), 0, count);
550                } else if (meta == -1) {
551                    // Special case for item instances used in fuzzy recipes
552                    item = new ItemBlock(Block.get(blockId), -1);
553                } else {
554                    BlockState state = BlockState.of(blockId, meta);
555                    try {
556                        state.validate();
557                        item = state.asItemBlock(count);
558                    } catch (InvalidBlockPropertyMetaException | InvalidBlockStateException e) {
559                        log.warn("Attempted to get an ItemBlock with invalid block state in memory: {}, trying to repair the block state...", state);
560                        log.catching(org.apache.logging.log4j.Level.DEBUG, e);
561                        Block repaired = state.getBlockRepairing(null, 0, 0, 0);
562                        item = repaired.asItemBlock(count);
563                        log.error("Attempted to get an illegal item block {}:{} ({}), the meta was changed to {}", id, meta, blockId, item.getDamage(), e);
564                    } catch (UnknownRuntimeIdException e) {
565                        log.warn("Attempted to get an illegal item block {}:{} ({}), the runtime id was unknown and the meta was changed to 0", id, meta, blockId, e);
566                        item = BlockState.of(id).asItemBlock(count);
567                    }
568                }
569            } else if (c == null) {
570                item = new Item(id, meta, count);
571            } else {
572                if (meta == -1) {
573                    item = ((Item) c.getConstructor(Integer.class, int.class).newInstance(0, count)).createFuzzyCraftingRecipe();
574                } else {
575                    item = ((Item) c.getConstructor(Integer.class, int.class).newInstance(meta, count));
576                }
577            }
578            if (tags.length != 0) {
579                item.setCompoundTag(tags);
580            }
581            return item;
582        } catch (Exception e) {
583            log.error("Error getting the item {}:{}{}! Returning an unsafe item stack!", id, meta, id < 0 ? " (" + (255 - id) + ")" : "", e);
584            return new Item(id, meta, count).setCompoundTag(tags);
585        }
586    }

588    @PowerNukkitDifference(since = "1.4.0.0-PN", info = "Improve namespaced name handling and allows to get custom blocks by name")
589    public static Item fromString(String str) {
590        String normalized = str.trim().replace(' ', '_').toLowerCase();
591        Matcher matcher = ITEM_STRING_PATTERN.matcher(normalized);
592        if (!matcher.matches()) {
593            return get(AIR);
594        }
595        String name = matcher.group(2);
596        OptionalInt meta = OptionalInt.empty();
597        String metaGroup;
598        if (name != null) {
599            metaGroup = matcher.group(3);
600        } else {
601            metaGroup = matcher.group(5);
602        }
603        if (metaGroup != null) {
604            meta = OptionalInt.of(Short.parseShort(metaGroup));
605        }
606        String numericIdGroup = matcher.group(4);
607        if (name != null) {
608            String namespaceGroup = matcher.group(1);
609            String namespacedId;
610            if (namespaceGroup != null) {
611                namespacedId = namespaceGroup + ":" + name;
612            } else {
613                namespacedId = "minecraft:" + name;
614            }
615            MinecraftItemID minecraftItemId = MinecraftItemID.getByNamespaceId(namespacedId);
616            if (minecraftItemId != null) {
617                Item item = minecraftItemId.get(1);
618                if (meta.isPresent()) {
619                    int damage = meta.getAsInt();
620                    if (damage < 0) {
621                        item = item.createFuzzyCraftingRecipe();
622                    } else {
623                        item.setDamage(damage);
624                    }
625                }
626                return item;
627            } else if (namespaceGroup != null && !namespaceGroup.equals("minecraft:")) {
628                return get(AIR);
629            }
630        } else if (numericIdGroup != null) {
631            int id = Integer.parseInt(numericIdGroup);
632            return get(id, meta.orElse(0));
633        }
634        if (name == null) {
635            return get(AIR);
636        }
637        int id = 0;
638        try {
639            id = ItemID.class.getField(name.toUpperCase()).getInt(null);
640        } catch (Exception ignore1) {
641            try {
642                id = BlockID.class.getField(name.toUpperCase()).getInt(null);
643                if (id > 255) {
644                    id = 255 - id;
645                }
646            } catch (Exception ignore2) {
647            }
648        }
649        return get(id, meta.orElse(0));
650    }

656    private static Item fromJson(Map<String, Object> data, boolean ignoreNegativeItemId) {
657        String nbt = (String) data.get("nbt_b64");
658        byte[] nbtBytes;
659        if (nbt != null) {
660            nbtBytes = Base64.getDecoder().decode(nbt);
661        } else {
662            // Support old format for backwards compat
663            nbt = (String) data.getOrDefault("nbt_hex", null);
664            if (nbt == null) {
665                nbtBytes = EmptyArrays.EMPTY_BYTES;
666            } else {
667                nbtBytes = Utils.parseHexBinary(nbt);
668            }
669        }
670        int id = Utils.toInt(data.get("id"));
671        if (ignoreNegativeItemId && id < 0) return null;
672        return get(id, Utils.toInt(data.getOrDefault("damage", 0)), Utils.toInt(data.getOrDefault("count", 1)), nbtBytes);
673    }

677    public static Item fromJsonNetworkId(Map<String, Object> data) {
678        String nbt = (String) data.get("nbt_b64");
679        byte[] nbtBytes;
680        if (nbt != null) {
681            nbtBytes = Base64.getDecoder().decode(nbt);
682        } else {
683            // Support old format for backwards compat
684            nbt = (String) data.getOrDefault("nbt_hex", null);
685            if (nbt == null) {
686                nbtBytes = EmptyArrays.EMPTY_BYTES;
687            } else {
688                nbtBytes = Utils.parseHexBinary(nbt);
689            }
690        }
691        int networkId = Utils.toInt(data.get("id"));
692        RuntimeItemMapping mapping = RuntimeItems.getRuntimeMapping();
693        int legacyFullId = mapping.getLegacyFullId(networkId);
694        int id = RuntimeItems.getId(legacyFullId);
695        OptionalInt meta = RuntimeItems.hasData(legacyFullId) ? OptionalInt.of(RuntimeItems.getData(legacyFullId)) : OptionalInt.empty();
696        if (data.containsKey("damage")) {
697            int jsonMeta = Utils.toInt(data.get("damage"));
698            if (jsonMeta != Short.MAX_VALUE) {
699                if (meta.isPresent() && jsonMeta != meta.getAsInt()) {
700                    throw new IllegalArgumentException("Conflicting damage value for " + mapping.getNamespacedIdByNetworkId(networkId) + ". From json: " + jsonMeta + ", from mapping: " + meta.getAsInt());
701                }
702                meta = OptionalInt.of(jsonMeta);
703            } else if (!meta.isPresent()) {
704                meta = OptionalInt.of(-1);
705            }
706        }
707        return get(id, meta.orElse(0), Utils.toInt(data.getOrDefault("count", 1)), nbtBytes);
708    }

850            tag = new CompoundTag();
851        } else {
852            tag = this.getNamedTag();

857            tag.putList(ench);

859            ench = tag.getList("ench", CompoundTag.class);
860        }
861        for (Enchantment enchantment : enchantments) {
862            boolean found = false;
863            for (int k = 0; k < ench.size(); k++) {
864                CompoundTag entry = ench.get(k);
865                if (entry.getShort("id") == enchantment.getId()) {
866                    ench.add(k, new CompoundTag().putShort("id", enchantment.getId()).putShort("lvl", enchantment.getLevel()));
867                    found = true;
868                    break;
869                }
870            }
871            if (!found) {
872                ench.add(new CompoundTag().putShort("id", enchantment.getId()).putShort("lvl", enchantment.getLevel()));
873            }
874        }
875        this.setNamedTag(tag);

1374    /**
1375     * Same as {@link #equals(Item, boolean)} but the enchantment order of the items does not affect the result.
1376     * @since 1.2.1.0-PN
1377     */
1378    public final boolean equalsIgnoringEnchantmentOrder(Item item, boolean checkDamage) {
1379        if (!this.equals(item, checkDamage, false)) {
1380            return false;
1381        }
1382        if (Arrays.equals(this.getCompoundTag(), item.getCompoundTag())) {
1383            return true;
1384        }
1385        if (!this.hasCompoundTag() || !item.hasCompoundTag()) {
1386            return false;
1387        }
1388        CompoundTag thisTags = this.getNamedTag();
1389        CompoundTag otherTags = item.getNamedTag();
1390        if (thisTags.equals(otherTags)) {
1391            return true;
1392        }
1393        if (!thisTags.contains("ench") || !otherTags.contains("ench") || !(thisTags.get("ench") instanceof ListTag) || !(otherTags.get("ench") instanceof ListTag) || thisTags.getList("ench").size() != otherTags.getList("ench").size()) {
1394            return false;
1395        }
1396        ListTag<CompoundTag> thisEnchantmentTags = thisTags.getList("ench", CompoundTag.class);
1397        ListTag<CompoundTag> otherEnchantmentTags = otherTags.getList("ench", CompoundTag.class);
1398        int size = thisEnchantmentTags.size();
1399        Int2IntMap enchantments = new Int2IntArrayMap(size);
1400        enchantments.defaultReturnValue(Integer.MIN_VALUE);
1401        for (int i = 0; i < size; i++) {
1402            CompoundTag tag = thisEnchantmentTags.get(i);
1403            enchantments.put(tag.getShort("id"), tag.getShort("lvl"));
1404        }
1405        for (int i = 0; i < size; i++) {
1406            CompoundTag tag = otherEnchantmentTags.get(i);
1407            if (enchantments.get(tag.getShort("id")) != tag.getShort("lvl")) {
1408                return false;
1409            }
1410        }
1411        return true;
1412    }
