001package cn.nukkit.item.enchantment;
002
003import cn.nukkit.api.Since;
004import cn.nukkit.block.BlockCarvedPumpkin;
005import cn.nukkit.item.*;
006
007import javax.annotation.Nonnull;
008
009/**
010 * @author MagicDroidX (Nukkit Project)
011 */
012public enum EnchantmentType {
013    ALL,
014    ARMOR,
015    ARMOR_HEAD,
016    ARMOR_TORSO,
017    ARMOR_LEGS,
018    ARMOR_FEET,
019    SWORD,
020    DIGGER,
021    FISHING_ROD,
022    BREAKABLE,
023    BOW,
024    WEARABLE,
025    TRIDENT,
026    @Since("1.4.0.0-PN") CROSSBOW;
027
028    public boolean canEnchantItem(@Nonnull Item item) {
029        if (this == ALL) {
030            return true;
031
032        } else if (this == BREAKABLE && item.getMaxDurability() >= 0) {
033            return true;
034
035        } else if (item instanceof ItemArmor) {
036            if (this == ARMOR || this == WEARABLE) {
037                return true;
038            }
039
040            switch (this) {
041                case ARMOR_HEAD:
042                    return item.isHelmet();
043                case ARMOR_TORSO:
044                    return item.isChestplate();
045                case ARMOR_LEGS:
046                    return item.isLeggings();
047                case ARMOR_FEET:
048                    return item.isBoots();
049                default:
050                    return false;
051            }
052
053        } else {
054            switch (this) {
055                case SWORD:
056                    return item.isSword() && !(item instanceof ItemTrident);
057                case DIGGER:
058                    return item.isPickaxe() || item.isShovel() || item.isAxe() || item.isHoe();
059                case BOW:
060                    return item instanceof ItemBow;
061                case FISHING_ROD:
062                    return item instanceof ItemFishingRod;
063                case WEARABLE:
064                    return item instanceof ItemSkull || item.getBlock() instanceof BlockCarvedPumpkin;
065                case TRIDENT:
066                    return item instanceof ItemTrident;
067                case CROSSBOW:
068                    return item instanceof ItemCrossbow;
069                default:
070                    return false;
071            }
072        }
073    }
074}
