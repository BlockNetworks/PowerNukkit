001package cn.nukkit.blockproperty;
002
003import cn.nukkit.api.PowerNukkitOnly;
004import cn.nukkit.api.Since;
005import cn.nukkit.blockproperty.exception.InvalidBlockPropertyMetaException;
006import cn.nukkit.blockproperty.exception.InvalidBlockPropertyPersistenceValueException;
007import cn.nukkit.blockproperty.exception.InvalidBlockPropertyValueException;
008import cn.nukkit.math.NukkitMath;
009import com.google.common.base.Preconditions;
010
011import javax.annotation.Nonnull;
012import javax.annotation.Nullable;
013
014@PowerNukkitOnly
015@Since("1.4.0.0-PN")
016public class IntBlockProperty extends BlockProperty<Integer> {
017    private static final long serialVersionUID = -2239010977496415152L;
018    
019    private final int minValue;
020    private final int maxValue;
021    
022    @PowerNukkitOnly
023    @Since("1.4.0.0-PN")
024    public IntBlockProperty(String name, boolean exportedToItem, int maxValue, int minValue, int bitSize, String persistenceName) {
025        super(name, exportedToItem, bitSize, persistenceName);
026        int delta = maxValue - minValue;
027        Preconditions.checkArgument(delta > 0, "maxValue must be higher than minValue. Got min:%s and max:%s", minValue, maxValue);
028        
029        int mask = -1 >>> (32 - bitSize);
030        Preconditions.checkArgument(delta <= mask, "The data range from %s to %s can't be stored in %s bits", minValue, maxValue, bitSize);
031
032        this.minValue = minValue;
033        this.maxValue = maxValue;
034    }
035
036    @PowerNukkitOnly
037    @Since("1.4.0.0-PN")
038    public IntBlockProperty(String name, boolean exportedToItem, int maxValue, int minValue, int bitSize) {
039        this(name, exportedToItem, maxValue, minValue, bitSize, name);
040    }
041
042    @PowerNukkitOnly
043    @Since("1.4.0.0-PN")
044    public IntBlockProperty(String name, boolean exportedToItem, int maxValue, int minValue) {
045        this(name, exportedToItem, maxValue, minValue, NukkitMath.bitLength(maxValue - minValue));
046    }
047
048    @PowerNukkitOnly
049    @Since("1.4.0.0-PN")
050    public IntBlockProperty(String name, boolean exportedToItem, int maxValue) {
051        this(name, exportedToItem, maxValue, 0);
052    }
053
054    @Since("1.4.0.0-PN")
055    @PowerNukkitOnly
056    @Override
057    public IntBlockProperty copy() {
058        return new IntBlockProperty(getName(), isExportedToItem(), getMaxValue(), getMinValue(), getBitSize(), getPersistenceName());
059    }
060
061    @Since("1.4.0.0-PN")
062    @PowerNukkitOnly
063    @Override
064    public IntBlockProperty exportingToItems(boolean exportedToItem) {
065        return new IntBlockProperty(getName(), exportedToItem, getMaxValue(), getMinValue(), getBitSize(), getPersistenceName());
066    }
067
068    @Override
069    public int getMetaForValue(@Nullable Integer value) {
070        if (value == null) {
071            return 0;
072        }
073        return getMetaForValue(value.intValue());
074    }
075
076    @PowerNukkitOnly
077    @Since("1.4.0.0-PN")
078    public int getMetaForValue(int value) {
079        try {
080            validateDirectly(value);
081        } catch (IllegalArgumentException e) {
082            throw new InvalidBlockPropertyValueException(this, null, value, e);
083        }
084        return value - minValue;
085    }
086
087    @Nonnull
088    @Override
089    public Integer getValueForMeta(int meta) {
090        return getIntValueForMeta(meta);
091    }
092
093    @Override
094    public int getIntValueForMeta(int meta) {
095        try {
096            validateMetaDirectly(meta);
097        } catch (IllegalArgumentException e) {
098            throw new InvalidBlockPropertyMetaException(this, meta, meta, e);
099        }
100        return minValue + meta;
101    }
102
103    @Override
104    public String getPersistenceValueForMeta(int meta) {
105        return String.valueOf(getIntValueForMeta(meta));
106    }
107
108    @Since("1.4.0.0-PN")
109    @PowerNukkitOnly
110    @Override
111    public int getMetaForPersistenceValue(@Nonnull String persistenceValue) {
112        try {
113            return getMetaForValue(Integer.parseInt(persistenceValue));
114        } catch (NumberFormatException | InvalidBlockPropertyValueException e) {
115            throw new InvalidBlockPropertyPersistenceValueException(this, null, persistenceValue, e);
116        }
117    }
118
119    @Override
120    protected void validateDirectly(@Nullable Integer value) {
121        if (value == null) {
122            return;
123        }
124        validateDirectly(value.intValue());
125    }
126    
127    private void validateDirectly(int newValue) {
128        Preconditions.checkArgument(newValue >= minValue, "New value (%s) must be higher or equals to %s", newValue, minValue);
129        Preconditions.checkArgument(maxValue >= newValue, "New value (%s) must be less or equals to %s", newValue, maxValue);
130    }
131
132    @Override
133    protected void validateMetaDirectly(int meta) {
134        int max = maxValue - minValue;
135        Preconditions.checkArgument(0 <= meta && meta <= max, "The meta %s is outside the range of 0 .. ", meta, max);
136    }
137
138    @PowerNukkitOnly
139    @Since("1.5.0.0-PN")
140    public int clamp(int value) {
141        return NukkitMath.clamp(value, getMinValue(), getMaxValue());
142    }
143
144    @PowerNukkitOnly
145    @Since("1.4.0.0-PN")
146    public int getMaxValue() {
147        return maxValue;
148    }
149
150    @PowerNukkitOnly
151    @Since("1.4.0.0-PN")
152    public int getMinValue() {
153        return minValue;
154    }
155
156    @PowerNukkitOnly
157    @Since("1.4.0.0-PN")
158    @Nonnull
159    public Integer getDefaultValue() {
160        return minValue;
161    }
162
163    @Since("1.4.0.0-PN")
164    @PowerNukkitOnly
165    @Override
166    public boolean isDefaultIntValue(int value) {
167        return minValue == value;
168    }
169
170    @Since("1.4.0.0-PN")
171    @PowerNukkitOnly
172    @Override
173    public int getDefaultIntValue() {
174        return minValue;
175    }
176
177    @Since("1.4.0.0-PN")
178    @PowerNukkitOnly
179    @Override
180    public boolean isDefaultValue(@Nullable Integer value) {
181        return value == null || minValue == value;
182    }
183
184    @Nonnull
185    @Override
186    public Class<Integer> getValueClass() {
187        return Integer.class;
188    }
189}
