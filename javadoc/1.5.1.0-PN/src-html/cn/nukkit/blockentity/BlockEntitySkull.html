001package cn.nukkit.blockentity;
002
003import cn.nukkit.api.PowerNukkitOnly;
004import cn.nukkit.api.Since;
005import cn.nukkit.block.Block;
006import cn.nukkit.level.format.FullChunk;
007import cn.nukkit.math.NukkitMath;
008import cn.nukkit.nbt.tag.CompoundTag;
009
010/**
011 * @author Snake1999
012 * @since 2016/2/3
013 */
014public class BlockEntitySkull extends BlockEntitySpawnable {
015    public BlockEntitySkull(FullChunk chunk, CompoundTag nbt) {
016        super(chunk, nbt);
017    }
018    
019    private boolean mouthMoving;
020    
021    private int mouthTickCount;
022
023    @Override
024    protected void initBlockEntity() {
025        if (!namedTag.contains("SkullType")) {
026            namedTag.putByte("SkullType", 0);
027        }
028        if (!namedTag.contains("Rot")) {
029            namedTag.putByte("Rot", 0);
030        }
031
032        if (namedTag.containsByte("MouthMoving")) {
033            mouthMoving = namedTag.getBoolean("MouthMoving");
034        }
035
036        if (namedTag.containsInt("MouthTickCount")) {
037            mouthTickCount = NukkitMath.clamp(namedTag.getInt("MouthTickCount"), 0, 60);
038        }
039
040        super.initBlockEntity();
041    }
042
043    @Override
044    public boolean onUpdate() {
045        if (isMouthMoving()) {
046            mouthTickCount++;
047            setDirty();
048            return true;
049        }
050        return false;
051    }
052
053    @PowerNukkitOnly
054    @Since("1.4.0.0-PN")
055    public void setMouthMoving(boolean mouthMoving) {
056        if (this.mouthMoving == mouthMoving) {
057            return;
058        }
059        this.mouthMoving = mouthMoving;
060        if (mouthMoving) {
061            scheduleUpdate();
062        }
063        this.level.updateComparatorOutputLevelSelective(this, true);
064        spawnToAll();
065        if (chunk != null) {
066            setDirty();
067        }
068    }
069
070    @Since("1.4.0.0-PN")
071    @PowerNukkitOnly
072    @Override
073    public boolean isObservable() {
074        return false;
075    }
076
077    @Override
078    public void setDirty() {
079        chunk.setChanged();
080    }
081
082    @PowerNukkitOnly
083    @Since("1.4.0.0-PN")
084    public boolean isMouthMoving() {
085        return mouthMoving;
086    }
087
088    @PowerNukkitOnly
089    @Since("1.4.0.0-PN")
090    public int getMouthTickCount() {
091        return mouthTickCount;
092    }
093
094    @PowerNukkitOnly
095    @Since("1.4.0.0-PN")
096    public void setMouthTickCount(int mouthTickCount) {
097        if (this.mouthTickCount == mouthTickCount) {
098            return;
099        }
100        this.mouthTickCount = mouthTickCount;
101        spawnToAll();
102    }
103
104    @Override
105    public void saveNBT() {
106        super.saveNBT();
107        this.namedTag
108                .putBoolean("MouthMoving", this.mouthMoving)
109                .putInt("MouthTickCount", mouthTickCount)
110                .remove("Creator");
111    }
112
113    @Override
114    public boolean isBlockEntityValid() {
115        return getBlock().getId() == Block.SKULL_BLOCK;
116    }
117
118    @Override
119    public CompoundTag getSpawnCompound() {
120        return new CompoundTag()
121                .putString("id", BlockEntity.SKULL)
122                .put("SkullType", this.namedTag.get("SkullType"))
123                .putInt("x", (int) this.x)
124                .putInt("y", (int) this.y)
125                .putInt("z", (int) this.z)
126                .put("Rot", this.namedTag.get("Rot"))
127                .putBoolean("MouthMoving", this.mouthMoving)
128                .putInt("MouthTickCount", mouthTickCount);
129    }
130
131}
