001package cn.nukkit.nbt.stream;
002
003import cn.nukkit.utils.VarInt;
004
005import java.io.DataOutput;
006import java.io.DataOutputStream;
007import java.io.IOException;
008import java.io.OutputStream;
009import java.nio.ByteOrder;
010import java.nio.charset.StandardCharsets;
011
012/**
013 * @author MagicDroidX (Nukkit Project)
014 */
015public class NBTOutputStream implements DataOutput, AutoCloseable {
016    private final DataOutputStream stream;
017    private final ByteOrder endianness;
018    private final boolean network;
019
020    public NBTOutputStream(OutputStream stream) {
021        this(stream, ByteOrder.BIG_ENDIAN);
022    }
023
024    public NBTOutputStream(OutputStream stream, ByteOrder endianness) {
025        this(stream, endianness, false);
026    }
027
028    public NBTOutputStream(OutputStream stream, ByteOrder endianness, boolean network) {
029        this.stream = stream instanceof DataOutputStream ? (DataOutputStream) stream : new DataOutputStream(stream);
030        this.endianness = endianness;
031        this.network = network;
032    }
033
034    public ByteOrder getEndianness() {
035        return endianness;
036    }
037
038    public boolean isNetwork() {
039        return network;
040    }
041
042    @Override
043    public void write(byte[] bytes) throws IOException {
044        this.stream.write(bytes);
045    }
046
047    @Override
048    public void write(byte[] b, int off, int len) throws IOException {
049        this.stream.write(b, off, len);
050    }
051
052    @Override
053    public void write(int b) throws IOException {
054        this.stream.write(b);
055    }
056
057    @Override
058    public void writeBoolean(boolean v) throws IOException {
059        this.stream.writeBoolean(v);
060    }
061
062    @Override
063    public void writeByte(int v) throws IOException {
064        this.stream.writeByte(v);
065    }
066
067    @Override
068    public void writeShort(int v) throws IOException {
069        if (endianness == ByteOrder.LITTLE_ENDIAN) {
070            v = Integer.reverseBytes(v) >> 16;
071        }
072        this.stream.writeShort(v);
073    }
074
075    @Override
076    public void writeChar(int v) throws IOException {
077        if (endianness == ByteOrder.LITTLE_ENDIAN) {
078            v = Character.reverseBytes((char) v);
079        }
080        this.stream.writeChar(v);
081    }
082
083    @Override
084    public void writeInt(int v) throws IOException {
085        if (network) {
086            VarInt.writeVarInt(this.stream, v);
087        } else {
088            if (endianness == ByteOrder.LITTLE_ENDIAN) {
089                v = Integer.reverseBytes(v);
090            }
091            this.stream.writeInt(v);
092        }
093    }
094
095    @Override
096    public void writeLong(long v) throws IOException {
097        if (network) {
098            VarInt.writeVarLong(this.stream, v);
099        } else {
100            if (endianness == ByteOrder.LITTLE_ENDIAN) {
101                v = Long.reverseBytes(v);
102            }
103            this.stream.writeLong(v);
104        }
105    }
106
107    @Override
108    public void writeFloat(float v) throws IOException {
109        int i = Float.floatToIntBits(v);
110        if (endianness == ByteOrder.LITTLE_ENDIAN) {
111            i = Integer.reverseBytes(i);
112        }
113        this.stream.writeInt(i);
114    }
115
116    @Override
117    public void writeDouble(double v) throws IOException {
118        long l = Double.doubleToLongBits(v);
119        if (endianness == ByteOrder.LITTLE_ENDIAN) {
120            l = Long.reverseBytes(l);
121        }
122        this.stream.writeLong(l);
123    }
124
125    @Override
126    public void writeBytes(String s) throws IOException {
127        this.stream.writeBytes(s);
128    }
129
130    @Override
131    public void writeChars(String s) throws IOException {
132        this.stream.writeChars(s);
133    }
134
135    @Override
136    public void writeUTF(String s) throws IOException {
137        byte[] bytes = s.getBytes(StandardCharsets.UTF_8);
138        if (network) {
139            VarInt.writeUnsignedVarInt(stream, bytes.length);
140        } else {
141            this.writeShort(bytes.length);
142        }
143        this.stream.write(bytes);
144    }
145
146    @Override
147    public void close() throws IOException {
148        this.stream.close();
149    }
150}
