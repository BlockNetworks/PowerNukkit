131    public static Effect getEffect(int id) {
132        if (id >= 0 && id < effects.length && effects[id] != null) {
133            return effects[id].clone();
134        } else {
135            throw new ServerException("Effect id: " + id + " not found");
136        }
137    }

139    public static Effect getEffectByName(String name) {
140        name = name.trim().replace(' ', '_').replace("minecraft:", "");
141        try {
142            int id = Effect.class.getField(name.toUpperCase()).getInt(null);
143            return getEffect(id);
144        } catch (Exception e) {
145            throw new RuntimeException(e);
146        }
147    }

226    public boolean canTick() {
227        int interval;
228        switch (this.id) {
229            case Effect.POISON: //POISON
230            case Effect.FATAL_POISON:
231                if ((interval = (25 >> this.amplifier)) > 0) {
232                    return (this.duration % interval) == 0;
233                }
234                return true;
235            case Effect.WITHER: //WITHER
236                if ((interval = (50 >> this.amplifier)) > 0) {
237                    return (this.duration % interval) == 0;
238                }
239                return true;
240            case Effect.REGENERATION: //REGENERATION
241                if ((interval = (40 >> this.amplifier)) > 0) {
242                    return (this.duration % interval) == 0;
243                }
244                return true;
245        }
246        return false;
247    }

249    public void applyEffect(Entity entity) {
250        switch (this.id) {
251            case Effect.POISON: //POISON
252            case Effect.FATAL_POISON:
253                if (entity.getHealth() > 1 || this.id == FATAL_POISON) {
254                    entity.attack(new EntityDamageEvent(entity, DamageCause.MAGIC, 1));
255                }
256                break;
257            case Effect.WITHER: //WITHER
258                entity.attack(new EntityDamageEvent(entity, DamageCause.MAGIC, 1));
259                break;
260            case Effect.REGENERATION: //REGENERATION
261                if (entity.getHealth() < entity.getMaxHealth()) {
262                    entity.heal(new EntityRegainHealthEvent(entity, 1, EntityRegainHealthEvent.CAUSE_MAGIC));
263                }
264                break;
265        }
266    }

276    public void add(Entity entity) {
277        Effect oldEffect = entity.getEffect(getId());
278        if (oldEffect != null && (Math.abs(this.getAmplifier()) < Math.abs(oldEffect.getAmplifier()) ||
279                Math.abs(this.getAmplifier()) == Math.abs(oldEffect.getAmplifier())
280                        && this.getDuration() < oldEffect.getDuration())) {
281            return;
282        }
283        if (entity instanceof Player) {
284            Player player = (Player) entity;
285
286            MobEffectPacket pk = new MobEffectPacket();
287            pk.eid = entity.getId();
288            pk.effectId = this.getId();
289            pk.amplifier = this.getAmplifier();
290            pk.particles = this.isVisible();
291            pk.duration = this.getDuration();
292            if (oldEffect != null) {
293                pk.eventId = MobEffectPacket.EVENT_MODIFY;
294            } else {
295                pk.eventId = MobEffectPacket.EVENT_ADD;
296            }
297
298            player.dataPacket(pk);
299
300            if (this.id == Effect.SPEED && (oldEffect == null || oldEffect.amplifier != this.amplifier)) {
301                if (oldEffect != null) {
302                    player.setMovementSpeed(player.getMovementSpeed() / (1 + 0.2f * (oldEffect.amplifier + 1)), false);
303                }
304                player.setMovementSpeed(player.getMovementSpeed() * (1 + 0.2f * (this.amplifier + 1)));
305            }
306
307            if (this.id == Effect.SLOWNESS && (oldEffect == null || oldEffect.amplifier != this.amplifier)) {
308                if (oldEffect != null) {
309                    player.setMovementSpeed(player.getMovementSpeed() / (1 - 0.15f * (oldEffect.amplifier + 1)), false);
310                }
311                player.setMovementSpeed(player.getMovementSpeed() * (1 - 0.15f * (this.amplifier + 1)));
312            }
313        }
314
315        if (this.id == Effect.INVISIBILITY) {
316            entity.setDataFlag(Entity.DATA_FLAGS, Entity.DATA_FLAG_INVISIBLE, true);
317            entity.setNameTagVisible(false);
318        }
319
320        if (this.id == Effect.ABSORPTION) {
321            int add = (this.amplifier + 1) * 4;
322            if (add > entity.getAbsorption()) entity.setAbsorption(add);
323        }
324    }

326    public void remove(Entity entity) {
327        if (entity instanceof Player) {
328            MobEffectPacket pk = new MobEffectPacket();
329            pk.eid = entity.getId();
330            pk.effectId = this.getId();
331            pk.eventId = MobEffectPacket.EVENT_REMOVE;
332
333            ((Player) entity).dataPacket(pk);
334
335            if (this.id == Effect.SPEED) {
336                ((Player) entity).setMovementSpeed(((Player) entity).getMovementSpeed() / (1 + 0.2f * (this.amplifier + 1)));
337            }
338            if (this.id == Effect.SLOWNESS) {
339                ((Player) entity).setMovementSpeed(((Player) entity).getMovementSpeed() / (1 - 0.15f * (this.amplifier + 1)));
340            }
341            if (this.id == Effect.HEALTH_BOOST) {
342                float max = entity.getMaxHealth();
343                float health = Math.min(entity.getHealth(), max);
344                entity.setHealth(health);
345            }
346        }
347
348        if (this.id == Effect.INVISIBILITY) {
349            entity.setDataFlag(Entity.DATA_FLAGS, Entity.DATA_FLAG_INVISIBLE, false);
350            entity.setNameTagVisible(true);
351        }
352
353        if (this.id == Effect.ABSORPTION) {
354            entity.setAbsorption(0);
355        }
356    }
