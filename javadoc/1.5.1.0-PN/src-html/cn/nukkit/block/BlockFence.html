024@PowerNukkitDifference(info = "Implements BlockConnectable only on PowerNukkit", since = "1.3.0.0-PN")
025public class BlockFence extends BlockTransparentMeta implements BlockConnectable {
026    public static final BlockProperties PROPERTIES = new BlockProperties(WoodType.PROPERTY);
027
028    @Deprecated @DeprecationDetails(reason = "Moved to the block property system", since = "1.4.0.0-PN", replaceWith = "getWoodType()")
029    public static final int FENCE_OAK = 0;
030    @Deprecated @DeprecationDetails(reason = "Moved to the block property system", since = "1.4.0.0-PN", replaceWith = "getWoodType()")

032    @Deprecated @DeprecationDetails(reason = "Moved to the block property system", since = "1.4.0.0-PN", replaceWith = "getWoodType()")
033    public static final int FENCE_BIRCH = 2;
034    @Deprecated @DeprecationDetails(reason = "Moved to the block property system", since = "1.4.0.0-PN", replaceWith = "getWoodType()")
035    public static final int FENCE_JUNGLE = 3;
036    @Deprecated @DeprecationDetails(reason = "Moved to the block property system", since = "1.4.0.0-PN", replaceWith = "getWoodType()")
037    public static final int FENCE_ACACIA = 4;
038    @Deprecated @DeprecationDetails(reason = "Moved to the block property system", since = "1.4.0.0-PN", replaceWith = "getWoodType()")
039    public static final int FENCE_DARK_OAK = 5;
040
041    public BlockFence() {
042        this(0);
043    }
044
045    public BlockFence(int meta) {
046        super(meta);
047    }
048
049    @Override
050    public int getId() {
051        return FENCE;
052    }
053
054    @Since("1.4.0.0-PN")
055    @PowerNukkitOnly
056    @Nonnull
057    @Override
058    public BlockProperties getProperties() {
059        return PROPERTIES;
060    }
061
062    @Override
063    public double getHardness() {
064        return 2;
065    }
066
067    @PowerNukkitOnly
068    @Override
069    public int getWaterloggingLevel() {
070        return 1;
071    }
072
073    @Override
074    public double getResistance() {
075        return 3;
076    }
077
078    @Override
079    public int getToolType() {
080        return ItemTool.TYPE_AXE;
081    }
082    
083    @PowerNukkitOnly
084    @Since("1.4.0.0-PN")
085    public void setWoodType(@Nullable WoodType woodType) {
086        setPropertyValue(WoodType.PROPERTY, woodType);
087    }
088
089    @PowerNukkitOnly
090    @Since("1.4.0.0-PN")
091    public Optional<WoodType> getWoodType() {
092        return Optional.of(getPropertyValue(WoodType.PROPERTY));
093    }
094
095    @Override
096    public String getName() {
097        return getPropertyValue(WoodType.PROPERTY).getEnglishName() + " Fence";
098    }
099
100    @Override
101    protected AxisAlignedBB recalculateBoundingBox() {
102        boolean north = this.canConnect(this.north());
103        boolean south = this.canConnect(this.south());
104        boolean west = this.canConnect(this.west());
105        boolean east = this.canConnect(this.east());
106        double n = north ? 0 : 0.375;
107        double s = south ? 1 : 0.625;
108        double w = west ? 0 : 0.375;
109        double e = east ? 1 : 0.625;
110        return new SimpleAxisAlignedBB(
111                this.x + w,
112                this.y,
113                this.z + n,
114                this.x + e,
115                this.y + 1.5,
116                this.z + s
117        );
118    }
119
120    @Override
121    public int getBurnChance() {
122        return 5;
123    }
124
125    @Override
126    public int getBurnAbility() {
127        return 20;
128    }
129
130    @Override
131    public boolean canConnect(Block block) {
132        if (block instanceof BlockFence) {
133            if (block.getId() == NETHER_BRICK_FENCE || this.getId() == NETHER_BRICK_FENCE) {
134                return block.getId() == this.getId();
135            }
136            return true;
137        }
138        if (block instanceof BlockTrapdoor) {
139            BlockTrapdoor trapdoor = (BlockTrapdoor) block;
140            return trapdoor.isOpen() && trapdoor.getBlockFace() == calculateFace(this, trapdoor);
141        }
142        return block instanceof BlockFenceGate || block.isSolid() && !block.isTransparent();
143    }
144
145    @Override
146    public BlockColor getColor() {
147        return getPropertyValue(WoodType.PROPERTY).getColor();
148    }
149}
