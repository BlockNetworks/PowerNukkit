001package cn.nukkit.block;
002
003import cn.nukkit.Player;
004import cn.nukkit.api.PowerNukkitOnly;
005import cn.nukkit.api.Since;
006import cn.nukkit.blockentity.BlockEntity;
007import cn.nukkit.blockentity.BlockEntityBarrel;
008import cn.nukkit.blockproperty.BlockProperties;
009import cn.nukkit.inventory.ContainerInventory;
010import cn.nukkit.item.Item;
011import cn.nukkit.item.ItemBlock;
012import cn.nukkit.item.ItemTool;
013import cn.nukkit.math.BlockFace;
014import cn.nukkit.nbt.tag.CompoundTag;
015import cn.nukkit.nbt.tag.ListTag;
016import cn.nukkit.nbt.tag.StringTag;
017import cn.nukkit.nbt.tag.Tag;
018import cn.nukkit.utils.BlockColor;
019import cn.nukkit.utils.Faceable;
020
021import javax.annotation.Nonnull;
022import java.util.Map;
023
024import static cn.nukkit.blockproperty.CommonBlockProperties.*;
025
026@PowerNukkitOnly
027public class BlockBarrel extends BlockSolidMeta implements Faceable, BlockEntityHolder<BlockEntityBarrel> {
028
029    @PowerNukkitOnly
030    @Since("1.5.0.0-PN")
031    public static final BlockProperties PROPERTIES = new BlockProperties(FACING_DIRECTION, OPEN);
032
033    @PowerNukkitOnly
034    public BlockBarrel() {
035        this(0);
036    }
037
038    @PowerNukkitOnly
039    public BlockBarrel(int meta) {
040        super(meta);
041    }
042
043    @Override
044    public String getName() {
045        return "Barrel";
046    }
047
048    @Override
049    public int getId() {
050        return BARREL;
051    }
052
053    @Since("1.4.0.0-PN")
054    @PowerNukkitOnly
055    @Nonnull
056    @Override
057    public BlockProperties getProperties() {
058        return PROPERTIES;
059    }
060
061    @PowerNukkitOnly
062    @Since("1.4.0.0-PN")
063    @Nonnull
064    @Override
065    public String getBlockEntityType() {
066        return BlockEntity.BARREL;
067    }
068
069    @Since("1.4.0.0-PN")
070    @PowerNukkitOnly
071    @Nonnull
072    @Override
073    public Class<? extends BlockEntityBarrel> getBlockEntityClass() {
074        return BlockEntityBarrel.class;
075    }
076
077    @Override
078    public boolean place(@Nonnull Item item, @Nonnull Block block, @Nonnull Block target, @Nonnull BlockFace face, double fx, double fy, double fz, Player player) {
079        if (Math.abs(player.x - this.x) < 2 && Math.abs(player.z - this.z) < 2) {
080            double y = player.y + player.getEyeHeight();
081
082            if (y - this.y > 2) {
083                this.setDamage(BlockFace.UP.getIndex());
084            } else if (this.y - y > 0) {
085                this.setDamage(BlockFace.DOWN.getIndex());
086            } else {
087                this.setDamage(player.getHorizontalFacing().getOpposite().getIndex());
088            }
089        } else {
090            this.setDamage(player.getHorizontalFacing().getOpposite().getIndex());
091        }
092
093        CompoundTag nbt = new CompoundTag().putList(new ListTag<>("Items"));
094
095        if (item.hasCustomName()) {
096            nbt.putString("CustomName", item.getCustomName());
097        }
098
099        if (item.hasCustomBlockData()) {
100            Map<String, Tag> customData = item.getCustomBlockData().getTags();
101            for (Map.Entry<String, Tag> tag : customData.entrySet()) {
102                nbt.put(tag.getKey(), tag.getValue());
103            }
104        }
105        
106        return BlockEntityHolder.setBlockAndCreateEntity(this, true, true, nbt) != null;
107    }
108
109    @Override
110    public boolean onActivate(@Nonnull Item item, Player player) {
111        if (player == null) {
112            return false;
113        }
114
115        BlockEntityBarrel barrel = getOrCreateBlockEntity();
116
117        if (barrel.namedTag.contains("Lock") && barrel.namedTag.get("Lock") instanceof StringTag 
118                && !barrel.namedTag.getString("Lock").equals(item.getCustomName())) {
119            return false;
120        }
121
122        player.addWindow(barrel.getInventory());
123        return true;
124    }
125
126    @Override
127    public boolean canBeActivated() {
128        return true;
129    }
130
131    @Override
132    public double getHardness() {
133        return 2.5;
134    }
135
136    @Override
137    public double getResistance() {
138        return 12.5;
139    }
140
141    @Override
142    public int getToolType() {
143        return ItemTool.TYPE_AXE;
144    }
145
146    @Override
147    public BlockColor getColor() {
148        return BlockColor.WOOD_BLOCK_COLOR;
149    }
150
151    @Override
152    public Item toItem() {
153        return new ItemBlock(new BlockBarrel());
154    }
155
156    @Override
157    public BlockFace getBlockFace() {
158        int index = getDamage() & 0x7;
159        return BlockFace.fromIndex(index);
160    }
161
162    @PowerNukkitOnly
163    @Since("1.3.0.0-PN")
164    @Override
165    public void setBlockFace(BlockFace face) {
166        setDamage((getDamage() & 0x8) | (face.getIndex() & 0x7));
167    }
168
169    @PowerNukkitOnly
170    public boolean isOpen() {
171        return (getDamage() & 0x8) == 0x8;
172    }
173
174    @PowerNukkitOnly
175    public void setOpen(boolean open) {
176        setDamage((getDamage() & 0x7) | (open? 0x8 : 0x0));
177    }
178
179    @Override
180    public boolean hasComparatorInputOverride() {
181        return true;
182    }
183
184    @Override
185    public int getComparatorInputOverride() {
186        BlockEntityBarrel blockEntity = getBlockEntity();
187
188        if (blockEntity != null) {
189            return ContainerInventory.calculateRedstone(blockEntity.getInventory());
190        }
191
192        return super.getComparatorInputOverride();
193    }
194}
