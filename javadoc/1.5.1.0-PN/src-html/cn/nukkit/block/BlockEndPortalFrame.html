133    @Since("1.3.0.0-PN")
134    public void createPortal() {
135        Vector3 centerSpot = this.searchCenter(new ArrayList<>());
136        if(centerSpot != null) {
137            for(int x = -2; x <= 2; x++) {
138                for(int z = -2; z <= 2; z++) {
139                    if((x == -2 || x == 2) && (z == -2 || z == 2))
140                        continue;
141                    if(x == -2 || x == 2 || z == -2 || z == 2) {
142                        if(!this.checkFrame(this.getLevel().getBlock(centerSpot.add(x, 0, z)), x, z)) {
143                            return;
144                        }
145                    }
146                }
147            }
148
149            for(int x = -1; x <= 1; x++) {
150                for(int z = -1; z <= 1; z++) {
151                    Vector3 vector3 = centerSpot.add(x, 0, z);
152                    if(this.getLevel().getBlock(vector3).getId() != Block.AIR) {
153                        this.getLevel().useBreakOn(vector3);
154                    }
155                    this.getLevel().setBlock(vector3, Block.get(Block.END_PORTAL));
156                }
157            }
158        }
159    }
160
161    private Vector3 searchCenter(List<Block> visited) {
162        for(int x = -2; x <= 2; x++) {
163            if(x == 0)
164                continue;
165            Block block = this.getLevel().getBlock(this.add(x, 0, 0));
166            Block iBlock = this.getLevel().getBlock(this.add(x * 2, 0, 0));
167            if(this.checkFrame(block) && !visited.contains(block)) {
168                visited.add(block);
169                if((x == -1 || x == 1) && this.checkFrame(iBlock))
170                    return ((BlockEndPortalFrame) block).searchCenter(visited);
171                for(int z = -4; z <= 4; z++) {
172                    if(z == 0)
173                        continue;
174                    block = this.getLevel().getBlock(this.add(x, 0, z));
175                    if(this.checkFrame(block)) {
176                        return this.add(x / 2, 0, z / 2);
177                    }
178                }
179            }
180        }
181        for(int z = -2; z <= 2; z++) {
182            if(z == 0)
183                continue;
184            Block block = this.getLevel().getBlock(this.add(0, 0, z));
185            Block iBlock = this.getLevel().getBlock(this.add(0, 0, z * 2));
186            if(this.checkFrame(block) && !visited.contains(block)) {
187                visited.add(block);
188                if((z == -1 || z == 1) && this.checkFrame(iBlock))
189                    return ((BlockEndPortalFrame) block).searchCenter(visited);
190                for(int x = -4; x <= 4; x++) {
191                    if(x == 0)
192                        continue;
193                    block = this.getLevel().getBlock(this.add(x, 0, z));
194                    if(this.checkFrame(block)) {
195                        return this.add(x / 2, 0, z / 2);
196                    }
197                }
198            }
199        }
200        return null;
201    }
202
203    private boolean checkFrame(Block block) {
204        return block.getId() == this.getId() && (block.getDamage() & 4) == 4;
205    }
206
207    private boolean checkFrame(Block block, int x, int z) {
208        return block.getId() == this.getId() && (block.getDamage() - 4) == (x == -2 ? 3 : x == 2 ? 1 : z == -2 ? 0 : z == 2 ? 2 : -1);
209    }
