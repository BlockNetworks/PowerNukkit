075    @PowerNukkitOnly

077        return (getDamage() & 0x8) > 0;
078    }
079
080    @PowerNukkitOnly
081    public void setStabilityCheck(boolean check) {
082        if (check) {
083            setDamage(getDamage() | 0x8);
084        } else {
085            setDamage(getDamage() & 0x7);
086        }
087    }

092    }

094    @Override
095    public boolean place(@Nonnull Item item, @Nonnull Block block, @Nonnull Block target, @Nonnull BlockFace face, double fx, double fy, double fz, Player player) {
096        if (block instanceof BlockLava) {
097            return false;
098        }
099
100        Block down = down();
101        if (target.getId() != SCAFFOLDING && down.getId() != SCAFFOLDING && down.getId() != AIR && !down.isSolid()) {
102            boolean scaffoldOnSide = false;
103            for (int i = 0; i < 4; i++) {
104                BlockFace sideFace = BlockFace.fromHorizontalIndex(i);
105                if (sideFace != face) {
106                    Block side = getSide(sideFace);
107                    if (side.getId() == SCAFFOLDING) {
108                        scaffoldOnSide = true;
109                        break;
110                    }
111                }
112            }
113            if (!scaffoldOnSide) {
114                return false;
115            }
116        }
117
118        setDamage(0x8);
119        this.getLevel().setBlock(this, this, true, true);
120        return true;
121    }
122
123    @Override
124    public int onUpdate(int type) {
125        if (type == Level.BLOCK_UPDATE_NORMAL) {
126            Block down = down();
127            if (down.isSolid()) {
128                if (getDamage() != 0) {
129                    setDamage(0);
130                    this.getLevel().setBlock(this, this, true, true);
131                }
132                return type;
133            }
134
135            int stability = 7;
136            for (BlockFace face : BlockFace.values()) {
137                if (face == BlockFace.UP) {
138                    continue;
139                }
140
141                Block otherBlock = getSide(face);
142                if (otherBlock.getId() == SCAFFOLDING) {
143                    BlockScaffolding other = (BlockScaffolding) otherBlock;
144                    int otherStability = other.getStability();
145                    if (otherStability < stability) {
146                        if (face == BlockFace.DOWN) {
147                            stability = otherStability;
148                        } else {
149                            stability = otherStability + 1;
150                        }
151                    }
152                }
153            }
154
155            if (stability >= 7) {
156                if (getStabilityCheck()) {
157                    super.onUpdate(type);
158                } else {
159                    this.getLevel().scheduleUpdate(this, 0);
160                }
161                return type;
162            }
163
164            setStabilityCheck(false);
165            setStability(stability);
166            this.getLevel().setBlock(this, this, true, true);
167            return type;
168        } else if (type == Level.BLOCK_UPDATE_SCHEDULED) {
169            this.getLevel().useBreakOn(this);
170            return type;
171        }
172
173        return 0;
174    }
