001package cn.nukkit.block;
002
003import cn.nukkit.api.PowerNukkitOnly;
004import cn.nukkit.api.Since;
005import cn.nukkit.blockproperty.BlockProperties;
006import cn.nukkit.blockproperty.BooleanBlockProperty;
007import cn.nukkit.blockproperty.value.WoodType;
008
009import javax.annotation.Nonnull;
010
011import static cn.nukkit.blockproperty.CommonBlockProperties.PILLAR_AXIS;
012
013@PowerNukkitOnly
014public class BlockWoodBark extends BlockWood {
015    @PowerNukkitOnly
016    @Since("1.4.0.0-PN")
017    public static final String STRIPPED_BIT = "stripped_bit";
018
019    @PowerNukkitOnly
020    @Since("1.4.0.0-PN")
021    public static final BlockProperties PROPERTIES = new BlockProperties(
022            WoodType.PROPERTY,
023            new BooleanBlockProperty(STRIPPED_BIT, true),
024            PILLAR_AXIS
025    );
026
027    @PowerNukkitOnly
028    @Since("1.4.0.0-PN")
029    public BlockWoodBark() {
030        this(0);
031    }
032
033    @PowerNukkitOnly
034    @Since("1.4.0.0-PN")
035    public BlockWoodBark(int meta) {
036        super(meta);
037    }
038    
039    @Override
040    public int getId() {
041        return WOOD_BARK;
042    }
043
044    @Since("1.4.0.0-PN")
045    @PowerNukkitOnly
046    @Nonnull
047    @Override
048    public BlockProperties getProperties() {
049        return PROPERTIES;
050    }
051
052    @Override
053    public String getName() {
054        return (isStripped()? "Stripped ": "") + getWoodType().getEnglishName() + " Wood";
055    }
056
057    @Override
058    public WoodType getWoodType() {
059        return getPropertyValue(WoodType.PROPERTY);
060    }
061
062    @Override
063    public void setWoodType(WoodType woodType) {
064        setPropertyValue(WoodType.PROPERTY, woodType);
065    }
066
067    @PowerNukkitOnly
068    @Since("1.4.0.0-PN")
069    public boolean isStripped() {
070        return getBooleanValue(STRIPPED_BIT);
071    }
072
073    @PowerNukkitOnly
074    @Since("1.4.0.0-PN")
075    public void setStripped(boolean stripped) {
076        setBooleanValue(STRIPPED_BIT, stripped);
077    }
078}
