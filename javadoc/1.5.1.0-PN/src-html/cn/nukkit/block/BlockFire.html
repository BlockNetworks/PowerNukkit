094    @Override
095    public void onEntityCollide(Entity entity) {
096        if (!entity.hasEffect(Effect.FIRE_RESISTANCE)) {
097            entity.attack(new EntityDamageByBlockEvent(this, entity, DamageCause.FIRE, 1));
098        }
099
100        EntityCombustByBlockEvent ev = new EntityCombustByBlockEvent(this, entity, 8);
101        if (entity instanceof EntityArrow) {
102            ev.setCancelled();
103        }
104        Server.getInstance().getPluginManager().callEvent(ev);
105        if (!ev.isCancelled() && entity.isAlive() && entity.noDamageTicks == 0) {
106            entity.setOnFire(ev.getDuration());
107        }
108    }

115    @PowerNukkitDifference(info = "Soul Fire Implementation", since = "1.4.0.0-PN")
116    @Override
117    public int onUpdate(int type) {
118        
119        if (type == Level.BLOCK_UPDATE_NORMAL || type == Level.BLOCK_UPDATE_RANDOM) {
120            Block down = down();
121            
122            if (type == Level.BLOCK_UPDATE_NORMAL) {
123                int downId = down.getId();
124                if (downId == Block.SOUL_SAND || downId == Block.SOUL_SOIL) {
125                    this.getLevel().setBlock(this, getCurrentState().withBlockId(BlockID.SOUL_FIRE).getBlock(this));
126                    return type;
127                }
128            }
129            
130            if (!this.isBlockTopFacingSurfaceSolid(down) && !this.canNeighborBurn()) {
131                BlockFadeEvent event = new BlockFadeEvent(this, get(AIR));
132                level.getServer().getPluginManager().callEvent(event);
133                if (!event.isCancelled()) {
134                    level.setBlock(this, event.getNewState(), true);
135                }
136            } else if (this.level.gameRules.getBoolean(GameRule.DO_FIRE_TICK) && !level.isUpdateScheduled(this, this)) {
137                level.scheduleUpdate(this, tickRate());
138            }
139
140            return Level.BLOCK_UPDATE_NORMAL;
141        } else if (type == Level.BLOCK_UPDATE_SCHEDULED && this.level.gameRules.getBoolean(GameRule.DO_FIRE_TICK)) {
142            Block down = down();
143            int downId = down.getId();
144
145            boolean forever = downId == BlockID.NETHERRACK || downId == BlockID.MAGMA 
146                    || downId == BlockID.BEDROCK && ((BlockBedrock)down).getBurnIndefinitely();
147            
148            ThreadLocalRandom random = ThreadLocalRandom.current();
149
150            //TODO: END
151
152            if (!this.isBlockTopFacingSurfaceSolid(down) && !this.canNeighborBurn()) {
153                BlockFadeEvent event = new BlockFadeEvent(this, get(AIR));
154                level.getServer().getPluginManager().callEvent(event);
155                if (!event.isCancelled()) {
156                    level.setBlock(this, event.getNewState(), true);
157                }
158            }
159
160            if (!forever && this.getLevel().isRaining() &&
161                    (this.getLevel().canBlockSeeSky(this) ||
162                            this.getLevel().canBlockSeeSky(this.east()) ||
163                            this.getLevel().canBlockSeeSky(this.west()) ||
164                            this.getLevel().canBlockSeeSky(this.south()) ||
165                            this.getLevel().canBlockSeeSky(this.north()))
166                    ) {
167                BlockFadeEvent event = new BlockFadeEvent(this, get(AIR));
168                level.getServer().getPluginManager().callEvent(event);
169                if (!event.isCancelled()) {
170                    level.setBlock(this, event.getNewState(), true);
171                }
172            } else {
173                int meta = this.getDamage();
174
175                if (meta < 15) {
176                    int newMeta = meta + random.nextInt(3);
177                    this.setDamage(Math.min(newMeta, 15));
178                    this.getLevel().setBlock(this, this, true);
179                }
180
181                this.getLevel().scheduleUpdate(this, this.tickRate() + random.nextInt(10));
182
183                if (!forever && !this.canNeighborBurn()) {
184                    if (!this.isBlockTopFacingSurfaceSolid(down) || meta > 3) {
185                        BlockFadeEvent event = new BlockFadeEvent(this, get(AIR));
186                        level.getServer().getPluginManager().callEvent(event);
187                        if (!event.isCancelled()) {
188                            level.setBlock(this, event.getNewState(), true);
189                        }
190                    }
191                } else if (!forever && !(down.getBurnAbility() > 0) && meta == 15 && random.nextInt(4) == 0) {
192                    BlockFadeEvent event = new BlockFadeEvent(this, get(AIR));
193                    level.getServer().getPluginManager().callEvent(event);
194                    if (!event.isCancelled()) {
195                        level.setBlock(this, event.getNewState(), true);
196                    }
197                } else {
198                    int o = 0;
199
200                    //TODO: decrease the o if the rainfall values are high
201
202                    this.tryToCatchBlockOnFire(this.east(), 300 + o, meta);
203                    this.tryToCatchBlockOnFire(this.west(), 300 + o, meta);
204                    this.tryToCatchBlockOnFire(down, 250 + o, meta);
205                    this.tryToCatchBlockOnFire(this.up(), 250 + o, meta);
206                    this.tryToCatchBlockOnFire(this.south(), 300 + o, meta);
207                    this.tryToCatchBlockOnFire(this.north(), 300 + o, meta);
208
209                    for (int x = (int) (this.x - 1); x <= (int) (this.x + 1); ++x) {
210                        for (int z = (int) (this.z - 1); z <= (int) (this.z + 1); ++z) {
211                            for (int y = (int) (this.y - 1); y <= (int) (this.y + 4); ++y) {
212                                if (x != (int) this.x || y != (int) this.y || z != (int) this.z) {
213                                    int k = 100;
214
215                                    if (y > this.y + 1) {
216                                        k += (y - (this.y + 1)) * 100;
217                                    }
218
219                                    Block block = this.getLevel().getBlock(new Vector3(x, y, z));
220                                    int chance = this.getChanceOfNeighborsEncouragingFire(block);
221
222                                    if (chance > 0) {
223                                        int t = (chance + 40 + this.getLevel().getServer().getDifficulty() * 7) / (meta + 30);
224
225                                        //TODO: decrease the t if the rainfall values are high
226
227                                        if (t > 0 && random.nextInt(k) <= t) {
228                                            int damage = meta + random.nextInt(5) / 4;
229
230                                            if (damage > 15) {
231                                                damage = 15;
232                                            }
233
234                                            BlockIgniteEvent e = new BlockIgniteEvent(block, this, null, BlockIgniteEvent.BlockIgniteCause.SPREAD);
235                                            this.level.getServer().getPluginManager().callEvent(e);
236
237                                            if (!e.isCancelled()) {
238                                                this.getLevel().setBlock(block, Block.get(BlockID.FIRE, damage), true);
239                                                this.getLevel().scheduleUpdate(block, this.tickRate());
240                                            }
241                                        }
242                                    }
243                                }
244                            }
245                        }
246                    }
247                }
248            }
249        }
250
251        return 0;
252    }

254    private void tryToCatchBlockOnFire(Block block, int bound, int damage) {
255        int burnAbility = block.getBurnAbility();
256
257        Random random = ThreadLocalRandom.current();
258
259        if (random.nextInt(bound) < burnAbility) {
260
261            if (random.nextInt(damage + 10) < 5) {
262                int meta = damage + random.nextInt(5) / 4;
263
264                if (meta > 15) {
265                    meta = 15;
266                }
267
268                BlockIgniteEvent e = new BlockIgniteEvent(block, this, null, BlockIgniteEvent.BlockIgniteCause.SPREAD);
269                this.level.getServer().getPluginManager().callEvent(e);
270
271                if (!e.isCancelled()) {
272                    this.getLevel().setBlock(block, Block.get(BlockID.FIRE, meta), true);
273                    this.getLevel().scheduleUpdate(block, this.tickRate());
274                }
275            } else {
276                BlockBurnEvent ev = new BlockBurnEvent(block);
277                this.getLevel().getServer().getPluginManager().callEvent(ev);
278
279                if (!ev.isCancelled()) {
280                    this.getLevel().setBlock(block, Block.get(BlockID.AIR), true);
281                }
282            }
283
284            if (block instanceof BlockTNT) {
285                ((BlockTNT) block).prime();
286            }
287        }
288    }
289
290    private int getChanceOfNeighborsEncouragingFire(Block block) {
291        if (block.getId() != AIR) {
292            return 0;
293        } else {
294            int chance = 0;
295            chance = Math.max(chance, block.east().getBurnChance());
296            chance = Math.max(chance, block.west().getBurnChance());
297            chance = Math.max(chance, block.down().getBurnChance());
298            chance = Math.max(chance, block.up().getBurnChance());
299            chance = Math.max(chance, block.south().getBurnChance());
300            chance = Math.max(chance, block.north().getBurnChance());
301            return chance;
302        }
303    }

305    public boolean canNeighborBurn() {
306        for (BlockFace face : BlockFace.values()) {
307            if (this.getSide(face).getBurnChance() > 0) {
308                return true;
309            }
310        }
311
312        return false;
313    }

315    public boolean isBlockTopFacingSurfaceSolid(Block block) {
316        if (block != null) {
317            if (block.isSolid()) {
318                return true;
319            } else {
320                if (block instanceof BlockStairs &&
321                        (block.getDamage() & 4) == 4) {
322
323                    return true;
324                } else if (block instanceof BlockSlab &&
325                        (block.getDamage() & 8) == 8) {
326
327                    return true;
328                } else if (block instanceof BlockSnowLayer &&
329                        (block.getDamage() & 7) == 7) {
330
331                    return true;
332                }
333            }
334        }
335
336        return false;
337    }
