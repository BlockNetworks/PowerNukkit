003import cn.nukkit.Player;
004import cn.nukkit.api.PowerNukkitOnly;

016import javax.annotation.Nonnull;
017import java.util.concurrent.ThreadLocalRandom;
018
019import static cn.nukkit.block.BlockCoral.COLOR;
020
021@PowerNukkitOnly
022public class BlockCoralFan extends BlockFlowable implements Faceable {
023    @PowerNukkitOnly
024    @Since("1.5.0.0-PN")
025    public static final ArrayBlockProperty<BlockFace.Axis> FAN_DIRECTION = new ArrayBlockProperty<>("coral_fan_direction", false,
026            new BlockFace.Axis[]{BlockFace.Axis.X, BlockFace.Axis.Z}
027    ).ordinal(true);
028
029    @PowerNukkitOnly
030    @Since("1.5.0.0-PN")
031    public static final BlockProperties PROPERTIES = new BlockProperties(COLOR, FAN_DIRECTION);
032
033    @PowerNukkitOnly
034    public BlockCoralFan() {
035        this(0);
036    }
037
038    @PowerNukkitOnly
039    public BlockCoralFan(int meta) {
040        super(meta);
041    }
042    
043    @Override
044    public int getId() {
045        return CORAL_FAN;
046    }
047
048    @Since("1.4.0.0-PN")
049    @PowerNukkitOnly
050    @Nonnull
051    @Override
052    public BlockProperties getProperties() {
053        return PROPERTIES;
054    }
055
056    @Override
057    public String getName() {
058        String[] names = new String[] {
059                "Tube Coral Fan",
060                "Brain Coral Fan",
061                "Bubble Coral Fan",
062                "Fire Coral Fan",
063                "Horn Coral Fan"
064        };
065        return names[getType()];
066    }
067    
068    @Override
069    public BlockColor getColor() {
070        BlockColor[] colors = new BlockColor[] {
071                BlockColor.BLUE_BLOCK_COLOR,
072                BlockColor.PINK_BLOCK_COLOR,
073                BlockColor.PURPLE_BLOCK_COLOR,
074                BlockColor.RED_BLOCK_COLOR,
075                BlockColor.YELLOW_BLOCK_COLOR
076        };
077        return colors[getType()];
078    }
079    
080    @PowerNukkitOnly
081    @Override
082    public int getWaterloggingLevel() {
083        return 2;
084    }
085
086    @PowerNukkitOnly
087    public boolean isDead() {
088        return false;
089    }
090
091    @PowerNukkitOnly
092    public int getType() {
093        return getDamage() & 0x7;
094    }
095    
096    @Override
097    public BlockFace getBlockFace() {
098        return BlockFace.fromHorizontalIndex(((getDamage() & 0x8) >> 3) + 1);
099    }
100
101    @PowerNukkitOnly
102    public BlockFace getRootsFace() {
103        return BlockFace.DOWN;
104    }
105    
106    @Override
107    public int onUpdate(int type) {
108        if (type == Level.BLOCK_UPDATE_NORMAL) {
109            Block side = getSide(getRootsFace());
110            if (!side.isSolid() || side.getId() == MAGMA || side.getId() == SOUL_SAND) {
111                this.getLevel().useBreakOn(this);
112            } else {
113                this.getLevel().scheduleUpdate(this, 60 + ThreadLocalRandom.current().nextInt(40));
114            }
115            return type;
116        } else if (type == Level.BLOCK_UPDATE_SCHEDULED) {
117            Block side = getSide(getRootsFace());
118            if (side.getId() == ICE) {
119                this.getLevel().useBreakOn(this);
120                return type;
121            }
122
123            if (!isDead() && !(getLevelBlockAtLayer(1) instanceof BlockWater) && !(getLevelBlockAtLayer(1) instanceof BlockIceFrosted)) {
124                BlockFadeEvent event = new BlockFadeEvent(this, new BlockCoralFanDead(getDamage()));
125                if (!event.isCancelled()) {
126                    this.getLevel().setBlock(this, event.getNewState(), true, true);
127                }
128            }
129            return type;
130        } else if (type == Level.BLOCK_UPDATE_RANDOM) {
131            if ((getDamage() & 0x8) == 0) {
132                setDamage(getDamage() | 0x8);
133            } else {
134                setDamage(getDamage() ^ 0x8);
135            }
136            this.getLevel().setBlock(this, this, true, true);
137            return type;
138        }
139        return 0;
140    }
141    
142    @Override
143    public boolean place(@Nonnull Item item, @Nonnull Block block, @Nonnull Block target, @Nonnull BlockFace face, double fx, double fy, double fz, Player player) {
144        if (face == BlockFace.DOWN) {
145            return false;
146        }
147        
148        Block layer1 = block.getLevelBlockAtLayer(1);
149        boolean hasWater = layer1 instanceof BlockWater;
150        if (layer1.getId() != Block.AIR && (!hasWater || layer1.getDamage() != 0 && layer1.getDamage() != 8)) {
151            return false;
152        }
153        
154        if (hasWater && layer1.getDamage() == 8) {
155            this.getLevel().setBlock(this, 1, new BlockWater(), true, false);
156        }
157
158        if (!target.isSolid() || target.getId() == MAGMA || target.getId() == SOUL_SAND) {
159            return false;
160        }
161        
162        if (face == BlockFace.UP) {
163            double rotation = player.yaw % 360;
164            if (rotation < 0) {
165                rotation += 360.0;
166            }
167            int axisBit = rotation >= 0 && rotation < 12 || (342 <= rotation && rotation < 360)? 0x0 : 0x8;
168            setDamage(getDamage() & 0x7 | axisBit);
169            this.getLevel().setBlock(this, 0, hasWater? new BlockCoralFan(getDamage()) : new BlockCoralFanDead(getDamage()), true, true);
170        } else {
171            int type = getType();
172            int typeBit = type % 2;
173            int deadBit = isDead()? 0x1 : 0;
174            int faceBit;
175            switch (face) {
176                case WEST:
177                    faceBit = 0;
178                    break;
179                case EAST:
180                    faceBit = 1;
181                    break;
182                case NORTH:
183                    faceBit = 2;
184                    break;
185                default:
186                case SOUTH:
187                    faceBit = 3;
188                    break;
189            }
190            int deadData = faceBit << 2 | deadBit << 1 | typeBit;
191            int deadBlockId;
192            switch (type) {
193                default:
194                case BlockCoral.TYPE_TUBE:
195                case BlockCoral.TYPE_BRAIN:
196                    deadBlockId = CORAL_FAN_HANG;
197                    break;
198                case BlockCoral.TYPE_BUBBLE:
199                case BlockCoral.TYPE_FIRE:
200                    deadBlockId = CORAL_FAN_HANG2;
201                    break;
202                case BlockCoral.TYPE_HORN:
203                    deadBlockId = CORAL_FAN_HANG3;
204                    break;
205            }
206            this.getLevel().setBlock(this, 0, Block.get(deadBlockId, deadData), true, true);
207        }
208
209        return true;
210    }
211    
212    @Override
213    public boolean canSilkTouch() {
214        return true;
215    }
216    
217    @Override
218    public Item toItem() {
219        return Item.get(getItemId(), getDamage() ^ 0x8);
220    }
221    
222    @Override
223    public Item[] getDrops(Item item) {
224        if (item.getEnchantment(Enchantment.ID_SILK_TOUCH) != null) {
225            return super.getDrops(item);
226        } else {
227            return Item.EMPTY_ARRAY;
228        }
229    }
