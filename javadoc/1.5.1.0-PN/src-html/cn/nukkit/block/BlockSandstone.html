001package cn.nukkit.block;
002
003import cn.nukkit.api.DeprecationDetails;
004import cn.nukkit.api.PowerNukkitOnly;
005import cn.nukkit.api.Since;
006import cn.nukkit.blockproperty.ArrayBlockProperty;
007import cn.nukkit.blockproperty.BlockProperties;
008import cn.nukkit.blockproperty.BlockProperty;
009import cn.nukkit.blockproperty.value.SandStoneType;
010import cn.nukkit.item.Item;
011import cn.nukkit.item.ItemBlock;
012import cn.nukkit.item.ItemTool;
013import cn.nukkit.utils.BlockColor;
014
015import javax.annotation.Nonnull;
016
017/**
018 * @author MagicDroidX (Nukkit Project)
019 */
020public class BlockSandstone extends BlockSolidMeta {
021    @PowerNukkitOnly
022    @Since("1.5.0.0-PN")
023    public static final BlockProperty<SandStoneType> SAND_STONE_TYPE = new ArrayBlockProperty<>("sand_stone_type", true, SandStoneType.class);
024
025    @PowerNukkitOnly
026    @Since("1.5.0.0-PN")
027    public static final BlockProperties PROPERTIES = new BlockProperties(SAND_STONE_TYPE);
028
029    @Deprecated
030    @DeprecationDetails(since = "1.5.0.0-PN", replaceWith = "getSandstoneBlockType()", reason = "Use the BlockProperty API instead")
031    public static final int NORMAL = 0;
032
033    @Deprecated
034    @DeprecationDetails(since = "1.5.0.0-PN", replaceWith = "getSandstoneBlockType()", reason = "Use the BlockProperty API instead")
035    public static final int CHISELED = 1;
036
037    @Deprecated
038    @DeprecationDetails(since = "1.5.0.0-PN", replaceWith = "getSandstoneBlockType()", reason = "Use the BlockProperty API instead")
039    public static final int SMOOTH = 2;
040
041    public BlockSandstone() {
042        this(0);
043    }
044
045    public BlockSandstone(int meta) {
046        super(meta);
047    }
048
049    @Override
050    public int getId() {
051        return SANDSTONE;
052    }
053
054    @PowerNukkitOnly
055    @Since("1.5.0.0-PN")
056    @Nonnull
057    @Override
058    public BlockProperties getProperties() {
059        return PROPERTIES;
060    }
061
062    @PowerNukkitOnly
063    @Since("1.5.0.0-PN")
064    public void setSandstoneType(SandStoneType sandStoneType) {
065        setPropertyValue(SAND_STONE_TYPE, sandStoneType);
066    }
067
068    public SandStoneType getSandstoneType() {
069        return getPropertyValue(SAND_STONE_TYPE);
070    }
071
072    @Override
073    public double getHardness() {
074        return SandStoneType.SMOOTH.equals(getSandstoneType()) ? 2 : 0.8;
075    }
076
077    @Override
078    public double getResistance() {
079        return SandStoneType.SMOOTH.equals(getSandstoneType()) ? 6 : 0.8;
080    }
081
082    @Override
083    public String getName() {
084        return getSandstoneType().getEnglishName();
085    }
086
087    @Override
088    public int getToolTier() {
089        return ItemTool.TIER_WOODEN;
090    }
091
092    @Override
093    public int getToolType() {
094        return ItemTool.TYPE_PICKAXE;
095    }
096
097    @Override
098    public BlockColor getColor() {
099        return BlockColor.SAND_BLOCK_COLOR;
100    }
101
102    @Override
103    public boolean canHarvestWithHand() {
104        return false;
105    }
106}
