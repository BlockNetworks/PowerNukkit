001package cn.nukkit.block;
002
003import cn.nukkit.Player;
004import cn.nukkit.api.PowerNukkitDifference;
005import cn.nukkit.event.redstone.RedstoneUpdateEvent;
006import cn.nukkit.item.Item;
007import cn.nukkit.level.Level;
008import cn.nukkit.math.BlockFace;
009import cn.nukkit.utils.BlockColor;
010import cn.nukkit.utils.RedstoneComponent;
011
012import javax.annotation.Nonnull;
013
014/**
015 * @author Angelic47 (Nukkit Project)
016 */
017@PowerNukkitDifference(info = "Implements RedstoneComponent and uses methods from it.", since = "1.4.0.0-PN")
018public class BlockRedstoneTorch extends BlockTorch implements RedstoneComponent {
019
020    public BlockRedstoneTorch() {
021        this(0);
022    }
023
024    public BlockRedstoneTorch(int meta) {
025        super(meta);
026    }
027
028    @Override
029    public String getName() {
030        return "Redstone Torch";
031    }
032
033    @Override
034    public int getId() {
035        return REDSTONE_TORCH;
036    }
037
038    @Override
039    public int getLightLevel() {
040        return 7;
041    }
042
043    @Override
044    public boolean place(@Nonnull Item item, @Nonnull Block block, @Nonnull Block target, @Nonnull BlockFace face, double fx, double fy, double fz, Player player) {
045        if (!super.place(item, block, target, face, fx, fy, fz, player)) {
046            return false;
047        }
048
049        if (this.level.getServer().isRedstoneEnabled()) {
050            if (!checkState()) {
051                updateAllAroundRedstone(getBlockFace().getOpposite());
052            }
053
054            checkState();
055        }
056
057        return true;
058    }
059
060    @Override
061    public int getWeakPower(BlockFace side) {
062        return getBlockFace() != side ? 15 : 0;
063    }
064
065    @Override
066    public int getStrongPower(BlockFace side) {
067        return side == BlockFace.DOWN ? this.getWeakPower(side) : 0;
068    }
069
070    @Override
071    public boolean onBreak(Item item) {
072        if (!super.onBreak(item)) {
073            return false;
074        }
075
076        if (this.level.getServer().isRedstoneEnabled()) {
077            updateAllAroundRedstone(getBlockFace().getOpposite());
078        }
079        return true;
080    }
081
082    @Override
083    public int onUpdate(int type) {
084        if (super.onUpdate(type) == 0) {
085            if (!this.level.getServer().isRedstoneEnabled()) {
086                return 0;
087            }
088
089            if (type == Level.BLOCK_UPDATE_NORMAL || type == Level.BLOCK_UPDATE_REDSTONE) {
090                this.level.scheduleUpdate(this, tickRate());
091            } else if (type == Level.BLOCK_UPDATE_SCHEDULED) {
092                RedstoneUpdateEvent ev = new RedstoneUpdateEvent(this);
093                getLevel().getServer().getPluginManager().callEvent(ev);
094
095                if (ev.isCancelled()) {
096                    return 0;
097                }
098
099                if (checkState()) {
100                    return 1;
101                }
102            }
103        }
104
105        return 0;
106    }
107
108    private boolean checkState() {
109        if (isPoweredFromSide()) {
110            this.level.setBlock(getLocation(), Block.get(BlockID.UNLIT_REDSTONE_TORCH, getDamage()), false, true);
111
112            updateAllAroundRedstone(getBlockFace().getOpposite());
113            return true;
114        }
115
116        return false;
117    }
118
119    @PowerNukkitDifference(info = "Check if the side block is piston and if piston is getting power.",
120            since = "1.4.0.0-PN")
121    protected boolean isPoweredFromSide() {
122        BlockFace face = getBlockFace().getOpposite();
123        if (this.getSide(face) instanceof BlockPistonBase && this.getSide(face).isGettingPower()) {
124            return true;
125        }
126
127        return this.level.isSidePowered(this.getLocation().getSide(face), face);
128    }
129
130    @Override
131    public int tickRate() {
132        return 2;
133    }
134
135    @Override
136    public boolean isPowerSource() {
137        return true;
138    }
139
140    @Override
141    public BlockColor getColor() {
142        return BlockColor.AIR_BLOCK_COLOR;
143    }
144}
