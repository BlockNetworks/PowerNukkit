001package cn.nukkit.block;
002
003import cn.nukkit.Player;
004import cn.nukkit.Server;
005import cn.nukkit.api.PowerNukkitOnly;
006import cn.nukkit.blockentity.BlockEntity;
007import cn.nukkit.blockentity.BlockEntityCauldron;
008import cn.nukkit.entity.Entity;
009import cn.nukkit.event.entity.EntityCombustByBlockEvent;
010import cn.nukkit.event.entity.EntityDamageByBlockEvent;
011import cn.nukkit.event.entity.EntityDamageEvent;
012import cn.nukkit.event.player.PlayerBucketEmptyEvent;
013import cn.nukkit.event.player.PlayerBucketFillEvent;
014import cn.nukkit.item.Item;
015import cn.nukkit.item.ItemBucket;
016import cn.nukkit.item.MinecraftItemID;
017import cn.nukkit.level.Sound;
018import cn.nukkit.math.AxisAlignedBB;
019import cn.nukkit.potion.Effect;
020
021import javax.annotation.Nonnull;
022
023@PowerNukkitOnly
024public class BlockCauldronLava extends BlockCauldron {
025    @PowerNukkitOnly
026    public BlockCauldronLava() {
027        this(0x8);
028    }
029
030    @PowerNukkitOnly
031    public BlockCauldronLava(int meta) {
032        super(meta);
033    }
034    
035    @Override
036    public String getName() {
037        return "Lava Cauldron";
038    }
039    
040    @Override
041    public int getId() {
042        return LAVA_CAULDRON;
043    }
044    
045    @Override
046    public int getLightLevel() {
047        return 15;
048    }
049    
050    @Override
051    public boolean hasEntityCollision() {
052        return true;
053    }
054    
055    @Override
056    protected AxisAlignedBB recalculateCollisionBoundingBox() {
057        return shrink(0.3, 0.3, 0.3);
058    }
059    
060    @Override
061    public void setFillLevel(int fillLevel) {
062        super.setFillLevel(fillLevel);
063        setDamage(getDamage() | 0x8);
064    }
065    
066    @Override
067    public void onEntityCollide(Entity entity) {
068        // Always setting the duration to 15 seconds? TODO
069        EntityCombustByBlockEvent ev = new EntityCombustByBlockEvent(this, entity, 15);
070        Server.getInstance().getPluginManager().callEvent(ev);
071        if (!ev.isCancelled()
072                // Making sure the entity is actually alive and not invulnerable.
073                && entity.isAlive()
074                && entity.noDamageTicks == 0) {
075            entity.setOnFire(ev.getDuration());
076        }
077    
078        if (!entity.hasEffect(Effect.FIRE_RESISTANCE)) {
079            entity.attack(new EntityDamageByBlockEvent(this, entity, EntityDamageEvent.DamageCause.LAVA, 4));
080        }
081    }
082    
083    @Override
084    public boolean onActivate(@Nonnull Item item, Player player) {
085        BlockEntity be = this.level.getBlockEntity(this);
086    
087        if (!(be instanceof BlockEntityCauldron)) {
088            return false;
089        }
090    
091        BlockEntityCauldron cauldron = (BlockEntityCauldron) be;
092    
093        switch (item.getId()) {
094            case Item.BUCKET:
095                ItemBucket bucket = (ItemBucket) item;
096                if (bucket.getFishEntityId() != null) {
097                    break;
098                }
099                if (item.getDamage() == 0) { //empty
100                    if (!isFull() || cauldron.isCustomColor() || cauldron.hasPotion()) {
101                        break;
102                    }
103    
104                    PlayerBucketFillEvent ev = new PlayerBucketFillEvent(player, this, null, this, item, MinecraftItemID.LAVA_BUCKET.get(1, bucket.getCompoundTag()));
105                    this.level.getServer().getPluginManager().callEvent(ev);
106                    if (!ev.isCancelled()) {
107                        replaceBucket(bucket, player, ev.getItem());
108                        this.setFillLevel(FILL_LEVEL.getMinValue());//empty
109                        this.level.setBlock(this, new BlockCauldron(0), true);
110                        cauldron.clearCustomColor();
111                        this.getLevel().addSound(this.add(0.5, 1, 0.5), Sound.BUCKET_FILL_LAVA);
112                    }
113                } else if (bucket.isWater() || bucket.isLava()) { //water or lava bucket
114                    if (isFull() && !cauldron.isCustomColor() && !cauldron.hasPotion() && item.getDamage() == 10) {
115                        break;
116                    }
117    
118                    PlayerBucketEmptyEvent ev = new PlayerBucketEmptyEvent(player, this, null, this, item, MinecraftItemID.BUCKET.get(1, bucket.getCompoundTag()));
119                    this.level.getServer().getPluginManager().callEvent(ev);
120                    if (!ev.isCancelled()) {
121                        replaceBucket(bucket, player, ev.getItem());
122    
123                        if (cauldron.hasPotion()) {//if has potion
124                            clearWithFizz(cauldron);
125                        } else if (bucket.isLava()) { //lava bucket
126                            this.setFillLevel(FILL_LEVEL.getMaxValue());//fill
127                            cauldron.clearCustomColor();
128                            this.level.setBlock(this, this, true);
129                            this.getLevel().addSound(this.add(0.5, 1, 0.5), Sound.BUCKET_EMPTY_LAVA);
130                        } else {
131                            if (isEmpty()) {
132                                this.level.setBlock(this, new BlockCauldron(6), true, true);
133                                cauldron.clearCustomColor();
134                                this.getLevel().addSound(this.add(0.5, 1, 0.5), Sound.CAULDRON_FILLWATER);
135                            } else {
136                                clearWithFizz(cauldron);
137                            }
138                        }
139                    }
140                }
141                break;
142            case Item.POTION:
143            case Item.SPLASH_POTION:
144            case Item.LINGERING_POTION:
145                if (!isEmpty() && (cauldron.hasPotion()? cauldron.getPotionId() != item.getDamage() : item.getDamage() != 0)) {
146                    clearWithFizz(cauldron);
147                    break;
148                }
149                return super.onActivate(item, player);
150            case Item.GLASS_BOTTLE:
151                if (!isEmpty() && cauldron.hasPotion()) {
152                    return super.onActivate(item, player);
153                }
154            default:
155                return true;
156        }
157    
158        this.level.updateComparatorOutputLevel(this);
159        return true;
160    }
161}
