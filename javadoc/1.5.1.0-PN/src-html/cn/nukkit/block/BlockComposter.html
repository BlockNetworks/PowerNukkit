121    @PowerNukkitDifference(info = "Player is null when is called from BlockEntityHopper")
122    @Override
123    public boolean onActivate(@Nonnull Item item, Player player) {
124        if (item.getCount() <= 0 || item.getId() == Item.AIR) {
125            return false;
126        }
127
128        if (isFull()) {
129            ComposterEmptyEvent event = new ComposterEmptyEvent(this, player, item, MinecraftItemID.BONE_MEAL.get(1), 0);
130            this.level.getServer().getPluginManager().callEvent(event);
131            if (!event.isCancelled()) {
132                setDamage(event.getNewLevel());
133                this.level.setBlock(this, this, true, true);
134                this.level.dropItem(add(0.5, 0.85, 0.5), event.getDrop(), event.getMotion(), false, 10);
135                this.level.addSound(add(0.5 , 0.5, 0.5), Sound.BLOCK_COMPOSTER_EMPTY);
136            }
137            return true;
138        }
139
140        int chance = getChance(item);
141        if (chance <= 0) {
142            return false;
143        }
144
145        boolean success = new Random().nextInt(100) < chance;
146        ComposterFillEvent event = new ComposterFillEvent(this, player, item, chance, success);
147        this.level.getServer().getPluginManager().callEvent(event);
148
149        if (event.isCancelled()) {
150            return true;
151        }
152
153        if (player != null && !player.isCreative()) {
154            item.setCount(item.getCount() - 1);
155        }
156
157        if (event.isSuccess()) {
158            if (incrementLevel()) {
159                level.addSound(this.add(0.5, 0.5, 0.5), Sound.BLOCK_COMPOSTER_READY);
160            } else {
161                level.addSound(this.add(0.5, 0.5, 0.5), Sound.BLOCK_COMPOSTER_FILL_SUCCESS);
162            }
163        } else {
164            level.addSound(this.add(0.5, 0.5, 0.5), Sound.BLOCK_COMPOSTER_FILL);
165        }
166
167        return true;
168    }

170    @PowerNukkitOnly
171    public Item empty() {

173    }

175    @PowerNukkitOnly
176    public Item empty(@Nullable Item item, @Nullable Player player) {
177        ComposterEmptyEvent event = new ComposterEmptyEvent(this, player, item, new ItemDye(DyeColor.WHITE), 0);
178        this.level.getServer().getPluginManager().callEvent(event);
179        if (!event.isCancelled()) {
180            setPropertyValue(COMPOSTER_FILL_LEVEL, event.getNewLevel());
181            this.level.setBlock(this, this, true, true);
182            if (item != null) {
183                this.level.dropItem(add(0.5, 0.85, 0.5), event.getDrop(), event.getMotion(), false, 10);
184            }
185            this.level.addSound(add(0.5 , 0.5, 0.5), Sound.BLOCK_COMPOSTER_EMPTY);
186            return event.getDrop();
187        }
188        return null;
189    }

228    @PowerNukkitOnly
229    public static void register(int chance, Item item) {
230        registerItem(chance, item.getId(), item.getDamage());
231    }
232
233    @PowerNukkitOnly
234    public static int getChance(Item item) {
235        int chance = compostableItems.get(item.getId() << 6 | item.getDamage());
236        if (chance == 0) {
237            chance = compostableItems.get(item.getId() << 6);
238        }
239        return chance;
240    }
