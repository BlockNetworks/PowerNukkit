1522  public MovingObjectPosition calculateIntercept(Vector3 pos1, Vector3 pos2) {
1523    AxisAlignedBB bb = this.getBoundingBox();
1524    if (bb == null) {
1525      return null;
1526    }
1527    Vector3 v1 = pos1.getIntermediateWithXValue(pos2, bb.getMinX());
1528    Vector3 v2 = pos1.getIntermediateWithXValue(pos2, bb.getMaxX());
1529    Vector3 v3 = pos1.getIntermediateWithYValue(pos2, bb.getMinY());
1530    Vector3 v4 = pos1.getIntermediateWithYValue(pos2, bb.getMaxY());
1531    Vector3 v5 = pos1.getIntermediateWithZValue(pos2, bb.getMinZ());
1532    Vector3 v6 = pos1.getIntermediateWithZValue(pos2, bb.getMaxZ());
1533    if (v1 != null && !bb.isVectorInYZ(v1)) {
1534      v1 = null;
1535    }
1536    if (v2 != null && !bb.isVectorInYZ(v2)) {
1537      v2 = null;
1538    }
1539    if (v3 != null && !bb.isVectorInXZ(v3)) {
1540      v3 = null;
1541    }
1542    if (v4 != null && !bb.isVectorInXZ(v4)) {
1543      v4 = null;
1544    }
1545    if (v5 != null && !bb.isVectorInXY(v5)) {
1546      v5 = null;
1547    }
1548    if (v6 != null && !bb.isVectorInXY(v6)) {
1549      v6 = null;
1550    }
1551    Vector3 vector = v1;
1552    if (v2 != null && (vector == null || pos1.distanceSquared(v2) < pos1.distanceSquared(vector))) {
1553      vector = v2;
1554    }
1555    if (v3 != null && (vector == null || pos1.distanceSquared(v3) < pos1.distanceSquared(vector))) {
1556      vector = v3;
1557    }
1558    if (v4 != null && (vector == null || pos1.distanceSquared(v4) < pos1.distanceSquared(vector))) {
1559      vector = v4;
1560    }
1561    if (v5 != null && (vector == null || pos1.distanceSquared(v5) < pos1.distanceSquared(vector))) {
1562      vector = v5;
1563    }
1564    if (v6 != null && (vector == null || pos1.distanceSquared(v6) < pos1.distanceSquared(vector))) {
1565      vector = v6;
1566    }
1567    if (vector == null) {
1568      return null;
1569    }
1570    BlockFace f = null;
1571    if (vector == v1) {
1572      f = BlockFace.WEST;
1573    } else if (vector == v2) {
1574      f = BlockFace.EAST;
1575    } else if (vector == v3) {
1576      f = BlockFace.DOWN;
1577    } else if (vector == v4) {
1578      f = BlockFace.UP;
1579    } else if (vector == v5) {
1580      f = BlockFace.NORTH;
1581    } else if (vector == v6) {
1582      f = BlockFace.SOUTH;
1583    }
1584    return MovingObjectPosition.fromBlock((int) this.x, (int) this.y, (int) this.z, f, vector.add(this.x, this.y, this.z));
1585  }

1666  @PowerNukkitOnly
1667  @Since("1.4.0.0-PN")
1668  public boolean isSideFull(BlockFace face) {
1669    AxisAlignedBB boundingBox = getBoundingBox();
1670    if (boundingBox == null) {
1671      return false;
1672    }
1673    if (face.getAxis().getPlane() == BlockFace.Plane.HORIZONTAL) {
1674      if (boundingBox.getMinY() != getY() || boundingBox.getMaxY() != getY() + 1) {
1675        return false;
1676      }
1677      int offset = face.getXOffset();
1678      if (offset < 0) {
1679        return boundingBox.getMinX() == getX() && boundingBox.getMinZ() == getZ() && boundingBox.getMaxZ() == getZ() + 1;
1680      } else if (offset > 0) {
1681        return boundingBox.getMaxX() == getX() + 1 && boundingBox.getMaxZ() == getZ() + 1 && boundingBox.getMinZ() == getZ();
1682      }
1683      offset = face.getZOffset();
1684      if (offset < 0) {
1685        return boundingBox.getMinZ() == getZ() && boundingBox.getMinX() == getX() && boundingBox.getMaxX() == getX() + 1;
1686      }
1687      return boundingBox.getMaxZ() == getZ() + 1 && boundingBox.getMaxX() == getX() + 1 && boundingBox.getMinX() == getX();
1688    }
1689    if (boundingBox.getMinX() != getX() || boundingBox.getMaxX() != getX() + 1 || boundingBox.getMinZ() != getZ() || boundingBox.getMaxZ() != getZ() + 1) {
1690      return false;
1691    }
1692    if (face.getYOffset() < 0) {
1693      return boundingBox.getMinY() == getY();
1694    }
1695    return boundingBox.getMaxY() == getY() + 1;
1696  }
