014/**

028    public static final int ACACIA = 0;
029    public static final int DARK_OAK = 1;
030
031    public BlockWood2() {

033    }
034

036        super(meta);
037    }
038
039    @Override
040    public int getId() {
041        return WOOD2;
042    }
043
044    @Since("1.4.0.0-PN")
045    @PowerNukkitOnly
046    @Nonnull
047    @Override
048    public BlockProperties getProperties() {
049        return PROPERTIES;
050    }
051
052    @Override

054        return getPropertyValue(NEW_LOG_TYPE);
055    }
056
057    @Override
058    public void setWoodType(WoodType woodType) {
