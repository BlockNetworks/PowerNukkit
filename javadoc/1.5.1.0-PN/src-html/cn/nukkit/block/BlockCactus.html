001package cn.nukkit.block;

024/**
025 * @author Nukkit Project Team
026 */
027public class BlockCactus extends BlockTransparentMeta {
028    @PowerNukkitOnly
029    @Since("1.5.0.0-PN")
030    public static final BlockProperties PROPERTIES = new BlockProperties(CommonBlockProperties.AGE_15);
031
032    public BlockCactus(int meta) {
033        super(meta);
034    }
035
036    public BlockCactus() {
037        this(0);
038    }
039
040    @Override
041    public int getId() {
042        return CACTUS;
043    }
044
045    @Since("1.4.0.0-PN")
046    @PowerNukkitOnly
047    @Nonnull
048    @Override
049    public BlockProperties getProperties() {
050        return PROPERTIES;
051    }
052
053    @Override
054    public double getHardness() {
055        return 0.4;
056    }
057
058    @Override
059    public double getResistance() {
060        return 2;
061    }
062
063    @PowerNukkitOnly
064    @Override
065    public int getWaterloggingLevel() {
066        return 1;
067    }
068
069    @Override
070    public boolean hasEntityCollision() {
071        return true;
072    }
073
074    @Override
075    public double getMinX() {
076        return this.x + 0.0625;
077    }
078
079    @Override
080    public double getMinY() {
081        return this.y;
082    }
083
084    @Override
085    public double getMinZ() {
086        return this.z + 0.0625;
087    }
088
089    @Override
090    public double getMaxX() {
091        return this.x + 0.9375;
092    }
093
094    @Override
095    public double getMaxY() {
096        return this.y + 0.9375;
097    }
098
099    @Override
100    public double getMaxZ() {
101        return this.z + 0.9375;
102    }
103
104    @Override
105    protected AxisAlignedBB recalculateCollisionBoundingBox() {
106        return new SimpleAxisAlignedBB(this.x, this.y, this.z, this.x + 1, this.y + 1, this.z + 1);
107    }
108
109    @Override
110    public void onEntityCollide(Entity entity) {
111        entity.attack(new EntityDamageByBlockEvent(this, entity, DamageCause.CONTACT, 1));
112    }
113
114    @Override
115    public int onUpdate(int type) {
116        if (type == Level.BLOCK_UPDATE_NORMAL) {
117            Block down = down();
118            if (down.getId() != SAND && down.getId() != CACTUS) {
119                this.getLevel().useBreakOn(this);
120            } else {
121                for (int side = 2; side <= 5; ++side) {
122                    Block block = getSide(BlockFace.fromIndex(side));
123                    if (!block.canBeFlowedInto()) {
124                        this.getLevel().useBreakOn(this);
125                    }
126                }
127            }
128        } else if (type == Level.BLOCK_UPDATE_RANDOM) {
129            if (down().getId() != CACTUS) {
130                if (this.getDamage() == 0x0F) {
131                    for (int y = 1; y < 3; ++y) {
132                        Block b = this.getLevel().getBlock(new Vector3(this.x, this.y + y, this.z));
133                        if (b.getId() == AIR) {
134                            BlockGrowEvent event = new BlockGrowEvent(b, Block.get(BlockID.CACTUS));
135                            Server.getInstance().getPluginManager().callEvent(event);
136                            if (!event.isCancelled()) {
137                                this.getLevel().setBlock(b, event.getNewState(), true);
138                            }
139                        }
140                    }
141                    this.setDamage(0);
142                    this.getLevel().setBlock(this, this);
143                } else {
144                    this.setDamage(this.getDamage() + 1);
145                    this.getLevel().setBlock(this, this);
146                }
147            }
148        }
149
150        return 0;
151    }
152
153    @Override
154    public boolean place(@Nonnull Item item, @Nonnull Block block, @Nonnull Block target, @Nonnull BlockFace face, double fx, double fy, double fz, @Nullable Player player) {
155        Block down = this.down();
156        if (down.getId() == SAND || down.getId() == CACTUS) {
157            Block block0 = north();
158            Block block1 = south();
159            Block block2 = west();
160            Block block3 = east();
161            if (block0.canBeFlowedInto() && block1.canBeFlowedInto() && block2.canBeFlowedInto() && block3.canBeFlowedInto()) {
162                this.getLevel().setBlock(this, this, true);
163
164                return true;
165            }
166        }
167        return false;
168    }
169
170    @Override
171    public String getName() {
172        return "Cactus";
173    }
174
175    @Override
176    public BlockColor getColor() {
177        return BlockColor.FOLIAGE_BLOCK_COLOR;
178    }
179
180    @Override
181    public Item[] getDrops(Item item) {
182        return new Item[]{
183                Item.getBlock(BlockID.CACTUS, 0, 1)
184        };
185    }
186
187    @Override
188    public boolean breaksWhenMoved() {
189        return true;
190    }
191
192    @Override
193    public boolean sticksToPiston() {
194        return false;
195    }
196}
