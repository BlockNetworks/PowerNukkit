026@PowerNukkitDifference(since = "1.4.0.0-PN", info = "Implements BlockEntityHolder only in PowerNukkit")
027@PowerNukkitDifference(info = "Implements RedstoneComponent and uses methods from it.", since = "1.4.0.0-PN")
028public class BlockDaylightDetector extends BlockTransparentMeta implements RedstoneComponent, BlockEntityHolder<BlockEntityDaylightDetector> {
029
030    @Since("1.5.0.0-PN")
031    @PowerNukkitOnly
032    public static final BlockProperties PROPERTIES = CommonBlockProperties.REDSTONE_SIGNAL_BLOCK_PROPERTY;
033
034    public BlockDaylightDetector() {
035        // Does nothing
036    }
037
038    @Override
039    public int getId() {
040        return DAYLIGHT_DETECTOR;
041    }
042
043    @Since("1.4.0.0-PN")
044    @PowerNukkitOnly
045    @Nonnull
046    @Override
047    public BlockProperties getProperties() {
048        return PROPERTIES;
049    }
050
051    @Override
052    public String getName() {
053        return "Daylight Detector";
054    }
055
056    @PowerNukkitOnly
057    @Since("1.4.0.0-PN")
058    @Nonnull
059    @Override
060    public String getBlockEntityType() {
061        return BlockEntity.DAYLIGHT_DETECTOR;
062    }
063
064    @Since("1.4.0.0-PN")
065    @PowerNukkitOnly
066    @Nonnull
067    @Override
068    public Class<? extends BlockEntityDaylightDetector> getBlockEntityClass() {
069        return BlockEntityDaylightDetector.class;
070    }
071
072    @Override
073    public double getHardness() {
074        return 0.2;
075    }
076
077    @PowerNukkitOnly
078    @Override
079    public int getWaterloggingLevel() {
080        return 1;
081    }
082
083    @Override
084    public int getToolType() {
085        return ItemTool.TYPE_AXE;
086    }
087
088    @Override
089    public BlockColor getColor() {
090        return BlockColor.WOOD_BLOCK_COLOR;
091    }
092
093    @Override
094    public Item toItem() {
095        return new ItemBlock(this, 0);
096    }
097
098    @Override
099    public boolean canBeActivated() {
100        return true;
101    }
102
103    @Override
104    public boolean place(@Nonnull Item item, @Nonnull Block block, @Nonnull Block target, @Nonnull BlockFace face, double fx, double fy, double fz, Player player) {
105        BlockEntityDaylightDetector detector = BlockEntityHolder.setBlockAndCreateEntity(this);
106        if (detector == null) {
107            return false;
108        }
109        if (getLevel().getDimension() == Level.DIMENSION_OVERWORLD) {
110            updatePower();
111        }
112        return true;
113    }
114
115    @Override
116    public boolean onActivate(@Nonnull Item item, Player player) {
117        BlockDaylightDetectorInverted block = new BlockDaylightDetectorInverted();
118        getLevel().setBlock(this, block, true, true);
119        block.updatePower();
120        return true;
121    }
122
123    @Override
124    public boolean onBreak(Item item) {
125        if (super.onBreak(item)) {
126            if (getLevel().getDimension() == Level.DIMENSION_OVERWORLD) {
127                updateAroundRedstone();
128            }
129            return true;
130        }
131        return false;
132    }
133
134    @Override
135    public int getWeakPower(BlockFace face) {
136        return getLevel().getBlockDataAt(getFloorX(), getFloorY(), getFloorZ());
137    }
138
139    @Override
140    public boolean isPowerSource() {
141        return true;
142    }
143
144    public boolean isInverted() {
145        return false;
146    }
147
148    public void updatePower() {
149        int i;
150        if (getLevel().getDimension() == Level.DIMENSION_OVERWORLD) {
151            i = getLevel().getBlockSkyLightAt((int) x, (int) y, (int) z) - getLevel().calculateSkylightSubtracted(1.0F);
152            float f = getLevel().getCelestialAngle(1.0F);
153
154            if (this.isInverted()) {
155                i = 15 - i;
156            }
157
158            if (i > 0 && !this.isInverted()) {
159                float f1 = f < (float) Math.PI ? 0.0F : ((float) Math.PI * 2F);
160                f = f + (f1 - f) * 0.2F;
161                i = Math.round((float) i * MathHelper.cos(f));
162            }
163
164            i = MathHelper.clamp(i, 0, 15);
165        } else i = 0;
166
167        if (i != getLevel().getBlockDataAt(getFloorX(), getFloorY(), getFloorZ())) {
168            getLevel().setBlockDataAt(getFloorX(), getFloorY(), getFloorZ(), i);
169            updateAroundRedstone();
170        }
171    }
172}
