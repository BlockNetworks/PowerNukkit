147            return 0;
148        } else {
149            if (!this.level.getServer().isRedstoneEnabled()) {
150                return 0;
151            }
152            // We can't use getOrCreateBlockEntity(), because the update method is called on block place,
153            // before the "real" BlockEntity is set. That means, if we'd use the other method here,
154            // it would create two BlockEntities.
155            BlockEntityPistonArm arm = this.getBlockEntity();
156            boolean powered = this.isGettingPower();
157            this.updateAroundRedstoneTorches(powered);
158            if (arm == null || !arm.finished) return 0;
159            if (arm.state % 2 == 0 && arm.powered != powered && checkState(powered)) {
160                arm.powered = powered;
161                if (arm.chunk != null) {
162                    arm.chunk.setChanged();
163                }
164            }
165            return type;
166        }

226        BlockFace direction = getBlockFace();
227        BlocksCalculator calculator = new BlocksCalculator(level, this, getBlockFace(), extending, sticky);
228        boolean canMove = calculator.canMove();
229        if (!canMove && extending) {

231        }
232        List<BlockVector3> attached = Collections.emptyList();
233        BlockPistonEvent event = new BlockPistonEvent(this, direction, calculator.getBlocksToMove(), calculator.getBlocksToDestroy(), extending);

235        if (event.isCancelled()) {
236            return false;
237        }
238        if (canMove && (this.sticky || extending)) {
239            List<Block> destroyBlocks = calculator.getBlocksToDestroy();
240            for (int i = destroyBlocks.size() - 1; i >= 0; --i) {
241                Block block = destroyBlocks.get(i);
242                this.level.useBreakOn(block, null, null, false);
243            }
244            List<Block> newBlocks = calculator.getBlocksToMove();
245            attached = newBlocks.stream().map(Vector3::asBlockVector3).collect(Collectors.toList());
246            BlockFace side = extending ? direction : direction.getOpposite();
247            List<CompoundTag> tags = new ArrayList<>();
248            for (Block oldBlock : newBlocks) {

250                BlockEntity be = this.level.getBlockEntity(oldBlock);

252                    be.saveNBT();
253                    tag = new CompoundTag(be.namedTag.getTags());

256                tags.add(tag);
257            }
258            int i = 0;

260                Vector3 oldPos = newBlock.add(0);

262                CompoundTag nbt = new CompoundTag().putInt("pistonPosX", this.getFloorX()).putInt("pistonPosY", this.getFloorY()).putInt("pistonPosZ", this.getFloorZ()).putCompound("movingBlock",  //only for nukkit purpose

289    public static boolean canPush(Block block, BlockFace face, boolean destroyBlocks, boolean extending) {
290        if (block.getY() >= 0 && (face != BlockFace.DOWN || block.getY() != 0) && block.getY() <= 255 && (face != BlockFace.UP || block.getY() != 255)) {
291            if (extending && !block.canBePushed() || !extending && !block.canBePulled()) {
292                return false;
293            }
294            if (block.breaksWhenMoved()) {
295                return destroyBlocks || block.sticksToPiston();
296            }
297            BlockEntity be = block.level.getBlockEntity(block);
298            return be == null || be.isMovable();
299        }
300        return false;
301    }

326        public BlocksCalculator(Level level, Block pos, BlockFace face, boolean extending, boolean sticky) {
327            this.pistonPos = pos.getLocation();
328            this.extending = extending;
329            this.sticky = sticky;
330            if (!extending) {
331                this.armPos = pistonPos.getSide(face);
332            }
333            if (extending) {
334                this.moveDirection = face;
335                this.blockToMove = pos.getSide(face);
336            } else {
337                this.moveDirection = face.getOpposite();
338                if (sticky) {

346        public boolean canMove() {
347            if (!sticky && !extending) {
348                return true;
349            }
350            this.toMove.clear();
351            this.toDestroy.clear();
352            Block block = this.blockToMove;
353            if (!canPush(block, this.moveDirection, true, extending)) {
354                return false;
355            }
356            if (block.breaksWhenMoved()) {
357                if (extending || block.sticksToPiston()) {
358                    this.toDestroy.add(this.blockToMove);
359                }
360                return true;
361            }
362            if (!this.addBlockLine(this.blockToMove, this.blockToMove.getSide(this.moveDirection.getOpposite()), true)) {
363                return false;
364            }
365            for (int i = 0; i < this.toMove.size(); ++i) {
366                Block b = this.toMove.get(i);
367                int blockId = b.getId();
368                if ((blockId == SLIME_BLOCK || blockId == HONEY_BLOCK) && !this.addBranchingBlocks(b)) {
369                    return false;
370                }
371            }
372            return true;
373        }

375        private boolean addBlockLine(Block origin, Block from, boolean mainBlockLine) {
376            Block block = origin.clone();
377            if (block.getId() == AIR) {
378                return true;
379            }
380            if (!mainBlockLine && (block.getId() == SLIME_BLOCK && from.getId() == HONEY_BLOCK || block.getId() == HONEY_BLOCK && from.getId() == SLIME_BLOCK)) {
381                return true;
382            }
383            if (!canPush(origin, this.moveDirection, false, extending)) {
384                return true;
385            }
386            if (origin.equals(this.pistonPos)) {
387                return true;
388            }
389            if (this.toMove.contains(origin)) {
390                return true;
391            }
392            if (this.toMove.size() >= 12) {
393                return false;
394            }
395            this.toMove.add(block);
396            int count = 1;
397            List<Block> sticked = new ArrayList<>();
398            while (block.getId() == SLIME_BLOCK || block.getId() == HONEY_BLOCK) {
399                Block oldBlock = block.clone();
400                block = origin.getSide(this.moveDirection.getOpposite(), count);
401                if (!extending && (block.getId() == SLIME_BLOCK && oldBlock.getId() == HONEY_BLOCK || block.getId() == HONEY_BLOCK && oldBlock.getId() == SLIME_BLOCK)) {
402                    break;
403                }
404                if (block.getId() == AIR || !canPush(block, this.moveDirection, false, extending) || block.equals(this.pistonPos)) {
405                    break;
406                }
407                if (block.breaksWhenMoved() && block.sticksToPiston()) {
408                    this.toDestroy.add(block);
409                    break;
410                }
411                if (count + this.toMove.size() > 12) {
412                    return false;
413                }
414                count++;
415                sticked.add(block);
416            }
417            int stickedCount = sticked.size();
418            if (stickedCount > 0) {
419                this.toMove.addAll(Lists.reverse(sticked));
420            }
421            int step = 1;
422            while (true) {
423                Block nextBlock = origin.getSide(this.moveDirection, step);
424                int index = this.toMove.indexOf(nextBlock);
425                if (index > -1) {
426                    this.reorderListAtCollision(stickedCount, index);
427                    for (int i = 0; i <= index + stickedCount; ++i) {
428                        Block b = this.toMove.get(i);
429                        if ((b.getId() == SLIME_BLOCK || b.getId() == HONEY_BLOCK) && !this.addBranchingBlocks(b)) {
430                            return false;
431                        }
432                    }
433                    return true;
434                }
435                if (nextBlock.getId() == AIR || nextBlock.equals(armPos)) {
436                    return true;
437                }
438                if (!canPush(nextBlock, this.moveDirection, true, extending) || nextBlock.equals(this.pistonPos)) {
439                    return false;
440                }
441                if (nextBlock.breaksWhenMoved()) {
442                    this.toDestroy.add(nextBlock);
443                    return true;
444                }
445                if (this.toMove.size() >= 12) {
446                    return false;
447                }
448                this.toMove.add(nextBlock);
449                ++stickedCount;
450                ++step;
451            }
452        }

454        private void reorderListAtCollision(int count, int index) {
455            List<Block> list = new ArrayList<>(this.toMove.subList(0, index));
456            List<Block> list1 = new ArrayList<>(this.toMove.subList(this.toMove.size() - count, this.toMove.size()));
457            List<Block> list2 = new ArrayList<>(this.toMove.subList(index, this.toMove.size() - count));
458            this.toMove.clear();
459            this.toMove.addAll(list);
460            this.toMove.addAll(list1);
461            this.toMove.addAll(list2);
462        }

464        private boolean addBranchingBlocks(Block block) {
465            for (BlockFace face : BlockFace.values()) {
466                if (face.getAxis() != this.moveDirection.getAxis() && !this.addBlockLine(block.getSide(face), block, false)) {
467                    return false;
468                }
469            }
470            return true;
471        }
