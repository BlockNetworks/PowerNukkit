001package cn.nukkit.block;
002
003import cn.nukkit.api.PowerNukkitOnly;
004import cn.nukkit.api.Since;
005import cn.nukkit.blockproperty.ArrayBlockProperty;
006import cn.nukkit.blockproperty.BlockProperties;
007import cn.nukkit.blockproperty.value.CoralType;
008import cn.nukkit.item.Item;
009import cn.nukkit.item.ItemBlock;
010import cn.nukkit.level.Level;
011import cn.nukkit.math.BlockFace;
012import cn.nukkit.utils.Faceable;
013
014import javax.annotation.Nonnull;
015
016import static cn.nukkit.blockproperty.CommonBlockProperties.PERMANENTLY_DEAD;
017
018@PowerNukkitOnly
019public class BlockCoralFanHang extends BlockCoralFan implements Faceable {
020    @PowerNukkitOnly
021    @Since("1.5.0.0-PN")
022    public static final ArrayBlockProperty<CoralType> HANG1_TYPE = new ArrayBlockProperty<>("coral_hang_type_bit", true,
023            new CoralType[]{CoralType.BLUE, CoralType.PINK}
024    ).ordinal(true);
025
026    @PowerNukkitOnly
027    @Since("1.5.0.0-PN")
028    public static final ArrayBlockProperty<BlockFace> HANG_DIRECTION = new ArrayBlockProperty<>("coral_direction", false,
029            new BlockFace[]{BlockFace.WEST, BlockFace.EAST, BlockFace.NORTH, BlockFace.SOUTH}
030    ).ordinal(true);
031
032    @PowerNukkitOnly
033    @Since("1.5.0.0-PN")
034    public static final BlockProperties PROPERTIES = new BlockProperties(HANG1_TYPE, PERMANENTLY_DEAD, HANG_DIRECTION);
035
036    @PowerNukkitOnly
037    public BlockCoralFanHang() {
038        this(0);
039    }
040
041    @PowerNukkitOnly
042    public BlockCoralFanHang(int meta) {
043        super(meta);
044    }
045    
046    @Override
047    public int getId() {
048        return CORAL_FAN_HANG;
049    }
050
051    @Since("1.4.0.0-PN")
052    @PowerNukkitOnly
053    @Nonnull
054    @Override
055    public BlockProperties getProperties() {
056        return PROPERTIES;
057    }
058
059    @Override
060    public String getName() {
061        String name = super.getName();
062        name = name.substring(0, name.length() - 4);
063        if (isDead()) {
064            return "Dead " + name + " Wall Fan";
065        } else {
066            return name + " Wall Fan";
067        }
068    }
069    
070    @Override
071    public boolean isDead() {
072        return (getDamage() & 0b10) == 0b10;
073    }
074    
075    @Override
076    public int onUpdate(int type) {
077        if (type == Level.BLOCK_UPDATE_RANDOM) {
078            return type;
079        } else {
080            return super.onUpdate(type);
081        }
082    }
083    
084    @Override
085    public int getType() {
086        if ((getDamage() & 0b1) == 0) {
087            return BlockCoral.TYPE_TUBE;
088        } else {
089            return BlockCoral.TYPE_BRAIN;
090        }
091    }
092    
093    @Override
094    public BlockFace getBlockFace() {
095        int face = getDamage() >> 2 & 0x3;
096        switch (face) {
097            case 0:
098                return BlockFace.WEST;
099            case 1:
100                return BlockFace.EAST;
101            case 2:
102                return BlockFace.NORTH;
103            default:
104            case 3:
105                return BlockFace.SOUTH;
106        }
107    }
108    
109    @Override
110    public BlockFace getRootsFace() {
111        return getBlockFace().getOpposite();
112    }
113    
114    @Override
115    public Item toItem() {
116        return new ItemBlock(isDead()? new BlockCoralFanDead() : new BlockCoralFan(), getType());
117    }
118}
