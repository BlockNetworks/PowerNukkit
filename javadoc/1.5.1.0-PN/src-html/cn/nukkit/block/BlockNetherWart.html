001package cn.nukkit.block;
002
003import cn.nukkit.Player;
004import cn.nukkit.Server;
005import cn.nukkit.api.PowerNukkitOnly;
006import cn.nukkit.api.Since;
007import cn.nukkit.blockproperty.BlockProperties;
008import cn.nukkit.blockproperty.IntBlockProperty;
009import cn.nukkit.event.block.BlockGrowEvent;
010import cn.nukkit.item.Item;
011import cn.nukkit.item.ItemNetherWart;
012import cn.nukkit.level.Level;
013import cn.nukkit.math.BlockFace;
014import cn.nukkit.utils.BlockColor;
015
016import javax.annotation.Nonnull;
017import java.util.Random;
018
019/**
020 * @author Leonidius20
021 * @since 22.03.17
022 */
023public class BlockNetherWart extends BlockFlowable {
024
025    @PowerNukkitOnly
026    @Since("1.5.0.0-PN")
027    public static final IntBlockProperty AGE = new IntBlockProperty("age", false, 3);
028
029    @PowerNukkitOnly
030    @Since("1.5.0.0-PN")
031    public static final BlockProperties PROPERTIES = new BlockProperties(AGE);
032
033    public BlockNetherWart() {
034        this(0);
035    }
036
037    public BlockNetherWart(int meta) {
038        super(meta);
039    }
040
041    @Override
042    public boolean place(@Nonnull Item item, @Nonnull Block block, @Nonnull Block target, @Nonnull BlockFace face, double fx, double fy, double fz, Player player) {
043        Block down = this.down();
044        if (down.getId() == SOUL_SAND) {
045            this.getLevel().setBlock(block, this, true, true);
046            return true;
047        }
048        return false;
049    }
050
051    @Override
052    public int onUpdate(int type) {
053        if (type == Level.BLOCK_UPDATE_NORMAL) {
054            if (this.down().getId() != SOUL_SAND) {
055                this.getLevel().useBreakOn(this);
056                return Level.BLOCK_UPDATE_NORMAL;
057            }
058        } else if (type == Level.BLOCK_UPDATE_RANDOM) {
059            if (new Random().nextInt(10) == 1) {
060                if (this.getDamage() < 0x03) {
061                    BlockNetherWart block = (BlockNetherWart) this.clone();
062                    block.setDamage(block.getDamage() + 1);
063                    BlockGrowEvent ev = new BlockGrowEvent(this, block);
064                    Server.getInstance().getPluginManager().callEvent(ev);
065
066                    if (!ev.isCancelled()) {
067                        this.getLevel().setBlock(this, ev.getNewState(), true, true);
068                    } else {
069                        return Level.BLOCK_UPDATE_RANDOM;
070                    }
071                }
072            } else {
073                return Level.BLOCK_UPDATE_RANDOM;
074            }
075        }
076
077        return 0;
078    }
079
080    @Override
081    public BlockColor getColor() {
082        return BlockColor.RED_BLOCK_COLOR;
083    }
084
085    @Override
086    public String getName() {
087        return "Nether Wart Block";
088    }
089
090    @Override
091    public int getId() {
092        return NETHER_WART_BLOCK;
093    }
094
095    @Since("1.4.0.0-PN")
096    @PowerNukkitOnly
097    @Nonnull
098    @Override
099    public BlockProperties getProperties() {
100        return PROPERTIES;
101    }
102
103    @Override
104    public Item[] getDrops(Item item) {
105        if (this.getDamage() == 0x03) {
106            return new Item[]{
107                    new ItemNetherWart(0, 2 + (int) (Math.random() * ((4 - 2) + 1)))
108            };
109        } else {
110            return new Item[]{
111                    new ItemNetherWart()
112            };
113        }
114    }
115
116    @Override
117    public Item toItem() {
118        return new ItemNetherWart();
119    }
120}
