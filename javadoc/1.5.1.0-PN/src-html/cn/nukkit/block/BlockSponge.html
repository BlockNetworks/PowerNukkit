024/**
025 * @author Angelic47 (Nukkit Project)
026 */
027public class BlockSponge extends BlockSolidMeta {
028
029    @PowerNukkitOnly
030    @Since("1.5.0.0-PN")
031    public static final ArrayBlockProperty<SpongeType> SPONGE_TYPE = new ArrayBlockProperty<>("sponge_type", true, SpongeType.class);
032
033    @PowerNukkitOnly
034    @Since("1.5.0.0-PN")
035    public static final BlockProperties PROPERTIES = new BlockProperties(SPONGE_TYPE);
036
037    public static final int DRY = 0;
038    public static final int WET = 1;
039    private static final String[] NAMES = new String[]{
040            "Sponge",
041            "Wet sponge"
042    };
043
044    public BlockSponge() {
045        this(0);
046    }
047
048    public BlockSponge(int meta) {
049        super(meta);
050    }
051
052    @Override
053    public int getId() {
054        return SPONGE;
055    }
056
057    @Since("1.4.0.0-PN")
058    @PowerNukkitOnly
059    @Nonnull
060    @Override
061    public BlockProperties getProperties() {
062        return PROPERTIES;
063    }
064
065    @Override
066    public double getHardness() {
067        return 0.6;
068    }
069
070    @Override
071    public double getResistance() {
072        return 3;
073    }
074
075    @Override
076    public int getToolType() {
077        return ItemTool.TYPE_HOE;
078    }
079
080    @Override
081    public String getName() {
082        return NAMES[this.getDamage() & 0b1];
083    }
084
085    @Override
086    public BlockColor getColor() {
087        return BlockColor.YELLOW_BLOCK_COLOR;
088    }
089
090    @Override
091    public boolean place(@Nonnull Item item, @Nonnull Block block, @Nonnull Block target, @Nonnull BlockFace face, double fx, double fy, double fz, Player player) {
092        if (this.getDamage() == WET && level.getDimension() == Level.DIMENSION_NETHER) {
093            level.setBlock(block, Block.get(BlockID.SPONGE, DRY), true, true);
094            this.getLevel().addLevelEvent(block.add(0.5, 0.875, 0.5), LevelEventPacket.EVENT_SOUND_EXPLODE);
095
096            ThreadLocalRandom random = ThreadLocalRandom.current();
097            for (int i = 0; i < 8; ++i) {
098                level.addParticle(new CloudParticle(block.getLocation().add(random.nextDouble(), 1, random.nextDouble())));
099            }
100
101            return true;
102        } else if (this.getDamage() == DRY && block instanceof BlockWater && performWaterAbsorb(block)) {
103            level.setBlock(block, Block.get(BlockID.SPONGE, WET), true, true);
104
105            for (int i = 0; i < 4; i++) {
106                LevelEventPacket packet = new LevelEventPacket();
107                packet.evid = LevelEventPacket.EVENT_PARTICLE_DESTROY;
108                packet.x = (float) block.getX() + 0.5f;
109                packet.y = (float) block.getY() + 1f;
110                packet.z = (float) block.getZ() + 0.5f;
111                packet.data = GlobalBlockPalette.getOrCreateRuntimeId(BlockID.WATER, 0);
112                level.addChunkPacket(getChunkX(), getChunkZ(), packet);
113            }
114
115            return true;
116        }
117
118        return super.place(item, block, target, face, fx, fy, fz, player);
119    }
120
121    private boolean performWaterAbsorb(Block block) {
122        Queue<Entry> entries = new ArrayDeque<>();
123
124        entries.add(new Entry(block, 0));
125
126        Entry entry;
127        int waterRemoved = 0;
128        while (waterRemoved < 64 && (entry = entries.poll()) != null) {
129            for (BlockFace face : BlockFace.values()) {
130                Block faceBlock = entry.block.getSideAtLayer(0, face);
131                Block faceBlock1 = faceBlock.getLevelBlockAtLayer(1);
132                
133                if (faceBlock instanceof BlockWater) {
134                    this.getLevel().setBlockStateAt(faceBlock.getFloorX(), faceBlock.getFloorY(), faceBlock.getFloorZ(), BlockState.AIR);
135                    this.getLevel().updateAround(faceBlock);
136                    waterRemoved++;
137                    if (entry.distance < 6) {
138                        entries.add(new Entry(faceBlock, entry.distance + 1));
139                    }
140                } else if (faceBlock1 instanceof BlockWater) {
141                    if (faceBlock.getId() == BlockID.BLOCK_KELP || faceBlock.getId() == BlockID.SEAGRASS || faceBlock.getId() == BlockID.SEA_PICKLE || faceBlock instanceof BlockCoralFan) {
142                        faceBlock.getLevel().useBreakOn(faceBlock);
143                    }
144                    this.getLevel().setBlockStateAt(faceBlock1.getFloorX(), faceBlock1.getFloorY(), faceBlock1.getFloorZ(), 1, BlockState.AIR);
145                    this.getLevel().updateAround(faceBlock1);
146                    waterRemoved++;
147                    if (entry.distance < 6) {
148                        entries.add(new Entry(faceBlock1, entry.distance + 1));
149                    }
150                }
151            }
152        }
153        
154        return waterRemoved > 0;
155    }
156
157    private static class Entry {
158        private final Block block;
159        private final int distance;
160
161        public Entry(Block block, int distance) {
162            this.block = block;
163            this.distance = distance;
164        }
165    }
166}
