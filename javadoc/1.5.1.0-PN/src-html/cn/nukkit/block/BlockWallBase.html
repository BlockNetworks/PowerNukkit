097    private boolean shouldBeTall(Block above, BlockFace face) {
098        switch (above.getId()) {
099        case AIR: 
100        case SKULL_BLOCK: 
101            return false;
102        // If the bell is standing and follow the path, make it tall
103        case BELL: 
104            BlockBell bell = (BlockBell) above;
105            return bell.getAttachment() == AttachmentType.STANDING && bell.getBlockFace().getAxis() != face.getAxis();
106        default: 
107            if (above instanceof BlockWallBase) {
108                return ((BlockWallBase) above).getConnectionType(face) != WallConnectionType.NONE;
109            } else if (above instanceof BlockConnectable) {
110                return ((BlockConnectable) above).isConnected(face);
111            } else if (above instanceof BlockPressurePlateBase || above instanceof BlockStairs) {
112                return true;
113            }
114            return above.isSolid() && !above.isTransparent() || shouldBeTallBasedOnBoundingBox(above, face);
115        }
116    }

118    private boolean shouldBeTallBasedOnBoundingBox(Block above, BlockFace face) {
119        AxisAlignedBB boundingBox = above.getBoundingBox();
120        if (boundingBox == null) {
121            return false;
122        }
123        boundingBox = boundingBox.getOffsetBoundingBox(-above.x, -above.y, -above.z);
124        if (boundingBox.getMinY() > 0) {
125            return false;
126        }
127        int offset = face.getXOffset();
128        if (offset < 0) {
129            return boundingBox.getMinX() < MIN_POST_BB && boundingBox.getMinZ() < MIN_POST_BB && MAX_POST_BB < boundingBox.getMaxZ();
130        } else if (offset > 0) {
131            return MAX_POST_BB < boundingBox.getMaxX() && MAX_POST_BB < boundingBox.getMaxZ() && boundingBox.getMinZ() < MAX_POST_BB;
132        } else {
133            offset = face.getZOffset();
134            if (offset < 0) {
135                return boundingBox.getMinZ() < MIN_POST_BB && boundingBox.getMinX() < MIN_POST_BB && MIN_POST_BB < boundingBox.getMaxX();
136            } else if (offset > 0) {
137                return MAX_POST_BB < boundingBox.getMaxZ() && MAX_POST_BB < boundingBox.getMaxX() && boundingBox.getMinX() < MAX_POST_BB;
138            }
139        }
140        return false;
141    }

272    private boolean recheckPostConditions(Block above) {
273        // If nothing is connected, it should be a post
274        if (!hasConnections()) {
275            return true;
276        }
277        // If it's not straight, it should be a post
278        Map<BlockFace, WallConnectionType> connections = getWallConnections();
279        if (connections.size() != 2) {
280            return true;
281        }
282        Iterator<Map.Entry<BlockFace, WallConnectionType>> iterator = connections.entrySet().iterator();
283        Map.Entry<BlockFace, WallConnectionType> entryA = iterator.next();
284        Map.Entry<BlockFace, WallConnectionType> entryB = iterator.next();
285        if (entryA.getValue() != entryB.getValue() || entryA.getKey().getOpposite() != entryB.getKey()) {
286            return true;
287        }
288        BlockFace.Axis axis = entryA.getKey().getAxis();
289        switch (above.getId()) {
290        // These special blocks forces walls to become a post
291        case FLOWER_POT_BLOCK: 
292        case SKULL_BLOCK: 
293        case CONDUIT: 
294        case STANDING_BANNER: 
295        case TURTLE_EGG: 
296            return true;
297        // End rods make it become a post if it's placed on the wall
298        case END_ROD: 
299            if (((Faceable) above).getBlockFace() == BlockFace.UP) {
300                return true;
301            }
302            break;
303        // If the bell is standing and don't follow the path, make it a post
304        case BELL: 
305            BlockBell bell = (BlockBell) above;
306            if (bell.getAttachment() == AttachmentType.STANDING && bell.getBlockFace().getAxis() == axis) {
307                return true;
308            }
309            break;
310        default: 
311            if (above instanceof BlockWallBase) {
312                // If the wall above is a post, it should also be a post
313                if (((BlockWallBase) above).isWallPost()) {
314                    return true;
315                }
316            } else if (above instanceof BlockLantern) {
317                // Lanterns makes this become a post if they are not hanging
318                if (!((BlockLantern) above).isHanging()) {
319                    return true;
320                }
321            } else if (above.getId() == LEVER || above instanceof BlockTorch || above instanceof BlockButton) {
322                // These blocks make this become a post if they are placed down (facing up)
323                if (((Faceable) above).getBlockFace() == BlockFace.UP) {
324                    return true;
325                }
326            } else if (above instanceof BlockFenceGate) {
327                // If the gate don't follow the path, make it a post
328                if (((Faceable) above).getBlockFace().getAxis() == axis) {
329                    return true;
330                }
331            } else if (above instanceof BlockConnectable) {
332                // If the connectable block above don't share 2 equal connections, then this should be a post
333                int shared = 0;
334                for (BlockFace connection : ((BlockConnectable) above).getConnections()) {
335                    if (connections.containsKey(connection) && ++shared == 2) {
336                        break;
337                    }
338                }
339                if (shared < 2) {
340                    return true;
341                }
342            }
343        }
344        // Sign posts always makes the wall become a post
345        return above instanceof BlockSignPost;
346    }

377    private boolean connect(BlockFace blockFace, Block above, boolean recheckPost) {
378        WallConnectionType type = shouldBeTall(above, blockFace) ? WallConnectionType.TALL : WallConnectionType.SHORT;
379        if (setConnection(blockFace, type)) {
380            if (recheckPost) {
381                recheckPostConditions(above);
382            }
383            return true;
384        }
385        return false;
386    }

401    @Override
402    protected AxisAlignedBB recalculateBoundingBox() {
403        boolean north = this.canConnect(this.getSide(BlockFace.NORTH));
404        boolean south = this.canConnect(this.getSide(BlockFace.SOUTH));
405        boolean west = this.canConnect(this.getSide(BlockFace.WEST));
406        boolean east = this.canConnect(this.getSide(BlockFace.EAST));
407        double n = north ? 0 : 0.25;
408        double s = south ? 1 : 0.75;
409        double w = west ? 0 : 0.25;
410        double e = east ? 1 : 0.75;
411        if (north && south && !west && !east) {
412            w = 0.3125;
413            e = 0.6875;
414        } else if (!north && !south && west && east) {
415            n = 0.3125;
416            s = 0.6875;
417        }
418        return new SimpleAxisAlignedBB(this.x + w, this.y, this.z + n, this.x + e, this.y + 1.5, this.z + s);
419    }

423    public boolean canConnect(Block block) {
424        switch (block.getId()) {
425        case GLASS_PANE: 
426        case STAINED_GLASS_PANE: 
427        case IRON_BARS: 
428        case GLASS: 
429        case STAINED_GLASS: 
430            return true;
431        default: 
432            if (block instanceof BlockWallBase) {
433                return true;
434            }
435            if (block instanceof BlockFenceGate) {
436                BlockFenceGate fenceGate = (BlockFenceGate) block;
437                return fenceGate.getBlockFace().getAxis() != calculateAxis(this, block);
438            }
439            if (block instanceof BlockStairs) {
440                return ((BlockStairs) block).getBlockFace().getOpposite() == calculateFace(this, block);
441            }
442            if (block instanceof BlockTrapdoor) {
443                BlockTrapdoor trapdoor = (BlockTrapdoor) block;
444                return trapdoor.isOpen() && trapdoor.getBlockFace() == calculateFace(this, trapdoor);
445            }
446            return block.isSolid() && !block.isTransparent();
447        }
