001package cn.nukkit.block;
002
003import cn.nukkit.Player;
004import cn.nukkit.api.PowerNukkitOnly;
005import cn.nukkit.api.Since;
006import cn.nukkit.blockproperty.ArrayBlockProperty;
007import cn.nukkit.blockproperty.BlockProperties;
008import cn.nukkit.item.Item;
009import cn.nukkit.item.ItemTool;
010import cn.nukkit.math.BlockFace;
011import cn.nukkit.utils.BlockColor;
012import cn.nukkit.utils.Faceable;
013
014import javax.annotation.Nonnull;
015
016import static cn.nukkit.blockproperty.CommonBlockProperties.DEPRECATED;
017
018/**
019 * @author CreeperFace
020 */
021public class BlockBone extends BlockSolidMeta implements Faceable {
022    private static final ArrayBlockProperty<String> SPECIAL_PILLAR_AXIS = new ArrayBlockProperty<>("pillar_axis", false,
023        new String[] {
024                "y",
025                "unused1",
026                "unused2",
027                "unused3",
028                "x",
029                "unused5",
030                "unused6",
031                "unused7",
032                "z",
033        }
034    );
035
036    @PowerNukkitOnly
037    @Since("1.5.0.0-PN")
038    public static final BlockProperties PROPERTIES = new BlockProperties(SPECIAL_PILLAR_AXIS, DEPRECATED);
039
040    private static final int[] FACES = {
041            0,
042            0,
043            0b1000,
044            0b1000,
045            0b0100,
046            0b0100
047    };
048
049    public BlockBone() {
050        this(0);
051    }
052
053    public BlockBone(int meta) {
054        super(meta);
055    }
056
057    @Override
058    public int getId() {
059        return BONE_BLOCK;
060    }
061
062    @Since("1.4.0.0-PN")
063    @PowerNukkitOnly
064    @Nonnull
065    @Override
066    public BlockProperties getProperties() {
067        return PROPERTIES;
068    }
069
070    @Override
071    public String getName() {
072        return "Bone Block";
073    }
074
075    @Override
076    public double getHardness() {
077        return 2;
078    }
079
080    @Override
081    public double getResistance() {
082        return 10;
083    }
084
085    @Override
086    public int getToolType() {
087        return ItemTool.TYPE_PICKAXE;
088    }
089
090    @Since("1.4.0.0-PN")
091    @PowerNukkitOnly
092    @Override
093    public int getToolTier() {
094        return ItemTool.TIER_WOODEN;
095    }
096
097    @Override
098    public BlockFace getBlockFace() {
099        return BlockFace.fromHorizontalIndex(this.getDamage() & 0x7);
100    }
101
102    @Override
103    public boolean place(@Nonnull Item item, @Nonnull Block block, @Nonnull Block target, @Nonnull BlockFace face, double fx, double fy, double fz, Player player) {
104        this.setDamage(((this.getDamage() & 0x3) | FACES[face.getIndex()]));
105        this.getLevel().setBlock(block, this, true);
106        return true;
107    }
108
109    @Override
110    public BlockColor getColor() {
111        return BlockColor.SAND_BLOCK_COLOR;
112    }
113}
