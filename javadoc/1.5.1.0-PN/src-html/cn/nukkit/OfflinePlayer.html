001package cn.nukkit;
002
003import cn.nukkit.metadata.MetadataValue;
004import cn.nukkit.nbt.tag.CompoundTag;
005import cn.nukkit.plugin.Plugin;
006
007import java.util.List;
008import java.util.UUID;
009
010/**
011 * 描述一个不在线的玩家的类。<br>
012 * Describes an offline player.
013 *
014 * @author MagicDroidX(code) @ Nukkit Project
015 * @author 粉鞋大妈(javadoc) @ Nukkit Project
016 * @see cn.nukkit.Player
017 * @since Nukkit 1.0 | Nukkit API 1.0.0
018 */
019public class OfflinePlayer implements IPlayer {
020    private final Server server;
021    private final CompoundTag namedTag;
022
023    /**
024     * 初始化这个{@code OfflinePlayer}对象。<br>
025     * Initializes the object {@code OfflinePlayer}.
026     *
027     * @param server 这个玩家所在服务器的{@code Server}对象。<br>
028     *               The server this player is in, as a {@code Server} object.
029     * @param uuid   这个玩家的UUID。<br>
030     *               UUID of this player.
031     * @since Nukkit 1.0 | Nukkit API 1.0.0
032     */
033    public OfflinePlayer(Server server, UUID uuid) {
034        this(server, uuid, null);
035    }
036
037    public OfflinePlayer(Server server, String name) {
038        this(server, null, name);
039    }
040
041    public OfflinePlayer(Server server, UUID uuid, String name) {
042        this.server = server;
043
044        CompoundTag nbt;
045        if (uuid != null) {
046            nbt = this.server.getOfflinePlayerData(uuid, false);
047        } else if (name != null) {
048            nbt = this.server.getOfflinePlayerData(name, false);
049        } else {
050            throw new IllegalArgumentException("Name and UUID cannot both be null");
051        }
052        if (nbt == null) {
053            nbt = new CompoundTag();
054        }
055        this.namedTag = nbt;
056
057        if (uuid != null) {
058            this.namedTag.putLong("UUIDMost", uuid.getMostSignificantBits());
059            this.namedTag.putLong("UUIDLeast", uuid.getLeastSignificantBits());
060        } else {
061            this.namedTag.putString("NameTag", name);
062        }
063    }
064
065    @Override
066    public boolean isOnline() {
067        return this.getPlayer() != null;
068    }
069
070    @Override
071    public String getName() {
072        if (namedTag != null && namedTag.contains("NameTag")) {
073            return namedTag.getString("NameTag");
074        }
075        return null;
076    }
077
078    @Override
079    public UUID getUniqueId() {
080        if (namedTag != null) {
081            long least = namedTag.getLong("UUIDLeast");
082            long most = namedTag.getLong("UUIDMost");
083
084            if (least != 0 && most != 0) {
085                return new UUID(most, least);
086            }
087        }
088        return null;
089    }
090
091    public Server getServer() {
092        return server;
093    }
094
095    @Override
096    public boolean isOp() {
097        return this.server.isOp(this.getName().toLowerCase());
098    }
099
100    @Override
101    public void setOp(boolean value) {
102        if (value == this.isOp()) {
103            return;
104        }
105
106        if (value) {
107            this.server.addOp(this.getName().toLowerCase());
108        } else {
109            this.server.removeOp(this.getName().toLowerCase());
110        }
111    }
112
113    @Override
114    public boolean isBanned() {
115        return this.server.getNameBans().isBanned(this.getName());
116    }
117
118    @Override
119    public void setBanned(boolean value) {
120        if (value) {
121            this.server.getNameBans().addBan(this.getName(), null, null, null);
122        } else {
123            this.server.getNameBans().remove(this.getName());
124        }
125    }
126
127    @Override
128    public boolean isWhitelisted() {
129        return this.server.isWhitelisted(this.getName().toLowerCase());
130    }
131
132    @Override
133    public void setWhitelisted(boolean value) {
134        if (value) {
135            this.server.addWhitelist(this.getName().toLowerCase());
136        } else {
137            this.server.removeWhitelist(this.getName().toLowerCase());
138        }
139    }
140
141    @Override
142    public Player getPlayer() {
143        return this.server.getPlayerExact(this.getName());
144    }
145
146    @Override
147    public Long getFirstPlayed() {
148        return this.namedTag != null ? this.namedTag.getLong("firstPlayed") : null;
149    }
150
151    @Override
152    public Long getLastPlayed() {
153        return this.namedTag != null ? this.namedTag.getLong("lastPlayed") : null;
154    }
155
156    @Override
157    public boolean hasPlayedBefore() {
158        return this.namedTag != null;
159    }
160
161    public void setMetadata(String metadataKey, MetadataValue newMetadataValue) {
162        this.server.getPlayerMetadata().setMetadata(this, metadataKey, newMetadataValue);
163    }
164
165    public List<MetadataValue> getMetadata(String metadataKey) {
166        return this.server.getPlayerMetadata().getMetadata(this, metadataKey);
167    }
168
169    public boolean hasMetadata(String metadataKey) {
170        return this.server.getPlayerMetadata().hasMetadata(this, metadataKey);
171    }
172
173    public void removeMetadata(String metadataKey, Plugin owningPlugin) {
174        this.server.getPlayerMetadata().removeMetadata(this, metadataKey, owningPlugin);
175    }
176
177}
