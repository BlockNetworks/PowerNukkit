187        if (taskMap.containsKey(taskId)) {

190            } catch (RuntimeException ex) {

196    public void cancelTask(Plugin plugin) {
197        if (plugin == null) {
198            throw new NullPointerException("Plugin cannot be null!");
199        }
200        for (Map.Entry<Integer, TaskHandler> entry : taskMap.entrySet()) {
201            TaskHandler taskHandler = entry.getValue();
202            // TODO: Remove the "taskHandler.getPlugin() == null" check
203            // It is only there for backwards compatibility!
204            if (taskHandler.getPlugin() == null || plugin.equals(taskHandler.getPlugin())) {
205                try {
206                    taskHandler.cancel(); /* It will remove from task map automatic in next main heartbeat. */
207                } catch (RuntimeException ex) {
208                    log.fatal("Exception while invoking onCancel", ex);
209                }
210            }
211        }
212    }

231    private TaskHandler addTask(Task task, int delay, int period, boolean asynchronous) {
232        return addTask(task instanceof PluginTask ? ((PluginTask) task).getOwner() : null, task, delay, period, asynchronous);
233    }

235    private TaskHandler addTask(Plugin plugin, Runnable task, int delay, int period, boolean asynchronous) {
236        if (plugin != null && plugin.isDisabled()) {
237            throw new PluginException("Plugin \'" + plugin.getName() + "\' attempted to register a task while disabled.");
238        }
239        if (delay < 0 || period < 0) {
240            throw new PluginException("Attempted to register a task with negative delay or period.");
241        }
242        TaskHandler taskHandler = new TaskHandler(plugin, task, nextTaskId(), asynchronous);
243        taskHandler.setDelay(delay);
244        taskHandler.setPeriod(period);
245        taskHandler.setNextRunTick(taskHandler.isDelayed() ? currentTick + taskHandler.getDelay() : currentTick);
246        if (task instanceof Task) {
247            ((Task) task).setHandler(taskHandler);
248        }
249        pending.offer(taskHandler);
250        taskMap.put(taskHandler.getTaskId(), taskHandler);
251        return taskHandler;
252    }

254    public void mainThreadHeartbeat(int currentTick) {
255        // Accepts pending.
256        TaskHandler task;
257        while ((task = pending.poll()) != null) {
258            int tick = Math.max(currentTick, task.getNextRunTick()); // Do not schedule in the past
259            ArrayDeque<TaskHandler> queue = Utils.getOrCreate(queueMap, ArrayDeque.class, tick);
260            queue.add(task);
261        }
262        if (currentTick - this.currentTick > queueMap.size()) {
263            // A large number of ticks have passed since the last execution
264            for (Map.Entry<Integer, ArrayDeque<TaskHandler>> entry : queueMap.entrySet()) {
265                int tick = entry.getKey();
266                if (tick <= currentTick) {
267                    runTasks(tick);
268                }
269            }
270        } else {
271            // Normal server tick
272            for (int i = this.currentTick + 1; i <= currentTick; i++) {
273                runTasks(currentTick);
274            }
275        }
276        this.currentTick = currentTick;
277        AsyncTask.collectTask();
278    }

280    private void runTasks(int currentTick) {
281        ArrayDeque<TaskHandler> queue = queueMap.remove(currentTick);
282        if (queue != null) {
283            for (TaskHandler taskHandler : queue) {
284                if (taskHandler.isCancelled()) {
285                    taskMap.remove(taskHandler.getTaskId());
286                    continue;
287                } else if (taskHandler.isAsynchronous()) {
288                    asyncPool.execute(taskHandler.getTask());
289                } else {
290                    taskHandler.timing.startTiming();
291                    try {
292                        taskHandler.run(currentTick);
293                    } catch (Throwable e) {
294                        log.fatal("Could not execute taskHandler {}", taskHandler.getTaskId(), e);
295                    }
296                    taskHandler.timing.stopTiming();
297                }
298                if (taskHandler.isRepeating()) {
299                    taskHandler.setNextRunTick(currentTick + taskHandler.getPeriod());
300                    pending.offer(taskHandler);
301                } else {
302                    try {
303                        TaskHandler removed = taskMap.remove(taskHandler.getTaskId());
304                        if (removed != null) removed.cancel();
305                    } catch (RuntimeException ex) {
306                        log.fatal("Exception while invoking onCancel", ex);
307                    }
308                }
309            }
310        }
311    }
