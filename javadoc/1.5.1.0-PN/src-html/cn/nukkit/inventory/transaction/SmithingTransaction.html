019package cn.nukkit.inventory.transaction;
020
021import cn.nukkit.Player;
022import cn.nukkit.api.PowerNukkitOnly;
023import cn.nukkit.api.Since;
024import cn.nukkit.event.inventory.SmithingTableEvent;
025import cn.nukkit.inventory.Inventory;
026import cn.nukkit.inventory.SmithingInventory;
027import cn.nukkit.inventory.transaction.action.CreativeInventoryAction;
028import cn.nukkit.inventory.transaction.action.InventoryAction;
029import cn.nukkit.inventory.transaction.action.SmithingItemAction;
030import cn.nukkit.item.Item;
031import cn.nukkit.item.ItemID;
032
033import java.util.List;
034
035/**
036 * @author joserobjr
037 * @since 2021-05-16
038 */
039@PowerNukkitOnly
040@Since("1.4.0.0-PN")
041public class SmithingTransaction extends InventoryTransaction {
042
043    private Item equipmentItem;
044    private Item ingredientItem;
045    private Item outputItem;
046
047    @PowerNukkitOnly
048    @Since("1.4.0.0-PN")
049    public SmithingTransaction(Player source, List<InventoryAction> actions) {
050        super(source, actions);
051    }
052
053    @Override
054    public void addAction(InventoryAction action) {
055        super.addAction(action);
056        if (action instanceof SmithingItemAction) {
057            switch (((SmithingItemAction) action).getType()) {
058                case 0: // input
059                    this.equipmentItem = action.getTargetItem();
060                    break;
061                case 2: // result
062                    this.outputItem = action.getSourceItem();
063                    break;
064                case 1: // ingredient
065                    this.ingredientItem = action.getTargetItem();
066                    break;
067            }
068        } else if (action instanceof CreativeInventoryAction) {
069            CreativeInventoryAction creativeAction = (CreativeInventoryAction) action;
070            if (creativeAction.getActionType() == 0
071                    && creativeAction.getSourceItem().isNull()
072                    && !creativeAction.getTargetItem().isNull() && creativeAction.getTargetItem().getId() == ItemID.NETHERITE_INGOT) {
073                this.ingredientItem = action.getTargetItem();
074            }
075        }
076    }
077
078    @Override
079    public boolean canExecute() {
080        Inventory inventory = getSource().getWindowById(Player.SMITHING_WINDOW_ID);
081        if (inventory == null) {
082            return false;
083        }
084        SmithingInventory grindstoneInventory = (SmithingInventory) inventory;
085        if (outputItem == null || outputItem.isNull() || 
086                ((equipmentItem == null || equipmentItem.isNull()) && (ingredientItem == null || ingredientItem.isNull()))) {
087            return false;
088        }
089        
090        Item air = Item.get(0);
091        Item equipment = equipmentItem != null? equipmentItem : air;
092        Item ingredient = ingredientItem != null? ingredientItem : air;
093        
094        return equipment.equals(grindstoneInventory.getEquipment(), true, true)
095                && ingredient.equals(grindstoneInventory.getIngredient(), true, true)
096                && outputItem.equals(grindstoneInventory.getResult(), true, true);
097    }
098
099    @Override
100    public boolean execute() {
101        if (this.hasExecuted() || !this.canExecute()) {
102            this.source.removeAllWindows(false);
103            this.sendInventories();
104            return false;
105        }
106        SmithingInventory inventory = (SmithingInventory) getSource().getWindowById(Player.SMITHING_WINDOW_ID);
107        Item air = Item.get(0);
108        Item equipment = equipmentItem != null? equipmentItem : air;
109        Item ingredient = ingredientItem != null? ingredientItem : air;
110        SmithingTableEvent event = new SmithingTableEvent(inventory, equipment, outputItem, ingredient, source);
111        this.source.getServer().getPluginManager().callEvent(event);
112        if (event.isCancelled()) {
113            this.source.removeAllWindows(false);
114            this.sendInventories();
115            return true;
116        }
117
118        for (InventoryAction action : this.actions) {
119            if (action.execute(this.source)) {
120                action.onExecuteSuccess(this.source);
121            } else {
122                action.onExecuteFail(this.source);
123            }
124        }
125        if (!source.isCreative()) {
126            source.sendAllInventories();
127            source.getCursorInventory().sendContents(source);
128        }
129        return true;
130    }
131
132    @PowerNukkitOnly
133    @Since("1.4.0.0-PN")
134    public Item getEquipmentItem() {
135        return equipmentItem == null? null : equipmentItem.clone();
136    }
137
138    @PowerNukkitOnly
139    @Since("1.4.0.0-PN")
140    public Item getIngredientItem() {
141        return ingredientItem == null? null : ingredientItem.clone();
142    }
143
144    @PowerNukkitOnly
145    @Since("1.4.0.0-PN")
146    public Item getOutputItem() {
147        return outputItem == null? null : outputItem.clone();
148    }
149
150    @PowerNukkitOnly
151    @Since("1.4.0.0-PN")
152    public static boolean checkForItemPart(List<InventoryAction> actions) {
153        return actions.stream().anyMatch(it-> it instanceof SmithingItemAction);
154    }
155}
