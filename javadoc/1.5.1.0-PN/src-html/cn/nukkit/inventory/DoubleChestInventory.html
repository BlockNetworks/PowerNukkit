001package cn.nukkit.inventory;
002
003import cn.nukkit.Player;
004import cn.nukkit.api.PowerNukkitDifference;
005import cn.nukkit.blockentity.BlockEntityChest;
006import cn.nukkit.item.Item;
007import cn.nukkit.level.Level;
008import cn.nukkit.level.Sound;
009import cn.nukkit.network.protocol.BlockEventPacket;
010import cn.nukkit.network.protocol.InventorySlotPacket;
011
012import java.util.HashMap;
013import java.util.Map;
014
015/**
016 * @author MagicDroidX (Nukkit Project)
017 */
018public class DoubleChestInventory extends ContainerInventory implements InventoryHolder {
019
020    private final ChestInventory left;
021    private final ChestInventory right;
022
023    public DoubleChestInventory(BlockEntityChest left, BlockEntityChest right) {
024        super(null, InventoryType.DOUBLE_CHEST);
025        this.holder = this;
026
027        this.left = left.getRealInventory();
028        this.left.setDoubleInventory(this);
029
030        this.right = right.getRealInventory();
031        this.right.setDoubleInventory(this);
032
033        Map<Integer, Item> items = new HashMap<>();
034        // First we add the items from the left chest
035        for (int idx = 0; idx < this.left.getSize(); idx++) {
036            if (this.left.getContents().containsKey(idx)) { // Don't forget to skip empty slots!
037                items.put(idx, this.left.getContents().get(idx));
038            }
039        }
040        // And them the items from the right chest
041        for (int idx = 0; idx < this.right.getSize(); idx++) {
042            if (this.right.getContents().containsKey(idx)) { // Don't forget to skip empty slots!
043                items.put(idx + this.left.getSize(), this.right.getContents().get(idx)); // idx + this.left.getSize() so we don't overlap left chest items
044            }
045        }
046
047        this.setContents(items);
048    }
049
050    @Override
051    public Inventory getInventory() {
052        return this;
053    }
054
055    @Override
056    public BlockEntityChest getHolder() {
057        return this.left.getHolder();
058    }
059
060    @Override
061    public Item getItem(int index) {
062        return index < this.left.getSize() ? this.left.getItem(index) : this.right.getItem(index - this.right.getSize());
063    }
064
065    @Override
066    public boolean setItem(int index, Item item, boolean send) {
067        return index < this.left.getSize() ? this.left.setItem(index, item, send) : this.right.setItem(index - this.right.getSize(), item, send);
068    }
069
070    @Override
071    public boolean clear(int index, boolean send) {
072        return index < this.left.getSize() ? this.left.clear(index, send) : this.right.clear(index - this.right.getSize(), send);
073    }
074
075    @Override
076    public Map<Integer, Item> getContents() {
077        Map<Integer, Item> contents = new HashMap<>();
078
079        for (int i = 0; i < this.getSize(); ++i) {
080            contents.put(i, this.getItem(i));
081        }
082
083        return contents;
084    }
085
086    @Override
087    public void setContents(Map<Integer, Item> items) {
088        if (items.size() > this.size) {
089            Map<Integer, Item> newItems = new HashMap<>();
090            for (int i = 0; i < this.size; i++) {
091                newItems.put(i, items.get(i));
092            }
093            items = newItems;
094        }
095
096        for (int i = 0; i < this.size; i++) {
097            if (!items.containsKey(i)) {
098                if (i < this.left.size) {
099                    if (this.left.slots.containsKey(i)) {
100                        this.clear(i);
101                    }
102                } else if (this.right.slots.containsKey(i - this.left.size)) {
103                    this.clear(i);
104                }
105            } else if (!this.setItem(i, items.get(i))) {
106                this.clear(i);
107            }
108        }
109    }
110
111    @PowerNukkitDifference(info = "Using new method to play sounds", since = "1.4.0.0-PN")
112    @Override
113    public void onOpen(Player who) {
114        super.onOpen(who);
115        this.left.viewers.add(who);
116        this.right.viewers.add(who);
117
118        if (this.getViewers().size() == 1) {
119            BlockEventPacket pk1 = new BlockEventPacket();
120            pk1.x = (int) this.left.getHolder().getX();
121            pk1.y = (int) this.left.getHolder().getY();
122            pk1.z = (int) this.left.getHolder().getZ();
123            pk1.case1 = 1;
124            pk1.case2 = 2;
125            Level level = this.left.getHolder().getLevel();
126            if (level != null) {
127                level.addSound(this.left.getHolder().add(0.5, 0.5, 0.5), Sound.RANDOM_CHESTOPEN);
128                level.addChunkPacket((int) this.left.getHolder().getX() >> 4, (int) this.left.getHolder().getZ() >> 4, pk1);
129            }
130
131            BlockEventPacket pk2 = new BlockEventPacket();
132            pk2.x = (int) this.right.getHolder().getX();
133            pk2.y = (int) this.right.getHolder().getY();
134            pk2.z = (int) this.right.getHolder().getZ();
135            pk2.case1 = 1;
136            pk2.case2 = 2;
137
138            level = this.right.getHolder().getLevel();
139            if (level != null) {
140                level.addSound(this.right.getHolder().add(0.5, 0.5, 0.5), Sound.RANDOM_CHESTOPEN);
141                level.addChunkPacket((int) this.right.getHolder().getX() >> 4, (int) this.right.getHolder().getZ() >> 4, pk2);
142            }
143        }
144    }
145
146    @PowerNukkitDifference(info = "Using new method to play sounds", since = "1.4.0.0-PN")
147    @Override
148    public void onClose(Player who) {
149        if (this.getViewers().size() == 1) {
150            BlockEventPacket pk1 = new BlockEventPacket();
151            pk1.x = (int) this.right.getHolder().getX();
152            pk1.y = (int) this.right.getHolder().getY();
153            pk1.z = (int) this.right.getHolder().getZ();
154            pk1.case1 = 1;
155            pk1.case2 = 0;
156
157            Level level = this.right.getHolder().getLevel();
158            if (level != null) {
159                level.addSound(this.right.getHolder().add(0.5, 0.5, 0.5), Sound.RANDOM_CHESTCLOSED);
160                level.addChunkPacket((int) this.right.getHolder().getX() >> 4, (int) this.right.getHolder().getZ() >> 4, pk1);
161            }
162
163            BlockEventPacket pk2 = new BlockEventPacket();
164            pk2.x = (int) this.left.getHolder().getX();
165            pk2.y = (int) this.left.getHolder().getY();
166            pk2.z = (int) this.left.getHolder().getZ();
167            pk2.case1 = 1;
168            pk2.case2 = 0;
169
170            level = this.left.getHolder().getLevel();
171            if (level != null) {
172                level.addSound(this.left.getHolder().add(0.5, 0.5, 0.5), Sound.RANDOM_CHESTCLOSED);
173                level.addChunkPacket((int) this.left.getHolder().getX() >> 4, (int) this.left.getHolder().getZ() >> 4, pk2);
174            }
175        }
176
177        this.left.viewers.remove(who);
178        this.right.viewers.remove(who);
179        super.onClose(who);
180    }
181
182    public ChestInventory getLeftSide() {
183        return this.left;
184    }
185
186    public ChestInventory getRightSide() {
187        return this.right;
188    }
189
190    public void sendSlot(Inventory inv, int index, Player... players) {
191        InventorySlotPacket pk = new InventorySlotPacket();
192        pk.slot = inv == this.right ? this.left.getSize() + index : index;
193        pk.item = inv.getItem(index).clone();
194
195        for (Player player : players) {
196            int id = player.getWindowId(this);
197            if (id == -1) {
198                this.close(player);
199                continue;
200            }
201            pk.inventoryId = id;
202            player.dataPacket(pk);
203        }
204    }
205}
