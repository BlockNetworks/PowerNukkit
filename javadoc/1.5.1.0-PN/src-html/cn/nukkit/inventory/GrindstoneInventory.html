049    @Override
050    public void onClose(Player who) {
051        super.onClose(who);
052        who.craftingType = Player.CRAFTING_SMALL;
053
054        Item[] drops = new Item[]{ getFirstItem(), getSecondItem() };
055        drops = who.getInventory().addItem(drops);
056        for (Item drop : drops) {
057            if (!who.dropItem(drop)) {
058                this.getHolder().getLevel().dropItem(this.getHolder().add(0.5, 0.5, 0.5), drop);
059            }
060        }
061
062        clear(SLOT_FIRST_ITEM);
063        clear(SLOT_SECOND_ITEM);
064
065        who.resetCraftingGridType();
066    }

119    @Override
120    public void onSlotChange(int index, Item before, boolean send) {
121        try {
122            if (index > 1) {
123                return;
124            }
125            updateResult(send);
126        } finally {
127            super.onSlotChange(index, before, send);
128        }
129    }
130
131    public boolean updateResult(boolean send) {
132        Item firstItem = getFirstItem();
133        Item secondItem = getSecondItem();
134        if (!firstItem.isNull() && !secondItem.isNull() && firstItem.getId() != secondItem.getId()) {
135            setResult(Item.get(0), send);
136            setResultExperience(0);
137            return false;
138        }
139
140        if (firstItem.isNull()) {
141            Item air = firstItem;
142            firstItem = secondItem;
143            secondItem = air;
144        }
145
146        if (firstItem.isNull()) {
147            setResult(Item.get(0), send);
148            setResultExperience(0);
149            return false;
150        }
151
152        if (firstItem.getId() == ItemID.ENCHANTED_BOOK) {
153            if (secondItem.isNull()) {
154                setResult(Item.get(ItemID.BOOK, 0, firstItem.getCount()), send);
155                recalculateResultExperience();
156            } else {
157                setResultExperience(0);
158                setResult(Item.get(0), send);
159            }
160            return false;
161        }
162
163        Item result = firstItem.clone();
164        CompoundTag tag = result.getNamedTag();
165        if (tag == null) tag = new CompoundTag();
166        tag.remove("ench");
167        
168        result.setCompoundTag(tag);
169        if (!secondItem.isNull() && firstItem.getMaxDurability() > 0) {
170            int first = firstItem.getMaxDurability() - firstItem.getDamage();
171            int second = secondItem.getMaxDurability() - secondItem.getDamage();
172            int reduction = first + second + firstItem.getMaxDurability() * 5 / 100;
173            int resultingDamage = Math.max(firstItem.getMaxDurability() - reduction + 1, 0);
174            result.setDamage(resultingDamage);
175        }
176        setResult(result, send);
177        recalculateResultExperience();
178        return true;
179    }
180
181    @PowerNukkitOnly
182    @Since("1.4.0.0-PN")
183    public void recalculateResultExperience() {
184        if (getResult().isNull()) {
185            setResultExperience(0);
186            return;
187        }
188
189        Item firstItem = getFirstItem();
190        Item secondItem = getSecondItem();
191        if (!firstItem.hasEnchantments() && !secondItem.hasEnchantments()) {
192            setResultExperience(0);
193            return;
194        }
195
196        int resultExperience = Stream.of(firstItem, secondItem)
197                .flatMap(item -> {
198                    // Support stacks of enchanted items and skips invalid stacks (e.g. negative stacks, enchanted air)
199                    if (item.isNull()) {
200                        return Stream.empty();
201                    } else if (item.getCount() == 1) {
202                        return Arrays.stream(item.getEnchantments());
203                    } else {
204                        Enchantment[][] enchantments = new Enchantment[item.getCount()][];
205                        Arrays.fill(enchantments, item.getEnchantments());
206                        return Arrays.stream(enchantments).flatMap(Arrays::stream);
207                    }
208                })
209                .mapToInt(enchantment-> enchantment.getMinEnchantAbility(enchantment.getLevel()))
210                .sum();
211
212        resultExperience = ThreadLocalRandom.current().nextInt(
213                NukkitMath.ceilDouble((double)resultExperience / 2),
214                resultExperience + 1
215        );
216
217        setResultExperience(resultExperience);
218    }

220    @Override
221    public Item getItem(int index) {
222        if (index < 0 || index > 3) {
223            return Item.get(0);
224        }
225        if (index == 2) {
226            index = SLOT_RESULT;
227        }
228        
229        return super.getItem(index);
230    }

232    @Override
233    public boolean setItem(int index, Item item, boolean send) {
234        if (index < 0 || index > 3) {
235            return false;
236        }
237        
238        if (index == 2) {
239            index = SLOT_RESULT;
240        }
241        
242        return super.setItem(index, item, send);
243    }
