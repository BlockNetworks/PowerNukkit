001package cn.nukkit.dispenser;
002
003import cn.nukkit.block.*;
004import cn.nukkit.item.Item;
005import cn.nukkit.item.ItemBucket;
006import cn.nukkit.item.MinecraftItemID;
007import cn.nukkit.math.BlockFace;
008
009/**
010 * @author CreeperFace
011 */
012public class BucketDispenseBehavior extends DefaultDispenseBehavior {
013
014    @Override
015    public Item dispense(BlockDispenser block, BlockFace face, Item item) {
016        if (!(item instanceof ItemBucket)) {
017            return super.dispense(block, face, item);
018        }
019        
020        ItemBucket bucket = (ItemBucket) item; 
021        Block target = block.getSide(face);
022        
023        if (!bucket.isEmpty()) {
024            if (target.canBeFlowedInto() || target.getId() == BlockID.NETHER_PORTAL) {
025                Block replace = bucket.getTargetBlock();
026
027                if (replace instanceof BlockLiquid) {
028                    if (target.getId() == BlockID.NETHER_PORTAL) {
029                        target.onBreak(null);
030                    }
031
032                    block.level.setBlock(target, replace);
033                    return MinecraftItemID.BUCKET.get(1, bucket.getCompoundTag());
034                }
035            }
036        } else if (target instanceof BlockWater && target.getDamage() == 0) {
037            target.level.setBlock(target, Block.get(BlockID.AIR));
038            return MinecraftItemID.WATER_BUCKET.get(1, bucket.getCompoundTag());
039        } else if (target instanceof BlockLava && target.getDamage() == 0) {
040            target.level.setBlock(target, Block.get(BlockID.AIR));
041            return MinecraftItemID.LAVA_BUCKET.get(1, bucket.getCompoundTag());
042        }
043
044        return super.dispense(block, face, item);
045    }
046}
