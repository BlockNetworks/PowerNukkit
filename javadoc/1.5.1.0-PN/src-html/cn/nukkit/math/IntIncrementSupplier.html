001package cn.nukkit.math;
002
003import cn.nukkit.api.PowerNukkitOnly;
004import cn.nukkit.api.Since;
005
006import java.util.function.IntSupplier;
007import java.util.stream.IntStream;
008
009@PowerNukkitOnly
010@Since("1.4.0.0-PN")
011public final class IntIncrementSupplier implements IntSupplier {
012    private int next;
013    private final int increment;
014
015    @PowerNukkitOnly
016    @Since("1.4.0.0-PN")
017    public IntIncrementSupplier(int first, int increment) {
018        next = first;
019        this.increment = increment;
020    }
021
022    @Override
023    public int getAsInt() {
024        int current = next;
025        next = current + increment;
026        return current;
027    }
028
029    @PowerNukkitOnly
030    @Since("1.4.0.0-PN")
031    public IntStream stream() {
032        return IntStream.generate(this); 
033    }
034}
