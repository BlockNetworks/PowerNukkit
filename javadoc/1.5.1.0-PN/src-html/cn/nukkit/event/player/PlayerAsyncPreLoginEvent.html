001package cn.nukkit.event.player;
002
003import cn.nukkit.Player;
004import cn.nukkit.Server;
005import cn.nukkit.api.DeprecationDetails;
006import cn.nukkit.api.PowerNukkitOnly;
007import cn.nukkit.api.Since;
008import cn.nukkit.entity.data.Skin;
009import cn.nukkit.event.HandlerList;
010import cn.nukkit.utils.ClientChainData;
011import cn.nukkit.utils.LoginChainData;
012import io.netty.util.internal.EmptyArrays;
013
014import java.util.ArrayList;
015import java.util.List;
016import java.util.UUID;
017import java.util.function.Consumer;
018
019/**
020 * This event is called asynchronously
021 *
022 * @author CreeperFace
023 */
024public class PlayerAsyncPreLoginEvent extends PlayerEvent {
025
026    private static final HandlerList handlers = new HandlerList();
027
028    public static HandlerList getHandlers() {
029        return handlers;
030    }
031
032    private final String name;
033    private final UUID uuid;
034    private final LoginChainData chainData;
035    private Skin skin;
036    private final String address;
037    private final int port;
038
039    private LoginResult loginResult = LoginResult.SUCCESS;
040    private String kickMessage = "Plugin Reason";
041
042    private final List<Consumer<Server>> scheduledActions = new ArrayList<>();
043
044    @Deprecated @DeprecationDetails(since = "1.4.0.0-PN", reason = "LoginChainData and Skin were added by refactoring this constructor",
045            replaceWith = "PlayerAsyncPreLoginEvent(String name, UUID uuid, LoginChainData chainData, Skin skin, String address, int port)")
046    @PowerNukkitOnly("The signature was changed in Cloudburst Nukkit and we re-added this constructor for backward-compatibility")
047    public PlayerAsyncPreLoginEvent(String name, UUID uuid, String address, int port) {
048        // TODO PowerNukkit: I think this might cause an exception...
049        this(name, uuid, ClientChainData.of(EmptyArrays.EMPTY_BYTES), null, address, port);
050    }
051    
052    @Since("1.4.0.0-PN")
053    public PlayerAsyncPreLoginEvent(String name, UUID uuid, LoginChainData chainData, Skin skin, String address, int port) {
054        this.name = name;
055        this.uuid = uuid;
056        this.chainData = chainData;
057        this.skin = skin;
058        this.address = address;
059        this.port = port;
060    }
061
062    @Override
063    public Player getPlayer() {
064        throw new UnsupportedOperationException("Could not get player instance in an async event");
065    }
066
067    public String getName() {
068        return this.name;
069    }
070
071    public UUID getUuid() {
072        return this.uuid;
073    }
074
075    @Since("1.4.0.0-PN")
076    public LoginChainData getChainData() {
077        return this.chainData;
078    }
079
080    @Since("1.4.0.0-PN")
081    public String getXuid() {
082        return this.chainData.getXUID();
083    }
084
085    public Skin getSkin() {
086        return this.skin;
087    }
088
089    public void setSkin(Skin skin) {
090        this.skin = skin;
091    }
092
093    public String getAddress() {
094        return this.address;
095    }
096
097    public int getPort() {
098        return this.port;
099    }
100
101    public LoginResult getLoginResult() {
102        return this.loginResult;
103    }
104
105    public void setLoginResult(LoginResult loginResult) {
106        this.loginResult = loginResult;
107    }
108
109    public String getKickMessage() {
110        return kickMessage;
111    }
112
113    public void setKickMessage(String kickMessage) {
114        this.kickMessage = kickMessage;
115    }
116
117    public void scheduleSyncAction(Consumer<Server> action) {
118        this.scheduledActions.add(action);
119    }
120
121    public List<Consumer<Server>> getScheduledActions() {
122        return new ArrayList<>(scheduledActions);
123    }
124
125    public void allow() {
126        this.loginResult = LoginResult.SUCCESS;
127    }
128
129    public void disAllow(String message) {
130        this.loginResult = LoginResult.KICK;
131        this.kickMessage = message;
132    }
133
134    public enum LoginResult {
135        SUCCESS,
136        KICK
137    }
138}
