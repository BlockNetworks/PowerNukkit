001package cn.nukkit.event.block;
002
003import cn.nukkit.Player;
004import cn.nukkit.api.PowerNukkitDifference;
005import cn.nukkit.block.Block;
006import cn.nukkit.event.Cancellable;
007import cn.nukkit.event.HandlerList;
008import cn.nukkit.item.Item;
009import cn.nukkit.math.Vector3;
010
011@PowerNukkitDifference(info = "The player and the item are null when they are empty by a hopper pulling the item", since = "1.4.0.0-PN")
012public class ComposterEmptyEvent extends BlockEvent implements Cancellable {
013
014    private static final HandlerList handlers = new HandlerList();
015
016    private final Player player;
017    private Item drop;
018    private Item itemUsed;
019    private int newLevel;
020    private Vector3 motion;
021
022    public ComposterEmptyEvent(Block block, Player player, Item itemUsed, Item drop, int newLevel) {
023        super(block);
024        this.player = player;
025        this.drop = drop;
026        this.itemUsed = itemUsed;
027        this.newLevel = Math.max(0, Math.min(newLevel, 8));
028    }
029
030    public Player getPlayer() {
031        return player;
032    }
033
034    public Item getDrop() {
035        return drop.clone();
036    }
037
038    public void setDrop(Item drop) {
039        if (drop == null) {
040            drop = Item.get(Item.AIR);
041        } else {
042            drop = drop.clone();
043        }
044        this.drop = drop;
045    }
046
047    public Item getItemUsed() {
048        return itemUsed;
049    }
050
051    public void setItemUsed(Item itemUsed) {
052        this.itemUsed = itemUsed;
053    }
054
055    public int getNewLevel() {
056        return newLevel;
057    }
058
059    public void setNewLevel(int newLevel) {
060        this.newLevel = Math.max(0, Math.min(newLevel, 8));
061    }
062
063    public Vector3 getMotion() {
064        return motion;
065    }
066
067    public void setMotion(Vector3 motion) {
068        this.motion = motion;
069    }
070
071    public static HandlerList getHandlers() {
072        return handlers;
073    }
074
075}
