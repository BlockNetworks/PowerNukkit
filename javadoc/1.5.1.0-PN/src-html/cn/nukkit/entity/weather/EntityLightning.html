001package cn.nukkit.entity.weather;

010import cn.nukkit.event.block.BlockIgniteEvent;

013import cn.nukkit.level.Sound;

017
018import java.util.concurrent.ThreadLocalRandom;
019
020/**
021 * @author boybook
022 * @since 2016/2/27
023 */
024public class EntityLightning extends Entity implements EntityLightningStrike {
025
026    public static final int NETWORK_ID = 93;
027
028    protected boolean isEffect = true;
029
030    public int state;
031    public int liveTime;
032
033
034    @Override
035    public int getNetworkId() {
036        return NETWORK_ID;
037    }
038
039    public EntityLightning(FullChunk chunk, CompoundTag nbt) {
040        super(chunk, nbt);
041    }
042
043    @Override
044    protected void initEntity() {
045        super.initEntity();
046
047        this.setHealth(4);
048        this.setMaxHealth(4);
049
050        this.state = 2;
051        this.liveTime = ThreadLocalRandom.current().nextInt(3) + 1;
052
053        if (isEffect && this.level.gameRules.getBoolean(GameRule.DO_FIRE_TICK) && (this.server.getDifficulty() >= 2)) {
054            Block block = this.getLevelBlock();
055            if (block.getId() == 0 || block.getId() == Block.TALL_GRASS) {
056                BlockFire fire = (BlockFire) Block.get(BlockID.FIRE);
057                fire.x = block.x;
058                fire.y = block.y;
059                fire.z = block.z;
060                fire.level = level;
061                this.getLevel().setBlock(fire, fire, true);
062                if (fire.isBlockTopFacingSurfaceSolid(fire.down()) || fire.canNeighborBurn()) {
063
064                    BlockIgniteEvent e = new BlockIgniteEvent(block, null, this, BlockIgniteEvent.BlockIgniteCause.LIGHTNING);
065                    getServer().getPluginManager().callEvent(e);
066
067                    if (!e.isCancelled()) {
068                        level.setBlock(fire, fire, true);
069                        level.scheduleUpdate(fire, fire.tickRate() + ThreadLocalRandom.current().nextInt(10));
070                    }
071                }
072            }
073        }
074    }
075
076    public boolean isEffect() {
077        return this.isEffect;
078    }
079
080    public void setEffect(boolean e) {
081        this.isEffect = e;
082    }
083
084    @Override
085    public boolean attack(EntityDamageEvent source) {
086        //false?
087        source.setDamage(0);
088        return super.attack(source);
089    }
090
091    @PowerNukkitDifference(info = "Using new method to play sounds", since = "1.4.0.0-PN")
092    @Override
093    public boolean onUpdate(int currentTick) {
094        if (this.closed) {
095            return false;
096        }
097
098        int tickDiff = currentTick - this.lastUpdate;
099
100        if (tickDiff <= 0 && !this.justCreated) {
101            return true;
102        }
103
104        this.lastUpdate = currentTick;
105
106        this.entityBaseTick(tickDiff);
107
108        if (this.state == 2) {
109            this.level.addSound(this, Sound.AMBIENT_WEATHER_THUNDER);
110            this.level.addSound(this, Sound.RANDOM_EXPLODE);
111        }
112
113        this.state--;
114
115        if (this.state < 0) {
116            if (this.liveTime == 0) {
117                this.close();
118                return false;
119            } else if (this.state < -ThreadLocalRandom.current().nextInt(10)) {
120                this.liveTime--;
121                this.state = 1;
122
123                if (this.isEffect && this.level.gameRules.getBoolean(GameRule.DO_FIRE_TICK)) {
124                    Block block = this.getLevelBlock();
125
126                    if (block.getId() == Block.AIR || block.getId() == Block.TALL_GRASS) {
127                        BlockIgniteEvent e = new BlockIgniteEvent(block, null, this, BlockIgniteEvent.BlockIgniteCause.LIGHTNING);
128                        getServer().getPluginManager().callEvent(e);
129
130                        if (!e.isCancelled()) {
131                            Block fire = Block.get(BlockID.FIRE);
132                            this.level.setBlock(block, fire);
133                            this.getLevel().scheduleUpdate(fire, fire.tickRate());
134                        }
135                    }
136                }
137            }
138        }
139
140        if (this.state >= 0) {
141            if (this.isEffect) {
142                AxisAlignedBB bb = getBoundingBox().grow(3, 3, 3);
143                bb.setMaxX(bb.getMaxX() + 6);
144
145                for (Entity entity : this.level.getCollidingEntities(bb, this)) {
146                    entity.onStruckByLightning(this);
147                }
148            }
149        }
150
151        return true;
152    }
153
154
155    @PowerNukkitOnly
156    @Since("1.5.1.0-PN")
157    @Override
158    public String getOriginalName() {
159        return "Lightning Bolt";
160    }
161}
