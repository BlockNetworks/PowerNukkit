071    public int getResultDamage() {
072        return NukkitMath.ceilDouble(Math.sqrt(this.motionX * this.motionX + this.motionY * this.motionY + this.motionZ * this.motionZ) * getDamage());
073    }

079    public void onCollideWithEntity(Entity entity) {
080        ProjectileHitEvent projectileHitEvent = new ProjectileHitEvent(this, MovingObjectPosition.fromEntity(entity));
081        this.server.getPluginManager().callEvent(projectileHitEvent);
082
083        if (projectileHitEvent.isCancelled()) {
084            return;
085        }
086
087        float damage = this.getResultDamage(entity);
088
089        EntityDamageEvent ev;
090        if (this.shootingEntity == null) {
091            ev = new EntityDamageByEntityEvent(this, entity, DamageCause.PROJECTILE, damage);
092        } else {
093            ev = new EntityDamageByChildEntityEvent(this.shootingEntity, this, entity, DamageCause.PROJECTILE, damage);
094        }
095        if (entity.attack(ev)) {
096            addHitEffect();
097            this.hadCollision = true;
098
099            if (this.fireTicks > 0) {
100                EntityCombustByEntityEvent event = new EntityCombustByEntityEvent(this, entity, 5);
101                this.server.getPluginManager().callEvent(ev);
102                if (!event.isCancelled()) {
103                    entity.setOnFire(event.getDuration());
104                }
105            }
106        }
107        afterCollisionWithEntity(entity);
108        if (closeOnCollide) {
109            this.close();
110        }
111    }

143    @PowerNukkitOnly

151    @Override
152    public boolean onUpdate(int currentTick) {
153        if (this.closed) {
154            return false;
155        }
156
157        int tickDiff = currentTick - this.lastUpdate;
158        if (tickDiff <= 0 && !this.justCreated) {
159            return true;
160        }
161        this.lastUpdate = currentTick;
162
163        boolean hasUpdate = this.entityBaseTick(tickDiff);
164
165        if (this.isAlive()) {
166
167            MovingObjectPosition movingObjectPosition = null;
168
169            if (!this.isCollided) {
170                updateMotion();
171            }
172
173            Vector3 moveVector = new Vector3(this.x + this.motionX, this.y + this.motionY, this.z + this.motionZ);
174
175            Entity[] list = this.getLevel().getCollidingEntities(this.boundingBox.addCoord(this.motionX, this.motionY, this.motionZ).expand(1, 1, 1), this);
176
177            double nearDistance = Integer.MAX_VALUE;
178            Entity nearEntity = null;
179
180            for (Entity entity : list) {
181                if (/*!entity.canCollideWith(this) or */
182                        (entity == this.shootingEntity && this.ticksLived < 5) ||
183                                (entity instanceof Player && ((Player) entity).getGamemode() == Player.SPECTATOR)
184                ) {
185                    continue;
186                }
187
188                AxisAlignedBB axisalignedbb = entity.boundingBox.grow(0.3, 0.3, 0.3);
189                MovingObjectPosition ob = axisalignedbb.calculateIntercept(this, moveVector);
190
191                if (ob == null) {
192                    continue;
193                }
194
195                double distance = this.distanceSquared(ob.hitVector);
196
197                if (distance < nearDistance) {
198                    nearDistance = distance;
199                    nearEntity = entity;
200                }
201            }
202
203            if (nearEntity != null) {
204                movingObjectPosition = MovingObjectPosition.fromEntity(nearEntity);
205            }
206
207            if (movingObjectPosition != null) {
208                if (movingObjectPosition.entityHit != null) {
209                    onCollideWithEntity(movingObjectPosition.entityHit);
210                    hasUpdate = true;
211                    if (closed) {
212                        return true;
213                    }
214                }
215            }
216
217            Position position = getPosition();
218            Vector3 motion = getMotion();
219            this.move(this.motionX, this.motionY, this.motionZ);
220
221            if (this.isCollided && !this.hadCollision) { //collide with block
222                this.hadCollision = true;
223
224                this.motionX = 0;
225                this.motionY = 0;
226                this.motionZ = 0;
227
228                this.server.getPluginManager().callEvent(new ProjectileHitEvent(this, MovingObjectPosition.fromBlock(this.getFloorX(), this.getFloorY(), this.getFloorZ(), -1, this)));
229                onCollideWithBlock(position, motion);
230                addHitEffect();
231                return false;
232            } else if (!this.isCollided && this.hadCollision) {
233                this.hadCollision = false;
234            }
235
236            if (!this.hadCollision || Math.abs(this.motionX) > 0.00001 || Math.abs(this.motionY) > 0.00001 || Math.abs(this.motionZ) > 0.00001) {
237                updateRotation();
238                hasUpdate = true;
239            }
240
241            this.updateMovement();
242
243        }
244
245        return hasUpdate;
246    }
247
248    public void updateRotation() {
249        double f = Math.sqrt((this.motionX * this.motionX) + (this.motionZ * this.motionZ));
250        this.yaw = Math.atan2(this.motionX, this.motionZ) * 180 / Math.PI;
251        this.pitch = Math.atan2(this.motionY, f) * 180 / Math.PI;
252    }

254    public void inaccurate(float modifier) {
255        Random rand = ThreadLocalRandom.current();
256
257        this.motionX += rand.nextGaussian() * 0.007499999832361937 * modifier;
258        this.motionY += rand.nextGaussian() * 0.007499999832361937 * modifier;
259        this.motionZ += rand.nextGaussian() * 0.007499999832361937 * modifier;
260    }
