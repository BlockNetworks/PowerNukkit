024 */
025public class EntityFallingBlock extends Entity {
026
027    public static final int NETWORK_ID = 66;
028
029    @Override
030    public float getWidth() {
031        return 0.98f;
032    }
033
034    @Override
035    public float getLength() {
036        return 0.98f;
037    }
038
039    @Override
040    public float getHeight() {
041        return 0.98f;
042    }
043
044    @Override
045    protected float getGravity() {
046        return 0.04f;
047    }
048
049    @Override
050    protected float getDrag() {
051        return 0.02f;
052    }
053
054    @Override
055    protected float getBaseOffset() {
056        return 0.49f;
057    }
058
059    @Override
060    public boolean canCollide() {
061        return false;
062    }
063
064    protected int blockId;
065    protected int damage;
066    protected boolean breakOnLava;
067
068    public EntityFallingBlock(FullChunk chunk, CompoundTag nbt) {
069        super(chunk, nbt);
070    }
071
072    @Override
073    protected void initEntity() {
074        super.initEntity();
075
076        if (namedTag != null) {
077            if (namedTag.contains("TileID")) {
078                blockId = namedTag.getInt("TileID");
079            } else if (namedTag.contains("Tile")) {
080                blockId = namedTag.getInt("Tile");
081                namedTag.putInt("TileID", blockId);
082            }
083
084            if (namedTag.contains("Data")) {
085                damage = namedTag.getByte("Data");
086            }
087        }
088
089        breakOnLava = namedTag.getBoolean("BreakOnLava");
090
091        if (blockId == 0) {
092            close();
093            return;
094        }
095
096        this.fireProof = true;
097        this.setDataFlag(DATA_FLAGS, DATA_FLAG_FIRE_IMMUNE, true);
098
099        setDataProperty(new IntEntityData(DATA_VARIANT, GlobalBlockPalette.getOrCreateRuntimeId(this.getBlock(), this.getDamage())));
100    }
101
102    public boolean canCollideWith(Entity entity) {
103        return false;
104    }
105
106    @Override
107    public boolean attack(EntityDamageEvent source) {
108        return source.getCause() == DamageCause.VOID && super.attack(source);
109    }
110
111    @Override
112    public boolean onUpdate(int currentTick) {
113
114        if (closed) {
115            return false;
116        }
117
118        this.timing.startTiming();
119
120        int tickDiff = currentTick - lastUpdate;
121        if (tickDiff <= 0 && !justCreated) {
122            return true;
123        }
124
125        lastUpdate = currentTick;
126
127        boolean hasUpdate = entityBaseTick(tickDiff);
128
129        if (isAlive()) {
130            motionY -= getGravity();
131
132            move(motionX, motionY, motionZ);
133
134            float friction = 1 - getDrag();
135
136            motionX *= friction;
137            motionY *= 1 - getDrag();
138            motionZ *= friction;
139
140            Vector3 pos = (new Vector3(x - 0.5, y, z - 0.5)).round();
141
142            if (breakOnLava && level.getBlock(pos.subtract(0, 1, 0)) instanceof BlockLava) {
143                close();
144                if (this.level.getGameRules().getBoolean(GameRule.DO_ENTITY_DROPS)) {
145                    getLevel().dropItem(this, Block.get(this.getBlock(), this.getDamage()).toItem());
146                }
147                level.addParticle(new DestroyBlockParticle(pos, Block.get(getBlock(), getDamage())));
148                return true;
149            }
150
151            if (onGround) {
152                close();
153                Block block = level.getBlock(pos);
154
155                Vector3 floorPos = (new Vector3(x - 0.5, y, z - 0.5)).floor();
156                Block floorBlock = this.level.getBlock(floorPos);
157                if (this.getBlock() == Block.SNOW_LAYER && floorBlock.getId() == Block.SNOW_LAYER && (floorBlock.getDamage() & 0x7) != 0x7) {
158                    int mergedHeight = (floorBlock.getDamage() & 0x7) + 1 + (this.getDamage() & 0x7) + 1;
159                    if (mergedHeight > 8) {
160                        EntityBlockChangeEvent event = new EntityBlockChangeEvent(this, floorBlock, Block.get(Block.SNOW_LAYER, 0x7));
161                        this.server.getPluginManager().callEvent(event);
162                        if (!event.isCancelled()) {
163                            this.level.setBlock(floorPos, event.getTo(), true);
164
165                            Vector3 abovePos = floorPos.up();
166                            Block aboveBlock = this.level.getBlock(abovePos);
167                            if (aboveBlock.getId() == Block.AIR) {
168                                EntityBlockChangeEvent event2 = new EntityBlockChangeEvent(this, aboveBlock, Block.get(Block.SNOW_LAYER, mergedHeight - 8 - 1));
169                                this.server.getPluginManager().callEvent(event2);
170                                if (!event2.isCancelled()) {
171                                    this.level.setBlock(abovePos, event2.getTo(), true);
172                                }
173                            }
174                        }
175                    } else {
176                        EntityBlockChangeEvent event = new EntityBlockChangeEvent(this, floorBlock, Block.get(Block.SNOW_LAYER, mergedHeight - 1));
177                        this.server.getPluginManager().callEvent(event);
178                        if (!event.isCancelled()) {
179                            this.level.setBlock(floorPos, event.getTo(), true);
180                        }
181                    }
182                } else if (block.getId() > 0 && block.isTransparent() && !block.canBeReplaced() || this.getBlock() == Block.SNOW_LAYER && block instanceof BlockLiquid) {
183                    if (this.getBlock() != Block.SNOW_LAYER ? this.level.getGameRules().getBoolean(GameRule.DO_ENTITY_DROPS) : this.level.getGameRules().getBoolean(GameRule.DO_TILE_DROPS)) {
184                        getLevel().dropItem(this, Block.get(this.getBlock(), this.getDamage()).toItem());
185                    }
186                } else {
187                    EntityBlockChangeEvent event = new EntityBlockChangeEvent(this, block, Block.get(getBlock(), getDamage()));
188                    server.getPluginManager().callEvent(event);
189                    if (!event.isCancelled()) {
190                        getLevel().setBlock(pos, event.getTo(), true);
191
192                        if (event.getTo().getId() == Item.ANVIL) {
193                            getLevel().addLevelEvent(block, LevelEventPacket.EVENT_SOUND_ANVIL_FALL);
194                        }
195                    }
196                }
197                hasUpdate = true;
198            }
199
200            updateMovement();
201        }
202
203        this.timing.stopTiming();
204
205        return hasUpdate || !onGround || Math.abs(motionX) > 0.00001 || Math.abs(motionY) > 0.00001 || Math.abs(motionZ) > 0.00001;
206    }
