199    @Override
200    public boolean onUpdate(int currentTick) {
201        if (this.closed) {
202            return false;
203        }
204
205        int tickDiff = currentTick - this.lastUpdate;
206
207        if (tickDiff <= 0 && !this.justCreated) {
208            return true;
209        }
210
211        this.lastUpdate = currentTick;
212
213        boolean hasUpdate = this.entityBaseTick(tickDiff);
214
215        if (this.isAlive()) {
216            hasUpdate = this.updateBoat(tickDiff) || hasUpdate;
217        }
218
219        return hasUpdate || !this.onGround || Math.abs(this.motionX) > 0.00001 || Math.abs(this.motionY) > 0.00001 || Math.abs(this.motionZ) > 0.00001;
220    }

222    private boolean updateBoat(int tickDiff) {
223        // The rolling amplitude
224        if (getRollingAmplitude() > 0) {
225            setRollingAmplitude(getRollingAmplitude() - 1);
226        }
227
228        // A killer task
229        if (y < -16) {
230            kill();
231            return false;
232        }
233
234        boolean hasUpdated = false;
235        double waterDiff = getWaterLevel();
236        if (!hasControllingPassenger()) {
237            hasUpdated = computeBuoyancy(waterDiff);
238            Iterator<Entity> iterator = ignoreCollision.iterator();
239            while (iterator.hasNext()) {
240                Entity ignored = iterator.next();
241                if (!ignored.isValid() || ignored.isClosed() || !ignored.isAlive()
242                        || !ignored.getBoundingBox().intersectsWith(getBoundingBox().grow(0.5, 0.5, 0.5))) {
243                    iterator.remove();
244                    hasUpdated = true;
245                }
246            }
247            moveBoat(waterDiff);
248        } else {
249            updateMovement();
250        }
251        hasUpdated = hasUpdated || positionChanged;
252        if (waterDiff >= -SINKING_DEPTH) {
253            if (ticksInWater != 0) {
254                ticksInWater = 0;
255                hasUpdated = true;
256            }
257            //hasUpdated = collectCollidingEntities() || hasUpdated;
258        } else {
259            hasUpdated = true;
260            ticksInWater += tickDiff;
261            if (ticksInWater >= 3 * 20) {
262                for (int i = passengers.size() - 1; i >= 0; i--) {
263                    dismountEntity(passengers.get(i));
264                }
265            }
266        }
267        this.getServer().getPluginManager().callEvent(new VehicleUpdateEvent(this));
268        return hasUpdated;
269    }

281    private void moveBoat(double waterDiff) {
282        checkObstruction(this.x, this.y, this.z);
283        move(this.motionX, this.motionY, this.motionZ);
284
285        double friction = 1 - this.getDrag();
286
287        if (this.onGround && (Math.abs(this.motionX) > 0.00001 || Math.abs(this.motionZ) > 0.00001)) {
288            friction *= this.getLevel().getBlock(this.temporalVector.setComponents((int) Math.floor(this.x), (int) Math.floor(this.y - 1), (int) Math.floor(this.z) - 1)).getFrictionFactor();
289        }
290
291        this.motionX *= friction;
292
293        if (!onGround || waterDiff > SINKING_DEPTH/* || sinking*/) {
294            this.motionY = waterDiff > 0.5 ? this.motionY - this.getGravity() : (this.motionY - SINKING_SPEED < -SINKING_MAX_SPEED ? this.motionY : this.motionY - SINKING_SPEED);
295        }
296
297        this.motionZ *= friction;
298
299        Location from = new Location(lastX, lastY, lastZ, lastYaw, lastPitch, level);
300        Location to = new Location(this.x, this.y, this.z, this.yaw, this.pitch, level);
301
302        if (!from.equals(to)) {
303            this.getServer().getPluginManager().callEvent(new VehicleMoveEvent(this, from, to));
304        }
305
306        //TODO: lily pad collision
307        this.updateMovement();
308    }

310    private boolean collectCollidingEntities() {
311        if (this.passengers.size() >= 2) {
312            return false;
313        }
314
315        for (Entity entity : this.level.getCollidingEntities(this.boundingBox.grow(0.20000000298023224, 0.0D, 0.20000000298023224), this)) {
316            if (entity.riding != null || !(entity instanceof EntityLiving) || entity instanceof Player || entity instanceof EntityWaterAnimal || isPassenger(entity)) {
317                continue;
318            }
319
320            this.mountEntity(entity);
321
322            if (this.passengers.size() >= 2) {
323                break;
324            }
325        }
326
327        return true;
328    }

330    private boolean computeBuoyancy(double waterDiff) {
331        boolean hasUpdated = false;
332        waterDiff -= getBaseOffset()/4;
333        if (waterDiff > SINKING_DEPTH && !sinking) {
334            sinking = true;
335        } else if (waterDiff < -SINKING_DEPTH && sinking) {
336            sinking = false;
337        }
338
339        if (waterDiff < -SINKING_DEPTH/1.7) {
340            this.motionY = Math.min(0.05/10, this.motionY + 0.005);
341            hasUpdated = true;
342        } else if (waterDiff < 0 || !sinking) {
343            this.motionY = this.motionY > (SINKING_MAX_SPEED/2) ? Math.max(this.motionY - 0.02, (SINKING_MAX_SPEED/2)) : this.motionY + SINKING_SPEED;
344            hasUpdated = true;
345        }
346        return hasUpdated;
347    }

353    public void updatePassengers(boolean sendLinks) {
354        if (this.passengers.isEmpty()) {
355            return;
356        }
357
358        for (Entity passenger : new ArrayList<>(passengers)) {
359            if (!passenger.isAlive()) {
360                dismountEntity(passenger);
361            }
362        }
363
364        Entity ent;
365
366        if (passengers.size() == 1) {
367            (ent = this.passengers.get(0)).setSeatPosition(getMountedOffset(ent));
368            super.updatePassengerPosition(ent);
369
370            if (sendLinks) {
371                broadcastLinkPacket(ent, SetEntityLinkPacket.TYPE_RIDE);
372            }
373        } else if (passengers.size() == 2) {
374            if (!((ent = passengers.get(0)) instanceof Player)) { //swap
375                Entity passenger2 = passengers.get(1);
376
377                if (passenger2 instanceof Player) {
378                    this.passengers.set(0, passenger2);
379                    this.passengers.set(1, ent);
380
381                    ent = passenger2;
382                }
383            }
384
385            ent.setSeatPosition(getMountedOffset(ent).add(RIDER_PASSENGER_OFFSET));
386            super.updatePassengerPosition(ent);
387            if (sendLinks) {
388                broadcastLinkPacket(ent, SetEntityLinkPacket.TYPE_RIDE);
389            }
390
391            (ent = this.passengers.get(1)).setSeatPosition(getMountedOffset(ent).add(PASSENGER_OFFSET));
392
393            super.updatePassengerPosition(ent);
394
395            if (sendLinks) {
396                broadcastLinkPacket(ent, SetEntityLinkPacket.TYPE_PASSENGER);
397            }
398
399            //float yawDiff = ent.getId() % 2 == 0 ? 90 : 270;
400            //ent.setRotation(this.yaw + yawDiff, ent.pitch);
401            //ent.updateMovement();
402        } else {
403            for (Entity passenger : passengers) {
404                super.updatePassengerPosition(passenger);
405            }
406        }
407    }

409    public double getWaterLevel() {
410        double maxY = this.boundingBox.getMinY() + getBaseOffset();
411        AxisAlignedBB.BBConsumer<Double> consumer = new AxisAlignedBB.BBConsumer<Double>() {
412
413            private double diffY = Double.MAX_VALUE;
414
415            @Override
416            public void accept(int x, int y, int z) {
417                Block block = EntityBoat.this.level.getBlock(EntityBoat.this.temporalVector.setComponents(x, y, z));
418
419                if (block instanceof BlockWater || ((block = block.getLevelBlockAtLayer(1)) instanceof BlockWater)) {
420                    double level = block.getMaxY();
421
422                    diffY = Math.min(maxY - level, diffY);
423                }
424            }
425
426            @Override
427            public Double get() {
428                return diffY;
429            }
430        };
431
432        this.boundingBox.forEach(consumer);
433
434        return consumer.get();
435    }

439        boolean player = this.passengers.size() >= 1 && this.passengers.get(0) instanceof Player;
440        byte mode = SetEntityLinkPacket.TYPE_PASSENGER;
441
442        if (!player && (entity instanceof Player || this.passengers.size() == 0)) {
443            mode = SetEntityLinkPacket.TYPE_RIDE;
444        }
445
446        return super.mountEntity(entity, mode);

450    public boolean mountEntity(Entity entity, byte mode) {
451        boolean r = super.mountEntity(entity, mode);
452        if (entity.riding == this) {
453            updatePassengers(true);
454
455            entity.setDataProperty(new ByteEntityData(DATA_RIDER_ROTATION_LOCKED, 1));
456            entity.setDataProperty(new FloatEntityData(DATA_RIDER_MAX_ROTATION, 90));
457            entity.setDataProperty(new FloatEntityData(DATA_RIDER_ROTATION_OFFSET, -90));
458            entity.setDataProperty(new FloatEntityData(DATA_RIDER_MIN_ROTATION, this.passengers.indexOf(entity) == 1 ? -90 : 1));
459            entity.setRotation(yaw, entity.pitch);
460            entity.updateMovement();
461        }
462        return r;
463    }

511    @Override
512    public void applyEntityCollision(Entity entity) {
513        if (this.riding == null && !hasControllingPassenger() && entity.riding != this
514                && !entity.passengers.contains(this) && !ignoreCollision.contains(entity)) {
515            if (!entity.boundingBox.intersectsWith(this.boundingBox.grow(0.20000000298023224, -0.1, 0.20000000298023224))
516                    || entity instanceof Player && ((Player) entity).isSpectator()) {
517                return;
518            }
519
520            double diffX = entity.x - this.x;
521            double diffZ = entity.z - this.z;
522
523            double direction = NukkitMath.getDirection(diffX, diffZ);
524
525            if (direction >= 0.009999999776482582D) {
526                direction = Math.sqrt(direction);
527                diffX /= direction;
528                diffZ /= direction;
529
530                double d3 = Math.min(1 / direction, 1);
531
532                diffX *= d3;
533                diffZ *= d3;
534                diffX *= 0.05000000074505806;
535                diffZ *= 0.05000000074505806;
536                diffX *= 1 + entityCollisionReduction;
537                diffZ *= 1 + entityCollisionReduction;
538
539                if (this.riding == null) {
540                    motionX -= diffX;
541                    motionZ -= diffZ;
542                }
543            }
544        }
