029/**
030 * @author CreeperFace
031 */
032public class EntityFirework extends Entity {
033
034    public static final int NETWORK_ID = 72;
035
036    private int fireworkAge;
037    private int lifetime;
038    private Item firework;
039    private boolean hadCollision;
040
041    @PowerNukkitDifference(info = "Will default to a black-creeper-face if the firework data is missing", since = "1.3.1.2-PN")
042    public EntityFirework(FullChunk chunk, CompoundTag nbt) {
043        super(chunk, nbt);
044
045        this.fireworkAge = 0;
046        Random rand = new Random();
047        this.lifetime = 30 + rand.nextInt(6) + rand.nextInt(7);
048
049        this.motionX = rand.nextGaussian() * 0.001D;
050        this.motionZ = rand.nextGaussian() * 0.001D;
051        this.motionY = 0.05D;
052
053        if (nbt.contains("FireworkItem")) {
054            firework = NBTIO.getItemHelper(nbt.getCompound("FireworkItem"));
055        } else {
056            firework = new ItemFirework();
057        }
058
059        if (!firework.hasCompoundTag() || !firework.getNamedTag().contains("Fireworks")) {
060            CompoundTag tag = firework.getNamedTag();
061            if (tag == null) {
062                tag = new CompoundTag();
063            }
064
065            CompoundTag ex = new CompoundTag()
066                    .putByteArray("FireworkColor", new byte[]{(byte) DyeColor.BLACK.getDyeData()})
067                    .putByteArray("FireworkFade", new byte[]{})
068                    .putBoolean("FireworkFlicker", false)
069                    .putBoolean("FireworkTrail", false)
070                    .putByte("FireworkType", ItemFirework.FireworkExplosion.ExplosionType.CREEPER_SHAPED.ordinal());
071
072            tag.putCompound("Fireworks", new CompoundTag("Fireworks")
073                    .putList(new ListTag<CompoundTag>("Explosions").add(ex))
074                    .putByte("Flight", 1)
075            );
076
077            firework.setNamedTag(tag);
078        }
079
080        this.setDataProperty(new NBTEntityData(Entity.DATA_DISPLAY_ITEM, firework.getNamedTag()));
081        this.setDataProperty(new IntEntityData(Entity.DATA_DISPLAY_OFFSET, 1));
082        this.setDataProperty(new ByteEntityData(Entity.DATA_HAS_DISPLAY, 1));
083    }
084
085    @Override
086    public int getNetworkId() {
087        return NETWORK_ID;
088    }
089
090    @Override
091    public boolean onUpdate(int currentTick) {
092        if (this.closed) {
093            return false;
094        }
095
096        int tickDiff = currentTick - this.lastUpdate;
097
098        if (tickDiff <= 0 && !this.justCreated) {
099            return true;
100        }
101
102        this.lastUpdate = currentTick;
103
104        this.timing.startTiming();
105
106
107        boolean hasUpdate = this.entityBaseTick(tickDiff);
108
109        if (this.isAlive()) {
110
111            this.motionX *= 1.15D;
112            this.motionZ *= 1.15D;
113            this.motionY += 0.04D;
114            Position position = getPosition();
115            Vector3 motion = getMotion();
116            this.move(this.motionX, this.motionY, this.motionZ);
117
118            if (this.isCollided && !this.hadCollision) { //collide with block
119                this.hadCollision = true;
120
121                for (Block collisionBlock : level.getCollisionBlocks(getBoundingBox().grow(0.1, 0.1, 0.1))) {
122                    collisionBlock.onProjectileHit(this, position, motion);
123                }
124                
125            } else if (!this.isCollided && this.hadCollision) {
126                this.hadCollision = false;
127            }
128
129            this.updateMovement();
130
131
132            float f = (float) Math.sqrt(this.motionX * this.motionX + this.motionZ * this.motionZ);
133            this.yaw = (float) (Math.atan2(this.motionX, this.motionZ) * (180D / Math.PI));
134
135            this.pitch = (float) (Math.atan2(this.motionY, f) * (180D / Math.PI));
136
137
138            if (this.fireworkAge == 0) {
139                this.getLevel().addSound(this, Sound.FIREWORK_LAUNCH);
140            }
141
142            this.fireworkAge++;
143
144            hasUpdate = true;
145            if (this.fireworkAge >= this.lifetime) {
146                EntityEventPacket pk = new EntityEventPacket();
147                pk.data = 0;
148                pk.event = EntityEventPacket.FIREWORK_EXPLOSION;
149                pk.eid = this.getId();
150
151                level.addLevelSoundEvent(this, LevelSoundEventPacket.SOUND_LARGE_BLAST, -1, NETWORK_ID);
152
153                Server.broadcastPacket(getViewers().values(), pk);
154
155                this.kill();
156                hasUpdate = true;
157            }
158        }
159
160        this.timing.stopTiming();
161
162        return hasUpdate || !this.onGround || Math.abs(this.motionX) > 0.00001 || Math.abs(this.motionY) > 0.00001 || Math.abs(this.motionZ) > 0.00001;
163    }
164
165    @Override
166    public boolean attack(EntityDamageEvent source) {
167        return (source.getCause() == DamageCause.VOID ||
168                source.getCause() == DamageCause.FIRE_TICK ||
169                source.getCause() == DamageCause.ENTITY_EXPLOSION ||
170                source.getCause() == DamageCause.BLOCK_EXPLOSION)
171                && super.attack(source);
172    }
173
174    public void setFirework(Item item) {
175        this.firework = item;
176        this.setDataProperty(new NBTEntityData(Entity.DATA_DISPLAY_ITEM, item.getNamedTag()));
177    }
178
179    @Override
180    public float getWidth() {
181        return 0.25f;
182    }
183
184    @Override
185    public float getHeight() {
186        return 0.25f;
187    }
188
189
190    @PowerNukkitOnly
191    @Since("1.5.1.0-PN")
192    @Override
193    public String getOriginalName() {
194        return "Firework Rocket";
195    }
196}
