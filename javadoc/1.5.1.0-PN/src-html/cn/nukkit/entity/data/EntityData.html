001package cn.nukkit.entity.data;
002
003import cn.nukkit.entity.Entity;
004import it.unimi.dsi.fastutil.ints.Int2ObjectMap;
005import it.unimi.dsi.fastutil.ints.Int2ObjectOpenHashMap;
006
007import java.lang.reflect.Modifier;
008import java.util.Arrays;
009import java.util.Objects;
010import java.util.SortedSet;
011import java.util.TreeSet;
012
013/**
014 * @author MagicDroidX (Nukkit Project)
015 */
016public abstract class EntityData<T> {
017    private static final Int2ObjectMap<String> KNOWN_ENTITY_DATA = new Int2ObjectOpenHashMap<>();
018    private static final Int2ObjectMap<String> KNOWN_ENTITY_FLAGS = new Int2ObjectOpenHashMap<>();
019    static {
020        Arrays.stream(Entity.class.getDeclaredFields())
021                .filter(it-> it.getModifiers() == (Modifier.PUBLIC | Modifier.STATIC | Modifier.FINAL))
022                .filter(it-> it.getType().equals(int.class))
023                .filter(it-> it.getName().startsWith("DATA_"))
024                .filter(it-> !it.getName().startsWith("DATA_FLAG_"))
025                .filter(it-> !it.getName().startsWith("DATA_TYPE_"))
026                .forEachOrdered(it-> {
027                    try {
028                        KNOWN_ENTITY_DATA.put(it.getInt(null), it.getName().substring(5));
029                    } catch (IllegalAccessException e) {
030                        throw new InternalError(e);
031                    }
032                });
033
034        Arrays.stream(Entity.class.getDeclaredFields())
035                .filter(it-> it.getModifiers() == (Modifier.PUBLIC | Modifier.STATIC | Modifier.FINAL))
036                .filter(it-> it.getType().equals(int.class))
037                .filter(it-> it.getName().startsWith("DATA_FLAG_"))
038                .forEachOrdered(it-> {
039                    try {
040                        KNOWN_ENTITY_FLAGS.put(it.getInt(null), it.getName().substring(10));
041                    } catch (IllegalAccessException e) {
042                        throw new InternalError(e);
043                    }
044                });
045    }
046
047    private int id;
048
049    protected EntityData(int id) {
050        this.id = id;
051    }
052
053    public abstract int getType();
054
055    public abstract T getData();
056
057    public abstract void setData(T data);
058
059    public int getId() {
060        return id;
061    }
062
063    public EntityData setId(int id) {
064        this.id = id;
065        return this;
066    }
067
068    @Override
069    public boolean equals(Object obj) {
070        return obj instanceof EntityData && ((EntityData) obj).getId() == this.getId() && Objects.equals(((EntityData) obj).getData(), this.getData());
071    }
072
073    private SortedSet<String> readFlags(int offset, long flags) {
074        SortedSet<String> flagSet = new TreeSet<>();
075        int limit = offset + 64;
076        for (Int2ObjectMap.Entry<String> flag : KNOWN_ENTITY_FLAGS.int2ObjectEntrySet()) {
077            if (flag.getIntKey() >= limit) continue;
078            long bit = 1L << (flag.getIntKey() - offset);
079            if ((flags & bit) != 0) {
080                flagSet.add(flag.getValue());
081            }
082        }
083        return flagSet;
084    }
085
086    @Override
087    public String toString() {
088        String idStr = KNOWN_ENTITY_DATA.get(id);
089        if (idStr == null) idStr = Integer.toString(id);
090        Object data = getData();
091        if (id == Entity.DATA_FLAGS) {
092            data = readFlags(0, (long) data);
093        } else if (id == Entity.DATA_FLAGS_EXTENDED) {
094            data = readFlags(64, (long) data);
095        }
096        return getClass().getSimpleName()+"{" +
097                "id=" + idStr + "," +
098                "data=" + data +
099                '}';
100    }
101}
