001package cn.nukkit.network.protocol;
002
003import cn.nukkit.api.PowerNukkitOnly;
004import cn.nukkit.api.Since;
005
006import java.util.ArrayList;
007import java.util.List;
008
009/**
010 * @author IWareQ
011 */
012 @PowerNukkitOnly
013@Since("1.5.1.0-PN")
014public class AnimateEntityPacket extends DataPacket {
015
016    @PowerNukkitOnly
017    @Since("1.5.1.0-PN")
018    public static final byte NETWORK_ID = ProtocolInfo.ANIMATE_ENTITY_PACKET;
019
020    private String animation;
021    private String nextState;
022    private String stopExpression;
023    private String controller;
024    private float blendOutTime;
025    private List<Long> entityRuntimeIds = new ArrayList<>();
026
027    @Override
028    public void decode() {
029        this.animation = this.getString();
030                this.nextState = this.getString();
031                this.stopExpression = this.getString();
032                this.controller = this.getString();
033                this.blendOutTime = this.getLFloat();
034                for (int i = 0, len = (int) this.getUnsignedVarInt(); i < len; i++) {
035                        this.entityRuntimeIds.add(this.getEntityRuntimeId());
036                }
037    }
038
039    @Override
040    public void encode() {
041        this.reset();
042        this.putString(this.animation);
043                this.putString(this.nextState);
044                this.putString(this.stopExpression);
045                this.putString(this.controller);
046                this.putLFloat(this.blendOutTime);
047                this.putUnsignedVarInt(this.entityRuntimeIds.size());
048                for (long entityRuntimeId : this.entityRuntimeIds){
049                        this.putEntityRuntimeId(entityRuntimeId);
050                }
051    }
052    
053    @Override
054    public byte pid() {
055        return NETWORK_ID;
056    }
057    
058    @PowerNukkitOnly
059    @Since("1.5.1.0-PN")
060    public void setAnimation(String animation) {
061        this.animation = animation;
062    }
063    
064    @PowerNukkitOnly
065    @Since("1.5.1.0-PN")
066    public String getAnimation() {
067        return this.animation;
068    }
069    
070    @PowerNukkitOnly
071    @Since("1.5.1.0-PN")
072    public void setNextState(String nextState) {
073        this.nextState = nextState;
074    }
075    
076    @PowerNukkitOnly
077    @Since("1.5.1.0-PN")
078    public String getNextState() {
079        return this.nextState;
080    }
081    
082    @PowerNukkitOnly
083    @Since("1.5.1.0-PN")
084    public void setStopExpression(String stopExpression) {
085        this.stopExpression = stopExpression;
086    }
087    
088    @PowerNukkitOnly
089    @Since("1.5.1.0-PN")
090    public String getStopExpression() {
091        return this.stopExpression;
092    }
093    
094    @PowerNukkitOnly
095    @Since("1.5.1.0-PN")
096    public void setController(String controller) {
097        this.controller = controller;
098    }
099    
100    @PowerNukkitOnly
101    @Since("1.5.1.0-PN")
102    public String getController() {
103        return this.controller;
104    }
105    
106    @PowerNukkitOnly
107    @Since("1.5.1.0-PN")
108    public void setBlendOutTime(float blendOutTime) {
109        this.blendOutTime = blendOutTime;
110    }
111    
112    @PowerNukkitOnly
113    @Since("1.5.1.0-PN")
114    public float getBlendOutTime() {
115        return this.blendOutTime;
116    }
117    
118    @PowerNukkitOnly
119    @Since("1.5.1.0-PN")
120    public void setEntityRuntimeIds(List<Long> entityRuntimeIds) {
121        this.entityRuntimeIds = entityRuntimeIds;
122    }
123    
124    @PowerNukkitOnly
125    @Since("1.5.1.0-PN")
126    public List<Long> getEntityRuntimeIds() {
127        return this.entityRuntimeIds;
128    }
129}
