001package cn.nukkit.blockstate.exception;
002
003import cn.nukkit.api.PowerNukkitOnly;
004import cn.nukkit.api.Since;
005
006import javax.annotation.Nonnull;
007import javax.annotation.Nullable;
008
009/**
010 * @author joserobjr
011 */
012@PowerNukkitOnly
013@Since("1.4.0.0-PN")
014public class InvalidBlockStateDataTypeException extends IllegalArgumentException {
015    private static final long serialVersionUID = 6883758182474914542L;
016
017    @PowerNukkitOnly
018    @Since("1.4.0.0-PN")
019    public InvalidBlockStateDataTypeException(@Nonnull Number blockData) {
020        super("The block data " + blockData + " has an unsupported type " + blockData.getClass());
021    }
022
023    @PowerNukkitOnly
024    @Since("1.4.0.0-PN")
025    public InvalidBlockStateDataTypeException(@Nonnull Number blockData, @Nullable Throwable cause) {
026        super("The block data " + blockData + " has an unsupported type " + blockData.getClass(), cause);
027    }
028}
