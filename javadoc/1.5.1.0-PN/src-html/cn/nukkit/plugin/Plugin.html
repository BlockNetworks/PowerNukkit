001package cn.nukkit.plugin;
002
003import cn.nukkit.Server;
004import cn.nukkit.api.PowerNukkitOnly;
005import cn.nukkit.api.Since;
006import cn.nukkit.command.CommandExecutor;
007import cn.nukkit.utils.Config;
008
009import java.io.File;
010import java.io.InputStream;
011
012/**
013 * 所有Nukkit插件必须实现的接口。<br>
014 * An interface what must be implemented by all Nukkit plugins.
015 *
016 * <p>对于插件作者，我们建议让插件主类继承{@link cn.nukkit.plugin.PluginBase}类，而不是实现这个接口。<br>
017 * For plugin developers: it's recommended to use {@link cn.nukkit.plugin.PluginBase} for an actual plugin
018 * instead of implement this interface.</p>
019 *
020 * @author MagicDroidX(code) @ Nukkit Project
021 * @author 粉鞋大妈(javadoc) @ Nukkit Project
022 * @see cn.nukkit.plugin.PluginBase
023 * @see cn.nukkit.plugin.PluginDescription
024 * @since Nukkit 1.0 | Nukkit API 1.0.0
025 */
026public interface Plugin extends CommandExecutor {
027    @PowerNukkitOnly
028    @Since("1.4.0.0-PN")
029    Plugin[] EMPTY_ARRAY = new Plugin[0];
030    
031    /**
032     * 在一个Nukkit插件被加载时调用的方法。这个方法会在{@link Plugin#onEnable()}之前调用。<br>
033     * Called when a Nukkit plugin is loaded, before {@link Plugin#onEnable()} .
034     *
035     * <p>应该填写加载插件时需要作出的动作。例如：初始化数组、初始化数据库连接。<br>
036     * Use this to init a Nukkit plugin, such as init arrays or init database connections.</p>
037     *
038     * @since Nukkit 1.0 | Nukkit API 1.0.0
039     */
040    void onLoad();
041
042    /**
043     * 在一个Nukkit插件被启用时调用的方法。<br>
044     * Called when a Nukkit plugin is enabled.
045     *
046     * <p>应该填写插件启用时需要作出的动作。例如：读取配置文件、读取资源、连接数据库。<br>
047     * Use this to open config files, open resources, connect databases.</p>
048     *
049     * <p>注意到可能存在的插件管理器插件，这个方法在插件多次重启时可能被调用多次。<br>
050     * Notes that there may be plugin manager plugins,
051     * this method can be called many times when a plugin is restarted many times.</p>
052     *
053     * @since Nukkit 1.0 | Nukkit API 1.0.0
054     */
055    void onEnable();
056
057    /**
058     * 返回这个Nukkit插件是否已启用。<br>
059     * Whether this Nukkit plugin is enabled.
060     *
061     * @return 这个插件是否已经启用。<br>Whether this plugin is enabled.
062     * @since Nukkit 1.0 | Nukkit API 1.0.0
063     */
064    boolean isEnabled();
065
066    /**
067     * 在一个Nukkit插件被停用时调用的方法。<br>
068     * Called when a Nukkit plugin is disabled.
069     *
070     * <p>应该填写插件停用时需要作出的动作。例如：关闭数据库，断开资源。<br>
071     * Use this to free open things and finish actions,
072     * such as disconnecting databases and close resources.</p>
073     *
074     * <p>注意到可能存在的插件管理器插件，这个方法在插件多次重启时可能被调用多次。<br>
075     * Notes that there may be plugin manager plugins,
076     * this method can be called many times when a plugin is restarted many times.</p>
077     *
078     * @since Nukkit 1.0 | Nukkit API 1.0.0
079     */
080    void onDisable();
081
082    /**
083     * 返回这个Nukkit插件是否已停用。<br>
084     * Whether this Nukkit plugin is disabled.
085     *
086     * @return 这个插件是否已经停用。<br>Whether this plugin is disabled.
087     * @since Nukkit 1.0 | Nukkit API 1.0.0
088     */
089    boolean isDisabled();
090
091    /**
092     * 返回这个Nukkit插件的数据文件夹。<br>
093     * The data folder of this Nukkit plugin.
094     *
095     * <p>一般情况下，数据文件夹名字与插件名字相同，而且都放在nukkit安装目录下的plugins文件夹里。<br>
096     * Under normal circumstances, the data folder has the same name with the plugin,
097     * and is placed in the 'plugins' folder inside the nukkit installation directory.</p>
098     *
099     * @return 这个插件的数据文件夹。<br>The data folder of this plugin.
100     * @since Nukkit 1.0 | Nukkit API 1.0.0
101     */
102    File getDataFolder();
103
104    /**
105     * 返回描述这个Nukkit插件的{@link PluginDescription}对象。<br>
106     * The description this Nukkit plugin as a {@link PluginDescription} object.
107     *
108     * <p>对于jar格式的Nukkit插件，插件的描述在plugin.yml文件内定义。<br>
109     * For jar-packed Nukkit plugins, the description is defined in the 'plugin.yml' file.</p>
110     *
111     * @return 这个插件的描述。<br>A description of this plugin.
112     * @see cn.nukkit.plugin.PluginDescription
113     * @since Nukkit 1.0 | Nukkit API 1.0.0
114     */
115    PluginDescription getDescription();
116
117    /**
118     * 读取这个插件特定的资源文件，并返回为{@code InputStream}对象。<br>
119     * Reads a resource of this plugin, and returns as an {@code InputStream} object.
120     *
121     * <p>对于jar格式的Nukkit插件，Nukkit会在jar包内的资源文件夹(一般为resources文件夹)寻找资源文件。<br>
122     * For jar-packed Nukkit plugins, Nukkit will look for your resource file in the resources folder,
123     * which is normally named 'resources' and placed in plugin jar file.</p>
124     *
125     * <p>当你需要把一个文件的所有内容读取为字符串，可以使用{@link cn.nukkit.utils.Utils#readFile}函数，
126     * 来从{@code InputStream}读取所有内容为字符串。例如：<br>
127     * When you need to read the whole file content as a String, you can use {@link cn.nukkit.utils.Utils#readFile}
128     * to read from a {@code InputStream} and get whole content as a String. For example:</p>
129     * <p><code>String string = Utils.readFile(this.getResource("string.txt"));</code></p>
130     *
131     * @param filename 要读取的资源文件名字。<br>The name of the resource file to read.
132     * @return 读取的资源文件的 {@code InputStream}对象。若错误会返回{@code null}<br>
133     * The resource as an {@code InputStream} object, or {@code null} when an error occurred.
134     * @since Nukkit 1.0 | Nukkit API 1.0.0
135     */
136    InputStream getResource(String filename);
137
138    /**
139     * 保存这个Nukkit插件的资源。<br>
140     * Saves the resource of this plugin.
141     *
142     * <p>对于jar格式的Nukkit插件，Nukkit会在jar包内的资源文件夹寻找资源文件，然后保存到数据文件夹。<br>
143     * For jar-packed Nukkit plugins, Nukkit will look for your resource file in the resources folder,
144     * which is normally named 'resources' and placed in plugin jar file, and copy it into data folder.</p>
145     *
146     * <p>这个函数通常用来在插件被加载(load)时，保存默认的资源文件。这样插件在启用(enable)时不会错误读取空的资源文件，
147     * 用户也无需从开发者处手动下载资源文件后再使用插件。<br>
148     * This is usually used to save the default plugin resource when the plugin is LOADED .If this is used,
149     * it won't happen to load an empty resource when plugin is ENABLED, and plugin users are not required to get
150     * default resources from the developer and place it manually. </p>
151     *
152     * <p>如果需要替换已存在的资源文件，建议使用{@link cn.nukkit.plugin.Plugin#saveResource(String, boolean)}<br>
153     * If you need to REPLACE an existing resource file, it's recommended
154     * to use {@link cn.nukkit.plugin.Plugin#saveResource(String, boolean)}.</p>
155     *
156     * @param filename 要保存的资源文件名字。<br>The name of the resource file to save.
157     * @return 保存是否成功。<br>true if the saving action is successful.
158     * @see cn.nukkit.plugin.Plugin#saveDefaultConfig
159     * @see cn.nukkit.plugin.Plugin#saveResource(String, boolean)
160     * @since Nukkit 1.0 | Nukkit API 1.0.0
161     */
162    boolean saveResource(String filename);
163
164    /**
165     * 保存或替换这个Nukkit插件的资源。<br>
166     * Saves or replaces the resource of this plugin.
167     *
168     * <p>对于jar格式的Nukkit插件，Nukkit会在jar包内的资源文件夹寻找资源文件，然后保存到数据文件夹。<br>
169     * For jar-packed Nukkit plugins, Nukkit will look for your resource file in the resources folder,
170     * which is normally named 'resources' and placed in plugin jar file, and copy it into data folder.</p>
171     *
172     * <p>如果需要保存默认的资源文件，建议使用{@link cn.nukkit.plugin.Plugin#saveResource(String)}<br>
173     * If you need to SAVE DEFAULT resource file, it's recommended
174     * to use {@link cn.nukkit.plugin.Plugin#saveResource(String)}.</p>
175     *
176     * @param filename 要保存的资源文件名字。<br>The name of the resource file to save.
177     * @param replace  是否替换目标文件。<br>if true, Nukkit will replace the target resource file.
178     * @return 保存是否成功。<br>true if the saving action is successful.
179     * @see cn.nukkit.plugin.Plugin#saveResource(String)
180     * @since Nukkit 1.0 | Nukkit API 1.0.0
181     */
182
183    boolean saveResource(String filename, boolean replace);
184
185    boolean saveResource(String filename, String outputName, boolean replace);
186
187    /**
188     * 返回这个Nukkit插件配置文件的{@link cn.nukkit.utils.Config}对象。<br>
189     * The config file this Nukkit plugin as a {@link cn.nukkit.utils.Config} object.
190     *
191     * <p>一般地，插件的配置保存在数据文件夹下的config.yml文件。<br>
192     * Normally, the plugin config is saved in the 'config.yml' file in its data folder.</p>
193     *
194     * @return 插件的配置文件。<br>The configuration of this plugin.
195     * @see cn.nukkit.plugin.Plugin#getDataFolder
196     * @since Nukkit 1.0 | Nukkit API 1.0.0
197     */
198    Config getConfig();
199
200    /**
201     * 保存这个Nukkit插件的配置文件。<br>
202     * Saves the plugin config.
203     *
204     * @see cn.nukkit.plugin.Plugin#getDataFolder
205     * @since Nukkit 1.0 | Nukkit API 1.0.0
206     */
207    void saveConfig();
208
209    /**
210     * 保存这个Nukkit插件的默认配置文件。<br>
211     * Saves the DEFAULT plugin config.
212     *
213     * <p>执行这个函数时，Nukkit会在资源文件夹内寻找开发者配置好的默认配置文件config.yml，然后保存在数据文件夹。
214     * 如果数据文件夹已经有一个config.yml文件，Nukkit不会替换这个文件。<br>
215     * When this is used, Nukkit will look for the default 'config.yml' file which is configured by plugin developer
216     * and save it to the data folder. If a config.yml file exists in the data folder, Nukkit won't replace it.</p>
217     *
218     * <p>这个函数通常用来在插件被加载(load)时，保存默认的配置文件。这样插件在启用(enable)时不会错误读取空的配置文件，
219     * 用户也无需从开发者处手动下载配置文件保存后再使用插件。<br>
220     * This is usually used to save the default plugin config when the plugin is LOADED .If this is used,
221     * it won't happen to load an empty config when plugin is ENABLED, and plugin users are not required to get
222     * default config from the developer and place it manually. </p>
223     *
224     * @see cn.nukkit.plugin.Plugin#getDataFolder
225     * @see cn.nukkit.plugin.Plugin#saveResource
226     * @since Nukkit 1.0 | Nukkit API 1.0.0
227     */
228    void saveDefaultConfig();
229
230    /**
231     * 重新读取这个Nukkit插件的默认配置文件。<br>
232     * Reloads the plugin config.
233     *
234     * <p>执行这个函数时，Nukkit会从数据文件夹中的config.yml文件重新加载配置。
235     * 这样用户在调整插件配置后，无需重启就可以马上使用新的配置。<br>
236     * By using this, Nukkit will reload the config from 'config.yml' file, then it isn't necessary to restart
237     * for plugin user who changes the config and needs to use new config at once.</p>
238     *
239     * @see cn.nukkit.plugin.Plugin#getDataFolder
240     * @since Nukkit 1.0 | Nukkit API 1.0.0
241     */
242    void reloadConfig();
243
244    /**
245     * 返回运行这个插件的服务器的{@link cn.nukkit.Server}对象。<br>
246     * Gets the server which is running this plugin, and returns as a {@link cn.nukkit.Server} object.
247     *
248     * @see cn.nukkit.Server
249     * @since Nukkit 1.0 | Nukkit API 1.0.0
250     */
251    Server getServer();
252
253    /**
254     * 返回这个插件的名字。<br>
255     * Returns the name of this plugin.
256     *
257     * <p>Nukkit会从已经读取的插件描述中获取插件的名字。<br>
258     * Nukkit will read plugin name from plugin description.</p>
259     *
260     * @see cn.nukkit.plugin.Plugin#getDescription
261     * @since Nukkit 1.0 | Nukkit API 1.0.0
262     */
263    String getName();
264
265    /**
266     * 返回这个插件的日志记录器为{@link cn.nukkit.plugin.PluginLogger}对象。<br>
267     * Returns the logger of this plugin as a {@link cn.nukkit.plugin.PluginLogger} object.
268     *
269     * <p>使用日志记录器，你可以在控制台和日志文件输出信息。<br>
270     * You can use a plugin logger to output messages to the console and log file.</p>
271     *
272     * @see cn.nukkit.plugin.PluginLogger
273     * @since Nukkit 1.0 | Nukkit API 1.0.0
274     */
275    PluginLogger getLogger();
276
277    /**
278     * 返回这个插件的加载器为{@link cn.nukkit.plugin.PluginLoader}对象。<br>
279     * Returns the loader of this plugin as a {@link cn.nukkit.plugin.PluginLoader} object.
280     *
281     * @see cn.nukkit.plugin.PluginLoader
282     * @since Nukkit 1.0 | Nukkit API 1.0.0
283     */
284    PluginLoader getPluginLoader();
