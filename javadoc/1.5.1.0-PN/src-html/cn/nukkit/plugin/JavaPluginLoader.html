033    @Override
034    public Plugin loadPlugin(File file) throws Exception {
035        PluginDescription description = this.getPluginDescription(file);
036        if (description != null) {
037            log.info(this.server.getLanguage().translateString("nukkit.plugin.load", description.getFullName()));
038            File dataFolder = new File(file.getParentFile(), description.getName());
039            if (dataFolder.exists() && !dataFolder.isDirectory()) {
040                throw new IllegalStateException("Projected dataFolder \'" + dataFolder.toString() + "\' for " + description.getName() + " exists and is not a directory");
041            }
042            String className = description.getMain();
043            PluginClassLoader classLoader = new PluginClassLoader(this, this.getClass().getClassLoader(), file);
044            this.classLoaders.put(description.getName(), classLoader);
045            PluginBase plugin;
046            try {
047                Class javaClass = classLoader.loadClass(className);
048                if (!PluginBase.class.isAssignableFrom(javaClass)) {
049                    throw new PluginException("Main class `" + description.getMain() + "\' does not extend PluginBase");
050                }
051                try {
052                    Class<PluginBase> pluginClass = (Class<PluginBase>) javaClass.asSubclass(PluginBase.class);
053                    plugin = pluginClass.newInstance();
054                    this.initPlugin(plugin, description, dataFolder, file);
055                    return plugin;
056                } catch (ClassCastException e) {
057                    throw new PluginException("Error whilst initializing main class `" + description.getMain() + "\'", e);
058                } catch (InstantiationException | IllegalAccessException e) {
059                    log.error("An exception happened while initializing the plgin {}, {}, {}, {}", file, className, description.getName(), description.getVersion(), e);
060                }
061            } catch (ClassNotFoundException e) {
062                throw new PluginException("Couldn\'t load plugin " + description.getName() + ": main class not found");
063            }
064        }
065        return null;
066    }

074    public PluginDescription getPluginDescription(File file) {
075        try (JarFile jar = new JarFile(file)) {
076            JarEntry entry = jar.getJarEntry("nukkit.yml");
077            if (entry == null) {
078                entry = jar.getJarEntry("plugin.yml");
079                if (entry == null) {
080                    return null;
081                }
082            }
083            try (InputStream stream = jar.getInputStream(entry)) {
084                return new PluginDescription(Utils.readFile(stream));
085            }
086        } catch (IOException e) {
087            return null;
088        }

117    public void disablePlugin(Plugin plugin) {
118        if (plugin instanceof PluginBase && plugin.isEnabled()) {
119            if (plugin == PowerNukkitPlugin.getInstance()) {
120                throw new UnsupportedOperationException("The PowerNukkitPlugin cannot be disabled");
121            }
122            log.info(this.server.getLanguage().translateString("nukkit.plugin.disable", plugin.getDescription().getFullName()));
123            this.server.getServiceManager().cancel(plugin);
124            this.server.getPluginManager().callEvent(new PluginDisableEvent(plugin));
125            ((PluginBase) plugin).setEnabled(false);
126        }
