002package cn.nukkit.plugin;

004import cn.nukkit.event.Event;
005import cn.nukkit.event.Listener;
006import cn.nukkit.utils.EventException;
007import java.lang.reflect.InvocationTargetException;
008import java.lang.reflect.Method;
009
010/**
011 * @author MagicDroidX (Nukkit Project)
012 */
013public class MethodEventExecutor implements EventExecutor {
014    @java.lang.SuppressWarnings("all")
015    private static final org.apache.logging.log4j.Logger log = org.apache.logging.log4j.LogManager.getLogger(MethodEventExecutor.class);
016    private final Method method;
017
018    public MethodEventExecutor(Method method) {
019        this.method = method;
020    }
021
022    @SuppressWarnings("unchecked")
023    @Override
024    public void execute(Listener listener, Event event) throws EventException {
025        try {
026            Class<Event>[] params = (Class<Event>[]) method.getParameterTypes();
027            for (Class<Event> param : params) {
028                if (param.isAssignableFrom(event.getClass())) {
029                    method.invoke(listener, event);
030                    break;
031                }
032            }
033        } catch (InvocationTargetException ex) {
034            throw new EventException(ex.getCause() != null ? ex.getCause() : ex);
035        } catch (ClassCastException ex) {
036            log.debug("Ignoring a ClassCastException", ex);
037        } catch (
038        // We are going to ignore ClassCastException because EntityDamageEvent can't be cast to EntityDamageByEntityEvent
039        Throwable t) {
040            throw new EventException(t);
041        }
042    }
043
044    public Method getMethod() {
045        return method;
046    }
047}
