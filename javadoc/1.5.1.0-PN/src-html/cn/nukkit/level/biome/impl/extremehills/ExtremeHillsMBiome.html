001package cn.nukkit.level.biome.impl.extremehills;
002
003import cn.nukkit.api.NewRakNetOnly;
004import cn.nukkit.api.PowerNukkitOnly;
005import cn.nukkit.api.Since;
006import cn.nukkit.blockstate.BlockState;
007import cn.nukkit.level.generator.noise.nukkit.f.SimplexF;
008import cn.nukkit.math.NukkitRandom;
009
010/**
011 * @author DaPorkchop_ (Nukkit Project)
012 * <p>
013 * make sure this is touching another extreme hills type or it'll look dumb
014 *
015 * very smooth hills with flat areas between
016 */
017public class ExtremeHillsMBiome extends ExtremeHillsPlusBiome {
018    private static final BlockState STATE_GRAVEL = BlockState.of(GRAVEL);
019    private static final BlockState STATE_GRASS = BlockState.of(GRASS);
020    private static final SimplexF gravelNoise = new SimplexF(new NukkitRandom(0), 1f, 1 / 4f, 1 / 64f);
021
022    public ExtremeHillsMBiome() {
023        this(true);
024    }
025
026    public ExtremeHillsMBiome(boolean tree) {
027        super(tree);
028
029        this.setBaseHeight(1f);
030        this.setHeightVariation(0.5f);
031    }
032
033    @Override
034    public String getName() {
035        return "Extreme Hills M";
036    }
037
038    @PowerNukkitOnly
039    @Since("1.4.0.0-PN")
040    @Override
041    public BlockState getSurfaceState(int x, int y, int z) {
042        return gravelNoise.noise2D(x, z, true) < -0.75f ? STATE_GRAVEL : STATE_GRASS;
043    }
044
045    @NewRakNetOnly
046    @Override
047    public int getSurfaceDepth(int x, int y, int z) {
048        return gravelNoise.noise2D(x, z, true) < -0.75f ? 4 : 1;
049    }
050    
051    @NewRakNetOnly
052    @Override
053    public int getGroundDepth(int x, int y, int z) {
054        return gravelNoise.noise2D(x, z, true) < -0.75f ? 0 : 4;
055    }
056
057    @Override
058    public boolean doesOverhang() {
059        return false;
060    }
061}
