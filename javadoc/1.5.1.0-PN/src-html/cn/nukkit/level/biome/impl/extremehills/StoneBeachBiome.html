001package cn.nukkit.level.biome.impl.extremehills;
002
003import cn.nukkit.api.RemovedFromNewRakNet;
004import cn.nukkit.api.Since;
005import cn.nukkit.level.biome.type.CoveredBiome;
006
007/**
008 * @author DaPorkchop_
009 * <p>
010 * Occurs when Extreme hills and variants touch the ocean.
011 *
012 * Nearly ertical cliffs, but no overhangs. Height difference is 2-7 near ocean, and pretty much flat everywhere else
013 */
014public class StoneBeachBiome extends CoveredBiome {
015    public StoneBeachBiome() {
016        this.setBaseHeight(0.1f);
017        this.setHeightVariation(0.8f);
018    }
019
020    @RemovedFromNewRakNet
021    @Since("1.4.0.0-PN")
022    @Override
023    public int getSurfaceDepth(int y) {
024        if (useNewRakNetSurfaceDepth()) {
025            return getSurfaceDepth(0,y,0);
026        }
027        return 0;
028    }
029
030    @RemovedFromNewRakNet
031    @Since("1.4.0.0-PN")
032    @Override
033    public int getSurfaceBlock(int y) {
034        if (useNewRakNetSurface()) {
035            return getSurfaceId(0,y,0) >> 4;
036        }
037        return 0;
038    }
039
040    @RemovedFromNewRakNet
041    @Since("1.4.0.0-PN")
042    @Override
043    public int getGroundDepth(int y) {
044        if (useNewRakNetGroundDepth()) {
045            return getGroundDepth(0,y,0);
046        }
047        return 0;
048    }
049
050    @RemovedFromNewRakNet
051    @Since("1.4.0.0-PN")
052    @Override
053    public int getGroundBlock(int y) {
054        if (useNewRakNetGround()) {
055            return getGroundId(0,y,0) >> 4;
056        }
057        return 0;
058    }
059
060    @Override
061    public String getName() {
062        return "Stone Beach";
063    }
064}
