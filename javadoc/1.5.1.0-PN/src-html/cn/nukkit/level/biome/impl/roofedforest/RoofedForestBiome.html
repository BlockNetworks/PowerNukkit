001package cn.nukkit.level.biome.impl.roofedforest;
002
003import cn.nukkit.level.biome.type.GrassyBiome;
004import cn.nukkit.level.generator.populator.impl.MushroomPopulator;
005import cn.nukkit.level.generator.populator.impl.PopulatorFlower;
006import cn.nukkit.level.generator.populator.impl.tree.DarkOakTreePopulator;
007
008public class RoofedForestBiome extends GrassyBiome {
009
010    public RoofedForestBiome() {
011        super();
012
013        DarkOakTreePopulator tree = new DarkOakTreePopulator();
014        tree.setBaseAmount(20);
015        tree.setRandomAmount(10);
016        this.addPopulator(tree);
017
018        PopulatorFlower flower = new PopulatorFlower();
019        flower.setBaseAmount(2);
020        this.addPopulator(flower);
021
022        MushroomPopulator mushroom = new MushroomPopulator();
023        mushroom.setBaseAmount(0);
024        mushroom.setRandomAmount(1);
025        this.addPopulator(mushroom);
026    }
027
028    @Override
029    public String getName() {
030        return "Roofed Forest";
031    }
032
033}
