001package cn.nukkit.level.biome;
002
003import cn.nukkit.api.PowerNukkitOnly;
004import cn.nukkit.api.Since;
005import cn.nukkit.block.BlockID;
006import cn.nukkit.level.ChunkManager;
007import cn.nukkit.level.format.FullChunk;
008import cn.nukkit.level.generator.populator.type.Populator;
009import cn.nukkit.math.NukkitRandom;
010
011import java.util.ArrayList;
012import java.util.List;
013
014/**
015 * @author MagicDroidX (Nukkit Project)
016 */
017public abstract class Biome implements BlockID {
018    public static final int MAX_BIOMES = 256;
019    public static final Biome[] biomes = new Biome[MAX_BIOMES];
020    public static final List<Biome> unorderedBiomes = new ArrayList<>();
021
022    private final ArrayList<Populator> populators = new ArrayList<>();
023    private int id;
024    private float baseHeight = 0.1f;
025    private float heightVariation = 0.3f;
026
027    protected static void register(int id, Biome biome) {
028        biome.setId(id);
029        biomes[id] = biome;
030        unorderedBiomes.add(biome);
031    }
032
033    public static Biome getBiome(int id) {
034        Biome biome = biomes[id];
035        return biome != null ? biome : EnumBiome.OCEAN.biome;
036    }
037
038    /**
039     * Get Biome by name.
040     *
041     * @param name Name of biome. Name could contain symbol "_" instead of space
042     * @return Biome. Null - when biome was not found
043     */
044    public static Biome getBiome(String name) {
045        for (Biome biome : biomes) {
046            if (biome != null) {
047                if (biome.getName().equalsIgnoreCase(name.replace("_", " "))) return biome;
048            }
049        }
050        return null;
051    }
052
053    public void clearPopulators() {
054        this.populators.clear();
055    }
056
057    public void addPopulator(Populator populator) {
058        this.populators.add(populator);
059    }
060
061    public void populateChunk(ChunkManager level, int chunkX, int chunkZ, NukkitRandom random) {
062        FullChunk chunk = level.getChunk(chunkX, chunkZ);
063        for (Populator populator : populators) {
064            populator.populate(level, chunkX, chunkZ, random, chunk);
065        }
066    }
067
068    public ArrayList<Populator> getPopulators() {
069        return populators;
070    }
071
072    public int getId() {
073        return id;
074    }
075
076    public void setId(int id) {
077        this.id = id;
078    }
079
080    public abstract String getName();
081
082    public void setBaseHeight(float baseHeight) {
083        this.baseHeight = baseHeight;
084    }
085
086    public void setHeightVariation(float heightVariation)   {
087        this.heightVariation = heightVariation;
088    }
089
090    public float getBaseHeight() {
091        return baseHeight;
092    }
093
094    public float getHeightVariation() {
095        return heightVariation;
096    }
097
098    @Override
099    public int hashCode() {
100        return getId();
101    }
102
103    @Override
104    public boolean equals(Object obj) {
105        return hashCode() == obj.hashCode();
106    }
107
108    //whether or not water should freeze into ice on generation
109    public boolean isFreezing() {
110        return false;
111    }
112
113    /**
114     * Whether or not overhangs should generate in this biome (places where solid blocks generate over air)
115     *
116     * This should probably be used with a custom max elevation or things can look stupid
117     *
118     * @return overhang
119     */
120    public boolean doesOverhang()   {
121        return false;
122    }
123
124    /**
125     * How much offset should be added to the min/max heights at this position
126     *
127     * @param x x
128     * @param z z
129     * @return height offset
130     */
131    public int getHeightOffset(int x, int z)    {
132        return 0;
133    }
134
135    public boolean canRain() {
136        return true;
137    }
138
139    @PowerNukkitOnly
140    @Since("1.4.0.0-PN")
141    public boolean isDry() {
142        return false;
143    }
144}
