001package cn.nukkit.level;
002
003import cn.nukkit.api.DeprecationDetails;
004import cn.nukkit.api.PowerNukkitOnly;
005import cn.nukkit.api.Since;
006
007import java.util.Optional;
008
009public enum GameRule {
010
011    COMMAND_BLOCKS_ENABLED("commandBlocksEnabled"),
012    COMMAND_BLOCK_OUTPUT("commandBlockOutput"),
013    DO_DAYLIGHT_CYCLE("doDaylightCycle"),
014    DO_ENTITY_DROPS("doEntityDrops"),
015    DO_FIRE_TICK("doFireTick"),
016    DO_INSOMNIA("doInsomnia"),
017    DO_IMMEDIATE_RESPAWN("doImmediateRespawn"),
018    DO_MOB_LOOT("doMobLoot"),
019    DO_MOB_SPAWNING("doMobSpawning"),
020    DO_TILE_DROPS("doTileDrops"),
021    DO_WEATHER_CYCLE("doWeatherCycle"),
022    DROWNING_DAMAGE("drowningDamage"),
023    FALL_DAMAGE("fallDamage"),
024    FIRE_DAMAGE("fireDamage"),
025    @Since("1.5.1.0-PN") FREEZE_DAMAGE("freezeDamage"),
026    FUNCTION_COMMAND_LIMIT("functionCommandLimit"),
027    KEEP_INVENTORY("keepInventory"),
028    MAX_COMMAND_CHAIN_LENGTH("maxCommandChainLength"),
029    MOB_GRIEFING("mobGriefing"),
030    NATURAL_REGENERATION("naturalRegeneration"),
031    PVP("pvp"),
032    RANDOM_TICK_SPEED("randomTickSpeed"),
033    SEND_COMMAND_FEEDBACK("sendCommandFeedback"),
034    SHOW_COORDINATES("showCoordinates"),
035    @Since("1.5.1.0-PN") SHOW_DEATH_MESSAGES("showDeathMessages"),
036
037    @Deprecated
038    @PowerNukkitOnly("Renamed to SHOW_DEATH_MESSAGE by NukkitX") 
039    @DeprecationDetails(since = "1.5.1.0-PN", 
040            reason = "Added by upstream with a different name",
041            replaceWith = "SHOW_DEATH_MESSAGES")
042    @SuppressWarnings("DeprecatedIsStillUsed")
043    SHOW_DEATH_MESSAGE(SHOW_DEATH_MESSAGES.name, true),
044    
045    SPAWN_RADIUS("spawnRadius"),
046    TNT_EXPLODES("tntExplodes"),
047    EXPERIMENTAL_GAMEPLAY("experimentalGameplay"),
048    SHOW_TAGS("showTags");
049
050    private final String name;
051    private final boolean deprecated;
052
053    @PowerNukkitOnly
054    @Since("1.4.0.0-PN")
055    public static final GameRule[] EMPTY_ARRAY = new GameRule[0];
056
057    GameRule(String name) {
058        this.name = name;
059        this.deprecated = false;
060    }
061
062    GameRule(String name, boolean deprecated) {
063        this.name = name;
064        this.deprecated = deprecated;
065    }
066
067    public static Optional<GameRule> parseString(String gameRuleString) {
068        //Backward compatibility
069        if ("showDeathMessage".equalsIgnoreCase(gameRuleString)) {
070            gameRuleString = "showDeathMessages";
071        }
072
073        for (GameRule gameRule: values()) {
074            if (gameRule.getName().equalsIgnoreCase(gameRuleString)) {
075                return Optional.of(gameRule);
076            }
077        }
078        return Optional.empty();
079    }
080
081    public static String[] getNames() {
082        String[] stringValues = new String[values().length];
083
084        for (int i = 0; i < values().length; i++) {
085            stringValues[i] = values()[i].getName();
086        }
087        return stringValues;
088    }
089
090    public String getName() {
091        return name;
092    }
093
094    @PowerNukkitOnly
095    @Since("1.5.1.0-PN")
096    public boolean isDeprecated() {
097        return deprecated;
098    }
099}
