001package cn.nukkit.level.particle;
002
003import cn.nukkit.math.Vector3;
004import cn.nukkit.utils.BlockColor;
005
006/**
007 * @author xtypr
008 * @since 2016/1/4
009 */
010public class InstantSpellParticle extends SpellParticle {
011    protected int data;
012
013    public InstantSpellParticle(Vector3 pos) {
014        this(pos, 0);
015    }
016
017    public InstantSpellParticle(Vector3 pos, int data) {
018        super(pos, data);
019    }
020
021    public InstantSpellParticle(Vector3 pos, BlockColor blockColor) {
022        //alpha is ignored
023        this(pos, blockColor.getRed(), blockColor.getGreen(), blockColor.getBlue());
024    }
025
026    public InstantSpellParticle(Vector3 pos, int r, int g, int b) {
027        //this 0x01 is the only difference between instant spell and non-instant one
028        super(pos, r, g, b, 0x01);
029    }
030
031}
