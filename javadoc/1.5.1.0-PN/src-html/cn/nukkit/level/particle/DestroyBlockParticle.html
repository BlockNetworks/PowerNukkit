001package cn.nukkit.level.particle;
002
003import cn.nukkit.block.Block;
004import cn.nukkit.level.GlobalBlockPalette;
005import cn.nukkit.math.Vector3;
006import cn.nukkit.network.protocol.DataPacket;
007import cn.nukkit.network.protocol.LevelEventPacket;
008
009/**
010 * @author xtypr
011 * @since 2015/11/21
012 */
013public class DestroyBlockParticle extends Particle {
014
015    protected final int data;
016
017    public DestroyBlockParticle(Vector3 pos, Block block) {
018        super(pos.x, pos.y, pos.z);
019        this.data = GlobalBlockPalette.getOrCreateRuntimeId(block.getId(), block.getDamage());
020    }
021
022    @Override
023    public DataPacket[] encode() {
024        LevelEventPacket pk = new LevelEventPacket();
025        pk.evid = LevelEventPacket.EVENT_PARTICLE_DESTROY;
026        pk.x = (float) this.x;
027        pk.y = (float) this.y;
028        pk.z = (float) this.z;
029        pk.data = this.data;
030
031        return new DataPacket[]{pk};
032    }
033}
