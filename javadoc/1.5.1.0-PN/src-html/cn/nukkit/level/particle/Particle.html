104    @Since("1.4.0.0-PN")
105    public static Integer getParticleIdByName(String name) {
106        name = name.toUpperCase();
107
108        try {
109            Field field = Particle.class.getDeclaredField(name.startsWith("TYPE_") ? name : "TYPE_" + name);
110
111            Class<?> type = field.getType();
112
113            if(type==int.class) {
114                return field.getInt(null);
115            }
116        } catch(NoSuchFieldException | IllegalAccessException e) {
117            // ignore
118        }
119        return null;
120    }
121
122    @Since("1.4.0.0-PN")
123    public static boolean particleExists(String name) {
124        return getParticleIdByName(name) != null;   
125    }
