101    public void setGameRules(GameRule gameRule, String value) throws IllegalArgumentException {
102        Preconditions.checkNotNull(gameRule, "gameRule");
103        Preconditions.checkNotNull(value, "value");
104
105        switch (getGameRuleType(gameRule)) {
106            case BOOLEAN:
107                if (value.equalsIgnoreCase("true")) {
108                    setGameRule(gameRule, true);
109                } else if (value.equalsIgnoreCase("false")) {
110                    setGameRule(gameRule, false);
111                } else {
112                    throw new IllegalArgumentException("Was not a boolean");
113                }
114                break;
115            case INTEGER:
116                setGameRule(gameRule, Integer.parseInt(value));
117                break;
118            case FLOAT:
119                setGameRule(gameRule, Float.parseFloat(value));
120        }
121    }

166    public void readNBT(CompoundTag nbt) {
167        Preconditions.checkNotNull(nbt);
168        for (String key : nbt.getTags().keySet()) {
169            Optional<GameRule> gameRule = GameRule.parseString(key);
170            if (!gameRule.isPresent()) {
171                continue;
172            }
173
174            setGameRules(gameRule.get(), nbt.getString(key));
175        }
176    }

178    public enum Type {
179        UNKNOWN {
180            @Override
181            void write(BinaryStream pk, Value value) {
182            }
183        },
184        BOOLEAN {
185            @Override
186            void write(BinaryStream pk, Value value) {
187                pk.putBoolean(value.getValueAsBoolean());
188            }
189        },
190        INTEGER {
191            @Override
192            void write(BinaryStream pk, Value value) {
193                pk.putUnsignedVarInt(value.getValueAsInteger());
194            }
195        },
196        FLOAT {
197            @Override
198            void write(BinaryStream pk, Value value) {
199                pk.putLFloat(value.getValueAsFloat());
200            }
201        };
202
203        abstract void write(BinaryStream pk, Value value);
204    }
205
206    public static class Value<T> {
207        private final Type type;
208        private T value;
209        private boolean canBeChanged;
210
211        public Value(Type type, T value) {
212            this.type = type;
213            this.value = value;
214        }
215
216        private void setValue(T value, Type type) {
217            if (this.type != type) {
218                throw new UnsupportedOperationException("Rule not of type " + type.name().toLowerCase());
219            }
220            this.value = value;
221        }
222
223        @Since("1.5.0.0-PN")
224        public boolean isCanBeChanged() {
225            return this.canBeChanged;
226        }
227
228        @Since("1.5.0.0-PN")
229        public void setCanBeChanged(boolean canBeChanged) {
230            this.canBeChanged = canBeChanged;
231        }
232
233        public Type getType() {
234            return this.type;
235        }
236
237        private boolean getValueAsBoolean() {
238            if (type != Type.BOOLEAN) {
239                throw new UnsupportedOperationException("Rule not of type boolean");
240            }
241            return (Boolean) value;
242        }
243
244        private int getValueAsInteger() {
245            if (type != Type.INTEGER) {
246                throw new UnsupportedOperationException("Rule not of type integer");
247            }
248            return (Integer) value;
249        }
250
251        private float getValueAsFloat() {
252            if (type != Type.FLOAT) {
253                throw new UnsupportedOperationException("Rule not of type float");
254            }
255            return (Float) value;
256        }
257
258        @Since("1.5.0.0-PN")
259        public void write(BinaryStream stream) {
260            stream.putBoolean(this.canBeChanged);
261            stream.putUnsignedVarInt(type.ordinal());
262            type.write(stream, this);
263        }
264    }
