019package cn.nukkit.level.format.anvil.util;
020
021import cn.nukkit.api.API;
022import cn.nukkit.api.PowerNukkitOnly;
023import cn.nukkit.api.Since;
024import cn.nukkit.blockstate.BlockState;
025import cn.nukkit.level.util.PalettedBlockStorage;
026
027import javax.annotation.Nonnull;
028import javax.annotation.Nullable;
029import java.util.BitSet;
030
031import static cn.nukkit.api.API.Definition.INTERNAL;
032import static cn.nukkit.api.API.Usage.BLEEDING;
033
034/**
035 * @author joserobjr
036 * @since 2020-10-02
037 */
038@PowerNukkitOnly
039@Since("1.4.0.0-PN")
040public final class ImmutableBlockStorage extends BlockStorage {
041    @PowerNukkitOnly
042    @Since("1.4.0.0-PN")
043    public static final ImmutableBlockStorage EMPTY = new BlockStorage().immutableCopy();
044
045    @PowerNukkitOnly
046    @Since("1.4.0.0-PN")
047    @API(definition = INTERNAL, usage = BLEEDING)
048    ImmutableBlockStorage(BlockState[] states, byte flags, PalettedBlockStorage palette, @Nullable BitSet denyStates) {
049        super(states.clone(), flags, palette.copy(), denyStates != null? (BitSet)denyStates.clone() : null);
050    }
051
052    @Override
053    protected BlockState setBlockState(int index, @Nonnull BlockState state) {
054        throw new UnsupportedOperationException("This BlockStorage is immutable");
055    }
056
057    @Since("1.4.0.0-PN")
058    @PowerNukkitOnly
059    @Override
060    public void delayPaletteUpdates() {
061        
062    }
063
064    @Since("1.4.0.0-PN")
065    @PowerNukkitOnly
066    @Override
067    public void recheckBlocks() {
068        
069    }
070
071    @Since("1.4.0.0-PN")
072    @PowerNukkitOnly
073    @Nonnull
074    @Override
075    public ImmutableBlockStorage immutableCopy() {
076        return this;
077    }
078}
