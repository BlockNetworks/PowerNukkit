495    private void inflate() {
496        try {
497            if (compressedLight != null && compressedLight.length != 0) {
498                byte[] inflated = Zlib.inflate(compressedLight);
499                blockLight = Arrays.copyOfRange(inflated, 0, 2048);
500                if (inflated.length > 2048) {
501                    skyLight = Arrays.copyOfRange(inflated, 2048, 4096);
502                } else {
503                    skyLight = new byte[2048];
504                    if (hasSkyLight) {
505                        Arrays.fill(skyLight, (byte) 255);
506                    }
507                }
508                compressedLight = null;
509            } else {
510                blockLight = new byte[2048];
511                skyLight = new byte[2048];
512                if (hasSkyLight) {
513                    Arrays.fill(skyLight, (byte) 255);
514                }
515            }

517            log.error("Failed to decompress a chunk section", e);

539        for (int i = 0; i < 4096; i++) {

554    @Nullable
555    private List<byte[]> saveData(BlockStorage storage, byte[] idsBase, @Nullable byte[] idsExtra, NibbleArray dataBase, @Nullable NibbleArray dataExtra) {
556        boolean huge = storage.hasBlockDataHuge();
557        boolean big = huge || storage.hasBlockDataBig();
558        List<byte[]> hugeList = big ? new ArrayList<>(huge ? 3 : 1) : null;
559        if (big) {
560            hugeList.add(new byte[BlockStorage.SECTION_SIZE]);
561        }
562        storage.iterateStates(((bx, by, bz, state) -> {
563            int anvil = getAnvilIndex(bx, by, bz);
564            int blockId = state.getBlockId();
565            if (blockId == 0) {
566                return;
567            }
568            idsBase[anvil] = (byte) (blockId & 255);
569            if (idsExtra != null) {
570                idsExtra[anvil] = (byte) (blockId >>> 8 & 255);
571            }
572            @SuppressWarnings("deprecation")
573            int unsignedIntData = state.getBigDamage();
574            dataBase.set(anvil, (byte) (unsignedIntData & 15));
575            if (dataExtra != null) {
576                dataExtra.set(anvil, (byte) (unsignedIntData >>> 4 & 15));
577            }
578            if (!big) {
579                return;
580            }
581            hugeList.get(0)[anvil] = (byte) (unsignedIntData >>> 8 & 255);
582            if (huge) {
583                saveHugeData(hugeList, state, anvil, unsignedIntData);
584            }
585        }));
586        return hugeList;
587    }

589    private void saveHugeData(List<byte[]> hugeList, BlockState state, int anvil, int intData) {
590        int bitSize = state.getBitSize();
591        if (bitSize <= 16) {
592            return;
593        }
594        intData >>>= 16;
595        int processedBits = 16;
596        int pos = 1;
597        for (; processedBits < 32 && processedBits <= bitSize; processedBits += 8, pos++, intData >>>= 8) {
598            byte[] blob = allocateBlob(hugeList, pos);
599            blob[anvil] = (byte) (intData & 255);
600        }
601        if (processedBits >= bitSize) {
602            return;
603        }
604        BigInteger hugeData = state.getHugeDamage().shiftRight(32);
605        for (; processedBits <= bitSize; processedBits += 8, pos++, hugeData = hugeData.shiftRight(8)) {
606            byte[] blob = allocateBlob(hugeList, pos);
607            blob[anvil] = hugeData.and(BYTE_MASK).byteValue();
608        }
609    }

611    private byte[] allocateBlob(List<byte[]> hugeList, int pos) {
612        byte[] blob;
613        if (hugeList.size() <= pos) {
614            blob = new byte[BlockStorage.SECTION_SIZE];
615            hugeList.add(blob);
616        } else {
617            blob = hugeList.get(pos);
618        }
619        return blob;
620    }

622    @Nonnull
623    @Override
624    public synchronized CompoundTag toNBT() {
625        CompoundTag s = new CompoundTag();
626        compressStorageLayers();
627        // For simplicity, not using the actual palette format to save in the disk
628        // And for better compatibility, attempting to use the closest to the old format as possible
629        // Version 0 = old format (single block storage, Blocks and Data tags only)
630        // Version 1 = old format extended same as 0 but may have BlocksExtra and DataExtra
631        // Version 7 = new format (multiple block storage, may have Blocks, BlocksExtra, Data and DataExtra)
632        // Version 8 = not the same as network version 8 because it's not pallet, it's like 7 but everything is filled even when an entire section is empty
633        s.putByte("Y", (getY()));
634        int version = SAVE_STORAGE_VERSION;
635        ListTag<CompoundTag> storageList = new ListTag<>(STORAGE_TAG_NAME);
636        int blockStorages = Math.max(1, layerStorage.size());
637        for (int layer = 0; layer < blockStorages; layer++) {
638            BlockStorage storage = layerStorage.getStorageOrEmpty(layer);
639            CompoundTag storageTag;
640            if (layer == 0 && blockStorages == 1) {
641                storageTag = s;
642                if (!storage.hasBlockDataExtras() && !storage.hasBlockIdExtras()) {
643                    version = 0;
644                } else {
645                    version = 1;
646                }
647            } else {
648                storageTag = new CompoundTag();
649            }
650            if (version == 0 || storage.hasBlockIds()) {
651                byte[] idsBase = new byte[BlockStorage.SECTION_SIZE];
652                byte[] idsExtra = storage.hasBlockIdExtras() ? new byte[BlockStorage.SECTION_SIZE] : null;
653                NibbleArray dataBase = new NibbleArray(BlockStorage.SECTION_SIZE);
654                NibbleArray dataExtra = storage.hasBlockDataExtras() ? new NibbleArray(BlockStorage.SECTION_SIZE) : null;
655                List<byte[]> dataHuge = saveData(storage, idsBase, idsExtra, dataBase, dataExtra);
656                storageTag.putByteArray("Blocks", idsBase);
657                storageTag.putByteArray("Data", dataBase.getData());
658                if (idsExtra != null) {
659                    storageTag.putByteArray("BlocksExtra", idsExtra);
660                }
661                if (dataExtra != null) {
662                    storageTag.putByteArray("DataExtra", dataExtra.getData());
663                }
664                if (dataHuge != null) {
665                    ListTag<ByteArrayTag> hugeDataListTag = new ListTag<>(HUGE_TAG_NAME);
666                    for (byte[] hugeData : dataHuge) {
667                        hugeDataListTag.add(new ByteArrayTag("", hugeData));
668                    }
669                    storageTag.putList(hugeDataListTag);
670                }
671            }
672            if (version >= SAVE_STORAGE_VERSION) {
673                storageList.add(storageTag);
674            }
675        }
676        s.putByte("Version", version);
677        s.putByte("ContentVersion", getContentVersion());
678        if (version >= SAVE_STORAGE_VERSION) {
679            s.putList(storageList);
680        }
681        s.putByteArray("BlockLight", getLightArray());
682        s.putByteArray("SkyLight", getSkyLightArray());
683        return s;
684    }

693    public boolean compress() {
694        if (blockLight != null) {
695            byte[] arr1 = blockLight;
696            hasBlockLight = !Utils.isByteArrayEmpty(arr1);
697            byte[] arr2;
698            if (skyLight != null) {
699                arr2 = skyLight;
700                hasSkyLight = !Utils.isByteArrayEmpty(arr2);
701            } else if (hasSkyLight) {
702                arr2 = EmptyChunkSection.EMPTY_SKY_LIGHT_ARR;
703            } else {
704                arr2 = EmptyChunkSection.EMPTY_LIGHT_ARR;
705                hasSkyLight = false;
706            }
707            blockLight = null;
708            skyLight = null;
709            byte[] toDeflate = null;
710            if (hasBlockLight && hasSkyLight && arr2 != EmptyChunkSection.EMPTY_SKY_LIGHT_ARR) {
711                toDeflate = Binary.appendBytes(arr1, arr2);
712            } else if (hasBlockLight) {
713                toDeflate = arr1;
714            }
715            if (toDeflate != null) {
716                try {
717                    compressedLight = Zlib.deflate(toDeflate, 1);
718                } catch (Exception e) {
719                    log.error("Error compressing the light data", e);
720                }
721            }
722            return true;
723        }
724        return false;
725    }

728    @PowerNukkitOnly
729    @Override
730    public synchronized List<Block> scanBlocks(LevelProvider provider, int offsetX, int offsetZ, BlockVector3 min, BlockVector3 max, BiPredicate<BlockVector3, BlockState> condition) {
731        BlockStorage storage = layerStorage.getStorageOrNull(0);
732        if (storage == null) {
733            return Collections.emptyList();
734        }
735        List<Block> results = new ArrayList<>();
736        BlockVector3 current = new BlockVector3();
737        int offsetY = getY() << 4;
738        int minX = Math.max(0, min.x - offsetX);
739        int minY = Math.max(0, min.y - offsetY);
740        int minZ = Math.max(0, min.z - offsetZ);
741        for (int x = Math.min(max.x - offsetX, 15); x >= minX; x--) {
742            current.x = offsetX + x;
743            for (int z = Math.min(max.z - offsetZ, 15); z >= minZ; z--) {
744                current.z = offsetZ + z;
745                for (int y = Math.min(max.y - offsetY, 15); y >= minY; y--) {
746                    current.y = offsetY + y;
747                    BlockState state = storage.getBlockState(x, y, z);
748                    if (condition.test(current, state)) {
749                        results.add(state.getBlockRepairing(provider.getLevel(), current, 0));
750                    }
751                }
752            }
753        }
754        return results;
755    }
