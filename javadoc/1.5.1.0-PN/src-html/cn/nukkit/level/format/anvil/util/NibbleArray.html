001package cn.nukkit.level.format.anvil.util;
002
003
004import com.google.common.base.Preconditions;
005
006public class NibbleArray implements Cloneable {
007    private final byte[] data;
008
009    public NibbleArray(int length) {
010        data = new byte[length / 2];
011    }
012
013    public NibbleArray(byte[] array) {
014        data = array;
015    }
016
017    public byte get(int index) {
018        Preconditions.checkElementIndex(index, data.length * 2);
019        byte val = data[index / 2];
020        if ((index & 1) == 0) {
021            return (byte) (val & 0x0f);
022        } else {
023            return (byte) ((val & 0xf0) >>> 4);
024        }
025    }
026
027    public void set(int index, byte value) {
028        Preconditions.checkArgument(value >= 0 && value < 16, "Nibbles must have a value between 0 and 15.");
029        Preconditions.checkElementIndex(index, data.length * 2);
030        value &= 0xf;
031        int half = index / 2;
032        byte previous = data[half];
033        if ((index & 1) == 0) {
034            data[half] = (byte) (previous & 0xf0 | value);
035        } else {
036            data[half] = (byte) (previous & 0x0f | value << 4);
037        }
038    }
039
040    public void fill(byte value) {
041        Preconditions.checkArgument(value >= 0 && value < 16, "Nibbles must have a value between 0 and 15.");
042        value &= 0xf;
043        for (int i = 0; i < data.length; i++) {
044            data[i] = (byte) ((value << 4) | value);
045        }
046    }
047
048    public void copyFrom(byte[] bytes) {
049        Preconditions.checkNotNull(bytes, "bytes");
050        Preconditions.checkArgument(bytes.length == data.length, "length of provided byte array is %s but expected %s", bytes.length,
051                data.length);
052        System.arraycopy(bytes, 0, data, 0, data.length);
053    }
054
055    public void copyFrom(NibbleArray array) {
056        Preconditions.checkNotNull(array, "array");
057        copyFrom(array.data);
058    }
059
060    public byte[] getData() {
061        return data;
062    }
063
064    public NibbleArray copy() {
065        return new NibbleArray(getData().clone());
066    }
067}
