001package cn.nukkit.level.format.updater;
002
003import cn.nukkit.api.PowerNukkitOnly;
004import cn.nukkit.api.Since;
005import cn.nukkit.block.BlockID;
006import cn.nukkit.blockstate.BlockState;
007import cn.nukkit.level.format.ChunkSection;
008import cn.nukkit.math.BlockFace;
009
010@PowerNukkitOnly
011@Since("1.4.0.0-PN")
012public class BeehiveUpdater implements Updater {
013    private final ChunkSection section;
014    private boolean updateDirection;
015
016    @PowerNukkitOnly
017    @Since("1.4.0.0-PN")
018    public BeehiveUpdater(ChunkSection section, boolean updateDirection) {
019        this.section = section;
020        this.updateDirection = updateDirection;
021    }
022
023    @Override
024    public boolean update(int offsetX, int offsetY, int offsetZ, int x, int y, int z, BlockState state) {
025        int blockId = state.getBlockId();
026        if (blockId == BlockID.BEEHIVE || blockId == BlockID.BEE_NEST) {
027            @SuppressWarnings("deprecation") 
028            int meta = state.getLegacyDamage();
029            BlockFace face;
030            int honeyLevel;
031            if (updateDirection) {
032                face = BlockFace.fromIndex(meta & 0b111);
033                honeyLevel = (meta >> 3) & 0b111;
034            } else {
035                face = BlockFace.fromHorizontalIndex(meta & 0b11);
036                honeyLevel = (meta >> 2) & 0b111;
037            }
038            
039            if (honeyLevel > 5) {
040                honeyLevel = 5;
041            }
042            
043            int newMeta = (honeyLevel << 2) | face.getHorizontalIndex();
044            if (newMeta != meta) {
045                section.setBlockState(x, y, z, state.withData(newMeta));
046                return true;
047            }
048        }
049        return false;
050    }
051
052}
