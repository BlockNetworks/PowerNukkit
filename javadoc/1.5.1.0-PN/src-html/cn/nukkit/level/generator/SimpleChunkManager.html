001package cn.nukkit.level.generator;
002
003import cn.nukkit.api.DeprecationDetails;
004import cn.nukkit.api.PowerNukkitOnly;
005import cn.nukkit.api.Since;
006import cn.nukkit.blockstate.BlockState;
007import cn.nukkit.level.ChunkManager;
008import cn.nukkit.level.format.FullChunk;
009
010/**
011 * @author MagicDroidX (Nukkit Project)
012 */
013public abstract class SimpleChunkManager implements ChunkManager {
014
015    protected long seed;
016
017    public SimpleChunkManager(long seed) {
018        this.seed = seed;
019    }
020
021    @Override
022    public int getBlockIdAt(int x, int y, int z) {
023        return getBlockIdAt(x, y, z, 0);
024    }
025
026    @Override
027    public int getBlockIdAt(int x, int y, int z, int layer) {
028        FullChunk chunk = this.getChunk(x >> 4, z >> 4);
029        if (chunk != null) {
030            return chunk.getBlockId(x & 0xf, y & 0xff, z & 0xf, layer);
031        }
032        return 0;
033    }
034
035    @Override
036    public BlockState getBlockStateAt(int x, int y, int z, int layer) {
037        FullChunk chunk = this.getChunk(x >> 4, z >> 4);
038        if (chunk != null) {
039            return chunk.getBlockState(x & 0xf, y & 0xff, z & 0xf, layer);
040        }
041        return BlockState.AIR;
042    }
043
044    @Override
045    public void setBlockIdAt(int x, int y, int z, int id) {
046        setBlockIdAt(x, y, z, 0, id);
047    }
048
049    @Override
050    public void setBlockIdAt(int x, int y, int z, int layer, int id) {
051        FullChunk chunk = this.getChunk(x >> 4, z >> 4);
052        if (chunk != null) {
053            chunk.setBlockId(x & 0xf, y & 0xff, z & 0xf, layer, id);
054        }
055    }
056
057    @Deprecated
058    @DeprecationDetails(reason = "The meta is limited to 32 bits", since = "1.4.0.0-PN")
059    @Override
060    public void setBlockAt(int x, int y, int z, int id, int data) {
061        setBlockAtLayer(x, y, z, 0, id, data);
062    }
063
064    @Deprecated
065    @DeprecationDetails(reason = "The meta is limited to 32 bits", since = "1.4.0.0-PN")
066    @Override
067    public boolean setBlockAtLayer(int x, int y, int z, int layer, int id, int data) {
068        FullChunk chunk = this.getChunk(x >> 4, z >> 4);
069        if (chunk != null) {
070            return chunk.setBlock(x & 0xf, y & 0xff, z & 0xf, id, data);
071        }
072        return false;
073    }
074
075    @Deprecated
076    @DeprecationDetails(reason = "The meta is limited to 32 bits", since = "1.4.0.0-PN")
077    @Override
078    public void setBlockFullIdAt(int x, int y, int z, int fullId) {
079        setBlockFullIdAt(x, y, z, 0, fullId);
080    }
081
082    @Deprecated
083    @DeprecationDetails(reason = "The meta is limited to 32 bits", since = "1.4.0.0-PN")
084    @Override
085    public void setBlockFullIdAt(int x, int y, int z, int layer, int fullId) {
086        FullChunk chunk = this.getChunk(x >> 4, z >> 4);
087        if (chunk != null) {
088            chunk.setFullBlockId(x & 0xf, y & 0xff, z & 0xf, layer, fullId);
089        }
090    }
091
092    @PowerNukkitOnly
093    @Since("1.4.0.0-PN")
094    @Override
095    public boolean setBlockStateAt(int x, int y, int z, int layer, BlockState state) {
096        FullChunk chunk = this.getChunk(x >> 4, z >> 4);
097        if (chunk != null) {
098            return chunk.setBlockStateAtLayer(x & 0xf, y & 0xff, z & 0xf, layer, state);
099        }
100        return false;
101    }
102
103    @Deprecated
104    @DeprecationDetails(reason = "The meta is limited to 32 bits", since = "1.4.0.0-PN")
105    @Override
106    public int getBlockDataAt(int x, int y, int z) {
107        return getBlockDataAt(x, y, z, 0);
108    }
109
110    @Deprecated
111    @DeprecationDetails(reason = "The meta is limited to 32 bits", since = "1.4.0.0-PN")
112    @Override
113    public int getBlockDataAt(int x, int y, int z, int layer) {
114        FullChunk chunk = this.getChunk(x >> 4, z >> 4);
115        if (chunk != null) {
116            return chunk.getBlockData(x & 0xf, y & 0xff, z & 0xf, layer);
117        }
118        return 0;
119    }
120
121    @Deprecated
122    @DeprecationDetails(reason = "The meta is limited to 32 bits", since = "1.4.0.0-PN")
123    @Override
124    public void setBlockDataAt(int x, int y, int z, int data) {
125        setBlockDataAt(x, y, z, data, 0);
126    }
127
128    @Deprecated
129    @DeprecationDetails(reason = "The meta is limited to 32 bits", since = "1.4.0.0-PN")
130    @Override
131    public void setBlockDataAt(int x, int y, int z, int layer, int data) {
132        FullChunk chunk = this.getChunk(x >> 4, z >> 4);
133        if (chunk != null) {
134            chunk.setBlockData(x & 0xf, y & 0xff, z & 0xf, layer, data);
135        }
136    }
137
138    @Override
139    public void setChunk(int chunkX, int chunkZ) {
140        this.setChunk(chunkX, chunkZ, null);
141    }
142
143    @Override
144    public long getSeed() {
145        return seed;
146    }
147
148    public void setSeed(long seed) {
149        this.seed = seed;
150    }
151
152    public void cleanChunks(long seed) {
153        this.seed = seed;
154    }
155}
