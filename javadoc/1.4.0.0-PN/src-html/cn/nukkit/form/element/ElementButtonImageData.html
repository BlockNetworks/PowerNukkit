001package cn.nukkit.form.element;
002
003public class ElementButtonImageData {
004
005    public static final String IMAGE_DATA_TYPE_PATH = "path";
006    public static final String IMAGE_DATA_TYPE_URL = "url";
007
008    private String type;
009    private String data;
010
011    public ElementButtonImageData(String type, String data) {
012        if (!type.equals(IMAGE_DATA_TYPE_URL) && !type.equals(IMAGE_DATA_TYPE_PATH)) return;
013        this.type = type;
014        this.data = data;
015    }
016
017    public String getType() {
018        return type;
019    }
020
021    public String getData() {
022        return data;
023    }
024
025    public void setType(String type) {
026        this.type = type;
027    }
028
029    public void setData(String data) {
030        this.data = data;
031    }
032}
