024@PowerNukkitOnly
025@Since("1.4.0.0-PN")
026@ParametersAreNonnullByDefault
027public interface IMutableBlockState extends IBlockState {
028    /**
029     * Replace all matching states of this block state with the same states of the given block state.
030     * <p>States that doesn't exists in the other state are ignored.
031     * <p>Only properties that matches each other will be copied, for example, if this state have an age property
032     * going from 0 to 7 and the other have an age from 0 to 15, the age property won't change.
033     * @throws UnsupportedOperationException If the state is from a different block id and property copying isn't supported by the implementation
034     * @throws InvalidBlockStateException If the given storage has invalid data properties
035     * @param state The states that will have the properties copied.
036     */
037    @PowerNukkitOnly
038    @Since("1.4.0.0-PN")
039    default void setState(IBlockState state) throws InvalidBlockStateException {
040        if (state.getBlockId() == getBlockId()) {
041            setDataStorage(state.getDataStorage());
042        } else {
043            //TODO Implement property value copying
044            throw new UnsupportedOperationException();
045        }
046    }

079        return setDataStorageFromInt(storage, repair, null);
080    }
081
082    /**
083     * @return if the storage was repaired
084     * @throws InvalidBlockStateException If repair is false and the storage has an invalid property state
085     * @throws InvalidBlockStateDataTypeException If the storage has an unsupported number type
086     */
087    @PowerNukkitOnly
088    @Since("1.4.0.0-PN")
089    default boolean setDataStorage(@Nonnegative Number storage, boolean repair, @Nullable Consumer<BlockStateRepair> callback) {
090        try {
091            setDataStorage(storage);
092            return false;
093        } catch (InvalidBlockStateException e) {
094            if (repair) {
095                BigInteger bigInteger;
096                try {
097                    bigInteger = new BigDecimal(storage.toString()).toBigIntegerExact();
098                } catch (NumberFormatException | ArithmeticException e2) {
099                    InvalidBlockStateDataTypeException ex = new InvalidBlockStateDataTypeException(storage, e2);
100                    ex.addSuppressed(e);
101                    throw ex;
102                }
103                
104                try {
105                    setDataStorage(repairStorage(getBlockId(), bigInteger, getProperties(), callback));
106                } catch (InvalidBlockPropertyException | InvalidBlockStateException e2) {
107                    InvalidBlockStateException ex = new InvalidBlockStateException(e.getState(), "The state is invalid and could not be repaired", e);
108                    ex.addSuppressed(e2);
109                    throw ex;
110                }
111                return true;
112            }
113            throw e;
114        }
115    }
116
117    /**
118     * @return if the storage was repaired
119     */
120    @PowerNukkitOnly
121    @Since("1.4.0.0-PN")
122    default boolean setDataStorageFromInt(@Nonnegative final int storage, boolean repair, @Nullable Consumer<BlockStateRepair> callback) {
123        try {
124            setDataStorageFromInt(storage);
125            return false;
126        } catch (IllegalStateException | InvalidBlockPropertyException e) {
127            if (repair) {
128                setDataStorage(repairStorage(getBlockId(), BigInteger.valueOf(storage), getProperties(), callback));
129                return true;
130            }
131            throw e;
132        }
133    }
134
135    @PowerNukkitOnly
136    @Since("1.4.0.0-PN")
137    default void setDataStorageFromItemBlockMeta(int itemBlockMeta) {
138        BlockProperties allProperties = getProperties();
139        BlockProperties itemBlockProperties = allProperties.getItemBlockProperties();
140        if (allProperties.equals(itemBlockProperties)) {
141            setDataStorageFromInt(itemBlockMeta);
142            return;
143        }
144        
145        MutableBlockState item = itemBlockProperties.createMutableState(getBlockId());
146        item.setDataStorageFromInt(itemBlockMeta);
147
148        MutableBlockState converted = allProperties.createMutableState(getBlockId());
149        itemBlockProperties.getItemPropertyNames().forEach(property ->
150                converted.setPropertyValue(property, item.getPropertyValue(property)));
151        setDataStorage(converted.getDataStorage());
152    }

198    @PowerNukkitOnly
199    @Since("1.4.0.0-PN")
200    @SuppressWarnings("unchecked")
201    @Nonnull
202    static BigInteger repairStorage(
203            @Nonnegative int blockId, @Nonnull final BigInteger storage, @Nonnull final BlockProperties properties, 
204            @Nullable final Consumer<BlockStateRepair> callback) {
205        Validation.checkPositive("blockId", blockId);
206        
207        int checkedBits = 0;
208        int repairs = 0;
209        BigInteger current = storage;
210        for (BlockProperties.RegisteredBlockProperty reg : properties.getAllProperties()) {
211            checkedBits += reg.getProperty().getBitSize();
212            try {
213                reg.validateMeta(current);
214            } catch (InvalidBlockPropertyException e) {
215                BlockProperty<?> property = reg.getProperty();
216                int offset = reg.getOffset();
217                BigInteger next = property.setValue(current, offset, null);
218                if (callback != null) {
219                    Serializable fixed = property.getValue(next, offset);
220                    BlockStateRepair stateRepair = new BlockStateRepair(
221                            blockId, properties,
222                            storage, current, next, repairs++, property, offset,
223                            property.getMetaFromBigInt(current, offset),
224                            fixed, fixed, e
225                    );
226                    callback.accept(stateRepair);
227                    Serializable proposed = stateRepair.getProposedPropertyValue();
228                    if (!fixed.equals(proposed)) {
229                        try {
230                            next = ((BlockProperty<Serializable>) property).setValue(current, offset, proposed);
231                        } catch (InvalidBlockPropertyException proposedFailed) {
232                            logIMutableBlockState.warn("Could not apply the proposed repair, using the default proposal. "+stateRepair, proposedFailed);
233                        }
234                    }
235                }
236                current = next;
237            }
238        }
239
240        if (NukkitMath.bitLength(current) > checkedBits) {
241            BigInteger validMask = BigInteger.ONE.shiftLeft(checkedBits).subtract(BigInteger.ONE);
242            BigInteger next = current.and(validMask);
243            if (callback != null) {
244                BlockStateRepair stateRepair = new BlockStateRepair(
245                        blockId, properties,
246                        storage, current, next, repairs, null,
247                        checkedBits, current.shiftRight(checkedBits).intValue(), 0, 0,
248                        null);
249                callback.accept(stateRepair);
250                if (!Integer.valueOf(0).equals(stateRepair.getProposedPropertyValue())) {
251                    logIMutableBlockState.warn("Could not apply the proposed repair, using the default proposal. "+stateRepair, 
252                            new IllegalStateException("Attempted to propose a value outside the properties boundary"));
253                }
254            }
255            current = next;
256        }
257        
258        return current;
259    }

261    @PowerNukkitOnly
262    @Since("1.4.0.0-PN")
263    @Nonnull
264    static RuntimeException handleUnsupportedStorageType(@Nonnegative int blockId, @Nonnegative Number storage, RuntimeException e) {
265        InvalidBlockStateException ex;
266        try {
267            ex = new InvalidBlockStateException(BlockState.of(blockId, storage), e);
268        } catch (InvalidBlockStateDataTypeException e2) {
269            e2.addSuppressed(e);
270            return e2;
271        }
272        return ex;
273    }
