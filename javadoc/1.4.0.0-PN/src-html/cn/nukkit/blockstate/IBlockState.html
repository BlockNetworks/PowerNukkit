186    /**
187     * Gets a unique persistence identification for this state based on the block properties.
188     * <p>If the state holds an invalid meta, the result of {@link #getLegacyStateId()} is returned.</p>
189     */
190    @PowerNukkitOnly
191    @Since("1.4.0.0-PN")
192    @Nonnull
193    default String getStateId() {
194        BlockProperties properties = getProperties();
195        Map<String, String> propertyMap = new TreeMap<>(HumanStringComparator.getInstance());
196        try {
197            properties.getNames().forEach(name -> propertyMap.put(properties.getBlockProperty(name).getPersistenceName(), getPersistenceValue(name)));
198        } catch (InvalidBlockPropertyException e) {
199            logIBlockState.debug("Attempted to get the stateId of an invalid state {}:{}\nProperties: {}", getBlockId(), getDataStorage(), properties, e);
200            return getLegacyStateId();
201        }
202
203        StringBuilder stateId = new StringBuilder(getPersistenceName());
204        propertyMap.forEach((name, value) -> stateId.append(';').append(name).append('=').append(value));
205        return stateId.toString();
206    }

263     * @throws InvalidBlockStateException if repair is false and the state contains invalid property values
264     */
265    @PowerNukkitOnly
266    @Since("1.4.0.0-PN")
267    @Nonnull
268    default Block getBlock(@Nullable Level level, int x, int y, int z, int layer, boolean repair, @Nullable Consumer<BlockStateRepair> callback) {
269        Block block = Block.get(getBlockId());
270        block.level = level;
271        block.x = x;
272        block.y = y;
273        block.z = z;
274        block.layer = layer;
275        BlockState currentState = getCurrentState();
276        try {
277            if (currentState.isCachedValidationValid()) {
278                block.setState(currentState);
279                return block;
280            }
281        } catch (Exception e) {
282            logIBlockState.error("Unexpected error while trying to set the cached valid state to the block. State: {}, Block: {}", currentState, block, e);
283        }
284        
285        try {
286            block.setDataStorage(currentState.getDataStorage(), repair, callback);
287        } catch (InvalidBlockStateException e) {
288            throw new InvalidBlockStateException(getCurrentState(), "Invalid block state in layer "+layer+" at: "+new Position(x, y, z, level), e);
289        }
290        return block;
291    }

374    @PowerNukkitOnly
375    @Since("1.4.0.0-PN")
376    @Nonnull
377    default Block getBlockRepairing(@Nullable Level level, int x, int y, int z, int layer, @Nullable Consumer<BlockStateRepair> callback) {
378        List<BlockStateRepair> repairs = new ArrayList<>(0);
379        
380        Consumer<BlockStateRepair> callbackChain = repairs::add;
381
382        if (!BlockStateRepairEvent.getHandlers().isEmpty()) {
383            PluginManager manager = Server.getInstance().getPluginManager();
384            callbackChain = callbackChain.andThen(repair -> manager.callEvent(new BlockStateRepairEvent(repair)));
385        }
386        
387        if (callback != null) {
388            callbackChain = callbackChain.andThen(callback);
389        }
390        
391        Block block = getBlock(level, x, y, z, layer, true, callbackChain);
392        
393        if (!BlockStateRepairFinishEvent.getHandlers().isEmpty()) {
394            BlockStateRepairFinishEvent event = new BlockStateRepairFinishEvent(repairs, block);
395            Server.getInstance().getPluginManager().callEvent(event);
396            block = event.getResult();
397        }
398
399        if (!repairs.isEmpty() && logIBlockState.isDebugEnabled()) {
400            logIBlockState.debug("The block that at Level:{}, X:{}, Y:{}, Z:{}, L:{} was repaired. Result: {}, Repairs: {}",
401                    level, x, y, z, layer, block, repairs,
402                    new Exception("Stacktrace")
403            );
404        }
405        
406        return block;
407    }
