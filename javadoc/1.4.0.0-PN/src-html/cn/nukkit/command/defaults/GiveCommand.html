001package cn.nukkit.command.defaults;
002
003import cn.nukkit.Player;
004import cn.nukkit.block.BlockUnknown;
005import cn.nukkit.command.Command;
006import cn.nukkit.command.CommandSender;
007import cn.nukkit.command.data.CommandEnum;
008import cn.nukkit.command.data.CommandParamType;
009import cn.nukkit.command.data.CommandParameter;
010import cn.nukkit.item.Item;
011import cn.nukkit.item.ItemBlock;
012import cn.nukkit.lang.TranslationContainer;
013import cn.nukkit.utils.TextFormat;
014
015import java.util.ArrayList;
016import java.util.List;
017
018/**
019 * @author xtypr
020 * @since 2015/12/9
021 */
022public class GiveCommand extends VanillaCommand {
023    public GiveCommand(String name) {
024        super(name, "%nukkit.command.give.description", "%nukkit.command.give.usage");
025        this.setPermission("nukkit.command.give");
026        this.commandParameters.clear();
027        this.commandParameters.put("default", new CommandParameter[]{
028                CommandParameter.newType("player", CommandParamType.TARGET),
029                CommandParameter.newEnum("itemName", CommandEnum.ENUM_ITEM),
030                CommandParameter.newType("amount", true, CommandParamType.INT),
031                CommandParameter.newType("tags", true, CommandParamType.RAWTEXT)
032        });
033        this.commandParameters.put("toPlayerById", new CommandParameter[]{
034                CommandParameter.newType("player", CommandParamType.TARGET),
035                CommandParameter.newType("itemId", CommandParamType.INT),
036                CommandParameter.newType("amount", true, CommandParamType.INT),
037                CommandParameter.newType("tags", true, CommandParamType.RAWTEXT)
038        });
039        this.commandParameters.put("toPlayerByIdMeta", new CommandParameter[]{
040                CommandParameter.newType("player", CommandParamType.TARGET),
041                CommandParameter.newType("itemAndData", CommandParamType.STRING),
042                CommandParameter.newType("amount", true, CommandParamType.INT),
043                CommandParameter.newType("tags", true, CommandParamType.RAWTEXT)
044        });
045    }
046
047    @Override
048    public boolean execute(CommandSender sender, String commandLabel, String[] args) {
049        if (!this.testPermission(sender)) {
050            return true;
051        }
052
053        if (args.length < 2) {
054            sender.sendMessage(new TranslationContainer("commands.generic.usage", this.usageMessage));
055
056            return true;
057        }
058
059        Player player = sender instanceof Player && "@p".equals(args[0])? (Player) sender : sender.getServer().getPlayer(args[0]);
060        Item item;
061
062        try {
063            item = Item.fromString(args[1]);
064        } catch (Exception e) {
065            sender.sendMessage(new TranslationContainer("commands.generic.usage", this.usageMessage));
066            return true;
067        }
068        
069        if (item.getDamage() < 0) {
070            sender.sendMessage(new TranslationContainer("commands.generic.usage", this.usageMessage));
071            return true;
072        }
073        
074        if (item instanceof ItemBlock && item.getBlock() instanceof BlockUnknown) {
075            sender.sendMessage(new TranslationContainer("commands.give.block.notFound", args[1]));
076            return true;
077        }
078
079        int count;
080        try {
081            if (args.length <= 2) {
082                count = 1;
083            } else {
084                count = Integer.parseInt(args[2]);
085            }
086        } catch (NumberFormatException e) {
087            count = 1;
088        }
089        if (count <= 0) {
090            sender.sendMessage(new TranslationContainer("commands.generic.usage", this.usageMessage));
091            return true;
092        }
093        item.setCount(count);
094
095        if (player == null) {
096            sender.sendMessage(new TranslationContainer(TextFormat.RED + "%commands.generic.player.notFound"));
097            return true;
098        }
099        
100        if (item.isNull()) {
101            sender.sendMessage(new TranslationContainer(TextFormat.RED + "%commands.give.item.notFound", args[1]));
102            return true;
103        }
104        
105        Item[] returns = player.getInventory().addItem(item.clone());
106        List<Item> drops = new ArrayList<>();
107        for (Item returned: returns) {
108            int maxStackSize = returned.getMaxStackSize();
109            if (returned.getCount() <= maxStackSize) {
110                drops.add(returned);
111            } else {
112                while (returned.getCount() > maxStackSize) {
113                    Item drop = returned.clone();
114                    int toDrop = Math.min(returned.getCount(), maxStackSize);
115                    drop.setCount(toDrop);
116                    returned.setCount(returned.getCount() - toDrop);
117                    drops.add(drop);
118                }
119                if (!returned.isNull()) {
120                    drops.add(returned);
121                }
122            }
123        }
124        
125        for (Item drop: drops) {
126            player.dropItem(drop);
127        }
128        
129        Command.broadcastCommandMessage(sender, new TranslationContainer(
130                "%commands.give.success",
131                item.getName() + " (" + item.getId() + ":" + item.getDamage() + ")",
132                String.valueOf(item.getCount()),
133                player.getName()));
134        return true;
135    }
136}
