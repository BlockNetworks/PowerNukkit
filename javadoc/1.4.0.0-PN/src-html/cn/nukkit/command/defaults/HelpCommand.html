001package cn.nukkit.command.defaults;
002
003import cn.nukkit.command.Command;
004import cn.nukkit.command.CommandSender;
005import cn.nukkit.command.ConsoleCommandSender;
006import cn.nukkit.command.data.CommandParamType;
007import cn.nukkit.command.data.CommandParameter;
008import cn.nukkit.lang.TranslationContainer;
009import cn.nukkit.utils.TextFormat;
010
011import java.util.Map;
012import java.util.TreeMap;
013
014/**
015 * @author MagicDroidX (Nukkit Project)
016 */
017public class HelpCommand extends VanillaCommand {
018
019    public HelpCommand(String name) {
020        super(name, "%nukkit.command.help.description", "%commands.help.usage", new String[]{"?"});
021        this.setPermission("nukkit.command.help");
022        this.commandParameters.clear();
023        this.commandParameters.put("default", new CommandParameter[]{
024                CommandParameter.newType("page", true, CommandParamType.INT)
025        });
026    }
027
028    @Override
029    public boolean execute(CommandSender sender, String commandLabel, String[] args) {
030        if (!this.testPermission(sender)) {
031            return true;
032        }
033        StringBuilder command = new StringBuilder();
034        int pageNumber = 1;
035        int pageHeight = 5;
036        if (args.length != 0) {
037            try {
038                pageNumber = Integer.parseInt(args[args.length - 1]);
039                if (pageNumber <= 0) {
040                    pageNumber = 1;
041                }
042
043                String[] newargs = new String[args.length - 1];
044                System.arraycopy(args, 0, newargs, 0, newargs.length);
045                args = newargs;
046                /*if (args.length > 1) {
047                    args = Arrays.copyOfRange(args, 0, args.length - 2);
048                } else {
049                    args = new String[0];
050                }*/
051                for (String arg : args) {
052                    if (!command.toString().equals("")) {
053                        command.append(" ");
054                    }
055                    command.append(arg);
056                }
057            } catch (NumberFormatException e) {
058                pageNumber = 1;
059                for (String arg : args) {
060                    if (!command.toString().equals("")) {
061                        command.append(" ");
062                    }
063                    command.append(arg);
064                }
065            }
066        }
067
068        if (sender instanceof ConsoleCommandSender) {
069            pageHeight = Integer.MAX_VALUE;
070        }
071
072        if (command.toString().equals("")) {
073            Map<String, Command> commands = new TreeMap<>();
074            for (Command cmd : sender.getServer().getCommandMap().getCommands().values()) {
075                if (cmd.testPermissionSilent(sender)) {
076                    commands.put(cmd.getName(), cmd);
077                }
078            }
079            int totalPage = commands.size() % pageHeight == 0 ? commands.size() / pageHeight : commands.size() / pageHeight + 1;
080            pageNumber = Math.min(pageNumber, totalPage);
081            if (pageNumber < 1) {
082                pageNumber = 1;
083            }
084
085            sender.sendMessage(new TranslationContainer("commands.help.header", String.valueOf(pageNumber), String.valueOf(totalPage)));
086            int i = 1;
087            for (Command command1 : commands.values()) {
088                if (i >= (pageNumber - 1) * pageHeight + 1 && i <= Math.min(commands.size(), pageNumber * pageHeight)) {
089                    sender.sendMessage(TextFormat.DARK_GREEN + "/" + command1.getName() + ": " + TextFormat.WHITE + command1.getDescription());
090                }
091                i++;
092            }
093
094            return true;
095        } else {
096            Command cmd = sender.getServer().getCommandMap().getCommand(command.toString().toLowerCase());
097            if (cmd != null) {
098                if (cmd.testPermissionSilent(sender)) {
099                    String message = TextFormat.YELLOW + "--------- " + TextFormat.WHITE + " Help: /" + cmd.getName() + TextFormat.YELLOW + " ---------\n";
100                    message += TextFormat.GOLD + "Description: " + TextFormat.WHITE + cmd.getDescription() + "\n";
101                    StringBuilder usage = new StringBuilder();
102                    String[] usages = cmd.getUsage().split("\n");
103                    for (String u : usages) {
104                        if (!usage.toString().equals("")) {
105                            usage.append("\n" + TextFormat.WHITE);
106                        }
107                        usage.append(u);
108                    }
109                    message += TextFormat.GOLD + "Usage: " + TextFormat.WHITE + usage + "\n";
110                    sender.sendMessage(message);
111                    return true;
112                }
113            }
114
115            sender.sendMessage(TextFormat.RED + "No help for " + command.toString().toLowerCase());
116            return true;
117        }
118    }
119}
