001package cn.nukkit.command.data;
002
003import cn.nukkit.api.Since;
004import cn.nukkit.block.BlockID;
005import cn.nukkit.item.MinecraftItemID;
006import com.google.common.collect.ImmutableList;
007
008import java.lang.reflect.Field;
009import java.util.Arrays;
010import java.util.List;
011import java.util.function.Function;
012import java.util.stream.Stream;
013
014/**
015 * @author CreeperFace
016 */
017public class CommandEnum {
018    
019    @Since("1.4.0.0-PN")
020    public static final CommandEnum ENUM_BOOLEAN = new CommandEnum("Boolean", ImmutableList.of("true", "false"));
021
022    @Since("1.4.0.0-PN")
023    public static final CommandEnum ENUM_GAMEMODE = new CommandEnum("GameMode",
024            ImmutableList.of("survival", "creative", "s", "c", "adventure", "a", "spectator", "view", "v", "spc"));
025
026    @Since("1.4.0.0-PN")
027    public static final CommandEnum ENUM_BLOCK;
028
029    @Since("1.4.0.0-PN")
030    public static final CommandEnum ENUM_ITEM;
031
032    static {
033        ImmutableList.Builder<String> blocks = ImmutableList.builder();
034        for (Field field : BlockID.class.getDeclaredFields()) {
035            blocks.add(field.getName().toLowerCase());
036        }
037        ENUM_BLOCK = new CommandEnum("Block", blocks.build());
038
039        ENUM_ITEM = new CommandEnum("Item", ImmutableList.copyOf(Arrays.stream(MinecraftItemID.values())
040            .filter(it -> !it.isTechnical())
041            .filter(it -> !it.isEducationEdition())
042            .flatMap(it -> Stream.of(Stream.of(it.getNamespacedId())/*, Arrays.stream(it.getAliases())*/).flatMap(Function.identity()))
043            .map(it-> it.substring(10).toLowerCase())
044            .toArray(String[]::new)
045        ));
046    }
047
048    private String name;
049    private List<String> values;
050
051    @Since("1.4.0.0-PN")
052    public CommandEnum(String name, String... values) {
053        this(name, Arrays.asList(values));
054    }
055
056    public CommandEnum(String name, List<String> values) {
057        this.name = name;
058        this.values = values;
059    }
060
061    public String getName() {
062        return name;
063    }
064
065    public List<String> getValues() {
066        return values;
067    }
068
069    public int hashCode() {
070        return name.hashCode();
071    }
072}
