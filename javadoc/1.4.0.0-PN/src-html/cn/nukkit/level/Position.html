001package cn.nukkit.level;
002
003import cn.nukkit.api.PowerNukkitDifference;
004import cn.nukkit.api.PowerNukkitOnly;
005import cn.nukkit.api.Since;
006import cn.nukkit.block.Block;
007import cn.nukkit.blockentity.BlockEntity;
008import cn.nukkit.blockstate.BlockState;
009import cn.nukkit.level.format.FullChunk;
010import cn.nukkit.math.BlockFace;
011import cn.nukkit.math.Vector3;
012import cn.nukkit.positiontracking.NamedPosition;
013import cn.nukkit.utils.LevelException;
014
015import javax.annotation.Nonnull;
016import javax.annotation.Nullable;
017
018/**
019 * @author MagicDroidX (Nukkit Project)
020 */
021@PowerNukkitDifference(since = "1.4.0.0-PN", info = "Overrides NamedPosition instead of Vector3")
022public class Position extends NamedPosition {
023    public Level level;
024
025    public Position() {
026        this(0, 0, 0, null);
027    }
028
029    public Position(double x) {
030        this(x, 0, 0, null);
031    }
032
033    public Position(double x, double y) {
034        this(x, y, 0, null);
035    }
036
037    public Position(double x, double y, double z) {
038        this(x, y, z, null);
039    }
040
041    public Position(double x, double y, double z, Level level) {
042        this.x = x;
043        this.y = y;
044        this.z = z;
045        this.level = level;
046    }
047
048    public static Position fromObject(Vector3 pos) {
049        return fromObject(pos, null);
050    }
051
052    public static Position fromObject(Vector3 pos, Level level) {
053        return new Position(pos.x, pos.y, pos.z, level);
054    }
055
056    public Level getLevel() {
057        return this.level;
058    }
059
060    public Position setLevel(Level level) {
061        this.level = level;
062        return this;
063    }
064
065    public boolean isValid() {
066        return this.level != null;
067    }
068
069    public boolean setStrong() {
070        return false;
071    }
072
073    public boolean setWeak() {
074        return false;
075    }
076
077    public Position getSide(BlockFace face) {
078        return this.getSide(face, 1);
079    }
080
081    public Position getSide(BlockFace face, int step) {
082        return Position.fromObject(super.getSide(face, step), getValidLevel());
083    }
084
085    @Override
086    public String toString() {
087        return "Position(level=" + (this.isValid() ? this.getLevel().getName() : "null") + ",x=" + this.x + ",y=" + this.y + ",z=" + this.z + ")";
088    }
089
090    @Override
091    public Position setComponents(double x, double y, double z) {
092        this.x = x;
093        this.y = y;
094        this.z = z;
095        return this;
096    }
097
098    @Since("1.4.0.0-PN")
099    @PowerNukkitOnly
100    @Override
101    public Position setComponents(Vector3 pos) {
102        super.setComponents(pos);
103        return this;
104    }
105
106    @PowerNukkitOnly
107    @Since("1.4.0.0-PN")
108    @Nullable
109    public BlockEntity getLevelBlockEntity() {
110        return getValidLevel().getBlockEntity(this);
111    }
112
113    @PowerNukkitOnly
114    @Since("1.4.0.0-PN")
115    @Nullable
116    public final <T extends BlockEntity> T getTypedBlockEntity(@Nonnull Class<T> type) {
117        BlockEntity blockEntity = getValidLevel().getBlockEntity(this);
118        return type.isInstance(blockEntity) ? type.cast(blockEntity) : null;
119    }
120
121    @PowerNukkitOnly
122    @Since("1.4.0.0-PN")
123    @Nonnull
124    public BlockState getLevelBlockState() {
125        return getLevelBlockState(0);
126    }
127
128    @PowerNukkitOnly
129    @Since("1.4.0.0-PN")
130    @Nonnull
131    public BlockState getLevelBlockState(int layer) {
132        return getValidLevel().getBlockStateAt(getFloorX(), getFloorY(), getFloorZ(), layer);
133    }
134
135    public Block getLevelBlock() {
136        return getValidLevel().getBlock(this);
137    }
138
139    public Block getLevelBlockAtLayer(int layer) {
140        return getValidLevel().getBlock(this, layer);
141    }
142
143    @Nonnull
144    public Location getLocation() {
145        return new Location(this.x, this.y, this.z, 0, 0, getValidLevel());
146    }
147
148
149    @Since("1.4.0.0-PN")
150    @PowerNukkitOnly
151    @Nonnull
152    @Override
153    public String getLevelName() {
154        return getValidLevel().getName();
155    }
156
157    @Since("1.4.0.0-PN")
158    @PowerNukkitOnly
159    @Nonnull
160    public final Level getValidLevel() {
161        Level level = this.level;
162        if (level == null) {
163            throw new LevelException("Undefined Level reference");
164        }
165        return level;
166    }
167
168    @Override
169    public Position add(double x) {
170        return this.add(x, 0, 0);
171    }
172
173    @Override
174    public Position add(double x, double y) {
175        return this.add(x, y, 0);
176    }
177
178    @Override
179    public Position add(double x, double y, double z) {
180        return new Position(this.x + x, this.y + y, this.z + z, this.level);
181    }
182
183    @Override
184    public Position add(Vector3 x) {
185        return new Position(this.x + x.getX(), this.y + x.getY(), this.z + x.getZ(), this.level);
186    }
187
188    @Override
189    public Position subtract() {
190        return this.subtract(0, 0, 0);
191    }
192
193    @Override
194    public Position subtract(double x) {
195        return this.subtract(x, 0, 0);
196    }
197
198    @Override
199    public Position subtract(double x, double y) {
200        return this.subtract(x, y, 0);
201    }
202
203    @Override
204    public Position subtract(double x, double y, double z) {
205        return this.add(-x, -y, -z);
206    }
207
208    @Override
209    public Position subtract(Vector3 x) {
210        return this.add(-x.getX(), -x.getY(), -x.getZ());
211    }
212
213    @Override
214    public Position multiply(double number) {
215        return new Position(this.x * number, this.y * number, this.z * number, this.level);
216    }
217
218    @Override
219    public Position divide(double number) {
220        return new Position(this.x / number, this.y / number, this.z / number, this.level);
221    }
222
223    @Override
224    public Position ceil() {
225        return new Position((int) Math.ceil(this.x), (int) Math.ceil(this.y), (int) Math.ceil(this.z), this.level);
226    }
227
228    @Override
229    public Position floor() {
230        return new Position(this.getFloorX(), this.getFloorY(), this.getFloorZ(), this.level);
231    }
232
233    @Override
234    public Position round() {
235        return new Position(Math.round(this.x), Math.round(this.y), Math.round(this.z), this.level);
236    }
237
238    @Override
239    public Position abs() {
240        return new Position((int) Math.abs(this.x), (int) Math.abs(this.y), (int) Math.abs(this.z), this.level);
241    }
242
243    @Override
244    public Position clone() {
245        return (Position) super.clone();
246    }
247
248    @Nullable
249    public FullChunk getChunk() {
250        return isValid() ? level.getChunk(getChunkX(), getChunkZ()) : null;
251    }
252}
