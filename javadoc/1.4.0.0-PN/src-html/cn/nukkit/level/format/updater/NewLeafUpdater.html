001package cn.nukkit.level.format.updater;
002
003import cn.nukkit.api.PowerNukkitOnly;
004import cn.nukkit.api.Since;
005import cn.nukkit.block.BlockID;
006import cn.nukkit.blockstate.BlockState;
007import cn.nukkit.level.format.ChunkSection;
008
009@PowerNukkitOnly
010@Since("1.4.0.0-PN")
011public class NewLeafUpdater implements Updater {
012    private final ChunkSection section;
013    private boolean forceOldSystem;
014
015    @PowerNukkitOnly
016    @Since("1.4.0.0-PN")
017    public NewLeafUpdater(ChunkSection section) {
018        this.section = section;
019    }
020
021    @Override
022    public boolean update(int offsetX, int offsetY, int offsetZ, int x, int y, int z, BlockState state) {
023        if (state.getBlockId() == BlockID.LEAVES2) {
024            @SuppressWarnings("deprecation") 
025            int legacyDamage = state.getLegacyDamage();
026            if ((legacyDamage & 0xE) == 0x0) {
027                // No flags, no conversion is needed
028                return false;
029            }
030            
031            boolean newSystemForSure = (legacyDamage & 0x8) == 0x8; // New check decay, invalid on old system
032            boolean oldSystemForSure = (legacyDamage & 0x2) == 0x2; // Old check decay, invalid on new system
033            if (newSystemForSure && oldSystemForSure) {
034                // Oh god! This shouldn't happen!
035                // But it's happening somehow: https://github.com/PowerNukkit/PowerNukkit/issues/482
036                // Keeping the type as old system, letting it check decay and making it non-persistent
037                int newData = legacyDamage & 0b0001; // Wood type
038                newData |= 0b1000; // Check-decay
039                BlockState fixed = state.withData(newData);
040                section.setBlockState(x, y, z, fixed);
041                return true;
042            }
043            if (newSystemForSure) {
044                // Ok, using the right flag positions as indicated in the wiki
045                // Nothing needs to be done
046                return false;
047            }
048            if (forceOldSystem || oldSystemForSure) {
049                // Using the old incorrect persistent flags, let's fix it and force a check decay
050                int newData = legacyDamage & 0b0001; // Wood type
051                boolean persistent = (legacyDamage & 0x04) == 0x04;
052                if (persistent) {
053                    newData |= 0b0100; // Persistent
054                } 
055                
056                if (oldSystemForSure || !persistent) {
057                    newData |= 0b1000; // Check Decay
058                }
059                
060                BlockState fixed = state.withData(newData);
061                if (newData == legacyDamage) {
062                    return false;
063                }
064                section.setBlockState(x, y, z, fixed);
065                return true;
066            }
067        }
068        return false;
069    }
070
071    @PowerNukkitOnly
072    @Since("1.4.0.0-PN")
073    public boolean isForceOldSystem() {
074        return this.forceOldSystem;
075    }
076
077    @PowerNukkitOnly
078    @Since("1.4.0.0-PN")
079    public void setForceOldSystem(boolean forceOldSystem) {
080        this.forceOldSystem = forceOldSystem;
081    }
082}
