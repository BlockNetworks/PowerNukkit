001package cn.nukkit.level.format.updater;
002
003import cn.nukkit.api.PowerNukkitOnly;
004import cn.nukkit.api.Since;
005import cn.nukkit.block.BlockDoor;
006import cn.nukkit.blockproperty.CommonBlockProperties;
007import cn.nukkit.blockstate.BlockState;
008import cn.nukkit.blockstate.BlockStateRegistry;
009import cn.nukkit.blockstate.MutableBlockState;
010import cn.nukkit.level.format.Chunk;
011import cn.nukkit.level.format.ChunkSection;
012
013import static cn.nukkit.block.BlockID.*;
014
015@PowerNukkitOnly
016@Since("1.4.0.0-PN")
017public class DoorUpdater implements Updater {
018    private static final int DOOR_OPEN_BIT = 0x04;
019    private static final int DOOR_TOP_BIT = 0x08;
020    private static final int DOOR_HINGE_BIT = 0x01;
021    
022    private final Chunk chunk;
023    private final ChunkSection section;
024
025    @PowerNukkitOnly
026    @Since("1.4.0.0-PN")
027    public DoorUpdater(Chunk chunk, ChunkSection section) {
028        this.chunk = chunk;
029        this.section = section;
030    }
031
032    @Override
033    public boolean update(int offsetX, int offsetY, int offsetZ, int x, int y, int z, BlockState state) {
034        switch (state.getBlockId()) {
035            case WOODEN_DOOR_BLOCK:
036            case DARK_OAK_DOOR_BLOCK:
037            case ACACIA_DOOR_BLOCK:
038            case BIRCH_DOOR_BLOCK:
039            case JUNGLE_DOOR_BLOCK:
040            case SPRUCE_DOOR_BLOCK:
041            case IRON_DOOR_BLOCK:
042                break;
043            default:
044                return false;
045        }
046        
047        @SuppressWarnings("deprecation") 
048        int legacy = state.getLegacyDamage();
049        MutableBlockState mutableState = BlockStateRegistry.createMutableState(state.getBlockId());
050        if ((legacy & DOOR_TOP_BIT) > 0) {
051            mutableState.setBooleanValue(CommonBlockProperties.UPPER_BLOCK, true);
052            mutableState.setBooleanValue(BlockDoor.DOOR_HINGE, (legacy & DOOR_HINGE_BIT) > 0);
053            
054            int underY = offsetY + y - 1;
055            if (underY >= 0) {
056                BlockState underState = chunk.getBlockState(x, underY, z);
057                if (underState.getBlockId() == state.getBlockId()) {
058                    mutableState.setPropertyValue(BlockDoor.DOOR_DIRECTION, underState.getPropertyValue(BlockDoor.DOOR_DIRECTION));
059                    mutableState.setBooleanValue(CommonBlockProperties.OPEN, underState.getPropertyValue(CommonBlockProperties.OPEN));
060                }
061            }
062        } else {
063            mutableState.setBooleanValue(CommonBlockProperties.UPPER_BLOCK, false);
064            mutableState.setPropertyValue(BlockDoor.DOOR_DIRECTION, BlockDoor.DOOR_DIRECTION.getValueForMeta(legacy & 0x3));
065            mutableState.setBooleanValue(CommonBlockProperties.OPEN, (legacy & DOOR_OPEN_BIT) > 0);
066        }
067        
068        return section.setBlockState(x, y, z, mutableState.getCurrentState());
069    }
070}
