001package cn.nukkit.level;
002
003import cn.nukkit.api.DeprecationDetails;
004import cn.nukkit.api.PowerNukkitDifference;
005import cn.nukkit.api.PowerNukkitOnly;
006import cn.nukkit.api.Since;
007import cn.nukkit.entity.Entity;
008import cn.nukkit.math.BlockFace;
009import cn.nukkit.math.Vector3;
010
011import javax.annotation.Nullable;
012
013/**
014 * @author MagicDroidX (Nukkit Project)
015 */
016public class MovingObjectPosition {
017
018    /**
019     * 0 = block, 1 = entity
020     */
021    public int typeOfHit;
022
023    public int blockX;
024    public int blockY;
025    public int blockZ;
026
027    /**
028     * Which side was hit. If its -1 then it went the full length of the ray trace.
029     * Bottom = 0, Top = 1, East = 2, West = 3, North = 4, South = 5.
030     */
031    @Deprecated
032    @DeprecationDetails(since = "1.4.0.0-PN", reason = "Magic numbers and not encapsulated", replaceWith = "getFaceHit(), setFaceHit(BlockFace)")
033    public int sideHit;
034
035    public Vector3 hitVector;
036
037    public Entity entityHit;
038    
039    @PowerNukkitOnly
040    @Since("1.4.0.0-PN")
041    @Nullable
042    @SuppressWarnings("java:S1874")
043    public BlockFace getFaceHit() {
044        switch (sideHit) {
045            case 0:
046                return BlockFace.DOWN;
047            case 1:
048                return BlockFace.UP;
049            case 2:
050                return BlockFace.EAST;
051            case 3:
052                return BlockFace.WEST;
053            case 4:
054                return BlockFace.NORTH;
055            case 5:
056                return BlockFace.SOUTH;
057            default:
058                return null;
059        }
060    }
061
062    @PowerNukkitOnly
063    @Since("1.4.0.0-PN")
064    @SuppressWarnings("java:S1874")
065    public void setFaceHit(@Nullable BlockFace face) {
066        if (face == null) {
067            sideHit = -1;
068            return;
069        }
070        
071        switch (face) {
072            case DOWN:
073                sideHit = 0;
074                break;
075            case UP:
076                sideHit = 1;
077                break;
078            case NORTH:
079                sideHit = 4;
080                break;
081            case SOUTH:
082                sideHit = 5;
083                break;
084            case WEST:
085                sideHit = 3;
086                break;
087            case EAST:
088                sideHit = 2;
089                break;
090            default:
091                sideHit = -1;
092        }
093    }
094
095    @PowerNukkitOnly
096    @Since("1.4.0.0-PN")
097    public static MovingObjectPosition fromBlock(int x, int y, int z, BlockFace face, Vector3 hitVector) {
098        MovingObjectPosition objectPosition = new MovingObjectPosition();
099        objectPosition.typeOfHit = 0;
100        objectPosition.blockX = x;
101        objectPosition.blockY = y;
102        objectPosition.blockZ = z;
103        objectPosition.hitVector = new Vector3(hitVector.x, hitVector.y, hitVector.z);
104        objectPosition.setFaceHit(face);
105        return objectPosition;
106    }
107
108    @PowerNukkitDifference(since = "1.4.0.0-PN", info = "Fixed: sideHit not being filled")
109    @Deprecated
110    @DeprecationDetails(since = "1.4.0.0-PN", reason = "Magic number in side param", replaceWith = "fromBlock(int,int,int,BlockFace,Vector3)")
111    public static MovingObjectPosition fromBlock(int x, int y, int z, int side, Vector3 hitVector) {
112        MovingObjectPosition objectPosition = new MovingObjectPosition();
113        objectPosition.typeOfHit = 0;
114        objectPosition.blockX = x;
115        objectPosition.blockY = y;
116        objectPosition.blockZ = z;
117        objectPosition.sideHit = side;
118        objectPosition.hitVector = new Vector3(hitVector.x, hitVector.y, hitVector.z);
119        return objectPosition;
120    }
121
122    public static MovingObjectPosition fromEntity(Entity entity) {
123        MovingObjectPosition objectPosition = new MovingObjectPosition();
124        objectPosition.typeOfHit = 1;
125        objectPosition.entityHit = entity;
126        objectPosition.hitVector = new Vector3(entity.x, entity.y, entity.z);
127        return objectPosition;
128    }
129
130    @Override
131    public String toString() {
132        return "MovingObjectPosition{" +
133                "typeOfHit=" + typeOfHit +
134                ", blockX=" + blockX +
135                ", blockY=" + blockY +
136                ", blockZ=" + blockZ +
137                ", sideHit=" + sideHit +
138                ", hitVector=" + hitVector +
139                ", entityHit=" + entityHit +
140                '}';
141    }
142}
