001package cn.nukkit.level.particle;
002
003import cn.nukkit.entity.Entity;
004import cn.nukkit.entity.data.EntityMetadata;
005import cn.nukkit.entity.data.Skin;
006import cn.nukkit.item.Item;
007import cn.nukkit.level.Level;
008import cn.nukkit.level.Location;
009import cn.nukkit.math.Vector3;
010import cn.nukkit.network.protocol.*;
011import cn.nukkit.utils.SerializedImage;
012import com.google.common.base.Strings;
013
014import java.util.ArrayList;
015import java.util.UUID;
016import java.util.concurrent.ThreadLocalRandom;
017
018/**
019 * @author xtypr
020 * @since 2015/11/21
021 */
022public class FloatingTextParticle extends Particle {
023    private static final Skin EMPTY_SKIN = new Skin();
024    private static final SerializedImage SKIN_DATA = SerializedImage.fromLegacy(new byte[8192]);
025
026    static {
027        EMPTY_SKIN.setSkinData(SKIN_DATA);
028        EMPTY_SKIN.generateSkinId("FloatingText");
029    }
030
031    protected UUID uuid = UUID.randomUUID();
032    protected final Level level;
033    protected long entityId = -1;
034    protected boolean invisible = false;
035    protected EntityMetadata metadata = new EntityMetadata();
036
037    public FloatingTextParticle(Location location, String title) {
038        this(location, title, null);
039    }
040
041    public FloatingTextParticle(Location location, String title, String text) {
042        this(location.getLevel(), location, title, text);
043    }
044
045    public FloatingTextParticle(Vector3 pos, String title) {
046        this(pos, title, null);
047    }
048
049    public FloatingTextParticle(Vector3 pos, String title, String text) {
050        this(null, pos, title, text);
051    }
052
053    private FloatingTextParticle(Level level, Vector3 pos, String title, String text) {
054        super(pos.x, pos.y, pos.z);
055        this.level = level;
056
057        long flags = (
058                1L << Entity.DATA_FLAG_NO_AI
059        );
060        metadata.putLong(Entity.DATA_FLAGS, flags)
061                .putLong(Entity.DATA_LEAD_HOLDER_EID,-1)
062                .putFloat(Entity.DATA_SCALE, 0.01f) //zero causes problems on debug builds?
063                .putFloat(Entity.DATA_BOUNDING_BOX_HEIGHT, 0.01f)
064                .putFloat(Entity.DATA_BOUNDING_BOX_WIDTH, 0.01f);
065        if (!Strings.isNullOrEmpty(title)) {
066            metadata.putString(Entity.DATA_NAMETAG, title);
067        }
068        if (!Strings.isNullOrEmpty(text)) {
069            metadata.putString(Entity.DATA_SCORE_TAG, text);
070        }
071    }
072
073    public String getText() {
074        return metadata.getString(Entity.DATA_SCORE_TAG);
075    }
076
077    public void setText(String text) {
078        this.metadata.putString(Entity.DATA_SCORE_TAG, text);
079        sendMetadata();
080    }
081
082    public String getTitle() {
083        return metadata.getString(Entity.DATA_NAMETAG);
084    }
085
086    public void setTitle(String title) {
087        this.metadata.putString(Entity.DATA_NAMETAG, title);
088        sendMetadata();
089    }
090
091    private void sendMetadata() {
092        if (level != null) {
093            SetEntityDataPacket packet = new SetEntityDataPacket();
094            packet.eid = entityId;
095            packet.metadata = metadata;
096            level.addChunkPacket(getChunkX(), getChunkZ(), packet);
097        }
098    }
099
100    public boolean isInvisible() {
101        return invisible;
102    }
103
104    public void setInvisible(boolean invisible) {
105        this.invisible = invisible;
106    }
107
108    public void setInvisible() {
109        this.setInvisible(true);
110    }
111    
112    public long getEntityId() {
113        return entityId;   
114    }
115
116    @Override
117    public DataPacket[] encode() {
118        ArrayList<DataPacket> packets = new ArrayList<>();
119
120        if (this.entityId == -1) {
121            this.entityId = 1095216660480L + ThreadLocalRandom.current().nextLong(0, 0x7fffffffL);
122        } else {
123            RemoveEntityPacket pk = new RemoveEntityPacket();
124            pk.eid = this.entityId;
125
126            packets.add(pk);
127        }
128
129        if (!this.invisible) {
130            PlayerListPacket.Entry[] entry = {new PlayerListPacket.Entry(uuid, entityId,
131                    metadata.getString(Entity.DATA_NAMETAG), EMPTY_SKIN)};
132            PlayerListPacket playerAdd = new PlayerListPacket();
133            playerAdd.entries = entry;
134            playerAdd.type = PlayerListPacket.TYPE_ADD;
135            packets.add(playerAdd);
136
137            AddPlayerPacket pk = new AddPlayerPacket();
138            pk.uuid = uuid;
139            pk.username = "";
140            pk.entityUniqueId = this.entityId;
141            pk.entityRuntimeId = this.entityId;
142            pk.x = (float) this.x;
143            pk.y = (float) (this.y - 0.75);
144            pk.z = (float) this.z;
145            pk.speedX = 0;
146            pk.speedY = 0;
147            pk.speedZ = 0;
148            pk.yaw = 0;
149            pk.pitch = 0;
150            pk.metadata = this.metadata;
151            pk.item = Item.get(Item.AIR);
152            packets.add(pk);
153
154            PlayerListPacket playerRemove = new PlayerListPacket();
155            playerRemove.entries = entry;
156            playerRemove.type = PlayerListPacket.TYPE_REMOVE;
157            packets.add(playerRemove);
158        }
159
160        return packets.toArray(DataPacket.EMPTY_ARRAY);
161    }
162}
