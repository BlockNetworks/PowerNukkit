001package cn.nukkit.level.generator.task;
002
003import cn.nukkit.Server;
004import cn.nukkit.level.Level;
005import cn.nukkit.level.format.generic.BaseFullChunk;
006import cn.nukkit.level.generator.Generator;
007import cn.nukkit.level.generator.SimpleChunkManager;
008import cn.nukkit.scheduler.AsyncTask;
009
010/**
011 * @author MagicDroidX (Nukkit Project)
012 */
013public class GenerationTask extends AsyncTask {
014    private final Level level;
015    public boolean state;
016    private BaseFullChunk chunk;
017
018
019    public GenerationTask(Level level, BaseFullChunk chunk) {
020        this.state = true;
021        this.chunk = chunk;
022        this.level = level;
023    }
024
025    @Override
026    public void onRun() {
027        Generator generator = level.getGenerator();
028        this.state = false;
029        if (generator == null) {
030            return;
031        }
032
033        SimpleChunkManager manager = (SimpleChunkManager) generator.getChunkManager();
034
035        if (manager == null) {
036            this.state = false;
037            return;
038        }
039
040        manager.cleanChunks(level.getSeed());
041        synchronized (manager) {
042            try {
043                BaseFullChunk chunk = this.chunk;
044
045                if (chunk == null) {
046                    return;
047                }
048
049                synchronized (chunk) {
050                    if (!chunk.isGenerated()) {
051                        manager.setChunk(chunk.getX(), chunk.getZ(), chunk);
052                        generator.generateChunk(chunk.getX(), chunk.getZ());
053                        chunk = manager.getChunk(chunk.getX(), chunk.getZ());
054                        chunk.setGenerated();
055                    }
056                }
057                this.chunk = chunk;
058                state = true;
059            } finally {
060                manager.cleanChunks(level.getSeed());
061            }
062        }
063
064    }
065
066    @Override
067    public void onCompletion(Server server) {
068        if (level != null) {
069            if (!this.state) {
070                return;
071            }
072
073            BaseFullChunk chunk = this.chunk;
074
075            if (chunk == null) {
076                return;
077            }
078
079            level.generateChunkCallback(chunk.getX(), chunk.getZ(), chunk);
080        }
081    }
082}
