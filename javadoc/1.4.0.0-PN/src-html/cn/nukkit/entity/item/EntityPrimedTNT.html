016/**
017 * @author MagicDroidX
018 */
019public class EntityPrimedTNT extends Entity implements EntityExplosive {
020
021    public static final int NETWORK_ID = 65;
022
023    @Override
024    public float getWidth() {
025        return 0.98f;
026    }
027
028    @Override
029    public float getLength() {
030        return 0.98f;
031    }
032
033    @Override
034    public float getHeight() {
035        return 0.98f;
036    }
037
038    @Override
039    protected float getGravity() {
040        return 0.04f;
041    }
042
043    @Override
044    protected float getDrag() {
045        return 0.02f;
046    }
047
048    @Override
049    protected float getBaseOffset() {
050        return 0.49f;
051    }
052
053    @Override
054    public boolean canCollide() {
055        return false;
056    }
057
058    protected int fuse;
059
060    protected Entity source;
061
062    public EntityPrimedTNT(FullChunk chunk, CompoundTag nbt) {
063        this(chunk, nbt, null);
064    }
065
066    public EntityPrimedTNT(FullChunk chunk, CompoundTag nbt, Entity source) {
067        super(chunk, nbt);
068        this.source = source;
069    }
070
071    @Override
072    public int getNetworkId() {
073        return NETWORK_ID;
074    }
075
076    @Override
077    public boolean attack(EntityDamageEvent source) {
078        return source.getCause() == DamageCause.VOID && super.attack(source);
079    }
080
081    @PowerNukkitDifference(info = "Using new method to play sounds", since = "1.4.0.0-PN")
082    protected void initEntity() {
083        super.initEntity();
084
085        if (namedTag.contains("Fuse")) {
086            fuse = namedTag.getByte("Fuse");
087        } else {
088            fuse = 80;
089        }
090
091        this.setDataFlag(DATA_FLAGS, DATA_FLAG_IGNITED, true);
092        this.setDataProperty(new IntEntityData(DATA_FUSE_LENGTH, fuse));
093
094        this.getLevel().addSound(this, Sound.RANDOM_FUSE);
095    }
096
097
098    public boolean canCollideWith(Entity entity) {
099        return false;
100    }
101
102    public void saveNBT() {
103        super.saveNBT();
104        namedTag.putByte("Fuse", fuse);
105    }
106
107    public boolean onUpdate(int currentTick) {
108
109        if (closed) {
110            return false;
111        }
112
113        this.timing.startTiming();
114
115        int tickDiff = currentTick - lastUpdate;
116
117        if (tickDiff <= 0 && !justCreated) {
118            return true;
119        }
120
121        if (fuse % 5 == 0) {
122            this.setDataProperty(new IntEntityData(DATA_FUSE_LENGTH, fuse));
123        }
124
125        lastUpdate = currentTick;
126
127        boolean hasUpdate = entityBaseTick(tickDiff);
128
129        if (isAlive()) {
130
131            motionY -= getGravity();
132
133            move(motionX, motionY, motionZ);
134
135            float friction = 1 - getDrag();
136
137            motionX *= friction;
138            motionY *= friction;
139            motionZ *= friction;
140
141            updateMovement();
142
143            if (onGround) {
144                motionY *= -0.5;
145                motionX *= 0.7;
146                motionZ *= 0.7;
147            }
148
149            fuse -= tickDiff;
150
151            if (fuse <= 0) {
152                if (this.level.getGameRules().getBoolean(GameRule.TNT_EXPLODES))
153                    explode();
154                kill();
155            }
156
157        }
158
159        this.timing.stopTiming();
160
161        return hasUpdate || fuse >= 0 || Math.abs(motionX) > 0.00001 || Math.abs(motionY) > 0.00001 || Math.abs(motionZ) > 0.00001;
162    }
163
164    public void explode() {
165        EntityExplosionPrimeEvent event = new EntityExplosionPrimeEvent(this, 4);
166        server.getPluginManager().callEvent(event);
167        if (event.isCancelled()) {
168            return;
169        }
170        Explosion explosion = new Explosion(this, event.getForce(), this);
171        explosion.setFireChance(event.getFireChance());
172        if (event.isBlockBreaking()) {
173            explosion.explodeA();
174        }
175        explosion.explodeB();
176    }
177
178    public Entity getSource() {
179        return source;
180    }
181}
