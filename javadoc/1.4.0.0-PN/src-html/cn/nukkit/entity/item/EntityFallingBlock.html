020/**
021 * @author MagicDroidX
022 */
023public class EntityFallingBlock extends Entity {
024
025    public static final int NETWORK_ID = 66;
026
027    @Override
028    public float getWidth() {
029        return 0.98f;
030    }
031
032    @Override
033    public float getLength() {
034        return 0.98f;
035    }
036
037    @Override
038    public float getHeight() {
039        return 0.98f;
040    }
041
042    @Override
043    protected float getGravity() {
044        return 0.04f;
045    }
046
047    @Override
048    protected float getDrag() {
049        return 0.02f;
050    }
051
052    @Override
053    protected float getBaseOffset() {
054        return 0.49f;
055    }
056
057    @Override
058    public boolean canCollide() {
059        return false;
060    }
061
062    protected int blockId;
063    protected int damage;
064    protected boolean breakOnLava;
065
066    public EntityFallingBlock(FullChunk chunk, CompoundTag nbt) {
067        super(chunk, nbt);
068    }
069
070    @Override
071    protected void initEntity() {
072        super.initEntity();
073
074        if (namedTag != null) {
075            if (namedTag.contains("TileID")) {
076                blockId = namedTag.getInt("TileID");
077            } else if (namedTag.contains("Tile")) {
078                blockId = namedTag.getInt("Tile");
079                namedTag.putInt("TileID", blockId);
080            }
081
082            if (namedTag.contains("Data")) {
083                damage = namedTag.getByte("Data");
084            }
085        }
086
087        breakOnLava = namedTag.getBoolean("BreakOnLava");
088
089        if (blockId == 0) {
090            close();
091            return;
092        }
093
094        this.fireProof = true;
095        this.setDataFlag(DATA_FLAGS, DATA_FLAG_FIRE_IMMUNE, true);
096
097        setDataProperty(new IntEntityData(DATA_VARIANT, GlobalBlockPalette.getOrCreateRuntimeId(this.getBlock(), this.getDamage())));
098    }
099
100    public boolean canCollideWith(Entity entity) {
101        return false;
102    }
103
104    @Override
105    public boolean attack(EntityDamageEvent source) {
106        return source.getCause() == DamageCause.VOID && super.attack(source);
107    }
108
109    @Override
110    public boolean onUpdate(int currentTick) {
111
112        if (closed) {
113            return false;
114        }
115
116        this.timing.startTiming();
117
118        int tickDiff = currentTick - lastUpdate;
119        if (tickDiff <= 0 && !justCreated) {
120            return true;
121        }
122
123        lastUpdate = currentTick;
124
125        boolean hasUpdate = entityBaseTick(tickDiff);
126
127        if (isAlive()) {
128            motionY -= getGravity();
129
130            move(motionX, motionY, motionZ);
131
132            float friction = 1 - getDrag();
133
134            motionX *= friction;
135            motionY *= 1 - getDrag();
136            motionZ *= friction;
137
138            Vector3 pos = (new Vector3(x - 0.5, y, z - 0.5)).round();
139
140            if (breakOnLava && level.getBlock(pos.subtract(0, 1, 0)) instanceof BlockLava) {
141                close();
142                if (this.level.getGameRules().getBoolean(GameRule.DO_ENTITY_DROPS)) {
143                    getLevel().dropItem(this, Block.get(this.getBlock(), this.getDamage()).toItem());
144                }
145                level.addParticle(new DestroyBlockParticle(pos, Block.get(getBlock(), getDamage())));
146                return true;
147            }
148
149            if (onGround) {
150                close();
151                Block block = level.getBlock(pos);
152
153                Vector3 floorPos = (new Vector3(x - 0.5, y, z - 0.5)).floor();
154                Block floorBlock = this.level.getBlock(floorPos);
155                if (this.getBlock() == Block.SNOW_LAYER && floorBlock.getId() == Block.SNOW_LAYER && (floorBlock.getDamage() & 0x7) != 0x7) {
156                    int mergedHeight = (floorBlock.getDamage() & 0x7) + 1 + (this.getDamage() & 0x7) + 1;
157                    if (mergedHeight > 8) {
158                        EntityBlockChangeEvent event = new EntityBlockChangeEvent(this, floorBlock, Block.get(Block.SNOW_LAYER, 0x7));
159                        this.server.getPluginManager().callEvent(event);
160                        if (!event.isCancelled()) {
161                            this.level.setBlock(floorPos, event.getTo(), true);
162
163                            Vector3 abovePos = floorPos.up();
164                            Block aboveBlock = this.level.getBlock(abovePos);
165                            if (aboveBlock.getId() == Block.AIR) {
166                                EntityBlockChangeEvent event2 = new EntityBlockChangeEvent(this, aboveBlock, Block.get(Block.SNOW_LAYER, mergedHeight - 8 - 1));
167                                this.server.getPluginManager().callEvent(event2);
168                                if (!event2.isCancelled()) {
169                                    this.level.setBlock(abovePos, event2.getTo(), true);
170                                }
171                            }
172                        }
173                    } else {
174                        EntityBlockChangeEvent event = new EntityBlockChangeEvent(this, floorBlock, Block.get(Block.SNOW_LAYER, mergedHeight - 1));
175                        this.server.getPluginManager().callEvent(event);
176                        if (!event.isCancelled()) {
177                            this.level.setBlock(floorPos, event.getTo(), true);
178                        }
179                    }
180                } else if (block.getId() > 0 && block.isTransparent() && !block.canBeReplaced() || this.getBlock() == Block.SNOW_LAYER && block instanceof BlockLiquid) {
181                    if (this.getBlock() != Block.SNOW_LAYER ? this.level.getGameRules().getBoolean(GameRule.DO_ENTITY_DROPS) : this.level.getGameRules().getBoolean(GameRule.DO_TILE_DROPS)) {
182                        getLevel().dropItem(this, Block.get(this.getBlock(), this.getDamage()).toItem());
183                    }
184                } else {
185                    EntityBlockChangeEvent event = new EntityBlockChangeEvent(this, block, Block.get(getBlock(), getDamage()));
186                    server.getPluginManager().callEvent(event);
187                    if (!event.isCancelled()) {
188                        getLevel().setBlock(pos, event.getTo(), true);
189
190                        if (event.getTo().getId() == Item.ANVIL) {
191                            getLevel().addLevelEvent(block, LevelEventPacket.EVENT_SOUND_ANVIL_FALL);
192                        }
193                    }
194                }
195                hasUpdate = true;
196            }
197
198            updateMovement();
199        }
200
201        this.timing.stopTiming();
202
203        return hasUpdate || !onGround || Math.abs(motionX) > 0.00001 || Math.abs(motionY) > 0.00001 || Math.abs(motionZ) > 0.00001;
204    }
205
206    public int getBlock() {
207        return blockId;
208    }
209
210    public int getDamage() {
211        return damage;
212    }
213
214    @Override
215    public int getNetworkId() {
216        return NETWORK_ID;
217    }
218
219    @Override
220    public void saveNBT() {
221        namedTag.putInt("TileID", blockId);
222        namedTag.putByte("Data", damage);
223    }
224
225    @Override
226    public boolean canBeMovedByCurrents() {
227        return false;
228    }
