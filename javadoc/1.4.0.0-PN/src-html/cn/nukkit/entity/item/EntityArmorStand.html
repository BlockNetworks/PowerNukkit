063    private static int getArmorSlot(ItemArmor armorItem) {
064        if (armorItem.isHelmet()) {
065            return EntityArmorInventory.SLOT_HEAD;
066        } else if (armorItem.isChestplate()) {
067            return EntityArmorInventory.SLOT_CHEST;
068        } else if (armorItem.isLeggings()) {
069            return EntityArmorInventory.SLOT_LEGS;
070        } else {
071            return EntityArmorInventory.SLOT_FEET;
072        }
073    }

097
098        this.setHealth(6);
099        this.setMaxHealth(6);
100        this.setImmobile(true);
101
102        super.initEntity();
103
104        this.equipmentInventory = new EntityEquipmentInventory(this);
105        this.armorInventory = new EntityArmorInventory(this);
106
107        if (this.namedTag.contains(TAG_MAINHAND)) {
108            this.equipmentInventory.setItemInHand(NBTIO.getItemHelper(this.namedTag.getCompound(TAG_MAINHAND)), true);
109        }
110
111        if (this.namedTag.contains(TAG_OFFHAND)) {
112            this.equipmentInventory.setItemInOffhand(NBTIO.getItemHelper(this.namedTag.getCompound(TAG_OFFHAND)), true);
113        }
114
115        if (this.namedTag.contains(TAG_ARMOR)) {
116            ListTag<CompoundTag> armorList = this.namedTag.getList(TAG_ARMOR, CompoundTag.class);
117            for (CompoundTag armorTag : armorList.getAll()) {
118                this.armorInventory.setItem(armorTag.getByte("Slot"), NBTIO.getItemHelper(armorTag));
119            }
120        }
121
122        if (this.namedTag.contains(TAG_POSE_INDEX)) {
123            this.setPose(this.namedTag.getInt(TAG_POSE_INDEX));
124        }
125    }

139    @Override
140    public boolean onInteract(Player player, Item item, Vector3 clickedPos) {
141        if (player.isSpectator() || !isValid()) {
142            return false;
143        }
144        
145        // Name tag
146        if (!item.isNull() && item.getId() == ItemID.NAME_TAG && playerApplyNameTag(player, item, false)) {
147            return true;
148        }
149        
150        //Pose
151        if (player.isSneaking()) {
152            if (this.getPose() >= 12) {
153                this.setPose(0);
154            } else {
155                this.setPose(this.getPose() + 1);
156            }
157            return false; // Returning true would consume the item
158        }
159        
160        //Inventory
161        boolean isArmor;
162        
163        boolean hasItemInHand = !item.isNull();
164        int slot;
165        
166        if (hasItemInHand && item instanceof ItemArmor) {
167            ItemArmor itemArmor = (ItemArmor) item;
168            isArmor = true;
169            slot = getArmorSlot(itemArmor);
170        } else if (hasItemInHand && (item.getId() == ItemID.SKULL) || item.getBlockId() == BlockID.CARVED_PUMPKIN) {
171            isArmor = true;
172            slot = EntityArmorInventory.SLOT_HEAD;
173        } else if (hasItemInHand) {
174            isArmor = false;
175            if (item.getId() == ItemID.SHIELD) {
176                slot = EntityEquipmentInventory.OFFHAND;
177            } else {
178                slot = EntityEquipmentInventory.MAIN_HAND;
179            }
180        } else {
181            double clickHeight = clickedPos.y - this.y;
182            if (clickHeight >= 0.1 && clickHeight < 0.55 && !armorInventory.getBoots().isNull()) {
183                isArmor = true;
184                slot = EntityArmorInventory.SLOT_FEET;
185            } else if (clickHeight >= 0.9 && clickHeight < 1.6) {
186                if (!equipmentInventory.getItemInOffhand().isNull()) {
187                    isArmor = false;
188                    slot = EntityEquipmentInventory.OFFHAND;
189                } else if (!equipmentInventory.getItemInHand().isNull()) {
190                    isArmor = false;
191                    slot = EntityEquipmentInventory.MAIN_HAND;
192                } else if (!armorInventory.getChestplate().isNull()) {
193                    isArmor = true;
194                    slot = EntityArmorInventory.SLOT_CHEST;
195                } else {
196                    return false;
197                }
198            } else if (clickHeight >= 0.4 && clickHeight < 1.2 && !armorInventory.getLeggings().isNull()) {
199                isArmor = true;
200                slot = EntityArmorInventory.SLOT_LEGS;
201            } else if (clickHeight >= 1.6 && !armorInventory.getHelmet().isNull()) {
202                isArmor = true;
203                slot = EntityArmorInventory.SLOT_HEAD;
204            } else if (!equipmentInventory.getItemInOffhand().isNull()) {
205                isArmor = false;
206                slot = EntityEquipmentInventory.OFFHAND;
207            } else if (!equipmentInventory.getItemInHand().isNull()) {
208                isArmor = false;
209                slot = EntityEquipmentInventory.MAIN_HAND;
210            } else {
211                return false;
212            }
213        }
214
215        boolean changed = false;
216        if (isArmor) {
217            changed = this.tryChangeEquipment(player, item, slot, true);
218            slot = EntityEquipmentInventory.MAIN_HAND;
219        }
220        
221        if (!changed) {
222            changed = this.tryChangeEquipment(player, item, slot, false);
223        }
224        
225        if (changed) {
226            level.addSound(this, Sound.MOB_ARMOR_STAND_PLACE);
227        }
228
229        return false; // Returning true would consume the item but tryChangeEquipment already manages the inventory
230    }

232    private boolean tryChangeEquipment(Player player, Item handItem, int slot, boolean isArmorSlot) {
233        BaseInventory inventory = isArmorSlot? armorInventory : equipmentInventory;
234        Item item = inventory.getItem(slot);
235        
236        if (item.isNull() && !handItem.isNull()) {
237            // Adding item to the armor stand
238            Item itemClone = handItem.clone();
239            itemClone.setCount(1);
240            inventory.setItem(slot, itemClone);
241            if (!player.isCreative()) {
242                handItem.count--;
243                player.getInventory().setItem(player.getInventory().getHeldItemIndex(), handItem);
244            }
245            return true;
246        } else if (!item.isNull()) {
247            Item itemtoAddToArmorStand = Item.getBlock(BlockID.AIR);
248            if (!handItem.isNull()) {
249                if (handItem.equals(item, true, true)) {
250                    // Attempted to replace with the same item type
251                    return false;
252                }
253                
254                if (item.count > 1) {
255                    // The armor stand have more items than 1, item swapping is not supported in this situation
256                    return false;
257                }
258                
259                Item itemToSetToPlayerInv;
260                if (handItem.count > 1) {
261                    itemtoAddToArmorStand = handItem.clone();
262                    itemtoAddToArmorStand.setCount(1);
263                    
264                    itemToSetToPlayerInv = handItem.clone();
265                    itemToSetToPlayerInv.count--;
266                } else {
267                    itemtoAddToArmorStand = handItem.clone();
268                    itemToSetToPlayerInv = Item.getBlock(BlockID.AIR);
269                }
270                player.getInventory().setItem(player.getInventory().getHeldItemIndex(), itemToSetToPlayerInv);
271            }
272            
273            // Removing item from the armor stand
274            Item[] notAdded = player.getInventory().addItem(item);
275            if (notAdded.length > 0) {
276                if (notAdded[0].count == item.count) {
277                    if (!handItem.isNull()) {
278                        player.getInventory().setItem(player.getInventory().getHeldItemIndex(), handItem);
279                    }
280                    return false;
281                }
282                
283                Item itemClone = item.clone();
284                itemClone.count -= notAdded[0].count;
285                inventory.setItem(slot, itemClone);
286            } else {
287                inventory.setItem(slot, itemtoAddToArmorStand);
288            }
289            return true;
290        }
291        return false;
292    }

299    private void setPose(int pose) {
300        this.dataProperties.putInt(Entity.DATA_ARMOR_STAND_POSE_INDEX, pose);
301        SetEntityDataPacket setEntityDataPacket = new SetEntityDataPacket();
302        setEntityDataPacket.eid = this.getId();
303        setEntityDataPacket.metadata = this.getDataProperties();
304        Server.getInstance().getOnlinePlayers().values().forEach(all -> all.dataPacket(setEntityDataPacket));
305    }

309        super.saveNBT();
310
311        this.namedTag.put(TAG_MAINHAND, NBTIO.putItemHelper(this.equipmentInventory.getItemInHand()));
312        this.namedTag.put(TAG_OFFHAND, NBTIO.putItemHelper(this.equipmentInventory.getItemInOffhand()));

315            ListTag<CompoundTag> armorTag = new ListTag<>(TAG_ARMOR);
316            for (int i = 0; i < 4; i++) {
317                armorTag.add(NBTIO.putItemHelper(this.armorInventory.getItem(i), i));
318            }
319            this.namedTag.putList(armorTag);
320        }
321
322        this.namedTag.putInt(TAG_POSE_INDEX, this.getPose());
323    }

349    @Override
350    public void kill() {
351        super.kill();
352        EntityDamageEvent lastDamageCause = this.lastDamageCause;
353        boolean byAttack = lastDamageCause != null && lastDamageCause.getCause() == EntityDamageEvent.DamageCause.ENTITY_ATTACK;
354        
355        Vector3 pos = getPosition();
356        
357        pos.y += 0.2;
358        level.dropItem(pos, armorInventory.getBoots());
359        
360        pos.y = y + 0.6;
361        level.dropItem(pos, armorInventory.getLeggings());
362        
363        pos.y = y + 1.4;
364        level.dropItem(byAttack? pos : this, Item.get(ItemID.ARMOR_STAND));
365        level.dropItem(pos, armorInventory.getChestplate());
366        equipmentInventory.getContents().values().forEach(items -> this.level.dropItem(this, items));
367        equipmentInventory.clearAll();
368        
369        pos.y = y + 1.8;
370        level.dropItem(pos, armorInventory.getHelmet());
371        armorInventory.clearAll();
372
373        level.addSound(this, Sound.MOB_ARMOR_STAND_BREAK);
374    }

376    @Override
377    public boolean attack(EntityDamageEvent source) {
378
379        switch (source.getCause()) {
380            case FALL:
381                source.setCancelled(true);
382                level.addSound(this, Sound.MOB_ARMOR_STAND_LAND);
383                break;
384            case CONTACT:
385            case HUNGER:
386            case MAGIC:
387            case DROWNING:
388            case SUFFOCATION:
389            case PROJECTILE:
390                source.setCancelled(true);
391                break;
392            case FIRE:
393            case FIRE_TICK:
394            case LAVA:
395                if (hasEffect(Effect.FIRE_RESISTANCE)) {
396                    return false;
397                }
398            default:
399        }
400        
401        if (source.getCause() != EntityDamageEvent.DamageCause.ENTITY_ATTACK) {
402            if (namedTag.getByte("InvulnerableTimer") > 0) {
403                source.setCancelled(true);
404            }
405            if (super.attack(source)) {
406                namedTag.putByte("InvulnerableTimer", 9);
407                return true;
408            }
409            return false;
410        }
411        
412        getServer().getPluginManager().callEvent(source);
413        if (source.isCancelled()) {
414            return false;
415        }
416        setLastDamageCause(source);
417
418        if (getDataPropertyInt(DATA_HURT_TIME) > 0) {
419            setHealth(0);
420            return true;
421        }
422        
423        if (source instanceof EntityDamageByEntityEvent) {
424            EntityDamageByEntityEvent event = (EntityDamageByEntityEvent) source;
425            if (event.getDamager() instanceof Player){
426                Player player = (Player) event.getDamager();
427                if (player.isCreative()) {
428                    this.level.addParticle(new DestroyBlockParticle(this, Block.get(BlockID.WOODEN_PLANKS)));
429                    this.close();
430                    return true;
431                }
432            }
433        }
434
435        setDataProperty(new IntEntityData(DATA_HURT_TIME, 9), true);
436        level.addSound(this, Sound.MOB_ARMOR_STAND_HIT);
437        
438        return true;
439    }

447    @Override
448    public boolean entityBaseTick(int tickDiff) {
449        boolean hasUpdate = super.entityBaseTick(tickDiff);
450
451        int hurtTime = getDataPropertyInt(DATA_HURT_TIME);
452        if (hurtTime > 0 && age % 2 == 0) {
453            setDataProperty(new IntEntityData(DATA_HURT_TIME, hurtTime - 1), true);
454            hasUpdate = true;
455        }
456        hurtTime = namedTag.getByte("InvulnerableTimer");
457        if (hurtTime > 0 && age % 2 == 0) {
458            namedTag.putByte("InvulnerableTimer", hurtTime - 1);
459        }
460
461        return hasUpdate;
462    }

475    @Override
476    public boolean onUpdate(int currentTick) {
477        int tickDiff = currentTick - lastUpdate;
478        boolean hasUpdated = super.onUpdate(currentTick);
479
480        if (closed || tickDiff <= 0 && !justCreated) {
481            return hasUpdated;
482        }
483
484        this.timing.startTiming();
485
486        lastUpdate = currentTick;
487
488        boolean hasUpdate = entityBaseTick(tickDiff);
489
490        if (isAlive()) {
491            if (getHealth() < getMaxHealth()) {
492                setHealth(getHealth() + 0.001f);
493            }
494            motionY -= getGravity();
495            
496            double highestPosition = this.highestPosition;
497            move(motionX, motionY, motionZ);
498
499            float friction = 1 - getDrag();
500
501            motionX *= friction;
502            motionY *= 1 - getDrag();
503            motionZ *= friction;
504
505            updateMovement();
506            hasUpdate = true;
507            if (onGround && (highestPosition - y) >= 3) {
508                level.addSound(this, Sound.MOB_ARMOR_STAND_LAND);
509            }
510        }
511
512        this.timing.stopTiming();
513
514        return hasUpdate || !onGround || Math.abs(motionX) > 0.00001 || Math.abs(motionY) > 0.00001 || Math.abs(motionZ) > 0.00001;
515    }
