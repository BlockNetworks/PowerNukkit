001package cn.nukkit.blockproperty.exception;
002
003import cn.nukkit.api.PowerNukkitOnly;
004import cn.nukkit.api.Since;
005import cn.nukkit.blockproperty.BlockProperty;
006
007import javax.annotation.Nonnull;
008import javax.annotation.ParametersAreNullableByDefault;
009
010@PowerNukkitOnly
011@Since("1.4.0.0-PN")
012@ParametersAreNullableByDefault
013public class InvalidBlockPropertyException extends IllegalArgumentException {
014    private static final long serialVersionUID = -6934630506175381230L;
015    
016    private final BlockProperty<?> property;
017
018    @PowerNukkitOnly
019    @Since("1.4.0.0-PN")
020    public InvalidBlockPropertyException(@Nonnull BlockProperty<?> property) {
021        super(buildMessage(property));
022        this.property = property;
023    }
024
025    @PowerNukkitOnly
026    @Since("1.4.0.0-PN")
027    public InvalidBlockPropertyException(@Nonnull BlockProperty<?> property, String message) {
028        super(buildMessage(property)+". "+message);
029        this.property = property;
030    }
031
032    @PowerNukkitOnly
033    @Since("1.4.0.0-PN")
034    public InvalidBlockPropertyException(@Nonnull BlockProperty<?> property, String message, Throwable cause) {
035        super(buildMessage(property)+". "+message, cause);
036        this.property = property;
037    }
038
039    @PowerNukkitOnly
040    @Since("1.4.0.0-PN")
041    public InvalidBlockPropertyException(@Nonnull BlockProperty<?> property, Throwable cause) {
042        super(buildMessage(property), cause);
043        this.property = property;
044    }
045
046    private static String buildMessage(@Nonnull BlockProperty<?> property) {
047        return "Property: "+property.getName();
048    }
049
050    @PowerNukkitOnly
051    @Since("1.4.0.0-PN")
052    @Nonnull
053    public BlockProperty<?> getProperty() {
054        return property;
055    }
056}
