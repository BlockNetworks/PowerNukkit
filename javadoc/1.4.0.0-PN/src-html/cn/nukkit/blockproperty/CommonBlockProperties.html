001package cn.nukkit.blockproperty;
002
003import cn.nukkit.api.PowerNukkitOnly;
004import cn.nukkit.api.Since;
005import cn.nukkit.block.Block;
006import cn.nukkit.math.BlockFace;
007
008@PowerNukkitOnly
009@Since("1.4.0.0-PN")
010public final class CommonBlockProperties {
011    @PowerNukkitOnly
012    @Since("1.4.0.0-PN")
013    public static final String LEGACY_PROPERTY_NAME = "nukkit-legacy";
014
015    @PowerNukkitOnly
016    @Since("1.4.0.0-PN")
017    public static final BlockProperties EMPTY_PROPERTIES = new BlockProperties();
018    @PowerNukkitOnly
019    @Since("1.4.0.0-PN")
020    public static final BlockProperties LEGACY_PROPERTIES = new BlockProperties(new IntBlockProperty(LEGACY_PROPERTY_NAME, true, Block.DATA_MASK));
021    @PowerNukkitOnly
022    @Since("1.4.0.0-PN")
023    public static final BlockProperties LEGACY_BIG_PROPERTIES = new BlockProperties(new UnsignedIntBlockProperty(LEGACY_PROPERTY_NAME, true, 0xFFFFFFFF));
024
025    @PowerNukkitOnly
026    @Since("1.4.0.0-PN")
027    public static final BooleanBlockProperty OPEN = new BooleanBlockProperty("open_bit", false);
028    @PowerNukkitOnly
029    @Since("1.4.0.0-PN")
030    public static final BooleanBlockProperty TOGGLE = new BooleanBlockProperty("toggle_bit", false);
031    @PowerNukkitOnly
032    @Since("1.4.0.0-PN")
033    public static final IntBlockProperty REDSTONE_SIGNAL = new IntBlockProperty("redstone_signal", false, 15);
034    @PowerNukkitOnly
035    @Since("1.4.0.0-PN")
036    public static final BooleanBlockProperty UPPER_BLOCK = new BooleanBlockProperty("upper_block_bit", false);
037
038    @PowerNukkitOnly
039    @Since("1.4.0.0-PN")
040    public static final BlockProperty<BlockFace> FACING_DIRECTION = new ArrayBlockProperty<>("facing_direction", false, new BlockFace[] {
041            // Index based
042            BlockFace.DOWN, BlockFace.UP,
043            BlockFace.NORTH, BlockFace.SOUTH,
044            BlockFace.WEST, BlockFace.EAST,
045    }).ordinal(true);
046
047    @PowerNukkitOnly
048    @Since("1.4.0.0-PN")
049    public static final BlockProperty<BlockFace> DIRECTION = new ArrayBlockProperty<>("direction", false, new BlockFace[]{
050            // Horizontal-index based
051            BlockFace.SOUTH, BlockFace.WEST,
052            BlockFace.NORTH, BlockFace.EAST,
053    }).ordinal(true);
054
055    @PowerNukkitOnly
056    @Since("1.4.0.0-PN")
057    public static final BlockProperty<BlockFace.Axis> PILLAR_AXIS = new ArrayBlockProperty<>("pillar_axis", false, new BlockFace.Axis[]{
058            BlockFace.Axis.Y, BlockFace.Axis.X, BlockFace.Axis.Z
059    });
060
061    @PowerNukkitOnly
062    @Since("1.4.0.0-PN")
063    public static final IntBlockProperty DEPRECATED = new IntBlockProperty("deprecated", false, 3);
064
065    private CommonBlockProperties() {
066        throw new UnsupportedOperationException("This is a utility class and cannot be instantiated");
067    }
068
069    @PowerNukkitOnly
070    @Since("1.4.0.0-PN")
071    public static final BooleanBlockProperty POWERED = new BooleanBlockProperty("powered_bit", false);
072}
