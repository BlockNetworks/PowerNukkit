001package cn.nukkit.blockproperty.value;
002
003import cn.nukkit.api.PowerNukkitOnly;
004import cn.nukkit.api.Since;
005import cn.nukkit.blockproperty.ArrayBlockProperty;
006import cn.nukkit.utils.BlockColor;
007
008import javax.annotation.Nonnull;
009import java.util.Arrays;
010import java.util.stream.Collectors;
011
012@PowerNukkitOnly
013@Since("1.4.0.0-PN")
014public enum StoneSlab4Type {
015    @PowerNukkitOnly @Since("1.4.0.0-PN") MOSSY_STONE_BRICK(BlockColor.STONE_BLOCK_COLOR),
016    @PowerNukkitOnly @Since("1.4.0.0-PN") SMOOTH_QUARTZ(BlockColor.QUARTZ_BLOCK_COLOR),
017    @PowerNukkitOnly @Since("1.4.0.0-PN") STONE(BlockColor.STONE_BLOCK_COLOR),
018    @PowerNukkitOnly @Since("1.4.0.0-PN") CUT_SANDSTONE(BlockColor.SAND_BLOCK_COLOR),
019    @PowerNukkitOnly @Since("1.4.0.0-PN") CUT_RED_SANDSTONE(BlockColor.ORANGE_BLOCK_COLOR);
020    @PowerNukkitOnly @Since("1.4.0.0-PN") public static final ArrayBlockProperty<StoneSlab4Type> PROPERTY = new ArrayBlockProperty<>("stone_slab_type_4", true, values());
021    private final BlockColor color;
022
023    private final String englishName;
024
025    StoneSlab4Type(BlockColor color) {
026        this.color = color;
027        englishName = Arrays.stream(name().split("_")).map(name-> name.substring(0, 1) + name.substring(1).toLowerCase()).collect(Collectors.joining(" "));
028    }
029
030    @PowerNukkitOnly
031    @Since("1.4.0.0-PN")
032    @Nonnull
033    public BlockColor getColor() {
034        return this.color;
035    }
036
037    @PowerNukkitOnly
038    @Since("1.4.0.0-PN")
039    @Nonnull
040    public String getEnglishName() {
041        return this.englishName;
042    }
043}
