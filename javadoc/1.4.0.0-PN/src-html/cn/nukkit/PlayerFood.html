001package cn.nukkit;
002
003import cn.nukkit.entity.Attribute;
004import cn.nukkit.event.entity.EntityDamageEvent;
005import cn.nukkit.event.entity.EntityDamageEvent.DamageCause;
006import cn.nukkit.event.entity.EntityRegainHealthEvent;
007import cn.nukkit.event.player.PlayerFoodLevelChangeEvent;
008import cn.nukkit.item.food.Food;
009import cn.nukkit.potion.Effect;
010
011/**
012 * @author funcraft
013 * @since 2015/11/11
014 */
015public class PlayerFood {
016
017    private int foodLevel = 20;
018    private final int maxFoodLevel;
019    private float foodSaturationLevel = 20f;
020    private int foodTickTimer = 0;
021    private double foodExpLevel = 0;
022
023    private final Player player;
024
025    public PlayerFood(Player player, int foodLevel, float foodSaturationLevel) {
026        this.player = player;
027        this.foodLevel = foodLevel;
028        this.maxFoodLevel = 20;
029        this.foodSaturationLevel = foodSaturationLevel;
030    }
031
032    public Player getPlayer() {
033        return this.player;
034    }
035
036    public int getLevel() {
037        return this.foodLevel;
038    }
039
040    public int getMaxLevel() {
041        return this.maxFoodLevel;
042    }
043
044    public void setLevel(int foodLevel) {
045        this.setLevel(foodLevel, -1);
046    }
047
048    public void setLevel(int foodLevel, float saturationLevel) {
049        if (foodLevel > 20) {
050            foodLevel = 20;
051        }
052
053        if (foodLevel < 0) {
054            foodLevel = 0;
055        }
056
057        if (foodLevel <= 6 && !(this.getLevel() <= 6)) {
058            if (this.getPlayer().isSprinting()) {
059                this.getPlayer().setSprinting(false);
060            }
061        }
062
063        PlayerFoodLevelChangeEvent ev = new PlayerFoodLevelChangeEvent(this.getPlayer(), foodLevel, saturationLevel);
064        this.getPlayer().getServer().getPluginManager().callEvent(ev);
065        if (ev.isCancelled()) {
066            this.sendFoodLevel(this.getLevel());
067            return;
068        }
069        int foodLevel0 = ev.getFoodLevel();
070        float fsl = ev.getFoodSaturationLevel();
071        this.foodLevel = foodLevel;
072        if (fsl != -1) {
073            if (fsl > foodLevel) fsl = foodLevel;
074            this.foodSaturationLevel = fsl;
075        }
076        this.foodLevel = foodLevel0;
077        this.sendFoodLevel();
078    }
079
080    public float getFoodSaturationLevel() {
081        return this.foodSaturationLevel;
082    }
083
084    public void setFoodSaturationLevel(float fsl) {
085        if (fsl > this.getLevel()) fsl = this.getLevel();
086        if (fsl < 0) fsl = 0;
087        PlayerFoodLevelChangeEvent ev = new PlayerFoodLevelChangeEvent(this.getPlayer(), this.getLevel(), fsl);
088        this.getPlayer().getServer().getPluginManager().callEvent(ev);
089        if (ev.isCancelled()) {
090            return;
091        }
092        fsl = ev.getFoodSaturationLevel();
093        this.foodSaturationLevel = fsl;
094    }
095
096    public void useHunger() {
097        this.useHunger(1);
098    }
099
100    public void useHunger(int amount) {
101        float sfl = this.getFoodSaturationLevel();
102        int foodLevel = this.getLevel();
103        if (sfl > 0) {
104            float newSfl = sfl - amount;
105            if (newSfl < 0) newSfl = 0;
106            this.setFoodSaturationLevel(newSfl);
107        } else {
108            this.setLevel(foodLevel - amount);
109        }
110    }
111
112    public void addFoodLevel(Food food) {
113        this.addFoodLevel(food.getRestoreFood(), food.getRestoreSaturation());
114    }
115
116    public void addFoodLevel(int foodLevel, float fsl) {
117        this.setLevel(this.getLevel() + foodLevel, this.getFoodSaturationLevel() + fsl);
118    }
119
120    public void sendFoodLevel() {
121        this.sendFoodLevel(this.getLevel());
122    }
123
124    public void reset() {
125        this.foodLevel = 20;
126        this.foodSaturationLevel = 20;
127        this.foodExpLevel = 0;
128        this.foodTickTimer = 0;
129        this.sendFoodLevel();
130    }
131
132    public void sendFoodLevel(int foodLevel) {
133        if (this.getPlayer().spawned) {
134            this.getPlayer().setAttribute(Attribute.getAttribute(Attribute.MAX_HUNGER).setValue(foodLevel));
135        }
136    }
137
138    public void update(int tickDiff) {
139        if (!this.getPlayer().isFoodEnabled()) return;
140        if (this.getPlayer().isAlive()) {
141            int diff = Server.getInstance().getDifficulty();
142            if (this.getLevel() > 17) {
143                this.foodTickTimer += tickDiff;
144                if (this.foodTickTimer >= 80) {
145                    if (this.getPlayer().getHealth() < this.getPlayer().getMaxHealth()) {
146                        EntityRegainHealthEvent ev = new EntityRegainHealthEvent(this.getPlayer(), 1, EntityRegainHealthEvent.CAUSE_EATING);
147                        this.getPlayer().heal(ev);
148                        this.updateFoodExpLevel(6);
149                    }
150                    this.foodTickTimer = 0;
151                }
152            } else if (this.getLevel() == 0) {
153                this.foodTickTimer += tickDiff;
154                if (this.foodTickTimer >= 80) {
155                    EntityDamageEvent ev = new EntityDamageEvent(this.getPlayer(), DamageCause.HUNGER, 1);
156                    float now = this.getPlayer().getHealth();
157                    if (diff == 1) {
158                        if (now > 10) this.getPlayer().attack(ev);
159                    } else if (diff == 2) {
160                        if (now > 1) this.getPlayer().attack(ev);
161                    } else {
162                        this.getPlayer().attack(ev);
163                    }
164
165                    this.foodTickTimer = 0;
166                }
167            }
168            if (this.getPlayer().hasEffect(Effect.HUNGER)) {
169                this.updateFoodExpLevel(0.025);
170            }
171        }
172    }
173
174    public void updateFoodExpLevel(double use) {
175        if (!this.getPlayer().isFoodEnabled()) return;
176        if (Server.getInstance().getDifficulty() == 0) return;
177        if (this.getPlayer().hasEffect(Effect.SATURATION)) return;
178        this.foodExpLevel += use;
179        if (this.foodExpLevel > 4) {
180            this.useHunger(1);
181            this.foodExpLevel = 0;
182        }
183    }
184
185    /**
186     * @deprecated use {@link #setLevel(int)} instead
187     * @param foodLevel level
188     **/
189    @Deprecated
190    public void setFoodLevel(int foodLevel) {
191        setLevel(foodLevel);
192    }
193
194    /**
195     * @deprecated use {@link #setLevel(int, float)} instead
196     * @param foodLevel level
197     * @param saturationLevel saturation
198     **/
199    @Deprecated
200    public void setFoodLevel(int foodLevel, float saturationLevel) {
201        setLevel(foodLevel, saturationLevel);
202    }
203}
