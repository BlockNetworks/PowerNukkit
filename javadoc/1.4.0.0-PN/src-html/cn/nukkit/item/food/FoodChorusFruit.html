001package cn.nukkit.item.food;
002
003import cn.nukkit.Player;
004import cn.nukkit.block.Block;
005import cn.nukkit.block.BlockLiquid;
006import cn.nukkit.event.player.PlayerTeleportEvent;
007import cn.nukkit.item.Item;
008import cn.nukkit.level.Level;
009import cn.nukkit.math.NukkitRandom;
010import cn.nukkit.math.Vector3;
011import cn.nukkit.network.protocol.LevelSoundEventPacket;
012
013/**
014 * @author Leonidius20
015 * @since 20.08.18
016 */
017public class FoodChorusFruit extends FoodNormal {
018
019    public FoodChorusFruit() {
020        super(4, 2.4F);
021        addRelative(Item.CHORUS_FRUIT);
022    }
023
024    @Override
025    protected boolean onEatenBy(Player player) {
026        super.onEatenBy(player);
027        // Teleportation
028        int minX = player.getFloorX() - 8;
029        int minY = player.getFloorY() - 8;
030        int minZ = player.getFloorZ() - 8;
031        int maxX = minX + 16;
032        int maxY = minY + 16;
033        int maxZ = minZ + 16;
034
035        Level level = player.getLevel();
036        if (level == null) return false;
037
038        NukkitRandom random = new NukkitRandom();
039        for (int attempts = 0; attempts < 128; attempts++) {
040            int x = random.nextRange(minX, maxX);
041            int y = random.nextRange(minY, maxY);
042            int z = random.nextRange(minZ, maxZ);
043
044            if (y < 0) continue;
045
046            while (y >= 0 && !level.getBlock(new Vector3(x, y + 1, z)).isSolid()) {
047                y--;
048            }
049            y++; // Back up to non solid
050
051            Block blockUp = level.getBlock(new Vector3(x, y + 1, z));
052            Block blockUp2 = level.getBlock(new Vector3(x, y + 2, z));
053
054            if (blockUp.isSolid() || blockUp instanceof BlockLiquid ||
055                    blockUp2.isSolid() || blockUp2 instanceof BlockLiquid) {
056                continue;
057            }
058
059            // Sounds are broadcast at both source and destination
060            level.addLevelSoundEvent(player, LevelSoundEventPacket.SOUND_TELEPORT);
061            player.teleport(new Vector3(x + 0.5, y + 1, z + 0.5), PlayerTeleportEvent.TeleportCause.CHORUS_FRUIT);
062            level.addLevelSoundEvent(player, LevelSoundEventPacket.SOUND_TELEPORT);
063
064            break;
065        }
066
067        return true;
068    }
069
070}
