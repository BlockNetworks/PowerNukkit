001package cn.nukkit.item;
002
003import cn.nukkit.api.PowerNukkitOnly;
004import cn.nukkit.api.Since;
005import cn.nukkit.block.Block;
006import cn.nukkit.block.BlockID;
007import cn.nukkit.entity.Entity;
008import cn.nukkit.item.enchantment.Enchantment;
009import cn.nukkit.nbt.tag.ByteTag;
010import cn.nukkit.nbt.tag.Tag;
011
012import javax.annotation.Nonnull;
013import java.util.Random;
014
015import static cn.nukkit.utils.Utils.dynamic;
016
017/**
018 * @author MagicDroidX (Nukkit Project)
019 */
020public abstract class ItemTool extends Item implements ItemDurable {
021    public static final int TIER_WOODEN = 1;
022    public static final int TIER_GOLD = 2;
023    public static final int TIER_STONE = 3;
024    public static final int TIER_IRON = 4;
025    public static final int TIER_DIAMOND = 5;
026    @Since("1.4.0.0-PN") public static final int TIER_NETHERITE = 6;
027
028    public static final int TYPE_NONE = 0;
029    public static final int TYPE_SWORD = 1;
030    public static final int TYPE_SHOVEL = 2;
031    public static final int TYPE_PICKAXE = 3;
032    public static final int TYPE_AXE = 4;
033    public static final int TYPE_SHEARS = 5;
034    @Since("1.4.0.0-PN") public static final int TYPE_HOE = 6;
035    
036    /**
037     * Same breaking speed independent of the tool.
038     */
039    @PowerNukkitOnly
040    public static final int TYPE_HANDS_ONLY = dynamic(Integer.MAX_VALUE);
041
042    public static final int DURABILITY_WOODEN = dynamic(60);
043    public static final int DURABILITY_GOLD = dynamic(33);
044    public static final int DURABILITY_STONE = dynamic(132);
045    public static final int DURABILITY_IRON = dynamic(251);
046    public static final int DURABILITY_DIAMOND = dynamic(1562);
047    @Since("1.4.0.0-PN") public static final int DURABILITY_NETHERITE = dynamic(2032);
048    public static final int DURABILITY_FLINT_STEEL = dynamic(65);
049    public static final int DURABILITY_SHEARS = dynamic(239);
050    public static final int DURABILITY_BOW = dynamic(385);
051    public static final int DURABILITY_TRIDENT = dynamic(251);
052    public static final int DURABILITY_FISHING_ROD = dynamic(65);
053    @Since("1.4.0.0-PN") public static final int DURABILITY_CROSSBOW = dynamic(465);
054    
055    @PowerNukkitOnly
056    @Since("1.4.0.0-PN")
057    @Nonnull
058    public static Item getBestTool(int toolType) {
059        switch (toolType) {
060            case TYPE_NONE:
061            case TYPE_PICKAXE:
062                return Item.get(ItemID.NETHERITE_PICKAXE);
063            case TYPE_AXE:
064                return Item.get(ItemID.NETHERITE_AXE);
065            case TYPE_SHOVEL:
066                return Item.get(ItemID.NETHERITE_SHOVEL);
067            case TYPE_SHEARS:
068                return Item.get(ItemID.SHEARS);
069            case TYPE_SWORD:
070                return Item.get(ItemID.NETHERITE_SWORD);
071            default:
072                // Can't use the switch-case syntax because they are dynamic types
073                if (toolType == TYPE_HOE) {
074                    return Item.get(ItemID.NETHERITE_HOE);
075                }
076                if (toolType == TYPE_HANDS_ONLY) {
077                    return Item.getBlock(BlockID.AIR);
078                }
079                return Item.get(ItemID.NETHERITE_PICKAXE);
080        }
081    }
082
083    public ItemTool(int id) {
084        this(id, 0, 1, UNKNOWN_STR);
085    }
086
087    public ItemTool(int id, Integer meta) {
088        this(id, meta, 1, UNKNOWN_STR);
089    }
090
091    public ItemTool(int id, Integer meta, int count) {
092        this(id, meta, count, UNKNOWN_STR);
093    }
094
095    public ItemTool(int id, Integer meta, int count, String name) {
096        super(id, meta, count, name);
097    }
098
099    @Override
100    public int getMaxStackSize() {
101        return 1;
102    }
103
104    @Override
105    public boolean useOn(Block block) {
106        if (this.isUnbreakable() || isDurable() || !damageWhenBreaking()) {
107            return true;
108        }
109
110        if (block.getToolType() == ItemTool.TYPE_PICKAXE && this.isPickaxe() ||
111                block.getToolType() == ItemTool.TYPE_SHOVEL && this.isShovel() ||
112                block.getToolType() == ItemTool.TYPE_AXE && this.isAxe() ||
113                block.getToolType() == ItemTool.TYPE_HOE && this.isHoe() ||
114                block.getToolType() == ItemTool.TYPE_SWORD && this.isSword() ||
115                block.getToolType() == ItemTool.TYPE_SHEARS && this.isShears() ||
116                block.getToolType() == ItemTool.TYPE_HOE && this.isHoe()
117                ) {
118            this.meta++;
119        } else if (!this.isShears() && block.calculateBreakTime(this) > 0) {
120            this.meta += 2;
121        } else if (this.isHoe()) {
122            if (block.getId() == GRASS || block.getId() == DIRT) {
123                this.meta++;
124            }
125        } else {
126            this.meta++;
127        }
128        return true;
129    }
130
131    @Override
132    public boolean useOn(Entity entity) {
133        if (this.isUnbreakable() || isDurable() || !damageWhenBreaking()) {
134            return true;
135        }
136
137        if ((entity != null) && !this.isSword()) {
138            this.meta += 2;
139        } else {
140            this.meta++;
141        }
142
143        return true;
144    }
145
146    private boolean isDurable() {
147        if (!hasEnchantments()) {
148            return false;
149        }
150
151        Enchantment durability = getEnchantment(Enchantment.ID_DURABILITY);
152        return durability != null && durability.getLevel() > 0 && (100 / (durability.getLevel() + 1)) <= new Random().nextInt(100);
153    }
154
155    @Override
156    public boolean isUnbreakable() {
157        Tag tag = this.getNamedTagEntry("Unbreakable");
158        return tag instanceof ByteTag && ((ByteTag) tag).data > 0;
159    }
160
161    @Override
162    public boolean isPickaxe() {
163        return false;
164    }
165
166    @Override
167    public boolean isAxe() {
168        return false;
169    }
170
171    @Override
172    public boolean isSword() {
173        return false;
174    }
175
176    @Override
177    public boolean isShovel() {
178        return false;
179    }
180
181    @Override
182    public boolean isHoe() {
183        return false;
184    }
185
186    @Override
187    public boolean isShears() {
188        return (this.id == SHEARS);
189    }
190
191    @Override
192    public boolean isTool() {
193        switch (this.id) {
194            case FLINT_STEEL:
195            case SHEARS:
196            case BOW:
197            case CROSSBOW:
198            case SHIELD:
199                return true;
200            default:
201                return this.isPickaxe() || this.isAxe() || this.isShovel() || this.isSword() || this.isHoe();
202        }
203    }
204
205    @Override
206    public int getEnchantAbility() {
207        int tier = this.getTier();
208        switch (tier) {
209            case TIER_STONE:
210                return 5;
211            case TIER_WOODEN:
212            case TIER_DIAMOND:
213                return 10;
214            case TIER_GOLD:
215                return 22;
216            case TIER_IRON:
217                return 14;
218        }
219        
220        if (tier == TIER_NETHERITE) {
221            return 15;
222        }
223        return 0;
224    }

226}
