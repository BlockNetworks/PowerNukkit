001package cn.nukkit.item;
002
003import cn.nukkit.Player;
004import cn.nukkit.api.PowerNukkitOnly;
005import cn.nukkit.api.Since;
006import cn.nukkit.block.Block;
007import cn.nukkit.entity.Entity;
008import cn.nukkit.event.entity.CreatureSpawnEvent;
009import cn.nukkit.event.entity.CreatureSpawnEvent.SpawnReason;
010import cn.nukkit.level.Level;
011import cn.nukkit.level.format.FullChunk;
012import cn.nukkit.math.BlockFace;
013import cn.nukkit.nbt.tag.CompoundTag;
014import cn.nukkit.nbt.tag.DoubleTag;
015import cn.nukkit.nbt.tag.FloatTag;
016import cn.nukkit.nbt.tag.ListTag;
017
018import java.util.Random;
019
020/**
021 * @author MagicDroidX (Nukkit Project)
022 */
023public class ItemSpawnEgg extends Item {
024    
025    public ItemSpawnEgg() {
026        this(0, 1);
027    }
028
029    public ItemSpawnEgg(Integer meta) {
030        this(meta, 1);
031    }
032
033    public ItemSpawnEgg(Integer meta, int count) {
034        super(SPAWN_EGG, meta, count, "Spawn EntityEgg");
035    }
036
037    @PowerNukkitOnly
038    @Since("1.4.0.0-PN")
039    protected ItemSpawnEgg(int id, Integer meta, int count, String name) {
040        super(id, meta, count, name);
041    }
042
043    @Override
044    public boolean canBeActivated() {
045        return true;
046    }
047
048    @Override
049    public boolean onActivate(Level level, Player player, Block block, Block target, BlockFace face, double fx, double fy, double fz) {
050        if (player.isAdventure()) {
051            return false;
052        }
053
054        FullChunk chunk = level.getChunk((int) block.getX() >> 4, (int) block.getZ() >> 4);
055
056        if (chunk == null) {
057            return false;
058        }
059
060        CompoundTag nbt = new CompoundTag()
061                .putList(new ListTag<DoubleTag>("Pos")
062                        .add(new DoubleTag("", block.getX() + 0.5))
063                        .add(new DoubleTag("", target.getBoundingBox() == null ? block.getY() : target.getBoundingBox().getMaxY() + 0.0001f))
064                        .add(new DoubleTag("", block.getZ() + 0.5)))
065                .putList(new ListTag<DoubleTag>("Motion")
066                        .add(new DoubleTag("", 0))
067                        .add(new DoubleTag("", 0))
068                        .add(new DoubleTag("", 0)))
069                .putList(new ListTag<FloatTag>("Rotation")
070                        .add(new FloatTag("", new Random().nextFloat() * 360))
071                        .add(new FloatTag("", 0)));
072
073        if (this.hasCustomName()) {
074            nbt.putString("CustomName", this.getCustomName());
075        }
076
077        int networkId = getEntityNetworkId();
078        CreatureSpawnEvent ev = new CreatureSpawnEvent(networkId, block, nbt, SpawnReason.SPAWN_EGG);
079        level.getServer().getPluginManager().callEvent(ev);
080
081        if (ev.isCancelled()) {
082            return false;
083        }
084
085        Entity entity = Entity.createEntity(networkId, chunk, nbt);
086
087        if (entity != null) {
088            if (player.isSurvival()) {
089                player.getInventory().decreaseCount(player.getInventory().getHeldItemIndex());
090            }
091            entity.spawnToAll();
092            return true;
093        }
094
095        return false;
096    }
097
098    @Since("1.4.0.0-PN")
099    @PowerNukkitOnly
100    public Item getLegacySpawnEgg() {
101        return Item.get(SPAWN_EGG, getEntityNetworkId(), getCount(), getCompoundTag());
102    }
103
104    @PowerNukkitOnly
105    @Since("1.4.0.0-PN")
106    public int getEntityNetworkId() {
107        return this.meta;
108    }
109}
