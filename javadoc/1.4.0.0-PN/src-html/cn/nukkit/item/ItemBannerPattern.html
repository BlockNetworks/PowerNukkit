001package cn.nukkit.item;
002
003import cn.nukkit.api.PowerNukkitOnly;
004import cn.nukkit.api.Since;
005import cn.nukkit.utils.BannerPattern;
006
007@Since("1.2.1.0-PN")
008@PowerNukkitOnly
009public class ItemBannerPattern extends Item {
010    @PowerNukkitOnly
011    public static final int PATTERN_CREEPER_CHARGE = 0;
012
013    @PowerNukkitOnly
014    public static final int PATTERN_SKULL_CHARGE = 1;
015
016    @PowerNukkitOnly
017    public static final int PATTERN_FLOWER_CHARGE = 2;
018
019    @PowerNukkitOnly
020    public static final int PATTERN_THING = 3;
021
022    @PowerNukkitOnly
023    public static final int PATTERN_FIELD_MASONED = 4;
024
025    @PowerNukkitOnly
026    public static final int PATTERN_BORDURE_INDENTED = 5;
027
028    @PowerNukkitOnly
029    public ItemBannerPattern() {
030        this(0, 1);
031    }
032
033    @PowerNukkitOnly
034    public ItemBannerPattern(Integer meta) {
035        this(meta, 1);
036    }
037
038    @PowerNukkitOnly
039    public ItemBannerPattern(Integer meta, int count) {
040        super(BANNER_PATTERN, meta, count, "Bone");
041        updateName();
042    }
043
044    @PowerNukkitOnly
045    @Since("1.4.0.0-PN")
046    protected ItemBannerPattern(int id, Integer meta, int count, String name) {
047        super(id, meta, count, name);
048    }
049
050    @Override
051    public int getMaxStackSize() {
052        return 1;
053    }
054
055    @Override
056    public void setDamage(Integer meta) {
057        super.setDamage(meta);
058        updateName();
059    }
060    
061    @PowerNukkitOnly
062    @Since("1.4.0.0-PN")
063    public BannerPattern.Type getPatternType() {
064        if (getId() != BANNER_PATTERN) {
065            return BannerPattern.Type.PATTERN_CREEPER;
066        }
067        switch (getDamage()) {
068            default:
069            case PATTERN_CREEPER_CHARGE: return BannerPattern.Type.PATTERN_CREEPER; 
070            case PATTERN_SKULL_CHARGE: return BannerPattern.Type.PATTERN_SKULL; 
071            case PATTERN_FLOWER_CHARGE: return BannerPattern.Type.PATTERN_FLOWER;
072            case PATTERN_THING: return BannerPattern.Type.PATTERN_MOJANG;
073            case PATTERN_FIELD_MASONED: return BannerPattern.Type.PATTERN_BRICK; 
074            case PATTERN_BORDURE_INDENTED: return BannerPattern.Type.PATTERN_CURLY_BORDER;
075        }
076    }
077
078    protected void updateName() {
079        if (getId() != BANNER_PATTERN) {
080            return;
081        }
082        switch (super.meta % 6) {
083            case PATTERN_CREEPER_CHARGE:
084                name = "Creeper Charge Banner Pattern";
085                return;
086            case PATTERN_SKULL_CHARGE:
087                name = "Skull Charge Banner Pattern";
088                return;
089            case PATTERN_FLOWER_CHARGE:
090                name = "Flower Charge Banner Pattern";
091                return;
092            case PATTERN_THING:
093                name = "Thing Banner Pattern";
094                return;
095            case PATTERN_FIELD_MASONED:
096                name = "Field Banner Pattern";
097                return;
098            case PATTERN_BORDURE_INDENTED:
099                name = "Bordure Idented Banner Pattern";
100                return;
101            default:
102                name = "Banner Pattern";
103        }
104    }
105}
