001package cn.nukkit.item;
002
003import cn.nukkit.api.API;
004import cn.nukkit.api.PowerNukkitOnly;
005import cn.nukkit.api.Since;
006import it.unimi.dsi.fastutil.ints.Int2IntMap;
007import it.unimi.dsi.fastutil.ints.Int2ObjectMap;
008import it.unimi.dsi.fastutil.ints.Int2ObjectOpenHashMap;
009
010import javax.annotation.Nonnull;
011import javax.annotation.Nullable;
012import java.util.AbstractMap;
013import java.util.LinkedHashMap;
014import java.util.Map;
015import java.util.OptionalInt;
016import java.util.stream.Collectors;
017
018/**
019 * Responsible for mapping item full ids, item network ids and item namespaced ids between each other.
020 * <ul>
021 * <li>A <b>full id</b> is a combination of <b>item id</b> and <b>item damage</b>. 
022 * The way they are combined may change in future, so you should not combine them by yourself and neither store them
023 * permanently. It's mainly used to preserve backward compatibility with plugins that don't support <em>namespaced ids</em>.
024 * <li>A <b>network id</b> is an id that is used to communicated with the client, it may change between executions of the
025 * same server version depending on how the plugins are setup.
026 * <li>A <b>namespaced id</b> is the new way Mojang saves the ids, a string like <code>minecraft:stone</code>. It may change
027 * in Minecraft updates but tends to be permanent, unless Mojang decides to change them for some random reasons...
028 */
029@Since("1.4.0.0-PN")
030public class RuntimeItemMapping {
031
032    private final Int2IntMap legacyNetworkMap;
033    private final Int2IntMap networkLegacyMap;
034    private final byte[] itemDataPalette;
035
036    private final Map<String, OptionalInt> namespaceNetworkMap;
037    private final Int2ObjectMap<String> networkNamespaceMap;
038
039    @Since("1.4.0.0-PN")
040    public RuntimeItemMapping(byte[] itemDataPalette, Int2IntMap legacyNetworkMap, Int2IntMap networkLegacyMap) {
041        this.itemDataPalette = itemDataPalette;
042        this.legacyNetworkMap = legacyNetworkMap;
043        this.networkLegacyMap = networkLegacyMap;
044        this.legacyNetworkMap.defaultReturnValue(-1);
045        this.networkLegacyMap.defaultReturnValue(-1);
046        this.namespaceNetworkMap = new LinkedHashMap<>();
047        this.networkNamespaceMap = new Int2ObjectOpenHashMap<>();
048    }
049
050    @PowerNukkitOnly
051    @Since("1.4.0.0-PN")
052    @API(definition = API.Definition.INTERNAL, usage = API.Usage.BLEEDING)
053    public RuntimeItemMapping(
054            byte[] itemDataPalette, Int2IntMap legacyNetworkMap, Int2IntMap networkLegacyMap,
055            Map<String, Integer> namespaceNetworkMap, Int2ObjectMap<String> networkNamespaceMap) {
056        this.itemDataPalette = itemDataPalette;
057        this.legacyNetworkMap = legacyNetworkMap;
058        this.networkLegacyMap = networkLegacyMap;
059        this.legacyNetworkMap.defaultReturnValue(-1);
060        this.networkLegacyMap.defaultReturnValue(-1);
061        this.networkNamespaceMap = networkNamespaceMap;
062        this.namespaceNetworkMap = namespaceNetworkMap.entrySet().stream()
063                .map(e-> new AbstractMap.SimpleEntry<>(e.getKey(), OptionalInt.of(e.getValue())))
064                .collect(Collectors.toMap(Map.Entry::getKey, Map.Entry::getValue));
065    }
066
067    /**
068     * Returns the <b>network id</b> based on the <b>full id</b> of the given item.
069     * @param item Given item
070     * @return The <b>network id</b>
071     * @throws IllegalArgumentException If the mapping of the <b>full id</b> to the <b>network id</b> is unknown
072     */
073    @Since("1.4.0.0-PN")
074    public int getNetworkFullId(Item item) {
075        int fullId = RuntimeItems.getFullId(item.getId(), item.hasMeta() ? item.getDamage() : -1);
076        int networkId = this.legacyNetworkMap.get(fullId);
077        if (networkId == -1) {
078            networkId = this.legacyNetworkMap.get(RuntimeItems.getFullId(item.getId(), 0));
079        }
080        if (networkId == -1) {
081            throw new IllegalArgumentException("Unknown item mapping " + item);
082        }
083
084        return networkId;
085    }
086
087    /**
088     * Returns the <b>full id</b> of a given <b>network id</b>.
089     * @param networkId The given <b>network id</b>
090     * @return The <b>full id</b>
091     * @throws IllegalArgumentException If the mapping of the <b>full id</b> to the <b>network id</b> is unknown
092     */
093    @Since("1.4.0.0-PN")
094    public int getLegacyFullId(int networkId) {
095        int fullId = networkLegacyMap.get(networkId);
096        if (fullId == -1) {
097            throw new IllegalArgumentException("Unknown network mapping: " + networkId);
098        }
099        return fullId;
100    }
101
102    @Since("1.4.0.0-PN")
103    public byte[] getItemDataPalette() {
104        return this.itemDataPalette;
105    }
106
107    /**
108     * Returns the <b>namespaced id</b> of a given <b>network id</b>.
109     * @param networkId The given <b>network id</b>
110     * @return The <b>namespace id</b> or {@code null} if it is unknown
111     */
112    @PowerNukkitOnly
113    @Since("1.4.0.0-PN")
114    @Nullable
115    public String getNamespacedIdByNetworkId(int networkId) {
116        return networkNamespaceMap.get(networkId);
117    }
118
119    /**
120     * Returns the <b>network id</b> of a given <b>namespaced id</b>.
121     * @param namespaceId The given <b>namespaced id</b>
122     * @return A <b>network id</b> wrapped in {@link OptionalInt} or an empty {@link OptionalInt} if it is unknown
123     */
124    @PowerNukkitOnly
125    @Since("1.4.0.0-PN")
126    @Nonnull
127    public OptionalInt getNetworkIdByNamespaceId(@Nonnull String namespaceId) {
128        return namespaceNetworkMap.getOrDefault(namespaceId, OptionalInt.empty());
129    }
130
131    /**
132     * Creates a new instance of the respective {@link Item} by the <b>namespaced id</b>.
133     * @param namespaceId The namespaced id
134     * @param amount How many items will be in the stack.
135     * @return The correct {@link Item} instance with the write <b>item id</b> and <b>item damage</b> values.
136     * @throws IllegalArgumentException If there are unknown mappings in the process. 
137     */
138    @PowerNukkitOnly
139    @Since("1.4.0.0-PN")
140    @Nonnull
141    public Item getItemByNamespaceId(@Nonnull String namespaceId, int amount) {
142        int legacyFullId = getLegacyFullId(
143                getNetworkIdByNamespaceId(namespaceId)
144                        .orElseThrow(()-> new IllegalArgumentException("The network id of \""+namespaceId+"\" is unknown"))
145        );
146        if (RuntimeItems.hasData(legacyFullId)) {
147            return Item.get(RuntimeItems.getId(legacyFullId), RuntimeItems.getData(legacyFullId), amount);
148        } else {
149            Item item = Item.get(RuntimeItems.getId(legacyFullId));
150            item.setCount(amount);
151            return item;
152        }
153    }
154}
