001package cn.nukkit.blockentity;
002
003import cn.nukkit.api.PowerNukkitOnly;
004import cn.nukkit.api.Since;
005import cn.nukkit.block.BlockID;
006import cn.nukkit.blockproperty.value.NetherReactorState;
007import cn.nukkit.level.format.FullChunk;
008import cn.nukkit.math.MathHelper;
009import cn.nukkit.nbt.tag.CompoundTag;
010
011/**
012 * This entity allows to manipulate the save state of a nether reactor core, but changing it
013 * will cause no visual change. To see the changes in the world it would be necessary to 
014 * change the block data value to {@code 0 1 or 3} but that is impossible in the recent versions
015 * because Minecraft Bedrock Edition has moved from block data to the block property & block state
016 * system and did not create a block property for the old nether reactor core block, making it 
017 * impossible for the server to tell the client to render the red and dark versions of the block.
018 */
019@PowerNukkitOnly
020@Since("1.4.0.0-PN")
021public class BlockEntityNetherReactor extends BlockEntitySpawnable {
022    private NetherReactorState reactorState;
023    private int progress;
024    
025    @PowerNukkitOnly
026    @Since("1.4.0.0-PN")
027    public BlockEntityNetherReactor(FullChunk chunk, CompoundTag nbt) {
028        super(chunk, nbt);
029    }
030
031    @Override
032    public boolean isBlockEntityValid() {
033        return getLevelBlock().getId() == BlockID.NETHER_REACTOR;
034    }
035
036    @PowerNukkitOnly
037    @Since("1.4.0.0-PN")
038    public NetherReactorState getReactorState() {
039        return reactorState;
040    }
041
042    @PowerNukkitOnly
043    @Since("1.4.0.0-PN")
044    public void setReactorState(NetherReactorState reactorState) {
045        this.reactorState = reactorState;
046    }
047
048    @PowerNukkitOnly
049    @Since("1.4.0.0-PN")
050    public int getProgress() {
051        return progress;
052    }
053
054    @PowerNukkitOnly
055    @Since("1.4.0.0-PN")
056    public void setProgress(int progress) {
057        this.progress = MathHelper.clamp(progress, 0, 900);
058    }
059
060    @Override
061    protected void initBlockEntity() {
062        reactorState = NetherReactorState.READY;
063        if (namedTag.containsShort("Progress")) {
064            progress = (short) namedTag.getShort("Progress");
065        }
066        
067        if (namedTag.containsByte("HasFinished") && namedTag.getBoolean("HasFinished")) {
068            reactorState = NetherReactorState.FINISHED;
069        } else if (namedTag.containsByte("IsInitialized") && namedTag.getBoolean("IsInitialized")) {
070            reactorState = NetherReactorState.INITIALIZED;
071        } else {
072            reactorState = NetherReactorState.READY;
073        }
074        super.initBlockEntity();
075    }
076
077    @Override
078    public void saveNBT() {
079        super.saveNBT();
080        NetherReactorState reactorState = getReactorState();
081        namedTag.putShort("Progress", getProgress());
082        namedTag.putBoolean("HasFinished", reactorState == NetherReactorState.FINISHED);
083        namedTag.putBoolean("IsInitialized", reactorState == NetherReactorState.INITIALIZED);
084    }
085
086    @Override
087    public CompoundTag getSpawnCompound() {
088        NetherReactorState reactorState = getReactorState();
089        return new CompoundTag()
090                .putString("id", BlockEntity.NETHER_REACTOR)
091                .putInt("x", (int) this.x)
092                .putInt("y", (int) this.y)
093                .putInt("z", (int) this.z)
094                .putShort("Progress", getProgress())
095                .putBoolean("HasFinished", reactorState == NetherReactorState.FINISHED)
096                .putBoolean("IsInitialized", reactorState == NetherReactorState.INITIALIZED);
097    }
098}
