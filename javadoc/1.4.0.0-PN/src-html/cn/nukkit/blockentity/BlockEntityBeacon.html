016/**
017 * @author Rover656
018 */
019public class BlockEntityBeacon extends BlockEntitySpawnable {
020
021    public BlockEntityBeacon(FullChunk chunk, CompoundTag nbt) {
022        super(chunk, nbt);
023    }
024
025    @Override
026    protected void initBlockEntity() {
027        if (!namedTag.contains("Lock")) {
028            namedTag.putString("Lock", "");
029        }
030
031        if (!namedTag.contains("Levels")) {
032            namedTag.putInt("Levels", 0);
033        }
034
035        if (!namedTag.contains("Primary")) {
036            namedTag.putInt("Primary", 0);
037        }
038
039        if (!namedTag.contains("Secondary")) {
040            namedTag.putInt("Secondary", 0);
041        }
042
043        scheduleUpdate();
044
045        super.initBlockEntity();
046    }
047
048    @Override
049    public boolean isBlockEntityValid() {
050        int blockID = getBlock().getId();
051        return blockID == Block.BEACON;
052    }
053
054    @Override
055    public CompoundTag getSpawnCompound() {
056        return new CompoundTag()
057                .putString("id", BlockEntity.BEACON)
058                .putInt("x", (int) this.x)
059                .putInt("y", (int) this.y)
060                .putInt("z", (int) this.z)
061                .putString("Lock", this.namedTag.getString("Lock"))
062                .putInt("Levels", this.namedTag.getInt("Levels"))
063                .putInt("primary", this.namedTag.getInt("Primary"))
064                .putInt("secondary", this.namedTag.getInt("Secondary"));
065    }
066
067    private long currentTick = 0;
068
069    @PowerNukkitDifference(info = "Using new method to play sounds", since = "1.4.0.0-PN")
070    @Override
071    public boolean onUpdate() {
072        //Only apply effects every 4 secs
073        if (currentTick++ % 80 != 0) {
074            return true;
075        }
076
077        int oldPowerLevel = this.getPowerLevel();
078        //Get the power level based on the pyramid
079        setPowerLevel(calculatePowerLevel());
080        int newPowerLevel = this.getPowerLevel();
081
082        //Skip beacons that do not have a pyramid or sky access
083        if (newPowerLevel < 1 || !hasSkyAccess()) {
084            if (oldPowerLevel > 0) {
085                this.getLevel().addSound(this, Sound.BEACON_DEACTIVATE);
086            }
087            return true;
088        } else if (oldPowerLevel < 1) {
089            this.getLevel().addSound(this, Sound.BEACON_ACTIVATE);
090        } else {
091            this.getLevel().addSound(this, Sound.BEACON_AMBIENT);
092        }
093
094        //Get all players in game
095        Map<Long, Player> players = this.level.getPlayers();
096
097        //Calculate vars for beacon power
098        int range = 10 + getPowerLevel() * 10;
099        int duration = 9 + getPowerLevel() * 2;
100
101        if (!isPrimaryAllowed(getPrimaryPower(), getPowerLevel())) {
102            return true;
103        }
104
105        for(Map.Entry<Long, Player> entry : players.entrySet()) {
106            Player p = entry.getValue();
107
108            //If the player is in range
109            if (p.distance(this) < range) {
110                Effect e;
111
112                if (getPrimaryPower() != 0) {
113                    //Apply the primary power
114                    e = Effect.getEffect(getPrimaryPower());
115
116                    //Set duration
117                    e.setDuration(duration * 20);
118
119                    //If secondary is selected as the primary too, apply 2 amplification
120                    if (getPowerLevel() == POWER_LEVEL_MAX && getSecondaryPower() == getPrimaryPower()) {
121                        e.setAmplifier(1);
122                    }
123
124                    //Add the effect
125                    p.addEffect(e);
126                }
127
128                //If we have a secondary power as regen, apply it
129                if (getPowerLevel() == POWER_LEVEL_MAX && getSecondaryPower() == Effect.REGENERATION) {
130                    //Get the regen effect
131                    e = Effect.getEffect(Effect.REGENERATION);
132
133                    //Set duration
134                    e.setDuration(duration * 20);
135
136                    //Add effect
137                    p.addEffect(e);
138                }
139            }
140        }
141
142        return true;
143    }
144
145    private static final int POWER_LEVEL_MAX = 4;
146
147    private boolean hasSkyAccess() {
148        int tileX = getFloorX();
149        int tileY = getFloorY();
150        int tileZ = getFloorZ();
151
152        //Check every block from our y coord to the top of the world
153        for (int y = tileY + 1; y <= 255; y++) {
154            int testBlockId = level.getBlockIdAt(tileX, y, tileZ);
155            if (!Block.transparent[testBlockId]) {
156                //There is no sky access
157                return false;
158            }
159        }
160
161        return true;
162    }
163
164    private int calculatePowerLevel() {
165        int tileX = getFloorX();
166        int tileY = getFloorY();
167        int tileZ = getFloorZ();
168
169        //The power level that we're testing for
170        for (int powerLevel = 1; powerLevel <= POWER_LEVEL_MAX; powerLevel++) {
171            int queryY = tileY - powerLevel; //Layer below the beacon block
172
173            for (int queryX = tileX - powerLevel; queryX <= tileX + powerLevel; queryX++) {
174                for (int queryZ = tileZ - powerLevel; queryZ <= tileZ + powerLevel; queryZ++) {
175
176                    int testBlockId = level.getBlockIdAt(queryX, queryY, queryZ);
177                    if (
178                            testBlockId != Block.IRON_BLOCK &&
179                                    testBlockId != Block.GOLD_BLOCK &&
180                                    testBlockId != Block.EMERALD_BLOCK &&
181                                    testBlockId != Block.DIAMOND_BLOCK &&
182                                    testBlockId != Block.NETHERITE_BLOCK
183                    ) {
184                        return powerLevel - 1;
185                    }
186
187                }
188            }
189        }
190
191        return POWER_LEVEL_MAX;
192    }
193
194    public int getPowerLevel() {
195        return namedTag.getInt("Levels");
196    }
197
198    public void setPowerLevel(int level) {
199        int currentLevel = getPowerLevel();
200        if (level != currentLevel) {
201            namedTag.putInt("Levels", level);
202            setDirty();
203            this.spawnToAll();
204        }
205    }
206
207    public int getPrimaryPower() {
208        return namedTag.getInt("Primary");
209    }
210
211    public void setPrimaryPower(int power) {
212        int currentPower = getPrimaryPower();
213        if (power != currentPower) {
214            namedTag.putInt("Primary", power);
215            setDirty();
216            this.spawnToAll();
217        }
218    }
219
220    public int getSecondaryPower() {
221        return namedTag.getInt("Secondary");
222    }
223
224    public void setSecondaryPower(int power) {
225        int currentPower = getSecondaryPower();
226        if (power != currentPower) {
227            namedTag.putInt("Secondary", power);
228            setDirty();
229            this.spawnToAll();
230        }
231    }
232
233    @PowerNukkitDifference(info = "Using new method to play sounds", since = "1.4.0.0-PN")
234    @Override
235    public boolean updateCompoundTag(CompoundTag nbt, Player player) {
236        if (!nbt.getString("id").equals(BlockEntity.BEACON)) {
237            return false;
238        }
239
240        int primary = nbt.getInt("primary");
241        if (!isPrimaryAllowed(primary, this.getPowerLevel())) {
242            return false;
243        }
244
245        int secondary = nbt.getInt("secondary");
246        if (secondary != 0 && secondary != primary && secondary != Effect.REGENERATION) {
247            return false;
248        }
249
250        this.setPrimaryPower(primary);
251        this.setSecondaryPower(secondary);
252
253        this.getLevel().addSound(this, Sound.BEACON_POWER);
254
255        BeaconInventory inv = (BeaconInventory)player.getWindowById(Player.BEACON_WINDOW_ID);
256
257        inv.setItem(0, new ItemBlock(Block.get(BlockID.AIR), 0, 0));
258        return true;
259    }
260
261    private static boolean isPrimaryAllowed(int primary, int powerLevel) {
262        return ((primary == Effect.SPEED || primary == Effect.HASTE) && powerLevel >= 1) ||
263                ((primary == Effect.DAMAGE_RESISTANCE || primary == Effect.JUMP) && powerLevel >= 2) ||
264                (primary == Effect.STRENGTH && powerLevel >= 3);
265    }
