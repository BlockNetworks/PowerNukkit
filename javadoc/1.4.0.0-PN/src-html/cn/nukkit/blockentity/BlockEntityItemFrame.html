001package cn.nukkit.blockentity;
002
003import cn.nukkit.Player;
004import cn.nukkit.api.PowerNukkitOnly;
005import cn.nukkit.api.Since;
006import cn.nukkit.block.Block;
007import cn.nukkit.block.BlockID;
008import cn.nukkit.entity.item.EntityItem;
009import cn.nukkit.event.block.ItemFrameDropItemEvent;
010import cn.nukkit.item.*;
011import cn.nukkit.level.Level;
012import cn.nukkit.level.format.FullChunk;
013import cn.nukkit.nbt.NBTIO;
014import cn.nukkit.nbt.tag.CompoundTag;
015import cn.nukkit.network.protocol.LevelEventPacket;
016
017import javax.annotation.Nullable;
018import java.util.concurrent.ThreadLocalRandom;
019
020/**
021 * @author Pub4Game
022 * @since 03.07.2016
023 */
024public class BlockEntityItemFrame extends BlockEntitySpawnable {
025
026    public BlockEntityItemFrame(FullChunk chunk, CompoundTag nbt) {
027        super(chunk, nbt);
028    }
029
030    @Override
031    protected void initBlockEntity() {
032        if (!namedTag.contains("Item")) {
033            namedTag.putCompound("Item", NBTIO.putItemHelper(new ItemBlock(Block.get(BlockID.AIR))));
034        }
035        if (!namedTag.contains("ItemRotation")) {
036            namedTag.putByte("ItemRotation", 0);
037        }
038        if (!namedTag.contains("ItemDropChance")) {
039            namedTag.putFloat("ItemDropChance", 1.0f);
040        }
041
042        this.level.updateComparatorOutputLevel(this);
043
044        super.initBlockEntity();
045    }
046
047    @Override
048    public String getName() {
049        return "Item Frame";
050    }
051
052    @Override
053    public boolean isBlockEntityValid() {
054        return this.getBlock().getId() == Block.ITEM_FRAME_BLOCK;
055    }
056
057    public int getItemRotation() {
058        return this.namedTag.getByte("ItemRotation");
059    }
060
061    public void setItemRotation(int itemRotation) {
062        this.namedTag.putByte("ItemRotation", itemRotation);
063        this.level.updateComparatorOutputLevel(this);
064        this.setDirty();
065    }
066
067    public Item getItem() {
068        CompoundTag NBTTag = this.namedTag.getCompound("Item");
069        return NBTIO.getItemHelper(NBTTag);
070    }
071
072    public void setItem(Item item) {
073        this.setItem(item, true);
074    }
075
076    public void setItem(Item item, boolean setChanged) {
077        this.namedTag.putCompound("Item", NBTIO.putItemHelper(item));
078        if (setChanged) {
079            this.setDirty();
080        }
081
082        this.level.updateComparatorOutputLevel(this);
083    }
084
085    public float getItemDropChance() {
086        return this.namedTag.getFloat("ItemDropChance");
087    }
088
089    public void setItemDropChance(float chance) {
090        this.namedTag.putFloat("ItemDropChance", chance);
091    }
092
093    public void setDirty() {
094        this.spawnToAll();
095        super.setDirty();
096    }
097
098    @Override
099    public CompoundTag getSpawnCompound() {
100        if (!this.namedTag.contains("Item")) {
101            this.setItem(new ItemBlock(Block.get(BlockID.AIR)), false);
102        }
103        Item item = getItem();
104        CompoundTag tag = new CompoundTag()
105                .putString("id", BlockEntity.ITEM_FRAME)
106                .putInt("x", (int) this.x)
107                .putInt("y", (int) this.y)
108                .putInt("z", (int) this.z);
109
110        if (!item.isNull()) {
111            CompoundTag itemTag = NBTIO.putItemHelper(item);
112            int networkFullId = item.getNetworkFullId();
113            int networkDamage = (networkFullId & 0x1) == 0x1? 0 : item.getDamage();
114            String namespacedId = RuntimeItems.getRuntimeMapping().getNamespacedIdByNetworkId(
115                    RuntimeItems.getNetworkId(networkFullId)
116            );
117            if (namespacedId != null) {
118                itemTag.remove("id");
119                itemTag.putShort("Damage", networkDamage);
120                itemTag.putString("Name", namespacedId);
121            }
122            tag.putCompound("Item", itemTag)
123                    .putByte("ItemRotation", this.getItemRotation());
124        }
125        return tag;
126    }
127
128    public int getAnalogOutput() {
129        return this.getItem() == null || this.getItem().getId() == 0 ? 0 : this.getItemRotation() % 8 + 1;
130    }
131
132    @Since("1.4.0.0-PN")
133    public boolean dropItem(Player player) {
134        Item before = this.getItem();
135        if (before == null || before.isNull()) {
136            return false;
137        }
138        EntityItem drop = dropItemAndGetEntity(player);
139        if (drop != null) {
140            return true;
141        }
142        Item after = this.getItem();
143        return after == null || after.isNull();
144    }
145
146    @PowerNukkitOnly
147    @Since("1.4.0.0-PN")
148    @Nullable
149    public EntityItem dropItemAndGetEntity(@Nullable Player player) {
150        Level level = getValidLevel();
151        Item drop = getItem();
152        if (drop.isNull()) {
153            if (player != null) {
154                spawnTo(player);
155            }
156            return null;
157        }
158
159        ItemFrameDropItemEvent event = new ItemFrameDropItemEvent(player, getLevelBlock(), this, drop);
160        level.getServer().getPluginManager().callEvent(event);
161        if (event.isCancelled()) {
162            if (player != null) {
163                spawnTo(player);
164            }
165            return null;
166        }
167
168        EntityItem itemEntity = null;
169        if (this.getItemDropChance() > ThreadLocalRandom.current().nextFloat()) {
170            itemEntity = level.dropAndGetItem(add(0.5, 0.25, 0.5), drop);
171            if (itemEntity == null) {
172                if (player != null) {
173                    spawnTo(player);
174                }
175                return null;
176            }
177        }
178        
179        setItem(MinecraftItemID.AIR.get(0), true);
180        setItemRotation(0);
181        spawnToAll();
182        level.addLevelEvent(this, LevelEventPacket.EVENT_SOUND_ITEM_FRAME_REMOVED);
183        
184        return itemEntity;
185    }
186}
