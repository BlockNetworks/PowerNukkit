001package cn.nukkit.blockentity;
002
003import cn.nukkit.api.PowerNukkitDifference;
004import cn.nukkit.block.BlockAir;
005import cn.nukkit.block.BlockID;
006import cn.nukkit.utils.RedstoneComponent;
007import cn.nukkit.item.Item;
008import cn.nukkit.item.ItemBlock;
009import cn.nukkit.level.format.FullChunk;
010import cn.nukkit.nbt.NBTIO;
011import cn.nukkit.nbt.tag.CompoundTag;
012import cn.nukkit.nbt.tag.IntTag;
013
014public class BlockEntityLectern extends BlockEntitySpawnable {
015
016    private int totalPages;
017
018    public BlockEntityLectern(FullChunk chunk, CompoundTag nbt) {
019        super(chunk, nbt);
020    }
021
022    @Override
023    protected void initBlockEntity() {
024        if (!(this.namedTag.get("book") instanceof CompoundTag)) {
025            this.namedTag.remove("book");
026        }
027
028        if (!(this.namedTag.get("page") instanceof IntTag)) {
029            this.namedTag.remove("page");
030        }
031
032        updateTotalPages();
033    }
034
035    @Override
036    public CompoundTag getSpawnCompound() {
037        CompoundTag c = new CompoundTag()
038                .putString("id", BlockEntity.LECTERN)
039                .putInt("x", (int) this.x)
040                .putInt("y", (int) this.y)
041                .putInt("z", (int) this.z)
042                .putBoolean("isMovable", this.movable);
043
044        Item book = getBook();
045        if (book.getId() != Item.AIR) {
046            c.putCompound("book", NBTIO.putItemHelper(book));
047            c.putBoolean("hasBook", true);
048            c.putInt("page", getRawPage());
049            c.putInt("totalPages", totalPages);
050        } else {
051            c.putBoolean("hasBook", false);
052        }
053
054        return c;
055    }
056
057    @Override
058    public boolean isBlockEntityValid() {
059        return getBlock().getId() == BlockID.LECTERN;
060    }
061
062    @Override
063    public void onBreak() {
064        level.dropItem(this, getBook());
065    }
066
067    public boolean hasBook() {
068        return this.namedTag.contains("book") && this.namedTag.get("book") instanceof CompoundTag;
069    }
070
071    public Item getBook() {
072        if (!hasBook()) {
073            return new ItemBlock(new BlockAir(), 0, 0);
074        } else {
075            return NBTIO.getItemHelper(this.namedTag.getCompound("book"));
076        }
077    }
078
079    public void setBook(Item item) {
080        if (item.getId() == Item.WRITTEN_BOOK || item.getId() == Item.BOOK_AND_QUILL) {
081            this.namedTag.putCompound("book", NBTIO.putItemHelper(item));
082        } else {
083            this.namedTag.remove("book");
084            this.namedTag.remove("page");
085        }
086        updateTotalPages();
087    }
088
089    public int getLeftPage() {
090        return (getRawPage() * 2) + 1;
091    }
092
093    public int getRightPage() {
094        return getLeftPage() + 1;
095    }
096
097    public void setLeftPage(int newLeftPage) {
098        setRawPage((newLeftPage - 1) /2);
099    }
100
101    public void setRightPage(int newRightPage) {
102        setLeftPage(newRightPage -1);
103    }
104
105    public void setRawPage(int page) {
106        this.namedTag.putInt("page", Math.min(page, totalPages));
107        this.getLevel().updateAround(this);
108    }
109
110    public int getRawPage() {
111        return this.namedTag.getInt("page");
112    }
113
114    public int getTotalPages() {
115        return totalPages;
116    }
117
118    @PowerNukkitDifference(info = "Use RedstoneComponent for redstone update.", since = "1.4.0.0-PN")
119    private void updateTotalPages() {
120        Item book = getBook();
121        if (book.getId() == Item.AIR || !book.hasCompoundTag()) {
122            totalPages = 0;
123        } else {
124            totalPages = book.getNamedTag().getList("pages", CompoundTag.class).size();
125        }
126        RedstoneComponent.updateAroundRedstone(this);
127    }
128}
