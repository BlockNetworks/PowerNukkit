102    public static byte[] writeMetadata(EntityMetadata metadata) {
103        BinaryStream stream = new BinaryStream();
104        Map<Integer, EntityData> map = metadata.getMap();
105        stream.putUnsignedVarInt(map.size());
106        for (int id : map.keySet()) {
107            EntityData d = map.get(id);
108            stream.putUnsignedVarInt(id);
109            stream.putUnsignedVarInt(d.getType());
110            switch (d.getType()) {
111                case Entity.DATA_TYPE_BYTE:
112                    stream.putByte(((ByteEntityData) d).getData().byteValue());
113                    break;
114                case Entity.DATA_TYPE_SHORT:
115                    stream.putLShort(((ShortEntityData) d).getData());
116                    break;
117                case Entity.DATA_TYPE_INT:
118                    stream.putVarInt(((IntEntityData) d).getData());
119                    break;
120                case Entity.DATA_TYPE_FLOAT:
121                    stream.putLFloat(((FloatEntityData) d).getData());
122                    break;
123                case Entity.DATA_TYPE_STRING:
124                    String s = ((StringEntityData) d).getData();
125                    stream.putUnsignedVarInt(s.getBytes(StandardCharsets.UTF_8).length);
126                    stream.put(s.getBytes(StandardCharsets.UTF_8));
127                    break;
128                case Entity.DATA_TYPE_NBT:
129                    NBTEntityData slot = (NBTEntityData) d;
130                    try {
131                        stream.put(NBTIO.write(slot.getData(), ByteOrder.LITTLE_ENDIAN, true));
132                    } catch (IOException e) {
133                        throw new UncheckedIOException(e);
134                    }
135                    break;
136                case Entity.DATA_TYPE_POS:
137                    IntPositionEntityData pos = (IntPositionEntityData) d;
138                    stream.putVarInt(pos.x);
139                    stream.putVarInt(pos.y);
140                    stream.putVarInt(pos.z);
141                    break;
142                case Entity.DATA_TYPE_LONG:
143                    stream.putVarLong(((LongEntityData) d).getData());
144                    break;
145                case Entity.DATA_TYPE_VECTOR3F:
146                    Vector3fEntityData v3data = (Vector3fEntityData) d;
147                    stream.putLFloat(v3data.x);
148                    stream.putLFloat(v3data.y);
149                    stream.putLFloat(v3data.z);
150                    break;
151            }
152        }
153        return stream.getBuffer();
154    }

156    public static EntityMetadata readMetadata(byte[] payload) {
157        BinaryStream stream = new BinaryStream();
158        stream.setBuffer(payload);
159        long count = stream.getUnsignedVarInt();
160        EntityMetadata m = new EntityMetadata();
161        for (int i = 0; i < count; i++) {
162            int key = (int) stream.getUnsignedVarInt();
163            int type = (int) stream.getUnsignedVarInt();
164            EntityData value = null;
165            switch (type) {
166                case Entity.DATA_TYPE_BYTE:
167                    value = new ByteEntityData(key, stream.getByte());
168                    break;
169                case Entity.DATA_TYPE_SHORT:
170                    value = new ShortEntityData(key, stream.getLShort());
171                    break;
172                case Entity.DATA_TYPE_INT:
173                    value = new IntEntityData(key, stream.getVarInt());
174                    break;
175                case Entity.DATA_TYPE_FLOAT:
176                    value = new FloatEntityData(key, stream.getLFloat());
177                    break;
178                case Entity.DATA_TYPE_STRING:
179                    value = new StringEntityData(key, stream.getString());
180                    break;
181                case Entity.DATA_TYPE_NBT:
182                    int offset = stream.getOffset();
183                    FastByteArrayInputStream fbais = new FastByteArrayInputStream(stream.get());
184                    try {
185                        CompoundTag tag = NBTIO.read(fbais, ByteOrder.LITTLE_ENDIAN, true);
186                        value = new NBTEntityData(key, tag);
187                    } catch (IOException e) {
188                        throw new UncheckedIOException(e);
189                    }
190                    stream.setOffset(offset + (int) fbais.position());
191                    break;
192                case Entity.DATA_TYPE_POS:
193                    BlockVector3 v3 = stream.getSignedBlockPosition();
194                    value = new IntPositionEntityData(key, v3.x, v3.y, v3.z);
195                    break;
196                case Entity.DATA_TYPE_LONG:
197                    value = new LongEntityData(key, stream.getVarLong());
198                    break;
199                case Entity.DATA_TYPE_VECTOR3F:
200                    value = new Vector3fEntityData(key, stream.getVector3f());
201                    break;
202            }
203            if (value != null) m.put(value);
204        }
205        return m;
206    }

398        byte[] newBytes = new byte[bytes.length];

409    public static String bytesToHexString(byte[] src, boolean blank) {
410        StringBuilder stringBuilder = new StringBuilder();
411        if (src == null || src.length <= 0) {
412            return null;
413        }
414
415        for (byte b : src) {
416            if (!(stringBuilder.length() == 0) && blank) {
417                stringBuilder.append(" ");
418            }
419            int v = b & 0xFF;
420            String hv = Integer.toHexString(v);
421            if (hv.length() < 2) {
422                stringBuilder.append(0);
423            }
424            stringBuilder.append(hv);
425        }
426        return stringBuilder.toString().toUpperCase();
427    }

429    public static byte[] hexStringToBytes(String hexString) {
430        if (hexString == null || hexString.equals("")) {
431            return null;
432        }
433        String str = "0123456789ABCDEF";
434        hexString = hexString.toUpperCase().replace(" ", "");
435        int length = hexString.length() / 2;
436        char[] hexChars = hexString.toCharArray();
437        byte[] d = new byte[length];
438        for (int i = 0; i < length; i++) {
439            int pos = i * 2;
440            d[i] = (byte) (((byte) str.indexOf(hexChars[pos]) << 4) | ((byte) str.indexOf(hexChars[pos + 1])));
441        }
442        return d;
443    }

454    public static byte[][] splitBytes(byte[] bytes, int chunkSize) {
455        byte[][] splits = new byte[(bytes.length + chunkSize - 1) / chunkSize][chunkSize];
456        int chunks = 0;
457
458        for (int i = 0; i < bytes.length; i += chunkSize) {
459            if ((bytes.length - i) > chunkSize) {
460                splits[chunks] = Arrays.copyOfRange(bytes, i, i + chunkSize);
461            } else {
462                splits[chunks] = Arrays.copyOfRange(bytes, i, bytes.length);
463            }
464            chunks++;
465        }
466
467        return splits;
468    }

470    public static byte[] appendBytes(byte[][] bytes) {
471        int length = 0;
472        for (byte[] b : bytes) {
473            length += b.length;
474        }
475
476        byte[] appendedBytes = new byte[length];
477        int index = 0;
478        for (byte[] b : bytes) {
479            System.arraycopy(b, 0, appendedBytes, index, b.length);
480            index += b.length;
481        }
482        return appendedBytes;
483    }

485    public static byte[] appendBytes(byte byte1, byte[]... bytes2) {
486        int length = 1;
487        for (byte[] bytes : bytes2) {
488            length += bytes.length;
489        }
490        ByteBuffer buffer = ByteBuffer.allocate(length);
491        buffer.put(byte1);
492        for (byte[] bytes : bytes2) {
493            buffer.put(bytes);
494        }
495        return buffer.array();
496    }

498    public static byte[] appendBytes(byte[] bytes1, byte[]... bytes2) {
499        int length = bytes1.length;
500        for (byte[] bytes : bytes2) {
501            length += bytes.length;
502        }
503
504        byte[] appendedBytes = new byte[length];
505        System.arraycopy(bytes1, 0, appendedBytes, 0, bytes1.length);
506        int index = bytes1.length;
507
508        for (byte[] b : bytes2) {
509            System.arraycopy(b, 0, appendedBytes, index, b.length);
510            index += b.length;
511        }
512        return appendedBytes;
513    }
