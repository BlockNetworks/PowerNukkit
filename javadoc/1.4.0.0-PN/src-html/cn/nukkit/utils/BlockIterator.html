001package cn.nukkit.utils;
002
003import cn.nukkit.block.Block;
004import cn.nukkit.level.Level;
005import cn.nukkit.math.BlockFace;
006import cn.nukkit.math.Vector3;
007
008import java.util.Iterator;
009
010/**
011 * @author MagicDroidX (Nukkit Project)
012 */
013public class BlockIterator implements Iterator<Block> {
014    private final Level level;
015    private final int maxDistance;
016
017    private static final int gridSize = 1 << 24;
018
019    private boolean end = false;
020
021    private final Block[] blockQueue;
022    private int currentBlock = 0;
023
024    private Block currentBlockObject = null;
025    private int currentDistance;
026    private int maxDistanceInt = 0;
027
028    private int secondError;
029    private int thirdError;
030
031    private final int secondStep;
032    private final int thirdStep;
033
034    private BlockFace mainFace;
035    private BlockFace secondFace;
036    private BlockFace thirdFace;
037
038    public BlockIterator(Level level, Vector3 start, Vector3 direction) {
039        this(level, start, direction, 0);
040    }
041
042    public BlockIterator(Level level, Vector3 start, Vector3 direction, double yOffset) {
043        this(level, start, direction, yOffset, 0);
044    }
045
046    public BlockIterator(Level level, Vector3 start, Vector3 direction, double yOffset, int maxDistance) {
047        this.level = level;
048        this.maxDistance = maxDistance;
049        this.blockQueue = new Block[3];
050
051        Vector3 startClone = new Vector3(start.x, start.y, start.z);
052        startClone.y += yOffset;
053
054        this.currentDistance = 0;
055
056        double mainDirection = 0;
057        double secondDirection = 0;
058        double thirdDirection = 0;
059
060        double mainPosition = 0;
061        double secondPosition = 0;
062        double thirdPosition = 0;
063
064        Vector3 pos = new Vector3(startClone.x, startClone.y, startClone.z);
065        Block startBlock = this.level.getBlock(new Vector3(Math.floor(pos.x), Math.floor(pos.y), Math.floor(pos.z)));
066
067        if (this.getXLength(direction) > mainDirection) {
068            this.mainFace = this.getXFace(direction);
069            mainDirection = this.getXLength(direction);
070            mainPosition = this.getXPosition(direction, startClone, startBlock);
071
072            this.secondFace = this.getYFace(direction);
073            secondDirection = this.getYLength(direction);
074            secondPosition = this.getYPosition(direction, startClone, startBlock);
075
076            this.thirdFace = this.getZFace(direction);
077            thirdDirection = this.getZLength(direction);
078            thirdPosition = this.getZPosition(direction, startClone, startBlock);
079        }
080        if (this.getYLength(direction) > mainDirection) {
081            this.mainFace = this.getYFace(direction);
082            mainDirection = this.getYLength(direction);
083            mainPosition = this.getYPosition(direction, startClone, startBlock);
084
085            this.secondFace = this.getZFace(direction);
086            secondDirection = this.getZLength(direction);
087            secondPosition = this.getZPosition(direction, startClone, startBlock);
088
089            this.thirdFace = this.getXFace(direction);
090            thirdDirection = this.getXLength(direction);
091            thirdPosition = this.getXPosition(direction, startClone, startBlock);
092        }
093        if (this.getZLength(direction) > mainDirection) {
094            this.mainFace = this.getZFace(direction);
095            mainDirection = this.getZLength(direction);
096            mainPosition = this.getZPosition(direction, startClone, startBlock);
097
098            this.secondFace = this.getXFace(direction);
099            secondDirection = this.getXLength(direction);
100            secondPosition = this.getXPosition(direction, startClone, startBlock);
101
102            this.thirdFace = this.getYFace(direction);
103            thirdDirection = this.getYLength(direction);
104            thirdPosition = this.getYPosition(direction, startClone, startBlock);
105        }
106
107        double d = mainPosition / mainDirection;
108        double secondd = secondPosition - secondDirection * d;
109        double thirdd = thirdPosition - thirdDirection * d;
110
111        this.secondError = (int) Math.floor(secondd * gridSize);
112        this.secondStep = (int) Math.round(secondDirection / mainDirection * gridSize);
113        this.thirdError = (int) Math.floor(thirdd * gridSize);
114        this.thirdStep = (int) Math.round(thirdDirection / mainDirection * gridSize);
115
116        if (this.secondError + this.secondStep <= 0) {
117            this.secondError = -this.secondStep + 1;
118        }
119
120        if (this.thirdError + this.thirdStep <= 0) {
121            this.thirdError = -this.thirdStep + 1;
122        }
123
124        Block lastBlock = startBlock.getSide(this.mainFace.getOpposite());
125
126        if (this.secondError < 0) {
127            this.secondError += gridSize;
128            lastBlock = lastBlock.getSide(this.secondFace.getOpposite());
129        }
130
131        if (this.thirdError < 0) {
132            this.thirdError += gridSize;
133            lastBlock = lastBlock.getSide(this.thirdFace.getOpposite());
134        }
135
136        this.secondError -= gridSize;
137        this.thirdError -= gridSize;
138
139        this.blockQueue[0] = lastBlock;
140
141        this.currentBlock = -1;
142
143        this.scan();
144
145        boolean startBlockFound = false;
146
147        for (int cnt = this.currentBlock; cnt >= 0; --cnt) {
148            if (this.blockEquals(this.blockQueue[cnt], startBlock)) {
149                this.currentBlock = cnt;
150                startBlockFound = true;
151                break;
152            }
153        }
154
155        if (!startBlockFound) {
156            throw new IllegalStateException("Start block missed in BlockIterator");
157        }
158
159        this.maxDistanceInt = (int) Math.round(maxDistance / (Math.sqrt(mainDirection * mainDirection + secondDirection * secondDirection + thirdDirection * thirdDirection) / mainDirection));
160    }
161
162    private boolean blockEquals(Block a, Block b) {
163        return a.x == b.x && a.y == b.y && a.z == b.z;
164    }
165
166    private BlockFace getXFace(Vector3 direction) {
167        return ((direction.x) > 0) ? BlockFace.EAST : BlockFace.WEST;
168    }
169
170    private BlockFace getYFace(Vector3 direction) {
171        return ((direction.y) > 0) ? BlockFace.UP : BlockFace.DOWN;
172    }
173
174    private BlockFace getZFace(Vector3 direction) {
175        return ((direction.z) > 0) ? BlockFace.SOUTH : BlockFace.NORTH;
176    }
177
178    private double getXLength(Vector3 direction) {
179        return Math.abs(direction.x);
180    }
181
182    private double getYLength(Vector3 direction) {
183        return Math.abs(direction.y);
184    }
185
186    private double getZLength(Vector3 direction) {
187        return Math.abs(direction.z);
188    }
189
190    private double getPosition(double direction, double position, double blockPosition) {
191        return direction > 0 ? (position - blockPosition) : (blockPosition + 1 - position);
192    }
193
194    private double getXPosition(Vector3 direction, Vector3 position, Block block) {
195        return this.getPosition(direction.x, position.x, block.x);
196    }
197
198    private double getYPosition(Vector3 direction, Vector3 position, Block block) {
199        return this.getPosition(direction.y, position.y, block.y);
200    }
201
202    private double getZPosition(Vector3 direction, Vector3 position, Block block) {
203        return this.getPosition(direction.z, position.z, block.z);
204    }
205
206    @Override
207    public Block next() {
208        this.scan();
209
210        if (this.currentBlock <= -1) {
211            throw new IndexOutOfBoundsException();
212        } else {
213            this.currentBlockObject = this.blockQueue[this.currentBlock--];
214        }
215        return this.currentBlockObject;
216    }
217
218    @Override
219    public boolean hasNext() {
220        this.scan();
221        return this.currentBlock != -1;
222    }
223
224    private void scan() {
225        if (this.currentBlock >= 0) {
226            return;
227        }
228
229        if (this.maxDistance != 0 && this.currentDistance > this.maxDistanceInt) {
230            this.end = true;
231            return;
232        }
233
234        if (this.end) {
235            return;
236        }
237
238        ++this.currentDistance;
239
240        this.secondError += this.secondStep;
241        this.thirdError += this.thirdStep;
242
243        if (this.secondError > 0 && this.thirdError > 0) {
244            this.blockQueue[2] = this.blockQueue[0].getSide(this.mainFace);
245
246            if ((this.secondStep * this.thirdError) < (this.thirdStep * this.secondError)) {
247                this.blockQueue[1] = this.blockQueue[2].getSide(this.secondFace);
248                this.blockQueue[0] = this.blockQueue[1].getSide(this.thirdFace);
249            } else {
250                this.blockQueue[1] = this.blockQueue[2].getSide(this.thirdFace);
251                this.blockQueue[0] = this.blockQueue[1].getSide(this.secondFace);
252            }
253
254            this.thirdError -= gridSize;
255            this.secondError -= gridSize;
256            this.currentBlock = 2;
257        } else if (this.secondError > 0) {
258            this.blockQueue[1] = this.blockQueue[0].getSide(this.mainFace);
259            this.blockQueue[0] = this.blockQueue[1].getSide(this.secondFace);
260            this.secondError -= gridSize;
261            this.currentBlock = 1;
262        } else if (this.thirdError > 0) {
263            this.blockQueue[1] = this.blockQueue[0].getSide(this.mainFace);
264            this.blockQueue[0] = this.blockQueue[1].getSide(this.thirdFace);
265            this.thirdError -= gridSize;
266            this.currentBlock = 1;
267        } else {
268            this.blockQueue[0] = this.blockQueue[0].getSide(this.mainFace);
269            this.currentBlock = 0;
270        }
271    }
272}
