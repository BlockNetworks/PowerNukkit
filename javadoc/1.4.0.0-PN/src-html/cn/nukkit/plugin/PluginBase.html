205    public boolean saveResource(String filename, String outputName, boolean replace) {
206        Preconditions.checkArgument(filename != null && outputName != null, "Filename can not be null!");
207        Preconditions.checkArgument(filename.trim().length() != 0 && outputName.trim().length() != 0, "Filename can not be empty!");
208        File out = new File(dataFolder, outputName);
209        if (!out.exists() || replace) {
210            try (InputStream resource = getResource(filename)) {
211                if (resource != null) {
212                    File outFolder = out.getParentFile();
213                    if (!outFolder.exists()) {
214                        outFolder.mkdirs();
215                    }
216                    Utils.writeFile(out, resource);
217                    return true;
218                }
219            } catch (IOException e) {
220                log.error("Error while saving resource {}, to {} (replace: {}, plugin:{})", filename, outputName, replace, getDescription().getName(), e);
221            }
222        }
223        return false;
224    }

249    public void reloadConfig() {
250        this.config = new Config(this.configFile);
251        InputStream configStream = this.getResource("config.yml");
252        if (configStream != null) {
253            DumperOptions dumperOptions = new DumperOptions();
254            dumperOptions.setDefaultFlowStyle(DumperOptions.FlowStyle.BLOCK);
255            Yaml yaml = new Yaml(dumperOptions);
256            try {
257                this.config.setDefault(yaml.loadAs(Utils.readFile(this.configFile), LinkedHashMap.class));
258            } catch (IOException e) {
259                log.error("Error while reloading configs for the plugin {}", getDescription().getName(), e);
260            }
261        }
