077    @Override
078    public boolean process() {
079        NukkitRakNetSession session;
080        while ((session = this.sessionCreationQueue.poll()) != null) {
081            InetSocketAddress address = session.raknet.getAddress();
082            PlayerCreationEvent ev = new PlayerCreationEvent(this, Player.class, Player.class, null, address);
083            this.server.getPluginManager().callEvent(ev);
084            Class<? extends Player> clazz = ev.getPlayerClass();
085            try {
086                Constructor<? extends Player> constructor = clazz.getConstructor(SourceInterface.class, Long.class, InetSocketAddress.class);
087                Player player = constructor.newInstance(this, ev.getClientId(), ev.getSocketAddress());
088                this.server.addPlayer(address, player);
089                session.player = player;
090                this.sessions.put(address, session);
091            } catch (NoSuchMethodException | InvocationTargetException | InstantiationException | IllegalAccessException e) {
092                log.error("Error while creating the player class {}", clazz, e);
093            }
094        }
095        Iterator<NukkitRakNetSession> iterator = this.sessions.values().iterator();
096        while (iterator.hasNext()) {
097            NukkitRakNetSession nukkitSession = iterator.next();
098            Player player = nukkitSession.player;
099            if (nukkitSession.disconnectReason != null) {
100                player.close(player.getLeaveMessage(), nukkitSession.disconnectReason, false);
101                iterator.remove();
102                continue;
103            }
104            DataPacket packet;
105            while ((packet = nukkitSession.inbound.poll()) != null) {
106                try {
107                    nukkitSession.player.handleDataPacket(packet);
108                } catch (Exception e) {
109                    log.error(new FormattedMessage("An error occurred whilst handling {} for {}", new Object[] {packet.getClass().getSimpleName(), nukkitSession.player.getName()}, e));
110                }
111            }
112        }
113        return true;
114    }

275        private void sendOutbound() {
276            List<DataPacket> toBatch = new ObjectArrayList<>();
277            DataPacket packet;
278            while ((packet = this.outbound.poll()) != null) {
279                if (packet.pid() == ProtocolInfo.BATCH_PACKET) {
280                    if (!toBatch.isEmpty()) {
281                        this.sendPackets(toBatch);
282                        toBatch.clear();
283                    }
284                    this.sendPacket(((BatchPacket) packet).payload);
285                } else {
286                    toBatch.add(packet);
287                }
288            }
289            if (!toBatch.isEmpty()) {
290                this.sendPackets(toBatch);
291            }
292        }

294        private void sendPackets(Collection<DataPacket> packets) {
295            BinaryStream batched = new BinaryStream();
296            for (DataPacket packet : packets) {
297                Preconditions.checkArgument(!(packet instanceof BatchPacket), "Cannot batch BatchPacket");
298                Preconditions.checkState(packet.isEncoded, "Packet should have already been encoded");
299                byte[] buf = packet.getBuffer();
300                batched.putUnsignedVarInt(buf.length);
301                batched.put(buf);
302            }
303            try {
304                this.sendPacket(Network.deflateRaw(batched.getBuffer(), network.getServer().networkCompressionLevel));
305            } catch (IOException e) {
306                log.error("Unable to compress batched packets", e);
307            }
308        }
309
310        private void sendPacket(byte[] payload) {
311            ByteBuf byteBuf = ByteBufAllocator.DEFAULT.ioBuffer(1 + payload.length);
312            byteBuf.writeByte(254);
313            byteBuf.writeBytes(payload);
314            this.raknet.send(byteBuf);
315        }
316
317        private void sendPacketImmediately(DataPacket packet) {
318            BinaryStream batched = new BinaryStream();
319            Preconditions.checkArgument(!(packet instanceof BatchPacket), "Cannot batch BatchPacket");
320            Preconditions.checkState(packet.isEncoded, "Packet should have already been encoded");
321            byte[] buf = packet.getBuffer();
322            batched.putUnsignedVarInt(buf.length);
323            batched.put(buf);
324            try {
325                byte[] payload = Network.deflateRaw(batched.getBuffer(), network.getServer().networkCompressionLevel);
326                ByteBuf byteBuf = ByteBufAllocator.DEFAULT.ioBuffer(1 + payload.length);
327                byteBuf.writeByte(254);
328                byteBuf.writeBytes(payload);
329                this.raknet.send(byteBuf, RakNetPriority.IMMEDIATE);
330            } catch (Exception e) {
331                log.error("Error occured while sending a packet immediately", e);
332            }
333        }
