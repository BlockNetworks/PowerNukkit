001package cn.nukkit.block;
002
003import cn.nukkit.Player;
004import cn.nukkit.api.PowerNukkitDifference;
005import cn.nukkit.api.PowerNukkitOnly;
006import cn.nukkit.api.Since;
007import cn.nukkit.inventory.StonecutterInventory;
008import cn.nukkit.item.Item;
009import cn.nukkit.item.ItemBlock;
010import cn.nukkit.item.ItemTool;
011import cn.nukkit.math.BlockFace;
012import cn.nukkit.network.protocol.types.ContainerIds;
013import cn.nukkit.utils.BlockColor;
014import cn.nukkit.utils.Faceable;
015
016import javax.annotation.Nonnull;
017import javax.annotation.Nullable;
018
019@PowerNukkitOnly
020public class BlockStonecutterBlock extends BlockTransparentMeta implements Faceable {
021
022    @PowerNukkitOnly
023    public BlockStonecutterBlock() {
024        this(0);
025    }
026
027    @PowerNukkitOnly
028    public BlockStonecutterBlock(int meta) {
029        super(meta);
030    }
031
032    @Override
033    public int getId() {
034        return STONECUTTER_BLOCK;
035    }
036
037    @Override
038    public String getName() {
039        return "Stonecutter";
040    }
041
042    @Override
043    @PowerNukkitOnly
044    @Since("1.3.0.0-PN")
045    public void setBlockFace(BlockFace face) {
046        int horizontalIndex = face.getHorizontalIndex();
047        if (horizontalIndex > -1) {
048            setDamage(horizontalIndex);
049        }
050    }
051
052    @Override
053    public BlockFace getBlockFace() {
054        return BlockFace.fromHorizontalIndex(this.getDamage() & 0x7);
055    }
056
057    @Override
058    public boolean place(@Nonnull Item item, @Nonnull Block block, @Nonnull Block target, @Nonnull BlockFace face, double fx, double fy, double fz, @Nullable Player player) {
059        int[] faces = {2, 5, 3, 4};
060        this.setDamage(faces[player != null ? player.getDirection().getHorizontalIndex() : 0]);
061        this.getLevel().setBlock(block, this, true, true);
062        return true;
063    }
064
065    @Override
066    public boolean canBeActivated() {
067        return true;
068    }
069
070    @Override
071    public boolean onActivate(@Nonnull Item item, @Nullable Player player) {
072        if (player != null) {
073            player.addWindow(new StonecutterInventory(player.getUIInventory(), this), ContainerIds.NONE);
074            player.craftingType = Player.CRAFTING_STONECUTTER;
075        }
076        return true;
077    }
078
079    @Override
080    public BlockColor getColor() {
081        return BlockColor.STONE_BLOCK_COLOR;
082    }
083
084    @Override
085    public double getHardness() {
086        return 3.5;
087    }
088
089    @Override
090    public double getResistance() {
091        return 17.5;
092    }
093
094    @Override
095    public boolean canHarvestWithHand() {
096        return false;
097    }
098
099    @PowerNukkitOnly
100    @Override
101    public int getWaterloggingLevel() {
102        return 1;
103    }
104
105    @Override
106    public int getToolType() {
107        return ItemTool.TYPE_PICKAXE;
108    }
109
110    @Override
111    public int getToolTier() {
112        return ItemTool.TIER_WOODEN;
113    }
114
115    @Override
116    public Item[] getDrops(Item item) {
117        return new Item[] {  toItem() };
118    }
119
120    @Override
121    public Item toItem() {
122        return new ItemBlock(new BlockStonecutterBlock());
123    }
124
125    @PowerNukkitDifference(since = "1.4.0.0-PN", info = "Fixed the returned value")
126    @Override
127    public double getMaxY() {
128        return y + 9/16.0;
129    }
130}
