001package cn.nukkit.block;
002
003import cn.nukkit.api.DeprecationDetails;
004import cn.nukkit.api.PowerNukkitDifference;
005import cn.nukkit.api.PowerNukkitOnly;
006import cn.nukkit.api.Since;
007import cn.nukkit.blockproperty.ArrayBlockProperty;
008import cn.nukkit.blockproperty.BlockProperties;
009import cn.nukkit.blockproperty.BlockProperty;
010import cn.nukkit.item.ItemTool;
011import cn.nukkit.utils.BlockColor;
012
013import javax.annotation.Nonnull;
014import java.util.Arrays;
015import java.util.stream.Collectors;
016
017import static cn.nukkit.utils.BlockColor.*;
018
019/**
020 * @author MagicDroidX (Nukkit Project)
021 * @apiNote Implements BlockConnectable only on PowerNukkit
022 */
023@PowerNukkitDifference(info = "Extends BlockWallBase and implements BlockConnectable only on PowerNukkit", since = "1.4.0.0-PN")
024public class BlockWall extends BlockWallBase {
025    @PowerNukkitOnly
026    @Since("1.4.0.0-PN")
027    public static final BlockProperty<WallType> WALL_BLOCK_TYPE = new ArrayBlockProperty<>("wall_block_type", true, WallType.class);
028    
029    @PowerNukkitOnly
030    @Since("1.4.0.0-PN")
031    public static final BlockProperties PROPERTIES = new BlockProperties(
032            WALL_BLOCK_TYPE,
033            WALL_CONNECTION_TYPE_SOUTH,
034            WALL_CONNECTION_TYPE_WEST,
035            WALL_CONNECTION_TYPE_NORTH,
036            WALL_CONNECTION_TYPE_EAST,
037            WALL_POST_BIT
038    );
039    
040    @Deprecated
041    @DeprecationDetails(reason = "No longer matches the meta directly", replaceWith = "WallType.COBBLESTONE", since = "1.3.0.0-PN")
042    public static final int NONE_MOSSY_WALL = 0;
043    
044    @Deprecated
045    @DeprecationDetails(reason = "No longer matches the meta directly", replaceWith = "WallType.MOSSY_COBBLESTONE", since = "1.3.0.0-PN")
046    public static final int MOSSY_WALL = 1;
047    
048    public BlockWall() {
049        this(0);
050    }
051
052    public BlockWall(int meta) {
053        super(meta);
054    }
055
056    @Override
057    public int getId() {
058        return STONE_WALL;
059    }
060    
061    @PowerNukkitOnly
062    @Since("1.4.0.0-PN")
063    @Nonnull
064    @Override
065    public BlockProperties getProperties() {
066        return PROPERTIES;
067    }
068
069
070    @PowerNukkitDifference(since = "1.3.0.0-PN", info = "Return the actual material color instead of transparent")
071    @Override
072    public BlockColor getColor() {
073        return getWallType().color;
074    }
075
076    @PowerNukkitOnly
077    @Since("1.3.0.0-PN")
078    public WallType getWallType() {
079        return getPropertyValue(WALL_BLOCK_TYPE);
080    }
081
082    @PowerNukkitOnly
083    @Since("1.3.0.0-PN")
084    public void setWallType(WallType type) {
085        setPropertyValue(WALL_BLOCK_TYPE, type);
086    }
087    
088    @Override
089    public String getName() {
090        return getWallType().getTypeName();
091    }
092
093    @Override
094    public int getToolTier() {
095        return ItemTool.TIER_WOODEN;
096    }
097    
098    @Since("1.3.0.0-PN")
099    @PowerNukkitOnly
100    public enum WallConnectionType {
101        @PowerNukkitOnly @Since("1.3.0.0-PN") NONE,
102        @PowerNukkitOnly @Since("1.3.0.0-PN") SHORT,
103        @PowerNukkitOnly @Since("1.3.0.0-PN") TALL
104    }
105    
106    @Since("1.3.0.0-PN")
107    @PowerNukkitOnly
108    public enum WallType {
109        @PowerNukkitOnly @Since("1.3.0.0-PN") COBBLESTONE,
110        @PowerNukkitOnly @Since("1.3.0.0-PN") MOSSY_COBBLESTONE,
111        @PowerNukkitOnly @Since("1.3.0.0-PN") GRANITE(DIRT_BLOCK_COLOR),
112        @PowerNukkitOnly @Since("1.3.0.0-PN") DIORITE(QUARTZ_BLOCK_COLOR),
113        @PowerNukkitOnly @Since("1.3.0.0-PN") ANDESITE, SANDSTONE(SAND_BLOCK_COLOR),
114        @PowerNukkitOnly @Since("1.3.0.0-PN") BRICK(RED_BLOCK_COLOR),
115        @PowerNukkitOnly @Since("1.3.0.0-PN") STONE_BRICK, MOSSY_STONE_BRICK,
116        @PowerNukkitOnly @Since("1.3.0.0-PN") NETHER_BRICK(NETHERRACK_BLOCK_COLOR),
117        @PowerNukkitOnly @Since("1.3.0.0-PN") END_STONE_BRICK(SAND_BLOCK_COLOR),
118        @PowerNukkitOnly @Since("1.3.0.0-PN") PRISMARINE(CYAN_BLOCK_COLOR),
119        @PowerNukkitOnly @Since("1.3.0.0-PN") RED_SANDSTONE(ORANGE_BLOCK_COLOR),
120        @PowerNukkitOnly @Since("1.3.0.0-PN") RED_NETHER_BRICK(NETHERRACK_BLOCK_COLOR);
121        
122        private final BlockColor color;
123        private final String typeName;
124        
125        WallType(BlockColor color) {
126            this.color = color;
127            String name = Arrays.stream(name().split("_"))
128                    .map(part-> part.substring(0,1) + part.substring(1).toLowerCase())
129                    .collect(Collectors.joining(" "));
130            
131            // Concatenation separated to workaround https://bugs.openjdk.java.net/browse/JDK-8077605
132            // https://www.reddit.com/r/learnprogramming/comments/32bfle/can_you_explain_this_strange_java8_error/
133            typeName = name + " Wall";
134        }
135
136        WallType() {
137            this(STONE_BLOCK_COLOR);
138        }
139
140        @Since("1.3.0.0-PN")
141        public BlockColor getColor() {
142            return color;
143        }
144
145        @Since("1.4.0.0-PN")
146        public String getTypeName() {
147            return typeName;
148        }
149    }
150}
