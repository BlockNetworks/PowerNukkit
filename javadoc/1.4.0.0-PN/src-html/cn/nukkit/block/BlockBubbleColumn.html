112    @Override
113    public void onEntityCollide(Entity entity) {
114        if (entity.canBeMovedByCurrents()) {
115            if (up().getId() == AIR) {
116                if (getDamage() == 1) {
117                    entity.motionY = Math.max(-0.9, entity.motionY - 0.03);
118                } else {
119                    entity.motionY = Math.min(1.8, entity.motionY + 0.1);
120                }
121                
122                ThreadLocalRandom random = ThreadLocalRandom.current();
123                for(int i = 0; i < 2; ++i) {
124                    level.addParticle(new SplashParticle(add(random.nextFloat(), random.nextFloat() + 1, random.nextFloat())));
125                    level.addParticle(new BubbleParticle(add(random.nextFloat(), random.nextFloat() + 1, random.nextFloat())));
126                }
127                
128            } else {
129                if (getDamage() == 1) {
130                    entity.motionY = Math.max(-0.3, entity.motionY - 0.3);
131                } else {
132                    entity.motionY = Math.min(0.7, entity.motionY + 0.06);
133                }
134            }
135            entity.motionChanged = true;
136            entity.resetFallDistance();
137        }
138    }

170    @Override
171    public int onUpdate(int type) {
172        if (type == Level.BLOCK_UPDATE_NORMAL) {
173            Block water = getLevelBlockAtLayer(1);
174            if (!(water instanceof BlockWater) || water.getDamage() != 0 && water.getDamage() != 8) {
175                fadeOut(water);
176                return type;
177            }
178
179            if (water.getDamage() == 8) {
180                water.setDamage(0);
181                this.getLevel().setBlock(this, 1, water, true, false);
182            }
183
184            Block down = down();
185            if (down.getId() == BUBBLE_COLUMN) {
186                if (down.getDamage() != this.getDamage()) {
187                    this.getLevel().setBlock(this, down, true, true);
188                }
189            } else if (down.getId() == MAGMA) {
190                if (this.getDamage() != 1) {
191                    setDamage(1);
192                    this.getLevel().setBlock(this, this, true, true);
193                }
194            } else if (down.getId() == SOUL_SAND) {
195                if (this.getDamage() != 0) {
196                    setDamage(0);
197                    this.getLevel().setBlock(this, this, true, true);
198                }
199            } else {
200                fadeOut(water);
201                return type;
202            }
203
204            Block up = up();
205            if (up instanceof BlockWater && (up.getDamage() == 0 || up.getDamage() == 8)) {
206                BlockFromToEvent event = new BlockFromToEvent(this, up);
207                if (!event.isCancelled()) {
208                    this.getLevel().setBlock(up, 1, new BlockWater(), true, false);
209                    this.getLevel().setBlock(up, 0, new BlockBubbleColumn(this.getDamage()), true, true);
210                }
211            }
212
213            return type;
214        }
215
216        return 0;
217    }
218
219    private void fadeOut(Block water) {
220        BlockFadeEvent event = new BlockFadeEvent(this, water.clone());
221        if (!event.isCancelled()) {
222            this.getLevel().setBlock(this, 1, new BlockAir(), true, false);
223            this.getLevel().setBlock(this, 0, event.getNewState(), true, true);
224        }
225    }
