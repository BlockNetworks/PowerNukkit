001package cn.nukkit.block;
002
003
004import cn.nukkit.Player;
005import cn.nukkit.api.PowerNukkitOnly;
006import cn.nukkit.api.Since;
007import cn.nukkit.blockproperty.BlockProperties;
008import cn.nukkit.item.Item;
009import cn.nukkit.item.ItemChain;
010import cn.nukkit.item.ItemTool;
011import cn.nukkit.math.BlockFace;
012
013import javax.annotation.Nonnull;
014import javax.annotation.Nullable;
015
016import static cn.nukkit.blockproperty.CommonBlockProperties.PILLAR_AXIS;
017
018@PowerNukkitOnly
019@Since("1.4.0.0-PN")
020public class BlockChain extends BlockTransparent {
021    
022    @PowerNukkitOnly
023    @Since("1.4.0.0-PN")
024    public BlockChain() {
025        // Does nothing
026    }
027
028    @Override
029    public String getName() {
030        return "Chain";
031    }
032
033    @Override
034    public int getId() {
035        return CHAIN_BLOCK;
036    }
037
038    @Since("1.4.0.0-PN")
039    @PowerNukkitOnly
040    @Nonnull
041    @Override
042    public BlockProperties getProperties() {
043        return BlockLog.PILLAR_PROPERTIES;
044    }
045
046    @PowerNukkitOnly
047    @Since("1.4.0.0-PN")
048    public BlockFace.Axis getPillarAxis() {
049        return getPropertyValue(PILLAR_AXIS);
050    }
051
052    @PowerNukkitOnly
053    @Since("1.4.0.0-PN")
054    public void setPillarAxis(BlockFace.Axis axis) {
055        setPropertyValue(PILLAR_AXIS, axis);
056    }
057
058    @Override
059    public boolean place(@Nonnull Item item, @Nonnull Block block, @Nonnull Block target, @Nonnull BlockFace face, double fx, double fy, double fz, @Nullable Player player) {
060        setPillarAxis(face.getAxis());
061        return super.place(item, block, target, face, fx, fy, fz, player);
062    }
063
064    @Override
065    public double getHardness() {
066        return 5;
067    }
068
069    @PowerNukkitOnly
070    @Override
071    public int getWaterloggingLevel() {
072        return 1;
073    }
074
075    @Override
076    public double getResistance() {
077        return 6;
078    }
079
080    @Override
081    public int getToolType() {
082        return ItemTool.TYPE_PICKAXE;
083    }
084
085    @Override
086    public double getMinX() {
087        return x + 7/16.0;
088    }
089
090    @Override
091    public double getMaxX() {
092        return x + 9/16.0;
093    }
094
095    @Override
096    public double getMinZ() {
097        return z + 7/16.0;
098    }
099
100    @Override
101    public double getMaxZ() {
102        return z + 9/16.0;
103    }
104
105    @Override
106    public Item toItem() {
107        return new ItemChain();
108    }
109
110    @Since("1.4.0.0-PN")
111    @PowerNukkitOnly
112    @Override
113    public int getToolTier() {
114        return ItemTool.TIER_WOODEN;
115    }
116
117    @Override
118    public boolean canHarvestWithHand() {
119        return false;
120    }
121    
122}
