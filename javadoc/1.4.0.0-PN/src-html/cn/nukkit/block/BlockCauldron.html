083    public boolean isFull() {
084        return (this.getDamage() & 0x06) == 0x06;
085    }
086
087    public boolean isEmpty() {
088        return this.getDamage() == 0x00;
089    }

091    public int getFillLevel() {
092        return (getDamage() & 0x6) >> 1;
093    }
094    
095    public void setFillLevel(int fillLevel) {
096        fillLevel = MathHelper.clamp(fillLevel, 0, 3);
097        setDamage(fillLevel << 1);
098    }

100    @Override
101    public boolean onActivate(@Nonnull Item item, Player player) {
102        BlockEntityCauldron cauldron = getBlockEntity();
103
104        if (cauldron == null) {
105            return false;
106        }
107
108        switch (item.getId()) {
109            case Item.BUCKET:
110                ItemBucket bucket = (ItemBucket) item;
111                if (bucket.getFishEntityId() != null) {
112                    break;
113                }
114                if (bucket.isEmpty()) {
115                    if (!isFull() || cauldron.isCustomColor() || cauldron.hasPotion()) {
116                        break;
117                    }
118                    
119                    PlayerBucketFillEvent ev = new PlayerBucketFillEvent(player, this, null, this, item, MinecraftItemID.WATER_BUCKET.get(1, bucket.getCompoundTag()));
120                    this.level.getServer().getPluginManager().callEvent(ev);
121                    if (!ev.isCancelled()) {
122                        replaceBucket(bucket, player, ev.getItem());
123                        this.setFillLevel(0);//empty
124                        this.level.setBlock(this, this, true);
125                        cauldron.clearCustomColor();
126                        this.getLevel().addLevelEvent(this.add(0.5, 0.375 + this.getDamage() * 0.125, 0.5), LevelEventPacket.EVENT_CAULDRON_TAKE_WATER);
127                    }
128                } else if (bucket.isWater() || bucket.isLava()) {
129                    if (isFull() && !cauldron.isCustomColor() && !cauldron.hasPotion() && item.getDamage() == 8) {
130                        break;
131                    }
132
133                    PlayerBucketEmptyEvent ev = new PlayerBucketEmptyEvent(player, this, null, this, item, MinecraftItemID.BUCKET.get(1, bucket.getCompoundTag()));
134                    this.level.getServer().getPluginManager().callEvent(ev);
135                    if (!ev.isCancelled()) {
136                        if (player.isSurvival() || player.isAdventure()) {
137                            replaceBucket(bucket, player, ev.getItem());
138                        }
139                        if (cauldron.hasPotion()) {//if has potion
140                            clearWithFizz(cauldron);
141                        } else if (bucket.isWater()) { //water bucket
142                            this.setFillLevel(3);//fill
143                            cauldron.clearCustomColor();
144                            this.level.setBlock(this, this, true);
145                            this.getLevel().addSound(this.add(0.5, 1, 0.5), Sound.CAULDRON_FILLWATER);
146                        } else { // lava bucket
147                            if (isEmpty()) {
148                                BlockCauldronLava cauldronLava = new BlockCauldronLava(0xE);
149                                cauldronLava.setFillLevel(3);
150                                this.level.setBlock(this, cauldronLava, true, true);
151                                cauldron.clearCustomColor();
152                                cauldron.setType(BlockEntityCauldron.PotionType.LAVA);
153                                this.getLevel().addSound(this.add(0.5, 1, 0.5), Sound.BUCKET_EMPTY_LAVA);
154                            } else {
155                                clearWithFizz(cauldron);
156                            }
157                        }
158                        //this.update();
159                    }
160                }
161                break;
162            case ItemID.DYE:
163                if (isEmpty() || cauldron.hasPotion()) {
164                    break;
165                }
166    
167                if (player.isSurvival() || player.isAdventure()) {
168                    item.setCount(item.getCount() - 1);
169                    player.getInventory().setItemInHand(item);
170                }
171    
172                BlockColor color = new ItemDye(item.getDamage()).getDyeColor().getLeatherColor();
173                if (!cauldron.isCustomColor()) {
174                    cauldron.setCustomColor(color);
175                } else {
176                    BlockColor current = cauldron.getCustomColor();
177                    BlockColor mixed = new BlockColor(
178                            (int)Math.round(Math.sqrt(color.getRed() * current.getRed()) * 0.965),
179                            (int)Math.round(Math.sqrt(color.getGreen() * current.getGreen()) * 0.965),
180                            (int)Math.round(Math.sqrt(color.getBlue() * current.getBlue()) * 0.965)
181                    );
182                    cauldron.setCustomColor(mixed);
183                }
184                this.level.addSound(this.add(0.5, 0.5, 0.5), Sound.CAULDRON_ADDDYE);
185                
186                break;
187            
188            case ItemID.LEATHER_CAP:
189            case ItemID.LEATHER_TUNIC:
190            case ItemID.LEATHER_PANTS:
191            case ItemID.LEATHER_BOOTS:
192            case ItemID.LEATHER_HORSE_ARMOR:
193                if (isEmpty() || cauldron.hasPotion()) {
194                    break;
195                }
196                
197                if (cauldron.isCustomColor()) {
198                    CompoundTag compoundTag = item.hasCompoundTag() ? item.getNamedTag() : new CompoundTag();
199                    compoundTag.putInt("customColor", cauldron.getCustomColor().getRGB());
200                    item.setCompoundTag(compoundTag);
201                    player.getInventory().setItemInHand(item);
202                    
203                    setFillLevel(getFillLevel() - 1);
204                    this.level.setBlock(this, this, true, true);
205                    this.level.addSound(add(0.5, 0.5, 0.5), Sound.CAULDRON_DYEARMOR);
206                } else {
207                    if (!item.hasCompoundTag()) {
208                        break;
209                    }
210                    
211                    CompoundTag compoundTag = item.getNamedTag();
212                    if (!compoundTag.exist("customColor")) {
213                        break;
214                    }
215                    
216                    compoundTag.remove("customColor");
217                    item.setCompoundTag(compoundTag);
218                    player.getInventory().setItemInHand(item);
219                    
220                    setFillLevel(getFillLevel() - 1);
221                    this.level.setBlock(this, this, true, true);
222                    this.getLevel().addSound(this.add(0.5, 1, 0.5), Sound.CAULDRON_TAKEWATER);
223                }
224                
225                break;
226            case ItemID.POTION:
227            case ItemID.SPLASH_POTION:
228            case ItemID.LINGERING_POTION:
229                if (!isEmpty() && (cauldron.hasPotion()? cauldron.getPotionId() != item.getDamage() : item.getDamage() != 0)) {
230                    clearWithFizz(cauldron);
231                    consumePotion(item, player);
232                    break;
233                }
234                if (isFull()) {
235                    break;
236                }
237                
238                if (item.getDamage() != 0 && isEmpty()) {
239                    cauldron.setPotionId(item.getDamage());
240                }
241                
242                cauldron.setPotionType(
243                        item.getId() == ItemID.POTION? BlockEntityCauldron.POTION_TYPE_NORMAL :
244                                item.getId() == ItemID.SPLASH_POTION? BlockEntityCauldron.POTION_TYPE_SPLASH :
245                                        BlockEntityCauldron.POTION_TYPE_LINGERING
246                );
247                cauldron.spawnToAll();
248                
249                setFillLevel(getFillLevel() + 1);
250                this.level.setBlock(this, this, true);
251    
252                consumePotion(item, player);
253    
254                this.level.addLevelEvent(this.add(0.5, 0.375 + this.getDamage() * 0.125, 0.5), LevelEventPacket.EVENT_CAULDRON_FILL_POTION);
255                break;
256            case ItemID.GLASS_BOTTLE:
257                if (isEmpty()) {
258                    break;
259                }
260
261                int meta = cauldron.hasPotion() ? cauldron.getPotionId() : 0;
262                
263                Item potion;
264                if (meta == 0) {
265                    potion = new ItemPotion();
266                } else {
267                    switch (cauldron.getPotionType()) {
268                        case BlockEntityCauldron.POTION_TYPE_SPLASH:
269                            potion = new ItemPotionSplash(meta);
270                            break;
271                        case BlockEntityCauldron.POTION_TYPE_LINGERING:
272                            potion = new ItemPotionLingering(meta);
273                            break;
274                        case BlockEntityCauldron.POTION_TYPE_NORMAL:
275                        default:
276                            potion = new ItemPotion(meta);
277                            break;
278                    }
279                }
280
281                setFillLevel(getFillLevel() - 1);
282                if (isEmpty()) {
283                    cauldron.setPotionId(-1);//reset potion
284                    cauldron.clearCustomColor();
285                }
286                this.level.setBlock(this, this, true);
287                
288                boolean consumeBottle = player.isSurvival() || player.isAdventure();
289                if (consumeBottle && item.getCount() == 1) {
290                    player.getInventory().setItemInHand(potion);
291                } else if (item.getCount() > 1) {
292                    if (consumeBottle) {
293                        item.setCount(item.getCount() - 1);
294                        player.getInventory().setItemInHand(item);
295                    }
296    
297                    if (player.getInventory().canAddItem(potion)) {
298                        player.getInventory().addItem(potion);
299                    } else {
300                        player.getLevel().dropItem(player.add(0, 1.3, 0), potion, player.getDirectionVector().multiply(0.4));
301                    }
302                }
303
304                this.level.addLevelEvent(this.add(0.5, 0.375 + this.getDamage() * 0.125, 0.5), LevelEventPacket.EVENT_CAULDRON_TAKE_POTION);
305                break;
306            case ItemID.BANNER:
307                if (isEmpty() || cauldron.isCustomColor() || cauldron.hasPotion()) {
308                    break;
309                }
310                
311                ItemBanner banner = (ItemBanner) item;
312                if (!banner.hasPattern()) {
313                    break;
314                }
315                
316                banner.removePattern(banner.getPatternsSize() - 1);
317                boolean consumeBanner = player.isSurvival() || player.isAdventure();
318                if (consumeBanner && item.getCount() < item.getMaxStackSize()) {
319                    player.getInventory().setItemInHand(banner);
320                } else {
321                    if (consumeBanner) {
322                        item.setCount(item.getCount() - 1);
323                        player.getInventory().setItemInHand(item);
324                    }
325                    
326                    if (player.getInventory().canAddItem(banner)) {
327                        player.getInventory().addItem(banner);
328                    } else {
329                        player.getLevel().dropItem(player.add(0, 1.3, 0), banner, player.getDirectionVector().multiply(0.4));
330                    }
331                }
332                
333                setFillLevel(getFillLevel() - 1);
334                this.level.setBlock(this, this, true, true);
335                this.getLevel().addSound(this.add(0.5, 1, 0.5), Sound.CAULDRON_TAKEWATER);
336                
337                break;
338            default:
339                if (item instanceof ItemDye) {
340                    if (isEmpty() || cauldron.hasPotion()) {
341                        break;
342                    }
343
344                    if (player.isSurvival() || player.isAdventure()) {
345                        item.setCount(item.getCount() - 1);
346                        player.getInventory().setItemInHand(item);
347                    }
348
349                    color = ((ItemDye)item).getDyeColor().getColor();
350                    if (!cauldron.isCustomColor()) {
351                        cauldron.setCustomColor(color);
352                    } else {
353                        BlockColor current = cauldron.getCustomColor();
354                        BlockColor mixed = new BlockColor(
355                                current.getRed() + (color.getRed() - current.getRed()) / 2,
356                                current.getGreen() + (color.getGreen() - current.getGreen()) / 2,
357                                current.getBlue() + (color.getBlue() - current.getBlue()) / 2
358                        );
359                        cauldron.setCustomColor(mixed);
360                    }
361                    this.level.addSound(this.add(0.5, 0.5, 0.5), Sound.CAULDRON_ADDDYE);
362                } else {
363                    return true;
364                }
365        }
366
367        this.level.updateComparatorOutputLevel(this);
368        return true;
369    }

371    protected void replaceBucket(Item oldBucket, Player player, Item newBucket) {
372        if (player.isSurvival() || player.isAdventure()) {
373            if (oldBucket.getCount() == 1) {
374                player.getInventory().setItemInHand(newBucket);
375            } else {
376                oldBucket.setCount(oldBucket.getCount() - 1);
377                if (player.getInventory().canAddItem(newBucket)) {
378                    player.getInventory().addItem(newBucket);
379                } else {
380                    player.getLevel().dropItem(player.add(0, 1.3, 0), newBucket, player.getDirectionVector().multiply(0.4));
381                }
382            }
383        }
384    }
385    
386    private void consumePotion(Item item, Player player) {
387        if (player.isSurvival() || player.isAdventure()) {
388            if (item.getCount() == 1) {
389                player.getInventory().setItemInHand(new ItemBlock(new BlockAir()));
390            } else if (item.getCount() > 1) {
391                item.setCount(item.getCount() - 1);
392                player.getInventory().setItemInHand(item);
393
394                Item bottle = new ItemGlassBottle();
395                if (player.getInventory().canAddItem(bottle)) {
396                    player.getInventory().addItem(bottle);
397                } else {
398                    player.getLevel().dropItem(player.add(0, 1.3, 0), bottle, player.getDirectionVector().multiply(0.4));
399                }
400            }
401        }
402    }

404    @PowerNukkitOnly
405    public void clearWithFizz(BlockEntityCauldron cauldron) {
406        this.setFillLevel(0);//empty
407        cauldron.setPotionId(-1);//reset potion
408        cauldron.setSplashPotion(false);
409        cauldron.clearCustomColor();
410        this.level.setBlock(this, new BlockCauldron(0), true);
411        this.level.addSound(this.add(0.5, 0, 0.5), Sound.RANDOM_FIZZ);
412        for (int i = 0; i < 8; ++i) {
413            this.getLevel().addParticle(new SmokeParticle(add(Math.random(), 1.2, Math.random())));
414        }
415    }

417    @Override
418    public boolean place(@Nonnull Item item, @Nonnull Block block, @Nonnull Block target, @Nonnull BlockFace face, double fx, double fy, double fz, Player player) {
419        CompoundTag nbt = new CompoundTag()
420                .putShort("PotionId", -1)
421                .putByte("SplashPotion", 0);
422
423        if (item.hasCustomBlockData()) {
424            Map<String, Tag> customData = item.getCustomBlockData().getTags();
425            for (Map.Entry<String, Tag> tag : customData.entrySet()) {
426                nbt.put(tag.getKey(), tag.getValue());
427            }
428        }
429
430        return BlockEntityHolder.setBlockAndCreateEntity(this, true, true, nbt) != null;
431    }
