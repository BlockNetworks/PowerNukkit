113    @Override
114    protected AxisAlignedBB recalculateBoundingBox() {
115        BlockFace position = getBlockFace().getOpposite();
116        boolean isOpen = isOpen();
117        boolean isRight = isRightHinged();
118        
119        if (isOpen) {
120            return recalculateBoundingBoxWithPos(isRight? position.rotateYCCW() : position.rotateY());
121        } else {
122            return recalculateBoundingBoxWithPos(position);
123        }
124    }
125
126    private AxisAlignedBB recalculateBoundingBoxWithPos(BlockFace pos) {
127        if (pos.getAxisDirection() == AxisDirection.NEGATIVE) {
128            return new SimpleAxisAlignedBB (
129                    this.x,
130                    this.y,
131                    this.z,
132                    this.x + 1 + pos.getXOffset() - (THICKNESS * pos.getXOffset()),
133                    this.y + 1,
134                    this.z + 1 + pos.getZOffset() - (THICKNESS * pos.getZOffset())
135            );
136        } else {
137            return new SimpleAxisAlignedBB (
138                    this.x + pos.getXOffset() - (THICKNESS * pos.getXOffset()),
139                    this.y,
140                    this.z + pos.getZOffset() - (THICKNESS * pos.getZOffset()),
141                    this.x + 1,
142                    this.y + 1,
143                    this.z + 1
144            );
145        }
146    }

148    @PowerNukkitDifference(info = "Will drop the iron door item if the support is broken", since = "1.3.1.2-PN")
149    @Override
150    public int onUpdate(int type) {
151        if (type == Level.BLOCK_UPDATE_NORMAL) {
152            this.onNormalUpdate();
153            return type;
154        }
155        
156        if (type == Level.BLOCK_UPDATE_REDSTONE && level.getServer().isRedstoneEnabled()) {
157            this.onRedstoneUpdate();
158            return type;
159        }
160
161        return 0;
162    }
163
164    private void onNormalUpdate() {
165        Block down = this.down();
166        if (isTop()) {
167            if (down.getId() != this.getId() || down.getBooleanValue(UPPER_BLOCK)) {
168                level.setBlock(this, Block.get(AIR), false);
169            }
170
171            /* Doesn't work with redstone
172            boolean downIsOpen = down.getBooleanValue(OPEN);
173            if (downIsOpen != isOpen()) {
174                setOpen(downIsOpen);
175                level.setBlock(this, this, false, true);
176            }*/
177            return;
178        }
179
180        if (down.getId() == AIR) {
181            level.useBreakOn(this, getToolType() == ItemTool.TYPE_PICKAXE? Item.get(ItemID.DIAMOND_PICKAXE) : null);
182        }
183    }
184
185    @PowerNukkitDifference(info = "Checking if the door was opened/closed manually.")
186    private void onRedstoneUpdate() {
187        if ((this.isOpen() != this.isGettingPower()) && !this.getManualOverride()) {
188            if (this.isOpen() != this.isGettingPower()) {
189                level.getServer().getPluginManager().callEvent(new BlockRedstoneEvent(this, this.isOpen() ? 15 : 0, this.isOpen() ? 0 : 15));
190
191                this.setOpen(null, this.isGettingPower());
192            }
193        } else if (this.getManualOverride() && (this.isGettingPower() == this.isOpen())) {
194            this.setManualOverride(false);
195        }
196    }

198    @PowerNukkitOnly
199    @Since("1.4.0.0-PN")
200    public void setManualOverride(boolean val) {
201        Location down;
202        Location up;
203        if (this.isTop()) {
204            down = down().getLocation();
205            up = getLocation();
206        } else {
207            down = getLocation();
208            up = up().getLocation();
209        }
210
211        if (val) {
212            manualOverrides.add(up);
213            manualOverrides.add(down);
214        } else {
215            manualOverrides.remove(up);
216            manualOverrides.remove(down);
217        }
218    }

220    @PowerNukkitOnly
221    @Since("1.4.0.0-PN")
222    public boolean getManualOverride() {
223        Location down;
224        Location up;
225        if (this.isTop()) {
226            down = down().getLocation();
227            up = getLocation();
228        } else {
229            down = getLocation();
230            up = up().getLocation();
231        }
232
233        return manualOverrides.contains(up) || manualOverrides.contains(down);
234    }

236    @Override
237    @PowerNukkitOnly
238    @Since("1.4.0.0-PN")
239    public boolean isGettingPower() {
240        Location down;
241        Location up;
242        if (this.isTop()) {
243            down = down().getLocation();
244            up = getLocation();
245        } else {
246            down = getLocation();
247            up = up().getLocation();
248        }
249
250        for (BlockFace side : BlockFace.values()) {
251            Block blockDown = down.getSide(side).getLevelBlock();
252            Block blockUp = up.getSide(side).getLevelBlock();
253
254            if (this.level.isSidePowered(blockDown.getLocation(), side)
255                    || this.level.isSidePowered(blockUp.getLocation(), side)) {
256                return true;
257            }
258        }
259
260        return this.level.isBlockPowered(down) || this.level.isBlockPowered(up);
261    }

263    @PowerNukkitDifference(since = "1.4.0.0-PN", info = "Fixed support logic")
264    @Override
265    public boolean place(@Nonnull Item item, @Nonnull Block block, @Nonnull Block target, @Nonnull BlockFace face, double fx, double fy, double fz, @Nullable Player player) {
266        if (this.y > 254 || face != BlockFace.UP) {
267            return false;
268        }
269
270        Block blockUp = this.up();
271        Block blockDown = this.down();
272        if (!blockUp.canBeReplaced() || !blockDown.isSolid(BlockFace.UP) && !(blockDown instanceof BlockCauldron)) {
273            return false;
274        }
275
276        BlockFace direction = player.getDirection();
277        setBlockFace(direction);
278
279        Block left = this.getSide(direction.rotateYCCW());
280        Block right = this.getSide(direction.rotateY());
281        if (left.getId() == this.getId() || (!right.isTransparent() && left.isTransparent())) { //Door hinge
282            setRightHinged(true);
283        }
284
285        setTop(false);
286
287        level.setBlock(block, this, true, false); //Bottom
288
289        if (blockUp instanceof BlockLiquid && ((BlockLiquid) blockUp).usesWaterLogging()) {
290            level.setBlock(blockUp, 1, blockUp, true, false);
291        }
292
293        BlockDoor doorTop = (BlockDoor) clone();
294        doorTop.y++;
295        doorTop.setTop(true);
296        level.setBlock(doorTop, doorTop, true, true); //Top
297        
298        level.updateAround(block);
299
300        if (level.getServer().isRedstoneEnabled() && !this.isOpen() && this.isGettingPower()) {
301            this.setOpen(null, true);
302        }
303
304        return true;
305    }

307    @Override
308    public boolean onBreak(Item item) {
309        this.setManualOverride(false);
310        if (isTop()) {
311            Block down = this.down();
312            if (down.getId() == this.getId() && !down.getBooleanValue(UPPER_BLOCK)) {
313                level.setBlock(down, Block.get(AIR), true);
314            }
315        } else {
316            Block up = this.up();
317            if (up.getId() == this.getId() && up.getBooleanValue(UPPER_BLOCK)) {
318                level.setBlock(up, Block.get(BlockID.AIR), true);
319            }
320        }
321        this.getLevel().setBlock(this, Block.get(BlockID.AIR), true);
322
323        return true;
324    }

333    public void playOpenCloseSound() {
334        if (this.isTop() && down() instanceof BlockDoor) {
335            if (((BlockDoor) down()).isOpen()) {
336                this.playOpenSound();
337            } else {
338                this.playCloseSound();
339            }
340        } else if (up() instanceof BlockDoor) {
341            if (this.isOpen()) {
342                this.playOpenSound();
343            } else {
344                this.playCloseSound();
345            }
346        }
347    }

366    @PowerNukkitDifference(info = "Using direct values, instead of toggling (fixes a redstone bug, that the door won't open). " +
367            "Also adding possibility to detect, whether a player or redstone recently opened/closed the door.", since = "1.4.0.0-PN")
368    @PowerNukkitOnly
369    public boolean setOpen(Player player, boolean open) {
370        if (open == this.isOpen()) {
371            return false;
372        }
373
374        DoorToggleEvent event = new DoorToggleEvent(this, player);
375        level.getServer().getPluginManager().callEvent(event);
376
377        if (event.isCancelled()) {
378            return false;
379        }
380
381        player = event.getPlayer();
382
383        Block down;
384        Block up;
385        if (this.isTop()) {
386            down = down();
387            up = this;
388        } else {
389            down = this;
390            up = up();
391        }
392
393        up.setBooleanValue(OPEN, open);
394        up.level.setBlock(up, up, true, true);
395
396        down.setBooleanValue(OPEN, open);
397        down.level.setBlock(down, down, true, true);
398
399        if (player != null) {
400            this.setManualOverride(this.isGettingPower() || isOpen());
401        }
402
403        playOpenCloseSound();
404        return true;
405    }
