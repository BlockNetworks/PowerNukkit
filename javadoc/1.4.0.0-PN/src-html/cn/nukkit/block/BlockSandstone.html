001package cn.nukkit.block;
002
003import cn.nukkit.item.Item;
004import cn.nukkit.item.ItemBlock;
005import cn.nukkit.item.ItemTool;
006import cn.nukkit.utils.BlockColor;
007
008/**
009 * @author MagicDroidX (Nukkit Project)
010 */
011public class BlockSandstone extends BlockSolidMeta {
012    public static final int NORMAL = 0;
013    public static final int CHISELED = 1;
014    public static final int SMOOTH = 2;
015
016    public BlockSandstone() {
017        this(0);
018    }
019
020    public BlockSandstone(int meta) {
021        super(meta);
022    }
023
024    @Override
025    public int getId() {
026        return SANDSTONE;
027    }
028
029    @Override
030    public double getHardness() {
031        return 0.8;
032    }
033
034    @Override
035    public double getResistance() {
036        return 4;
037    }
038
039    @Override
040    public String getName() {
041        String[] names = new String[]{
042                "Sandstone",
043                "Chiseled Sandstone",
044                "Smooth Sandstone",
045                ""
046        };
047
048        return names[this.getDamage() & 0x03];
049    }
050
051    @Override
052    public int getToolTier() {
053        return ItemTool.TIER_WOODEN;
054    }
055
056    @Override
057    public Item toItem() {
058        return new ItemBlock(this, this.getDamage() & 0x03);
059    }
060
061    @Override
062    public int getToolType() {
063        return ItemTool.TYPE_PICKAXE;
064    }
065
066    @Override
067    public BlockColor getColor() {
068        return BlockColor.SAND_BLOCK_COLOR;
069    }
070
071    @Override
072    public boolean canHarvestWithHand() {
073        return false;
074    }
075}
