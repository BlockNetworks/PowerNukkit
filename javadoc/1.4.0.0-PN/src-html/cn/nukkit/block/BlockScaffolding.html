022    @PowerNukkitOnly
023    public BlockScaffolding() {
024        // Does nothing
025    }

027    @PowerNukkitOnly
028    public BlockScaffolding(int meta) {
029        super(meta);
030    }
031
032    @Override
033    public int getId() {
034        return SCAFFOLDING;
035    }
036
037    @Override
038    public String getName() {
039        return "Scaffolding";
040    }
041
042    @PowerNukkitOnly
043    public int getStability() {
044        return getDamage() & 0x7;
045    }
046
047    @PowerNukkitOnly
048    public void setStability(int stability) {
049        setDamage(stability & 0x7 | (getDamage() & 0x8));
050    }
051
052    @PowerNukkitOnly
053    public boolean getStabilityCheck() {
054        return (getDamage() & 0x8) > 0;
055    }
056
057    @PowerNukkitOnly
058    public void setStabilityCheck(boolean check) {
059        if (check) {
060            setDamage(getDamage() | 0x8);
061        } else {
062            setDamage(getDamage() & 0x7);
063        }
064    }
065
066    @Override
067    public Item toItem() {
068        return new ItemBlock(new BlockScaffolding(0));
069    }
070
071    @Override
072    public boolean place(@Nonnull Item item, @Nonnull Block block, @Nonnull Block target, @Nonnull BlockFace face, double fx, double fy, double fz, Player player) {
073        if (block instanceof BlockLava) {
074            return false;
075        }
076
077        Block down = down();
078        if (target.getId() != SCAFFOLDING && down.getId() != SCAFFOLDING && down.getId() != AIR && !down.isSolid()) {
079            boolean scaffoldOnSide = false;
080            for (int i = 0; i < 4; i++) {
081                BlockFace sideFace = BlockFace.fromHorizontalIndex(i);
082                if (sideFace != face) {
083                    Block side = getSide(sideFace);
084                    if (side.getId() == SCAFFOLDING) {
085                        scaffoldOnSide = true;
086                        break;
087                    }
088                }
089            }
090            if (!scaffoldOnSide) {
091                return false;
092            }
093        }
094
095        setDamage(0x8);
096        this.getLevel().setBlock(this, this, true, true);
097        return true;
098    }
099
100    @Override
101    public int onUpdate(int type) {
102        if (type == Level.BLOCK_UPDATE_NORMAL) {
103            Block down = down();
104            if (down.isSolid()) {
105                if (getDamage() != 0) {
106                    setDamage(0);
107                    this.getLevel().setBlock(this, this, true, true);
108                }
109                return type;
110            }
111
112            int stability = 7;
113            for (BlockFace face : BlockFace.values()) {
114                if (face == BlockFace.UP) {
115                    continue;
116                }
117
118                Block otherBlock = getSide(face);
119                if (otherBlock.getId() == SCAFFOLDING) {
120                    BlockScaffolding other = (BlockScaffolding) otherBlock;
121                    int otherStability = other.getStability();
122                    if (otherStability < stability) {
123                        if (face == BlockFace.DOWN) {
124                            stability = otherStability;
125                        } else {
126                            stability = otherStability + 1;
127                        }
128                    }
129                }
130            }
131
132            if (stability >= 7) {
133                if (getStabilityCheck()) {
134                    super.onUpdate(type);
135                } else {
136                    this.getLevel().scheduleUpdate(this, 0);
137                }
138                return type;
139            }
140
141            setStabilityCheck(false);
142            setStability(stability);
143            this.getLevel().setBlock(this, this, true, true);
144            return type;
145        } else if (type == Level.BLOCK_UPDATE_SCHEDULED) {
146            this.getLevel().useBreakOn(this);
147            return type;
148        }
149
150        return 0;
151    }
