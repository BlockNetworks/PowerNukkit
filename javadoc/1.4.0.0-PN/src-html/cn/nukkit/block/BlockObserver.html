029@PowerNukkitDifference(info = "Implements RedstoneComponent and uses methods from it.", since = "1.4.0.0-PN")
030public class BlockObserver extends BlockSolidMeta implements RedstoneComponent, Faceable {
031    @PowerNukkitOnly
032    @Since("1.4.0.0-PN")
033    public static final BlockProperties PROPERTIES = new BlockProperties(FACING_DIRECTION, POWERED);
034
035    public BlockObserver() {
036        this(0);
037    }
038
039    public BlockObserver(int meta) {
040        super(meta);
041    }
042
043    @Override
044    public String getName() {
045        return "Observer";
046    }
047
048    @Override
049    public int getId() {
050        return OBSERVER;
051    }
052
053    @Since("1.4.0.0-PN")
054    @PowerNukkitOnly
055    @Nonnull
056    @Override
057    public BlockProperties getProperties() {
058        return PROPERTIES;
059    }
060
061    @Override
062    public boolean place(@Nonnull Item item, @Nonnull Block block, @Nonnull Block target, @Nonnull BlockFace face, double fx, double fy, double fz, @Nullable Player player) {
063        if (player != null) {
064            if (Math.abs(player.getFloorX() - this.x) <= 1 && Math.abs(player.getFloorZ() - this.z) <= 1) {
065                double y = player.y + player.getEyeHeight();
066                if (y - this.y > 2) {
067                    setBlockFace(BlockFace.DOWN);
068                } else if (this.y - y > 0) {
069                    setBlockFace(BlockFace.UP);
070                } else {
071                    setBlockFace(player.getHorizontalFacing());
072                }
073            } else {
074                setBlockFace(player.getHorizontalFacing());
075            }
076        }
077
078        this.getLevel().setBlock(block, this, true, true);
079        return true;
080    }
081
082    @PowerNukkitDifference(since = "1.4.0.0-PN", info = "Implemented")
083    @Override
084    public boolean isPowerSource() {
085        return true;
086    }
087
088    @PowerNukkitDifference(since = "1.4.0.0-PN", info = "Implemented")
089    @Override
090    public int getStrongPower(BlockFace side) {
091        return isPowered() && side == getBlockFace()? 15 : 0;
092    }
093
094    @PowerNukkitDifference(since = "1.4.0.0-PN", info = "Implemented")
095    @Override
096    public int getWeakPower(BlockFace face) {
097        return getStrongPower(face);
098    }
099
100    @PowerNukkitDifference(since = "1.4.0.0-PN", info = "Implemented")
101    @Override
102    public int onUpdate(int type) {
103        if (type == Level.BLOCK_UPDATE_SCHEDULED) {
104            RedstoneUpdateEvent ev = new RedstoneUpdateEvent(this);
105            PluginManager pluginManager = level.getServer().getPluginManager();
106            pluginManager.callEvent(ev);
107            if (ev.isCancelled()) {
108                return 0;
109            }
110
111            if (!isPowered()) {
112                level.getServer().getPluginManager().callEvent(new BlockRedstoneEvent(this, 0, 15));
113                setPowered(true);
114
115                if (level.setBlock(this, this)) {
116                    getSide(getBlockFace().getOpposite()).onUpdate(Level.BLOCK_UPDATE_REDSTONE);
117                    RedstoneComponent.updateAroundRedstone(getSide(getBlockFace().getOpposite()));
118                    level.scheduleUpdate(this, 2);
119                }
120            } else {
121                pluginManager.callEvent(new BlockRedstoneEvent(this, 15, 0));
122                setPowered(false);
123
124                level.setBlock(this, this);
125                getSide(getBlockFace().getOpposite()).onUpdate(Level.BLOCK_UPDATE_REDSTONE);
126                RedstoneComponent.updateAroundRedstone(getSide(getBlockFace().getOpposite()));
127            }
128            return Level.BLOCK_UPDATE_SCHEDULED;
129        } else if (type == Level.BLOCK_UPDATE_MOVED) {
130            onNeighborChange(getBlockFace());
131            return type;
132        }
133        return 0;
134    }
135
136    @Since("1.4.0.0-PN")
137    @PowerNukkitOnly
138    @Override
139    public void onNeighborChange(@Nonnull BlockFace side) {
140        Server server = level.getServer();
141        BlockFace blockFace = getBlockFace();
142        if (!server.isRedstoneEnabled() || isPowered() || side != blockFace || level.isUpdateScheduled(this, this)) {
143            return;
144        }
145
146        RedstoneUpdateEvent ev = new RedstoneUpdateEvent(this);
147        server.getPluginManager().callEvent(ev);
148        if (ev.isCancelled()) {
149            return;
150        }
151
152        level.scheduleUpdate(this, 5);
153    }
154

156    public boolean canHarvestWithHand() {
157        return false;
158    }

161    public int getToolType() {
162        return ItemTool.TYPE_PICKAXE;
163    }

168    }

172        return 3.5;
173    }
174
175    @Override
176    public double getResistance() {
177        return 17.5;
178    }

181    @Since("1.4.0.0-PN")
182    public boolean isPowered() {
183        return getBooleanValue(POWERED);
184    }
185    
186    @PowerNukkitOnly
187    @Since("1.4.0.0-PN")
188    public void setPowered(boolean powered) {
189        setBooleanValue(POWERED, powered);
190    }
191    
192    @Override
193    public BlockFace getBlockFace() {
194        return getPropertyValue(FACING_DIRECTION);
195    }
196
197    @Since("1.4.0.0-PN")
198    @PowerNukkitOnly
199    @Override
200    public void setBlockFace(BlockFace face) {
201        setPropertyValue(FACING_DIRECTION, face);
202    }
