015/**
016 * @author Nukkit Project Team
017 */
018@PowerNukkitDifference(info = "Implements RedstoneComponent.", since = "1.4.0.0-PN")
019public class BlockRailActivator extends BlockRail implements RedstoneComponent {
020
021    public BlockRailActivator(int meta) {
022        super(meta);
023    }
024
025    public BlockRailActivator() {
026        this(0);
027        canBePowered = true;
028    }
029
030    @Override
031    public String getName() {
032        return "Activator Rail";
033    }
034
035    @Override
036    public int getId() {
037        return ACTIVATOR_RAIL;
038    }
039
040    @Since("1.4.0.0-PN")
041    @PowerNukkitOnly
042    @Nonnull
043    @Override
044    public BlockProperties getProperties() {
045        return ACTIVABLE_PROPERTIES;
046    }
047
048    @PowerNukkitDifference(info = "Using new method for checking if powered", since = "1.4.0.0-PN")
049    @Override
050    public int onUpdate(int type) {
051        if (type == Level.BLOCK_UPDATE_NORMAL || type == Level.BLOCK_UPDATE_REDSTONE || type == Level.BLOCK_UPDATE_SCHEDULED) {
052            super.onUpdate(type);
053            boolean wasPowered = isActive();
054            boolean isPowered = this.isGettingPower()
055                    || checkSurrounding(this, true, 0)
056                    || checkSurrounding(this, false, 0);
057            boolean hasUpdate = false;
058
059            if (wasPowered != isPowered) {
060                setActive(isPowered);
061                hasUpdate = true;
062            }
063
064            if (hasUpdate) {
065                level.updateAround(down());
066                if (getOrientation().isAscending()) {
067                    level.updateAround(up());
068                }
069            }
070            return type;
071        }
072        return 0;
073    }
074
075    /**
076     * Check the surrounding of the rail
077     *
078     * @param pos      The rail position
079     * @param relative The relative of the rail that will be checked
080     * @param power    The count of the rail that had been counted
081     * @return Boolean of the surrounding area. Where the powered rail on!
082     */
083    protected boolean checkSurrounding(Vector3 pos, boolean relative, int power) {
084        if (power >= 8) {
085            return false;
086        }
087        int dx = pos.getFloorX();
088        int dy = pos.getFloorY();
089        int dz = pos.getFloorZ();
090
091        BlockRail block;
092        Block block2 = level.getBlock(new Vector3(dx, dy, dz));
093
094        if (Rail.isRailBlock(block2)) {
095            block = (BlockRail) block2;
096        } else {
097            return false;
098        }
099
100        Rail.Orientation base = null;
101        boolean onStraight = true;
102
103        switch (block.getOrientation()) {
104            case STRAIGHT_NORTH_SOUTH:
105                if (relative) {
106                    dz++;
107                } else {
108                    dz--;
109                }
110                break;
111            case STRAIGHT_EAST_WEST:
112                if (relative) {
113                    dx--;
114                } else {
115                    dx++;
116                }
117                break;
118            case ASCENDING_EAST:
119                if (relative) {
120                    dx--;
121                } else {
122                    dx++;
123                    dy++;
124                    onStraight = false;
125                }
126                base = Rail.Orientation.STRAIGHT_EAST_WEST;
127                break;
128            case ASCENDING_WEST:
129                if (relative) {
130                    dx--;
131                    dy++;
132                    onStraight = false;
133                } else {
134                    dx++;
135                }
136                base = Rail.Orientation.STRAIGHT_EAST_WEST;
137                break;
138            case ASCENDING_NORTH:
139                if (relative) {
140                    dz++;
141                } else {
142                    dz--;
143                    dy++;
144                    onStraight = false;
145                }
146                base = Rail.Orientation.STRAIGHT_NORTH_SOUTH;
147                break;
148            case ASCENDING_SOUTH:
149                if (relative) {
150                    dz++;
151                    dy++;
152                    onStraight = false;
153                } else {
154                    dz--;
155                }
156                base = Rail.Orientation.STRAIGHT_NORTH_SOUTH;
157                break;
158            default:
159                return false;
160        }
161
162        return canPowered(new Vector3(dx, dy, dz), base, power, relative)
163                || onStraight && canPowered(new Vector3(dx, dy - 1., dz), base, power, relative);
164    }
165
166    @PowerNukkitDifference(info = "Using new method for checking if powered", since = "1.4.0.0-PN")
167    protected boolean canPowered(Vector3 pos, Rail.Orientation state, int power, boolean relative) {
168        Block block = level.getBlock(pos);
169
170        if (!(block instanceof BlockRailActivator)) {
171            return false;
172        }
173
174        Rail.Orientation base = ((BlockRailActivator) block).getOrientation();
175
176        return (state != Rail.Orientation.STRAIGHT_EAST_WEST
177                || base != Rail.Orientation.STRAIGHT_NORTH_SOUTH
178                && base != Rail.Orientation.ASCENDING_NORTH
179                && base != Rail.Orientation.ASCENDING_SOUTH)
180                && (state != Rail.Orientation.STRAIGHT_NORTH_SOUTH
181                || base != Rail.Orientation.STRAIGHT_EAST_WEST
182                && base != Rail.Orientation.ASCENDING_EAST
183                && base != Rail.Orientation.ASCENDING_WEST)
184                && (this.isGettingPower() || checkSurrounding(pos, relative, power + 1));
185    }
186
187    @Override
188    public boolean isActive() {
189        return getBooleanValue(ACTIVE);
190    }
191
192    @Since("1.4.0.0-PN")
193    @PowerNukkitOnly
194    @Override
195    public OptionalBoolean isRailActive() {
196        return OptionalBoolean.of(getBooleanValue(ACTIVE));
197    }
198
199    @Since("1.4.0.0-PN")
200    @PowerNukkitOnly
201    @Override
202    public void setRailActive(boolean active) {
203        setBooleanValue(ACTIVE, active);
204    }
