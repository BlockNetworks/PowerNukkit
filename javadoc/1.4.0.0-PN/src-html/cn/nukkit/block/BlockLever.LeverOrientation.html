107    @Override
108    public boolean onActivate(@Nonnull Item item, Player player) {
109        this.level.getServer().getPluginManager().callEvent(new BlockRedstoneEvent(this, isPowerOn() ? 15 : 0, isPowerOn() ? 0 : 15));
110        toggleBooleanProperty(OPEN);
111
112        this.getLevel().setBlock(this, this, false, true);
113        this.getLevel().addSound(this, Sound.RANDOM_CLICK, 0.8f, isPowerOn() ? 0.58f : 0.5f );
114
115        LeverOrientation orientation = getLeverOrientation();
116        BlockFace face = orientation.getFacing();
117
118        if (this.level.getServer().isRedstoneEnabled()) {
119            updateAroundRedstone();
120            RedstoneComponent.updateAroundRedstone(getSide(face.getOpposite()), face);
121        }
122        return true;
123    }

125    @PowerNukkitDifference(info = "Now, can be placed on solid blocks", since= "1.4.0.0-PN")
126    @Override
127    public int onUpdate(int type) {
128        if (type == Level.BLOCK_UPDATE_NORMAL) {
129            BlockFace blockFace = getLeverOrientation().getFacing().getOpposite();
130            Block side = this.getSide(blockFace);
131            if (!isSupportValid(side, blockFace.getOpposite())) {
132                this.level.useBreakOn(this);
133            }
134        }
135        return 0;
136    }

138    @PowerNukkitDifference(info = "Allows to be placed on walls", since = "1.3.0.0-PN")
139    @PowerNukkitDifference(info = "Now, can be placed on solid blocks and always returns false if the placement fails", since= "1.4.0.0-PN")
140    @Override
141    public boolean place(@Nonnull Item item, @Nonnull Block block, @Nonnull Block target, @Nonnull BlockFace face, double fx, double fy, double fz, Player player) {
142        if (target.canBeReplaced()) {
143            target = target.down();
144            face = BlockFace.UP;
145        }
146        
147        if (!isSupportValid(target, face)) {
148            return false;
149        }
150        setLeverOrientation(LeverOrientation.forFacings(face, player.getHorizontalFacing()));
151        return this.getLevel().setBlock(block, this, true, true);
152    }

154    /**
155     * Check if the given block and its block face is a valid support for a lever
156     * @param support The block that the lever is being placed against
157     * @param face The face that the torch will be touching the block
158     * @return If the support and face combinations can hold the lever
159     */
160    @PowerNukkitOnly
161    @Since("1.4.0.0-PN")
162    public static boolean isSupportValid(Block support, BlockFace face) {
163        switch (support.getId()) {
164            case FARMLAND:
165            case GRASS_PATH:
166                return true;
167            default:
168        }
169        
170        if (face == BlockFace.DOWN) {
171            return support.isSolid(BlockFace.DOWN) && (support.isFullBlock() || !support.isTransparent());
172        }
173        
174        if (support.isSolid(face)) {
175            return true;
176        }
177
178        if (support instanceof BlockWallBase || support instanceof BlockFence) {
179            return face == BlockFace.UP;
180        }
181        
182        return false;
183    }

185    @Override
186    @PowerNukkitDifference(info = "Update redstone", since = "1.4.0.0-PN")
187    public boolean onBreak(Item item) {
188        this.getLevel().setBlock(this, Block.get(BlockID.AIR), true, true);
189
190        if (isPowerOn()) {
191            BlockFace face = getLeverOrientation().getFacing();
192            this.level.updateAround(this.getLocation().getSide(face.getOpposite()));
193
194            if (level.getServer().isRedstoneEnabled()) {
195                updateAroundRedstone();
196                RedstoneComponent.updateAroundRedstone(getSide(face.getOpposite()), face);
197            }
198        }
199        return true;
200    }

216    public enum LeverOrientation {
217        DOWN_X(0, "down_x", BlockFace.DOWN),
218        EAST(1, "east", BlockFace.EAST),
219        WEST(2, "west", BlockFace.WEST),
220        SOUTH(3, "south", BlockFace.SOUTH),
221        NORTH(4, "north", BlockFace.NORTH),
222        UP_Z(5, "up_z", BlockFace.UP),
223        UP_X(6, "up_x", BlockFace.UP),
224        DOWN_Z(7, "down_z", BlockFace.DOWN);
225
226        private static final LeverOrientation[] META_LOOKUP = new LeverOrientation[values().length];
227        private final int meta;
228        private final String name;
229        private final BlockFace facing;
230
231        LeverOrientation(int meta, String name, BlockFace face) {
232            this.meta = meta;
233            this.name = name;
234            this.facing = face;
235        }
236

246            return this.name;
247        }
248
249        public static LeverOrientation byMetadata(int meta) {
250            if (meta < 0 || meta >= META_LOOKUP.length) {
251                meta = 0;
252            }
253
254            return META_LOOKUP[meta];
255        }
256
257        public static LeverOrientation forFacings(BlockFace clickedSide, BlockFace playerDirection) {
258            switch (clickedSide) {
259                case DOWN:
260                    switch (playerDirection.getAxis()) {
261                        case X:
262                            return DOWN_X;
263
264                        case Z:
265                            return DOWN_Z;
266
267                        default:
268                            throw new IllegalArgumentException("Invalid entityFacing " + playerDirection + " for facing " + clickedSide);
269                    }
270
271                case UP:
272                    switch (playerDirection.getAxis()) {
273                        case X:
274                            return UP_X;
275
276                        case Z:
277                            return UP_Z;
278
279                        default:
280                            throw new IllegalArgumentException("Invalid entityFacing " + playerDirection + " for facing " + clickedSide);
281                    }
282
283                case NORTH:
284                    return NORTH;
285
286                case SOUTH:
287                    return SOUTH;
288
289                case WEST:
290                    return WEST;
291
292                case EAST:
293                    return EAST;
294
295                default:
296                    throw new IllegalArgumentException("Invalid facing: " + clickedSide);
297            }
298        }
299
300        public String getName() {
301            return this.name;
302        }
303
304        static {
305            for (LeverOrientation face : values()) {
306                META_LOOKUP[face.getMetadata()] = face;
307            }
308        }
