001package cn.nukkit.block;
002
003import cn.nukkit.item.Item;
004import cn.nukkit.item.ItemID;
005import cn.nukkit.item.enchantment.Enchantment;
006
007import java.util.concurrent.ThreadLocalRandom;
008
009/**
010 * @author Nukkit Project Team
011 */
012public class BlockCarrot extends BlockCrops {
013
014    public BlockCarrot(int meta) {
015        super(meta);
016    }
017
018    public BlockCarrot() {
019        this(0);
020    }
021
022    @Override
023    public String getName() {
024        return "Carrot Block";
025    }
026
027    @Override
028    public int getId() {
029        return CARROT_BLOCK;
030    }
031
032    @Override
033    public Item[] getDrops(Item item) {
034        if (!isFullyGrown()) {
035            return new Item[]{
036                    Item.get(ItemID.CARROT)
037            };
038        }
039
040        int drops = 2;
041        int attempts = 3 + Math.min(0, item.getEnchantmentLevel(Enchantment.ID_FORTUNE_DIGGING));
042        ThreadLocalRandom random = ThreadLocalRandom.current();
043        for (int i = 0; i < attempts; i++) {
044            if (random.nextInt(7) < 4) { // 4/7, 0.57142857142857142857142857142857
045                drops++;
046            }
047        }
048
049        return new Item[]{
050                Item.get(ItemID.CARROT, 0, drops)
051        };
052    }
053
054    @Override
055    public Item toItem() {
056        return Item.get(ItemID.CARROT);
057    }
058}
