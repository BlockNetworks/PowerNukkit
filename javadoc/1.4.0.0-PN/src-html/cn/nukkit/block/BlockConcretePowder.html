001package cn.nukkit.block;
002
003import cn.nukkit.Player;
004import cn.nukkit.api.PowerNukkitDifference;
005import cn.nukkit.item.Item;
006import cn.nukkit.item.ItemTool;
007import cn.nukkit.level.Level;
008import cn.nukkit.math.BlockFace;
009
010import javax.annotation.Nonnull;
011
012/**
013 * @author CreeperFace
014 * @since 2.6.2017
015 */
016@PowerNukkitDifference(info = "Extends BlockFallableMeta instead of BlockFallable")
017public class BlockConcretePowder extends BlockFallableMeta {
018    public BlockConcretePowder() {
019        // Does nothing
020    }
021
022    public BlockConcretePowder(int meta) {
023        super(meta);
024    }
025
026    @Override
027    public int getId() {
028        return CONCRETE_POWDER;
029    }
030
031    @Override
032    public String getName() {
033        return "Concrete Powder";
034    }
035
036    @Override
037    public double getResistance() {
038        return 2.5;
039    }
040
041    @Override
042    public double getHardness() {
043        return 0.5;
044    }
045
046    @Override
047    public int getToolType() {
048        return ItemTool.TYPE_SHOVEL;
049    }
050    
051    @Override
052    public int onUpdate(int type) {
053        if (type == Level.BLOCK_UPDATE_NORMAL) {
054            super.onUpdate(Level.BLOCK_UPDATE_NORMAL);
055
056            for (int side = 1; side <= 5; side++) {
057                Block block = this.getSide(BlockFace.fromIndex(side));
058                if (block.getId() == Block.WATER || block.getId() == Block.STILL_WATER) {
059                    this.level.setBlock(this, Block.get(Block.CONCRETE, getDamage()), true, true);
060                }
061            }
062
063            return Level.BLOCK_UPDATE_NORMAL;
064        }
065        return 0;
066    }
067
068    @Override
069    public boolean place(@Nonnull Item item, @Nonnull Block b, @Nonnull Block target, @Nonnull BlockFace face, double fx, double fy, double fz, Player player) {
070        boolean concrete = false;
071
072        for (int side = 1; side <= 5; side++) {
073            Block block = this.getSide(BlockFace.fromIndex(side));
074            if (block.getId() == Block.WATER || block.getId() == Block.STILL_WATER) {
075                concrete = true;
076                break;
077            }
078        }
079
080        if (concrete) {
081            this.level.setBlock(this, Block.get(Block.CONCRETE, this.getDamage()), true, true);
082        } else {
083            this.level.setBlock(this, this, true, true);
084        }
085
086        return true;
087    }
088}
