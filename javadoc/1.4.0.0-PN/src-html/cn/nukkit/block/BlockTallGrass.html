014
015import javax.annotation.Nonnull;

020/**
021 * @author Angelic47 (Nukkit Project)
022 */
023public class BlockTallGrass extends BlockFlowable {
024
025    public BlockTallGrass() {
026        this(1);
027    }
028
029    public BlockTallGrass(int meta) {
030        super(meta);
031    }
032
033    @Override
034    public int getId() {
035        return TALL_GRASS;
036    }
037
038    @Override
039    public String getName() {
040        String[] names = new String[]{
041                "Grass",
042                "Grass",
043                "Fern",
044                "Fern"
045        };
046        return names[this.getDamage() & 0x03];
047    }
048
049    @Override
050    public boolean canBeActivated() {
051        return true;
052    }
053
054    @Override
055    public boolean canBeReplaced() {
056        return true;
057    }
058
059    @Override
060    public int getBurnChance() {
061        return 60;
062    }
063
064    @Override
065    public int getBurnAbility() {
066        return 100;
067    }
068
069    @Override
070    public boolean place(@Nonnull Item item, @Nonnull Block block, @Nonnull Block target, @Nonnull BlockFace face, double fx, double fy, double fz, Player player) {
071        if (BlockSweetBerryBush.isSupportValid(down())) {
072            this.getLevel().setBlock(block, this, true);
073            return true;
074        }
075        return false;
076    }
077    
078    @PowerNukkitDifference(since = "1.4.0.0-PN", info = "Will break on block update if the supporting block is invalid")
079    @Override
080    public int onUpdate(int type) {
081        if (type == Level.BLOCK_UPDATE_NORMAL) {
082            if (!BlockSweetBerryBush.isSupportValid(down(1, 0))) {
083                this.getLevel().useBreakOn(this);
084                return Level.BLOCK_UPDATE_NORMAL;
085            }
086        }
087        return 0;
088    }
089
090    @Override
091    public boolean onActivate(@Nonnull Item item, Player player) {
092        if (item.isFertilizer()) {
093            Block up = this.up();
094
095            if (up.getId() == AIR) {
096                DoublePlantType type;
097
098                switch (this.getDamage()) {
099                    case 0:
100                    case 1:
101                        type = DoublePlantType.TALL_GRASS;
102                        break;
103                    case 2:
104                    case 3:
105                        type = DoublePlantType.LARGE_FERN;
106                        break;
107                    default:
108                        type = null;
109                }
110
111                if (type != null) {
112                    if (player != null && !player.isCreative()) {
113                        item.count--;
114                    }
115
116                    BlockDoublePlant doublePlant = (BlockDoublePlant) Block.get(BlockID.DOUBLE_PLANT);
117                    doublePlant.setDoublePlantType(type);
118                    doublePlant.setTopHalf(false);
119
120                    this.level.addParticle(new BoneMealParticle(this));
121                    this.level.setBlock(this, doublePlant, true, false);
122
123                    doublePlant.setTopHalf(true);
124                    this.level.setBlock(up, doublePlant, true);
125                    this.level.updateAround(this);
126                }
127            }
128
129            return true;
130        }
131
132        return false;
133    }
134
135    @Override
136    public Item[] getDrops(Item item) {
137        // https://minecraft.gamepedia.com/Fortune#Grass_and_ferns
138        List<Item> drops = new ArrayList<>(2);
139        if (item.isShears()) {
140            drops.add(getCurrentState().asItemBlock());
141        }
142
143        ThreadLocalRandom random = ThreadLocalRandom.current();
144        if (random.nextInt(8) == 0) {
145            Enchantment fortune = item.getEnchantment(Enchantment.ID_FORTUNE_DIGGING);
146            int fortuneLevel = fortune != null? fortune.getLevel() : 0;
147            int amount = fortuneLevel == 0? 1 : 1 + random.nextInt(fortuneLevel * 2);
148            drops.add(Item.get(ItemID.WHEAT_SEEDS, 0, amount));
149        }
150        
151        return drops.toArray(Item.EMPTY_ARRAY);
152    }
153
154    @Override
155    public int getToolType() {
156        return ItemTool.TYPE_SHEARS;
157    }
158
159    @Override
160    public BlockColor getColor() {
161        return BlockColor.FOLIAGE_BLOCK_COLOR;
162    }
163}
