001package cn.nukkit.block;
002
003import cn.nukkit.item.Item;
004import cn.nukkit.item.ItemBlock;
005import cn.nukkit.item.ItemTool;
006import cn.nukkit.math.NukkitRandom;
007import cn.nukkit.utils.BlockColor;
008
009/**
010 * @author Pub4Game
011 * @since 28.01.2016
012 */
013public class BlockHugeMushroomBrown extends BlockSolidMeta {
014
015    public BlockHugeMushroomBrown() {
016        this(0);
017    }
018
019    public BlockHugeMushroomBrown(int meta) {
020        super(meta);
021    }
022
023    @Override
024    public String getName() {
025        return "Brown Mushroom Block";
026    }
027
028    @Override
029    public int getId() {
030        return BROWN_MUSHROOM_BLOCK;
031    }
032
033    @Override
034    public int getToolType() {
035        return ItemTool.TYPE_AXE;
036    }
037
038    @Override
039    public double getHardness() {
040        return 0.2;
041    }
042
043    @Override
044    public double getResistance() {
045        return 1;
046    }
047
048    @Override
049    public Item[] getDrops(Item item) {
050        if (new NukkitRandom().nextRange(1, 20) == 0) {
051            return new Item[]{
052                    new ItemBlock(Block.get(BlockID.BROWN_MUSHROOM))
053            };
054        } else {
055            return Item.EMPTY_ARRAY;
056        }
057    }
058
059    @Override
060    public boolean canSilkTouch() {
061        return true;
062    }
063
064    @Override
065    public BlockColor getColor() {
066        return BlockColor.DIRT_BLOCK_COLOR;
067    }
068}
