001package cn.nukkit.block;
002
003import cn.nukkit.Player;
004import cn.nukkit.api.PowerNukkitOnly;
005import cn.nukkit.api.Since;
006import cn.nukkit.blockproperty.BlockProperties;
007import cn.nukkit.blockproperty.CommonBlockProperties;
008import cn.nukkit.blockstate.BlockState;
009import cn.nukkit.item.Item;
010import cn.nukkit.item.ItemTool;
011import cn.nukkit.math.BlockFace;
012
013import javax.annotation.Nonnull;
014
015import static cn.nukkit.blockproperty.CommonBlockProperties.PILLAR_AXIS;
016
017@PowerNukkitOnly
018@Since("1.4.0.0-PN")
019public abstract class BlockLog extends BlockSolidMeta {
020    @PowerNukkitOnly
021    @Since("1.4.0.0-PN")
022    protected static final BlockProperties PILLAR_PROPERTIES = new BlockProperties(CommonBlockProperties.PILLAR_AXIS);
023
024    @PowerNukkitOnly
025    @Since("1.4.0.0-PN")
026    protected BlockLog(int meta) {
027        super(meta);
028    }
029
030    @Since("1.4.0.0-PN")
031    @PowerNukkitOnly
032    @Nonnull
033    @Override
034    public abstract BlockProperties getProperties();
035
036    @PowerNukkitOnly
037    @Since("1.4.0.0-PN")
038    protected abstract BlockState getStrippedState();
039
040    @PowerNukkitOnly
041    @Since("1.4.0.0-PN")
042    public BlockFace.Axis getPillarAxis() {
043        return getPropertyValue(PILLAR_AXIS);
044    }
045
046    @PowerNukkitOnly
047    @Since("1.4.0.0-PN")
048    public void setPillarAxis(BlockFace.Axis axis) {
049        setPropertyValue(PILLAR_AXIS, axis);
050    }
051
052    @Override
053    public boolean place(@Nonnull Item item, @Nonnull Block block, @Nonnull Block target, @Nonnull BlockFace face, double fx, double fy, double fz, Player player) {
054        setPillarAxis(face.getAxis());
055        getLevel().setBlock(block, this, true, true);
056        return true;
057    }
058
059    @Override
060    public boolean canBeActivated() {
061        return true;
062    }
063
064    @Override
065    public boolean onActivate(@Nonnull Item item, Player player) {
066        if (item.isAxe()) {
067            Block strippedBlock = getStrippedState().getBlock(this);
068            item.useOn(this);
069            this.level.setBlock(this, strippedBlock, true, true);
070            return true;
071        }
072        return false;
073    }
074
075    @Override
076    public int getToolType() {
077        return ItemTool.TYPE_AXE;
078    }
079}
