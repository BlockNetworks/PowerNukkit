029@PowerNukkitOnly
030@Since("1.4.0.0-PN")
031public class BlockTarget extends BlockTransparent implements RedstoneComponent, BlockEntityHolder<BlockEntityTarget> {
032
033    @PowerNukkitOnly
034    @Since("1.4.0.0-PN")
035    public BlockTarget() {
036        // Does nothing
037    }
038
039    @Override
040    public int getId() {
041        return TARGET;
042    }
043
044    @Override
045    public String getName() {
046        return "Target";
047    }
048
049    @Since("1.4.0.0-PN")
050    @PowerNukkitOnly
051    @Nonnull
052    @Override
053    public Class<? extends BlockEntityTarget> getBlockEntityClass() {
054        return BlockEntityTarget.class;
055    }
056
057    @Since("1.4.0.0-PN")
058    @PowerNukkitOnly
059    @Nonnull
060    @Override
061    public String getBlockEntityType() {
062        return BlockEntity.TARGET;
063    }
064
065    @Override
066    public boolean isPowerSource() {
067        return true;
068    }
069
070    @Override
071    public int getWeakPower(BlockFace face) {
072        BlockEntityTarget target = getBlockEntity();
073        return target == null? 0 : target.getActivePower();
074    }
075
076    @PowerNukkitOnly
077    @Since("1.4.0.0-PN")
078    public boolean activatePower(int power) {
079        return activatePower(power, 4 * 2);
080    }
081
082    @PowerNukkitOnly
083    @Since("1.4.0.0-PN")
084    public boolean activatePower(int power, int ticks) {
085        Level level = getLevel();
086        if (power <= 0 || ticks <= 0) {
087            return deactivatePower();
088        }
089        
090        if (!level.getServer().isRedstoneEnabled()) {
091            return false;
092        }
093
094        BlockEntityTarget target = getOrCreateBlockEntity();
095        int previous = target.getActivePower();
096        level.cancelSheduledUpdate(this, this);
097        level.scheduleUpdate(this, ticks);
098        target.setActivePower(power);
099        if (previous != power) {
100            updateAroundRedstone();
101        }
102        return true;
103    }
104
105    @PowerNukkitOnly
106    @Since("1.4.0.0-PN")
107    public boolean deactivatePower() {
108        BlockEntityTarget target = getBlockEntity();
109        if (target != null) {
110            int currentPower = target.getActivePower();
111            target.setActivePower(0);
112            target.close();
113            if (currentPower != 0 && level.getServer().isRedstoneEnabled()) {
114                updateAroundRedstone();
115            }
116            return true;
117        }
118        return false;
119    }
120
121    @Override
122    public int onUpdate(int type) {
123        if (type == Level.BLOCK_UPDATE_SCHEDULED) {
124            deactivatePower();
125            return type;
126        }
127        return 0;
128    }
129
130    @Since("1.4.0.0-PN")
131    @PowerNukkitOnly
132    @Override
133    public boolean onProjectileHit(@Nonnull Entity projectile, @Nonnull Position position, @Nonnull Vector3 motion) {
134        int ticks = 8;
135        if (projectile instanceof EntityArrow || projectile instanceof EntityThrownTrident) {
136            ticks = 20;
137        }
138
139        MovingObjectPosition intercept = calculateIntercept(position, position.add(motion.multiply(2)));
140        if (intercept == null) {
141            return false;
142        }
143
144        BlockFace faceHit = intercept.getFaceHit();
145        if (faceHit == null) {
146            return false;
147        }
148
149        Vector3 hitVector = intercept.hitVector.subtract(x*2, y*2, z*2);
150        List<Axis> axes = new ArrayList<>(Arrays.asList(Axis.values()));
151        axes.remove(faceHit.getAxis());
152        
153        double[] coords = new double[] { hitVector.getAxis(axes.get(0)), hitVector.getAxis(axes.get(1)) };
154
155        for (int i = 0; i < 2 ; i++) {
156            if (coords[i] == 0.5) {
157                coords[i] = 1;
158            } else if (coords[i] <= 0 || coords[i] >= 1) {
159                coords[i] = 0;
160            } else if (coords[i] < 0.5) {
161                coords[i] *= 2;
162            } else {
163                coords[i] = (coords[i] / (-0.5)) + 2;
164            }
165        }
166
167        double scale = (coords[0] + coords[1]) / 2;
168        activatePower(NukkitMath.ceilDouble(16 * scale), ticks);
169        return true;
170    }
