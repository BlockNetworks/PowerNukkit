001package cn.nukkit.block;
002
003import cn.nukkit.Player;
004import cn.nukkit.api.PowerNukkitDifference;
005import cn.nukkit.api.PowerNukkitOnly;
006import cn.nukkit.api.Since;

022import javax.annotation.Nonnull;
023import javax.annotation.Nullable;
024
025@PowerNukkitOnly
026public class BlockLectern extends BlockTransparentMeta implements RedstoneComponent, Faceable, BlockEntityHolder<BlockEntityLectern> {
027    @PowerNukkitOnly
028    public BlockLectern() {
029        this(0);
030    }
031
032    @PowerNukkitOnly
033    public BlockLectern(int meta) {
034        super(meta);
035    }
036
037    @Override
038    public String getName() {
039        return "Lectern";
040    }
041
042    @Override
043    public int getId() {
044        return LECTERN;
045    }
046
047    @Since("1.4.0.0-PN")
048    @PowerNukkitOnly
049    @Nonnull
050    @Override
051    public Class<? extends BlockEntityLectern> getBlockEntityClass() {
052        return BlockEntityLectern.class;
053    }
054
055    @PowerNukkitOnly
056    @Since("1.4.0.0-PN")
057    @Nonnull
058    @Override
059    public String getBlockEntityType() {
060        return BlockEntity.LECTERN;
061    }
062
063    @PowerNukkitOnly
064    @Override
065    public int getWaterloggingLevel() {
066        return 1;
067    }
068
069    @Override
070    public boolean canBeActivated() {
071        return true;
072    }
073
074    @Override
075    public double getHardness() {
076        return 2;
077    }
078
079    @Override
080    public double getResistance() {
081        return 12.5;
082    }
083
084    @Override
085    public int getToolType() {
086        return ItemTool.TYPE_AXE;
087    }
088
089    @Override
090    public double getMaxY() {
091        return y + 0.89999;
092    }
093
094    @Override
095    public boolean hasComparatorInputOverride() {
096        return true;
097    }
098
099    @Override
100    public int getComparatorInputOverride() {
101        int power = 0;
102        int page = 0;
103        int maxPage = 0;
104        BlockEntityLectern lectern = getBlockEntity();
105        if (lectern != null && lectern.hasBook()) {
106            maxPage = lectern.getTotalPages();
107            page = lectern.getLeftPage() + 1;
108            power = (int) (((float) page / maxPage) * 16);
109        }
110        return power;
111    }
112    
113    @Override
114    public BlockFace getBlockFace() {
115        return BlockFace.fromHorizontalIndex(getDamage() & 0b11);
116    }
117
118    @Override
119    @PowerNukkitOnly
120    @Since("1.3.0.0-PN")
121    public void setBlockFace(BlockFace face) {
122        int horizontalIndex = face.getHorizontalIndex();
123        if (horizontalIndex >= 0) {
124            setDamage(getDamage() & (DATA_MASK ^ 0b11) | (horizontalIndex & 0b11));
125        }
126    }
127    
128    @Override
129    public boolean place(@Nonnull Item item, @Nonnull Block block, @Nonnull Block target, @Nonnull BlockFace face, double fx, double fy, double fz, @Nullable Player player) {
130        setBlockFace(player != null ? player.getDirection().getOpposite() : BlockFace.SOUTH);
131        return BlockEntityHolder.setBlockAndCreateEntity(this) != null;
132    }
133
134    @Since("1.4.0.0-PN")
135    @PowerNukkitOnly
136    @Override
137    public int onTouch(@Nullable Player player, Action action) {
138        if (action == Action.LEFT_CLICK_BLOCK && (player == null || (!player.isCreative() && !player.isSpectator()))) {
139            dropBook(player);
140        }
141        return super.onTouch(player, action);
142    }
143
144    @Override
145    public boolean onActivate(@Nonnull Item item, @Nullable Player player) {
146        BlockEntityLectern lectern = getOrCreateBlockEntity();
147        Item currentBook = lectern.getBook();
148        if (!currentBook.isNull()) {
149            return false;
150        }
151
152        if (item.getId() != ItemID.WRITTEN_BOOK && item.getId() != ItemID.BOOK_AND_QUILL) {
153            return false;
154        }
155        
156        if (player == null || !player.isCreative()) {
157            item.count--;
158        }
159
160        Item newBook = item.clone();
161        newBook.setCount(1);
162        lectern.setBook(newBook);
163        lectern.spawnToAll();
164        this.getLevel().addSound(this.add(0.5, 0.5, 0.5), Sound.ITEM_BOOK_PUT);
165        return true;
166    }
167
168    @Override
169    public boolean isPowerSource() {
170        return true;
171    }
172
173    @PowerNukkitOnly
174    public boolean isActivated() {
175        return (this.getDamage() & 0x04) == 0x04;
176    }
177
178    @PowerNukkitOnly
179    public void setActivated(boolean activated) {
180        if (activated) {
181            setDamage(getDamage() | 0x04);
182        } else {
183            setDamage(getDamage() ^ 0x04);
184        }
185    }
186
187    @PowerNukkitDifference(info = "Down side is strongly powered.", since = "1.4.0.0-PN")
188    public void executeRedstonePulse() {
189        if (isActivated()) {
190            level.cancelSheduledUpdate(this, this);
191        } else {
192            this.level.getServer().getPluginManager().callEvent(new BlockRedstoneEvent(this, 0, 15));
193        }
194
195        level.scheduleUpdate(this, this, 4);
196        setActivated(true);
197        level.setBlock(this, this, true, false);
198        level.addSound(this.add(0.5, 0.5, 0.5), Sound.ITEM_BOOK_PAGE_TURN);
199
200        updateAroundRedstone();
201        RedstoneComponent.updateAroundRedstone(getSide(BlockFace.DOWN), BlockFace.UP);
202    }
203
204    @Override
205    public int getWeakPower(BlockFace face) {
206        return isActivated()? 15 : 0;
207    }
208
209    @Override
210    @PowerNukkitDifference(info = "Down side is strongly powered.", since = "1.4.0.0-PN")
211    public int getStrongPower(BlockFace face) {
212        return face == BlockFace.DOWN ? this.getWeakPower(face) : 0;
213    }
214
215    @Override
216    @PowerNukkitDifference(info = "Down side is strongly powered.", since = "1.4.0.0-PN")
217    public int onUpdate(int type) {
218        if (type == Level.BLOCK_UPDATE_SCHEDULED) {
219            if (isActivated()) {
220                this.level.getServer().getPluginManager().callEvent(new BlockRedstoneEvent(this, 15, 0));
221
222                setActivated(false);
223                level.setBlock(this, this, true, false);
224                updateAroundRedstone();
225                RedstoneComponent.updateAroundRedstone(getSide(BlockFace.DOWN), BlockFace.UP);
226            }
227
228            return Level.BLOCK_UPDATE_SCHEDULED;
229        }
230
231        return 0;
232    }
233
234    @Override
235    public BlockColor getColor() {
236        return BlockColor.WOOD_BLOCK_COLOR;
237    }
238
239    @PowerNukkitOnly
240    public void dropBook(Player player) {
241        BlockEntityLectern lectern = getBlockEntity();
242        if (lectern == null) {
243            return;
244        }
245
246        Item book = lectern.getBook();
247        if (book.isNull()) {
248            return;
249        }
250        
251        LecternDropBookEvent dropBookEvent = new LecternDropBookEvent(player, lectern, book);
252        this.getLevel().getServer().getPluginManager().callEvent(dropBookEvent);
253        if (dropBookEvent.isCancelled()) {
254            return;
255        }
256        
257        lectern.setBook(Item.getBlock(BlockID.AIR));
258        lectern.spawnToAll();
259        this.level.dropItem(lectern.add(0.5f, 0.6f, 0.5f), dropBookEvent.getBook());
260    }
261}
