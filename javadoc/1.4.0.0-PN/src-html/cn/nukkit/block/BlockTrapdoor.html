121    //<editor-fold desc="pre-computing the bounding boxes" defaultstate="collapsed">
122    static {
123        for (int damage = 0; damage < boundingBoxDamage.length; damage++) {
124            AxisAlignedBB bb;
125            if (PROPERTIES.getBooleanValue(damage, OPEN.getName())) {
126                BlockFace face = (BlockFace) PROPERTIES.getValue(damage, TRAPDOOR_DIRECTION.getName());
127                face = face.getOpposite();
128                if (face.getAxisDirection() == AxisDirection.NEGATIVE) {
129                    bb = new SimpleAxisAlignedBB (
130                            0,
131                            0,
132                            0,
133                            1 + face.getXOffset() - (THICKNESS * face.getXOffset()),
134                            1,
135                            1 + face.getZOffset() - (THICKNESS * face.getZOffset())
136                    );
137                } else {
138                    bb = new SimpleAxisAlignedBB (
139                            face.getXOffset() - (THICKNESS * face.getXOffset()),
140                            0,
141                            face.getZOffset() - (THICKNESS * face.getZOffset()),
142                            1,
143                            1,
144                            1
145                    );
146                }
147            } else if (PROPERTIES.getBooleanValue(damage, UPSIDE_DOWN.getName())) {
148                bb = new SimpleAxisAlignedBB(
149                        0,
150                        1 - THICKNESS,
151                        0,
152                        1,
153                        1,
154                        1
155                );
156            } else {
157                bb = new SimpleAxisAlignedBB(
158                        0,
159                        0,
160                        0,
161                        1,
162                        0 + THICKNESS,
163                        1
164                );
165            }
166            
167            boundingBoxDamage[damage] = bb;
168        }
169    }

209    @PowerNukkitDifference(info = "Checking if the door was opened/closed manually and using new powered checks.", since = "1.4.0.0-PN")
210    @Override
211    public int onUpdate(int type) {
212        if (type == Level.BLOCK_UPDATE_REDSTONE && this.level.getServer().isRedstoneEnabled()) {
213            if ((this.isOpen() != this.isGettingPower()) && !this.getManualOverride()) {
214                if (this.isOpen() != this.isGettingPower()) {
215                    level.getServer().getPluginManager().callEvent(new BlockRedstoneEvent(this, this.isOpen() ? 15 : 0, this.isOpen() ? 0 : 15));
216
217                    this.setOpen(null, this.isGettingPower());
218                }
219            } else if (this.getManualOverride() && (this.isGettingPower() == this.isOpen())) {
220                this.setManualOverride(false);
221            }
222            return type;
223        }
224
225        return 0;
226    }

250    @PowerNukkitDifference(info = "Will return false if setBlock fails and the direction is relative to where the player is facing", since = "1.4.0.0-PN")
251    @Override
252    public boolean place(@Nonnull Item item, @Nonnull Block block, @Nonnull Block target, @Nonnull BlockFace face, double fx, double fy, double fz, @Nullable Player player) {
253        if (face.getAxis().isHorizontal()) {
254            setBlockFace(player == null? face : player.getDirection().getOpposite());
255            setTop(fy > 0.5);
256        } else {
257            setBlockFace(player.getDirection().getOpposite());
258            setTop(face != BlockFace.UP);
259        }
260
261        if (!this.getLevel().setBlock(block, this, true, true)) {
262            return false;
263        }
264
265        if (level.getServer().isRedstoneEnabled() && !this.isOpen() && this.isGettingPower()) {
266            this.setOpen(null, true);
267        }
268
269        return true;
270    }

282    @PowerNukkitDifference(info = "Returns false if setBlock fails", since = "1.4.0.0-PN")
283    @PowerNukkitDifference(info = "Using direct values, instead of toggling (fixes a redstone bug, that the door won't open). " +
284            "Also adding possibility to detect, whether a player or redstone recently opened/closed the door.", since = "1.4.0.0-PN")
285    @PowerNukkitOnly
286    public boolean setOpen(Player player, boolean open) {
287        if (open == this.isOpen()) {
288            return false;
289        }
290
291        DoorToggleEvent event = new DoorToggleEvent(this, player);
292        level.getServer().getPluginManager().callEvent(event);
293
294        if (event.isCancelled()) {
295            return false;
296        }
297
298        player = event.getPlayer();
299
300        setBooleanValue(OPEN, open);
301        if (!level.setBlock(this, this, true, true))
302            return false;
303
304        if (player != null) {
305            this.setManualOverride(this.isGettingPower() || isOpen());
306        }
307
308        playOpenCloseSound();
309        return true;
310    }
