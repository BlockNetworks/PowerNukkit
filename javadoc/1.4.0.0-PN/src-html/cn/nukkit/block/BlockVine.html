109    @Override
110    protected AxisAlignedBB recalculateBoundingBox() {
111        double f1 = 1;
112        double f2 = 1;
113        double f3 = 1;
114        double f4 = 0;
115        double f5 = 0;
116        double f6 = 0;
117        boolean flag = this.getDamage() > 0;
118        if ((this.getDamage() & 0x02) > 0) {
119            f4 = Math.max(f4, 0.0625);
120            f1 = 0;
121            f2 = 0;
122            f5 = 1;
123            f3 = 0;
124            f6 = 1;
125            flag = true;
126        }
127        if ((this.getDamage() & 0x08) > 0) {
128            f1 = Math.min(f1, 0.9375);
129            f4 = 1;
130            f2 = 0;
131            f5 = 1;
132            f3 = 0;
133            f6 = 1;
134            flag = true;
135        }
136        if ((this.getDamage() & 0x01) > 0) {
137            f3 = Math.min(f3, 0.9375);
138            f6 = 1;
139            f1 = 0;
140            f4 = 1;
141            f2 = 0;
142            f5 = 1;
143            flag = true;
144        }
145        if (!flag && this.up().isSolid()) {
146            f2 = Math.min(f2, 0.9375);
147            f5 = 1;
148            f1 = 0;
149            f4 = 1;
150            f3 = 0;
151            f6 = 1;
152        }
153        return new SimpleAxisAlignedBB(
154                this.x + f1,
155                this.y + f2,
156                this.z + f3,
157                this.x + f4,
158                this.y + f5,
159                this.z + f6
160        );
161    }

190    @Override
191    public int onUpdate(int type) {
192        if (type == Level.BLOCK_UPDATE_NORMAL) {
193            Block up = this.up();
194            Set<BlockFace> upFaces = up instanceof BlockVine ? ((BlockVine) up).getFaces() : null;
195            Set<BlockFace> faces = this.getFaces();
196            for (BlockFace face : BlockFace.Plane.HORIZONTAL) {
197                if (!this.getSide(face).isSolid() && (upFaces == null || !upFaces.contains(face))) {
198                    faces.remove(face);
199                }
200            }
201            if (faces.isEmpty() && !up.isSolid()) {
202                this.getLevel().useBreakOn(this, null, null, true);
203                return Level.BLOCK_UPDATE_NORMAL;
204            }
205            int meta = getMetaFromFaces(faces);
206            if (meta != this.getDamage()) {
207                this.level.setBlock(this, Block.get(VINE, meta), true);
208                return Level.BLOCK_UPDATE_NORMAL;
209            }
210        } else if (type == Level.BLOCK_UPDATE_RANDOM) {
211            Random random = ThreadLocalRandom.current();
212            if (random.nextInt(4) == 0) {
213                BlockFace face = BlockFace.random(random);
214                Block block = this.getSide(face);
215                int faceMeta = getMetaFromFace(face);
216                int meta = this.getDamage();
217
218                if (this.y < 255 && face == BlockFace.UP && block.getId() == AIR) {
219                    if (this.canSpread()) {
220                        for (BlockFace horizontalFace : BlockFace.Plane.HORIZONTAL) {
221                            if (random.nextBoolean() || !this.getSide(horizontalFace).getSide(face).isSolid()) {
222                                meta &= ~getMetaFromFace(horizontalFace);
223                            }
224                        }
225                        putVineOnHorizontalFace(block, meta, this);
226                    }
227                } else if (face.getHorizontalIndex() != -1 && (meta & faceMeta) != faceMeta) {
228                    if (this.canSpread()) {
229                        if (block.getId() == AIR) {
230                            BlockFace cwFace = face.rotateY();
231                            BlockFace ccwFace = face.rotateYCCW();
232                            Block cwBlock = block.getSide(cwFace);
233                            Block ccwBlock = block.getSide(ccwFace);
234                            int cwMeta = getMetaFromFace(cwFace);
235                            int ccwMeta = getMetaFromFace(ccwFace);
236                            boolean onCw = (meta & cwMeta) == cwMeta;
237                            boolean onCcw = (meta & ccwMeta) == ccwMeta;
238
239                            if (onCw && cwBlock.isSolid()) {
240                                putVine(block, getMetaFromFace(cwFace), this);
241                            } else if (onCcw && ccwBlock.isSolid()) {
242                                putVine(block, getMetaFromFace(ccwFace), this);
243                            } else if (onCw && cwBlock.getId() == AIR && this.getSide(cwFace).isSolid()) {
244                                putVine(cwBlock, getMetaFromFace(face.getOpposite()), this);
245                            } else if (onCcw && ccwBlock.getId() == AIR && this.getSide(ccwFace).isSolid()) {
246                                putVine(ccwBlock, getMetaFromFace(face.getOpposite()), this);
247                            } else if (block.up().isSolid()) {
248                                putVine(block, 0, this);
249                            }
250                        } else if (!block.isTransparent()) {
251                            meta |= getMetaFromFace(face);
252                            putVine(this, meta, null);
253                        }
254                    }
255                } else if (this.y > 0) {
256                    Block below = this.down();
257                    int id = below.getId();
258                    if (id == AIR || id == VINE) {
259                        for (BlockFace horizontalFace : BlockFace.Plane.HORIZONTAL) {
260                            if (random.nextBoolean()) {
261                                meta &= ~getMetaFromFace(horizontalFace);
262                            }
263                        }
264                        putVineOnHorizontalFace(below, below.getDamage() | meta, id == AIR ? this : null);
265                    }
266                }
267                return Level.BLOCK_UPDATE_RANDOM;
268            }
269        }
270        return 0;
271    }

273    private boolean canSpread() {
274        int blockX = this.getFloorX();
275        int blockY = this.getFloorY();
276        int blockZ = this.getFloorZ();
277
278        int count = 0;
279        for (int x = blockX - 4; x <= blockX + 4; x++) {
280            for (int z = blockZ - 4; z <= blockZ + 4; z++) {
281                for (int y = blockY - 1; y <= blockY + 1; y++) {
282                    if (this.level.getBlock(x, y, z).getId() == VINE) {
283                        if (++count >= 5) return false;
284                    }
285                }
286            }
287        }
288        return true;
289    }

291    private void putVine(Block block, int meta, Block source) {
292        if (block.getId() == VINE && block.getDamage() == meta) return;
293        Block vine = get(VINE, meta);
294        BlockGrowEvent event;
295        if (source != null) {
296            event = new BlockSpreadEvent(block, source, vine);
297        } else {
298            event = new BlockGrowEvent(block, vine);
299        }
300        this.level.getServer().getPluginManager().callEvent(event);
301        if (!event.isCancelled()) {
302            this.level.setBlock(block, vine, true);
303        }
304    }
305
306    private void putVineOnHorizontalFace(Block block, int meta, Block source) {
307        if (block.getId() == VINE && block.getDamage() == meta) return;
308        boolean isOnHorizontalFace = false;
309        for (BlockFace face : BlockFace.Plane.HORIZONTAL) {
310            int faceMeta = getMetaFromFace(face);
311            if ((meta & faceMeta) == faceMeta) {
312                isOnHorizontalFace = true;
313                break;
314            }
315        }
316        if (isOnHorizontalFace) {
317            putVine(block, meta, source);
318        }
319    }

321    private Set<BlockFace> getFaces() {
322        Set<BlockFace> faces = EnumSet.noneOf(BlockFace.class);
323
324        int meta = this.getDamage();
325        if ((meta & 1) > 0) {
326            faces.add(BlockFace.SOUTH);
327        }
328        if ((meta & 2) > 0) {
329            faces.add(BlockFace.WEST);
330        }
331        if ((meta & 4) > 0) {
332            faces.add(BlockFace.NORTH);
333        }
334        if ((meta & 8) > 0) {
335            faces.add(BlockFace.EAST);
336        }
337
338        return faces;
339    }
340
341    private static int getMetaFromFaces(Set<BlockFace> faces) {
342        int meta = 0;
343
344        for (BlockFace face : faces) {
345            meta |= getMetaFromFace(face);
346
347        }
348
349        return meta;
350    }
351
352    private static int getMetaFromFace(BlockFace face) {
353        switch (face) {
354            case SOUTH:
355            default:
356                return 0x01;
357            case WEST:
358                return 0x02;
359            case NORTH:
360                return 0x04;
361            case EAST:
362                return 0x08;
363        }
364    }
