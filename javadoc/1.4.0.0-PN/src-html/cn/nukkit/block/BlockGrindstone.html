001package cn.nukkit.block;

017import javax.annotation.Nonnull;

019@PowerNukkitOnly
020public class BlockGrindstone extends BlockTransparentMeta implements Faceable {
021    @PowerNukkitOnly
022    public static final int TYPE_ATTACHMENT_STANDING = 0;
023    @PowerNukkitOnly
024    public static final int TYPE_ATTACHMENT_HANGING = 1;
025    @PowerNukkitOnly
026    public static final int TYPE_ATTACHMENT_SIDE = 2;
027    @PowerNukkitOnly
028    public static final int TYPE_ATTACHMENT_MULTIPLE = 3;
029
030    @PowerNukkitOnly
031    public BlockGrindstone() {
032        this(0);
033    }
034
035    @PowerNukkitOnly
036    public BlockGrindstone(int meta) {
037        super(meta);
038    }
039
040    @Override
041    public int getId() {
042        return GRINDSTONE;
043    }
044
045    @Override
046    public String getName() {
047        return "Grindstone";
048    }
049
050    @Override
051    public int getToolType() {
052        return ItemTool.TYPE_PICKAXE;
053    }
054
055    @Override
056    public int getToolTier() {
057        return ItemTool.TIER_WOODEN;
058    }
059
060    @Override
061    public boolean canHarvestWithHand() {
062        return false;
063    }
064
065    @Override
066    public Item toItem() {
067        return new ItemBlock(new BlockGrindstone());
068    }
069
070    @PowerNukkitOnly
071    @Override
072    public int getWaterloggingLevel() {
073        return 1;
074    }
075
076    @Override
077    public BlockColor getColor() {
078        return BlockColor.IRON_BLOCK_COLOR;
079    }
080
081    @Override
082    public double getHardness() {
083        return 2;
084    }
085
086    @Override
087    public double getResistance() {
088        return 6;
089    }
090
091    @Override
092    public BlockFace getBlockFace() {
093        return BlockFace.fromHorizontalIndex(getDamage() & 0b11);
094    }
095
096    @Override
097    @PowerNukkitOnly
098    @Since("1.3.0.0-PN")
099    public void setBlockFace(BlockFace face) {
100        if (face.getHorizontalIndex() == -1) {
101            return;
102        }
103        setDamage(getDamage() & (DATA_MASK ^ 0b11) | face.getHorizontalIndex());
104    }
105
106    @PowerNukkitOnly
107    public int getAttachmentType() {
108        return (getDamage() & 0b1100) >> 2 & 0b11;
109    }
110
111    @PowerNukkitOnly
112    public void setAttachmentType(int attachmentType) {
113        attachmentType = attachmentType & 0b11;
114        setDamage(getDamage() & (DATA_MASK ^ 0b1100) | (attachmentType << 2));
115    }
116
117    private boolean isConnectedTo(BlockFace connectedFace, int attachmentType, BlockFace blockFace) {
118        BlockFace.Axis faceAxis = connectedFace.getAxis();
119        switch (attachmentType) {
120            case TYPE_ATTACHMENT_STANDING:
121                if (faceAxis == BlockFace.Axis.Y) {
122                    return connectedFace == BlockFace.DOWN;
123                } else {
124                    return false;
125                }
126            case TYPE_ATTACHMENT_HANGING:
127                return connectedFace == BlockFace.UP;
128            case TYPE_ATTACHMENT_SIDE:
129            case TYPE_ATTACHMENT_MULTIPLE:
130                return connectedFace == blockFace.getOpposite();
131        }
132        return false;
133    }
134
135    @Override
136    public int onUpdate(int type) {
137        if (type == Level.BLOCK_UPDATE_NORMAL) {
138            if (!checkSupport()) {
139                this.level.useBreakOn(this, Item.get(Item.DIAMOND_PICKAXE));
140            }
141            return type;
142        }
143        return 0;
144    }
145
146    @Override
147    public boolean place(@Nonnull Item item, @Nonnull Block block, @Nonnull Block target, @Nonnull BlockFace face, double fx, double fy, double fz, Player player) {
148        if (block.getId() != AIR && block.canBeReplaced()) {
149            face = BlockFace.UP;
150        }
151        switch (face) {
152            case UP:
153                setAttachmentType(TYPE_ATTACHMENT_STANDING);
154                setBlockFace(player.getDirection().getOpposite());
155                break;
156            case DOWN:
157                setAttachmentType(TYPE_ATTACHMENT_HANGING);
158                setBlockFace(player.getDirection().getOpposite());
159                break;
160            default:
161                setBlockFace(face);
162                setAttachmentType(TYPE_ATTACHMENT_SIDE);
163        }
164        if (!checkSupport()) {
165            return false;
166        }
167        this.level.setBlock(this, this, true, true);
168        return true;
169    }
170
171    @SuppressWarnings("BooleanMethodIsAlwaysInverted")
172    private boolean checkSupport() {
173        switch (getAttachmentType()) {
174            case TYPE_ATTACHMENT_STANDING:
175                if (checkSupport(down())) {
176                    return true;
177                }
178                break;
179            case TYPE_ATTACHMENT_HANGING:
180                if (checkSupport(up())) {
181                    return true;
182                }
183                break;
184            case TYPE_ATTACHMENT_SIDE:
185                BlockFace blockFace = getBlockFace();
186                if (checkSupport(getSide(blockFace.getOpposite()))) {
187                    return true;
188                }
189                break;
190        }
191
192        return false;
193    }
194
195    private boolean checkSupport(Block support) {
196        int id = support.getId();
197        return id != AIR && id != BUBBLE_COLUMN && !(support instanceof BlockLiquid);
198    }
199
200    @Override
201    protected AxisAlignedBB recalculateBoundingBox() {
202        int attachmentType = getAttachmentType();
203        BlockFace blockFace = getBlockFace();
204        boolean south = this.isConnectedTo(BlockFace.SOUTH, attachmentType, blockFace);
205        boolean north = this.isConnectedTo(BlockFace.NORTH, attachmentType, blockFace);
206        boolean west = this.isConnectedTo(BlockFace.WEST, attachmentType, blockFace);
207        boolean east = this.isConnectedTo(BlockFace.EAST, attachmentType, blockFace);
208        boolean up = this.isConnectedTo(BlockFace.UP, attachmentType, blockFace);
209        boolean down = this.isConnectedTo(BlockFace.DOWN, attachmentType, blockFace);
210
211        double pixels = (2.0/16);
212
213        double n = north ? 0 : pixels;
214        double s = south ? 1 : 1 - pixels;
215        double w = west ? 0 : pixels;
216        double e = east ? 1 : 1 - pixels;
217        double d = down ? 0 : pixels;
218        double u = up ? 1 : 1 - pixels;
219
220        return new SimpleAxisAlignedBB(
221                this.x + w,
222                this.y + d,
223                this.z + n,
224                this.x + e,
225                this.y + u,
226                this.z + s
227        );
228    }
229
230    @Override
231    public boolean canBeActivated() {
232        return true;
233    }
234
235    @Override
236    public boolean onActivate(@Nonnull Item item, Player player) {
237        if (player != null) {
238            player.addWindow(new GrindstoneInventory(player.getUIInventory(), this), Player.GRINDSTONE_WINDOW_ID);
239        }
240        return true;
241    }
