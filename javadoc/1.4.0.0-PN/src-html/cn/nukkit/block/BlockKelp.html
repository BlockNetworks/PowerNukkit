001package cn.nukkit.block;
002
003import cn.nukkit.Player;
004import cn.nukkit.Server;
005import cn.nukkit.api.PowerNukkitOnly;
006import cn.nukkit.api.Since;
007import cn.nukkit.blockproperty.BlockProperties;
008import cn.nukkit.blockproperty.IntBlockProperty;
009import cn.nukkit.blockstate.BlockState;
010import cn.nukkit.event.block.BlockGrowEvent;
011import cn.nukkit.item.Item;
012import cn.nukkit.item.ItemKelp;
013import cn.nukkit.level.Level;
014import cn.nukkit.level.particle.BoneMealParticle;
015import cn.nukkit.math.BlockFace;
016
017import javax.annotation.Nonnull;
018import java.util.concurrent.ThreadLocalRandom;
019
020@PowerNukkitOnly
021public class BlockKelp extends BlockFlowable {
022    @PowerNukkitOnly
023    @Since("1.4.0.0-PN")
024    public static final IntBlockProperty KELP_AGE = new IntBlockProperty("kelp_age", false, 25);
025
026    @PowerNukkitOnly
027    @Since("1.4.0.0-PN")
028    public static final BlockProperties PROPERTIES = new BlockProperties(KELP_AGE);
029
030    @PowerNukkitOnly
031    @Since("1.4.0.0-PN")
032    public BlockKelp() {
033        this(0);
034    }
035
036    @PowerNukkitOnly
037    @Since("1.4.0.0-PN")
038    public BlockKelp(int meta) {
039        super(meta);
040    }
041    
042    @Override
043    public int getId() {
044        return BLOCK_KELP;
045    }
046
047    @Since("1.4.0.0-PN")
048    @PowerNukkitOnly
049    @Nonnull
050    @Override
051    public BlockProperties getProperties() {
052        return PROPERTIES;
053    }
054
055    @Override
056    public String getName() {
057        return "Kelp";
058    }
059
060    @PowerNukkitOnly
061    @Since("1.4.0.0-PN")
062    public int getAge() {
063        return getIntValue(KELP_AGE);
064    }
065
066    @PowerNukkitOnly
067    @Since("1.4.0.0-PN")
068    public void setAge(int age) {
069        setIntValue(KELP_AGE, age);
070    }
071    
072    @Override
073    public boolean place(@Nonnull Item item, @Nonnull Block block, @Nonnull Block target, @Nonnull BlockFace face, double fx, double fy, double fz, Player player) {
074        Block down = down();
075        Block layer1Block = block.getLevelBlockAtLayer(1);
076        if ((down.getId() == BLOCK_KELP || down.isSolid()) && down.getId() != MAGMA && down.getId() != ICE && down.getId() != SOUL_SAND &&
077                (layer1Block instanceof BlockWater && ((BlockWater) layer1Block).isSourceOrFlowingDown())
078        ) {
079            if (((BlockWater) layer1Block).isFlowingDown()) {
080                this.getLevel().setBlock(this, 1, get(WATER), true, false);
081            }
082
083            int maxAge = KELP_AGE.getMaxValue();
084            if (down.getId() == BLOCK_KELP && down.getIntValue(KELP_AGE) != maxAge - 1) {
085                down.setIntValue(KELP_AGE, maxAge - 1);
086                this.getLevel().setBlock(down, down, true, true);
087            }
088
089            //Placing it by hand gives it a random age value between 0 and 24.
090            setAge(ThreadLocalRandom.current().nextInt(maxAge));
091            this.getLevel().setBlock(this, this, true, true);
092            return true;
093        } else {
094            return false;
095        }
096    }
097    
098    @Override
099    public int onUpdate(int type) {
100        if (type == Level.BLOCK_UPDATE_NORMAL) {
101            Block blockLayer1 = getLevelBlockAtLayer(1);
102            if (!(blockLayer1 instanceof BlockIceFrosted) &&
103                    (!(blockLayer1 instanceof BlockWater) || !((BlockWater)blockLayer1).isSourceOrFlowingDown())) {
104                this.getLevel().useBreakOn(this);
105                return type;
106            }
107        
108            Block down = down();
109            if ((!down.isSolid() && down.getId() != BLOCK_KELP) || down.getId() == MAGMA || down.getId() == ICE || down.getId() == SOUL_SAND) {
110                this.getLevel().useBreakOn(this);
111                return type;
112            }
113        
114            if (blockLayer1 instanceof BlockWater && ((BlockWater)blockLayer1).isFlowingDown()) {
115                this.getLevel().setBlock(this, 1, get(WATER), true, false);
116            }
117            return type;
118        } else if (type == Level.BLOCK_UPDATE_RANDOM) {
119            if (ThreadLocalRandom.current().nextInt(100) <= 14) {
120                grow();
121            }
122            return type;
123        }
124        return super.onUpdate(type);
125    }
126    
127    @PowerNukkitOnly
128    public boolean grow() {
129        int age = getAge();
130        int maxValue = KELP_AGE.getMaxValue();
131        if (age < maxValue) {
132            Block up = up();
133            if (up instanceof BlockWater && ((BlockWater)up).isSourceOrFlowingDown()) {
134                Block grown = BlockState.of(BLOCK_KELP, age + 1).getBlock();
135                BlockGrowEvent ev = new BlockGrowEvent(this, grown);
136                Server.getInstance().getPluginManager().callEvent(ev);
137                if (!ev.isCancelled()) {
138                    this.setAge(maxValue);
139                    this.getLevel().setBlock(this, 0, this, true, true);
140                    this.getLevel().setBlock(up, 1, get(WATER), true, false);
141                    this.getLevel().setBlock(up, 0, ev.getNewState(), true, true);
142                    return true;
143                }
144            }
145        }
146        return false;
147    }
148    
149    @Override
150    public boolean onBreak(Item item) {
151        Block down = down();
152        if (down.getId() == BLOCK_KELP) {
153            this.getLevel().setBlock(down, BlockState.of(BLOCK_KELP, ThreadLocalRandom.current().nextInt(KELP_AGE.getMaxValue())).getBlock(), true, true);
154        }
155        this.getLevel().setBlock(this, get(AIR), true, true);
156        return true;
157    }
158    
159    @Override
160    public boolean onActivate(@Nonnull Item item, Player player) {
161        //Bone meal
162        if (item.isFertilizer()) {
163            int x = (int) this.x;
164            int z = (int) this.z;
165            for (int y = (int) this.y + 1; y < 255; y++) {
166                BlockState blockStateAbove = getLevel().getBlockStateAt(x, y, z);
167                int blockIdAbove = blockStateAbove.getBlockId();
168                if (blockIdAbove != BLOCK_KELP) {
169                    if (blockIdAbove == WATER || blockIdAbove == STILL_WATER) {
170                        if (((BlockWater)blockStateAbove.getBlock()).isSourceOrFlowingDown()) {
171                            BlockKelp highestKelp = (BlockKelp) getLevel().getBlock(x, y - 1, z);
172                            if (highestKelp.grow()) {
173                                this.level.addParticle(new BoneMealParticle(this));
174    
175                                if (player != null && (player.gamemode & 0x01) == 0) {
176                                    item.count--;
177                                }
178                                
179                                return true;
180                            }
181                        }
182                    }
183                    return false;
184                }
185            }
186            
187            return true;
188        }
189        
190        return false;
191    }
192    
193    @Override
194    public Item toItem() {
195        return new ItemKelp();
196    }
197    
198    @PowerNukkitOnly
199    @Override
200    public int getWaterloggingLevel() {
201        return 2;
202    }
203    
204    @Override
205    public boolean canBeActivated() {
206        return true;
207    }
208}
