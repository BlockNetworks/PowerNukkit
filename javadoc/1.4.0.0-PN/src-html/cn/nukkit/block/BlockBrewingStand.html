001package cn.nukkit.block;
002
003
004import cn.nukkit.Player;
005import cn.nukkit.api.PowerNukkitDifference;
006import cn.nukkit.api.PowerNukkitOnly;
007import cn.nukkit.blockentity.BlockEntity;
008import cn.nukkit.blockentity.BlockEntityBrewingStand;
009import cn.nukkit.inventory.ContainerInventory;
010import cn.nukkit.item.Item;
011import cn.nukkit.item.ItemBrewingStand;
012import cn.nukkit.item.ItemTool;
013import cn.nukkit.math.BlockFace;
014import cn.nukkit.nbt.tag.CompoundTag;
015import cn.nukkit.nbt.tag.ListTag;
016import cn.nukkit.nbt.tag.StringTag;
017import cn.nukkit.nbt.tag.Tag;
018import cn.nukkit.utils.BlockColor;
019
020import javax.annotation.Nonnull;
021import java.util.Map;
022
023public class BlockBrewingStand extends BlockSolidMeta {
024
025    public BlockBrewingStand() {
026        this(0);
027    }
028
029    public BlockBrewingStand(int meta) {
030        super(meta);
031    }
032
033    @Override
034    public String getName() {
035        return "Brewing Stand";
036    }
037
038    @Override
039    public boolean canBeActivated() {
040        return true;
041    }
042
043    @Override
044    public double getHardness() {
045        return 0.5;
046    }
047
048    @Override
049    public double getResistance() {
050        return 2.5;
051    }
052
053    @PowerNukkitOnly
054    @Override
055    public int getWaterloggingLevel() {
056        return 1;
057    }
058
059    @Override
060    public int getToolType() {
061        return ItemTool.TYPE_PICKAXE;
062    }
063
064    @Override
065    public int getId() {
066        return BREWING_STAND_BLOCK;
067    }
068
069    @Override
070    public int getLightLevel() {
071        return 1;
072    }
073
074    @PowerNukkitDifference(info = "Remove placement restrictions, they don't exists in vanilla", since = "1.3.1.2-PN")
075    @Override
076    public boolean place(@Nonnull Item item, @Nonnull Block block, @Nonnull Block target, @Nonnull BlockFace face, double fx, double fy, double fz, Player player) {
077        getLevel().setBlock(block, this, true, true);
078
079        CompoundTag nbt = new CompoundTag()
080                .putList(new ListTag<>("Items"))
081                .putString("id", BlockEntity.BREWING_STAND)
082                .putInt("x", (int) this.x)
083                .putInt("y", (int) this.y)
084                .putInt("z", (int) this.z);
085
086        if (item.hasCustomName()) {
087            nbt.putString("CustomName", item.getCustomName());
088        }
089
090        if (item.hasCustomBlockData()) {
091            Map<String, Tag> customData = item.getCustomBlockData().getTags();
092            for (Map.Entry<String, Tag> tag : customData.entrySet()) {
093                nbt.put(tag.getKey(), tag.getValue());
094            }
095        }
096
097        BlockEntityBrewingStand brewing = (BlockEntityBrewingStand) BlockEntity.createBlockEntity(BlockEntity.BREWING_STAND, getLevel().getChunk((int) this.x >> 4, (int) this.z >> 4), nbt);
098        return brewing != null;
099    }
100    
101    @Override
102    public boolean onActivate(@Nonnull Item item, Player player) {
103        if (player != null) {
104            BlockEntity t = getLevel().getBlockEntity(this);
105            BlockEntityBrewingStand brewing;
106            if (t instanceof BlockEntityBrewingStand) {
107                brewing = (BlockEntityBrewingStand) t;
108            } else {
109                CompoundTag nbt = new CompoundTag()
110                        .putList(new ListTag<>("Items"))
111                        .putString("id", BlockEntity.BREWING_STAND)
112                        .putInt("x", (int) this.x)
113                        .putInt("y", (int) this.y)
114                        .putInt("z", (int) this.z);
115                brewing = (BlockEntityBrewingStand) BlockEntity.createBlockEntity(BlockEntity.BREWING_STAND, this.getLevel().getChunk((int) (this.x) >> 4, (int) (this.z) >> 4), nbt);
116                if (brewing == null) {
117                    return false;
118                }
119            }
120
121            if (brewing.namedTag.contains("Lock") && brewing.namedTag.get("Lock") instanceof StringTag) {
122                if (!brewing.namedTag.getString("Lock").equals(item.getCustomName())) {
123                    return false;
124                }
125            }
126
127            player.addWindow(brewing.getInventory());
128        }
129
130        return true;
131    }
132
133    @Override
134    public Item toItem() {
135        return new ItemBrewingStand();
136    }
137
138    @Override
139    public int getToolTier() {
140        return ItemTool.TIER_WOODEN;
141    }
142
143    @Override
144    public BlockColor getColor() {
145        return BlockColor.IRON_BLOCK_COLOR;
146    }
147
148    public boolean hasComparatorInputOverride() {
149        return true;
150    }
151
152    @Override
153    public int getComparatorInputOverride() {
154        BlockEntity blockEntity = this.level.getBlockEntity(this);
155
156        if (blockEntity instanceof BlockEntityBrewingStand) {
157            return ContainerInventory.calculateRedstone(((BlockEntityBrewingStand) blockEntity).getInventory());
158        }
159
160        return super.getComparatorInputOverride();
161    }
162
163    @Override
164    public boolean canHarvestWithHand() {
165        return false;
166    }
167}
