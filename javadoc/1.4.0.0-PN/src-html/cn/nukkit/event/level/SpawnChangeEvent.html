001package cn.nukkit.event.level;
002
003import cn.nukkit.event.HandlerList;
004import cn.nukkit.level.Level;
005import cn.nukkit.level.Position;
006
007/**
008 * @author MagicDroidX (Nukkit Project)
009 */
010public class SpawnChangeEvent extends LevelEvent {
011
012    private static final HandlerList handlers = new HandlerList();
013
014    public static HandlerList getHandlers() {
015        return handlers;
016    }
017
018    private final Position previousSpawn;
019
020    public SpawnChangeEvent(Level level, Position previousSpawn) {
021        super(level);
022        this.previousSpawn = previousSpawn;
023    }
024
025    public Position getPreviousSpawn() {
026        return previousSpawn;
027    }
028}
