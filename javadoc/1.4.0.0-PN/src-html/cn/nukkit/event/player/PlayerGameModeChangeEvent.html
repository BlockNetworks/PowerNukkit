001package cn.nukkit.event.player;
002
003import cn.nukkit.AdventureSettings;
004import cn.nukkit.Player;
005import cn.nukkit.event.Cancellable;
006import cn.nukkit.event.HandlerList;
007
008public class PlayerGameModeChangeEvent extends PlayerEvent implements Cancellable {
009    private static final HandlerList handlers = new HandlerList();
010
011    public static HandlerList getHandlers() {
012        return handlers;
013    }
014
015    protected final int gamemode;
016
017    protected AdventureSettings newAdventureSettings;
018
019    public PlayerGameModeChangeEvent(Player player, int newGameMode, AdventureSettings newAdventureSettings) {
020        this.player = player;
021        this.gamemode = newGameMode;
022        this.newAdventureSettings = newAdventureSettings;
023    }
024
025    public int getNewGamemode() {
026        return gamemode;
027    }
028
029    public AdventureSettings getNewAdventureSettings() {
030        return newAdventureSettings;
031    }
032
033    public void setNewAdventureSettings(AdventureSettings newAdventureSettings) {
034        this.newAdventureSettings = newAdventureSettings;
035    }
036}
