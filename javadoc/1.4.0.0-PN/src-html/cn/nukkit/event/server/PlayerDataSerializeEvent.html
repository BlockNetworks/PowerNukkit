001package cn.nukkit.event.server;
002
003import cn.nukkit.event.HandlerList;
004import cn.nukkit.utils.PlayerDataSerializer;
005import com.google.common.base.Preconditions;
006
007import java.util.Optional;
008import java.util.UUID;
009
010@SuppressWarnings("OptionalUsedAsFieldOrParameterType")
011public class PlayerDataSerializeEvent extends ServerEvent {
012    private static HandlerList handlers = new HandlerList();
013
014    private final Optional<String> name;
015    private final Optional<UUID> uuid;
016    private PlayerDataSerializer serializer;
017
018    public PlayerDataSerializeEvent(String name, PlayerDataSerializer serializer) {
019        Preconditions.checkNotNull(name);
020        this.serializer = Preconditions.checkNotNull(serializer);
021        UUID uuid = null;
022        try {
023            uuid = UUID.fromString(name);
024        } catch (Exception e) {
025            // ignore
026        }
027        this.uuid = Optional.ofNullable(uuid);
028        this.name = this.uuid.isPresent() ? Optional.empty() : Optional.of(name);
029    }
030
031    public Optional<String> getName() {
032        return name;
033    }
034
035    public Optional<UUID> getUuid() {
036        return uuid;
037    }
038
039    public PlayerDataSerializer getSerializer() {
040        return serializer;
041    }
042
043    public void setSerializer(PlayerDataSerializer serializer) {
044        this.serializer = Preconditions.checkNotNull(serializer, "serializer");
045    }
046
047    public static HandlerList getHandlers() {
048        return handlers;
049    }
050}
