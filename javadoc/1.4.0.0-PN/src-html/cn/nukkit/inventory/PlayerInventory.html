048    /**
049     * Called when a client equips a hotbar inventorySlot. This method should not be used by plugins.
050     * This method will call PlayerItemHeldEvent.
051     *
052     * @param slot hotbar slot Number of the hotbar slot to equip.
053     * @return boolean if the equipment change was successful, false if not.
054     */
055    public boolean equipItem(int slot) {
056        if (!isHotbarSlot(slot)) {
057            this.sendContents((Player) this.getHolder());
058            return false;
059        }
060
061        if (this.getHolder() instanceof Player) {
062            Player player = (Player) this.getHolder();
063            PlayerItemHeldEvent ev = new PlayerItemHeldEvent(player, this.getItem(slot), slot);
064            this.getHolder().getLevel().getServer().getPluginManager().callEvent(ev);
065
066            if (ev.isCancelled()) {
067                this.sendContents(this.getViewers());
068                return false;
069            }
070
071            if (player.fishing != null) {
072                if (!(this.getItem(slot).equals(player.fishing.rod))) {
073                    player.stopFishing(false);
074                }
075            }
076        }
077
078        this.setHeldItemIndex(slot, false);
079        return true;
080    }

083        return slot >= 0 && slot <= this.getHotbarSize();

148    public void sendHeldItem(Player... players) {
149        Item item = this.getItemInHand();
150
151        MobEquipmentPacket pk = new MobEquipmentPacket();
152        pk.item = item;
153        pk.inventorySlot = pk.hotbarSlot = this.getHeldItemIndex();
154
155        for (Player player : players) {
156            pk.eid = this.getHolder().getId();
157            if (player.equals(this.getHolder())) {
158                pk.eid = player.getId();
159                this.sendSlot(this.getHeldItemIndex(), player);
160            }
161
162            player.dataPacket(pk);
163        }
164    }

170    @Override
171    public void onSlotChange(int index, Item before, boolean send) {
172        EntityHuman holder = this.getHolder();
173        if (holder instanceof Player && !((Player) holder).spawned) {
174            return;
175        }
176
177        if (index >= this.getSize()) {
178            this.sendArmorSlot(index, this.getViewers());
179            this.sendArmorSlot(index, this.getHolder().getViewers().values());
180        } else {
181            super.onSlotChange(index, before, send);
182        }
183    }

238    private boolean setItem(int index, Item item, boolean send, boolean ignoreArmorEvents) {
239        if (index < 0 || index >= this.size) {
240            return false;
241        } else if (item.getId() == 0 || item.getCount() <= 0) {
242            return this.clear(index);
243        }
244
245        //Armor change
246        if (!ignoreArmorEvents && index >= this.getSize()) {
247            EntityArmorChangeEvent ev = new EntityArmorChangeEvent(this.getHolder(), this.getItem(index), item, index);
248            Server.getInstance().getPluginManager().callEvent(ev);
249            if (ev.isCancelled() && this.getHolder() != null) {
250                this.sendArmorSlot(index, this.getViewers());
251                return false;
252            }
253            item = ev.getNewItem();
254        } else {
255            EntityInventoryChangeEvent ev = new EntityInventoryChangeEvent(this.getHolder(), this.getItem(index), item, index);
256            Server.getInstance().getPluginManager().callEvent(ev);
257            if (ev.isCancelled()) {
258                this.sendSlot(index, this.getViewers());
259                return false;
260            }
261            item = ev.getNewItem();
262        }
263        Item old = this.getItem(index);
264        this.slots.put(index, item.clone());
265        this.onSlotChange(index, old, send);
266        return true;
267    }

269    @Override
270    public boolean clear(int index, boolean send) {
271        if (this.slots.containsKey(index)) {
272            Item item = new ItemBlock(Block.get(BlockID.AIR), null, 0);
273            Item old = this.slots.get(index);
274            if (index >= this.getSize() && index < this.size) {
275                EntityArmorChangeEvent ev = new EntityArmorChangeEvent(this.getHolder(), old, item, index);
276                Server.getInstance().getPluginManager().callEvent(ev);
277                if (ev.isCancelled()) {
278                    if (index >= this.size) {
279                        this.sendArmorSlot(index, this.getViewers());
280                    } else {
281                        this.sendSlot(index, this.getViewers());
282                    }
283                    return false;
284                }
285                item = ev.getNewItem();
286            } else {
287                EntityInventoryChangeEvent ev = new EntityInventoryChangeEvent(this.getHolder(), old, item, index);
288                Server.getInstance().getPluginManager().callEvent(ev);
289                if (ev.isCancelled()) {
290                    if (index >= this.size) {
291                        this.sendArmorSlot(index, this.getViewers());
292                    } else {
293                        this.sendSlot(index, this.getViewers());
294                    }
295                    return false;
296                }
297                item = ev.getNewItem();
298            }
299
300            if (item.getId() != Item.AIR) {
301                this.slots.put(index, item.clone());
302            } else {
303                this.slots.remove(index);
304            }
305
306            this.onSlotChange(index, old, send);
307        }
308
309        return true;
310    }

334    public void sendArmorContents(Player[] players) {
335        Item[] armor = this.getArmorContents();
336
337        MobArmorEquipmentPacket pk = new MobArmorEquipmentPacket();
338        pk.eid = this.getHolder().getId();
339        pk.slots = armor;
340        pk.tryEncode();
341
342        for (Player player : players) {
343            if (player.equals(this.getHolder())) {
344                InventoryContentPacket pk2 = new InventoryContentPacket();
345                pk2.inventoryId = InventoryContentPacket.SPECIAL_ARMOR;
346                pk2.slots = armor;
347                player.dataPacket(pk2);
348            } else {
349                player.dataPacket(pk);
350            }
351        }
352    }

354    public void setArmorContents(Item[] items) {
355        if (items.length < 4) {
356            Item[] newItems = new Item[4];
357            System.arraycopy(items, 0, newItems, 0, items.length);
358            items = newItems;
359        }
360
361        for (int i = 0; i < 4; ++i) {
362            if (items[i] == null) {
363                items[i] = new ItemBlock(Block.get(BlockID.AIR), null, 0);
364            }
365
366            if (items[i].getId() == Item.AIR) {
367                this.clear(this.getSize() + i);
368            } else {
369                this.setItem(this.getSize() + i, items[i]);
370            }
371        }
372    }

417    @Override
418    public void sendContents(Player[] players) {
419        InventoryContentPacket pk = new InventoryContentPacket();
420        pk.slots = new Item[this.getSize()];
421        for (int i = 0; i < this.getSize(); ++i) {
422            pk.slots[i] = this.getItem(i);
423        }
424
425        /*//Because PE is stupid and shows 9 less slots than you send it, give it 9 dummy slots so it shows all the REAL slots.
426        for(int i = this.getSize(); i < this.getSize() + this.getHotbarSize(); ++i){
427            pk.slots[i] = new ItemBlock(Block.get(BlockID.AIR));
428        }
429            pk.slots[i] = new ItemBlock(Block.get(BlockID.AIR));
430        }*/
431
432        for (Player player : players) {
433            int id = player.getWindowId(this);
434            if (id == -1 || !player.spawned) {
435                if (this.getHolder() != player) this.close(player);
436                continue;
437            }
438            pk.inventoryId = id;
439            player.dataPacket(pk.clone());
440
441        }
442    }

454    @Override
455    public void sendSlot(int index, Player... players) {
456        InventorySlotPacket pk = new InventorySlotPacket();
457        pk.slot = index;
458        pk.item = this.getItem(index).clone();
459
460        for (Player player : players) {
461            if (player.equals(this.getHolder())) {
462                pk.inventoryId = ContainerIds.INVENTORY;
463                player.dataPacket(pk);
464            } else {
465                int id = player.getWindowId(this);
466                if (id == -1) {
467                    this.close(player);
468                    continue;
469                }
470                pk.inventoryId = id;
471                player.dataPacket(pk.clone());
472            }
473        }
474    }

476    public void sendCreativeContents() {
477        if (!(this.getHolder() instanceof Player)) {
478            return;
479        }
480        Player p = (Player) this.getHolder();
481
482        CreativeContentPacket pk = new CreativeContentPacket();

485            pk.entries = Item.getCreativeItems().toArray(Item.EMPTY_ARRAY);
486        }
487
488        p.dataPacket(pk);

496    @Override
497    public void onOpen(Player who) {
498        super.onOpen(who);
499        if (who.spawned) {
500            ContainerOpenPacket pk = new ContainerOpenPacket();
501            pk.windowId = who.getWindowId(this);
502            pk.type = this.getType().getNetworkType();
503            pk.x = who.getFloorX();
504            pk.y = who.getFloorY();
505            pk.z = who.getFloorZ();
506            pk.entityId = who.getId();
507            who.dataPacket(pk);
508        }
509    }

511    @Override
512    public void onClose(Player who) {
513        ContainerClosePacket pk = new ContainerClosePacket();
514        pk.windowId = who.getWindowId(this);
515        pk.wasServerInitiated = who.getClosingWindowId() != pk.windowId;
516        who.dataPacket(pk);
517        // player can never stop viewing their own inventory
518        if (who != holder) {
519            super.onClose(who);
520        }
521    }
