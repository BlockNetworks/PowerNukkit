001package cn.nukkit.inventory;
002
003import cn.nukkit.Player;
004import cn.nukkit.api.PowerNukkitDifference;
005import cn.nukkit.block.BlockID;
006import cn.nukkit.blockentity.BlockEntityShulkerBox;
007import cn.nukkit.item.Item;
008import cn.nukkit.level.Level;
009import cn.nukkit.level.Sound;
010import cn.nukkit.network.protocol.BlockEventPacket;
011
012/**
013 * @author PetteriM1
014 */
015public class ShulkerBoxInventory extends ContainerInventory {
016
017    public ShulkerBoxInventory(BlockEntityShulkerBox box) {
018        super(box, InventoryType.SHULKER_BOX);
019    }
020
021    @Override
022    public BlockEntityShulkerBox getHolder() {
023        return (BlockEntityShulkerBox) this.holder;
024    }
025
026    @PowerNukkitDifference(info = "Using new method to play sounds", since = "1.4.0.0-PN")
027    @Override
028    public void onOpen(Player who) {
029        super.onOpen(who);
030
031        if (this.getViewers().size() == 1) {
032            BlockEventPacket pk = new BlockEventPacket();
033            pk.x = (int) this.getHolder().getX();
034            pk.y = (int) this.getHolder().getY();
035            pk.z = (int) this.getHolder().getZ();
036            pk.case1 = 1;
037            pk.case2 = 2;
038
039            Level level = this.getHolder().getLevel();
040            if (level != null) {
041                level.addSound(this.getHolder().add(0.5, 0.5, 0.5), Sound.RANDOM_SHULKERBOXOPEN);
042                level.addChunkPacket((int) this.getHolder().getX() >> 4, (int) this.getHolder().getZ() >> 4, pk);
043            }
044        }
045    }
046
047    @PowerNukkitDifference(info = "Using new method to play sounds", since = "1.4.0.0-PN")
048    @Override
049    public void onClose(Player who) {
050        if (this.getViewers().size() == 1) {
051            BlockEventPacket pk = new BlockEventPacket();
052            pk.x = (int) this.getHolder().getX();
053            pk.y = (int) this.getHolder().getY();
054            pk.z = (int) this.getHolder().getZ();
055            pk.case1 = 1;
056            pk.case2 = 0;
057
058            Level level = this.getHolder().getLevel();
059            if (level != null) {
060                level.addSound(this.getHolder().add(0.5, 0.5, 0.5), Sound.RANDOM_SHULKERBOXCLOSED);
061                level.addChunkPacket((int) this.getHolder().getX() >> 4, (int) this.getHolder().getZ() >> 4, pk);
062            }
063        }
064
065        super.onClose(who);
066    }
067
068    @Override
069    public boolean canAddItem(Item item) {
070        if (item.getId() == BlockID.SHULKER_BOX || item.getId() == BlockID.UNDYED_SHULKER_BOX) {
071            // Do not allow nested shulker boxes.
072            return false;
073        }
074        return super.canAddItem(item);
075    }
076
077    @Override
078    public void sendSlot(int index, Player... players) {
079        super.sendSlot(index, players);
080    }
081}
