001package cn.nukkit.inventory;
002
003import cn.nukkit.Player;
004import cn.nukkit.api.PowerNukkitDifference;
005import cn.nukkit.block.BlockEnderChest;
006import cn.nukkit.entity.EntityHuman;
007import cn.nukkit.entity.EntityHumanType;
008import cn.nukkit.level.Level;
009import cn.nukkit.level.Sound;
010import cn.nukkit.network.protocol.BlockEventPacket;
011import cn.nukkit.network.protocol.ContainerClosePacket;
012import cn.nukkit.network.protocol.ContainerOpenPacket;
013
014public class PlayerEnderChestInventory extends BaseInventory {
015
016    public PlayerEnderChestInventory(EntityHumanType player) {
017        super(player, InventoryType.ENDER_CHEST);
018    }
019
020    @Override
021    public EntityHuman getHolder() {
022        return (EntityHuman) this.holder;
023    }
024
025    @PowerNukkitDifference(info = "Using new method to play sounds", since = "1.4.0.0-PN")
026    @Override
027    public void onOpen(Player who) {
028        if (who != this.getHolder()) {
029            return;
030        }
031        super.onOpen(who);
032        ContainerOpenPacket containerOpenPacket = new ContainerOpenPacket();
033        containerOpenPacket.windowId = who.getWindowId(this);
034        containerOpenPacket.type = this.getType().getNetworkType();
035        BlockEnderChest chest = who.getViewingEnderChest();
036        if (chest != null) {
037            containerOpenPacket.x = (int) chest.getX();
038            containerOpenPacket.y = (int) chest.getY();
039            containerOpenPacket.z = (int) chest.getZ();
040        } else {
041            containerOpenPacket.x = containerOpenPacket.y = containerOpenPacket.z = 0;
042        }
043
044        who.dataPacket(containerOpenPacket);
045
046        this.sendContents(who);
047
048        if (chest != null && chest.getViewers().size() == 1) {
049            BlockEventPacket blockEventPacket = new BlockEventPacket();
050            blockEventPacket.x = (int) chest.getX();
051            blockEventPacket.y = (int) chest.getY();
052            blockEventPacket.z = (int) chest.getZ();
053            blockEventPacket.case1 = 1;
054            blockEventPacket.case2 = 2;
055
056            Level level = this.getHolder().getLevel();
057            if (level != null) {
058                level.addSound(this.getHolder().add(0.5, 0.5, 0.5), Sound.RANDOM_ENDERCHESTOPEN);
059                level.addChunkPacket((int) this.getHolder().getX() >> 4, (int) this.getHolder().getZ() >> 4, blockEventPacket);
060            }
061        }
062    }
063
064    @PowerNukkitDifference(info = "Using new method to play sounds", since = "1.4.0.0-PN")
065    @Override
066    public void onClose(Player who) {
067        ContainerClosePacket containerClosePacket = new ContainerClosePacket();
068        containerClosePacket.windowId = who.getWindowId(this);
069        containerClosePacket.wasServerInitiated = who.getClosingWindowId() != containerClosePacket.windowId;
070        who.dataPacket(containerClosePacket);
071        super.onClose(who);
072
073        BlockEnderChest chest = who.getViewingEnderChest();
074        if (chest != null && chest.getViewers().size() == 1) {
075            BlockEventPacket blockEventPacket = new BlockEventPacket();
076            blockEventPacket.x = (int) chest.getX();
077            blockEventPacket.y = (int) chest.getY();
078            blockEventPacket.z = (int) chest.getZ();
079            blockEventPacket.case1 = 1;
080            blockEventPacket.case2 = 0;
081
082            Level level = this.getHolder().getLevel();
083            if (level != null) {
084                level.addSound(this.getHolder().add(0.5, 0.5, 0.5), Sound.RANDOM_ENDERCHESTCLOSED);
085                level.addChunkPacket((int) this.getHolder().getX() >> 4, (int) this.getHolder().getZ() >> 4, blockEventPacket);
086            }
087
088            who.setViewingEnderChest(null);
089        }
090
091        super.onClose(who);
092    }
093}
