001package cn.nukkit.inventory;
002
003import cn.nukkit.Player;
004import cn.nukkit.api.PowerNukkitOnly;
005import cn.nukkit.api.Since;
006import cn.nukkit.entity.Entity;
007import cn.nukkit.item.Item;
008import cn.nukkit.network.protocol.MobEquipmentPacket;
009
010import java.util.HashSet;
011import java.util.Set;
012
013@PowerNukkitOnly
014@Since("1.4.0.0-PN")
015public class EntityEquipmentInventory extends BaseInventory {
016
017    private final Entity entity;
018
019    @PowerNukkitOnly @Since("1.4.0.0-PN") public static final int MAIN_HAND = 0;
020    @PowerNukkitOnly @Since("1.4.0.0-PN") public static final int OFFHAND = 1;
021
022    /**
023     * @param entity an Entity which implements {@link InventoryHolder}.
024     * @throws ClassCastException if the entity does not implements {@link InventoryHolder}
025     */
026    @PowerNukkitOnly
027    @Since("1.4.0.0-PN")
028    public EntityEquipmentInventory(Entity entity) {
029        super((InventoryHolder) entity, InventoryType.ENTITY_EQUIPMENT);
030        this.entity = entity;
031    }
032
033    @Override
034    public String getName() {
035        return "Entity Equipment";
036    }
037
038    @Override
039    public int getSize() {
040        return 2;
041    }
042
043    @PowerNukkitOnly
044    @Since("1.4.0.0-PN")
045    public Entity getEntity() {
046        return entity;
047    }
048    
049    @Override
050    public InventoryHolder getHolder() {
051        return this.holder;
052    }
053
054    @Override
055    public void sendSlot( int index, Player... players ) {
056        for ( Player player : players ) {
057            this.sendSlot( index, player );
058        }
059    }
060
061    @Override
062    public void sendSlot( int index, Player player ) {
063        MobEquipmentPacket mobEquipmentPacket = new MobEquipmentPacket();
064        mobEquipmentPacket.eid = this.entity.getId();
065        mobEquipmentPacket.inventorySlot = mobEquipmentPacket.hotbarSlot = index;
066        mobEquipmentPacket.item = this.getItem( index );
067        player.dataPacket( mobEquipmentPacket );
068    }
069
070    @Override
071    public Set<Player> getViewers() {
072        Set<Player> viewers = new HashSet<>(this.viewers);
073        viewers.addAll(entity.getViewers().values());
074        return viewers;
075    }
076
077    @Override
078    public boolean open( Player who ) {
079        return this.viewers.add( who );
080    }
081
082    @Override
083    public void onClose( Player who ) {
084        this.viewers.remove( who );
085    }
086
087    @PowerNukkitOnly
088    @Since("1.4.0.0-PN")
089    public Item getItemInHand() {
090        return this.getItem( MAIN_HAND);
091    }
092
093    @PowerNukkitOnly
094    @Since("1.4.0.0-PN")
095    public Item getItemInOffhand() {
096        return this.getItem( OFFHAND );
097    }
098
099    @PowerNukkitOnly
100    @Since("1.4.0.0-PN")
101    public boolean setItemInHand( Item item, boolean send ) {
102        return this.setItem( MAIN_HAND, item, send );
103    }
104
105    @PowerNukkitOnly
106    @Since("1.4.0.0-PN")
107    public boolean setItemInOffhand(Item item, boolean send ) {
108        return this.setItem( OFFHAND, item, send );
109    }
110
111    @Override
112    public void sendContents( Player target ) {
113        this.sendSlot( MAIN_HAND, target );
114        this.sendSlot( OFFHAND, target );
115    }
116
117    @Override
118    public void sendContents( Player... target ) {
119        for ( Player player : target ) {
120            this.sendContents( player );
121        }
122    }
123}
