001package cn.nukkit.form.window;
002
003import cn.nukkit.form.response.FormResponseModal;
004
005public class FormWindowModal extends FormWindow {
006
007    private final String type = "modal"; //This variable is used for JSON import operations. Do NOT delete :) -- @Snake1999
008    private String title = "";
009    private String content = "";
010    private String button1 = "";
011    private String button2 = "";
012
013    private FormResponseModal response = null;
014
015    public FormWindowModal(String title, String content, String trueButtonText, String falseButtonText) {
016        this.title = title;
017        this.content = content;
018        this.button1 = trueButtonText;
019        this.button2 = falseButtonText;
020    }
021
022    public String getTitle() {
023        return title;
024    }
025
026    public void setTitle(String title) {
027        this.title = title;
028    }
029
030    public String getContent() {
031        return content;
032    }
033
034    public void setContent(String content) {
035        this.content = content;
036    }
037
038    public String getButton1() {
039        return button1;
040    }
041
042    public void setButton1(String button1) {
043        this.button1 = button1;
044    }
045
046    public String getButton2() {
047        return button2;
048    }
049
050    public void setButton2(String button2) {
051        this.button2 = button2;
052    }
053
054    public FormResponseModal getResponse() {
055        return response;
056    }
057
058    public void setResponse(String data) {
059        if (data.equals("null")) {
060            closed = true;
061            return;
062        }
063        if (data.equals("true")) response = new FormResponseModal(0, button1);
064        else response = new FormResponseModal(1, button2);
065    }
066
067}
