001package cn.nukkit.block;
002
003import cn.nukkit.Player;
004import cn.nukkit.entity.Entity;
005import cn.nukkit.entity.EntityLiving;
006import cn.nukkit.item.Item;
007import cn.nukkit.math.AxisAlignedBB;
008import cn.nukkit.potion.Effect;
009
010public class BlockWitherRose extends BlockFlower {
011    public BlockWitherRose() {
012        this(0);
013    }
014
015    public BlockWitherRose(int meta) {
016        super(0);
017    }
018
019    @Override
020    public String getName() {
021        return "Wither Rose";
022    }
023
024    @Override
025    public int getId() {
026        return WITHER_ROSE;
027    }
028
029    @Override
030    public boolean canPlantOn(Block block) {
031        return super.canPlantOn(block) || block.getId() == Block.NETHERRACK || block.getId() == Block.SOUL_SAND;
032    }
033
034    @Override
035    public boolean onActivate(Item item, Player player) {
036        return false;
037    }
038
039    @Override
040    public void onEntityCollide(Entity entity) {
041        if (level.getServer().getDifficulty() != 0 && entity instanceof EntityLiving) {
042            EntityLiving living = (EntityLiving) entity;
043            if (!living.invulnerable && !living.hasEffect(Effect.WITHER)
044                    && (!(living instanceof Player) || !((Player) living).isCreative() && !((Player) living).isSpectator())) {
045                Effect effect = Effect.getEffect(Effect.WITHER);
046                effect.setDuration(40);
047                effect.setAmplifier(1);
048                living.addEffect(effect);
049            }
050        }
051    }
052
053    @Override
054    protected AxisAlignedBB recalculateCollisionBoundingBox() {
055        return this;
056    }
057
058    @Override
059    public boolean hasEntityCollision() {
060        return true;
061    }
062}
