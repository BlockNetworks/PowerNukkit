001package cn.nukkit.block;
002
003import cn.nukkit.api.PowerNukkitDifference;
004import cn.nukkit.math.AxisAlignedBB;
005import cn.nukkit.math.SimpleAxisAlignedBB;
006import cn.nukkit.utils.LevelException;
007
008/**
009 * Created on 2015/12/6 by xtypr.
010 * Package cn.nukkit.block in project Nukkit .
011 * @apiNote Implements BlockConnectable only in PowerNukkit
012 */
013@PowerNukkitDifference(info = "Made it implement BlockConnectable")
014public abstract class BlockThin extends BlockTransparent implements BlockConnectable {
015
016    protected BlockThin() {
017    }
018
019    @Override
020    public boolean isSolid() {
021        return false;
022    }
023
024    protected AxisAlignedBB recalculateBoundingBox() {
025        final double offNW = 7.0 / 16.0;
026        final double offSE = 9.0 / 16.0;
027        final double onNW = 0.0;
028        final double onSE = 1.0;
029        double w = offNW;
030        double e = offSE;
031        double n = offNW;
032        double s = offSE;
033        try {
034            boolean north = this.canConnect(this.north());
035            boolean south = this.canConnect(this.south());
036            boolean west = this.canConnect(this.west());
037            boolean east = this.canConnect(this.east());
038            w = west ? onNW : offNW;
039            e = east ? onSE : offSE;
040            n = north ? onNW : offNW;
041            s = south ? onSE : offSE;
042        } catch (LevelException ignore) {
043            //null sucks
044        }
045        return new SimpleAxisAlignedBB(
046                this.x + w,
047                this.y,
048                this.z + n,
049                this.x + e,
050                this.y + 1,
051                this.z + s
052        );
053    }
054
055    @PowerNukkitDifference(info = "Fixed connection logic for BE 1.16.0", since = "1.3.0.0-PN")
056    @Override
057    public boolean canConnect(Block block) {
058        switch (block.getId()) {
059            case GLASS_PANE:
060            case STAINED_GLASS_PANE:
061            case IRON_BARS:
062            case COBBLE_WALL:
063                return true;
064            default:
065                return block.isSolid();
066        }
067    }
068}
