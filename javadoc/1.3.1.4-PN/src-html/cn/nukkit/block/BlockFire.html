069    @Override
070    public void onEntityCollide(Entity entity) {
071        if (!entity.hasEffect(Effect.FIRE_RESISTANCE)) {
072            entity.attack(new EntityDamageByBlockEvent(this, entity, DamageCause.FIRE, 1));
073        }
074
075        EntityCombustByBlockEvent ev = new EntityCombustByBlockEvent(this, entity, 8);
076        if (entity instanceof EntityArrow) {
077            ev.setCancelled();
078        }
079        Server.getInstance().getPluginManager().callEvent(ev);
080        if (!ev.isCancelled() && entity.isAlive() && entity.noDamageTicks == 0) {
081            entity.setOnFire(ev.getDuration());
082        }
083    }

090    @Override
091    public int onUpdate(int type) {
092        if (type == Level.BLOCK_UPDATE_NORMAL || type == Level.BLOCK_UPDATE_RANDOM) {
093            if (!this.isBlockTopFacingSurfaceSolid(this.down()) && !this.canNeighborBurn()) {
094                BlockFadeEvent event = new BlockFadeEvent(this, get(AIR));
095                level.getServer().getPluginManager().callEvent(event);
096                if (!event.isCancelled()) {
097                    level.setBlock(this, event.getNewState(), true);
098                }
099            } else if (this.level.gameRules.getBoolean(GameRule.DO_FIRE_TICK) && !level.isUpdateScheduled(this, this)) {
100                level.scheduleUpdate(this, tickRate());
101            }
102
103            return Level.BLOCK_UPDATE_NORMAL;
104        } else if (type == Level.BLOCK_UPDATE_SCHEDULED && this.level.gameRules.getBoolean(GameRule.DO_FIRE_TICK)) {
105            boolean forever = this.down().getId() == Block.NETHERRACK || this.down().getId() == Block.MAGMA;
106
107            ThreadLocalRandom random = ThreadLocalRandom.current();
108
109            //TODO: END
110
111            if (!this.isBlockTopFacingSurfaceSolid(this.down()) && !this.canNeighborBurn()) {
112                BlockFadeEvent event = new BlockFadeEvent(this, get(AIR));
113                level.getServer().getPluginManager().callEvent(event);
114                if (!event.isCancelled()) {
115                    level.setBlock(this, event.getNewState(), true);
116                }
117            }
118
119            if (!forever && this.getLevel().isRaining() &&
120                    (this.getLevel().canBlockSeeSky(this) ||
121                            this.getLevel().canBlockSeeSky(this.east()) ||
122                            this.getLevel().canBlockSeeSky(this.west()) ||
123                            this.getLevel().canBlockSeeSky(this.south()) ||
124                            this.getLevel().canBlockSeeSky(this.north()))
125                    ) {
126                BlockFadeEvent event = new BlockFadeEvent(this, get(AIR));
127                level.getServer().getPluginManager().callEvent(event);
128                if (!event.isCancelled()) {
129                    level.setBlock(this, event.getNewState(), true);
130                }
131            } else {
132                int meta = this.getDamage();
133
134                if (meta < 15) {
135                    int newMeta = meta + random.nextInt(3);
136                    this.setDamage(Math.min(newMeta, 15));
137                    this.getLevel().setBlock(this, this, true);
138                }
139
140                this.getLevel().scheduleUpdate(this, this.tickRate() + random.nextInt(10));
141
142                if (!forever && !this.canNeighborBurn()) {
143                    if (!this.isBlockTopFacingSurfaceSolid(this.down()) || meta > 3) {
144                        BlockFadeEvent event = new BlockFadeEvent(this, get(AIR));
145                        level.getServer().getPluginManager().callEvent(event);
146                        if (!event.isCancelled()) {
147                            level.setBlock(this, event.getNewState(), true);
148                        }
149                    }
150                } else if (!forever && !(this.down().getBurnAbility() > 0) && meta == 15 && random.nextInt(4) == 0) {
151                    BlockFadeEvent event = new BlockFadeEvent(this, get(AIR));
152                    level.getServer().getPluginManager().callEvent(event);
153                    if (!event.isCancelled()) {
154                        level.setBlock(this, event.getNewState(), true);
155                    }
156                } else {
157                    int o = 0;
158
159                    //TODO: decrease the o if the rainfall values are high
160
161                    this.tryToCatchBlockOnFire(this.east(), 300 + o, meta);
162                    this.tryToCatchBlockOnFire(this.west(), 300 + o, meta);
163                    this.tryToCatchBlockOnFire(this.down(), 250 + o, meta);
164                    this.tryToCatchBlockOnFire(this.up(), 250 + o, meta);
165                    this.tryToCatchBlockOnFire(this.south(), 300 + o, meta);
166                    this.tryToCatchBlockOnFire(this.north(), 300 + o, meta);
167
168                    for (int x = (int) (this.x - 1); x <= (int) (this.x + 1); ++x) {
169                        for (int z = (int) (this.z - 1); z <= (int) (this.z + 1); ++z) {
170                            for (int y = (int) (this.y - 1); y <= (int) (this.y + 4); ++y) {
171                                if (x != (int) this.x || y != (int) this.y || z != (int) this.z) {
172                                    int k = 100;
173
174                                    if (y > this.y + 1) {
175                                        k += (y - (this.y + 1)) * 100;
176                                    }
177
178                                    Block block = this.getLevel().getBlock(new Vector3(x, y, z));
179                                    int chance = this.getChanceOfNeighborsEncouragingFire(block);
180
181                                    if (chance > 0) {
182                                        int t = (chance + 40 + this.getLevel().getServer().getDifficulty() * 7) / (meta + 30);
183
184                                        //TODO: decrease the t if the rainfall values are high
185
186                                        if (t > 0 && random.nextInt(k) <= t) {
187                                            int damage = meta + random.nextInt(5) / 4;
188
189                                            if (damage > 15) {
190                                                damage = 15;
191                                            }
192
193                                            BlockIgniteEvent e = new BlockIgniteEvent(block, this, null, BlockIgniteEvent.BlockIgniteCause.SPREAD);
194                                            this.level.getServer().getPluginManager().callEvent(e);
195
196                                            if (!e.isCancelled()) {
197                                                this.getLevel().setBlock(block, Block.get(BlockID.FIRE, damage), true);
198                                                this.getLevel().scheduleUpdate(block, this.tickRate());
199                                            }
200                                        }
201                                    }
202                                }
203                            }
204                        }
205                    }
206                }
207            }
208        }
209
210        return 0;
211    }

213    private void tryToCatchBlockOnFire(Block block, int bound, int damage) {
214        int burnAbility = block.getBurnAbility();
215
216        Random random = ThreadLocalRandom.current();
217
218        if (random.nextInt(bound) < burnAbility) {
219
220            if (random.nextInt(damage + 10) < 5) {
221                int meta = damage + random.nextInt(5) / 4;
222
223                if (meta > 15) {
224                    meta = 15;
225                }
226
227                BlockIgniteEvent e = new BlockIgniteEvent(block, this, null, BlockIgniteEvent.BlockIgniteCause.SPREAD);
228                this.level.getServer().getPluginManager().callEvent(e);
229
230                if (!e.isCancelled()) {
231                    this.getLevel().setBlock(block, Block.get(BlockID.FIRE, meta), true);
232                    this.getLevel().scheduleUpdate(block, this.tickRate());
233                }
234            } else {
235                BlockBurnEvent ev = new BlockBurnEvent(block);
236                this.getLevel().getServer().getPluginManager().callEvent(ev);
237
238                if (!ev.isCancelled()) {
239                    this.getLevel().setBlock(block, Block.get(BlockID.AIR), true);
240                }
241            }
242
243            if (block instanceof BlockTNT) {
244                ((BlockTNT) block).prime();
245            }
246        }
247    }
248
249    private int getChanceOfNeighborsEncouragingFire(Block block) {
250        if (block.getId() != AIR) {
251            return 0;
252        } else {
253            int chance = 0;
254            chance = Math.max(chance, block.east().getBurnChance());
255            chance = Math.max(chance, block.west().getBurnChance());
256            chance = Math.max(chance, block.down().getBurnChance());
257            chance = Math.max(chance, block.up().getBurnChance());
258            chance = Math.max(chance, block.south().getBurnChance());
259            chance = Math.max(chance, block.north().getBurnChance());
260            return chance;
261        }
262    }

264    public boolean canNeighborBurn() {
265        for (BlockFace face : BlockFace.values()) {
266            if (this.getSide(face).getBurnChance() > 0) {
267                return true;
268            }
269        }
270
271        return false;
272    }
273
274    public boolean isBlockTopFacingSurfaceSolid(Block block) {
275        if (block != null) {
276            if (block.isSolid()) {
277                return true;
278            } else {
279                if (block instanceof BlockStairs &&
280                        (block.getDamage() & 4) == 4) {
281
282                    return true;
283                } else if (block instanceof BlockSlab &&
284                        (block.getDamage() & 8) == 8) {
285
286                    return true;
287                } else if (block instanceof BlockSnowLayer &&
288                        (block.getDamage() & 7) == 7) {
289
290                    return true;
291                }
292            }
293        }
294
295        return false;
296    }
