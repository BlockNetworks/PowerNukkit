001package cn.nukkit.inventory;
002
003import cn.nukkit.Player;
004import cn.nukkit.item.Item;
005import cn.nukkit.network.protocol.InventorySlotPacket;
006import cn.nukkit.network.protocol.types.ContainerIds;
007
008import java.util.HashMap;
009
010public class PlayerUIInventory extends BaseInventory {
011    private final Player player;
012
013    private final PlayerCursorInventory cursorInventory;
014    private final CraftingGrid craftingGrid;
015    private final BigCraftingGrid bigCraftingGrid;
016
017    public PlayerUIInventory(Player player) {
018        super(player, InventoryType.UI, new HashMap<>(), 51);
019        this.player = player;
020
021        this.cursorInventory = new PlayerCursorInventory(this);
022        this.craftingGrid = new CraftingGrid(this);
023        this.bigCraftingGrid = new BigCraftingGrid(this);
024    }
025
026    public PlayerCursorInventory getCursorInventory() {
027        return cursorInventory;
028    }
029
030    public CraftingGrid getCraftingGrid() {
031        return craftingGrid;
032    }
033
034    public BigCraftingGrid getBigCraftingGrid() {
035        return bigCraftingGrid;
036    }
037
038    @Override
039    public void setSize(int size) {
040        throw new UnsupportedOperationException("UI size is immutable");
041    }
042
043    @Override
044    public void sendSlot(int index, Player... target) {
045        InventorySlotPacket pk = new InventorySlotPacket();
046        pk.slot = index;
047        pk.item = this.getItem(index);
048
049        for (Player p : target) {
050            if (p == this.getHolder()) {
051                pk.inventoryId = ContainerIds.UI;
052            } else {
053                int id;
054
055                if ((id = p.getWindowId(this)) == ContainerIds.NONE) {
056                    this.close(p);
057                    continue;
058                }
059                pk.inventoryId = id;
060            }
061            p.dataPacket(pk);
062        }
063    }
064
065    @Override
066    public void sendContents(Player... target) {
067        //doesn't work here
068    }
069    
070    @Override
071    public void onSlotChange(int index, Item before, boolean send) {
072        switch (player.craftingType) {
073            case Player.CRAFTING_GRINDSTONE:
074                if (index >= GrindstoneInventory.OFFSET) {
075                    Inventory inventory = player.getWindowById(Player.ANVIL_WINDOW_ID);
076                    if (inventory instanceof GrindstoneInventory && (index == 50 || index - GrindstoneInventory.OFFSET < inventory.getSize())) {
077                        inventory.onSlotChange(index == 50 ? 2 : index - GrindstoneInventory.OFFSET, before, send);
078                    }
079                }
080            case Player.CRAFTING_ANVIL:
081                if (index >= AnvilInventory.OFFSET) {
082                    Inventory inventory = player.getWindowById(Player.ANVIL_WINDOW_ID);
083                    if (inventory instanceof AnvilInventory && (index == 50 || index - AnvilInventory.OFFSET < inventory.getSize())) {
084                        inventory.onSlotChange(index == 50 ? 2 : index - AnvilInventory.OFFSET, before, send);
085                    }
086                }
087                return;
088        }
089        super.onSlotChange(index, before, send);
090    }
091    
092    public void onSlotChangeBase(int index, Item before, boolean send) {
093        super.onSlotChange(index, before, send);
094    }
095    
096    @Override
097    public int getSize() {
098        return 51;
099    }
100
101    @Override
102    public Player getHolder() {
103        return player;
104    }
105}
