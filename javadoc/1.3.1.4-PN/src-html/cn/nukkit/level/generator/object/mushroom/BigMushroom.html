001package cn.nukkit.level.generator.object.mushroom;
002
003import cn.nukkit.block.Block;
004import cn.nukkit.block.BlockID;
005import cn.nukkit.level.ChunkManager;
006import cn.nukkit.level.generator.object.BasicGenerator;
007import cn.nukkit.math.NukkitRandom;
008import cn.nukkit.math.Vector3;
009
010public class BigMushroom extends BasicGenerator {
011    public static final int NORTH_WEST = 1;
012    public static final int NORTH = 2;
013    public static final int NORTH_EAST = 3;
014    public static final int WEST = 4;
015    public static final int CENTER = 5;
016    public static final int EAST = 6;
017    public static final int SOUTH_WEST = 7;
018    public static final int SOUTH = 8;
019    public static final int SOUTH_EAST = 9;
020    public static final int STEM = 10;
021    public static final int ALL_INSIDE = 0;
022    public static final int ALL_OUTSIDE = 14;
023    public static final int ALL_STEM = 15;
024
025    public static final int BROWN = 0;
026    public static final int RED = 1;
027    /**
028     * The mushroom type. 0 for brown, 1 for red.
029     */
030    private int mushroomType;
031
032    public BigMushroom(int mushroomType) {
033        this.mushroomType = mushroomType;
034    }
035
036    public BigMushroom() {
037        this.mushroomType = -1;
038    }
039
040    public boolean generate(ChunkManager level, NukkitRandom rand, Vector3 position) {
041        int block = this.mushroomType;
042        if (block < 0) {
043            block = rand.nextBoolean() ? RED : BROWN;
044        }
045
046        Block mushroom = block == 0 ? Block.get(BlockID.BROWN_MUSHROOM_BLOCK) : Block.get(BlockID.RED_MUSHROOM_BLOCK);
047
048        int i = rand.nextBoundedInt(3) + 4;
049
050        if (rand.nextBoundedInt(12) == 0) {
051            i *= 2;
052        }
053
054        boolean flag = true;
055
056        if (position.getY() >= 1 && position.getY() + i + 1 < 256) {
057            for (int j = position.getFloorY(); j <= position.getY() + 1 + i; ++j) {
058                int k = 3;
059
060                if (j <= position.getY() + 3) {
061                    k = 0;
062                }
063
064                Vector3 pos = new Vector3();
065
066                for (int l = position.getFloorX() - k; l <= position.getX() + k && flag; ++l) {
067                    for (int i1 = position.getFloorZ() - k; i1 <= position.getZ() + k && flag; ++i1) {
068                        if (j >= 0 && j < 256) {
069                            pos.setComponents(l, j, i1);
070                            int material = level.getBlockIdAt(pos.getFloorX(), pos.getFloorY(), pos.getFloorZ());
071
072                            if (material != Block.AIR && material != Block.LEAVES) {
073                                flag = false;
074                            }
075                        } else {
076                            flag = false;
077                        }
078                    }
079                }
080            }
081
082            if (!flag) {
083                return false;
084            } else {
085                Vector3 pos2 = position.down();
086                int block1 = level.getBlockIdAt(pos2.getFloorX(), pos2.getFloorY(), pos2.getFloorZ());
087
088                if (block1 != Block.DIRT && block1 != Block.GRASS && block1 != Block.MYCELIUM) {
089                    return false;
090                } else {
091                    int k2 = position.getFloorY() + i;
092
093                    if (block == RED) {
094                        k2 = position.getFloorY() + i - 3;
095                    }
096
097                    for (int l2 = k2; l2 <= position.getY() + i; ++l2) {
098                        int j3 = 1;
099
100                        if (l2 < position.getY() + i) {
101                            ++j3;
102                        }
103
104                        if (block == BROWN) {
105                            j3 = 3;
106                        }
107
108                        int k3 = position.getFloorX() - j3;
109                        int l3 = position.getFloorX() + j3;
110                        int j1 = position.getFloorZ() - j3;
111                        int k1 = position.getFloorZ() + j3;
112
113                        for (int l1 = k3; l1 <= l3; ++l1) {
114                            for (int i2 = j1; i2 <= k1; ++i2) {
115                                int j2 = 5;
116
117                                if (l1 == k3) {
118                                    --j2;
119                                } else if (l1 == l3) {
120                                    ++j2;
121                                }
122
123                                if (i2 == j1) {
124                                    j2 -= 3;
125                                } else if (i2 == k1) {
126                                    j2 += 3;
127                                }
128
129                                int meta = j2;
130
131                                if (block == BROWN || l2 < position.getY() + i) {
132                                    if ((l1 == k3 || l1 == l3) && (i2 == j1 || i2 == k1)) {
133                                        continue;
134                                    }
135
136                                    if (l1 == position.getX() - (j3 - 1) && i2 == j1) {
137                                        meta = NORTH_WEST;
138                                    }
139
140                                    if (l1 == k3 && i2 == position.getZ() - (j3 - 1)) {
141                                        meta = NORTH_WEST;
142                                    }
143
144                                    if (l1 == position.getX() + (j3 - 1) && i2 == j1) {
145                                        meta = NORTH_EAST;
146                                    }
147
148                                    if (l1 == l3 && i2 == position.getZ() - (j3 - 1)) {
149                                        meta = NORTH_EAST;
150                                    }
151
152                                    if (l1 == position.getX() - (j3 - 1) && i2 == k1) {
153                                        meta = SOUTH_WEST;
154                                    }
155
156                                    if (l1 == k3 && i2 == position.getZ() + (j3 - 1)) {
157                                        meta = SOUTH_WEST;
158                                    }
159
160                                    if (l1 == position.getX() + (j3 - 1) && i2 == k1) {
161                                        meta = SOUTH_EAST;
162                                    }
163
164                                    if (l1 == l3 && i2 == position.getZ() + (j3 - 1)) {
165                                        meta = SOUTH_EAST;
166                                    }
167                                }
168
169                                if (meta == CENTER && l2 < position.getY() + i) {
170                                    meta = ALL_INSIDE;
171                                }
172
173                                if (position.getY() >= position.getY() + i - 1 || meta != ALL_INSIDE) {
174                                    Vector3 blockPos = new Vector3(l1, l2, i2);
175
176                                    if (!Block.solid[level.getBlockIdAt(blockPos.getFloorX(), blockPos.getFloorY(), blockPos.getFloorZ())]) {
177                                        mushroom.setDamage(meta);
178                                        this.setBlockAndNotifyAdequately(level, blockPos, mushroom);
179                                    }
180                                }
181                            }
182                        }
183                    }
184
185                    for (int i3 = 0; i3 < i; ++i3) {
186                        Vector3 pos = position.up(i3);
187                        int id = level.getBlockIdAt(pos.getFloorX(), pos.getFloorY(), pos.getFloorZ());
188
189                        if (!Block.solid[id]) {
190                            mushroom.setDamage(STEM);
191                            this.setBlockAndNotifyAdequately(level, pos, mushroom);
192                        }
193                    }
194
195                    return true;
196                }
197            }
198        } else {
199            return false;
200        }
201    }
202}
