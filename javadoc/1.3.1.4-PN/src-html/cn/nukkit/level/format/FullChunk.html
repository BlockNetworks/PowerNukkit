019public interface FullChunk extends Cloneable {
020
021    int getX();
022
023    int getZ();
024
025    default void setPosition(int x, int z) {
026        setX(x);
027        setZ(z);
028    }
029
030    void setX(int x);
031
032    void setZ(int z);
033
034    long getIndex();
035
036    LevelProvider getProvider();
037
038    void setProvider(LevelProvider provider);
039
040    @Deprecated
041    @DeprecationDetails(reason = "Does not support hyper ids", since = "1.3.0.0-PN")
042    int getFullBlock(int x, int y, int z);
043    
044    @Deprecated
045    @DeprecationDetails(reason = "Does not support hyper ids", since = "1.3.0.0-PN")
046    int getFullBlock(int x, int y, int z, int layer);
047
048    @PowerNukkitOnly
049    @Since("1.3.0.0-PN")
050    default int getBlockRuntimeId(int x, int y, int z) {
051        return getBlockRuntimeId(x, y, z, 0);
052    }
053    
054    @PowerNukkitOnly
055    @Since("1.3.0.0-PN")
056    default int getBlockRuntimeId(int x, int y, int z, int layer) {
057        return GlobalBlockPalette.getOrCreateRuntimeId(getBlockId(x, y, z, layer), getBlockData(x, y, z, layer));
058    }
059    
060    @PowerNukkitOnly
061    @Since("1.3.0.0-PN")
062    default int[] getBlockState(int x, int y, int z) {
063        return getBlockState(x, y, z, 0);
064    }
065
066    @PowerNukkitOnly
067    @Since("1.3.0.0-PN")
068    default int[] getBlockState(int x, int y, int z, int layer) {
069        int full = getFullBlock(x, y, z, layer);
070        return new int[] { full >> Block.DATA_BITS, full & Block.DATA_MASK };
071    }
072
073    Block getAndSetBlock(int x, int y, int z, Block block);
074    Block getAndSetBlock(int x, int y, int z, int layer, Block block);
075
076    @Deprecated
077    @DeprecationDetails(reason = "Does not support hyper ids", since = "1.3.0.0-PN", replaceWith = "setBlock(int x, int y, int z, int  blockId, int  meta)")
078    default boolean setFullBlockId(int x, int y, int z, int fullId) {
079        return setFullBlockId(x, y, z, 0, fullId >> Block.DATA_BITS);
080    }
081
082    @Deprecated
083    @DeprecationDetails(reason = "Does not support hyper ids", since = "1.3.0.0-PN", replaceWith = "setBlockAtLayer(int x, int y, int z, int layer, int  blockId)")
084    default boolean setFullBlockId(int x, int y, int z, int layer, int fullId) {
085        return setBlockAtLayer(x, y, z, layer, fullId >> Block.DATA_BITS, fullId & Block.DATA_MASK);
086    }
087
088    boolean setBlock(int x, int y, int z, int blockId);
089    boolean setBlockAtLayer(int x, int y, int z, int layer, int  blockId);
090
091    boolean setBlock(int x, int y, int z, int  blockId, int  meta);
092    boolean setBlockAtLayer(int x, int y, int z, int layer, int blockId, int  meta);
093
094    int getBlockId(int x, int y, int z);
095    int getBlockId(int x, int y, int z, int layer);
096
097    void setBlockId(int x, int y, int z, int id);
098    void setBlockId(int x, int y, int z, int layer, int id);
099
100    int getBlockData(int x, int y, int z);
101    int getBlockData(int x, int y, int z, int layer);
102
103    void setBlockData(int x, int y, int z, int data);
104    void setBlockData(int x, int y, int z, int layer, int data);
105
106    int getBlockExtraData(int x, int y, int z);
107
108    void setBlockExtraData(int x, int y, int z, int data);
109
110    int getBlockSkyLight(int x, int y, int z);
111
112    void setBlockSkyLight(int x, int y, int z, int level);
113
114    int getBlockLight(int x, int y, int z);
115
116    void setBlockLight(int x, int y, int z, int level);
117
118    int getHighestBlockAt(int x, int z);
119
120    int getHighestBlockAt(int x, int z, boolean cache);
121
122    int getHeightMap(int x, int z);
123
124    void setHeightMap(int x, int z, int value);
125
126    void recalculateHeightMap();
127
128    int recalculateHeightMapColumn(int chunkX, int chunkZ);
129
130    void populateSkyLight();
131
132    int getBiomeId(int x, int z);
133
134    void setBiomeId(int x, int z, byte biomeId);
135
136    default void setBiomeId(int x, int z, int biomeId)  {
137        setBiomeId(x, z, (byte) biomeId);
138    }
139
140    default void setBiome(int x, int z, Biome biome) {
141        setBiomeId(x, z, (byte) biome.getId());
142    }
143
144    boolean isLightPopulated();
145
146    void setLightPopulated();
147
148    void setLightPopulated(boolean value);
149
150    boolean isPopulated();
151
152    void setPopulated();
153
154    void setPopulated(boolean value);
155
156    boolean isGenerated();
157
158    void setGenerated();
159
160    void setGenerated(boolean value);
161
162    void addEntity(Entity entity);
163
164    void removeEntity(Entity entity);
165
166    void addBlockEntity(BlockEntity blockEntity);
167
168    void removeBlockEntity(BlockEntity blockEntity);
169
170    Map<Long, Entity> getEntities();
171
172    Map<Long, BlockEntity> getBlockEntities();
173
174    BlockEntity getTile(int x, int y, int z);
175
176    boolean isLoaded();
177
178    boolean load() throws IOException;
179
180    boolean load(boolean generate) throws IOException;
181
182    boolean unload() throws Exception;
183
184    boolean unload(boolean save) throws Exception;
185
186    boolean unload(boolean save, boolean safe) throws Exception;
187
188    void initChunk();
189
190    byte[] getBiomeIdArray();
191
192    byte[] getHeightMapArray();
193
194    byte[] getBlockIdArray(int layer);
195    default byte[] getBlockIdArray() {
196        return getBlockIdArray(0);
197    }
198
199    byte[] getBlockDataArray(int layer);
200    default byte[] getBlockDataArray() {
201        return getBlockDataArray(0);
202    }
203
204    Map<Integer, Integer> getBlockExtraDataArray();
205
206    byte[] getBlockSkyLightArray();
207
208    byte[] getBlockLightArray();
209
210    byte[] toBinary();
211
212    byte[] toFastBinary();
213
214    boolean hasChanged();
215
216    void setChanged();
217
218    void setChanged(boolean changed);
