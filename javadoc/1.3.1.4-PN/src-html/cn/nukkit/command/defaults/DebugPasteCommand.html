001package cn.nukkit.command.defaults;
002
003import cn.nukkit.Server;
004import cn.nukkit.command.CapturingCommandSender;
005import cn.nukkit.command.CommandSender;
006import cn.nukkit.network.protocol.ProtocolInfo;
007import cn.nukkit.plugin.Plugin;
008import cn.nukkit.plugin.PluginDescription;
009import cn.nukkit.scheduler.AsyncTask;
010import cn.nukkit.utils.HastebinUtility;
011import cn.nukkit.utils.MainLogger;
012import cn.nukkit.utils.Utils;
013
014import java.io.File;
015import java.io.IOException;
016import java.lang.management.ManagementFactory;
017
018public class DebugPasteCommand extends VanillaCommand {
019
020    public DebugPasteCommand(String name) {
021        super(name, "%nukkit.command.debug.description", "%nukkit.command.debug.usage");
022        this.setPermission("nukkit.command.debug.perform");
023    }
024
025    @Override
026    public boolean execute(CommandSender sender, String commandLabel, String[] args) {
027        if (!this.testPermission(sender)) {
028            return true;
029        }
030        Server server = Server.getInstance();
031        server.getScheduler().scheduleAsyncTask(new AsyncTask() {
032            @Override
033            public void onRun() {
034                try {
035                    CapturingCommandSender capturing = new CapturingCommandSender();
036                    capturing.setOp(true);
037                    new StatusCommand("status").execute(capturing, "status", new String[]{});
038                    String dataPath = server.getDataPath();
039                    String status = HastebinUtility.upload(capturing.getCleanCapture());
040                    String nukkitYML = HastebinUtility.upload(new File(dataPath, "nukkit.yml"));
041                    String serverProperties = HastebinUtility.upload(new File(dataPath, "server.properties"));
042                    String latestLog = HastebinUtility.upload(new File(dataPath, "/logs/server.log"));
043                    String threadDump = HastebinUtility.upload(Utils.getAllThreadDumps());
044
045                    StringBuilder b = new StringBuilder();
046                    b.append("# Files\n");
047                    b.append("links.nukkit_yml: ").append(nukkitYML).append('\n');
048                    b.append("links.server_properties: ").append(serverProperties).append('\n');
049                    b.append("links.server_status: ").append(status).append('\n');
050                    b.append("links.server_log: ").append(latestLog).append('\n');
051                    b.append("links.thread_dump: ").append(threadDump).append('\n');
052                    b.append("\n# Server Information\n");
053    
054                    b.append("server.name: ").append(server.getName()).append('\n');
055                    b.append("version.api: ").append(server.getApiVersion()).append('\n');
056                    b.append("version.nukkit: ").append(server.getNukkitVersion()).append('\n');
057                    b.append("version.git: ").append(server.getGitCommit()).append('\n');
058                    b.append("version.codename: ").append(server.getCodename()).append('\n');
059                    b.append("version.minecraft: ").append(server.getVersion()).append('\n');
060                    b.append("version.protocol: ").append(ProtocolInfo.CURRENT_PROTOCOL).append('\n');
061                    b.append("plugins:");
062                    for (Plugin plugin : server.getPluginManager().getPlugins().values()) {
063                        boolean enabled = plugin.isEnabled();
064                        String name = plugin.getName();
065                        PluginDescription desc = plugin.getDescription();
066                        String version = desc.getVersion();
067                        b.append("\n  ")
068                                .append(name)
069                                .append(":\n    ")
070                                .append("version: '")
071                                .append(version)
072                                .append('\'')
073                                .append("\n    enabled: ")
074                                .append(enabled);
075                    }
076                    b.append("\n\n# Java Details\n");
077                    Runtime runtime = Runtime.getRuntime();
078                    b.append("memory.free: ").append(runtime.freeMemory()).append('\n');
079                    b.append("memory.max: ").append(runtime.maxMemory()).append('\n');
080                    b.append("cpu.runtime: ").append(ManagementFactory.getRuntimeMXBean().getUptime()).append('\n');
081                    b.append("cpu.processors: ").append(runtime.availableProcessors()).append('\n');
082                    b.append("java.specification.version: '").append(System.getProperty("java.specification.version")).append("'\n");
083                    b.append("java.vendor: '").append(System.getProperty("java.vendor")).append("'\n");
084                    b.append("java.version: '").append(System.getProperty("java.version")).append("'\n");
085                    b.append("os.arch: '").append(System.getProperty("os.arch")).append("'\n");
086                    b.append("os.name: '").append(System.getProperty("os.name")).append("'\n");
087                    b.append("os.version: '").append(System.getProperty("os.version")).append("'\n\n");
088                    b.append("\n# Create a ticket: https://github.com/PowerNukkit/PowerNukkit/issues/new");
089                    String link = HastebinUtility.upload(b.toString());
090                    sender.sendMessage(link);
091                } catch (IOException e) {
092                    MainLogger.getLogger().logException(e);
093                }
094            }
095        });
096        return true;
097    }
098}
