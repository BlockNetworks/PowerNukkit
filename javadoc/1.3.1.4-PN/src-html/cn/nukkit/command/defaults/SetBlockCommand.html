001package cn.nukkit.command.defaults;
002
003import cn.nukkit.Player;
004import cn.nukkit.block.Block;
005import cn.nukkit.block.BlockID;
006import cn.nukkit.command.CommandSender;
007import cn.nukkit.command.data.CommandParamType;
008import cn.nukkit.command.data.CommandParameter;
009import cn.nukkit.item.Item;
010import cn.nukkit.lang.TranslationContainer;
011import cn.nukkit.level.Level;
012import cn.nukkit.level.Position;
013import cn.nukkit.math.BlockFace;
014import cn.nukkit.math.Vector3;
015
016import java.util.Arrays;
017
018public class SetBlockCommand extends VanillaCommand {
019    public SetBlockCommand(String name) {
020        super(name, "%nukkit.command.setblock.description", "%nukkit.command.setblock.usage");
021        this.setPermission("nukkit.command.setblock");
022        this.commandParameters.clear();
023        this.commandParameters.put("default", new CommandParameter[]{
024                new CommandParameter("position", CommandParamType.POSITION, false),
025                new CommandParameter("tileName", false, Arrays.stream(BlockID.class.getDeclaredFields()).map(f-> f.getName().toLowerCase()).toArray(String[]::new)),
026                new CommandParameter("tileData", CommandParamType.INT, true),
027                new CommandParameter("oldBlockHandling", true, new String[]{"destroy", "keep", "replace"})
028        });
029    }
030    
031    @Override
032    public boolean execute(CommandSender sender, String commandLabel, String[] args) {
033        if (!this.testPermission(sender)) {
034            return true;
035        }
036
037        if (args.length < 4) {
038            sender.sendMessage(new TranslationContainer("commands.generic.usage", this.usageMessage));
039
040            return true;
041        }
042
043        if (!(sender instanceof Player)) {
044            sender.sendMessage(new TranslationContainer("commands.setblock.outOfWorld"));
045            return true;
046        }
047        Player player = (Player) sender;
048
049        double x;
050        double y;
051        double z;
052        int data = 0;
053        try {
054            x = parseTilde(args[0], player.x);
055            y = parseTilde(args[1], player.y);
056            z = parseTilde(args[2], player.z);
057
058            if (args.length > 4) {
059                data = Integer.parseInt(args[4]);
060            }
061        } catch (NumberFormatException|IndexOutOfBoundsException ignored) {
062            sender.sendMessage(new TranslationContainer("commands.generic.usage", this.usageMessage));
063            return true;
064        }
065
066        String oldBlockHandling = "replace";
067        if (args.length > 5) {
068            oldBlockHandling = args[5].toLowerCase();
069            switch (oldBlockHandling) {
070                case "destroy":
071                case "keep":
072                case "replace":
073                    break;
074                default:
075                    sender.sendMessage(new TranslationContainer("commands.generic.usage", this.usageMessage));
076                    return true;
077            }
078        }
079
080        Block block;
081        try {
082            int blockId = Integer.parseInt(args[3]);
083            block = Block.get(blockId, data);
084        } catch (NullPointerException|NumberFormatException|IndexOutOfBoundsException ignored) {
085            try {
086                int blockId = BlockID.class.getField(args[3].toUpperCase()).getInt(null);
087                block = Block.get(blockId, data);
088            } catch (NullPointerException|IndexOutOfBoundsException|ReflectiveOperationException ignored2) {
089                sender.sendMessage(new TranslationContainer("commands.setblock.notFound", args[3]));
090                return true;
091            }
092        }
093
094        if (y < 0 || y > 255) {
095            sender.sendMessage(new TranslationContainer("commands.setblock.outOfWorld"));
096            return true;
097        }
098
099        Level level = player.getLevel();
100
101        Position position = new Position(x, y, z, player.getLevel());
102        Block current = level.getBlock(position);
103        if (current.getId() != Block.AIR) {
104            switch (oldBlockHandling) {
105                case "destroy":
106                    level.useBreakOn(position, null, Item.get(Item.AIR), player, true, true);
107                    current = level.getBlock(position);
108                    break;
109                case "keep":
110                    sender.sendMessage(new TranslationContainer("commands.setblock.noChange"));
111                    return true;
112            }
113        }
114
115        if (current.getId() == block.getId() && current.getDamage() == block.getDamage()) {
116            sender.sendMessage(new TranslationContainer("commands.setblock.noChange"));
117            return true;
118        }
119
120        Item item = block.toItem();
121        block.position(position);
122        if(block.place(item, block, block.down(), BlockFace.UP, 0.5, 0.5, 0.5, player)) {
123            if (args.length > 4) {
124                level.setBlockDataAt((int) x, (int) y, (int) z, data);
125            }
126        //if (level.setBlock(position, block, true, true)) {
127            sender.sendMessage(new TranslationContainer("commands.setblock.success"));
128        } else {
129            sender.sendMessage(new TranslationContainer("commands.setblock.failed"));
130        }
131        return true;
132    }
133}
