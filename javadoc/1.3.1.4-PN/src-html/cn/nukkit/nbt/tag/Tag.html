001package cn.nukkit.nbt.tag;
002
003import cn.nukkit.nbt.stream.NBTInputStream;
004import cn.nukkit.nbt.stream.NBTOutputStream;
005
006import java.io.IOException;
007import java.io.PrintStream;
008import java.util.Objects;
009
010public abstract class Tag {
011    public static final byte TAG_End = 0;
012    public static final byte TAG_Byte = 1;
013    public static final byte TAG_Short = 2;
014    public static final byte TAG_Int = 3;
015    public static final byte TAG_Long = 4;
016    public static final byte TAG_Float = 5;
017    public static final byte TAG_Double = 6;
018    public static final byte TAG_Byte_Array = 7;
019    public static final byte TAG_String = 8;
020    public static final byte TAG_List = 9;
021    public static final byte TAG_Compound = 10;
022    public static final byte TAG_Int_Array = 11;
023
024    private String name;
025
026    abstract void write(NBTOutputStream dos) throws IOException;
027
028    abstract void load(NBTInputStream dis) throws IOException;
029
030    public abstract String toString();
031
032    public abstract byte getId();
033
034    protected Tag(String name) {
035        if (name == null) {
036            this.name = "";
037        } else {
038            this.name = name;
039        }
040    }
041
042    @Override
043    public boolean equals(Object obj) {
044        if (!(obj instanceof Tag)) {
045            return false;
046        }
047        Tag o = (Tag) obj;
048        return getId() == o.getId() && !(name == null && o.name != null || name != null && o.name == null) && !(name != null && !name.equals(o.name));
049    }
050    
051    @Override
052    public int hashCode() {
053        return Objects.hash(getId(), name);
054    }
055    
056    public void print(PrintStream out) {
057        print("", out);
058    }
059
060    public void print(String prefix, PrintStream out) {
061        String name = getName();
062
063        out.print(prefix);
064        out.print(getTagName(getId()));
065        if (name.length() > 0) {
066            out.print("(\"" + name + "\")");
067        }
068        out.print(": ");
069        if (getId() != TAG_Compound && getId() != TAG_List) {
070            out.println(parseValue());
071        }
072    }
073
074    public Tag setName(String name) {
075        if (name == null) {
076            this.name = "";
077        } else {
078            this.name = name;
079        }
080        return this;
081    }
082
083    public String getName() {
084        if (name == null) return "";
085        return name;
086    }
087
088    public static Tag readNamedTag(NBTInputStream dis) throws IOException {
089        byte type = dis.readByte();
090        if (type == 0) return new EndTag();
091
092        String name = dis.readUTF();
093
094        Tag tag = newTag(type, name);
095
096        tag.load(dis);
097        return tag;
098    }
099
100    public static void writeNamedTag(Tag tag, NBTOutputStream dos) throws IOException {
101        writeNamedTag(tag, tag.getName(), dos);
102    }
103
104    public static void writeNamedTag(Tag tag, String name, NBTOutputStream dos) throws IOException {
105        dos.writeByte(tag.getId());
106        if (tag.getId() == Tag.TAG_End) return;
107        dos.writeUTF(name);
108
109        tag.write(dos);
110    }
111
112    public static Tag newTag(byte type, String name) {
113        switch (type) {
114            case TAG_End:
115                return new EndTag();
116            case TAG_Byte:
117                return new ByteTag(name);
118            case TAG_Short:
119                return new ShortTag(name);
120            case TAG_Int:
121                return new IntTag(name);
122            case TAG_Long:
123                return new LongTag(name);
124            case TAG_Float:
125                return new FloatTag(name);
126            case TAG_Double:
127                return new DoubleTag(name);
128            case TAG_Byte_Array:
129                return new ByteArrayTag(name);
130            case TAG_Int_Array:
131                return new IntArrayTag(name);
132            case TAG_String:
133                return new StringTag(name);
134            case TAG_List:
135                return new ListTag<>(name);
136            case TAG_Compound:
137                return new CompoundTag(name);
138        }
139        return new EndTag();
140    }
141
142    public static String getTagName(byte type) {
143        switch (type) {
144            case TAG_End:
145                return "TAG_End";
146            case TAG_Byte:
147                return "TAG_Byte";
148            case TAG_Short:
149                return "TAG_Short";
150            case TAG_Int:
151                return "TAG_Int";
152            case TAG_Long:
153                return "TAG_Long";
154            case TAG_Float:
155                return "TAG_Float";
156            case TAG_Double:
157                return "TAG_Double";
158            case TAG_Byte_Array:
159                return "TAG_Byte_Array";
160            case TAG_Int_Array:
161                return "TAG_Int_Array";
162            case TAG_String:
163                return "TAG_String";
164            case TAG_List:
165                return "TAG_List";
166            case TAG_Compound:
167                return "TAG_Compound";
168        }
169        return "UNKNOWN";
170    }
171
172    public abstract Tag copy();
173
174    public abstract Object parseValue();
175}
