001package cn.nukkit.command.defaults;
002
003import cn.nukkit.Player;
004import cn.nukkit.command.Command;
005import cn.nukkit.command.CommandSender;
006import cn.nukkit.command.data.CommandParamType;
007import cn.nukkit.command.data.CommandParameter;
008import cn.nukkit.event.player.PlayerKickEvent;
009import cn.nukkit.lang.TranslationContainer;
010import cn.nukkit.nbt.NBTIO;
011import cn.nukkit.nbt.tag.CompoundTag;
012
013import java.io.File;
014import java.io.FileInputStream;
015import java.io.IOException;
016import java.net.InetAddress;
017import java.net.UnknownHostException;
018import java.util.ArrayList;
019import java.util.regex.Pattern;
020
021/**
022 * author: MagicDroidX
023 * Nukkit Project
024 */
025public class BanIpCommand extends VanillaCommand {
026
027    public BanIpCommand(String name) {
028        super(name, "%nukkit.command.ban.ip.description", "%commands.banip.usage");
029        this.setPermission("nukkit.command.ban.ip");
030        this.setAliases(new String[]{"banip"});
031        this.commandParameters.clear();
032        this.commandParameters.put("default", new CommandParameter[]{
033                new CommandParameter("player", CommandParamType.TARGET, false),
034                new CommandParameter("reason", CommandParamType.STRING, true)
035        });
036    }
037
038    @Override
039    public boolean execute(CommandSender sender, String commandLabel, String[] args) {
040        if (!this.testPermission(sender)) {
041            return true;
042        }
043
044        if (args.length == 0) {
045            sender.sendMessage(new TranslationContainer("commands.generic.usage", this.usageMessage));
046
047            return false;
048        }
049
050        String value = args[0];
051        String reason = "";
052        for (int i = 1; i < args.length; i++) {
053            reason += args[i] + " ";
054        }
055
056        if (reason.length() > 0) {
057            reason = reason.substring(0, reason.length() - 1);
058        }
059
060        if (Pattern.matches("^(25[0-5]|2[0-4][0-9]|[0-1]{1}[0-9]{2}|[1-9]{1}[0-9]{1}|[1-9])\\.(25[0-5]|2[0-4][0-9]|[0-1]{1}[0-9]{2}|[1-9]{1}[0-9]{1}|[1-9]|0)\\.(25[0-5]|2[0-4][0-9]|[0-1]{1}[0-9]{2}|[1-9]{1}[0-9]{1}|[1-9]|0)\\.(25[0-5]|2[0-4][0-9]|[0-1]{1}[0-9]{2}|[1-9]{1}[0-9]{1}|[0-9])$", value)) {
061            this.processIPBan(value, sender, reason);
062
063            Command.broadcastCommandMessage(sender, new TranslationContainer("commands.banip.success", value));
064        } else {
065            Player player = sender.getServer().getPlayer(value);
066            if (player != null) {
067                this.processIPBan(player.getAddress(), sender, reason);
068
069                Command.broadcastCommandMessage(sender, new TranslationContainer("commands.banip.success.players", player.getAddress(), player.getName()));
070            } else {
071                String name = value.toLowerCase();
072                String path = sender.getServer().getDataPath() + "players/";
073                File file = new File(path + name + ".dat");
074                CompoundTag nbt = null;
075                if (file.exists()) {
076                    try {
077                        nbt = NBTIO.readCompressed(new FileInputStream(file));
078                    } catch (IOException e) {
079                        throw new RuntimeException(e);
080                    }
081                }
082
083                if (nbt != null && nbt.contains("lastIP") && Pattern.matches("^(25[0-5]|2[0-4][0-9]|[0-1]{1}[0-9]{2}|[1-9]{1}[0-9]{1}|[1-9])\\.(25[0-5]|2[0-4][0-9]|[0-1]{1}[0-9]{2}|[1-9]{1}[0-9]{1}|[1-9]|0)\\.(25[0-5]|2[0-4][0-9]|[0-1]{1}[0-9]{2}|[1-9]{1}[0-9]{1}|[1-9]|0)\\.(25[0-5]|2[0-4][0-9]|[0-1]{1}[0-9]{2}|[1-9]{1}[0-9]{1}|[0-9])$", (value = nbt.getString("lastIP")))) {
084                    this.processIPBan(value, sender, reason);
085
086                    Command.broadcastCommandMessage(sender, new TranslationContainer("commands.banip.success", value));
087                } else {
088                    sender.sendMessage(new TranslationContainer("commands.banip.invalid"));
089                    return false;
090                }
091            }
092        }
093
094        return true;
095    }
096
097    private void processIPBan(String ip, CommandSender sender, String reason) {
098        sender.getServer().getIPBans().addBan(ip, reason, null, sender.getName());
099
100        for (Player player : new ArrayList<>(sender.getServer().getOnlinePlayers().values())) {
101            if (player.getAddress().equals(ip)) {
102                player.kick(PlayerKickEvent.Reason.IP_BANNED, !reason.isEmpty() ? reason : "IP banned");
103            }
104        }
105
106        try {
107            sender.getServer().getNetwork().blockAddress(InetAddress.getByName(ip), -1);
108        } catch (UnknownHostException e) {
109            // ignore
110        }
111    }
112}
