127            return null;
128        case SOURCE_WORLD:

133            return new DropItemAction(this.oldItem, this.newItem);
134        case SOURCE_CREATIVE: 
135            int type;

137            case InventoryTransactionPacket.ACTION_MAGIC_SLOT_CREATIVE_DELETE_ITEM:

140            case InventoryTransactionPacket.ACTION_MAGIC_SLOT_CREATIVE_CREATE_ITEM: 
141                type = CreativeInventoryAction.TYPE_CREATE_ITEM;
142                break;
143            default:

145                return null;
146            }
147            return new CreativeInventoryAction(this.oldItem, this.newItem, type);
148        case SOURCE_CRAFT_SLOT: 
149        case SOURCE_TODO: 
150            //These types need special handling.
151            switch (this.windowId) {
152            case SOURCE_TYPE_CRAFTING_ADD_INGREDIENT: 
153            case SOURCE_TYPE_CRAFTING_REMOVE_INGREDIENT: 
154                return new SlotChangeAction(player.getCraftingGrid(), this.inventorySlot, this.oldItem, this.newItem);
155            case SOURCE_TYPE_CONTAINER_DROP_CONTENTS:

157                if (!inventory.isPresent()) {
158                    // No window open?

162            case SOURCE_TYPE_CRAFTING_RESULT: 
163                return new CraftingTakeResultAction(this.oldItem, this.newItem);
164            case SOURCE_TYPE_CRAFTING_USE_INGREDIENT: 
165                return new CraftingTransferMaterialAction(this.oldItem, this.newItem, this.inventorySlot);
166            }

168                //anvil actions
