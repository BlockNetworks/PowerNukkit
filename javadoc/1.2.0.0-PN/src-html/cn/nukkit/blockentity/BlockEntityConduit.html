042    @Override
043    protected void initBlockEntity() {
044        validBlocks = -1;
045        if (!namedTag.contains("Target")) {
046            namedTag.putLong("Target", -1);
047            target = -1;
048            targetEntity = null;
049        } else {
050            target = namedTag.getLong("Target");
051        }
052
053        active = namedTag.getBoolean("Active");
054
055        super.initBlockEntity();
056        this.scheduleUpdate();
057    }

060    public void saveNBT() {
061        Entity targetEntity = this.targetEntity;
062        namedTag.putLong("Target", targetEntity != null? targetEntity.getId() : -1);
063        namedTag.putBoolean("Active", active);
064        super.saveNBT();
065    }

072    @Override
073    public boolean onUpdate() {
074        if (closed) {
075            return false;
076        }
077
078        boolean activeBeforeUpdate = active;
079        Entity targetBeforeUpdate = targetEntity;
080        if (validBlocks == -1) {
081            active = scanStructure();
082        }
083
084        if (level.getCurrentTick() % 20 == 0) {
085            active = scanStructure();
086        }
087
088        if (target != -1) {
089            targetEntity = getLevel().getEntity(target);
090            target = -1;
091        }
092
093        if (activeBeforeUpdate != active || targetBeforeUpdate != targetEntity) {
094            this.spawnToAll();
095            if (activeBeforeUpdate && !active) {
096                level.addSound(add(0, 0.5, 0), Sound.CONDUIT_DEACTIVATE);
097                level.getServer().getPluginManager().callEvent(new ConduitDeactivateEvent(getBlock()));
098            } else if (!activeBeforeUpdate && active) {
099                level.addSound(add(0, 0.5, 0), Sound.CONDUIT_ACTIVATE);
100                level.getServer().getPluginManager().callEvent(new ConduitActivateEvent(getBlock()));
101            }
102        }
103
104        if (!active) {
105            targetEntity = null;
106            target = -1;
107        } else if (level.getCurrentTick() % 40 == 0) {
108            attackMob();
109            addEffectToPlayers();
110        }
111
112        return true;
113    }

137    public void addEffectToPlayers() {
138        int radius = getPlayerRadius();
139        if (radius <= 0) {
140            return;
141        }
142        final int radiusSquared = radius * radius;
143
144        Vector2 conduitPos = new Vector2(x, z);
145
146        this.getLevel().getPlayers().values().stream()
147                .filter(this::canAffect)
148                .filter(p -> conduitPos.distanceSquared(p.x, p.z) <= radiusSquared)
149                .forEach(p -> p.addEffect(Effect.getEffect(Effect.CONDUIT_POWER)
150                                .setDuration(260)
151                                .setVisible(true)
152                                .setAmplifier(0)
153                                .setAmbient(true)
154                    )
155                );
156    }

158    public void attackMob() {
159        int radius = getAttackRadius();
160        if (radius <= 0) {
161            return;
162        }
163
164        boolean updated = false;
165
166        Entity target = this.targetEntity;
167        if (target != null && !canAttack(target)) {
168            target = null;
169            updated = true;
170            this.targetEntity = null;
171            this.target = -1;
172        }
173
174        if (target == null) {
175            Entity[] mobs = Arrays.stream(level.getCollidingEntities(new SimpleAxisAlignedBB(x - radius, y - radius, z - radius, x + 1 + radius, y + 1 + radius, z + 1 + radius)))
176                    .filter(this::canAttack)
177                    .toArray(Entity[]::new);
178            if (mobs.length == 0) {
179                if (updated) {
180                    spawnToAll();
181                }
182                return;
183            }
184            target = mobs[ThreadLocalRandom.current().nextInt(mobs.length)];
185            this.targetEntity = target;
186            updated = true;
187        }
188
189        if (!target.attack(new EntityDamageByBlockEvent(getBlock(), target, EntityDamageEvent.DamageCause.MAGIC, 4))) {
190            this.targetEntity = null;
191            updated = true;
192        }
193
194        if (updated) {
195            spawnToAll();
196        }
197    }
198
199    public boolean canAttack(Entity target) {
200        return target instanceof EntityMob && canAffect(target);
201    }
202
203    public boolean canAffect(Entity target) {
204        return target.isTouchingWater()
205                || target.level.isRaining() && target.level.canBlockSeeSky(target)
206                        && !(Biome.getBiome(target.level.getBiomeId(target.getFloorX(), target.getFloorZ())) instanceof SnowyBiome);
207    }

209    private boolean scanWater() {
210        int x = getFloorX();
211        int y = getFloorY();
212        int z = getFloorZ();
213        for (int ix = -1; ix <= 1; ix++) {
214            for (int iz = -1; iz <= 1; iz++) {
215                for (int iy = -1; iy <= 1; iy++) {
216                    int blockId = this.getLevel().getBlockIdAt(x + ix, y + iy, z + iz, 0);
217                    if (blockId != Block.WATER && blockId != Block.STILL_WATER) {
218                        blockId = this.getLevel().getBlockIdAt(x + ix, y + iy, z + iz, 1);
219                        if (blockId != Block.WATER && blockId != Block.STILL_WATER) {
220                            return false;
221                        }
222                    }
223                }
224            }
225        }
226
227        return true;
228    }

230    private int scanFrame() {
231        int validBlocks = 0;
232        int x = getFloorX();
233        int y = getFloorY();
234        int z = getFloorZ();
235        for (int iy = -2; iy <= 2; iy++) {
236            if (iy == 0) {
237                for (int ix = -2; ix <= 2; ix++) {
238                    for (int iz = -2; iz <= 2; iz++) {
239                        if (Math.abs(iz) != 2 && Math.abs(ix) != 2) {
240                            continue;
241                        }
242
243                        int blockId = level.getBlockIdAt(x + ix, y, z + iz);
244                        //validBlocks++;
245                        //level.setBlock(x + ix, y, z + iz, new BlockPlanks(), true, true);
246                        if (VALID_STRUCTURE_BLOCKS.contains(blockId)) {
247                            validBlocks++;
248                        }
249                    }
250                }
251            } else {
252                int absIY = Math.abs(iy);
253                for (int ix = -2; ix <= 2; ix++) {
254                    if (absIY != 2 && ix == 0) {
255                        continue;
256                    }
257
258                    if (absIY == 2 || Math.abs(ix) == 2) {
259                        int blockId = level.getBlockIdAt(x + ix, y + iy, z);
260                        //validBlocks++;
261                        //level.setBlock(x + ix, y + iy, z, new BlockWood(), true, true);
262                        if (VALID_STRUCTURE_BLOCKS.contains(blockId)) {
263                            validBlocks++;
264                        }
265                    }
266                }
267
268                for (int iz = -2; iz <= 2; iz++) {
269                    if (absIY != 2 && iz == 0) {
270                        continue;
271                    }
272
273                    if (absIY == 2 && iz != 0 || Math.abs(iz) == 2) {
274                        int blockId = level.getBlockIdAt(x, y + iy, z + iz);
275                        //validBlocks++;
276                        //level.setBlock(x, y + iy, z + iz, new BlockWood(), true, true);
277                        if (VALID_STRUCTURE_BLOCKS.contains(blockId)) {
278                            validBlocks++;
279                        }
280                    }
281                }
282            }
283        }
284
285        return validBlocks;
286    }

288    public List<Block> scanEdgeBlock() {
289        List<Block> validBlocks = new ArrayList<>();
290        int x = getFloorX();
291        int y = getFloorY();
292        int z = getFloorZ();
293        for (int iy = -2; iy <= 2; iy++) {
294            if (iy != 0) {
295                for (int ix = -2; ix <= 2; ix++) {
296                    for (int iz = -2; iz <= 2; iz++) {
297                        if (Math.abs(iy) != 2 && Math.abs(iz) < 2 && Math.abs(ix) < 2) {
298                            continue;
299                        }
300
301                        if (ix == 0 || iz == 0) {
302                            continue;
303                        }
304
305                        Block block = level.getBlock(x + ix, y + iy, z + iz);
306                        //validBlocks++;
307                        //level.setBlock(x + ix, y + iy, z + iz, new BlockDiamond(), true, true);
308                        if (VALID_STRUCTURE_BLOCKS.contains(block.getId())) {
309                            validBlocks.add(block);
310                        }
311                    }
312                }
313            }
314        }
315
316        return validBlocks;
317    }
318
319    public boolean scanStructure() {
320        if(!scanWater()) {
321            this.validBlocks = 0;
322            return false;
323        }
324
325        int validBlocks = scanFrame();
326        if (validBlocks < 16) {
327            this.validBlocks = 0;
328            return false;
329        }
330
331        this.validBlocks = validBlocks;
332
333        return true;
334    }
335
336    public int getValidBlocks() {
337        return validBlocks;
338    }
339
340    public int getPlayerRadius() {
341        int radius = validBlocks / 7;
342        return radius * 16;
343    }
344
345    public int getAttackRadius() {
346        if (validBlocks >= 42) {
347            return 8;
348        } else {
349            return 0;
350        }
351    }

353    @Override
354    public CompoundTag getSpawnCompound() {
355        CompoundTag tag = new CompoundTag()
356                .putString("id", BlockEntity.CONDUIT)
357                .putInt("x", (int) this.x)
358                .putInt("y", (int) this.y)
359                .putInt("z", (int) this.z)
360                .putBoolean("Active", this.active)
361                .putBoolean("isMovable", isMovable());
362        Entity targetEntity = this.targetEntity;
363        tag.putLong("Target", targetEntity != null? targetEntity.getId() : -1);
364        return tag;
365    }
