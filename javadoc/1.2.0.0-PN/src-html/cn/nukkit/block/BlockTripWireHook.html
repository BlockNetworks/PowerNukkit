001package cn.nukkit.block;
002
003import cn.nukkit.Player;
004import cn.nukkit.event.block.BlockRedstoneEvent;
005import cn.nukkit.item.Item;
006import cn.nukkit.item.ItemBlock;
007import cn.nukkit.level.Level;
008import cn.nukkit.math.BlockFace;
009import cn.nukkit.math.Vector3;
010import cn.nukkit.network.protocol.LevelSoundEventPacket;
011
012/**
013 * @author CreeperFace
014 */
015public class BlockTripWireHook extends BlockTransparentMeta {
016
017    public BlockTripWireHook() {
018        this(0);
019    }
020
021    public BlockTripWireHook(int meta) {
022        super(meta);
023    }
024
025    @Override
026    public String getName() {
027        return "Tripwire Hook";
028    }
029
030    @Override
031    public int getId() {
032        return TRIPWIRE_HOOK;
033    }
034
035    public BlockFace getFacing() {
036        return BlockFace.fromHorizontalIndex(getDamage() & 0b11);
037    }
038
039    @Override
040    public int onUpdate(int type) {
041        if (type == Level.BLOCK_UPDATE_NORMAL) {
042            if (!this.getSide(this.getFacing().getOpposite()).isNormalBlock()) {
043                this.level.useBreakOn(this);
044            }
045
046            return type;
047        } else if (type == Level.BLOCK_UPDATE_SCHEDULED) {
048            this.calculateState(false, true, -1, null);
049            return type;
050        }
051
052        return 0;
053    }
054
055    @Override
056    public boolean place(Item item, Block block, Block target, BlockFace face, double fx, double fy, double fz, Player player) {
057        if (!this.getSide(face.getOpposite()).isNormalBlock() || face == BlockFace.DOWN || face == BlockFace.UP) {
058            return false;
059        }
060
061        if (face.getAxis().isHorizontal()) {
062            this.setFace(face);
063        }
064
065        this.level.setBlock(this, this);
066
067        if (player != null) {
068            this.calculateState(false, false, -1, null);
069        }
070        return true;
071    }
072
073    @Override
074    public boolean onBreak(Item item) {
075        super.onBreak(item);
076        boolean attached = isAttached();
077        boolean powered = isPowered();
078
079        if (attached || powered) {
080            this.calculateState(true, false, -1, null);
081        }
082
083        if (powered) {
084            this.level.updateAroundRedstone(this, null);
085            this.level.updateAroundRedstone(this.getLocation().getSide(getFacing().getOpposite()), null);
086        }
087
088        return true;
089    }
090
091    public void calculateState(boolean onBreak, boolean updateAround, int pos, Block block) {
092        if (!this.level.getServer().isRedstoneEnabled()) {
093            return;
094        }
095
096        BlockFace facing = getFacing();
097        Vector3 v = this.getLocation();
098        boolean attached = isAttached();
099        boolean powered = isPowered();
100        boolean canConnect = !onBreak;
101        boolean nextPowered = false;
102        int distance = 0;
103        Block[] blocks = new Block[42];
104
105        for (int i = 1; i < 42; ++i) {
106            Vector3 vector = v.getSide(facing, i);
107            Block b = this.level.getBlock(vector);
108
109            if (b instanceof BlockTripWireHook) {
110                if (((BlockTripWireHook) b).getFacing() == facing.getOpposite()) {
111                    distance = i;
112                }
113                break;
114            }
115
116            if (b.getId() != Block.TRIPWIRE && i != pos) {
117                blocks[i] = null;
118                canConnect = false;
119            } else {
120                if (i == pos) {
121                    b = block != null ? block : b;
122                }
123
124                if (b instanceof BlockTripWire) {
125                    boolean disarmed = !((BlockTripWire) b).isDisarmed();
126                    boolean wirePowered = ((BlockTripWire) b).isPowered();
127                    nextPowered |= disarmed && wirePowered;
128
129                    if (i == pos) {
130                        this.level.scheduleUpdate(this, 10);
131                        canConnect &= disarmed;
132                    }
133                }
134                blocks[i] = b;
135            }
136        }
137
138        canConnect = canConnect & distance > 1;
139        nextPowered = nextPowered & canConnect;
140        BlockTripWireHook hook = (BlockTripWireHook) Block.get(BlockID.TRIPWIRE_HOOK);
141        hook.setAttached(canConnect);
142        hook.setPowered(nextPowered);
143
144
145        if (distance > 0) {
146            Vector3 vec = v.getSide(facing, distance);
147            BlockFace face = facing.getOpposite();
148            hook.setFace(face);
149            this.level.setBlock(vec, hook, true, false);
150            this.level.updateAroundRedstone(vec, null);
151            this.level.updateAroundRedstone(vec.getSide(face.getOpposite()), null);
152            this.addSound(vec, canConnect, nextPowered, attached, powered);
153        }
154
155        this.addSound(v, canConnect, nextPowered, attached, powered);
156
157        if (!onBreak) {
158            hook.setFace(facing);
159            this.level.setBlock(v, hook, true, false);
160
161            if (updateAround) {
162                this.level.updateAroundRedstone(v, null);
163                this.level.updateAroundRedstone(v.getSide(facing.getOpposite()), null);
164            }
165        }
166
167        if (attached != canConnect) {
168            for (int i = 1; i < distance; i++) {
169                Vector3 vc = v.getSide(facing, i);
170                block = blocks[i];
171
172                if (block != null && this.level.getBlockIdAt(vc.getFloorX(), vc.getFloorY(), vc.getFloorZ()) != Block.AIR) {
173                    if (canConnect ^ ((block.getDamage() & 0x04) > 0)) {
174                        block.setDamage(block.getDamage() ^ 0x04);
175                    }
176
177                    this.level.setBlock(vc, block, true, false);
178                }
179            }
180        }
181    }
182
183    private void addSound(Vector3 pos, boolean canConnect, boolean nextPowered, boolean attached, boolean powered) {
184        if (nextPowered && !powered) {
185            this.level.addLevelSoundEvent(pos, LevelSoundEventPacket.SOUND_POWER_ON);
186            this.level.getServer().getPluginManager().callEvent(new BlockRedstoneEvent(this, 0, 15));
187        } else if (!nextPowered && powered) {
188            this.level.addLevelSoundEvent(pos, LevelSoundEventPacket.SOUND_POWER_OFF);
189            this.level.getServer().getPluginManager().callEvent(new BlockRedstoneEvent(this, 15, 0));
190        } else if (canConnect && !attached) {
191            this.level.addLevelSoundEvent(pos, LevelSoundEventPacket.SOUND_ATTACH);
192        } else if (!canConnect && attached) {
193            this.level.addLevelSoundEvent(pos, LevelSoundEventPacket.SOUND_DETACH);
194        }
195    }
196
197    public boolean isAttached() {
198        return (getDamage() & 0x04) > 0;
199    }
200
201    public boolean isPowered() {
202        return (this.getDamage() & 0x08) > 0;
203    }
204
205    public void setPowered(boolean value) {
206        if (value ^ this.isPowered()) {
207            this.setDamage(this.getDamage() ^ 0x08);
208        }
209    }
210
211    public void setAttached(boolean value) {
212        if (value ^ this.isAttached()) {
213            this.setDamage(this.getDamage() ^ 0x04);
214        }
215    }
216
217    public void setFace(BlockFace face) {
218        this.setDamage(this.getDamage() - this.getDamage() % 4);
219        this.setDamage(this.getDamage() | face.getHorizontalIndex());
220    }
221
222    @Override
223    public boolean isPowerSource() {
224        return true;
225    }
226
227    @Override
228    public int getWeakPower(BlockFace face) {
229        return isPowered() ? 15 : 0;
230    }
231
232    @Override
233    public int getStrongPower(BlockFace side) {
234        return !isPowered() ? 0 : getFacing() == side ? 15 : 0;
235    }
236
237    @Override
238    public int getWaterloggingLevel() {
239        return 2;
240    }
241
242    @Override
243    public boolean canBeFlowedInto() {
244        return false;
245    }
246
247    @Override
248    public Item toItem() {
249        return new ItemBlock(this, 0);
250    }
251}
