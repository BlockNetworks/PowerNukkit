018    public BlockRailPowered() {
019        this(0);

024        super(meta);

027    @Override
028    public int getId() {
029        return POWERED_RAIL;

033    public String getName() {
034        return "Powered Rail";

037    @Override
038    public int onUpdate(int type) {
039        // Warning: I din't recommended this on slow networks server or slow client
040        //          Network below 86Kb/s. This will became unresponsive to clients 
041        //          When updating the block state. Espicially on the world with many rails. 
042        //          Trust me, I tested this on my server.
043        if (type == Level.BLOCK_UPDATE_NORMAL || type == Level.BLOCK_UPDATE_REDSTONE || type == Level.BLOCK_UPDATE_SCHEDULED) {
044            if (super.onUpdate(type) == Level.BLOCK_UPDATE_NORMAL) {
045                return 0; // Already broken
046            }
047
048            if (!this.level.getServer().isRedstoneEnabled()) {
049                return 0;
050            }
051            boolean wasPowered = isActive();
052            boolean isPowered = level.isBlockPowered(this.getLocation())
053                    || checkSurrounding(this, true, 0)
054                    || checkSurrounding(this, false, 0);
055
056            // Avoid Block minstake
057            if (wasPowered != isPowered) {
058                setActive(isPowered);
059                level.updateAround(down());
060                if (getOrientation().isAscending()) {
061                    level.updateAround(up());
062                }
063            }
064            return type;
065        }
066        return 0;
067    }
068
069    /**
070     * Check the surrounding of the rail
071     *
072     * @param pos      The rail position
073     * @param relative The relative of the rail that will be checked
074     * @param power    The count of the rail that had been counted
075     * @return Boolean of the surrounding area. Where the powered rail on!
076     */
077    private boolean checkSurrounding(Vector3 pos, boolean relative, int power) {
078        // The powered rail can power up to 8 blocks only
079        if (power >= 8) {
080            return false;
081        }
082        // The position of the floor numbers
083        int dx = pos.getFloorX();
084        int dy = pos.getFloorY();
085        int dz = pos.getFloorZ();
086        // First: get the base block
087        BlockRail block;
088        Block block2 = level.getBlock(new Vector3(dx, dy, dz));
089
090        // Second: check if the rail is Powered rail
091        if (Rail.isRailBlock(block2)) {
092            block = (BlockRail) block2;
093        } else {
094            return false;
095        }
096
097        // Used to check if the next ascending rail should be what
098        Rail.Orientation base = null;
099        boolean onStraight = true;
100        // Third: Recalculate the base position
101        switch (block.getOrientation()) {
102            case STRAIGHT_NORTH_SOUTH:
103                if (relative) {
104                    dz++;
105                } else {
106                    dz--;
107                }
108                break;
109            case STRAIGHT_EAST_WEST:
110                if (relative) {
111                    dx--;
112                } else {
113                    dx++;
114                }
115                break;
116            case ASCENDING_EAST:
117                if (relative) {
118                    dx--;
119                } else {
120                    dx++;
121                    dy++;
122                    onStraight = false;
123                }
124                base = Rail.Orientation.STRAIGHT_EAST_WEST;
125                break;
126            case ASCENDING_WEST:
127                if (relative) {
128                    dx--;
129                    dy++;
130                    onStraight = false;
131                } else {
132                    dx++;
133                }
134                base = Rail.Orientation.STRAIGHT_EAST_WEST;
135                break;
136            case ASCENDING_NORTH:
137                if (relative) {
138                    dz++;
139                } else {
140                    dz--;
141                    dy++;
142                    onStraight = false;
143                }
144                base = Rail.Orientation.STRAIGHT_NORTH_SOUTH;
145                break;
146            case ASCENDING_SOUTH:
147                if (relative) {
148                    dz++;
149                    dy++;
150                    onStraight = false;
151                } else {
152                    dz--;
153                }
154                base = Rail.Orientation.STRAIGHT_NORTH_SOUTH;
155                break;
156            default:
157                // Unable to determinate the rail orientation
158                // Wrong rail?
159                return false;
160        }
161        // Next check the if rail is on power state
162        return canPowered(new Vector3(dx, dy, dz), base, power, relative)
163                || onStraight && canPowered(new Vector3(dx, dy - 1, dz), base, power, relative);
164    }
165
166    protected boolean canPowered(Vector3 pos, Rail.Orientation state, int power, boolean relative) {
167        Block block = level.getBlock(pos);
168        // What! My block is air??!! Impossible! XD
169        if (!(block instanceof BlockRailPowered)) {
170            return false;
171        }
172
173        // Sometimes the rails are diffrent orientation
174        Rail.Orientation base = ((BlockRailPowered) block).getOrientation();
175
176        // Possible way how to know when the rail is activated is rail were directly powered
177        // OR recheck the surrounding... Which will returns here =w=        
178        return (state != Rail.Orientation.STRAIGHT_EAST_WEST
179                || base != Rail.Orientation.STRAIGHT_NORTH_SOUTH
180                && base != Rail.Orientation.ASCENDING_NORTH
181                && base != Rail.Orientation.ASCENDING_SOUTH)
182                && (state != Rail.Orientation.STRAIGHT_NORTH_SOUTH
183                || base != Rail.Orientation.STRAIGHT_EAST_WEST
184                && base != Rail.Orientation.ASCENDING_EAST
185                && base != Rail.Orientation.ASCENDING_WEST)
186                && (level.isBlockPowered(pos) || checkSurrounding(pos, relative, power + 1));
187    }
188
189    @Override
190    public Item[] getDrops(Item item) {
191        return new Item[]{
192                Item.get(Item.POWERED_RAIL, 0, 1)
193        };
194    }
