001package cn.nukkit.block;
002
003import cn.nukkit.Player;
004import cn.nukkit.item.Item;
005import cn.nukkit.item.ItemBlock;
006import cn.nukkit.item.ItemTool;
007import cn.nukkit.level.Level;
008import cn.nukkit.math.AxisAlignedBB;
009import cn.nukkit.math.BlockFace;
010import cn.nukkit.utils.BlockColor;
011
012public class BlockLantern extends BlockFlowable {
013
014    public BlockLantern() {
015        this(0);
016    }
017
018    public BlockLantern(int meta) {
019        super(meta);
020    }
021
022    @Override
023    public int getId() {
024        return LANTERN;
025    }
026
027    @Override
028    public String getName() {
029        return "Lantern";
030    }
031
032    private boolean isBlockAboveValid() {
033        Block up = up();
034        if (up instanceof BlockLeaves) {
035            return false;
036        } else if (up instanceof BlockFence) {
037            return true;
038        } else if (up instanceof BlockSlab) {
039            return (up.getDamage() & 0x08) == 0x00;
040        } else if (up instanceof BlockStairs) {
041            return (up.getDamage() & 0x04) == 0x00;
042        } else if (up.isSolid()) {
043            return true;
044        } else {
045            return false;
046        }
047    }
048
049    private boolean isBlockUnderValid() {
050        Block down = down();
051        if (down instanceof BlockLeaves) {
052            return false;
053        } else if (down instanceof BlockFence) {
054            return true;
055        } else if (down instanceof BlockSlab) {
056            return (down.getDamage() & 0x08) == 0x08;
057        } else if (down instanceof BlockStairs) {
058            return (down.getDamage() & 0x04) == 0x04;
059        } else if (down.isSolid()) {
060            return true;
061        } else {
062            return false;
063        }
064    }
065
066    @Override
067    public boolean place(Item item, Block block, Block target, BlockFace face, double fx, double fy, double fz, Player player) {
068        if(this.getLevelBlock() instanceof BlockLiquid || this.getLevelBlockAtLayer(1) instanceof BlockLiquid) {
069            return false;
070        }
071
072        boolean hanging = false;
073        if (face == BlockFace.DOWN) {
074            if (isBlockAboveValid()) {
075                hanging = true;
076            } else if (!isBlockUnderValid()) {
077                return false;
078            }
079        } else if (face == BlockFace.UP && isBlockUnderValid()) {
080            hanging = false;
081        } else if (isBlockAboveValid()) {
082            hanging = true;
083        } else {
084            return false;
085        }
086
087        if (hanging) {
088            setDamage(1);
089        } else {
090            setDamage(0);
091        }
092
093        this.getLevel().setBlock(this, this, true, true);
094        return true;
095    }
096
097    @Override
098    public int onUpdate(int type) {
099        if (type == Level.BLOCK_UPDATE_NORMAL) {
100            if (getDamage() == 0) {
101                if (!isBlockUnderValid()) {
102                    level.useBreakOn(this);
103                }
104            } else if (!isBlockAboveValid()) {
105                level.useBreakOn(this);
106            }
107            return type;
108        }
109        return 0;
110    }
111
112    @Override
113    public int getLightLevel() {
114        return 15;
115    }
116
117    @Override
118    public double getResistance() {
119        return 17.5;
120    }
121
122    @Override
123    public double getHardness() {
124        return 5.0;
125    }
126
127    @Override
128    public boolean canHarvestWithHand() {
129        return false;
130    }
131
132    @Override
133    public int getToolType() {
134        return ItemTool.TYPE_PICKAXE;
135    }
136
137    @Override
138    public double getMinX() {
139        return x + (5.0/16);
140    }
141
142    @Override
143    public double getMinY() {
144        return y + (getDamage()==0?0: 1./16);
145    }
146
147    @Override
148    public double getMinZ() {
149        return z + (5.0/16);
150    }
151
152    @Override
153    public double getMaxX() {
154        return x + (11.0/16);
155    }
156
157    @Override
158    public double getMaxY() {
159        return y + (getDamage()==0? 7.0/16 : 8.0/16);
160    }
161
162    @Override
163    public double getMaxZ() {
164        return z + (11.0/16);
165    }
166
167    @Override
168    public Item toItem() {
169        return new ItemBlock(new BlockLantern());
170    }
171
172    @Override
173    public boolean canPassThrough() {
174        return false;
175    }
176
177    @Override
178    protected AxisAlignedBB recalculateBoundingBox() {
179        return this;
180    }
181
182    @Override
183    public BlockColor getColor() {
184        return BlockColor.IRON_BLOCK_COLOR;
185    }
186
187}
