001package cn.nukkit.inventory;
002
003import cn.nukkit.item.Item;
004
005import java.util.ArrayList;
006import java.util.Collection;
007import java.util.List;
008
009public class RepairRecipe implements Recipe {
010    
011    private final Item result;
012    private final List<Item> ingredients;
013    private final InventoryType inventoryType;
014    
015    public RepairRecipe(InventoryType inventoryType, Item result, Collection<Item> ingredients) {
016        this.inventoryType = inventoryType;
017        this.result = result.clone();
018        this.ingredients = new ArrayList<>();
019    
020        for (Item item : ingredients) {
021            if (item.getCount() < 1) {
022                throw new IllegalArgumentException("Recipe Ingredient amount was not 1 (value: " + item.getCount() + ")");
023            }
024            this.ingredients.add(item.clone());
025        }
026    }
027    
028    @Override
029    public Item getResult() {
030        return result.clone();
031    }
032    
033    public List<Item> getIngredientList() {
034        List<Item> ingredients = new ArrayList<>();
035        for (Item ingredient : this.ingredients) {
036            ingredients.add(ingredient.clone());
037        }
038        
039        return ingredients;
040    }
041    
042    @Override
043    public void registerToCraftingManager(CraftingManager manager) {
044    
045    }
046    
047    @Override
048    public RecipeType getType() {
049        return RecipeType.REPAIR;
050    }
051    
052    public InventoryType getInventoryType() {
053        return inventoryType;
054    }
055}
