003import cn.nukkit.Player;
004import cn.nukkit.event.inventory.InventoryClickEvent;
005import cn.nukkit.event.inventory.InventoryTransactionEvent;
006import cn.nukkit.inventory.Inventory;

013import java.util.*;
014import java.util.Map.Entry;

016/**
017 * @author CreeperFace
018 */
019public class InventoryTransaction {
020
021    private long creationTime;
022    protected boolean hasExecuted;
023
024    protected Player source;
025
026    protected Set<Inventory> inventories = new HashSet<>();
027
028    protected Set<InventoryAction> actions = new LinkedHashSet<>();
029
030    public InventoryTransaction(Player source, List<InventoryAction> actions) {
031        this(source, actions, true);
032    }
033
034    public InventoryTransaction(Player source, List<InventoryAction> actions, boolean init) {
035        if (init) {
036            init(source, actions);
037        }
038    }
039
040    protected void init(Player source, List<InventoryAction> actions) {
041        creationTime = System.currentTimeMillis();
042        this.source = source;
043
044        for (InventoryAction action : actions) {
045            this.addAction(action);
046        }
047    }
048
049    public Player getSource() {
050        return source;
051    }
052
053    public long getCreationTime() {
054        return creationTime;
055    }
056
057    public Set<Inventory> getInventories() {
058        return inventories;
059    }
060
061    public Set<InventoryAction> getActions() {
062        return actions;
063    }
064
065    public void addAction(InventoryAction action) {
066        if (!this.actions.contains(action)) {
067            this.actions.add(action);
068            action.onAddToTransaction(this);
069        } else {
070            throw new RuntimeException("Tried to add the same action to a transaction twice");
071        }
072    }
073
074    /**
075     * This method should not be used by plugins, it's used to add tracked inventories for InventoryActions
076     * involving inventories.
077     *
078     * @param inventory to add
079     */
080    public void addInventory(Inventory inventory) {
081        this.inventories.add(inventory);
082    }
083
084    protected boolean matchItems(List<Item> needItems, List<Item> haveItems) {
085        for (InventoryAction action : this.actions) {
086            if (action.getTargetItem().getId() != Item.AIR) {
087                needItems.add(action.getTargetItem());
088            }
089
090            if (!action.isValid(this.source)) {
091                return false;
092            }
093
094            if (action.getSourceItem().getId() != Item.AIR) {
095                haveItems.add(action.getSourceItem());
096            }
097        }
098
099        for (Item needItem : new ArrayList<>(needItems)) {
100            for (Item haveItem : new ArrayList<>(haveItems)) {
101                if (needItem.equals(haveItem)) {
102                    int amount = Math.min(haveItem.getCount(), needItem.getCount());
103                    needItem.setCount(needItem.getCount() - amount);
104                    haveItem.setCount(haveItem.getCount() - amount);
105                    if (haveItem.getCount() == 0) {
106                        haveItems.remove(haveItem);
107                    }
108                    if (needItem.getCount() == 0) {
109                        needItems.remove(needItem);
110                        break;
111                    }
112                }
113            }
114        }
115
116        return haveItems.isEmpty() && needItems.isEmpty();
117    }
118
119    protected void sendInventories() {
120        for (Inventory inventory : this.inventories) {
121            inventory.sendContents(this.source);
122            if (inventory instanceof PlayerInventory) {
123                ((PlayerInventory) inventory).sendArmorContents(this.source);
124            }
125        }
126    }
127
128    /**
129     * Iterates over SlotChangeActions in this transaction and compacts any which refer to the same inventorySlot in the same
130     * inventory so they can be correctly handled.
131     * <p>
132     * Under normal circumstances, the same inventorySlot would never be changed more than once in a single transaction. However,
133     * due to the way things like the crafting grid are "implemented" in MCPE 1.2 (a.k.a. hacked-in), we may get
134     * multiple inventorySlot changes referring to the same inventorySlot in a single transaction. These multiples are not even guaranteed
135     * to be in the correct order (inventorySlot splitting in the crafting grid for example, causes the actions to be sent in the
136     * wrong order), so this method also tries to chain them into order.
137     * </p>
138     *
139     * @return successful
140     */
141    protected boolean squashDuplicateSlotChanges() {
142        Map<Integer, List<SlotChangeAction>> slotChanges = new HashMap<>();
143
144        for (InventoryAction action : this.actions) {
145            if (action instanceof SlotChangeAction) {
146                int hash = Objects.hash(((SlotChangeAction) action).getInventory(), ((SlotChangeAction) action).getSlot());
147
148                List<SlotChangeAction> list = slotChanges.get(hash);
149                if (list == null) {
150                    list = new ArrayList<>();
151                }
152
153                list.add((SlotChangeAction) action);
154
155                slotChanges.put(hash, list);
156            }
157        }
158
159        for (Entry<Integer, List<SlotChangeAction>> entry : new ArrayList<>(slotChanges.entrySet())) {
160            int hash = entry.getKey();
161            List<SlotChangeAction> list = entry.getValue();
162
163            if (list.size() == 1) { //No need to compact inventorySlot changes if there is only one on this inventorySlot
164                slotChanges.remove(hash);
165                continue;
166            }
167
168            List<SlotChangeAction> originalList = new ArrayList<>(list);
169
170            SlotChangeAction originalAction = null;
171            Item lastTargetItem = null;
172
173            for (int i = 0; i < list.size(); i++) {
174                SlotChangeAction action = list.get(i);
175
176                if (action.isValid(this.source)) {
177                    originalAction = action;
178                    lastTargetItem = action.getTargetItem();
179                    list.remove(i);
180                    break;
181                }
182            }
183
184            if (originalAction == null) {
185                return false; //Couldn't find any actions that had a source-item matching the current inventory inventorySlot
186            }
187
188            int sortedThisLoop;
189
190            do {
191                sortedThisLoop = 0;
192                for (int i = 0; i < list.size(); i++) {
193                    SlotChangeAction action = list.get(i);
194
195                    Item actionSource = action.getSourceItem();
196                    if (actionSource.equalsExact(lastTargetItem)) {
197                        lastTargetItem = action.getTargetItem();
198                        list.remove(i);
199                        sortedThisLoop++;
200                    }
201                    else if (actionSource.equals(lastTargetItem)) {
202                        lastTargetItem.count -= actionSource.count;
203                        list.remove(i);
204                        if (lastTargetItem.count == 0) sortedThisLoop++;
205                    }
206                }
207            } while (sortedThisLoop > 0);
208
209            if (list.size() > 0) { //couldn't chain all the actions together
210                MainLogger.getLogger().debug("Failed to compact " + originalList.size() + " actions for " + this.source.getName());
211                return false;
212            }
213
214            for (SlotChangeAction action : originalList) {
215                this.actions.remove(action);
216            }
217
218            this.addAction(new SlotChangeAction(originalAction.getInventory(), originalAction.getSlot(), originalAction.getSourceItem(), lastTargetItem));
219
220            MainLogger.getLogger().debug("Successfully compacted " + originalList.size() + " actions for " + this.source.getName());
221        }
222
223        return true;
224    }
225
226    public boolean canExecute() {
227        this.squashDuplicateSlotChanges();
228
229        List<Item> haveItems = new ArrayList<>();
230        List<Item> needItems = new ArrayList<>();
231        return matchItems(needItems, haveItems) && this.actions.size() > 0 && haveItems.size() == 0 && needItems.size() == 0;
232    }
233
234    protected boolean callExecuteEvent() {
235        InventoryTransactionEvent ev = new InventoryTransactionEvent(this);
236        this.source.getServer().getPluginManager().callEvent(ev);
237
238        SlotChangeAction from = null;
239        SlotChangeAction to = null;
240        Player who = null;
241
242        for (InventoryAction action : this.actions) {
243            if (!(action instanceof SlotChangeAction)) {
244                continue;
245            }
246            SlotChangeAction slotChange = (SlotChangeAction) action;
247
248            if (slotChange.getInventory() instanceof PlayerInventory) {
249                who = (Player) slotChange.getInventory().getHolder();
250            }
251
252            if (from == null) {
253                from = slotChange;
254            } else {
255                to = slotChange;
256            }
257        }
258
259        if (who != null && to != null) {
260            if (from.getTargetItem().getCount() > from.getSourceItem().getCount()) {
261                from = to;
262            }
263
264            InventoryClickEvent ev2 = new InventoryClickEvent(who, from.getInventory(), from.getSlot(), from.getSourceItem(), from.getTargetItem());
265            this.source.getServer().getPluginManager().callEvent(ev2);
266
267            if (ev2.isCancelled()) {
268                return false;
269            }
270        }
271
272        return !ev.isCancelled();
273    }
274
275    public boolean execute() {
276        if (this.hasExecuted() || !this.canExecute()) {
277            this.sendInventories();
278            return false;
279        }
280
281
282        if (!callExecuteEvent()) {
283            this.sendInventories();
284            return true;
285        }
286
287        for (InventoryAction action : this.actions) {
288            if (!action.onPreExecute(this.source)) {
289                this.sendInventories();
290                return true;
291            }
292        }
293
294        for (InventoryAction action : this.actions) {
295            if (action.execute(this.source)) {
296                action.onExecuteSuccess(this.source);
297            } else {
298                action.onExecuteFail(this.source);
299            }
300        }
301
302        this.hasExecuted = true;
303        return true;
304    }
305
306    public boolean hasExecuted() {
307        return this.hasExecuted;
308    }
