033public class Explosion {
034
035    private final int rays = 16; //Rays
036    private final Level level;
037    private final Position source;
038    private final double size;
039
040    private List<Block> affectedBlocks = new ArrayList<>();
041    private final double stepLen = 0.3d;
042
043    private final Object what;
044
045    public Explosion(Position center, double size, Entity what) {
046        this.level = center.getLevel();
047        this.source = center;
048        this.size = Math.max(size, 0);
049        this.what = what;
050    }
051
052    /**
053     * @return bool
054     * @deprecated
055     */
056    public boolean explode() {
057        if (explodeA()) {
058            return explodeB();
059        }
060        return false;
061    }
062
063    /**
064     * @return bool
065     */
066    public boolean explodeA() {
067        if (this.size < 0.1) {
068            return false;
069        }
070
071        Vector3 vector = new Vector3(0, 0, 0);
072        Vector3 vBlock = new Vector3(0, 0, 0);
073
074        int mRays = this.rays - 1;
075        for (int i = 0; i < this.rays; ++i) {
076            for (int j = 0; j < this.rays; ++j) {
077                for (int k = 0; k < this.rays; ++k) {
078                    if (i == 0 || i == mRays || j == 0 || j == mRays || k == 0 || k == mRays) {
079                        vector.setComponents((double) i / (double) mRays * 2d - 1, (double) j / (double) mRays * 2d - 1, (double) k / (double) mRays * 2d - 1);
080                        double len = vector.length();
081                        vector.setComponents((vector.x / len) * this.stepLen, (vector.y / len) * this.stepLen, (vector.z / len) * this.stepLen);
082                        double pointerX = this.source.x;
083                        double pointerY = this.source.y;
084                        double pointerZ = this.source.z;
085
086                        for (double blastForce = this.size * (ThreadLocalRandom.current().nextInt(700, 1301)) / 1000d; blastForce > 0; blastForce -= this.stepLen * 0.75d) {
087                            int x = (int) pointerX;
088                            int y = (int) pointerY;
089                            int z = (int) pointerZ;
090                            vBlock.x = pointerX >= x ? x : x - 1;
091                            vBlock.y = pointerY >= y ? y : y - 1;
092                            vBlock.z = pointerZ >= z ? z : z - 1;
093                            if (vBlock.y < 0 || vBlock.y > 255) {
094                                break;
095                            }
096                            Block block = this.level.getBlock(vBlock);
097
098                            if (block.getId() != 0) {
099                                blastForce -= (block.getResistance() / 5 + 0.3d) * this.stepLen;
100                                if (blastForce > 0) {
101                                    if (!this.affectedBlocks.contains(block)) {
102                                        this.affectedBlocks.add(block);
103                                    }
104                                }
105                            }
106                            pointerX += vector.x;
107                            pointerY += vector.y;
108                            pointerZ += vector.z;
109                        }
110                    }
111                }
112            }
113        }
114
115        return true;
116    }

118    public boolean explodeB() {
119
120        LongArraySet updateBlocks = new LongArraySet();
121        List<Vector3> send = new ArrayList<>();
122
123        Vector3 source = (new Vector3(this.source.x, this.source.y, this.source.z)).floor();
124        double yield = (1d / this.size) * 100d;
125
126        if (this.what instanceof Entity) {
127            EntityExplodeEvent ev = new EntityExplodeEvent((Entity) this.what, this.source, this.affectedBlocks, yield);
128            this.level.getServer().getPluginManager().callEvent(ev);
129            if (ev.isCancelled()) {
130                return false;
131            } else {
132                yield = ev.getYield();
133                this.affectedBlocks = ev.getBlockList();
134            }
135        }
136
137        double explosionSize = this.size * 2d;
138        double minX = NukkitMath.floorDouble(this.source.x - explosionSize - 1);
139        double maxX = NukkitMath.ceilDouble(this.source.x + explosionSize + 1);
140        double minY = NukkitMath.floorDouble(this.source.y - explosionSize - 1);
141        double maxY = NukkitMath.ceilDouble(this.source.y + explosionSize + 1);
142        double minZ = NukkitMath.floorDouble(this.source.z - explosionSize - 1);
143        double maxZ = NukkitMath.ceilDouble(this.source.z + explosionSize + 1);
144
145        AxisAlignedBB explosionBB = new SimpleAxisAlignedBB(minX, minY, minZ, maxX, maxY, maxZ);
146
147        Entity[] list = this.level.getNearbyEntities(explosionBB, this.what instanceof Entity ? (Entity) this.what : null);
148        for (Entity entity : list) {
149            double distance = entity.distance(this.source) / explosionSize;
150
151            if (distance <= 1) {
152                Vector3 motion = entity.subtract(this.source).normalize();
153                int exposure = 1;
154                double impact = (1 - distance) * exposure;
155                int damage = (int) (((impact * impact + impact) / 2) * 8 * explosionSize + 1);
156
157                if (this.what instanceof Entity) {
158                    entity.attack(new EntityDamageByEntityEvent((Entity) this.what, entity, DamageCause.ENTITY_EXPLOSION, damage));
159                } else if (this.what instanceof Block) {
160                    entity.attack(new EntityDamageByBlockEvent((Block) this.what, entity, DamageCause.BLOCK_EXPLOSION, damage));
161                } else {
162                    entity.attack(new EntityDamageEvent(entity, DamageCause.BLOCK_EXPLOSION, damage));
163                }
164
165                if (!(entity instanceof EntityItem || entity instanceof EntityXPOrb)) {
166                    entity.setMotion(motion.multiply(impact));
167                }
168            }
169        }
170
171        ItemBlock air = new ItemBlock(Block.get(BlockID.AIR));
172
173        //Iterator iter = this.affectedBlocks.entrySet().iterator();
174        for (Block block : this.affectedBlocks) {
175            //Block block = (Block) ((HashMap.Entry) iter.next()).getValue();
176            if (block.getId() == Block.TNT) {
177                ((BlockTNT) block).prime(new NukkitRandom().nextRange(10, 30), this.what instanceof Entity ? (Entity) this.what : null);
178            } else if (block.getId() == Block.CHEST || block.getId() == Block.TRAPPED_CHEST) {
179                BlockEntity chest = block.getLevel().getBlockEntity(block);
180                if (chest != null) {
181                    for (Item drop : ((BlockEntityChest) chest).getInventory().getContents().values()) {
182                        this.level.dropItem(block.add(0.5, 0.5, 0.5), drop);
183                    }
184                    ((BlockEntityChest) chest).getInventory().clearAll();
185                }
186            } else if (Math.random() * 100 < yield) {
187                for (Item drop : block.getDrops(air)) {
188                    this.level.dropItem(block.add(0.5, 0.5, 0.5), drop);
189                }
190            }
191
192            this.level.setBlockAt((int) block.x, (int) block.y, (int) block.z, BlockID.AIR);
193
194            Vector3 pos = new Vector3(block.x, block.y, block.z);
195
196            for (BlockFace side : BlockFace.values()) {
197                Vector3 sideBlock = pos.getSide(side);
198                long index = Hash.hashBlock((int) sideBlock.x, (int) sideBlock.y, (int) sideBlock.z);
199                if (!this.affectedBlocks.contains(sideBlock) && !updateBlocks.contains(index)) {
200                    BlockUpdateEvent ev = new BlockUpdateEvent(this.level.getBlock(sideBlock));
201                    this.level.getServer().getPluginManager().callEvent(ev);
202                    if (!ev.isCancelled()) {
203                        ev.getBlock().onUpdate(Level.BLOCK_UPDATE_NORMAL);
204                    }
205                    updateBlocks.add(index);
206                }
207            }
208            send.add(new Vector3(block.x - source.x, block.y - source.y, block.z - source.z));
209        }
210
211        this.level.addParticle(new HugeExplodeSeedParticle(this.source));
212        this.level.addLevelSoundEvent(source, LevelSoundEventPacket.SOUND_EXPLODE);
213
214        return true;
215    }
