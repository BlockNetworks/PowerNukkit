064    public Flat() {

068    public Flat(Map<String, Object> options) {
069        this.preset = "2;7,2x3,2;1;";
070        this.options = options;
071
072        if (this.options.containsKey("decoration")) {
073            PopulatorOre ores = new PopulatorOre(BlockID.STONE, new OreType[]{
074                    new OreType(Block.get(BlockID.COAL_ORE), 20, 16, 0, 128),
075                    new OreType(Block.get(BlockID.IRON_ORE), 20, 8, 0, 64),
076                    new OreType(Block.get(BlockID.REDSTONE_ORE), 8, 7, 0, 16),
077                    new OreType(Block.get(BlockID.LAPIS_ORE), 1, 6, 0, 32),
078                    new OreType(Block.get(BlockID.GOLD_ORE), 2, 8, 0, 32),
079                    new OreType(Block.get(BlockID.DIAMOND_ORE), 1, 7, 0, 16),
080                    new OreType(Block.get(BlockID.DIRT), 20, 32, 0, 128),
081                    new OreType(Block.get(BlockID.GRAVEL), 20, 16, 0, 128),
082            });
083            this.populators.add(ores);
084        }
085    }
086
087    protected void parsePreset(String preset, int chunkX, int chunkZ) {
088        try {
089            this.preset = preset;
090            String[] presetArray = preset.split(";");
091            int version = Integer.valueOf(presetArray[0]);
092            String blocks = presetArray.length > 1 ? presetArray[1] : "";
093            this.biome = presetArray.length > 2 ? Integer.valueOf(presetArray[2]) : 1;
094            String options = presetArray.length > 3 ? presetArray[1] : "";
095            this.structure = new int[256][];
096            int y = 0;
097            for (String block : blocks.split(",")) {
098                int id, meta = 0, cnt = 1;
099                if (Pattern.matches("^[0-9]{1,3}x[0-9]$", block)) {
100                    //AxB
101                    String[] s = block.split("x");
102                    cnt = Integer.valueOf(s[0]);
103                    id = Integer.valueOf(s[1]);
104                } else if (Pattern.matches("^[0-9]{1,3}:[0-9]{0,2}$", block)) {
105                    //A:B
106                    String[] s = block.split(":");
107                    id = Integer.valueOf(s[0]);
108                    meta = Integer.valueOf(s[1]);
109                } else if (Pattern.matches("^[0-9]{1,3}$", block)) {
110                    //A
111                    id = Integer.valueOf(block);
112                } else {
113                    continue;
114                }
115                int cY = y;
116                y += cnt;
117                if (y > 0xFF) {
118                    y = 0xFF;
119                }
120                for (; cY < y; ++cY) {
121                    this.structure[cY] = new int[]{id, meta};
122                }
123            }
124            this.floorLevel = y;
125            for (; y <= 0xFF; ++y) {
126                this.structure[y] = new int[]{0, 0};
127            }
128            for (String option : options.split(",")) {
129                if (Pattern.matches("^[0-9a-z_]+$", option)) {
130                    this.options.put(option, true);
131                } else if (Pattern.matches("^[0-9a-z_]+\\([0-9a-z_ =]+\\)$", option)) {
132                    String name = option.substring(0, option.indexOf("("));
133                    String extra = option.substring(option.indexOf("(") + 1, option.indexOf(")"));
134                    Map<String, Float> map = new HashMap<>();
135                    for (String kv : extra.split(" ")) {
136                        String[] data = kv.split("=");
137                        map.put(data[0], Float.valueOf(data[1]));
138                    }
139                    this.options.put(name, map);
140                }
141            }
142        } catch (Exception e) {
143            Server.getInstance().getLogger().error("error while parsing the preset", e);
144            throw new RuntimeException(e);
145        }
146    }
147
148    @Override
149    public void init(ChunkManager level, NukkitRandom random) {
150        this.level = level;
151        this.random = random;
152    }
153
154    @Override
155    public void generateChunk(int chunkX, int chunkZ) {
156        if (!this.init) {
157            init = true;
158            if (this.options.containsKey("preset") && !"".equals(this.options.get("preset"))) {
159                this.parsePreset((String) this.options.get("preset"), chunkX, chunkZ);
160            } else {
161                this.parsePreset(this.preset, chunkX, chunkZ);
162            }
163        }
164        this.generateChunk(level.getChunk(chunkX, chunkZ));
165    }
166
167    private void generateChunk(FullChunk chunk) {
168        chunk.setGenerated();
169
170        for (int Z = 0; Z < 16; ++Z) {
171            for (int X = 0; X < 16; ++X) {
172                chunk.setBiomeId(X, Z, biome);
173
174                for (int y = 0; y < 256; ++y) {
175                    int k = this.structure[y][0];
176                    int l = this.structure[y][1];
177                    chunk.setBlock(X, y, Z, this.structure[y][0], this.structure[y][1]);
178                }
179            }
180        }
181    }
182
183    @Override
184    public void populateChunk(int chunkX, int chunkZ) {
185        BaseFullChunk chunk = level.getChunk(chunkX, chunkZ);
186        this.random.setSeed(0xdeadbeef ^ (chunkX << 8) ^ chunkZ ^ this.level.getSeed());
187        for (Populator populator : this.populators) {
188            populator.populate(this.level, chunkX, chunkZ, this.random, chunk);
189        }
190    }
191
192    @Override
193    public Vector3 getSpawn() {
194        return new Vector3(128, this.floorLevel, 128);
195    }
