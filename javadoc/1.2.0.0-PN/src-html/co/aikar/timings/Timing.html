024package co.aikar.timings;
025
026import java.util.HashMap;
027import java.util.Map;
028
029public class Timing implements AutoCloseable {
030    private static int idPool = 1;
031    final int id = idPool++;
032
033    final String name;
034    private final boolean verbose;
035
036    final Map<Integer, TimingData> children = new HashMap<>();
037    private Timing parent;
038
039    private final Timing groupTiming;
040    final TimingData record;
041
042    private long start = 0;
043    private int timingDepth = 0;
044    private boolean added;
045    boolean timed;
046    boolean enabled;
047
048    Timing(TimingIdentifier id) {
049        if (id.name.startsWith("##")) {
050            this.verbose = true;
051            this.name = id.name.substring(3);
052        } else {
053            this.name = id.name;
054            this.verbose = false;
055        }
056
057        this.record = new TimingData(this.id);
058        this.groupTiming = id.groupTiming;
059
060        TimingIdentifier.getGroup(id.group).timings.add(this);
061        this.checkEnabled();
062    }
063
064    final void checkEnabled() {
065        this.enabled = Timings.isTimingsEnabled() && (!this.verbose || Timings.isVerboseEnabled());
066    }
067
068    void tick(boolean violated) {
069        if (this.timingDepth != 0 || this.record.curTickCount == 0) {
070            this.timingDepth = 0;
071            this.start = 0;
072            return;
073        }
074
075        this.record.tick(violated);
076        for (TimingData data : this.children.values()) {
077            data.tick(violated);
078        }
079    }
080
081    public Timing startTiming() {
082        if (!this.enabled) {
083            return this;
084        }
085
086        if (++this.timingDepth == 1) {
087            this.start = System.nanoTime();
088            this.parent = TimingsManager.CURRENT;
089            TimingsManager.CURRENT = this;
090        }
091
092        return this;
093    }
094
095    public void stopTiming() {
096        if (!this.enabled) {
097            return;
098        }
099
100        if (--this.timingDepth == 0 && this.start != 0) {
101            this.addDiff(System.nanoTime() - this.start);
102            this.start = 0;
103        }
104    }
105
106    public void abort() {
107        if (this.enabled && this.timingDepth > 0) {
108            this.start = 0;
109        }
110    }
111
112    void addDiff(long diff) {
113        if (TimingsManager.CURRENT == this) {
114            TimingsManager.CURRENT = this.parent;
115            if (this.parent != null) {
116                if (!this.parent.children.containsKey(this.id))
117                    this.parent.children.put(this.id, new TimingData(this.id));
118                this.parent.children.get(this.id).add(diff);
119            }
120        }
121
122        this.record.add(diff);
123        if (!this.added) {
124            this.added = true;
125            this.timed = true;
126            TimingsManager.TIMINGS.add(this);
127        }
128
129        if (this.groupTiming != null) {
130            this.groupTiming.addDiff(diff);
131
132            if (!this.groupTiming.children.containsKey(this.id))
133                this.groupTiming.children.put(this.id, new TimingData(this.id));
134            this.groupTiming.children.get(this.id).add(diff);
135        }
136    }
137
138    void reset(boolean full) {
139        this.record.reset();
140        if (full) {
141            this.timed = false;
142        }
143        this.start = 0;
144        this.timingDepth = 0;
145        this.added = false;
146        this.children.clear();
147        this.checkEnabled();
148    }
149
150    @Override
151    public boolean equals(Object o) {
152        return (o instanceof Timing && this == o);
153    }
154
155    @Override
156    public int hashCode() {
157        return this.id;
158    }
159
160    //For try-with-resources
161    @Override
162    public void close() {
163        this.stopTiming();
164    }
165
166    boolean isSpecial() {
167        return this == Timings.fullServerTickTimer || this == Timings.timingsTickTimer;
168    }
169}
