064    /**
065     * Builds a JSON timings report and sends it to Aikar's viewer
066     *
067     * @param sender Sender that issued the command
068     */
069    public static void reportTimings(CommandSender sender) {
070        JsonObject out = new JsonObject();
071        out.addProperty("version", Server.getInstance().getVersion());
072        out.addProperty("maxplayers", Server.getInstance().getMaxPlayers());
073        out.addProperty("start", TimingsManager.timingStart / 1000);
074        out.addProperty("end", System.currentTimeMillis() / 1000);
075        out.addProperty("sampletime", (System.currentTimeMillis() - TimingsManager.timingStart) / 1000);
076        if (!Timings.isPrivacy()) {
077            out.addProperty("server", Server.getInstance().getName());
078            out.addProperty("motd", Server.getInstance().getMotd());
079            out.addProperty("online-mode", false); //In MCPE we have permanent offline mode.
080            out.addProperty("icon", ""); //"data:image/png;base64,"
081        }
082        final Runtime runtime = Runtime.getRuntime();
083        RuntimeMXBean runtimeBean = ManagementFactory.getRuntimeMXBean();
084        JsonObject system = new JsonObject();
085        system.addProperty("timingcost", getCost());
086        system.addProperty("name", System.getProperty("os.name"));
087        system.addProperty("version", System.getProperty("os.version"));
088        system.addProperty("jvmversion", System.getProperty("java.version"));
089        system.addProperty("arch", System.getProperty("os.arch"));
090        system.addProperty("maxmem", runtime.maxMemory());
091        system.addProperty("cpu", runtime.availableProcessors());
092        system.addProperty("runtime", ManagementFactory.getRuntimeMXBean().getUptime());
093        system.addProperty("flags", String.join(" ", runtimeBean.getInputArguments()));
094        system.add("gc", JsonUtil.mapToObject(ManagementFactory.getGarbageCollectorMXBeans(), input -> new JsonUtil.JSONPair(input.getName(), JsonUtil.toArray(input.getCollectionCount(), input.getCollectionTime()))));
095        out.add("system", system);
096        TimingsHistory[] history = HISTORY.toArray(new TimingsHistory[HISTORY.size() + 1]);
097        history[HISTORY.size()] = new TimingsHistory(); //Current snapshot
098        JsonObject timings = new JsonObject();
099        for (TimingIdentifier.TimingGroup group : TimingIdentifier.GROUP_MAP.values()) {
100            for (Timing id : group.timings) {
101                if (!id.timed && !id.isSpecial()) {
102                    continue;
103                }
104                timings.add(String.valueOf(id.id), JsonUtil.toArray(group.id, id.name));
105            }
106        }
107        JsonObject idmap = new JsonObject();
108        idmap.add("groups", JsonUtil.mapToObject(TimingIdentifier.GROUP_MAP.values(), group -> new JsonUtil.JSONPair(group.id, group.name)));
109        idmap.add("handlers", timings);
110        idmap.add("worlds", JsonUtil.mapToObject(TimingsHistory.levelMap.entrySet(), entry -> new JsonUtil.JSONPair(entry.getValue(), entry.getKey())));
111        idmap.add("tileentity", JsonUtil.mapToObject(TimingsHistory.blockEntityMap.entrySet(), entry -> new JsonUtil.JSONPair(entry.getKey(), entry.getValue())));
112        idmap.add("entity", JsonUtil.mapToObject(TimingsHistory.entityMap.entrySet(), entry -> new JsonUtil.JSONPair(entry.getKey(), entry.getValue())));
113        out.add("idmap", idmap);
114        //Information about loaded plugins
115        out.add("plugins", JsonUtil.mapToObject(Server.getInstance().getPluginManager().getPlugins().values(), plugin -> {
116            JsonObject jsonPlugin = new JsonObject();
117            jsonPlugin.addProperty("version", plugin.getDescription().getVersion());
118            jsonPlugin.addProperty("description", plugin.getDescription().getDescription());// Sounds legit
119            jsonPlugin.addProperty("website", plugin.getDescription().getWebsite());
120            jsonPlugin.addProperty("authors", String.join(", ", plugin.getDescription().getAuthors()));
121            return new JsonUtil.JSONPair(plugin.getName(), jsonPlugin);
122        }));
123        //Information on the users Config
124        JsonObject config = new JsonObject();
125        if (!Timings.getIgnoredConfigSections().contains("all")) {
126            JsonObject nukkit = JsonUtil.toObject(Server.getInstance().getConfig().getRootSection());
127            Timings.getIgnoredConfigSections().forEach(nukkit::remove);
128            config.add("nukkit", nukkit);
129        } else {
130            config.add("nukkit", null);
131        }
132        out.add("config", config);
133        new TimingsExport(sender, out, history).start();
134    }

136    private static long getCost() {
137        int passes = 200;
138        Timing SAMPLER1 = TimingsManager.getTiming(null, "Timings sampler 1", null);
139        Timing SAMPLER2 = TimingsManager.getTiming(null, "Timings sampler 2", null);
140        Timing SAMPLER3 = TimingsManager.getTiming(null, "Timings sampler 3", null);
141        Timing SAMPLER4 = TimingsManager.getTiming(null, "Timings sampler 4", null);
142        Timing SAMPLER5 = TimingsManager.getTiming(null, "Timings sampler 5", null);
143        Timing SAMPLER6 = TimingsManager.getTiming(null, "Timings sampler 6", null);
144        long start = System.nanoTime();
145        for (int i = 0; i < passes; i++) {
146            SAMPLER1.startTiming();
147            SAMPLER2.startTiming();
148            SAMPLER3.startTiming();
149            SAMPLER4.startTiming();
150            SAMPLER5.startTiming();
151            SAMPLER6.startTiming();
152            SAMPLER6.stopTiming();
153            SAMPLER5.stopTiming();
154            SAMPLER4.stopTiming();
155            SAMPLER3.stopTiming();
156            SAMPLER2.stopTiming();
157            SAMPLER1.stopTiming();
158        }
159        long timingsCost = (System.nanoTime() - start) / passes / 6;
160        SAMPLER1.reset(true);
161        SAMPLER2.reset(true);
162        SAMPLER3.reset(true);
163        SAMPLER4.reset(true);
164        SAMPLER5.reset(true);
165        SAMPLER6.reset(true);
166        return timingsCost;
167    }

179    @Override
180    public void run() {
181        this.sender.sendMessage(new TranslationContainer("nukkit.command.timings.uploadStart"));
182        this.out.add("data", JsonUtil.mapToArray(this.history, TimingsHistory::export));
183        String response = null;
184        try {
185            HttpURLConnection con = (HttpURLConnection) new URL("http://timings.aikar.co/post").openConnection();
186            con.setDoOutput(true);
187            con.setRequestProperty("User-Agent", "Nukkit/" + Server.getInstance().getName() + "/" + InetAddress.getLocalHost().getHostName());
188            con.setRequestMethod("POST");
189            con.setInstanceFollowRedirects(false);
190            PGZIPOutputStream request = new PGZIPOutputStream(con.getOutputStream());
191            request.setLevel(Deflater.BEST_COMPRESSION);
192            request.write(new Gson().toJson(this.out).getBytes(StandardCharsets.UTF_8));
193            request.close();
194            response = getResponse(con);
195            if (con.getResponseCode() != 302) {
196                this.sender.sendMessage(new TranslationContainer("nukkit.command.timings.uploadError", String.valueOf(con.getResponseCode()), con.getResponseMessage()));
197                if (response != null) {
198                    log.warn(response);
199                }
200                return;
201            }
202            String location = con.getHeaderField("Location");
203            this.sender.sendMessage(new TranslationContainer("nukkit.command.timings.timingsLocation", location));
204            if (!(this.sender instanceof ConsoleCommandSender)) {
205                log.info(Server.getInstance().getLanguage().translateString("nukkit.command.timings.timingsLocation", location));
206            }
207            if (response != null && !response.isEmpty()) {
208                log.info(Server.getInstance().getLanguage().translateString("nukkit.command.timings.timingsResponse", response));
209            }
210            File timingFolder = new File(Server.getInstance().getDataPath() + File.separator + "timings");
211            timingFolder.mkdirs();
212            String fileName = timingFolder + File.separator + new SimpleDateFormat("\'timings-\'yyyy-MM-dd-hh-mm\'.txt\'").format(new Date());
213            FileWriter writer = new FileWriter(fileName);
214            writer.write(Server.getInstance().getLanguage().translateString("nukkit.command.timings.timingsLocation", location) + "\n\n");
215            writer.write(new GsonBuilder().setPrettyPrinting().create().toJson(this.out));
216            writer.close();
217            log.info(Server.getInstance().getLanguage().translateString("nukkit.command.timings.timingsWrite", fileName));
218        } catch (IOException exception) {
219            this.sender.sendMessage(TextFormat.RED + "" + new TranslationContainer("nukkit.command.timings.reportError"));
220            if (response != null) {
221                log.warn(response);
222            }
223            log.error("An error has occurred while exporting the timings report", exception);
224        }
225    }

227    private String getResponse(HttpURLConnection con) throws IOException {
228        try (InputStream is = con.getInputStream()) {
229            ByteArrayOutputStream bos = new ByteArrayOutputStream();
230            byte[] b = new byte[1024];
231            int bytesRead;
232            while ((bytesRead = is.read(b)) != -1) {
233                bos.write(b, 0, bytesRead);
234            }
235            return bos.toString();
236        } catch (IOException exception) {
237            this.sender.sendMessage(TextFormat.RED + "" + new TranslationContainer("nukkit.command.timings.reportError"));
238            log.error("An error has occurred while getting the timings response", exception);
239            return null;
240        }
241    }
