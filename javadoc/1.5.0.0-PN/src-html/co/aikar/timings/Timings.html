157    public static void setHistoryLength(int length) {
158        //Cap at 12 History Frames, 1 hour at 5 minute frames.
159        int maxLength = historyInterval * MAX_HISTORY_FRAMES;
160        //For special cases of servers with special permission to bypass the max.
161        //This max helps keep data file sizes reasonable for processing on Aikar's Timing parser side.
162        //Setting this will not help you bypass the max unless Aikar has added an exception on the API side.
163        if (Server.getInstance().getConfig().getBoolean("timings.bypass-max", false)) {
164            maxLength = Integer.MAX_VALUE;
165        }
166        historyLength = Math.max(Math.min(maxLength, length), historyInterval);
167        Queue<TimingsHistory> oldQueue = TimingsManager.HISTORY;
168        int frames = (getHistoryLength() / getHistoryInterval());
169        if (length > maxLength) {
170            log.warn("Timings Length too high. Requested {}, max is {}. To get longer history, you must increase your interval. Set Interval to {} to achieve this length.", length, maxLength, Math.ceil((float) length / MAX_HISTORY_FRAMES));
171        }
172        TimingsManager.HISTORY = new TimingsManager.BoundedQueue<>(frames);
173        TimingsManager.HISTORY.addAll(oldQueue);
174    }

184    public static Timing getTaskTiming(TaskHandler handler, long period) {
185        String repeating = " ";
186        if (period > 0) {
187            repeating += "(interval:" + period + ")";
188        } else {
189            repeating += "(Single)";
190        }
191        if (handler.getTask() instanceof PluginTask) {
192            String owner = ((PluginTask) handler.getTask()).getOwner().getName();
193            return TimingsManager.getTiming(owner, "PluginTask: " + handler.getTaskId() + repeating, schedulerSyncTimer);
194        } else if (!handler.isAsynchronous()) {
195            return TimingsManager.getTiming(DEFAULT_GROUP.name, "Task: " + handler.getTaskId() + repeating, schedulerSyncTimer);
196        } else {
197            return null;
198        }
199    }
