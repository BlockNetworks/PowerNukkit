108    @PowerNukkitDifference(info = "Using new method to play sounds", since = "1.4.0.0-PN")
109    @Override
110    public boolean attack(EntityDamageEvent source) {
111        if (this.noDamageTicks > 0) {
112            return false;
113        } else if (this.attackTime > 0 && !attackTimeByShieldKb) {
114            EntityDamageEvent lastCause = this.getLastDamageCause();
115            if (lastCause != null && lastCause.getDamage() >= source.getDamage()) {
116                return false;
117            }
118        }
119
120        if (isBlocking() && this.blockedByShield(source)) {
121            return false;
122        }
123
124        if (super.attack(source)) {
125            if (source instanceof EntityDamageByEntityEvent) {
126                Entity damager = ((EntityDamageByEntityEvent) source).getDamager();
127                if (source instanceof EntityDamageByChildEntityEvent) {
128                    damager = ((EntityDamageByChildEntityEvent) source).getChild();
129                }
130
131                //Critical hit
132                if (damager instanceof Player && !damager.onGround) {
133                    AnimatePacket animate = new AnimatePacket();
134                    animate.action = AnimatePacket.Action.CRITICAL_HIT;
135                    animate.eid = getId();
136
137                    this.getLevel().addChunkPacket(damager.getChunkX(), damager.getChunkZ(), animate);
138                    this.getLevel().addSound(this, Sound.GAME_PLAYER_ATTACK_STRONG);
139
140                    source.setDamage(source.getDamage() * 1.5f);
141                }
142
143                if (damager.isOnFire() && !(damager instanceof Player)) {
144                    this.setOnFire(2 * this.server.getDifficulty());
145                }
146
147                double deltaX = this.x - damager.x;
148                double deltaZ = this.z - damager.z;
149                this.knockBack(damager, source.getDamage(), deltaX, deltaZ, ((EntityDamageByEntityEvent) source).getKnockBack());
150            }
151
152            EntityEventPacket pk = new EntityEventPacket();
153            pk.eid = this.getId();
154            pk.event = this.getHealth() <= 0 ? EntityEventPacket.DEATH_ANIMATION : EntityEventPacket.HURT_ANIMATION;
155            Server.broadcastPacket(this.hasSpawned.values(), pk);
156
157            this.attackTime = source.getAttackCooldown();
158            this.attackTimeByShieldKb = false;
159            this.scheduleUpdate();
160
161            return true;
162        } else {
163            return false;
164        }
165    }

171    public void knockBack(Entity attacker, double damage, double x, double z, double base) {
172        double f = Math.sqrt(x * x + z * z);
173        if (f <= 0) {
174            return;
175        }
176
177        f = 1 / f;
178
179        Vector3 motion = new Vector3(this.motionX, this.motionY, this.motionZ);
180
181        motion.x /= 2d;
182        motion.y /= 2d;
183        motion.z /= 2d;
184        motion.x += x * f * base;
185        motion.y += base;
186        motion.z += z * f * base;
187
188        if (motion.y > base) {
189            motion.y = base;
190        }
191
192        this.setMotion(motion);
193    }

195    @Override
196    public void kill() {
197        if (!this.isAlive()) {
198            return;
199        }
200        super.kill();
201        EntityDeathEvent ev = new EntityDeathEvent(this, this.getDrops());
202        this.server.getPluginManager().callEvent(ev);
203
204        if (this.level.getGameRules().getBoolean(GameRule.DO_ENTITY_DROPS)) {
205            for (cn.nukkit.item.Item item : ev.getDrops()) {
206                this.getLevel().dropItem(this, item);
207            }
208        }
209    }

216    @Override
217    public boolean entityBaseTick(int tickDiff) {
218        Timings.livingEntityBaseTickTimer.startTiming();
219        boolean isBreathing = !this.isInsideOfWater();
220
221        if (this instanceof Player) {
222            if (isBreathing && ((Player) this).getInventory().getHelmet() instanceof ItemTurtleShell) {
223                turtleTicks = 200;
224            } else if (turtleTicks > 0) {
225                isBreathing = true;
226                turtleTicks--;
227            }
228
229            if ((((Player) this).isCreative() || ((Player) this).isSpectator())) {
230                isBreathing = true;
231            }
232        }
233        
234        this.setDataFlag(DATA_FLAGS, DATA_FLAG_BREATHING, isBreathing);
235
236        boolean hasUpdate = super.entityBaseTick(tickDiff);
237
238        if (this.isAlive()) {
239
240            if (this.isInsideOfSolid()) {
241                hasUpdate = true;
242                this.attack(new EntityDamageEvent(this, DamageCause.SUFFOCATION, 1));
243            }
244
245            if (this.isOnLadder() || this.hasEffect(Effect.LEVITATION) || this.hasEffect(Effect.SLOW_FALLING)) {
246                this.resetFallDistance();
247            }
248
249            if (!this.hasEffect(Effect.WATER_BREATHING) && !this.hasEffect(Effect.CONDUIT_POWER) && this.isInsideOfWater()) {
250                if (this instanceof EntityWaterAnimal || (this instanceof Player && (((Player) this).isCreative() || ((Player) this).isSpectator()))) {
251                    this.setAirTicks(400);
252                } else {
253                    if (turtleTicks == 0 || turtleTicks == 200) {
254                        hasUpdate = true;
255                        int airTicks = this.getAirTicks() - tickDiff;
256
257                        if (airTicks <= -20) {
258                            airTicks = 0;
259                            this.attack(new EntityDamageEvent(this, DamageCause.DROWNING, 2));
260                        }
261
262                        setAirTicks(airTicks);
263                    }
264                }
265            } else {
266                if (this instanceof EntityWaterAnimal) {
267                    hasUpdate = true;
268                    int airTicks = getAirTicks() - tickDiff;
269
270                    if (airTicks <= -20) {
271                        airTicks = 0;
272                        this.attack(new EntityDamageEvent(this, DamageCause.SUFFOCATION, 2));
273                    }
274
275                    setAirTicks(airTicks);
276                } else {
277                    int airTicks = getAirTicks();
278
279                    if (airTicks < 400) {
280                        setAirTicks(Math.min(400, airTicks + tickDiff * 5));
281                    }
282                }
283            }
284        }
285
286        if (this.attackTime > 0) {
287            this.attackTime -= tickDiff;
288            if (this.attackTime <= 0) {
289                attackTimeByShieldKb = false;
290            }
291            hasUpdate = true;
292        }
293
294        if (this.riding == null) {
295            for (Entity entity : level.getNearbyEntities(this.boundingBox.grow(0.20000000298023224, 0.0D, 0.20000000298023224), this)) {
296                if (entity instanceof EntityRideable) {
297                    this.collidingWith(entity);
298                }
299            }
300        }
301
302        // Used to check collisions with magma blocks
303        Block block = this.level.getBlock((int) x, (int) y - 1, (int) z);
304        if (block instanceof BlockMagma) block.onEntityCollide(this);
305
306        Timings.livingEntityBaseTickTimer.stopTiming();
307
308        return hasUpdate;
309    }

328    public Block[] getLineOfSight(int maxDistance, int maxLength, Integer[] transparent) {
329        if (maxDistance > 120) {
330            maxDistance = 120;
331        }
332
333        if (transparent != null && transparent.length == 0) {
334            transparent = null;
335        }
336
337        List<Block> blocks = new ArrayList<>();
338
339        BlockIterator itr = new BlockIterator(this.level, this.getPosition(), this.getDirectionVector(), this.getEyeHeight(), maxDistance);
340
341        while (itr.hasNext()) {
342            Block block = itr.next();
343            blocks.add(block);
344
345            if (maxLength != 0 && blocks.size() > maxLength) {
346                blocks.remove(0);
347            }
348
349            int id = block.getId();
350
351            if (transparent == null) {
352                if (id != 0) {
353                    break;
354                }
355            } else {
356                if (Arrays.binarySearch(transparent, id) < 0) {
357                    break;
358                }
359            }
360        }
361
362        return blocks.toArray(Block.EMPTY_ARRAY);
363    }

374    public Block getTargetBlock(int maxDistance, Integer[] transparent) {
375        try {
376            Block[] blocks = this.getLineOfSight(maxDistance, 1, transparent);
377            Block block = blocks[0];
378            if (block != null) {
379                if (transparent != null && transparent.length != 0) {
380                    if (Arrays.binarySearch(transparent, block.getId()) < 0) {
381                        return block;
382                    }
383                } else {
384                    return block;
385                }
386            }
387        } catch (Exception ignored) {
388
389        }
390
391        return null;
392    }

410    @PowerNukkitOnly
411    @Since("1.4.0.0-PN")
412    protected boolean blockedByShield(EntityDamageEvent source) {
413        Entity damager = null;
414        if (source instanceof EntityDamageByChildEntityEvent) {
415            damager = ((EntityDamageByChildEntityEvent) source).getChild();
416        } else if (source instanceof EntityDamageByEntityEvent) {
417            damager = ((EntityDamageByEntityEvent) source).getDamager();
418        }
419        if (damager == null || damager instanceof EntityWeather || !this.isBlocking()) {
420            return false;
421        }
422
423        Vector3 entityPos = damager.getPosition();
424        Vector3 direction = this.getDirectionVector();
425        Vector3 normalizedVector = this.getPosition().subtract(entityPos).normalize();
426        boolean blocked = (normalizedVector.x * direction.x) + (normalizedVector.z * direction.z) < 0.0;
427        boolean knockBack = !(damager instanceof EntityProjectile);
428        EntityDamageBlockedEvent event = new EntityDamageBlockedEvent(this, source, knockBack, true);
429        if (!blocked || !source.canBeReducedByArmor()) {
430            event.setCancelled();
431        }
432
433        getServer().getPluginManager().callEvent(event);
434        if (event.isCancelled()) {
435            return false;
436        }
437
438        if (event.getKnockBackAttacker() && damager instanceof EntityLiving) {
439            EntityLiving attacker = (EntityLiving) damager;
440            double deltaX = attacker.getX() - this.getX();
441            double deltaZ = attacker.getZ() - this.getZ();
442            attacker.knockBack(this, 0, deltaX, deltaZ);
443            attacker.attackTime = 10;
444            attacker.attackTimeByShieldKb = true;
445        }
446
447        onBlock(damager, event.getAnimation());
448        return true;
449    }
