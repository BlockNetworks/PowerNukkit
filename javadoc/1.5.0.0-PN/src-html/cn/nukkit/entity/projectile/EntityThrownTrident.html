145    @Override
146    protected void initEntity() {
147        super.initEntity();
148        
149        this.closeOnCollide = false;
150        
151        this.hasAge = false;
152        
153        if (namedTag.contains("Trident")) {
154            this.trident = NBTIO.getItemHelper(namedTag.getCompound("Trident"));
155            this.loyaltyLevel = this.trident.getEnchantmentLevel(Enchantment.ID_TRIDENT_LOYALTY);
156            this.hasChanneling = this.trident.hasEnchantment(Enchantment.ID_TRIDENT_CHANNELING);
157            this.riptideLevel = this.trident.getEnchantmentLevel(Enchantment.ID_TRIDENT_RIPTIDE);
158            this.impalingLevel = this.trident.getEnchantmentLevel(Enchantment.ID_TRIDENT_IMPALING);
159        } else {
160            this.trident = Item.get(0);
161            this.loyaltyLevel = 0;
162            this.hasChanneling = false;
163            this.riptideLevel = 0;
164            this.impalingLevel = 0;
165        }
166        
167        if (namedTag.contains("damage")) {
168            this.damage = namedTag.getDouble("damage");
169        } else {
170            this.damage = 8;
171        }
172        
173        if (namedTag.contains("CollisionPos")) {
174            ListTag<DoubleTag> collisionPosList = this.namedTag.getList("CollisionPos", DoubleTag.class);
175            collisionPos = new Vector3(collisionPosList.get(0).data, collisionPosList.get(1).data, collisionPosList.get(2).data);
176        } else {
177            collisionPos = this.defaultCollisionPos.clone();
178        }
179        
180        if (namedTag.contains("StuckToBlockPos")) {
181            ListTag<IntTag> stuckToBlockPosList = this.namedTag.getList("StuckToBlockPos", IntTag.class);
182            stuckToBlockPos = new BlockVector3(stuckToBlockPosList.get(0).data, stuckToBlockPosList.get(1).data, stuckToBlockPosList.get(2).data);
183        } else {
184            stuckToBlockPos = this.defaultStuckToBlockPos.clone();
185        }
186        
187        if (namedTag.contains("favoredSlot")) {
188            this.favoredSlot = namedTag.getInt("favoredSlot");
189        } else {
190            this.favoredSlot = -1;
191        }
192        
193        if (namedTag.contains("isCreative")) {
194            this.isCreative = namedTag.getBoolean("isCreative");
195        } else {
196            this.isCreative = false;
197        }
198        
199        if (namedTag.contains("player")) {
200            this.player = namedTag.getBoolean("player");
201        } else {
202            this.player = true;
203        }
204    }

210        this.namedTag.put("Trident", NBTIO.putItemHelper(this.trident));

250    @Override
251    public int getResultDamage() {
252        int base = super.getResultDamage();
253
254        if (this.isCritical()) {
255            base += ThreadLocalRandom.current().nextInt(base / 2 + 2);

266    @PowerNukkitDifference(info = "Using new method to play sounds", since = "1.4.0.0-PN")
267    @Override
268    public boolean onUpdate(int currentTick) {
269        if (this.closed) {
270            return false;
271        }
272
273        this.timing.startTiming();
274
275        if (this.isCollided && !this.hadCollision) {
276            this.getLevel().addSound(this, Sound.ITEM_TRIDENT_HIT_GROUND);
277        }
278
279        boolean hasUpdate = super.onUpdate(currentTick);
280
281        if (this.onGround || this.hadCollision) {
282            this.setCritical(false);
283        }
284        
285        if (this.noClip) {
286            if (this.canReturnToShooter()) {
287                Entity shooter = this.shootingEntity;
288                Vector3 vector3 = new Vector3(shooter.x - this.x, shooter.y + shooter.getEyeHeight() - this.y, shooter.z - this.z);
289                this.setPosition(new Vector3(this.x, this.y + vector3.y * 0.015 * ((double) loyaltyLevel), this.z));
290                this.setMotion(this.getMotion().multiply(0.95).add(vector3.multiply(loyaltyLevel * 0.05)));
291                hasUpdate = true;
292            } else {
293                if (level.getGameRules().getBoolean(GameRule.DO_ENTITY_DROPS) && !this.closed) {
294                    this.level.dropItem(this, this.trident);
295                }
296                this.close();
297            }
298        }
299        
300        this.timing.stopTiming();
301
302        return hasUpdate;
303    }

307        AddEntityPacket pk = new AddEntityPacket();
308        pk.type = NETWORK_ID;

317        pk.yaw = (float) this.yaw;

319        pk.metadata = this.dataProperties;

322        super.spawnTo(player);

325    @PowerNukkitDifference(info = "Using new method to play sounds", since = "1.4.0.0-PN")
326    @Override
327    public void onCollideWithEntity(Entity entity) {
328        if (this.noClip) {
329            return;
330        }
331        
332        this.server.getPluginManager().callEvent(new ProjectileHitEvent(this, MovingObjectPosition.fromEntity(entity)));
333        float damage = this.getResultDamage();
334        if (this.impalingLevel > 0 && (entity.isTouchingWater() || (entity.getLevel().isRaining() && entity.getLevel().canBlockSeeSky(entity)))) {
335            damage = damage + (2.5f * (float) this.impalingLevel);
336        }
337        
338        EntityDamageEvent ev;
339        if (this.shootingEntity == null) {
340            ev = new EntityDamageByEntityEvent(this, entity, DamageCause.PROJECTILE, damage);
341        } else {
342            ev = new EntityDamageByChildEntityEvent(this.shootingEntity, this, entity, DamageCause.PROJECTILE, damage);
343        }
344        entity.attack(ev);
345        this.getLevel().addSound(this, Sound.ITEM_TRIDENT_HIT);
346        this.hadCollision = true;
347        this.setCollisionPos(this);
348        this.setMotion(new Vector3(this.getMotion().getX() * -0.01, this.getMotion().getY() * -0.1, this.getMotion().getZ() * -0.01));
349        
350        if (this.hasChanneling) {
351            if (this.level.isThundering() && this.level.canBlockSeeSky(this)) {
352                Position pos = this.getPosition();
353                EntityLightning lighting = new EntityLightning(pos.getChunk(), getDefaultNBT(pos));
354                lighting.spawnToAll();
355                this.getLevel().addSound(this, Sound.ITEM_TRIDENT_THUNDER);
356            }
357        }
358        
359        if (this.canReturnToShooter()) {
360            this.getLevel().addSound(this, Sound.ITEM_TRIDENT_RETURN);
361            this.setNoClip(true);
362            this.hadCollision = false;
363            this.setTridentRope(true);
364        }
365    }

367    public Entity create(Object type, Position source, Object... args) {
368        FullChunk chunk = source.getLevel().getChunk((int) source.x >> 4, (int) source.z >> 4);
369        if (chunk == null) return null;
370
371        CompoundTag nbt = Entity.getDefaultNBT(
372                source.add(0.5, 0, 0.5), 
373                null, 
374                new Random().nextFloat() * 360, 0
375        );
376
377        return Entity.createEntity(type.toString(), chunk, nbt, args);
378    }

380    @PowerNukkitOnly
381    @Since("1.4.0.0-PN")
382    @Override
383    protected void onCollideWithBlock(Position position, Vector3 motion) {
384        if (this.noClip) {
385            return;
386        }
387        
388        for (Block collisionBlock : level.getCollisionBlocks(getBoundingBox().grow(0.1, 0.1, 0.1))) {
389            this.setStuckToBlockPos(new BlockVector3(collisionBlock.getFloorX(), collisionBlock.getFloorY(), collisionBlock.getFloorZ()));
390            if (this.canReturnToShooter()) {
391                this.getLevel().addSound(this, Sound.ITEM_TRIDENT_RETURN);
392                this.setNoClip(true);
393                this.setTridentRope(true);
394                return;
395            }
396            onCollideWithBlock(position, motion, collisionBlock);
397        }
398    }

545    @PowerNukkitOnly
546    @Since("1.4.0.0-PN")
547    public boolean canReturnToShooter() {
548        if (this.loyaltyLevel <= 0) {
549            return false;
550        }
551        
552        if (this.getCollisionPos().equals(this.defaultCollisionPos) && this.getStuckToBlockPos().equals(this.defaultStuckToBlockPos)) {
553            return false;
554        }
555        
556        Entity shooter = this.shootingEntity;
557        if (shooter != null) {
558            if (shooter.isAlive() && shooter instanceof Player) {
559                return !(((Player) shooter).isSpectator());
560            }
561        }
562        return false;
563    }
