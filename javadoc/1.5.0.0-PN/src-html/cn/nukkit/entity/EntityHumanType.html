053    @Override
054    protected void initEntity() {
055        this.inventory = new PlayerInventory(this);
056        if (namedTag.containsNumber("SelectedInventorySlot")) {
057            this.inventory.setHeldItemSlot(NukkitMath.clamp(this.namedTag.getInt("SelectedInventorySlot"), 0, 8));
058        }
059        this.offhandInventory = new PlayerOffhandInventory(this);
060
061        if (this.namedTag.contains("Inventory") && this.namedTag.get("Inventory") instanceof ListTag) {
062            ListTag<CompoundTag> inventoryList = this.namedTag.getList("Inventory", CompoundTag.class);
063            for (CompoundTag item : inventoryList.getAll()) {
064                int slot = item.getByte("Slot");
065                if (slot >= 0 && slot < 9) { //hotbar
066                    //Old hotbar saving stuff, remove it (useless now)
067                    inventoryList.remove(item);
068                } else if (slot >= 100 && slot < 104) {
069                    this.inventory.setItem(this.inventory.getSize() + slot - 100, NBTIO.getItemHelper(item));
070                } else if (slot == -106) {
071                    this.offhandInventory.setItem(0, NBTIO.getItemHelper(item));
072                } else {
073                    this.inventory.setItem(slot - 9, NBTIO.getItemHelper(item));
074                }
075            }
076        }
077
078        this.enderChestInventory = new PlayerEnderChestInventory(this);
079
080        if (this.namedTag.contains("EnderItems") && this.namedTag.get("EnderItems") instanceof ListTag) {
081            ListTag<CompoundTag> inventoryList = this.namedTag.getList("EnderItems", CompoundTag.class);
082            for (CompoundTag item : inventoryList.getAll()) {
083                this.enderChestInventory.setItem(item.getByte("Slot"), NBTIO.getItemHelper(item));
084            }
085        }
086
087        super.initEntity();
088    }

090    @Override
091    public void saveNBT() {
092        super.saveNBT();
093
094        ListTag<CompoundTag> inventoryTag = null;
095        if (this.inventory != null) {
096            inventoryTag = new ListTag<>("Inventory");
097            this.namedTag.putList(inventoryTag);
098
099            for (int slot = 0; slot < 9; ++slot) {
100                inventoryTag.add(new CompoundTag()
101                        .putByte("Count", 0)
102                        .putShort("Damage", 0)
103                        .putByte("Slot", slot)
104                        .putByte("TrueSlot", -1)
105                        .putShort("id", 0)
106                );
107            }
108
109            int slotCount = Player.SURVIVAL_SLOTS + 9;
110            for (int slot = 9; slot < slotCount; ++slot) {
111                Item item = this.inventory.getItem(slot - 9);
112                inventoryTag.add(NBTIO.putItemHelper(item, slot));
113            }
114
115            for (int slot = 100; slot < 104; ++slot) {
116                Item item = this.inventory.getItem(this.inventory.getSize() + slot - 100);
117                if (item != null && item.getId() != Item.AIR) {
118                    inventoryTag.add(NBTIO.putItemHelper(item, slot));
119                }
120            }
121            
122            this.namedTag.putInt("SelectedInventorySlot", this.inventory.getHeldItemIndex());
123        }
124
125        if (this.offhandInventory != null) {
126            Item item = this.offhandInventory.getItem(0);
127            if (item.getId() != Item.AIR) {
128                if (inventoryTag == null) {
129                    inventoryTag = new ListTag<>("Inventory");
130                    this.namedTag.putList(inventoryTag);
131                }
132                inventoryTag.add(NBTIO.putItemHelper(item, -106));
133            }
134        }
135
136        this.namedTag.putList(new ListTag<CompoundTag>("EnderItems"));
137        if (this.enderChestInventory != null) {
138            for (int slot = 0; slot < 27; ++slot) {
139                Item item = this.enderChestInventory.getItem(slot);
140                if (item != null && item.getId() != Item.AIR) {
141                    this.namedTag.getList("EnderItems", CompoundTag.class).add(NBTIO.putItemHelper(item, slot));
142                }
143            }
144        }
145    }

147    @Override
148    public Item[] getDrops() {
149        if (this.inventory != null) {
150            List<Item> drops = new ArrayList<>(this.inventory.getContents().values());
151            drops.addAll(this.offhandInventory.getContents().values());
152            return drops.toArray(Item.EMPTY_ARRAY);
153        }
154        return Item.EMPTY_ARRAY;
155    }

157    @Override
158    public boolean attack(EntityDamageEvent source) {
159        if (this.isClosed() || !this.isAlive()) {
160            return false;
161        }
162
163        if (source.getCause() != DamageCause.VOID && source.getCause() != DamageCause.CUSTOM && source.getCause() != DamageCause.MAGIC) {
164            int armorPoints = 0;
165            int epf = 0;
166            int toughness = 0;
167
168            for (Item armor : inventory.getArmorContents()) {
169                armorPoints += armor.getArmorPoints();
170                epf += calculateEnchantmentProtectionFactor(armor, source);
171                //toughness += armor.getToughness();
172            }
173
174            if (source.canBeReducedByArmor()) {
175                source.setDamage(-source.getFinalDamage() * armorPoints * 0.04f, DamageModifier.ARMOR);
176            }
177
178            source.setDamage(-source.getFinalDamage() * Math.min(NukkitMath.ceilFloat(Math.min(epf, 25) * ((float) ThreadLocalRandom.current().nextInt(50, 100) / 100)), 20) * 0.04f,
179                    DamageModifier.ARMOR_ENCHANTMENTS);
180
181            source.setDamage(-Math.min(this.getAbsorption(), source.getFinalDamage()), DamageModifier.ABSORPTION);
182        }
183
184        if (super.attack(source)) {
185            Entity damager = null;
186
187            if (source instanceof EntityDamageByEntityEvent) {
188                damager = ((EntityDamageByEntityEvent) source).getDamager();
189            }
190
191            for (int slot = 0; slot < 4; slot++) {
192                Item armor = damageArmor(this.inventory.getArmorItem(slot), damager);
193                inventory.setArmorItem(slot, armor, armor.getId() != BlockID.AIR);
194            }
195
196            return true;
197        } else {
198            return false;
199        }
200    }
201
202    protected double calculateEnchantmentProtectionFactor(Item item, EntityDamageEvent source) {
203        if (!item.hasEnchantments()) {
204            return 0;
205        }
206
207        double epf = 0;
208
209        for (Enchantment ench : item.getEnchantments()) {
210            epf += ench.getProtectionFactor(source);
211        }
212
213        return epf;
214    }

216    @Override
217    public void setOnFire(int seconds) {
218        int level = 0;
219
220        for (Item armor : this.inventory.getArmorContents()) {
221            Enchantment fireProtection = armor.getEnchantment(Enchantment.ID_PROTECTION_FIRE);
222
223            if (fireProtection != null && fireProtection.getLevel() > 0) {
224                level = Math.max(level, fireProtection.getLevel());
225            }
226        }
227
228        seconds = (int) (seconds * (1 - level * 0.15));
229
230        super.setOnFire(seconds);
231    }

245    @PowerNukkitOnly
246    @Since("1.4.0.0-PN")
247    protected Item damageArmor(Item armor, Entity damager) {
248        if (armor.hasEnchantments()) {
249            if (damager != null) {
250                for (Enchantment enchantment : armor.getEnchantments()) {
251                    enchantment.doPostAttack(damager, this);
252                }
253            }
254
255            Enchantment durability = armor.getEnchantment(Enchantment.ID_DURABILITY);
256            if (durability != null
257                    && durability.getLevel() > 0
258                    && (100 / (durability.getLevel() + 1)) <= Utils.random.nextInt(100)) {
259                return armor;
260            }
261        }
262
263        if (armor.isUnbreakable() || armor.getMaxDurability() < 0) {
264            return armor;
265        }
266
267        armor.setDamage(armor.getDamage() + 1);
268
269        if (armor.getDamage() >= armor.getMaxDurability()) {
270            getLevel().addSound(this, Sound.RANDOM_BREAK);
271            return Item.get(BlockID.AIR, 0, 0);
272        }
273
274        return armor;
275    }
