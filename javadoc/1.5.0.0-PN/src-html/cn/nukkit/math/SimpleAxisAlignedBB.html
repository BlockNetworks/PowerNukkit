001package cn.nukkit.math;
002
003/**
004 * @author MagicDroidX (Nukkit Project)
005 */
006public class SimpleAxisAlignedBB implements AxisAlignedBB {
007
008    private double minX;
009    private double minY;
010    private double minZ;
011    private double maxX;
012    private double maxY;
013    private double maxZ;
014
015    public SimpleAxisAlignedBB(Vector3 pos1, Vector3 pos2) {
016        this.minX = Math.min(pos1.x, pos2.x);
017        this.minY = Math.min(pos1.y, pos2.y);
018        this.minZ = Math.min(pos1.z, pos2.z);
019        this.maxX = Math.max(pos1.x, pos2.x);
020        this.maxY = Math.max(pos1.y, pos2.y);
021        this.maxZ = Math.max(pos1.z, pos2.z);
022    }
023
024    public SimpleAxisAlignedBB(double minX, double minY, double minZ, double maxX, double maxY, double maxZ) {
025        this.minX = minX;
026        this.minY = minY;
027        this.minZ = minZ;
028        this.maxX = maxX;
029        this.maxY = maxY;
030        this.maxZ = maxZ;
031    }
032
033    @Override
034    public String toString() {
035        return "AxisAlignedBB(" + this.getMinX() + ", " + this.getMinY() + ", " + this.getMinZ() + ", " + this.getMaxX() + ", " + this.getMaxY() + ", " + this.getMaxZ() + ")";
036    }
037
038    @Override
039    public double getMinX() {
040        return minX;
041    }
042
043    @Override
044    public void setMinX(double minX) {
045        this.minX = minX;
046    }
047
048    @Override
049    public double getMinY() {
050        return minY;
051    }
052
053    @Override
054    public void setMinY(double minY) {
055        this.minY = minY;
056    }
057
058    @Override
059    public double getMinZ() {
060        return minZ;
061    }
062
063    @Override
064    public void setMinZ(double minZ) {
065        this.minZ = minZ;
066    }
067
068    @Override
069    public double getMaxX() {
070        return maxX;
071    }
072
073    @Override
074    public void setMaxX(double maxX) {
075        this.maxX = maxX;
076    }
077
078    @Override
079    public double getMaxY() {
080        return maxY;
081    }
082
083    @Override
084    public void setMaxY(double maxY) {
085        this.maxY = maxY;
086    }
087
088    @Override
089    public double getMaxZ() {
090        return maxZ;
091    }
092
093    @Override
094    public void setMaxZ(double maxZ) {
095        this.maxZ = maxZ;
096    }
097
098    @Override
099    public AxisAlignedBB clone() {
100        return new SimpleAxisAlignedBB(minX, minY, minZ, maxX, maxY, maxZ);
101    }
102}
