001package cn.nukkit.math;
002
003import cn.nukkit.api.PowerNukkitOnly;
004import cn.nukkit.api.Since;
005
006/**
007 * @author MagicDroidX (Nukkit Project)
008 */
009
010public abstract class VectorMath {
011
012    public static Vector2 getDirection2D(double azimuth) {
013        return new Vector2(Math.cos(azimuth), Math.sin(azimuth));
014    }
015
016    @PowerNukkitOnly
017    @Since("1.4.0.0-PN")
018    public static  BlockFace.Axis calculateAxis(Vector3 base, Vector3 side) {
019        Vector3 vector = side.subtract(base);
020        return vector.x != 0? BlockFace.Axis.X : vector.z != 0? BlockFace.Axis.Z : BlockFace.Axis.Y;
021    }
022
023    @PowerNukkitOnly
024    @Since("1.4.0.0-PN")
025    public static  BlockFace calculateFace(Vector3 base, Vector3 side) {
026        Vector3 vector = side.subtract(base);
027        BlockFace.Axis axis = vector.x != 0? BlockFace.Axis.X : vector.z != 0? BlockFace.Axis.Z : BlockFace.Axis.Y;
028        double direction = vector.getAxis(axis);
029        return BlockFace.fromAxis(direction < 0? BlockFace.AxisDirection.NEGATIVE : BlockFace.AxisDirection.POSITIVE, axis);
030    }
031
032}
