001package cn.nukkit.scheduler;
002
003
004import cn.nukkit.plugin.Plugin;
005
006/**
007 * 插件创建的任务。<br>Task that created by a plugin.
008 *
009 * <p>对于插件作者，通过继承这个类创建的任务，可以在插件被禁用时不被执行。<br>
010 * For plugin developers: Tasks that extend this class, won't be executed when the plugin is disabled.</p>
011 *
012 * <p>另外，继承这个类的任务可以通过{@link #getOwner()}来获得这个任务所属的插件。<br>
013 * Otherwise, tasks that extend this class can use {@link #getOwner()} to get its owner.</p>
014 *
015 * 下面是一个插件创建任务的例子：<br>An example for plugin create a task:
016 * <pre>
017 *     public class ExampleTask extends PluginTask&lt;ExamplePlugin&gt;{
018 *         public ExampleTask(ExamplePlugin plugin){
019 *             super(plugin);
020 *         }
021 *
022 *        {@code @Override}
023 *         public void onRun(int currentTick){
024 *             getOwner().getLogger().info("Task is executed in tick "+currentTick);
025 *         }
026 *     }
027 * </pre>
028 *
029 * <p>如果要让Nukkit能够延时或循环执行这个任务，请使用{@link ServerScheduler}。<br>
030 * If you want Nukkit to execute this task with delay or repeat, use {@link ServerScheduler}.</p>
031 *
032 * @param <T> 这个任务所属的插件。<br>The plugin that owns this task.
033 * @author MagicDroidX(code) @ Nukkit Project
034 * @author 粉鞋大妈(javadoc) @ Nukkit Project
035 * @since Nukkit 1.0 | Nukkit API 1.0.0
036 */
037public abstract class PluginTask<T extends Plugin> extends Task {
038
039    protected final T owner;
040
041    /**
042     * 构造一个插件拥有的任务的方法。<br>Constructs a plugin-owned task.
043     *
044     * @param owner 这个任务的所有者插件。<br>The plugin object that owns this task.
045     * @since Nukkit 1.0 | Nukkit API 1.0.0
046     */
047    public PluginTask(T owner) {
048        this.owner = owner;
049    }
050
051    /**
052     * 返回这个任务的所有者插件。<br>
053     * Returns the owner of this task.
054     *
055     * @return 这个任务的所有者插件。<br>The plugin that owns this task.
056     * @since Nukkit 1.0 | Nukkit API 1.0.0
057     */
058    public final T getOwner() {
059        return this.owner;
060    }
061
062}
