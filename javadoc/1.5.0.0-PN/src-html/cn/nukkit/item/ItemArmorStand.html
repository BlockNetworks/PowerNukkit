001package cn.nukkit.item;
002
003import cn.nukkit.Player;
004import cn.nukkit.api.PowerNukkitOnly;
005import cn.nukkit.api.Since;
006import cn.nukkit.block.Block;
007import cn.nukkit.block.BlockID;
008import cn.nukkit.entity.Entity;
009import cn.nukkit.entity.item.EntityArmorStand;
010import cn.nukkit.level.Level;
011import cn.nukkit.level.Sound;
012import cn.nukkit.level.format.FullChunk;
013import cn.nukkit.math.BlockFace;
014import cn.nukkit.math.CompassRoseDirection;
015import cn.nukkit.math.SimpleAxisAlignedBB;
016import cn.nukkit.math.Vector3;
017import cn.nukkit.nbt.tag.CompoundTag;
018
019import static cn.nukkit.math.CompassRoseDirection.Precision.PRIMARY_INTER_CARDINAL;
020
021@PowerNukkitOnly
022@Since("1.4.0.0-PN")
023public class ItemArmorStand extends Item {
024
025    @PowerNukkitOnly
026    @Since("1.4.0.0-PN")
027    public ItemArmorStand() {
028        this(0);
029    }
030
031    @PowerNukkitOnly
032    @Since("1.4.0.0-PN")
033    public ItemArmorStand(Integer meta) {
034        this(meta, 1);
035    }
036
037    @PowerNukkitOnly
038    @Since("1.4.0.0-PN")
039    public ItemArmorStand(Integer meta, int count) {
040        super(ARMOR_STAND, meta, count, "Armor Stand");
041    }
042
043    @Override
044    public boolean canBeActivated() {
045        return true;
046    }
047
048    @Override
049    public int getMaxStackSize() {
050        return 64;
051    }
052
053    @Override
054    public boolean onActivate(Level level, Player player, Block block, Block target, BlockFace face, double fx, double fy, double fz) {
055        if (player.isAdventure()) {
056            return false;
057        }
058        
059        FullChunk chunk = block.getChunk();
060        if (chunk == null) {
061            return false;
062        }
063        
064        if (!block.canBeReplaced() || !block.up().canBeReplaced()) {
065            return false;
066        }
067        
068        for (Entity collidingEntity : level.getCollidingEntities(new SimpleAxisAlignedBB(block.x, block.y, block.z, block.x + 1, block.y + 1, block.z + 1))) {
069            if (collidingEntity instanceof EntityArmorStand) {
070                return false;
071            }
072        }
073
074        CompassRoseDirection direction = CompassRoseDirection.getClosestFromYaw(player.yaw, PRIMARY_INTER_CARDINAL).getOppositeFace();
075        CompoundTag nbt = Entity.getDefaultNBT(block.add(0.5, 0, 0.5), new Vector3(), direction.getYaw(), 0f);
076        if (this.hasCustomName()) {
077            nbt.putString("CustomName", this.getCustomName());
078        }
079        
080        if (!removeForPlacement(block) || !removeForPlacement(block.up())) {
081            return false;
082        }
083
084        Entity entity = Entity.createEntity(EntityArmorStand.NETWORK_ID, chunk, nbt);
085        if (entity == null) {
086            return false;
087        }
088        
089        if (!player.isCreative()) {
090            player.getInventory().decreaseCount(player.getInventory().getHeldItemIndex());
091        }
092        
093        entity.spawnToAll();
094        player.getLevel().addSound(entity, Sound.MOB_ARMOR_STAND_PLACE);
095        return true;
096    }
097
098    /**
099     * @param block The block which is in the same space as the armor stand.
100     * @return {@code true} if the armor stand entity can be placed
101     */
102    @PowerNukkitOnly
103    @Since("1.4.0.0-PN")
104    protected boolean removeForPlacement(Block block) {
105        switch (block.getId()) {
106            case AIR:
107                return true;
108            case SNOW_LAYER:
109                return block.canBeReplaced();
110            default:
111        }
112        
113        return block.getLevel().setBlock(block, Block.get(BlockID.AIR));
114    }
115}
