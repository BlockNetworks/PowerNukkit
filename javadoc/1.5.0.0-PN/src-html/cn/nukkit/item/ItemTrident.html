001package cn.nukkit.item;
002
003import cn.nukkit.Player;
004import cn.nukkit.Server;
005import cn.nukkit.api.PowerNukkitDifference;
006import cn.nukkit.entity.Entity;
007import cn.nukkit.entity.projectile.EntityProjectile;
008import cn.nukkit.entity.projectile.EntityThrownTrident;
009import cn.nukkit.event.entity.EntityShootBowEvent;
010import cn.nukkit.event.entity.ProjectileLaunchEvent;
011import cn.nukkit.item.enchantment.Enchantment;
012import cn.nukkit.level.Sound;
013import cn.nukkit.math.Vector3;
014import cn.nukkit.nbt.tag.CompoundTag;
015import cn.nukkit.nbt.tag.DoubleTag;
016import cn.nukkit.nbt.tag.FloatTag;
017import cn.nukkit.nbt.tag.ListTag;
018
019/**
020 * @author PetteriM1
021 */
022public class ItemTrident extends ItemTool {
023
024    public ItemTrident() {
025        this(0, 1);
026    }
027
028    public ItemTrident(Integer meta) {
029        this(meta, 1);
030    }
031
032    public ItemTrident(Integer meta, int count) {
033        super(TRIDENT, meta, count, "Trident");
034    }
035
036    @Override
037    public int getMaxDurability() {
038        return ItemTool.DURABILITY_TRIDENT;
039    }
040    
041    @Override
042    public boolean isSword() {
043        return true;
044    }
045    
046    @Override
047    public int getAttackDamage() {
048        return 9;
049    }
050
051    @Override
052    public boolean onClickAir(Player player, Vector3 directionVector) {
053        return true;
054    }
055
056    @PowerNukkitDifference(info = "Using new method to play sounds", since = "1.4.0.0-PN")
057    @Override
058    public boolean onRelease(Player player, int ticksUsed) {
059        if (this.getEnchantmentLevel(Enchantment.ID_TRIDENT_RIPTIDE) > 0) {
060            return true;
061        }
062        
063        this.useOn(player);
064
065        CompoundTag nbt = new CompoundTag()
066                .putList(new ListTag<DoubleTag>("Pos")
067                        .add(new DoubleTag("", player.x))
068                        .add(new DoubleTag("", player.y + player.getEyeHeight()))
069                        .add(new DoubleTag("", player.z)))
070                .putList(new ListTag<DoubleTag>("Motion")
071                        .add(new DoubleTag("", -Math.sin(player.yaw / 180 * Math.PI) * Math.cos(player.pitch / 180 * Math.PI)))
072                        .add(new DoubleTag("", -Math.sin(player.pitch / 180 * Math.PI)))
073                        .add(new DoubleTag("", Math.cos(player.yaw / 180 * Math.PI) * Math.cos(player.pitch / 180 * Math.PI))))
074                .putList(new ListTag<FloatTag>("Rotation")
075                        .add(new FloatTag("", (player.yaw > 180 ? 360 : 0) - (float) player.yaw))
076                        .add(new FloatTag("", (float) -player.pitch)));
077
078        double p = (double) ticksUsed / 20;
079
080        double f = Math.min((p * p + p * 2) / 3, 1) * 2;
081        EntityThrownTrident trident = (EntityThrownTrident) Entity.createEntity("ThrownTrident", player.chunk, nbt, player, f == 2);
082
083        if (trident == null) {
084            return false;
085        }
086
087        trident.setItem(this);
088        
089        if (player.isCreative()) {
090            trident.setCreative(true);
091        }
092        
093        trident.setFavoredSlot(player.getInventory().getHeldItemIndex());
094        
095        EntityShootBowEvent entityShootBowEvent = new EntityShootBowEvent(player, this, trident, f);
096
097        if (f < 0.1 || ticksUsed < 5) {
098            entityShootBowEvent.setCancelled();
099        }
100
101        Server.getInstance().getPluginManager().callEvent(entityShootBowEvent);
102        if (entityShootBowEvent.isCancelled()) {
103            entityShootBowEvent.getProjectile().kill();
104        } else {
105            entityShootBowEvent.getProjectile().setMotion(entityShootBowEvent.getProjectile().getMotion().multiply(entityShootBowEvent.getForce()));
106            if (entityShootBowEvent.getProjectile() instanceof EntityProjectile) {
107                ProjectileLaunchEvent ev = new ProjectileLaunchEvent(entityShootBowEvent.getProjectile());
108                Server.getInstance().getPluginManager().callEvent(ev);
109                if (ev.isCancelled()) {
110                    entityShootBowEvent.getProjectile().kill();
111                } else {
112                    entityShootBowEvent.getProjectile().spawnToAll();
113                    player.getLevel().addSound(player, Sound.ITEM_TRIDENT_THROW);
114                    if (!player.isCreative()) {
115                        this.count--;
116                        player.getInventory().setItemInHand(this);
117                    }
118                }
119            }
120        }
121
122        return true;
123    }
124}
