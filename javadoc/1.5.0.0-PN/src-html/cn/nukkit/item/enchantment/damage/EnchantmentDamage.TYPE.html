001package cn.nukkit.item.enchantment.damage;
002
003import cn.nukkit.api.DeprecationDetails;
004import cn.nukkit.api.PowerNukkitOnly;
005import cn.nukkit.api.Since;
006import cn.nukkit.item.Item;
007import cn.nukkit.item.enchantment.Enchantment;
008import cn.nukkit.item.enchantment.EnchantmentType;
009
010/**
011 * @author MagicDroidX (Nukkit Project)
012 */
013public abstract class EnchantmentDamage extends Enchantment {
014
015    public enum TYPE {
016        ALL,
017        SMITE,
018        ARTHROPODS
019    }
020
021    protected final TYPE damageType;
022
023    @PowerNukkitOnly("Re-added for backward compatibility")
024    @Deprecated @DeprecationDetails(since = "1.4.0.0-PN", by = "Cloudburst Nukkit", 
025            reason = "The signature was changed and it doesn't exists anymore in Cloudburst Nukkit",
026            replaceWith = "EnchantmentDamage(int id, String name, Rarity rarity, TYPE type)")
027    protected EnchantmentDamage(int id, String name, int weight, TYPE type) {
028        this(id, name, Rarity.fromWeight(weight), type);
029    }
030    
031    @Since("1.4.0.0-PN")
032    protected EnchantmentDamage(int id, String name, Rarity rarity, TYPE type) {
033        super(id, name, rarity, EnchantmentType.SWORD);
034        this.damageType = type;
035    }
036
037    @Override
038    public boolean checkCompatibility(Enchantment enchantment) {
039        return !(enchantment instanceof EnchantmentDamage);
040    }
041
042    @Override
043    public boolean canEnchant(Item item) {
044        return item.isAxe() || super.canEnchant(item);
045    }
046
047    @Override
048    public int getMaxLevel() {
049        return 5;
050    }
051
052    @Override
053    public String getName() {
054        return "%enchantment.damage." + this.name;
055    }
056
057    @Override
058    public boolean isMajor() {
059        return true;
060    }
061
062    @Override
063    public boolean isItemAcceptable(Item item) {
064        if (item.isAxe()) {
065            return true;
066        }
067        return super.isItemAcceptable(item);
068    }
069}
