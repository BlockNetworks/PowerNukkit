001package cn.nukkit.item.enchantment.trident;
002
003import cn.nukkit.api.DeprecationDetails;
004import cn.nukkit.api.PowerNukkitOnly;
005import cn.nukkit.api.Since;
006import cn.nukkit.item.enchantment.Enchantment;
007import cn.nukkit.item.enchantment.EnchantmentType;
008
009public abstract class EnchantmentTrident extends Enchantment {
010    @PowerNukkitOnly("Re-added for backward compatibility")
011    @Deprecated @DeprecationDetails(since = "1.4.0.0-PN", by = "Cloudburst Nukkit",
012            reason = "The signature was changed and it doesn't exists anymore in Cloudburst Nukkit",
013            replaceWith = "EnchantmentTrident(int id, String name, Rarity rarity)")
014    protected EnchantmentTrident(int id, String name, int weight) {
015        this(id, name, Rarity.fromWeight(weight));
016    }
017
018    @Since("1.4.0.0-PN")
019    protected EnchantmentTrident(int id, String name, Rarity rarity) {
020        super(id, name, rarity, EnchantmentType.TRIDENT);
021    }
022
023    @Override
024    public int getMaxEnchantAbility(int level) {
025        return 50;
026    }
027}
