048    public ItemDye() {
049        this(0, 1);
050    }
051
052    public ItemDye(Integer meta) {
053        this(meta, 1);
054    }
055
056    public ItemDye(DyeColor dyeColor) {
057        this(dyeColor.getDyeData(), 1);
058    }
059
060    public ItemDye(DyeColor dyeColor, int amount) {
061        this(dyeColor.getDyeData(), amount);
062    }
063
064    public ItemDye(Integer meta, int amount) {
065        super(DYE, meta, amount, meta <= 15? DyeColor.getByDyeData(meta).getDyeName() : DyeColor.getByDyeData(meta).getName() + " Dye");
066
067        if (this.meta == DyeColor.BROWN.getDyeData()) {
068            this.block = Block.get(BlockID.COCOA_BLOCK);
069        }
070    }
071
072    @PowerNukkitOnly
073    @Since("1.4.0.0-PN")
074    protected ItemDye(int id, Integer meta, int count, String name) {
075        super(id, meta, count, name);
076    }
077
078    @Since("1.4.0.0-PN")
079    @PowerNukkitOnly
080    @Override
081    public boolean isFertilizer() {
082        return getId() == DYE && getDyeColor().equals(DyeColor.WHITE);
083    }
084    
085    @Since("1.4.0.0-PN")
086    @PowerNukkitOnly
087    public boolean isLapisLazuli() {
088        return getId() == DYE && getDyeColor().equals(DyeColor.BLUE);
089    }
090
091    @Since("1.4.0.0-PN")
092    @PowerNukkitOnly
093    public boolean isCocoaBeans() {
094        return getId() == DYE && getDyeColor().equals(DyeColor.BROWN);
095    }
096
097    @Deprecated
098    public static BlockColor getColor(int meta) {
099        return DyeColor.getByDyeData(meta).getColor();
100    }
101
102    public DyeColor getDyeColor() {
103        return DyeColor.getByDyeData(meta);
104    }
105
106    @Deprecated
107    public static String getColorName(int meta) {
108        return DyeColor.getByDyeData(meta).getName();
109    }
110}
