001package cn.nukkit.inventory;
002
003import cn.nukkit.item.Item;
004
005public class BlastFurnaceRecipe implements SmeltingRecipe {
006
007    private final Item output;
008
009    private Item ingredient;
010
011    public BlastFurnaceRecipe(Item result, Item ingredient) {
012        this.output = result.clone();
013        this.ingredient = ingredient.clone();
014    }
015
016    public void setInput(Item item) {
017        this.ingredient = item.clone();
018    }
019
020    @Override
021    public Item getInput() {
022        return this.ingredient.clone();
023    }
024
025    @Override
026    public Item getResult() {
027        return this.output.clone();
028    }
029
030    @Override
031    public void registerToCraftingManager(CraftingManager manager) {
032        manager.registerBlastFurnaceRecipe(this);
033    }
034
035    @Override
036    public RecipeType getType() {
037        return this.ingredient.hasMeta() ? RecipeType.BLAST_FURNACE_DATA : RecipeType.BLAST_FURNACE;
038    }
039}
