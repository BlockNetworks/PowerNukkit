001package cn.nukkit;
002
003import cn.nukkit.utils.TextFormat;
004
005import java.util.HashMap;
006
007/**
008 * @author CreeperFace
009 * @since 9. 11. 2016
010 */
011public class Achievement {
012
013    public static final HashMap<String, Achievement> achievements = new HashMap<String, Achievement>() {
014        {
015            put("mineWood", new Achievement("Getting Wood"));
016            put("buildWorkBench", new Achievement("Benchmarking", "mineWood"));
017            put("buildPickaxe", new Achievement("Time to Mine!", "buildWorkBench"));
018            put("buildFurnace", new Achievement("Hot Topic", "buildPickaxe"));
019            put("acquireIron", new Achievement("Acquire hardware", "buildFurnace"));
020            put("buildHoe", new Achievement("Time to Farm!", "buildWorkBench"));
021            put("makeBread", new Achievement("Bake Bread", "buildHoe"));
022            put("bakeCake", new Achievement("The Lie", "buildHoe"));
023            put("buildBetterPickaxe", new Achievement("Getting an Upgrade", "buildPickaxe"));
024            put("buildSword", new Achievement("Time to Strike!", "buildWorkBench"));
025            put("diamonds", new Achievement("DIAMONDS!", "acquireIron"));
026        }
027    };
028
029    public static boolean broadcast(Player player, String achievementId) {
030        if (!achievements.containsKey(achievementId)) {
031            return false;
032        }
033        String translation = Server.getInstance().getLanguage().translateString("chat.type.achievement", player.getDisplayName(), TextFormat.GREEN + achievements.get(achievementId).getMessage() + TextFormat.RESET);
034
035        if (Server.getInstance().getPropertyBoolean("announce-player-achievements", true)) {
036            Server.getInstance().broadcastMessage(translation);
037        } else {
038            player.sendMessage(translation);
039        }
040        return true;
041    }
042
043    public static boolean add(String name, Achievement achievement) {
044        if (achievements.containsKey(name)) {
045            return false;
046        }
047
048        achievements.put(name, achievement);
049        return true;
050    }
051
052    public final String message;
053    public final String[] requires;
054
055    public Achievement(String message, String... requires) {
056        this.message = message;
057        this.requires = requires;
058    }
059
060    public String getMessage() {
061        return message;
062    }
063
064    public void broadcast(Player player) {
065        String translation = Server.getInstance().getLanguage().translateString("chat.type.achievement", player.getDisplayName(), TextFormat.GREEN + this.getMessage(), null);
066
067        if (Server.getInstance().getPropertyBoolean("announce-player-achievements", true)) {
068            Server.getInstance().broadcastMessage(translation);
069        } else {
070            player.sendMessage(translation);
071        }
072    }
073}
