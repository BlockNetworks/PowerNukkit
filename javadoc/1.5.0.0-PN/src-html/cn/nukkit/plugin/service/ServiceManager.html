001package cn.nukkit.plugin.service;
002
003import cn.nukkit.plugin.Plugin;
004
005import java.util.List;
006
007/**
008 * @since 16-11-20
009 */
010public interface ServiceManager {
011
012    /**
013     * Register an object as a service's provider.
014     *
015     * @param service  the service
016     * @param provider the service provider
017     * @param plugin   the plugin
018     * @param priority the priority
019     * @return {@code true}, or {@code false} only if {@code provider}
020     * already registered
021     */
022    <T> boolean register(Class<T> service, T provider, Plugin plugin, ServicePriority priority);
023
024    /**
025     * Cancel service's provider(s) offered this plugin.
026     *
027     * @param plugin the plugin
028     * @return a {@link com.google.common.collect.ImmutableList}
029     * contains cancelled {@link RegisteredServiceProvider}
030     */
031    List<RegisteredServiceProvider<?>> cancel(Plugin plugin);
032
033    /**
034     * Cancel a service's provider.
035     *
036     * @param service  the service
037     * @param provider the provider
038     * @return the cancelled {@link RegisteredServiceProvider}, or {@code null} if not
039     * any provider cancelled
040     */
041    <T> RegisteredServiceProvider<T> cancel(Class<T> service, T provider);
042
043    /**
044     * Return the service's provider.
045     *
046     * @param service the target service
047     * @return a {@link RegisteredServiceProvider} registered highest priority, or
048     * {@code null} if not exists
049     */
050    <T> RegisteredServiceProvider<T> getProvider(Class<T> service);
051
052    /**
053     * Return the known service(s).
054     *
055     * @return a {@link com.google.common.collect.ImmutableList} contains the
056     * known service(s)
057     */
058    List<Class<?>> getKnownService();
059
060    List<RegisteredServiceProvider<?>> getRegistrations(Plugin plugin);
061
062    <T> List<RegisteredServiceProvider<T>> getRegistrations(Class<T> service);
063
064    <T> boolean isProvidedFor(Class<T> service);
065}
