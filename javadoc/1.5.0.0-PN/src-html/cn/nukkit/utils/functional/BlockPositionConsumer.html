001package cn.nukkit.utils.functional;
002
003import cn.nukkit.api.PowerNukkitOnly;
004import cn.nukkit.api.Since;
005import cn.nukkit.math.IntIncrementSupplier;
006import cn.nukkit.math.NukkitMath;
007import com.google.common.base.Preconditions;
008
009import java.util.stream.IntStream;
010
011@PowerNukkitOnly
012@Since("1.4.0.0-PN")
013@FunctionalInterface
014public interface BlockPositionConsumer {
015    @PowerNukkitOnly
016    @Since("1.4.0.0-PN")
017    void accept(int x, int y, int z);
018
019    @PowerNukkitOnly
020    @Since("1.4.0.0-PN")
021    static void validate(int fromX, int fromY, int fromZ, int toX, int toY, int toZ, int xInc, int yInc, int zInc) {
022        if (fromX <= toX) {
023            Preconditions.checkArgument(xInc > 0, "Invalid xInc");
024        } else {
025            Preconditions.checkArgument(xInc < 0, "Invalid xInc");
026        }
027
028        if (fromY <= toY) {
029            Preconditions.checkArgument(yInc > 0, "Invalid yInc");
030        } else {
031            Preconditions.checkArgument(yInc < 0, "Invalid yInc");
032        }
033
034        if (fromZ <= toZ) {
035            Preconditions.checkArgument(zInc > 0, "Invalid zInc");
036        } else {
037            Preconditions.checkArgument(zInc < 0, "Invalid zInc");
038        }
039    }
040
041    @PowerNukkitOnly
042    @Since("1.4.0.0-PN")
043    static void xzy(int fromX, int fromY, int fromZ, int toX, int toY, int toZ, int xInc, int yInc, int zInc, BlockPositionConsumer iterator) {
044        validate(fromX, fromY, fromZ, toX, toY, toZ, xInc, yInc, zInc);
045        
046        IntStream xStream = new IntIncrementSupplier(fromX, xInc).stream().limit(NukkitMath.floorFloat((toX - fromX)/(float)xInc));
047        IntStream yStream = new IntIncrementSupplier(fromY, yInc).stream().limit(NukkitMath.floorFloat((toY - fromY)/(float)yInc));
048        IntStream zStream = new IntIncrementSupplier(fromZ, zInc).stream().limit(NukkitMath.floorFloat((toZ - fromZ)/(float)zInc));
049        
050        xStream.forEachOrdered(x-> 
051                zStream.forEachOrdered(z-> 
052                        yStream.forEachOrdered(y-> 
053                                iterator.accept(x, y, z)
054        )));
055    }
056
057    @PowerNukkitOnly
058    @Since("1.4.0.0-PN")
059    static void xzy(int fromX, int fromY, int fromZ, int toX, int toY, int toZ, BlockPositionConsumer iterator) {
060        xzy(fromX, fromY, fromZ, toX, toY, toZ, fromX <= toX? 1 : -1, fromY <= toY? 1 : -1, fromZ <= toZ? 1 : -1, iterator);
061    }
062
063    @PowerNukkitOnly
064    @Since("1.4.0.0-PN")
065    static void xzy(int toX, int toY, int toZ, BlockPositionConsumer iterator) {
066        xzy(0, 0, 0, toX, toY, toZ, iterator);
067    }
068}
