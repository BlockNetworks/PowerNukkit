001package cn.nukkit.utils.functional;
002
003import cn.nukkit.api.PowerNukkitOnly;
004import cn.nukkit.api.Since;
005
006import java.util.Objects;
007import java.util.function.Function;
008
009/**
010 * Represents a function that accepts three arguments and produces a result.
011 * This is the three-arity specialization of {@link Function}.
012 *
013 * <p>This is a functional interface
014 * whose functional method is {@link #apply(Object, Object, Object)}.
015 *
016 * @param <F> the type of the first argument to the function
017 * @param <S> the type of the second argument to the function
018 * @param <T> the type of the third argument to the function
019 * @param <R> the type of the result of the function
020 *
021 * @see Function
022 * @since 1.4.0.0-PN
023 */
024@PowerNukkitOnly
025@Since("1.4.0.0-PN")
026@FunctionalInterface
027public interface TriFunction<F, S, T, R> {
028
029    /**
030     * Applies this function to the given arguments.
031     *
032     * @param f the first function argument
033     * @param s the second function argument
034     * @param t the third function argument
035     * @return the function result
036     */
037    @PowerNukkitOnly
038    @Since("1.4.0.0-PN")
039    R apply(F f, S s, T t);
040
041    /**
042     * Returns a composed function that first applies this function to
043     * its input, and then applies the {@code after} function to the result.
044     * If evaluation of either function throws an exception, it is relayed to
045     * the caller of the composed function.
046     *
047     * @param <V> the type of output of the {@code after} function, and of the
048     *           composed function
049     * @param after the function to apply after this function is applied
050     * @return a composed function that first applies this function and then
051     * applies the {@code after} function
052     * @throws NullPointerException if after is null
053     */
054    @PowerNukkitOnly
055    @Since("1.4.0.0-PN")
056    default <V> TriFunction<F, S, T, V> andThen(Function<? super R, ? extends V> after) {
057        Objects.requireNonNull(after);
058        return (F f, S s, T t) -> after.apply(apply(f, s, t));
059    }
060}
