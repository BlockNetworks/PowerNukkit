040    public static void safeWrite(File currentFile, Consumer<File> operation) throws IOException {
041        File parent = currentFile.getParentFile();
042        File newFile = new File(parent, currentFile.getName() + "_new");
043        File oldFile = new File(parent, currentFile.getName() + "_old");
044        File olderFile = new File(parent, currentFile.getName() + "_older");
045        if (olderFile.isFile() && !olderFile.delete()) {
046            log.fatal("Could not delete the file {}", olderFile.getAbsolutePath());
047        }
048        if (newFile.isFile() && !newFile.delete()) {
049            log.fatal("Could not delete the file {}", newFile.getAbsolutePath());
050        }
051        try {
052            operation.accept(newFile);
053        } catch (Exception e) {
054            throw new IOException(e);
055        }
056        if (oldFile.isFile()) {
057            if (olderFile.isFile()) {
058                Utils.copyFile(oldFile, olderFile);
059            } else if (!oldFile.renameTo(olderFile)) {
060                throw new IOException("Could not rename the " + oldFile + " to " + olderFile);
061            }
062        }
063        if (currentFile.isFile() && !currentFile.renameTo(oldFile)) {
064            throw new IOException("Could not rename the " + currentFile + " to " + oldFile);
065        }
066        if (!newFile.renameTo(currentFile)) {
067            throw new IOException("Could not rename the " + newFile + " to " + currentFile);
068        }
069    }

135    public static void copyFile(File from, File to) throws IOException {
136        if (!from.exists()) {
137            throw new FileNotFoundException();
138        }
139        if (from.isDirectory() || to.isDirectory()) {
140            throw new FileNotFoundException();
141        }
142        FileInputStream fi = null;
143        FileChannel in = null;
144        FileOutputStream fo = null;
145        FileChannel out = null;
146        try {
147            if (!to.exists()) {
148                to.createNewFile();
149            }
150            fi = new FileInputStream(from);
151            in = fi.getChannel();
152            fo = new FileOutputStream(to);
153            out = fo.getChannel();
154            in.transferTo(0, in.size(), out);
155        } finally {
156            if (fi != null) fi.close();
157            if (in != null) in.close();
158            if (fo != null) fo.close();
159            if (out != null) out.close();
160        }
161    }

233    public static Object[][] splitArray(Object[] arrayToSplit, int chunkSize) {
234        if (chunkSize <= 0) {
235            return null;
236        }
237        int rest = arrayToSplit.length % chunkSize;
238        int chunks = arrayToSplit.length / chunkSize + (rest > 0 ? 1 : 0);
239        Object[][] arrays = new Object[chunks][];
240        for (int i = 0; i < (rest > 0 ? chunks - 1 : chunks); i++) {
241            arrays[i] = Arrays.copyOfRange(arrayToSplit, i * chunkSize, i * chunkSize + chunkSize);
242        }
243        if (rest > 0) {
244            arrays[chunks - 1] = Arrays.copyOfRange(arrayToSplit, (chunks - 1) * chunkSize, (chunks - 1) * chunkSize + rest);
245        }
246        return arrays;
247    }

254        T[] data = array;
255        if (copy) {
256            data = Arrays.copyOf(array, array.length);
257        }
258        for (int left = 0, right = data.length - 1; left < right; left++, right--) {
259            // swap the values at the left and right indices
260            T temp = data[left];
261            data[left] = data[right];
262            data[right] = temp;
263        }
264        return data;

288        U existing = map.get(key);
289        if (existing != null) {
290            return existing;
291        }
292        try {
293            U toPut = clazz.newInstance();
294            existing = map.putIfAbsent(key, toPut);
295            if (existing == null) {
296                return toPut;
297            }
298            return existing;
299        } catch (InstantiationException | IllegalAccessException e) {
300            throw new RuntimeException(e);

311    public static byte[] parseHexBinary(String s) {
312        final int len = s.length();
313        // "111" is not a valid hex encoding.
314        if (len % 2 != 0) throw new IllegalArgumentException("hexBinary needs to be even-length: " + s);
315        byte[] out = new byte[len / 2];
316        for (int i = 0; i < len; i += 2) {
317            int h = hexToBin(s.charAt(i));
318            int l = hexToBin(s.charAt(i + 1));
319            if (h == -1 || l == -1) throw new IllegalArgumentException("contains illegal character for hexBinary: " + s);
320            out[i / 2] = (byte) (h * 16 + l);
321        }
322        return out;
323    }

325    private static int hexToBin(char ch) {
326        if ('0' <= ch && ch <= '9') return ch - '0';
327        if ('A' <= ch && ch <= 'F') return ch - 'A' + 10;
328        if ('a' <= ch && ch <= 'f') return ch - 'a' + 10;
329        return -1;
330    }

386    public static void zipFolder(Path sourceFolderPath, Path zipPath) throws IOException {
387        try (ZipOutputStream zos = new ZipOutputStream(new FileOutputStream(zipPath.toFile()))) {
388            Files.walkFileTree(sourceFolderPath, new SimpleFileVisitor<Path>() {
389                public FileVisitResult visitFile(Path file, BasicFileAttributes attrs) throws IOException {
390                    zos.putNextEntry(new ZipEntry(sourceFolderPath.relativize(file).toString()));
391                    Files.copy(file, zos);
392                    zos.closeEntry();
393                    return FileVisitResult.CONTINUE;
394                }
