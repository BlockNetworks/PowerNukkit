001package cn.nukkit.utils;
002
003import cn.nukkit.api.PowerNukkitOnly;
004import cn.nukkit.api.Since;
005import cn.nukkit.utils.functional.BooleanConsumer;
006
007import javax.annotation.Nullable;
008import java.util.NoSuchElementException;
009import java.util.Objects;
010import java.util.function.BooleanSupplier;
011import java.util.function.Supplier;
012
013/**
014 * @author joserobjr
015 */
016@PowerNukkitOnly
017@Since("1.4.0.0-PN")
018public enum OptionalBoolean {
019    @PowerNukkitOnly
020    @Since("1.4.0.0-PN")
021    TRUE(Boolean.TRUE),
022    
023    @PowerNukkitOnly
024    @Since("1.4.0.0-PN")
025    FALSE(Boolean.FALSE),
026
027    @PowerNukkitOnly
028    @Since("1.4.0.0-PN")
029    EMPTY(null);
030    
031    @Nullable
032    private final Boolean value;
033
034    OptionalBoolean(@Nullable Boolean value) {
035        this.value = value;
036    }
037
038    @PowerNukkitOnly
039    @Since("1.4.0.0-PN")
040    public static OptionalBoolean of(Boolean value) {
041        return of(Objects.requireNonNull(value).booleanValue());
042    }
043
044    @PowerNukkitOnly
045    @Since("1.4.0.0-PN")
046    public static OptionalBoolean of(boolean value) {
047        return value? TRUE : FALSE;
048    }
049
050    @PowerNukkitOnly
051    @Since("1.4.0.0-PN")
052    public static OptionalBoolean ofNullable(Boolean value) {
053        return value == null? EMPTY : of(value);
054    }
055
056    @PowerNukkitOnly
057    @Since("1.4.0.0-PN")
058    public static OptionalBoolean empty() {
059        return EMPTY;
060    }
061
062    @PowerNukkitOnly
063    @Since("1.4.0.0-PN")
064    public boolean getAsBoolean() {
065        if (value == null) {
066            throw new NoSuchElementException("No value present");
067        }
068        return value;
069    }
070
071    @PowerNukkitOnly
072    @Since("1.4.0.0-PN")
073    public boolean isPresent() {
074        return value != null;
075    }
076
077    @PowerNukkitOnly
078    @Since("1.4.0.0-PN")
079    public void ifPresent(BooleanConsumer consumer) {
080        if (value != null) {
081            consumer.accept(value);
082        }
083    }
084
085    @PowerNukkitOnly
086    @Since("1.4.0.0-PN")
087    public boolean orElse(boolean other) {
088        return value != null? value : other;
089    }
090
091    @PowerNukkitOnly
092    @Since("1.4.0.0-PN")
093    public boolean orElseGet(BooleanSupplier other) {
094        return value != null? value : other.getAsBoolean();
095    }
096
097    @PowerNukkitOnly
098    @Since("1.4.0.0-PN")
099    public<X extends Throwable> boolean orElseThrow(Supplier<X> exceptionSupplier) throws X {
100        if (value != null) {
101            return value;
102        } else {
103            throw exceptionSupplier.get();
104        }
105    }
106
107    @Override
108    public String toString() {
109        return value == null? "OptionalBoolean.empty" : 
110                value? "OptionalBoolean[true]" : 
111                        "OptionalBoolean[false]";
112    }
113}
