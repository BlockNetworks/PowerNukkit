001package cn.nukkit.utils;
002
003import cn.nukkit.api.PowerNukkitDifference;
004import cn.nukkit.api.PowerNukkitOnly;
005import cn.nukkit.api.Since;
006import cn.nukkit.math.MathHelper;
007
008import java.util.Arrays;
009
010public enum DyeColor {
011
012
013    BLACK(0, 15, 16, "Black", "Ink Sack", BlockColor.BLACK_BLOCK_COLOR, new BlockColor(0x1D1D21)),
014    RED(1, 14, 1, "Red", "Rose Red", BlockColor.RED_BLOCK_COLOR, new BlockColor(0xB02E26)),
015    GREEN(2, 13, 2, "Green", "Cactus Green", BlockColor.GREEN_BLOCK_COLOR, new BlockColor(0x5E7C16)),
016    BROWN(3, 12, 17, "Brown", "Cocoa Beans", BlockColor.BROWN_BLOCK_COLOR, new BlockColor(0x835432)),
017    BLUE(4, 11, 18, "Blue", "Lapis Lazuli", BlockColor.BLUE_BLOCK_COLOR, new BlockColor(0x3C44AA)),
018    PURPLE(5, 10, 5, "Purple", BlockColor.PURPLE_BLOCK_COLOR, new BlockColor(0x8932B8)),
019    CYAN(6, 9, 6, "Cyan", BlockColor.CYAN_BLOCK_COLOR, new BlockColor(0x169C9C)),
020    LIGHT_GRAY(7, 8, 7, "Light Gray", BlockColor.LIGHT_GRAY_BLOCK_COLOR, new BlockColor(0x9D9D97)),
021    GRAY(8, 7, 8, "Gray", BlockColor.GRAY_BLOCK_COLOR, new BlockColor(0x474F52)),
022    PINK(9, 6, 9, "Pink", BlockColor.PINK_BLOCK_COLOR, new BlockColor(0xF38BAA)),
023    LIME(10, 5, 10, "Lime", BlockColor.LIME_BLOCK_COLOR, new BlockColor(0x80C71F)),
024    YELLOW(11, 4, 11, "Yellow", "Dandelion Yellow", BlockColor.YELLOW_BLOCK_COLOR, new BlockColor(0xFED83D)),
025    LIGHT_BLUE(12, 3, 12, "Light Blue", BlockColor.LIGHT_BLUE_BLOCK_COLOR, new BlockColor(0x3AB3DA)),
026    MAGENTA(13, 2, 13, "Magenta", BlockColor.MAGENTA_BLOCK_COLOR, new BlockColor(0xC74EBD)),
027    ORANGE(14, 1, 14, "Orange", BlockColor.ORANGE_BLOCK_COLOR, new BlockColor(0xFF9801)),
028    WHITE(15, 0, 19, "White", "Bone Meal", BlockColor.WHITE_BLOCK_COLOR, new BlockColor(0xF0F0F0));
029
030
031    private int dyeColorMeta;
032    private int itemDyeMeta;
033    private int woolColorMeta;
034    private String colorName;
035    private String dyeName;
036    private BlockColor blockColor;
037    private BlockColor leatherColor;
038
039
040    private final static DyeColor[] BY_WOOL_DATA;
041    private final static DyeColor[] BY_DYE_DATA;
042
043    DyeColor(int dyeColorMeta, int woolColorMeta, int itemDyeMeta, String colorName, BlockColor blockColor) {
044        this(dyeColorMeta, woolColorMeta, itemDyeMeta, colorName, blockColor, blockColor);
045    }
046
047    DyeColor(int dyeColorMeta, int woolColorMeta, int itemDyeMeta, String colorName, BlockColor blockColor, BlockColor leatherColor) {
048        this(dyeColorMeta, woolColorMeta, itemDyeMeta, colorName, colorName + " Dye", blockColor, leatherColor);
049    }
050
051    DyeColor(int dyeColorMeta, int woolColorMeta, int itemDyeMeta, String colorName, String dyeName, BlockColor blockColor) {
052        this(dyeColorMeta, woolColorMeta, itemDyeMeta, colorName, blockColor, blockColor);
053    }
054
055    DyeColor(int dyeColorMeta, int woolColorMeta, int itemDyeMeta, String colorName, String dyeName, BlockColor blockColor, BlockColor leatherColor) {
056        this.dyeColorMeta = dyeColorMeta;
057        this.woolColorMeta = woolColorMeta;
058        this.itemDyeMeta = itemDyeMeta;
059        this.colorName = colorName;
060        this.blockColor = blockColor;
061        this.dyeName = dyeName;
062        this.leatherColor = leatherColor;
063    }
064
065    public BlockColor getColor() {
066        return this.blockColor;
067    }
068
069    /**
070     * The {@code minecraft:dye} meta from `0-15` that represents the source of a dye. Includes
071     * ink_sac, bone_meal, cocoa_beans, and lapis_lazuli.
072     */
073    public int getDyeData() {
074        return this.dyeColorMeta;
075    }
076
077    /**
078     * The {@code minecraft:dye} meta that actually represents the item dye for that color.
079     * Uses black_dye instead of ink_sac, white_dye instead of bone_meal, and so on.
080     */
081    @PowerNukkitOnly
082    @Since("1.4.0.0-PN")
083    public int getItemDyeMeta() {
084        return itemDyeMeta;
085    }
086
087    @PowerNukkitOnly
088    @Since("1.4.0.0-PN")
089    public BlockColor getLeatherColor() {
090        return leatherColor;
091    }
092
093    public int getWoolData() {
094        return this.woolColorMeta;
095    }
096
097    public String getName() {
098        return this.colorName;
099    }
100
101    public String getDyeName() {
102        return this.dyeName;
103    }
104
105    static {
106        BY_WOOL_DATA = values();
107        BY_DYE_DATA = new DyeColor[Arrays.stream(BY_WOOL_DATA).mapToInt(DyeColor::getItemDyeMeta).max().orElse(0) + 1];
108
109        for (DyeColor dyeColor : BY_WOOL_DATA) {
110            BY_DYE_DATA[dyeColor.dyeColorMeta] = dyeColor;
111            BY_DYE_DATA[dyeColor.itemDyeMeta] = dyeColor;
112        }
113
114        for (DyeColor color : values()) {
115            BY_WOOL_DATA[color.woolColorMeta & 0x0f] = color;
116        }
117    }
118
119    @PowerNukkitDifference(since = "1.4.0.0-PN", info = "When overflowed, instead of wrapping, the meta will be clamped, accepts the new dye metas")
120    public static DyeColor getByDyeData(int dyeColorMeta) {
121        return BY_DYE_DATA[MathHelper.clamp(dyeColorMeta, 0, BY_DYE_DATA.length - 1)];
122    }
123
124    public static DyeColor getByWoolData(int woolColorMeta) {
125        return BY_WOOL_DATA[woolColorMeta & 0x0f];
126    }
127}
