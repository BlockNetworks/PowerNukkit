001package cn.nukkit.block;
002
003import cn.nukkit.api.PowerNukkitOnly;
004import cn.nukkit.api.Since;
005import cn.nukkit.blockproperty.BlockProperties;
006import cn.nukkit.item.Item;
007
008import javax.annotation.Nonnull;
009
010@PowerNukkitOnly
011@Since("1.4.0.0-PN")
012public abstract class BlockDoubleSlabBase extends BlockSolidMeta {
013    @PowerNukkitOnly
014    @Since("1.4.0.0-PN")
015    public BlockDoubleSlabBase(int meta) {
016        super(meta);
017    }
018
019    @Override
020    public String getName() {
021        return "Double "+getSlabName()+" Slab";
022    }
023
024    @Since("1.4.0.0-PN")
025    @PowerNukkitOnly
026    @Nonnull
027    @Override
028    public BlockProperties getProperties() {
029        return BlockSlab.SIMPLE_SLAB_PROPERTIES;
030    }
031
032    @PowerNukkitOnly
033    @Since("1.4.0.0-PN")
034    public abstract String getSlabName();
035    
036    @PowerNukkitOnly
037    @Since("1.4.0.0-PN")
038    public abstract int getSingleSlabId();
039
040    @Override
041    public Item toItem() {
042        return getCurrentState().forItem().withBlockId(getSingleSlabId()).asItemBlock();
043    }
044
045    @PowerNukkitOnly
046    @Since("1.4.0.0-PN")
047    protected boolean isCorrectTool(Item item) {
048        return canHarvestWithHand() || canHarvest(item);
049    }
050
051    @Override
052    public Item[] getDrops(Item item) {
053        if (isCorrectTool(item)) {
054            Item slab = toItem();
055            slab.setCount(2);
056            return new Item[]{ slab };
057        } else {
058            return Item.EMPTY_ARRAY;
059        }
060    }
061}
