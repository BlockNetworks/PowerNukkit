001package cn.nukkit.block;
002
003import cn.nukkit.api.PowerNukkitOnly;
004import cn.nukkit.api.Since;
005import cn.nukkit.blockproperty.BlockProperties;
006import cn.nukkit.blockproperty.value.StoneSlab4Type;
007import cn.nukkit.item.ItemTool;
008import cn.nukkit.utils.BlockColor;
009
010import javax.annotation.Nonnull;
011
012public class BlockSlabStone4 extends BlockSlab {
013    @PowerNukkitOnly
014    @Since("1.4.0.0-PN")
015    public static final BlockProperties PROPERTIES = new BlockProperties(
016            StoneSlab4Type.PROPERTY,
017            TOP_SLOT_PROPERTY
018    );
019
020    public static final int MOSSY_STONE_BRICKS = 0;
021    public static final int SMOOTH_QUARTZ = 1;
022    public static final int STONE = 2;
023    public static final int CUT_SANDSTONE = 3;
024    public static final int CUT_RED_SANDSTONE = 4;
025
026
027    public BlockSlabStone4() {
028        this(0);
029    }
030
031    public BlockSlabStone4(int meta) {
032        super(meta, DOUBLE_STONE_SLAB4);
033    }
034
035    @Override
036    public int getId() {
037        return STONE_SLAB4;
038    }
039
040    @Since("1.4.0.0-PN")
041    @PowerNukkitOnly
042    @Nonnull
043    @Override
044    public BlockProperties getProperties() {
045        return PROPERTIES;
046    }
047
048    @PowerNukkitOnly
049    @Since("1.4.0.0-PN")
050    public StoneSlab4Type getSlabType() {
051        return getPropertyValue(StoneSlab4Type.PROPERTY);
052    }
053
054    @PowerNukkitOnly
055    @Since("1.4.0.0-PN")
056    public void setSlabType(StoneSlab4Type type) {
057        setPropertyValue(StoneSlab4Type.PROPERTY, type);
058    }
059
060    @Override
061    public String getSlabName() {
062        return getSlabType().getEnglishName();
063    }
064
065    @Override
066    public boolean isSameType(BlockSlab slab) {
067        return slab.getId() == getId() && getSlabType().equals(slab.getPropertyValue(StoneSlab4Type.PROPERTY));
068    }
069
070
071    @Override
072    public BlockColor getColor() {
073        return getSlabType().getColor();
074    }
075
076    @Override
077    public int getToolTier() {
078        return ItemTool.TIER_WOODEN;
079    }
080
081    @Override
082    public int getToolType() {
083        return ItemTool.TYPE_PICKAXE;
084    }
085
086    @Override
087    public boolean canHarvestWithHand() {
088        return false;
089    }
090}
