033

037

041

043    @Since("1.5.0.0-PN")
044    public static final BlockProperties PROPERTIES = new BlockProperties(EXPLODE_ON_BREAK, ALLOW_UNDERWATER);

048

052    }

064        return PROPERTIES;
065    }

069        return 0;

074        return 0;

079        return true;

092    public void prime() {
093        this.prime(80);
094    }
095
096    public void prime(int fuse) {
097        prime(fuse, null);
098    }
099
100    @PowerNukkitDifference(info = "TNT Sound handled by EntityPrimedTNT", since = "1.4.0.0-PN")
101    public void prime(int fuse, Entity source) {
102        this.getLevel().setBlock(this, Block.get(BlockID.AIR), true);
103        double mot = (new NukkitRandom()).nextSignedFloat() * Math.PI * 2;
104        CompoundTag nbt = new CompoundTag()
105                .putList(new ListTag<DoubleTag>("Pos")
106                        .add(new DoubleTag("", this.x + 0.5))
107                        .add(new DoubleTag("", this.y))
108                        .add(new DoubleTag("", this.z + 0.5)))
109                .putList(new ListTag<DoubleTag>("Motion")
110                        .add(new DoubleTag("", -Math.sin(mot) * 0.02))
111                        .add(new DoubleTag("", 0.2))
112                        .add(new DoubleTag("", -Math.cos(mot) * 0.02)))
113                .putList(new ListTag<FloatTag>("Rotation")
114                        .add(new FloatTag("", 0))
115                        .add(new FloatTag("", 0)))
116                .putShort("Fuse", fuse);
117        Entity tnt = Entity.createEntity("PrimedTnt",
118                this.getLevel().getChunk(this.getFloorX() >> 4, this.getFloorZ() >> 4),
119                nbt, source
120        );
121        if(tnt == null) {
122            return;
123        }
124        tnt.spawnToAll();
125    }
126
127    @Override
128    @PowerNukkitDifference(info = "Using new method for checking if powered", since = "1.4.0.0-PN")
129    public int onUpdate(int type) {
130        if (!this.level.getServer().isRedstoneEnabled()) {
131            return 0;
132        }
133
134        if ((type == Level.BLOCK_UPDATE_NORMAL || type == Level.BLOCK_UPDATE_REDSTONE) && this.isGettingPower()) {
135            this.prime();
136        }
137
138        return 0;
139    }
140
141    @Override
142    public boolean onActivate(@Nonnull Item item, @Nullable Player player) {
143        if (item.getId() == Item.FLINT_STEEL) {
144            item.useOn(this);
145            this.prime(80, player);
146            return true;
147        } else if (item.getId() == Item.FIRE_CHARGE) {
148            if (!player.isCreative()) item.count--;
149            this.prime(80, player);
150            return true;
151        } else if (item.hasEnchantment(Enchantment.ID_FIRE_ASPECT)) {
152            item.useOn(this);
153            this.prime(80, player);
154            return true;
155        }
156        return false;
157    }
158
159    @Since("1.4.0.0-PN")
160    @PowerNukkitOnly
161    @Override
162    public boolean onProjectileHit(@Nonnull Entity projectile, @Nonnull Position position, @Nonnull Vector3 motion) {
163        if (projectile.isOnFire() && projectile instanceof EntityArrow) {
164            prime(80, projectile);
165            return true;
166        }
167        return false;
168    }
169
170    @Override
171    public BlockColor getColor() {
172        return BlockColor.TNT_BLOCK_COLOR;
173    }
