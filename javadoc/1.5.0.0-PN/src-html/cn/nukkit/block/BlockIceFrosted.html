001package cn.nukkit.block;
002
003import cn.nukkit.Player;
004import cn.nukkit.api.PowerNukkitOnly;
005import cn.nukkit.api.Since;
006import cn.nukkit.blockproperty.BlockProperties;
007import cn.nukkit.blockproperty.IntBlockProperty;
008import cn.nukkit.item.Item;
009import cn.nukkit.level.Level;
010import cn.nukkit.math.BlockFace;
011import cn.nukkit.utils.BlockColor;
012
013import javax.annotation.Nonnull;
014import java.util.concurrent.ThreadLocalRandom;
015
016public class BlockIceFrosted extends BlockTransparentMeta {
017
018    @PowerNukkitOnly
019    @Since("1.5.0.0-PN")
020    public static final IntBlockProperty AGE = new IntBlockProperty("age", false, 3);
021
022    @PowerNukkitOnly
023    @Since("1.5.0.0-PN")
024    public static final BlockProperties PROPERTIES = new BlockProperties(AGE);
025
026    public BlockIceFrosted() {
027        this(0);
028    }
029
030    public BlockIceFrosted(int meta) {
031        super(meta);
032    }
033
034    @Override
035    public int getId() {
036        return ICE_FROSTED;
037    }
038
039    @Since("1.4.0.0-PN")
040    @PowerNukkitOnly
041    @Nonnull
042    @Override
043    public BlockProperties getProperties() {
044        return PROPERTIES;
045    }
046
047    @Override
048    public String getName() {
049        return "Frosted Ice";
050    }
051
052    @Override
053    public double getResistance() {
054        return 2.5;
055    }
056
057    @Override
058    public double getHardness() {
059        return 0.5;
060    }
061
062    @Override
063    public double getFrictionFactor() {
064        return 0.98;
065    }
066
067    @Override
068    public boolean place(@Nonnull Item item, @Nonnull Block block, @Nonnull Block target, @Nonnull BlockFace face, double fx, double fy, double fz, Player player) {
069        boolean success = super.place(item, block, target, face, fx, fy, fz, player);
070        if (success) {
071            level.scheduleUpdate(this, ThreadLocalRandom.current().nextInt(20, 40));
072        }
073        return success;
074    }
075
076    @Override
077    public boolean onBreak(Item item) {
078        level.setBlock(this, get(WATER), true);
079        return true;
080    }
081
082    @Override
083    public int onUpdate(int type) {
084        if (type == Level.BLOCK_UPDATE_SCHEDULED) {
085            if (level.getBlockLightAt(getFloorX(), getFloorY(), getFloorZ()) > 11 && (ThreadLocalRandom.current().nextInt(3) == 0 || countNeighbors() < 4)) {
086                slightlyMelt(true);
087            } else {
088                level.scheduleUpdate(this, ThreadLocalRandom.current().nextInt(20, 40));
089            }
090        } else if (type == Level.BLOCK_UPDATE_NORMAL) {
091            if (countNeighbors() < 2) {
092                level.setBlock(this, layer, get(WATER), true);
093            }
094        }
095        return super.onUpdate(type);
096    }
097
098    @Override
099    public Item toItem() {
100        return Item.get(AIR);
101    }
102
103    @Override
104    public BlockColor getColor() {
105        return BlockColor.ICE_BLOCK_COLOR;
106    }
107
108    @Override
109    public boolean canHarvestWithHand() {
110        return false;
111    }
112
113    protected void slightlyMelt(boolean isSource) {
114        int age = getDamage();
115        if (age < 3) {
116            setDamage(age + 1);
117            level.setBlock(this, layer, this, true);
118            level.scheduleUpdate(level.getBlock(this), ThreadLocalRandom.current().nextInt(20, 40));
119        } else {
120            level.setBlock(this, layer, get(WATER), true);
121            if (isSource) {
122                for (BlockFace face : BlockFace.values()) {
123                    Block block = getSide(face);
124                    if (block instanceof BlockIceFrosted) {
125                        ((BlockIceFrosted) block).slightlyMelt(false);
126                    }
127                }
128            }
129        }
130    }
131
132    private int countNeighbors() {
133        int neighbors = 0;
134        for (BlockFace face : BlockFace.values()) {
135            if (getSide(face).getId() == ICE_FROSTED && ++neighbors >= 4) {
136                return neighbors;
137            }
138        }
139        return neighbors;
140    }
141}
