001package cn.nukkit.block;
002
003import cn.nukkit.Player;
004import cn.nukkit.api.PowerNukkitOnly;
005import cn.nukkit.api.Since;
006import cn.nukkit.blockproperty.BlockProperties;
007import cn.nukkit.blockproperty.IntBlockProperty;
008import cn.nukkit.item.Item;
009import cn.nukkit.item.ItemTool;
010import cn.nukkit.level.Level;
011import cn.nukkit.level.particle.DestroyBlockParticle;
012import cn.nukkit.math.BlockFace;
013
014import javax.annotation.Nonnull;
015import javax.annotation.Nullable;
016import java.util.Map;
017
018public class BlockChorusFlower extends BlockTransparent {
019
020    @PowerNukkitOnly
021    @Since("1.5.0.0-PN")
022    public static final IntBlockProperty AGE = new IntBlockProperty("age", false, 5);
023
024    @PowerNukkitOnly
025    @Since("1.5.0.0-PN")
026    public static final BlockProperties PROPERTIES = new BlockProperties(AGE);
027
028    public BlockChorusFlower() {
029    }
030
031    @Override
032    public int getId() {
033        return CHORUS_FLOWER;
034    }
035
036    @Since("1.4.0.0-PN")
037    @PowerNukkitOnly
038    @Nonnull
039    @Override
040    public BlockProperties getProperties() {
041        return PROPERTIES;
042    }
043
044    @Override
045    public String getName() {
046        return "Chorus Flower";
047    }
048
049    @Override
050    public double getHardness() {
051        return 0.4;
052    }
053
054    @Override
055    public double getResistance() {
056        return 0.4;
057    }
058
059    @Override
060    public int getToolType() {
061        return ItemTool.TYPE_NONE;
062    }
063
064    private boolean isPositionValid() {
065        // Chorus flowers must be above end stone or chorus plant, or be above air and horizontally adjacent to exactly one chorus plant.
066        // If these conditions are not met, the block breaks without dropping anything.
067        Block down = down();
068        if (down.getId() == CHORUS_PLANT || down.getId() == END_STONE) {
069            return true;
070        }
071        if (down.getId() != AIR) {
072            return false;
073        }
074        boolean foundPlant = false;
075        for (BlockFace face : BlockFace.Plane.HORIZONTAL) {
076            Block side = getSide(face);
077            if (side.getId() == CHORUS_PLANT) {
078                if (foundPlant) {
079                    return false;
080                }
081                foundPlant = true;
082            }
083        }
084
085        return foundPlant;
086    }
087
088    @Override
089    public int onUpdate(int type) {
090
091        if (type == Level.BLOCK_UPDATE_NORMAL) {
092            if (!isPositionValid()) {
093                level.scheduleUpdate(this, 1);
094                return type;
095            }
096        } else if (type == Level.BLOCK_UPDATE_SCHEDULED) {
097            Map<Integer, Player> players = level.getChunkPlayers((int) x >> 4, (int) z >> 4);
098            level.addParticle(new DestroyBlockParticle(this, this), players.values());
099            level.setBlock(this, Block.get(AIR));
100            return type;
101        }
102
103        return 0;
104    }
105
106    @Override
107    public boolean place(@Nonnull Item item, @Nonnull Block block, @Nonnull Block target, @Nonnull BlockFace face, double fx, double fy, double fz, @Nullable Player player) {
108        if (!isPositionValid()) {
109            return false;
110        }
111        return super.place(item, block, target, face, fx, fy, fz, player);
112    }
113
114    @Override
115    public Item[] getDrops(Item item) {
116        return new Item[]{this.toItem()};
117    }
118
119    @Override
120    public boolean breaksWhenMoved() {
121        return true;
122    }
123
124    @Override
125    public boolean sticksToPiston() {
126        return false;
127    }
128}
