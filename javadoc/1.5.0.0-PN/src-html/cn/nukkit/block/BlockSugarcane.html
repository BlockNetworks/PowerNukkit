020/**
021 * @author Pub4Game
022 * @since 09.01.2016
023 */
024public class BlockSugarcane extends BlockFlowable {
025
026    @PowerNukkitOnly
027    @Since("1.5.0.0-PN")
028    public static final IntBlockProperty AGE = CommonBlockProperties.AGE_15;
029
030    @PowerNukkitOnly
031    @Since("1.5.0.0-PN")
032    public static final BlockProperties PROPERTIES = new BlockProperties(AGE);
033
034    public BlockSugarcane() {
035        this(0);
036    }
037
038    public BlockSugarcane(int meta) {
039        super(meta);
040    }
041
042    @Override
043    public String getName() {
044        return "Sugarcane";
045    }
046
047    @Override
048    public int getId() {
049        return SUGARCANE_BLOCK;
050    }
051
052    @Since("1.4.0.0-PN")
053    @PowerNukkitOnly
054    @Nonnull
055    @Override
056    public BlockProperties getProperties() {
057        return PROPERTIES;
058    }
059
060    @Override
061    public Item toItem() {
062        return new ItemSugarcane();
063    }
064
065    @Override
066    public boolean canBeActivated() {
067        return true;
068    }
069
070    @Override
071    public boolean onActivate(@Nonnull Item item, Player player) {
072        if (item.isFertilizer()) { //Bonemeal
073            int count = 1;
074
075            for (int i = 1; i <= 2; i++) {
076                int id = this.level.getBlockIdAt(this.getFloorX(), this.getFloorY() - i, this.getFloorZ());
077
078                if (id == SUGARCANE_BLOCK) {
079                    count++;
080                }
081            }
082
083            if (count < 3) {
084                boolean success = false;
085                int toGrow = 3 - count;
086
087                for (int i = 1; i <= toGrow; i++) {
088                    Block block = this.up(i);
089                    if (block.getId() == 0) {
090                        BlockGrowEvent ev = new BlockGrowEvent(block, Block.get(BlockID.SUGARCANE_BLOCK));
091                        Server.getInstance().getPluginManager().callEvent(ev);
092
093                        if (!ev.isCancelled()) {
094                            this.getLevel().setBlock(block, ev.getNewState(), true);
095                            success = true;
096                        }
097                    } else if (block.getId() != SUGARCANE_BLOCK) {
098                        break;
099                    }
100                }
101
102                if (success) {
103                    if (player != null && (player.gamemode & 0x01) == 0) {
104                        item.count--;
105                    }
106
107                    this.level.addParticle(new BoneMealParticle(this));
108                }
109            }
110
111            return true;
112        }
113        return false;
114    }
115
116    @Override
117    public int onUpdate(int type) {
118        Level level = getLevel();
119        if (type == Level.BLOCK_UPDATE_NORMAL) {
120            level.scheduleUpdate(this, 0);
121            return type;
122        }
123        
124        if (type == Level.BLOCK_UPDATE_SCHEDULED) {
125            if (!isSupportValid()) {
126                level.useBreakOn(this);
127            }
128            return type;
129        }
130        
131        if (type == Level.BLOCK_UPDATE_RANDOM) {
132            if (!isSupportValid()) {
133                level.scheduleUpdate(this, 0);
134                return type;
135            }
136            if (getDamage() < 15) {
137                setDamage(this.getDamage() + 1);
138                level.setBlock(this, this, false);
139                return type;
140            }
141            Block up = up();
142            if (up.getId() != AIR) {
143                return type;
144            }
145            
146            int height = 0;
147            for (Block current = this; height < 3 && current.getId() == SUGARCANE_BLOCK; height++) {
148                current = current.down();
149            }
150            if (height >= 3) {
151                return type;
152            }
153
154            BlockGrowEvent ev = new BlockGrowEvent(up, Block.get(BlockID.SUGARCANE_BLOCK));
155            Server.getInstance().getPluginManager().callEvent(ev);
156
157            if (ev.isCancelled()) {
158                return type;
159            }
160            
161            if (!level.setBlock(up, Block.get(BlockID.SUGARCANE_BLOCK), false)) {
162                return type;
163            }
164            
165            setDamage(0);
166            level.setBlock(this, this, false);
167            return type;
168        }
169        return 0;
170    }
171
172    @Override
173    public boolean place(@Nonnull Item item, @Nonnull Block block, @Nonnull Block target, @Nonnull BlockFace face, double fx, double fy, double fz, Player player) {
174        if (block.getId() != AIR) {
175            return false;
176        }
177        if (isSupportValid()) {
178            this.getLevel().setBlock(block, this, true);
179            return true;
180        }
181        return false;
182    }
183
184    /**
185     * @since 1.2.0.2-PN
186     */
187    private boolean isSupportValid() {
188        Block down = this.down();
189        int downId = down.getId();
190        if (downId == SUGARCANE_BLOCK) {
191            return true;
192        }
193        if (downId != GRASS && downId != DIRT && downId != SAND || down.getId() == PODZOL) {
194            return false;
195        }
196        for (BlockFace face : BlockFace.Plane.HORIZONTAL) {
197            Block possibleWater = down.getSide(face);
198            if (possibleWater instanceof BlockWater
199                    || possibleWater instanceof BlockIceFrosted
200                    || possibleWater.getLevelBlockAtLayer(1) instanceof BlockWater) {
201                return true;
202            }
203        }
204        return false;
205    }
206
207    @Override
208    public BlockColor getColor() {
209        return BlockColor.FOLIAGE_BLOCK_COLOR;
210    }
211}
