019package cn.nukkit.block;
020
021import cn.nukkit.Server;
022import cn.nukkit.api.PowerNukkitOnly;
023import cn.nukkit.api.Since;
024import cn.nukkit.blockproperty.BlockProperties;
025import cn.nukkit.event.block.BlockGrowEvent;
026import cn.nukkit.item.Item;
027import cn.nukkit.level.Level;
028import cn.nukkit.math.BlockFace;
029import cn.nukkit.math.NukkitMath;
030import cn.nukkit.utils.Faceable;
031
032import javax.annotation.Nonnull;
033import java.util.concurrent.ThreadLocalRandom;
034
035import static cn.nukkit.blockproperty.CommonBlockProperties.FACING_DIRECTION;
036
037/**
038 * @author joserobjr
039 * @since 2020-09-15
040 */
041@PowerNukkitOnly
042@Since("1.4.0.0-PN")
043public abstract class BlockCropsStem extends BlockCrops implements Faceable {
044    @PowerNukkitOnly
045    @Since("1.4.0.0-PN")
046    public static final BlockProperties PROPERTIES = new BlockProperties(GROWTH, FACING_DIRECTION);
047
048    //https://minecraft.gamepedia.com/Melon_Seeds#Breaking
049    private static final double[][] dropChances = new double[][]{
050            {.8130,.1742,.0124,.0003}, //0
051            {.6510,.3004,.0462,.0024}, //1
052            {.5120,.3840,.0960,.0080}, //2
053            {.3944,.4302,.1564,.0190}, //3
054            {.2913,.4444,.2222,.0370}, //4
055            {.2160,.4320,.2880,.0640}, //5
056            {.1517,.3982,.3484,.1016}, //6
057            {.1016,.3484,.3982,.1517}  //7
058    };
059    
060    static {
061        for (double[] dropChance : dropChances) {
062            double last = dropChance[0];
063            for (int i = 1; i < dropChance.length; i++) {
064                last += dropChance[i];
065                assert last <= 1.0;
066                dropChance[i] = last;
067            }
068        }
069    }
070
071    @PowerNukkitOnly
072    @Since("1.4.0.0-PN")
073    protected BlockCropsStem(int meta) {
074        super(meta);
075    }
076
077    @Since("1.4.0.0-PN")
078    @PowerNukkitOnly
079    @Nonnull
080    @Override
081    public BlockProperties getProperties() {
082        return PROPERTIES;
083    }
084    
085    @PowerNukkitOnly
086    @Since("1.4.0.0-PN")
087    public abstract int getFruitId();
088
089    @PowerNukkitOnly
090    @Since("1.4.0.0-PN")
091    public abstract int getSeedsId();
092
093    @Override
094    public BlockFace getBlockFace() {
095        return getPropertyValue(FACING_DIRECTION);
096    }
097
098    @Since("1.3.0.0-PN")
099    @PowerNukkitOnly
100    @Override
101    public void setBlockFace(BlockFace face) {
102        setPropertyValue(FACING_DIRECTION, face);
103    }
104
105    @Override
106    public int onUpdate(int type) {
107        if (type == Level.BLOCK_UPDATE_NORMAL) {
108            if (this.down().getId() != FARMLAND) {
109                this.getLevel().useBreakOn(this);
110                return Level.BLOCK_UPDATE_NORMAL;
111            }
112            BlockFace blockFace = getBlockFace();
113            if (blockFace.getAxis().isHorizontal() && getSide(blockFace).getId() != getFruitId()) {
114                setBlockFace(null);
115                getLevel().setBlock(this, this);
116                return Level.BLOCK_UPDATE_NORMAL;
117            }
118            return 0;
119        }
120        
121        if (type != Level.BLOCK_UPDATE_RANDOM) {
122            return 0;
123        }
124        
125        if (ThreadLocalRandom.current().nextInt(1, 3) != 1 
126                || getLevel().getFullLight(this) < MINIMUM_LIGHT_LEVEL) {
127            return Level.BLOCK_UPDATE_RANDOM;
128        }
129        
130        int growth = getGrowth();
131        if (growth < GROWTH.getMaxValue()) {
132            BlockCropsStem block = this.clone();
133            block.setGrowth(growth + 1);
134            BlockGrowEvent ev = new BlockGrowEvent(this, block);
135            Server.getInstance().getPluginManager().callEvent(ev);
136            if (!ev.isCancelled()) {
137                this.getLevel().setBlock(this, ev.getNewState(), true);
138            }
139            return Level.BLOCK_UPDATE_RANDOM;
140        }
141        
142        growFruit();
143        return Level.BLOCK_UPDATE_RANDOM;
144    }
145    
146    @PowerNukkitOnly
147    @Since("1.4.0.0-PN")
148    public boolean growFruit() {
149        int fruitId = getFruitId();
150        for (BlockFace face : BlockFace.Plane.HORIZONTAL) {
151            Block b = this.getSide(face);
152            if (b.getId() == fruitId) {
153                return false;
154            }
155        }
156        
157        BlockFace sideFace = BlockFace.Plane.HORIZONTAL.random();
158        Block side = this.getSide(sideFace);
159        Block d = side.down();
160        if (side.getId() == AIR && (d.getId() == FARMLAND || d.getId() == GRASS || d.getId() == DIRT)) {
161            BlockGrowEvent ev = new BlockGrowEvent(side, Block.get(fruitId));
162            Server.getInstance().getPluginManager().callEvent(ev);
163            if (!ev.isCancelled()) {
164                this.getLevel().setBlock(side, ev.getNewState(), true);
165                setBlockFace(sideFace);
166                this.getLevel().setBlock(this, this, true);
167            }
168        }
169        return true;
170    }
171
172    @Override
173    public Item toItem() {
174        return Item.get(getSeedsId());
175    }
176
177    @Override
178    public Item[] getDrops(Item item) {
179        double[] dropChance = dropChances[NukkitMath.clamp(getGrowth(), 0, dropChances.length)];
180        
181        double dice = ThreadLocalRandom.current().nextDouble();
182        int count = 0;
183        while (dice > dropChance[count]) {
184            count++;
185        }
186        
187        if (count == 0) {
188            return Item.EMPTY_ARRAY;
189        }
190        
191        return new Item[]{
192                Item.get(getSeedsId(), 0, count)
193        };
194    }
195
196    @Override
197    public BlockCropsStem clone() {
198        return (BlockCropsStem) super.clone();
199    }
200}
