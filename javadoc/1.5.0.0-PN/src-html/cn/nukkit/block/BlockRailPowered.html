050    @Override
051    @PowerNukkitDifference(info = "Using new method for checking if powered", since = "1.4.0.0-PN")
052    public int onUpdate(int type) {
053        // Warning: I din't recommended this on slow networks server or slow client
054        //          Network below 86Kb/s. This will became unresponsive to clients 
055        //          When updating the block state. Espicially on the world with many rails. 
056        //          Trust me, I tested this on my server.
057        if (type == Level.BLOCK_UPDATE_NORMAL || type == Level.BLOCK_UPDATE_REDSTONE || type == Level.BLOCK_UPDATE_SCHEDULED) {
058            if (super.onUpdate(type) == Level.BLOCK_UPDATE_NORMAL) {
059                return 0; // Already broken
060            }
061
062            if (!this.level.getServer().isRedstoneEnabled()) {
063                return 0;
064            }
065            boolean wasPowered = isActive();
066            boolean isPowered = this.isGettingPower()
067                    || checkSurrounding(this, true, 0)
068                    || checkSurrounding(this, false, 0);
069
070            // Avoid Block mistake
071            if (wasPowered != isPowered) {
072                setActive(isPowered);
073                level.updateAround(down());
074                if (getOrientation().isAscending()) {
075                    level.updateAround(up());
076                }
077            }
078            return type;
079        }
080        return 0;
081    }

083    /**
084     * Check the surrounding of the rail
085     *
086     * @param pos      The rail position
087     * @param relative The relative of the rail that will be checked
088     * @param power    The count of the rail that had been counted
089     * @return Boolean of the surrounding area. Where the powered rail on!
090     */
091    private boolean checkSurrounding(Vector3 pos, boolean relative, int power) {
092        // The powered rail can power up to 8 blocks only
093        if (power >= 8) {
094            return false;
095        }
096        // The position of the floor numbers
097        int dx = pos.getFloorX();
098        int dy = pos.getFloorY();
099        int dz = pos.getFloorZ();
100        // First: get the base block
101        BlockRail block;
102        Block block2 = level.getBlock(new Vector3(dx, dy, dz));
103
104        // Second: check if the rail is Powered rail
105        if (Rail.isRailBlock(block2)) {
106            block = (BlockRail) block2;
107        } else {
108            return false;
109        }
110
111        // Used to check if the next ascending rail should be what
112        Rail.Orientation base = null;
113        boolean onStraight = true;
114        // Third: Recalculate the base position
115        switch (block.getOrientation()) {
116            case STRAIGHT_NORTH_SOUTH:
117                if (relative) {
118                    dz++;
119                } else {
120                    dz--;
121                }
122                break;
123            case STRAIGHT_EAST_WEST:
124                if (relative) {
125                    dx--;
126                } else {
127                    dx++;
128                }
129                break;
130            case ASCENDING_EAST:
131                if (relative) {
132                    dx--;
133                } else {
134                    dx++;
135                    dy++;
136                    onStraight = false;
137                }
138                base = Rail.Orientation.STRAIGHT_EAST_WEST;
139                break;
140            case ASCENDING_WEST:
141                if (relative) {
142                    dx--;
143                    dy++;
144                    onStraight = false;
145                } else {
146                    dx++;
147                }
148                base = Rail.Orientation.STRAIGHT_EAST_WEST;
149                break;
150            case ASCENDING_NORTH:
151                if (relative) {
152                    dz++;
153                } else {
154                    dz--;
155                    dy++;
156                    onStraight = false;
157                }
158                base = Rail.Orientation.STRAIGHT_NORTH_SOUTH;
159                break;
160            case ASCENDING_SOUTH:
161                if (relative) {
162                    dz++;
163                    dy++;
164                    onStraight = false;
165                } else {
166                    dz--;
167                }
168                base = Rail.Orientation.STRAIGHT_NORTH_SOUTH;
169                break;
170            default:
171                // Unable to determinate the rail orientation
172                // Wrong rail?
173                return false;
174        }
175        // Next check the if rail is on power state
176        return canPowered(new Vector3(dx, dy, dz), base, power, relative)
177                || onStraight && canPowered(new Vector3(dx, dy - 1., dz), base, power, relative);
178    }
179
180    @PowerNukkitDifference(info = "Using new method for checking if powered", since = "1.4.0.0-PN")
181    protected boolean canPowered(Vector3 pos, Rail.Orientation state, int power, boolean relative) {
182        Block block = level.getBlock(pos);
183        // What! My block is air??!! Impossible! XD
184        if (!(block instanceof BlockRailPowered)) {
185            return false;
186        }
187
188        // Sometimes the rails are diffrent orientation
189        Rail.Orientation base = ((BlockRailPowered) block).getOrientation();
190
191        // Possible way how to know when the rail is activated is rail were directly powered
192        // OR recheck the surrounding... Which will returns here =w=        
193        return (state != Rail.Orientation.STRAIGHT_EAST_WEST
194                || base != Rail.Orientation.STRAIGHT_NORTH_SOUTH
195                && base != Rail.Orientation.ASCENDING_NORTH
196                && base != Rail.Orientation.ASCENDING_SOUTH)
197                && (state != Rail.Orientation.STRAIGHT_NORTH_SOUTH
198                || base != Rail.Orientation.STRAIGHT_EAST_WEST
199                && base != Rail.Orientation.ASCENDING_EAST
200                && base != Rail.Orientation.ASCENDING_WEST)
201                && (this.isGettingPower() || checkSurrounding(pos, relative, power + 1));
202    }
