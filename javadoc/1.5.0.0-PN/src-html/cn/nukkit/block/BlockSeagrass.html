001package cn.nukkit.block;
002
003import cn.nukkit.Player;
004import cn.nukkit.api.PowerNukkitOnly;
005import cn.nukkit.api.Since;
006import cn.nukkit.blockproperty.ArrayBlockProperty;
007import cn.nukkit.blockproperty.BlockProperties;
008import cn.nukkit.blockproperty.value.SeaGrassType;
009import cn.nukkit.item.Item;
010import cn.nukkit.item.ItemBlock;
011import cn.nukkit.item.ItemTool;
012import cn.nukkit.level.Level;
013import cn.nukkit.level.particle.BoneMealParticle;
014import cn.nukkit.math.BlockFace;
015import cn.nukkit.utils.BlockColor;
016
017import javax.annotation.Nonnull;
018
019@PowerNukkitOnly
020public class BlockSeagrass extends BlockFlowable {
021
022    @PowerNukkitOnly
023    @Since("1.5.0.0-PN")
024    public static final ArrayBlockProperty<SeaGrassType> SEA_GRASS_TYPE = new ArrayBlockProperty<>("sea_grass_type", false, SeaGrassType.class);
025
026    @PowerNukkitOnly
027    @Since("1.5.0.0-PN")
028    public static final BlockProperties PROPERTIES = new BlockProperties(SEA_GRASS_TYPE);
029
030    @PowerNukkitOnly
031    public BlockSeagrass() {
032        this(0);
033    }
034
035    @PowerNukkitOnly
036    public BlockSeagrass(int meta) {
037        super(meta);
038    }
039    
040    @Override
041    public int getId() {
042        return SEAGRASS;
043    }
044
045    @Since("1.4.0.0-PN")
046    @PowerNukkitOnly
047    @Nonnull
048    @Override
049    public BlockProperties getProperties() {
050        return PROPERTIES;
051    }
052
053    @Override
054    public String getName() {
055        return "Seagrass";
056    }
057    
058    @Override
059    public boolean place(@Nonnull Item item, @Nonnull Block block, @Nonnull Block target, @Nonnull BlockFace face, double fx, double fy, double fz, Player player) {
060        Block down = down();
061        Block layer1Block = block.getLevelBlockAtLayer(1);
062        int waterDamage;
063        if (down.isSolid() && down.getId() != MAGMA && down.getId() != SOUL_SAND &&
064                (layer1Block instanceof BlockWater && ((waterDamage = (block.getDamage())) == 0 || waterDamage == 8))
065        ) {
066            if (waterDamage == 8) {
067                this.getLevel().setBlock(this, 1, new BlockWater(), true, false);
068            }
069            this.getLevel().setBlock(this, 0, this, true, true);
070            return true;
071        }
072        return false;
073    }
074    
075    @Override
076    public int onUpdate(int type) {
077        if (type == Level.BLOCK_UPDATE_NORMAL) {
078            Block blockLayer1 = getLevelBlockAtLayer(1);
079            int damage;
080            if (!(blockLayer1 instanceof BlockIceFrosted)
081                    && (!(blockLayer1 instanceof BlockWater) || ((damage = blockLayer1.getDamage()) != 0 && damage != 8))) {
082                this.getLevel().useBreakOn(this);
083                return Level.BLOCK_UPDATE_NORMAL;
084            }
085            
086            Block down = down();
087            damage = getDamage();
088            if (damage == 0 || damage == 2) {
089                if (!down.isSolid() || down.getId() == MAGMA || down.getId() == SOUL_SAND) {
090                    this.getLevel().useBreakOn(this);
091                    return Level.BLOCK_UPDATE_NORMAL;
092                }
093                
094                if (damage == 2) {
095                    Block up = up();
096                    if (up.getId() != getId() || up.getDamage() != 1) {
097                        this.getLevel().useBreakOn(this);
098                    }
099                }
100            } else if (down.getId() != getId() || down.getDamage() != 2) {
101                this.getLevel().useBreakOn(this);
102            }
103            
104            return Level.BLOCK_UPDATE_NORMAL;
105        }
106        
107        return 0;
108    }
109    
110    @Override
111    public boolean canBeActivated() {
112        return true;
113    }
114    
115    @Override
116    public boolean onActivate(@Nonnull Item item, Player player) {
117        if (getDamage() == 0 && item.isFertilizer()) {
118            Block up = this.up();
119            int damage;
120            if (up instanceof BlockWater && ((damage = up.getDamage()) == 0 || damage == 8)) {
121                if (player != null && (player.gamemode & 0x01) == 0) {
122                    item.count--;
123                }
124
125                this.level.addParticle(new BoneMealParticle(this));
126                this.level.setBlock(this, new BlockSeagrass(2), true, false);
127                this.level.setBlock(up, 1, up, true, false);
128                this.level.setBlock(up, 0, new BlockSeagrass(1), true);
129                return true;
130            }
131        }
132        
133        return false;
134    }
135    
136    @Override
137    public Item[] getDrops(Item item) {
138        if (item.isShears()) {
139            return new Item[] { toItem() };
140        } else {
141            return Item.EMPTY_ARRAY;
142        }
143    }
144    
145    @Override
146    public boolean canBeReplaced() {
147        return true;
148    }
149    
150    @PowerNukkitOnly
151    @Override
152    public int getWaterloggingLevel() {
153        return 2;
154    }
155    
156    @Override
157    public BlockColor getColor() {
158        return BlockColor.WATER_BLOCK_COLOR;
159    }
160    
161    @Override
162    public int getToolType() {
163        return ItemTool.TYPE_SHEARS;
164    }
165    
166    @Override
167    public Item toItem() {
168        return new ItemBlock(new BlockSeagrass(), 0);
169    }
170}
