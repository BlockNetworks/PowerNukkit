001package cn.nukkit.block;
002
003import cn.nukkit.Player;
004import cn.nukkit.api.PowerNukkitOnly;
005import cn.nukkit.api.Since;
006import cn.nukkit.blockproperty.BlockProperties;
007import cn.nukkit.item.Item;
008import cn.nukkit.item.ItemTool;
009import cn.nukkit.math.BlockFace;
010import cn.nukkit.utils.BlockColor;
011
012import javax.annotation.Nonnull;
013
014import static cn.nukkit.blockproperty.CommonBlockProperties.PILLAR_AXIS;
015
016@PowerNukkitOnly
017@Since("1.4.0.0-PN")
018public class BlockBasalt extends BlockSolidMeta {
019
020    @PowerNukkitOnly
021    @Since("1.4.0.0-PN")
022    public static final BlockProperties PROPERTIES = new BlockProperties(PILLAR_AXIS);
023
024    @PowerNukkitOnly
025    @Since("1.4.0.0-PN")
026    public BlockBasalt() { this(0); }
027
028    @PowerNukkitOnly
029    @Since("1.4.0.0-PN")
030    public BlockBasalt(int meta) {
031        super(meta);
032    }
033
034    @Override
035    public String getName() {
036        return "Basalt";
037    }
038
039    @Override
040    public int getId() {
041        return BlockID.BASALT;
042    }
043
044    @Since("1.4.0.0-PN")
045    @PowerNukkitOnly
046    @Nonnull
047    @Override
048    public BlockProperties getProperties() {
049        return PROPERTIES;
050    }
051
052    @Override
053    public double getHardness() {
054        return 1.25;
055    }
056
057    @Override
058    public double getResistance() {
059        return 4.2;
060    }
061
062    @Override
063    public int getToolType() {
064        return ItemTool.TYPE_PICKAXE;
065    }
066
067    @Override
068    public int getToolTier() {
069        return ItemTool.TIER_WOODEN;
070    }
071
072    @PowerNukkitOnly
073    @Since("1.4.0.0-PN")
074    public BlockFace.Axis getPillarAxis() {
075        return getPropertyValue(PILLAR_AXIS);
076    }
077
078    @PowerNukkitOnly
079    @Since("1.4.0.0-PN")
080    public void setPillarAxis(BlockFace.Axis axis) {
081        setPropertyValue(PILLAR_AXIS, axis);
082    }
083
084    @Override
085    public boolean place(@Nonnull Item item, @Nonnull Block block, @Nonnull Block target, @Nonnull BlockFace face, double fx, double fy, double fz, Player player) {
086        setPillarAxis(face.getAxis());
087        getLevel().setBlock(block, this, true, true);
088        return true;
089    }
090
091    @Override
092    public BlockColor getColor() {
093        return BlockColor.GRAY_BLOCK_COLOR;
094    }
095}
