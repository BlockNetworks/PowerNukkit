102    @PowerNukkitDifference(since = "1.4.0.0-PN", info = "Renders a bounding box that the entities stands on top")

105    protected AxisAlignedBB recalculateBoundingBox() {
106        int snowHeight = getSnowHeight();
107        if (snowHeight < 3) {
108            return null;
109        }
110        if (snowHeight == 3 || snowHeight == SNOW_HEIGHT.getMaxValue()) {
111            return this;
112        }
113        return new SimpleAxisAlignedBB(x, y, z, x + 1, y + 8/16.0, z + 1);
114    }

144    @PowerNukkitDifference(since = "1.4.0.0-PN", info = "Will increase the layers and behave as expected in vanilla and will cover grass blocks")
145    @Override
146    public boolean place(@Nonnull Item item, @Nonnull Block block, @Nonnull Block target, @Nonnull BlockFace face, double fx, double fy, double fz, Player player) {
147        Optional<BlockSnowLayer> increment = Stream.of(target, block)
148                .filter(b -> b.getId() == SNOW_LAYER).map(BlockSnowLayer.class::cast)
149                .filter(b -> b.getSnowHeight() < SNOW_HEIGHT.getMaxValue())
150                .findFirst();
151        
152        if (increment.isPresent()) {
153            BlockSnowLayer other = increment.get();
154            if (Arrays.stream(level.getCollidingEntities(new SimpleAxisAlignedBB(
155                    other.x, other.y, other.z, 
156                    other.x + 1, other.y + 1, other.z + 1
157                    ))).anyMatch(e-> e instanceof EntityLiving)) {
158                return false;
159            }
160            other.setSnowHeight(other.getSnowHeight() + 1);
161            return level.setBlock(other, other, true);
162        }
163
164        Block down = down();
165        if (!down.isSolid()) {
166            return false;
167        }
168        
169        switch (down.getId()) {
170            case BARRIER:
171            case STRUCTURE_VOID:
172                return false;
173            case GRASS:
174                setCovered(true);
175                break;
176            case TALL_GRASS:
177                if (!level.setBlock(this, 0, this, true)) {
178                    return false;
179                }
180                level.setBlock(block, 1, block, true, false);
181                return true;
182            default:
183        }
184        
185        return this.getLevel().setBlock(block, this, true);
186    }

198    @Since("1.2.1.0-PN")
199    @PowerNukkitOnly
200    @Override
201    public void afterRemoval(Block newBlock, boolean update) {
202        if (layer != 0 || newBlock.getId() == getId()) {
203            return;
204        }
205
206        Block layer1 = getLevelBlockAtLayer(1);
207        if (layer1.getId() != TALL_GRASS) {
208            return;
209        }
210        
211        // Clear the layer1 block and do a small hack as workaround a vanilla client rendering bug
212        Level level = getLevel();
213        level.setBlock(this, 0, layer1, true, false);
214        level.setBlock(this, 1, get(AIR), true, false);
215        level.setBlock(this, 0, newBlock, true, false);
216        Server.getInstance().getScheduler().scheduleDelayedTask(()-> {
217            Player[] target = level.getChunkPlayers(getChunkX(), getChunkZ()).values().toArray(Player.EMPTY_ARRAY);
218            Vector3[] blocks = {getLocation()};
219            level.sendBlocks(target, blocks, UpdateBlockPacket.FLAG_ALL_PRIORITY, 0, false);
220            level.sendBlocks(target, blocks, UpdateBlockPacket.FLAG_ALL_PRIORITY, 1, false);
221        }, 10);
222        
223        Player[] target = level.getChunkPlayers(getChunkX(), getChunkZ()).values().toArray(Player.EMPTY_ARRAY);
224        Vector3[] blocks = {getLocation()};
225        level.sendBlocks(target, blocks, UpdateBlockPacket.FLAG_ALL_PRIORITY, 0, false);
226        level.sendBlocks(target, blocks, UpdateBlockPacket.FLAG_ALL_PRIORITY, 1, false);
227    }

229    @PowerNukkitDifference(since = "1.4.0.0-PN", info = "Will melt on dry biomes and will melt gradually and will cover grass blocks")
230    @Override
231    public int onUpdate(int type) {
232        super.onUpdate(type);
233        if (type == Level.BLOCK_UPDATE_RANDOM) {
234            Biome biome = Biome.getBiome(getLevel().getBiomeId(getFloorX(), getFloorZ()));
235            if (biome.isDry() || this.getLevel().getBlockLightAt(getFloorX(), getFloorY(), getFloorZ()) >= 10) {
236                melt();
237                return Level.BLOCK_UPDATE_RANDOM;
238            }
239        } else if (type == Level.BLOCK_UPDATE_NORMAL) {
240            boolean covered = down().getId() == GRASS;
241            if (isCovered() != covered) {
242                setCovered(covered);
243                level.setBlock(this, this, true);
244                return type;
245            }
246        }
247        return 0;
248    }

256    @PowerNukkitOnly
257    @Since("1.4.0.0-PN")
258    public boolean melt(int layers) {
259        Preconditions.checkArgument(layers > 0, "Layers must be positive, got {}", layers);
260        Block toMelt = this;
261        while (toMelt.getIntValue(SNOW_HEIGHT) == SNOW_HEIGHT.getMaxValue()) {
262            Block up = toMelt.up();
263            if (up.getId() != SNOW_LAYER) {
264                break;
265            }
266            
267            toMelt = up;
268        }
269        
270        int snowHeight = toMelt.getIntValue(SNOW_HEIGHT) - layers;
271        Block newState = snowHeight < 0? get(AIR) : getCurrentState().withProperty(SNOW_HEIGHT, snowHeight).getBlock(toMelt);
272        BlockFadeEvent event = new BlockFadeEvent(toMelt, newState);
273        level.getServer().getPluginManager().callEvent(event);
274        if (event.isCancelled()) {
275            return false;
276        }
277        
278        return level.setBlock(toMelt, event.getNewState(), true);
279    }

287    @PowerNukkitDifference(since = "1.4.0.0-PN", info = "Fixed the amount of snowballs that are dropped")
288    @Override
289    public Item[] getDrops(Item item) {
290        if (!item.isShovel() || item.getTier() < ItemTool.TIER_WOODEN) {
291            return Item.EMPTY_ARRAY;
292        }
293        
294        int amount;
295        switch (getSnowHeight()) {
296            case 0:
297            case 1:
298            case 2:
299                amount = 1;
300                break;
301            case 3:
302            case 4:
303                amount = 2;
304                break;
305            case 5:
306            case 6:
307                amount = 3;
308                break;
309            default:
310            case 7:
311                amount = 4;
312        }
313        return new Item[]{ Item.get(ItemID.SNOWBALL, 0, amount) };
314    }
