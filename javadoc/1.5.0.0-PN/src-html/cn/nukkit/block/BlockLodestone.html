066    @Override
067    public boolean onActivate(@Nonnull Item item, @Nullable Player player) {
068        if (player == null || item.isNull() || item.getId() != ItemID.COMPASS && item.getId() != ItemID.LODESTONE_COMPASS) {
069            return false;
070        }
071        ItemCompassLodestone compass = (ItemCompassLodestone) Item.get(ItemID.LODESTONE_COMPASS);
072        if (item.hasCompoundTag()) {
073            compass.setCompoundTag(item.getCompoundTag().clone());
074        }
075        int trackingHandle;
076        try {
077            trackingHandle = getOrCreateBlockEntity().requestTrackingHandler();
078            compass.setTrackingHandle(trackingHandle);
079        } catch (Exception e) {
080            log.warn("Could not create a lodestone compass to {} for {}", getLocation(), player.getName(), e);
081            return false;
082        }
083        boolean added = true;
084        if (item.getCount() == 1) {
085            player.getInventory().setItemInHand(compass);
086        } else {
087            Item clone = item.clone();
088            clone.count--;
089            player.getInventory().setItemInHand(clone);
090            for (Item failed : player.getInventory().addItem(compass)) {
091                added = false;
092                player.getLevel().dropItem(player.getPosition(), failed);
093            }
094        }
095        getLevel().addSound(player.getPosition(), Sound.LODESTONE_COMPASS_LINK_COMPASS_TO_LODESTONE);
096        if (added) {
097            try {
098                getLevel().getServer().getPositionTrackingService().startTracking(player, trackingHandle, false);
099            } catch (IOException e) {
100                log.warn("Failed to make the player {} track {} at {}", player.getName(), trackingHandle, getLocation(), e);
101            }
102            getLevel().getServer().getScheduler().scheduleTask(null, player::updateTrackingPositions);
103        }
104        return true;
105    }
