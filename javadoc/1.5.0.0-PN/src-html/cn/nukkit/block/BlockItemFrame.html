036@PowerNukkitDifference(since = "1.4.0.0-PN", info = "Implements BlockEntityHolder and Faceable only in PowerNukkit")
037public class BlockItemFrame extends BlockTransparentMeta implements BlockEntityHolder<BlockEntityItemFrame>, Faceable {
038    @PowerNukkitOnly
039    @Since("1.4.0.0-PN")
040    public static final BooleanBlockProperty HAS_MAP = new BooleanBlockProperty("item_frame_map_bit", false);

047        this(0);
048    }
049

051        super(meta);
052    }
053

108    }
109

111    public String getName() {
112        return "Item Frame";
113    }
114
115    @PowerNukkitDifference(info = "Allow to stay in walls", since = "1.3.0.0-PN")
116    @Override
117    public int onUpdate(int type) {
118        if (type == Level.BLOCK_UPDATE_NORMAL) {
119            Block support = this.getSideAtLayer(0, getFacing().getOpposite());
120            if (!support.isSolid() && support.getId() != COBBLE_WALL) {
121                this.level.useBreakOn(this);
122                return type;
123            }
124        }
125
126        return 0;
127    }
128
129    @Override
130    public boolean canBeActivated() {
131        return true;
132    }
133
134    @PowerNukkitOnly
135    @Override
136    public int getWaterloggingLevel() {
137        return 1;
138    }
139
140    @Since("1.4.0.0-PN")
141    @PowerNukkitOnly
142    @Override
143    public int onTouch(@Nullable Player player, Action action) {
144        if (action == Action.LEFT_CLICK_BLOCK && (player == null || (!player.isCreative() && !player.isSpectator()))) {
145            getOrCreateBlockEntity().dropItem(player);
146        }
147        return super.onTouch(player, action);
148    }
149
150    @Override
151    public boolean onActivate(@Nonnull Item item, Player player) {
152        BlockEntityItemFrame itemFrame = getOrCreateBlockEntity();
153        if (itemFrame.getItem().isNull()) {
154                Item itemOnFrame = item.clone();
155                if (player != null && !player.isCreative()) {
156                        itemOnFrame.setCount(itemOnFrame.getCount() - 1);
157                player.getInventory().setItemInHand(itemOnFrame);
158                }
159            itemOnFrame.setCount(1);
160            itemFrame.setItem(itemOnFrame);
161            if (itemOnFrame.getId() == ItemID.MAP) {
162                setStoringMap(true);
163                this.getLevel().setBlock(this, this, true);
164            }
165            this.getLevel().addLevelEvent(this, LevelEventPacket.EVENT_SOUND_ITEM_FRAME_ITEM_ADDED);
166        } else {
167            itemFrame.setItemRotation((itemFrame.getItemRotation() + 1) % 8);
168            if (isStoringMap()) {
169                setStoringMap(false);
170                this.getLevel().setBlock(this, this, true);
171            }
172            this.getLevel().addLevelEvent(this, LevelEventPacket.EVENT_SOUND_ITEM_FRAME_ITEM_ROTATED);
173        }
174        return true;
175    }
176
177    @PowerNukkitDifference(info = "Allow to place on walls", since = "1.3.0.0-PN")
178    @Override
179    public boolean place(@Nonnull Item item, @Nonnull Block block, @Nonnull Block target, @Nonnull BlockFace face, double fx, double fy, double fz, @Nullable Player player) {
180        if (target.getId() != COBBLE_WALL && (!target.isSolid() || (block.isSolid() && !block.canBeReplaced()))) {
181            return false;
182        }
183
184        setBlockFace(face);
185        setStoringMap(item.getId() == ItemID.MAP);
186        CompoundTag nbt = new CompoundTag()
187                .putByte("ItemRotation", 0)
188                .putFloat("ItemDropChance", 1.0f);
189        if (item.hasCustomBlockData()) {
190            for (Tag aTag : item.getCustomBlockData().getAllTags()) {
191                nbt.put(aTag.getName(), aTag);
192            }
193        }
194        BlockEntityItemFrame frame = BlockEntityHolder.setBlockAndCreateEntity(this, true, true, nbt);
195        if (frame == null) {
196            return false;
197        }
198
199        this.getLevel().addSound(this, Sound.BLOCK_ITEMFRAME_PLACE);
200        return true;
201    }
202
203    @Override
204    public boolean onBreak(Item item) {
205        this.getLevel().setBlock(this, layer, Block.get(BlockID.AIR), true, true);
206        this.getLevel().addLevelEvent(this, LevelEventPacket.EVENT_SOUND_ITEM_FRAME_REMOVED);
207        return true;
208    }
209
210    @Override
211    public Item[] getDrops(Item item) {
212        BlockEntityItemFrame itemFrame = getBlockEntity();
213        if (itemFrame != null && ThreadLocalRandom.current().nextFloat() <= itemFrame.getItemDropChance()) {
214            return new Item[]{
215                    toItem(), itemFrame.getItem().clone()
216            };
217        } else {
218            return new Item[]{
219                    toItem()
220            };
221        }
222    }

269    @PowerNukkitOnly("Will calculate the correct AABB")
270    @Since("1.3.0.0-PN")
271    @Override
272    protected AxisAlignedBB recalculateBoundingBox() {
273        double[][] aabb = {
274                {2.0/16, 14.0/16},
275                {2.0/16, 14.0/16},
276                {2.0/16, 14.0/16}
277        };
278
279        BlockFace facing = getFacing();
280        if (facing.getAxisDirection() == POSITIVE) {
281            int axis = facing.getAxis().ordinal();
282            aabb[axis][0] = 0;
283            aabb[axis][1] = 1.0/16;
284        }
285
286        return new SimpleAxisAlignedBB(
287                aabb[0][0] + x, aabb[1][0] + y, aabb[2][0] + z,
288                aabb[0][1] + x, aabb[1][1] + y, aabb[2][1] + z
289        );
290    }
