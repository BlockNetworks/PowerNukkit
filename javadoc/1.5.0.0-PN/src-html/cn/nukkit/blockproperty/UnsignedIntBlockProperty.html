001package cn.nukkit.blockproperty;
002
003import cn.nukkit.api.PowerNukkitOnly;
004import cn.nukkit.api.Since;
005import cn.nukkit.blockproperty.exception.InvalidBlockPropertyMetaException;
006import cn.nukkit.blockproperty.exception.InvalidBlockPropertyPersistenceValueException;
007import cn.nukkit.blockproperty.exception.InvalidBlockPropertyValueException;
008import cn.nukkit.math.NukkitMath;
009import com.google.common.base.Preconditions;
010
011import javax.annotation.Nonnull;
012import javax.annotation.Nullable;
013
014@PowerNukkitOnly
015@Since("1.4.0.0-PN")
016public class UnsignedIntBlockProperty extends BlockProperty<Integer> {
017    private static final long serialVersionUID = 7896101036099245755L;
018    
019    private final long minValue;
020    private final long maxValue;
021    
022    @PowerNukkitOnly
023    @Since("1.4.0.0-PN")
024    public UnsignedIntBlockProperty(String name, boolean exportedToItem, int maxValue, int minValue, int bitSize, String persistenceName) {
025        super(name, exportedToItem, bitSize, persistenceName);
026        long unsignedMinValue = removeSign(minValue);
027        long unsignedMaxValue = removeSign(maxValue);
028        long delta = unsignedMaxValue - unsignedMinValue;
029        Preconditions.checkArgument(delta > 0, "maxValue must be higher than minValue. Got min:%s and max:%s", unsignedMinValue, unsignedMaxValue);
030        
031        long mask = removeSign(-1 >>> (32 - bitSize));
032        Preconditions.checkArgument(delta <= mask, "The data range from %s to %s can't be stored in %s bits", unsignedMinValue, unsignedMaxValue, bitSize);
033        
034        this.minValue = unsignedMinValue;
035        this.maxValue = unsignedMaxValue;
036    }
037
038    @PowerNukkitOnly
039    @Since("1.4.0.0-PN")
040    public UnsignedIntBlockProperty(String name, boolean exportedToItem, int maxValue, int minValue, int bitSize) {
041        this(name, exportedToItem, maxValue, minValue, bitSize, name);
042    }
043
044    @PowerNukkitOnly
045    @Since("1.4.0.0-PN")
046    public UnsignedIntBlockProperty(String name, boolean exportedToItem, int maxValue, int minValue) {
047        this(name, exportedToItem, maxValue, minValue, NukkitMath.bitLength(maxValue - minValue));
048    }
049
050    @PowerNukkitOnly
051    @Since("1.4.0.0-PN")
052    public UnsignedIntBlockProperty(String name, boolean exportedToItem, int maxValue) {
053        this(name, exportedToItem, maxValue, 0);
054    }
055
056    @Since("1.4.0.0-PN")
057    @PowerNukkitOnly
058    @Override
059    public UnsignedIntBlockProperty copy() {
060        return new UnsignedIntBlockProperty(getName(), isExportedToItem(), (int)getMaxValue(), (int)getMinValue(), getBitSize(), getPersistenceName());
061    }
062
063    @Since("1.4.0.0-PN")
064    @PowerNukkitOnly
065    @Override
066    public UnsignedIntBlockProperty exportingToItems(boolean exportedToItem) {
067        return new UnsignedIntBlockProperty(getName(), exportedToItem, (int)getMaxValue(), (int)getMinValue(), getBitSize(), getPersistenceName());
068    }
069
070    private static long removeSign(int value) {
071        return (long)value & 0xFFFFFFFFL;
072    }
073    
074    private static int addSign(long value) {
075        return (int)(value & 0xFFFFFFFFL);
076    }
077
078    @Override
079    public int getMetaForValue(@Nullable Integer value) {
080        if (value == null) {
081            return 0;
082        }
083        long unsigned = removeSign(value);
084        try {
085            validateDirectly(unsigned);
086        } catch (IllegalArgumentException e) {
087            throw new InvalidBlockPropertyValueException(this, null, value, e);
088        }
089        return (int) (unsigned - minValue);
090    }
091
092    @Nonnull
093    @Override
094    public Integer getValueForMeta(int meta) {
095        return getIntValueForMeta(meta);
096    }
097
098    @Override
099    public int getIntValueForMeta(int meta) {
100        try {
101            validateMetaDirectly(meta);
102        } catch (IllegalArgumentException e) {
103            throw new InvalidBlockPropertyMetaException(this, meta, meta, e);
104        }
105        return (int) (minValue + meta);
106    }
107
108    @Override
109    public String getPersistenceValueForMeta(int meta) {
110        return String.valueOf(removeSign(getIntValueForMeta(meta)));
111    }
112
113    @Since("1.4.0.0-PN")
114    @PowerNukkitOnly
115    @Override
116    public int getMetaForPersistenceValue(@Nonnull String persistenceValue) {
117        try {
118            return getMetaForValue(addSign(Long.parseLong(persistenceValue)));
119        } catch (NumberFormatException | InvalidBlockPropertyValueException e) {
120            throw new InvalidBlockPropertyPersistenceValueException(this, null, persistenceValue, e);
121        }
122    }
123
124    @Override
125    protected void validateDirectly(@Nullable Integer value) {
126        if (value == null) {
127            return;
128        }
129        validateDirectly(removeSign(value));
130    }
131
132    /**
133     * @throws RuntimeException Any runtime exception to indicate an invalid value
134     */
135    private void validateDirectly(long unsigned) {
136        Preconditions.checkArgument(unsigned >= minValue, "New value (%s) must be higher or equals to %s", unsigned, minValue);
137        Preconditions.checkArgument(maxValue >= unsigned, "New value (%s) must be less or equals to %s", unsigned, maxValue);
138    }
139
140    @Override
141    protected void validateMetaDirectly(int meta) {
142        long max = maxValue - minValue;
143        Preconditions.checkArgument(0 <= meta && meta <= max, "The meta %s is outside the range of 0 .. ", meta, max);
144    }
145
146    @Nonnull
147    @Override
148    public Class<Integer> getValueClass() {
149        return Integer.class;
150    }
151
152    @PowerNukkitOnly
153    @Since("1.4.0.0-PN")
154    public long getMaxValue() {
155        return maxValue;
156    }
157
158    @PowerNukkitOnly
159    @Since("1.4.0.0-PN")
160    public long getMinValue() {
161        return minValue;
162    }
163
164    @Nonnull
165    @PowerNukkitOnly
166    @Since("1.4.0.0-PN")
167    public Integer getDefaultValue() {
168        return (int)minValue;
169    }
170
171    @Since("1.4.0.0-PN")
172    @PowerNukkitOnly
173    @Override
174    public boolean isDefaultValue(@Nullable Integer value) {
175        return value == null || removeSign(value)==minValue;
176    }
177
178    @Since("1.4.0.0-PN")
179    @PowerNukkitOnly
180    @Override
181    public boolean isDefaultIntValue(int value) {
182        return removeSign(value)==minValue;
183    }
184
185    @Since("1.4.0.0-PN")
186    @PowerNukkitOnly
187    @Override
188    public int getDefaultIntValue() {
189        return (int)minValue;
190    }
191}
