001package cn.nukkit.blockproperty.exception;
002
003import cn.nukkit.api.PowerNukkitOnly;
004import cn.nukkit.api.Since;
005import cn.nukkit.blockproperty.BlockProperties;
006
007import java.util.NoSuchElementException;
008
009@PowerNukkitOnly
010@Since("1.5.0.0-PN")
011public class BlockPropertyNotFoundException extends NoSuchElementException {
012    private final String propertyName;
013
014    @PowerNukkitOnly
015    @Since("1.5.0.0-PN")
016    public BlockPropertyNotFoundException(String propertyName) {
017        super("The property \""+propertyName+"\" was not found.");
018        this.propertyName = propertyName;
019    }
020
021    @PowerNukkitOnly
022    @Since("1.5.0.0-PN")
023    public BlockPropertyNotFoundException(String propertyName, String details) {
024        super(propertyName+": " + details);
025        this.propertyName = propertyName;
026    }
027
028    @PowerNukkitOnly
029    @Since("1.5.0.0-PN")
030    public BlockPropertyNotFoundException(String propertyName, BlockProperties properties) {
031        super("The property \""+propertyName+"\" was not found. Valid properties: " + properties.getNames());
032        this.propertyName = propertyName;
033    }
034
035    @PowerNukkitOnly
036    @Since("1.5.0.0-PN")
037    public String getPropertyName() {
038        return propertyName;
039    }
040}
