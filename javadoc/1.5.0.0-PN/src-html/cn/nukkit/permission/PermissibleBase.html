001package cn.nukkit.permission;
002
003import cn.nukkit.Server;
004import cn.nukkit.plugin.Plugin;
005import cn.nukkit.utils.PluginException;
006import cn.nukkit.utils.ServerException;
007import co.aikar.timings.Timings;
008
009import java.util.HashMap;
010import java.util.HashSet;
011import java.util.Map;
012import java.util.Set;
013
014/**
015 * @author MagicDroidX (Nukkit Project)
016 */
017public class PermissibleBase implements Permissible {
018
019    ServerOperator opable = null;
020
021    private Permissible parent = null;
022
023    private final Set<PermissionAttachment> attachments = new HashSet<>();
024
025    private final Map<String, PermissionAttachmentInfo> permissions = new HashMap<>();
026
027    public PermissibleBase(ServerOperator opable) {
028        this.opable = opable;
029        if (opable instanceof Permissible) {
030            this.parent = (Permissible) opable;
031        }
032    }
033
034    @Override
035    public boolean isOp() {
036        return this.opable != null && this.opable.isOp();
037    }
038
039    @Override
040    public void setOp(boolean value) {
041        if (this.opable == null) {
042            throw new ServerException("Cannot change op value as no ServerOperator is set");
043        } else {
044            this.opable.setOp(value);
045        }
046    }
047
048    @Override
049    public boolean isPermissionSet(String name) {
050        return this.permissions.containsKey(name);
051    }
052
053    @Override
054    public boolean isPermissionSet(Permission permission) {
055        return this.isPermissionSet(permission.getName());
056    }
057
058    @Override
059    public boolean hasPermission(String name) {
060        if (this.isPermissionSet(name)) {
061            return this.permissions.get(name).getValue();
062        }
063
064        Permission perm = Server.getInstance().getPluginManager().getPermission(name);
065
066        if (perm != null) {
067            String permission = perm.getDefault();
068
069            return Permission.DEFAULT_TRUE.equals(permission) || (this.isOp() && Permission.DEFAULT_OP.equals(permission)) || (!this.isOp() && Permission.DEFAULT_NOT_OP.equals(permission));
070        } else {
071            return Permission.DEFAULT_TRUE.equals(Permission.DEFAULT_PERMISSION) || (this.isOp() && Permission.DEFAULT_OP.equals(Permission.DEFAULT_PERMISSION)) || (!this.isOp() && Permission.DEFAULT_NOT_OP.equals(Permission.DEFAULT_PERMISSION));
072        }
073    }
074
075    @Override
076    public boolean hasPermission(Permission permission) {
077        return this.hasPermission(permission.getName());
078    }
079
080    @Override
081    public PermissionAttachment addAttachment(Plugin plugin) {
082        return this.addAttachment(plugin, null, null);
083    }
084
085    @Override
086    public PermissionAttachment addAttachment(Plugin plugin, String name) {
087        return this.addAttachment(plugin, name, null);
088    }
089
090    @Override
091    public PermissionAttachment addAttachment(Plugin plugin, String name, Boolean value) {
092        if (!plugin.isEnabled()) {
093            throw new PluginException("Plugin " + plugin.getDescription().getName() + " is disabled");
094        }
095
096        PermissionAttachment result = new PermissionAttachment(plugin, this.parent != null ? this.parent : this);
097        this.attachments.add(result);
098        if (name != null && value != null) {
099            result.setPermission(name, value);
100        }
101        this.recalculatePermissions();
102
103        return result;
104    }
105
106    @Override
107    public void removeAttachment(PermissionAttachment attachment) {
108        if (this.attachments.contains(attachment)) {
109            this.attachments.remove(attachment);
110            PermissionRemovedExecutor ex = attachment.getRemovalCallback();
111            if (ex != null) {
112                ex.attachmentRemoved(attachment);
113            }
114            this.recalculatePermissions();
115        }
116    }
117
118    @Override
119    public void recalculatePermissions() {
120        Timings.permissibleCalculationTimer.startTiming();
121
122        this.clearPermissions();
123        Map<String, Permission> defaults = Server.getInstance().getPluginManager().getDefaultPermissions(this.isOp());
124        Server.getInstance().getPluginManager().subscribeToDefaultPerms(this.isOp(), this.parent != null ? this.parent : this);
125
126        for (Permission perm : defaults.values()) {
127            String name = perm.getName();
128            this.permissions.put(name, new PermissionAttachmentInfo(this.parent != null ? this.parent : this, name, null, true));
129            Server.getInstance().getPluginManager().subscribeToPermission(name, this.parent != null ? this.parent : this);
130            this.calculateChildPermissions(perm.getChildren(), false, null);
131        }
132
133        for (PermissionAttachment attachment : this.attachments) {
134            this.calculateChildPermissions(attachment.getPermissions(), false, attachment);
135        }
136        Timings.permissibleCalculationTimer.stopTiming();
137    }
138
139    public void clearPermissions() {
140        for (String name : this.permissions.keySet()) {
141            Server.getInstance().getPluginManager().unsubscribeFromPermission(name, this.parent != null ? this.parent : this);
142        }
143
144
145        Server.getInstance().getPluginManager().unsubscribeFromDefaultPerms(false, this.parent != null ? this.parent : this);
146        Server.getInstance().getPluginManager().unsubscribeFromDefaultPerms(true, this.parent != null ? this.parent : this);
147
148        this.permissions.clear();
149    }
150
151    private void calculateChildPermissions(Map<String, Boolean> children, boolean invert, PermissionAttachment attachment) {
152        for (Map.Entry<String, Boolean> entry : children.entrySet()) {
153            String name = entry.getKey();
154            Permission perm = Server.getInstance().getPluginManager().getPermission(name);
155            boolean v = entry.getValue();
156            boolean value = (v ^ invert);
157            this.permissions.put(name, new PermissionAttachmentInfo(this.parent != null ? this.parent : this, name, attachment, value));
158            Server.getInstance().getPluginManager().subscribeToPermission(name, this.parent != null ? this.parent : this);
159
160            if (perm != null) {
161                this.calculateChildPermissions(perm.getChildren(), !value, attachment);
162            }
163        }
164    }
165
166    @Override
167    public Map<String, PermissionAttachmentInfo> getEffectivePermissions() {
168        return this.permissions;
169    }
170}
