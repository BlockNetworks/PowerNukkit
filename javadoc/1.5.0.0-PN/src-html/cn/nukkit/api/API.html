001package cn.nukkit.api;
002
003import java.lang.annotation.ElementType;
004import java.lang.annotation.Retention;
005import java.lang.annotation.RetentionPolicy;
006import java.lang.annotation.Target;
007
008import static cn.nukkit.api.API.Definition.UNIVERSAL;
009import static cn.nukkit.api.API.Usage.BLEEDING;
010
011/**
012 * Describes an API element.
013 *
014 * @author Lin Mulan, Nukkit Project
015 * @see Usage
016 * @see Definition
017 */
018@PowerNukkitDifference(since = "1.4.0.0-PN", info = "Applicable to fields only in PowerNukkit")
019@Retention(RetentionPolicy.SOURCE)
020@Target({ElementType.CONSTRUCTOR, ElementType.METHOD, ElementType.ANNOTATION_TYPE, ElementType.TYPE, ElementType.FIELD})
021@API(usage = BLEEDING, definition = UNIVERSAL)
022@SuppressWarnings("unused")
023public @interface API {
024
025    /**
026     * Indicates the level of stability of an API element.
027     * The stability also describes when to use this API element.
028     *
029     * @return The stability
030     * @see Usage
031     */
032    Usage usage();
033
034    /**
035     * Indicates definition or the platforms this API element supports.
036     *
037     * @return The definition
038     * @see Definition
039     */
040    Definition definition();
041
042    /**
043     * Enum constant for API usage. Indicates when to use this API element.
044     *
045     * @see #DEPRECATED
046     * @see #INCUBATING
047     * @see #BLEEDING
048     * @see #EXPERIMENTAL
049     * @see #MAINTAINED
050     * @see #STABLE
051     */
052    enum Usage {
053
054        /**
055         * Should no longer be used, might disappear in the next minor release.
056         */
057        DEPRECATED,
058
059        /**
060         * Intended for features in drafts. Should only be used for tests.
061         *
062         * <p>Might contains notable new features, but will be moved to a new package before remarking to {@link #BLEEDING}.
063         * Could be unsafe, might be removed without prior notice. Warnings will be send if used.
064         */
065        INCUBATING,
066
067        /**
068         * Intended for features in early development. Should only be used for tests.
069         *
070         * <p>Might be unwrapped, unsafe or have unchecked parameters.
071         * Further contribution was demanded to enhance, strengthen or simplify before remarking to {@link #EXPERIMENTAL}.
072         * Might be removed or modified without prior notice.
073         */
074        BLEEDING,
075
076        /**
077         * Intended for new, experimental features where we are looking for feedback.
078         * At least stable for development.
079         *
080         * <p>Use with caution, might be remarked to {@link #MAINTAINED} or {@link #STABLE} in the future,
081         * but also might be removed without prior notice.
082         */
083        EXPERIMENTAL,
084
085        /**
086         * Intended for features that was tested, documented and at least stable for production use.
087         *
088         * <p>These features will not be modified in a backwards-incompatible way for at least next minor release
089         * of the current major version. Will be remarked to {@link #DEPRECATED} first if scheduled for removal.
090         */
091        MAINTAINED,
092
093        /**
094         * Intended for features that was tested, documented and is preferred in production use.
095         *
096         * <p>Will not be changed in a backwards-incompatible way in the current version.
097         */
098        STABLE
099    }
100
101    /**
102     * Enum constant for API definition. Indicates which client platform this API element supports.
103     *
104     * @see #INTERNAL
105     * @see #PLATFORM_NATIVE
106     * @see #UNIVERSAL
107     */
108    enum Definition {
109
110        /**
111         * Intended for features should only be used by Nukkit itself.
112         * Should not be used in production.
113         */
114        INTERNAL,
115
116        /**
117         * Intended for features only available on one or several client platforms.
118         *
119         * <p>By using {@code PLATFORM_NATIVE} features, program will lose some cross-platform features provided.
120         * Might not available in some client platforms. Read the documents carefully before using this API element.
121         */
122        PLATFORM_NATIVE,
123
124        /**
125         * Intended for features implemented in all client platforms.
126         *
127         * <p>Preferred to use for production use, but sometimes be lack of platform-native features.
128         */
129        UNIVERSAL
130    }
131}
