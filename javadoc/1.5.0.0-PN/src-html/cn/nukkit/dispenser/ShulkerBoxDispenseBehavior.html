001package cn.nukkit.dispenser;
002
003import cn.nukkit.block.Block;
004import cn.nukkit.block.BlockDispenser;
005import cn.nukkit.block.BlockUndyedShulkerBox;
006import cn.nukkit.item.Item;
007import cn.nukkit.math.BlockFace;
008
009public class ShulkerBoxDispenseBehavior extends DefaultDispenseBehavior {
010
011    @Override
012    public Item dispense(BlockDispenser block, BlockFace face, Item item) {
013        Block target = block.getSide(face);
014        
015        if (!target.canBeReplaced()) {
016            success = false;
017            return null;
018        }
019
020        BlockUndyedShulkerBox shulkerBox = (BlockUndyedShulkerBox) item.getBlock().clone();
021        shulkerBox.level = block.level;
022        shulkerBox.layer = 0;
023        shulkerBox.x = target.x;
024        shulkerBox.y = target.y;
025        shulkerBox.z = target.z;
026
027        BlockFace shulkerBoxFace = shulkerBox.down().isTransparent() ? face : BlockFace.UP;
028        
029        if (success = shulkerBox.place(item, target, target.getSide(shulkerBoxFace.getOpposite()), shulkerBoxFace, 0, 0, 0, null)) {
030            block.level.updateComparatorOutputLevel(target);
031        }
032
033        return null;
034    }
035}
