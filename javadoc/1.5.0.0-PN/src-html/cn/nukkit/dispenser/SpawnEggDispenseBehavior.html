001package cn.nukkit.dispenser;
002
003import cn.nukkit.block.BlockDispenser;
004import cn.nukkit.entity.Entity;
005import cn.nukkit.entity.EntityLiving;
006import cn.nukkit.item.Item;
007import cn.nukkit.item.ItemSpawnEgg;
008import cn.nukkit.math.BlockFace;
009import cn.nukkit.math.Vector3;
010
011public class SpawnEggDispenseBehavior extends DefaultDispenseBehavior {
012
013    @Override
014    public Item dispense(BlockDispenser block, BlockFace face, Item item) {
015        Vector3 pos = block.getSide(face).add(0.5, 0.7, 0.5);
016
017        Entity entity = Entity.createEntity(((ItemSpawnEgg)item).getEntityNetworkId(), block.level.getChunk(pos.getChunkX(), pos.getChunkZ()),
018                Entity.getDefaultNBT(pos));
019
020        this.success = entity != null;
021
022        if (this.success) {
023            if (item.hasCustomName() && entity instanceof EntityLiving) {
024                entity.setNameTag(item.getCustomName());
025            }
026
027            entity.spawnToAll();
028            return null;
029        }
030
031        return super.dispense(block, face, item);
032    }
033}
