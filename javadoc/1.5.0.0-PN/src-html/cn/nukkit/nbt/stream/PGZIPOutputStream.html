001package cn.nukkit.nbt.stream;
002
003import cn.nukkit.api.PowerNukkitOnly;
004import cn.nukkit.api.Since;
005import it.unimi.dsi.fastutil.ints.IntArrayList;
006import it.unimi.dsi.fastutil.ints.IntList;
007
008import java.io.FilterOutputStream;
009import java.io.IOException;
010import java.io.InterruptedIOException;
011import java.io.OutputStream;
012import java.nio.ByteBuffer;
013import java.nio.ByteOrder;
014import java.util.concurrent.*;
015import java.util.zip.CRC32;
016import java.util.zip.Deflater;
017import java.util.zip.DeflaterOutputStream;
018
019/**
020 * A multi-threaded version of {@link java.util.zip.GZIPOutputStream}.
021 *
022 * @author shevek
023 */
024public class PGZIPOutputStream extends FilterOutputStream {
025
026    private static final ExecutorService EXECUTOR = Executors.newCachedThreadPool();
027
028    public static ExecutorService getSharedThreadPool() {
029        return EXECUTOR;
030    }
031
032
033    // private static final Logger LOG = LoggerFactory.getLogger(PGZIPOutputStream.class);
034    private final static int GZIP_MAGIC = 0x8b1f;
035
036    // todo: remove after block guessing is implemented
037    // array list that contains the block sizes
038    private IntList blockSizes = new IntArrayList();
039
040    private int level = Deflater.BEST_SPEED;
041    private int strategy = Deflater.DEFAULT_STRATEGY;
042
043    protected Deflater newDeflater() {
044        Deflater def = new Deflater(level, true);
045        def.setStrategy(strategy);
046        return def;
047    }
048
049    public void setStrategy(int strategy) {
050        this.strategy = strategy;
051    }
052
053    public void setLevel(int level) {
054        this.level = level;
055    }
056
057    protected static DeflaterOutputStream newDeflaterOutputStream(OutputStream out, Deflater deflater) {
058        return new DeflaterOutputStream(out, deflater, 512, true);
059    }
060
061    // TODO: Share, daemonize.
062    private final ExecutorService executor;
063    private final int nthreads;
064    private final CRC32 crc = new CRC32();
065    private final BlockingQueue<Future<byte[]>> emitQueue;
066    private PGZIPBlock block = new PGZIPBlock(this/* 0 */);
067    /**
068     * Used as a sentinel for 'closed'.
069     */
070    private int bytesWritten = 0;
071
072    // Master thread only
073    public PGZIPOutputStream(OutputStream out, ExecutorService executor, int nthreads) throws IOException {
074        super(out);
075        this.executor = executor;
076        this.nthreads = nthreads;
077        this.emitQueue = new ArrayBlockingQueue<Future<byte[]>>(nthreads);
078        writeHeader();
079    }
080
081    /**
082     * Creates a PGZIPOutputStream
083     * using {@link PGZIPOutputStream#getSharedThreadPool()}.
084     *
085     * @param out the eventual output stream for the compressed data.
086     * @throws java.io.IOException if it all goes wrong.
087     */
088    public PGZIPOutputStream(OutputStream out, int nthreads) throws IOException {
089        this(out, PGZIPOutputStream.getSharedThreadPool(), nthreads);
090    }
091
092    /**
093     * Creates a PGZIPOutputStream
094     * using {@link PGZIPOutputStream#getSharedThreadPool()}
095     * and {@link Runtime#availableProcessors()}.
096     *
097     * @param out the eventual output stream for the compressed data.
098     * @throws java.io.IOException if it all goes wrong.
099     */
100    public PGZIPOutputStream(OutputStream out) throws IOException {
101        this(out, Runtime.getRuntime().availableProcessors());
102    }
103
104    /*
105     * @see http://www.gzip.org/zlib/rfc-gzip.html#file-format
106     */
107    private void writeHeader() throws IOException {
108        out.write(new byte[]{
109                (byte) GZIP_MAGIC, // ID1: Magic number (little-endian short)
110                (byte) (GZIP_MAGIC >> 8), // ID2: Magic number (little-endian short)
111                Deflater.DEFLATED, // CM: Compression method
112                0, // FLG: Flags (byte)
113                0, 0, 0, 0, // MTIME: Modification time (int)
114                0, // XFL: Extra flags
115                3 // OS: Operating system (3 = Linux)
116        });
117    }
118
119    // Master thread only
120    @Override
121    public void write(int b) throws IOException {
122        byte[] single = new byte[1];
123        single[0] = (byte) (b & 0xFF);
124        write(single);
125    }
126
127    // Master thread only
128    @Override
129    public void write(byte[] b) throws IOException {
130        write(b, 0, b.length);
131    }
132
133    // Master thread only
134    @Override
135    public void write(byte[] b, int off, int len) throws IOException {
136        crc.update(b, off, len);
137        bytesWritten += len;
138        while (len > 0) {
139            // assert block.in_length < block.in.length
140            int capacity = block.in.length - block.in_length;
141            if (len >= capacity) {
142                System.arraycopy(b, off, block.in, block.in_length, capacity);
143                block.in_length += capacity;   // == block.in.length
144                off += capacity;
145                len -= capacity;
146                submit();
147            } else {
148                System.arraycopy(b, off, block.in, block.in_length, len);
149                block.in_length += len;
150                // off += len;
151                // len = 0;
152                break;
153            }
154        }
155    }
156
157    // Master thread only
158    private void submit() throws IOException {
159        emitUntil(nthreads - 1);
160        emitQueue.add(executor.submit(block));
161        block = new PGZIPBlock(this/* block.index + 1 */);
162    }
163
164    // Emit If Available - submit always
165    // Emit At Least one - submit when executor is full
166    // Emit All Remaining - flush(), close()
167    // Master thread only
168    private void tryEmit() throws IOException, InterruptedException, ExecutionException {
169        for (; ; ) {
170            Future<byte[]> future = emitQueue.peek();
171            // LOG.info("Peeked future " + future);
172            if (future == null)
173                return;
174            if (!future.isDone())
175                return;
176            // It's an ordered queue. This MUST be the same element as above.
177            emitQueue.remove();
178            byte[] toWrite = future.get();
179            blockSizes.add(toWrite.length);  // todo: remove after block guessing is implemented
180            out.write(toWrite);
181        }
182    }
183
184    // Master thread only
185
186    /**
187     * Emits any opportunistically available blocks. Furthermore, emits blocks until the number of executing tasks is less than taskCountAllowed.
188     */
189    private void emitUntil(int taskCountAllowed) throws IOException {
190        try {
191            while (emitQueue.size() > taskCountAllowed) {
192                // LOG.info("Waiting for taskCount=" + emitQueue.size() + " -> " + taskCountAllowed);
193                Future<byte[]> future = emitQueue.remove(); // Valid because emitQueue.size() > 0
194                byte[] toWrite = future.get();  // Blocks until this task is done.
195                blockSizes.add(toWrite.length);  // todo: remove after block guessing is implemented
196                out.write(toWrite);
197            }
198            // We may have achieved more opportunistically available blocks
199            // while waiting for a block above. Let's emit them here.
200            tryEmit();
201        } catch (ExecutionException e) {
202            throw new IOException(e);
203        } catch (InterruptedException e) {
204            throw new InterruptedIOException();
205        }
206    }
207
208    // Master thread only
209    @Override
210    public void flush() throws IOException {
211        // LOG.info("Flush: " + block);
212        if (block.in_length > 0)
213            submit();
214        emitUntil(0);
215        super.flush();
216    }
217    
218    @PowerNukkitOnly
219    @Since("1.4.0.0-PN")
220    public void finish() throws IOException {
221        if (bytesWritten >= 0) {
222            flush();
223
224            newDeflaterOutputStream(out, newDeflater()).finish();
225
226            ByteBuffer buf = ByteBuffer.allocate(8);
227            buf.order(ByteOrder.LITTLE_ENDIAN);
228            // LOG.info("CRC is " + crc.getValue());
229            buf.putInt((int) crc.getValue());
230            buf.putInt(bytesWritten);
231            out.write(buf.array()); // allocate() guarantees a backing array.
232            // LOG.info("trailer is " + Arrays.toString(buf.array()));
233        }
234    }
235
236    // Master thread only
237    @Override
238    public void close() throws IOException {
239        // LOG.info("Closing: bytesWritten=" + bytesWritten);
240        if (bytesWritten >= 0) {
241            finish();
242
243            out.flush();
244            out.close();
245
246            bytesWritten = Integer.MIN_VALUE;
247            // } else {
248            // LOG.warn("Already closed.");
249        }
250    }
251}
