001package cn.nukkit.level.format.updater;
002
003import cn.nukkit.api.PowerNukkitOnly;
004import cn.nukkit.api.Since;
005import cn.nukkit.block.BlockID;
006import cn.nukkit.block.BlockSnowLayer;
007import cn.nukkit.blockstate.BlockState;
008import cn.nukkit.level.Level;
009import cn.nukkit.level.format.ChunkSection;
010
011@PowerNukkitOnly
012@Since("1.4.0.0-PN")
013public class SnowLayerUpdater implements Updater {
014    private final Level level;
015    private final ChunkSection section;
016
017    @PowerNukkitOnly
018    @Since("1.4.0.0-PN")
019    public SnowLayerUpdater(Level level, ChunkSection section) {
020        this.level = level;
021        this.section = section;
022    }
023
024    @Override
025    public boolean update(int offsetX, int offsetY, int offsetZ, int x, int y, int z, BlockState state) {
026        if (state.getBlockId() != BlockID.SNOW_LAYER) {
027            return false;
028        }
029        
030        if (y > 0) {
031            if (section.getBlockId(x, y-1, z) == BlockID.GRASS) {
032                section.setBlockState(x, y, z, state.withProperty(BlockSnowLayer.COVERED, true));
033                return true;
034            }
035        } else if (offsetY == 0) {
036            return false;
037        }
038        if (level.getBlockIdAt(offsetX + x, offsetY + y - 1, offsetZ + z, 0) == BlockID.GRASS) {
039            section.setBlockState(x, y, z, state.withProperty(BlockSnowLayer.COVERED, true));
040            return true;
041        }
042        
043        return false;
044    }
045}
