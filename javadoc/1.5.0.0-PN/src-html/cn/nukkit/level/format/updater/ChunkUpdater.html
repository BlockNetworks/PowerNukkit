042    public static void backwardCompatibilityUpdate(Level level, BaseChunk chunk) {
043        boolean updated = false;
044        for (ChunkSection section : chunk.getSections()) {
045            if (section.getContentVersion() >= getCurrentContentVersion()) {
046                continue;
047            }
048            if (section.getContentVersion() < 5) {
049                updated = updateToV8FromV0toV5(level, chunk, updated, section, section.getContentVersion());
050            } else if (section.getContentVersion() == 5 || section.getContentVersion() == 7) {
051                updated = updateBeehiveFromV5or6or7toV8(chunk, updated, section, false);
052            } else if (section.getContentVersion() == 6) {
053                updated = updateBeehiveFromV5or6or7toV8(chunk, updated, section, true);
054            }
055            if (section.getContentVersion() == 8) {
056                updated = upgradeWallsFromV8toV9(level, chunk, updated, section);
057            }
058            if (section.getContentVersion() == 9) {
059                updated = upgradeSnowLayersFromV9toV10(level, chunk, updated, section);
060            }
061            if (section.getContentVersion() == 10) {
062                updated = upgradeStrippedStemsFromV10toV11(chunk, updated, section);
063            }
064            if (section.getContentVersion() == 11) {
065                updated = upgradeFrameFromV11toV12(chunk, section, updated);
066            }
067        }
068        if (updated) {
069            chunk.setChanged();
070        }

105    private static boolean updateToV8FromV0toV5(Level level, BaseChunk chunk, boolean updated, ChunkSection section, int contentVersion) {
106        WallUpdater wallUpdater = new WallUpdater(level, section);
107        boolean sectionUpdated = walk(chunk, section, new GroupedUpdaters(new StemStrippedUpdater(section), new MesaBiomeUpdater(section), new NewLeafUpdater(section), new BeehiveUpdater(section, true), wallUpdater, contentVersion < 1 ? new StemUpdater(level, section, BlockID.MELON_STEM, BlockID.MELON_BLOCK) : null, contentVersion < 1 ? new StemUpdater(level, section, BlockID.PUMPKIN_STEM, BlockID.PUMPKIN) : null, contentVersion < 5 ? new OldWoodBarkUpdater(section, BlockID.LOG, 0) : null, contentVersion < 5 ? new OldWoodBarkUpdater(section, BlockID.LOG2, 4) : null, contentVersion < 5 ? new DoorUpdater(chunk, section) : null));
108        updated = updated || sectionUpdated;
109        int attempts = 0;
110        while (sectionUpdated) {
111            if (attempts++ >= 5) {
112                int x = chunk.getX() << 4 | 6;
113                int y = section.getY() << 4 | 6;
114                int z = chunk.getZ() << 4 | 6;
115                log.error("The chunk section at x:{}, y:{}, z:{} failed to complete the backward compatibility update 1 after {} attempts", x, y, z, attempts);
116                break;
117            }
118            sectionUpdated = walk(chunk, section, wallUpdater);
119        }
120        section.setContentVersion(8);
121        return updated;
122    }

124    private static boolean walk(BaseChunk chunk, ChunkSection section, Updater updater) {
125        int offsetX = chunk.getX() << 4;
126        int offsetZ = chunk.getZ() << 4;
127        int offsetY = section.getY() << 4;
128        boolean updated = false;
129        for (int x = 0; x <= 15; x++) {
130            for (int z = 0; z <= 15; z++) {
131                for (int y = 0; y <= 15; y++) {
132                    BlockState state = section.getBlockState(x, y, z, 0);
133                    updated |= updater.update(offsetX, offsetY, offsetZ, x, y, z, state);
134                }
135            }
136        }
137        return updated;
138    }
