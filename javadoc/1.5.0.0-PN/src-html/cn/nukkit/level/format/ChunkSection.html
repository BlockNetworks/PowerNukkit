184    @PowerNukkitOnly
185    @Since("1.4.0.0-PN")
186    default List<Block> scanBlocks(LevelProvider provider, int offsetX, int offsetZ, BlockVector3 min, BlockVector3 max, BiPredicate<BlockVector3, BlockState> condition) {
187        int offsetY = getY() << 4;
188        List<Block> results = new ArrayList<>();
189
190        BlockVector3 current = new BlockVector3();
191        
192        int minX = Math.max(0, min.x - offsetX);
193        int minY = Math.max(0, min.y - offsetY);
194        int minZ = Math.max(0, min.z - offsetZ);
195
196        for (int x = Math.min(max.x - offsetX, 15); x >= minX; x--) {
197            current.x = offsetX + x;
198            for (int z = Math.min(max.z - offsetZ, 15); z >= minZ; z--) {
199                current.z = offsetZ + z;
200                for (int y = Math.min(max.y - offsetY, 15); y >= minY; y--) {
201                    current.y = offsetY + y;
202                    BlockState state = getBlockState(x, y, z);
203                    if (condition.test(current, state)) {
204                        results.add(state.getBlockRepairing(provider.getLevel(), current, 0));
205                    }
206                }
207            }
208        }
209        
210        return results;
211    }
