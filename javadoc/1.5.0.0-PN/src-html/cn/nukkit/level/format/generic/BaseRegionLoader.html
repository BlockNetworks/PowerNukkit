001package cn.nukkit.level.format.generic;
002
003import cn.nukkit.api.DeprecationDetails;
004import cn.nukkit.api.PowerNukkitOnly;
005import cn.nukkit.api.Since;
006import cn.nukkit.level.format.FullChunk;
007import cn.nukkit.level.format.LevelProvider;
008import cn.nukkit.utils.Utils;
009import cn.nukkit.utils.collection.ConvertingMapWrapper;
010import it.unimi.dsi.fastutil.ints.Int2ObjectMap;
011import it.unimi.dsi.fastutil.ints.Int2ObjectOpenHashMap;
012
013import java.io.File;
014import java.io.IOException;
015import java.io.RandomAccessFile;
016import java.util.Arrays;
017import java.util.Map;
018
019/**
020 * @author MagicDroidX (Nukkit Project)
021 */
022abstract public class BaseRegionLoader {
023    public static final int VERSION = 1;
024    public static final byte COMPRESSION_GZIP = 1;
025    public static final byte COMPRESSION_ZLIB = 2;
026    public static final int MAX_SECTOR_LENGTH = 256 << 12;
027    public static final int COMPRESSION_LEVEL = 7;
028
029    protected int x;
030    protected int z;
031    protected int lastSector;
032    protected LevelProvider levelProvider;
033
034    private RandomAccessFile randomAccessFile;
035
036    @PowerNukkitOnly
037    @Since("1.4.0.0-PN")
038    protected final Int2ObjectMap<int[]> primitiveLocationTable = new Int2ObjectOpenHashMap<>();
039    
040    @Deprecated
041    @DeprecationDetails(since = "1.4.0.0-PN", reason = "Integer boxing was polluting the memory heap", replaceWith = "primitiveLocationTable")
042    protected final Map<Integer, Integer[]> locationTable = new ConvertingMapWrapper<>(
043            primitiveLocationTable,
044            table-> Arrays.stream(table).mapToInt(Integer::intValue).toArray(),
045            table -> Arrays.stream(table).boxed().toArray(Integer[]::new)
046    );
047
048    public long lastUsed;
049
050    public BaseRegionLoader(LevelProvider level, int regionX, int regionZ, String ext) {
051        try {
052            this.x = regionX;
053            this.z = regionZ;
054            this.levelProvider = level;
055            String filePath = this.levelProvider.getPath() + "region/r." + regionX + "." + regionZ + "." + ext;
056            File file = new File(filePath);
057            boolean exists = file.exists();
058            if (!exists) {
059                file.createNewFile();
060            }
061            // TODO: buffering is a temporary solution to chunk reading/writing being poorly optimized
062            //  - need to fix the code where it reads single bytes at a time from disk
063            this.randomAccessFile = new RandomAccessFile(filePath, "rw");
064            if (!exists) {
065                this.createBlank();
066            } else {
067                this.loadLocationTable();
068            }
069
070            this.lastUsed = System.currentTimeMillis();
071        } catch (IOException e) {
072            throw new RuntimeException(e);
073        }
074    }
075
076    public void compress() {
077        // TODO
078    }
079
080    public RandomAccessFile getRandomAccessFile() {
081        return randomAccessFile;
082    }
083
084    protected abstract boolean isChunkGenerated(int index);
085
086    public abstract BaseFullChunk readChunk(int x, int z) throws IOException;
087
088    protected abstract BaseFullChunk unserializeChunk(byte[] data);
089
090    public abstract boolean chunkExists(int x, int z);
091
092    protected abstract void saveChunk(int x, int z, byte[] chunkData) throws IOException;
093
094    public abstract void removeChunk(int x, int z);
095
096    public abstract void writeChunk(FullChunk chunk) throws Exception;
097
098    public void close() throws IOException {
099        if (randomAccessFile != null) randomAccessFile.close();
100    }
101
102    protected abstract void loadLocationTable() throws IOException;
103
104    public abstract int doSlowCleanUp() throws Exception;
105
106    protected abstract void writeLocationIndex(int index) throws IOException;
107
108    protected abstract void createBlank() throws IOException;
109
110    public abstract int getX();
111
112    public abstract int getZ();
113
114    @Deprecated
115    @DeprecationDetails(
116            since = "1.4.0.0-PN", by = "PowerNukkit",
117            reason = "Unnecessary int-boxing causing heap pollution", 
118            replaceWith = "getIntLocationIndexes()")
119    public Integer[] getLocationIndexes() {
120        return this.primitiveLocationTable.keySet().toArray(Utils.EMPTY_INTEGERS);
121    }
122
123    @PowerNukkitOnly
124    @Since("1.4.0.0-PN")
125    public int[] getIntLocationIndexes() {
126        return this.primitiveLocationTable.keySet().toIntArray();
127    }
128}
