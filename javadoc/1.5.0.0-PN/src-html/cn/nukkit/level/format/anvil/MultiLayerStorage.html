052    @Since("1.4.0.0-PN")
053    @PowerNukkitOnly
054    @Override
055    public void compress(Consumer<LayerStorage> setLayerStorage) {
056        // Remove unused storage layers
057        int newSize = storages.length;
058        for (int i = storages.length - 1; i >= 0; i--) {
059            BlockStorage storage = storages[i];
060            if (storage == null || storage == ImmutableBlockStorage.EMPTY) {
061                newSize--;
062            } else if (storage.hasBlockIds()) {
063                storage.recheckBlocks();
064                if (storage.hasBlockIds()) {
065                    break;
066                } else {
067                    newSize--;
068                }
069            } else {
070                newSize--;
071            }
072        }
073        if (newSize == 0) {
074            setLayerStorage.accept(EMPTY);
075        } else if (newSize == 1) {
076            setLayerStorage.accept(new SingleLayerStorage(storages[0]));
077        } else if (newSize != storages.length) {
078            storages = Arrays.copyOf(storages, newSize);
079        }
080    }

110    @PowerNukkitOnly
111    @Override
112    public BlockStorage getOrSetStorage(@Nullable Consumer<LayerStorage> setLayerStorage, IntSupplier contentVersion, int layer) {
113        int oldLen = storages.length;
114        if (layer >= oldLen) {
115            if (layer > 1) {
116                throw new IndexOutOfBoundsException("Only layer 0 and 1 are supported. Attempted: " + layer);
117            }
118            storages = Arrays.copyOf(storages, layer + 1);
119            Arrays.fill(storages, oldLen, layer, ImmutableBlockStorage.EMPTY);
120            BlockStorage storage = createBlockStorage(contentVersion.getAsInt());
121            storages[layer] = storage;
122            return storage;
123        } else if (layer < 0) {
124            throw new IndexOutOfBoundsException("Only layer 0 and 1 are supported. Attempted: " + layer);
125        }
126        BlockStorage storage = storages[layer];
127        if (storage != ImmutableBlockStorage.EMPTY) {
128            return storage;
129        }
130        storage = createBlockStorage(contentVersion.getAsInt());
131        storages[layer] = storage;
132        return storage;
133    }
