001package cn.nukkit.level.format.anvil.palette;
002
003import io.netty.util.internal.EmptyArrays;
004
005import java.util.Arrays;
006
007/**
008 * @author https://github.com/boy0001/
009 */
010public class BytePalette {
011    private static byte[] BYTE0 = EmptyArrays.EMPTY_BYTES;
012    private byte[] keys = BYTE0;
013    private byte lastIndex = Byte.MIN_VALUE;
014
015    public void add(byte key) {
016        keys = insert(key);
017        lastIndex = Byte.MIN_VALUE;
018    }
019
020    protected void set(byte[] keys) {
021        this.keys = keys;
022        lastIndex = Byte.MIN_VALUE;
023    }
024
025    private byte[] insert(byte val) {
026        lastIndex = Byte.MIN_VALUE;
027        if (keys.length == 0) {
028            return new byte[] { val };
029        }
030        else if (val < keys[0]) {
031            byte[] s = new byte[keys.length + 1];
032            System.arraycopy(keys, 0, s, 1, keys.length);
033            s[0] = val;
034            return s;
035        } else if (val > keys[keys.length - 1]) {
036            byte[] s = Arrays.copyOf(keys, keys.length + 1);
037            s[keys.length] = val;
038            return s;
039        }
040        byte[] s = Arrays.copyOf(keys, keys.length + 1);
041        for (int i = 0; i < s.length; i++) {
042            if (keys[i] < val) {
043                continue;
044            }
045            System.arraycopy(keys, i, s, i + 1, s.length - i - 1);
046            s[i] = val;
047            break;
048        }
049        return s;
050    }
051
052    public byte getKey(int index) {
053        return keys[index];
054    }
055
056    public byte getValue(byte key) {
057        byte lastTmp = lastIndex;
058        boolean hasLast = lastTmp != Byte.MIN_VALUE;
059        int index;
060        if (hasLast) {
061            byte lastKey = keys[lastTmp];
062            if (lastKey == key) return lastTmp;
063            if (lastKey > key) {
064                index = binarySearch0(0, lastTmp, key);
065            } else {
066                index = binarySearch0(lastTmp + 1, keys.length, key);
067            }
068        } else {
069            index = binarySearch0(0, keys.length, key);
070        }
071        if (index >= keys.length || index < 0) {
072            return lastIndex = Byte.MIN_VALUE;
073        } else {
074            return lastIndex = (byte) index;
075        }
076    }
077
078    private int binarySearch0(int fromIndex, int toIndex, byte key) {
079        int low = fromIndex;
080        int high = toIndex - 1;
081
082        while (low <= high) {
083            int mid = (low + high) >>> 1;
084            byte midVal = keys[mid];
085
086            if (midVal < key)
087                low = mid + 1;
088            else if (midVal > key)
089                high = mid - 1;
090            else
091                return mid; // key found
092        }
093        return -(low + 1);  // key not found.
094    }
095}
