001package cn.nukkit.level;
002
003import cn.nukkit.api.PowerNukkitOnly;
004import cn.nukkit.api.Since;
005import cn.nukkit.block.Block;
006import cn.nukkit.blockstate.BlockState;
007import cn.nukkit.level.format.generic.BaseFullChunk;
008
009import java.util.ArrayList;
010import java.util.List;
011import java.util.Optional;
012
013@Since("1.4.0.0-PN")
014public class ListChunkManager implements ChunkManager {
015
016    private ChunkManager parent;
017    private List<Block> blocks;
018
019    @Since("1.4.0.0-PN")
020    public ListChunkManager(ChunkManager parent) {
021        this.parent = parent;
022        this.blocks = new ArrayList<>();
023    }
024
025    @Override
026    public int getBlockIdAt(int x, int y, int z) {
027        return getBlockIdAt(x, y, z, 0);
028    }
029    
030    private Optional<Block> findBlockAt(int x, int y, int z, int layer) {
031        return this.blocks.stream().filter(block -> 
032                block.getFloorX() == x 
033                        && block.getFloorY() == y
034                        && block.getFloorZ() == z 
035                        && block.layer == layer
036        ).findAny();
037    }
038    
039    @Override
040    public int getBlockIdAt(int x, int y, int z, int layer) {
041        return findBlockAt(x, y, z, layer).map(Block::getId).orElseGet(() -> this.parent.getBlockIdAt(x, y, z, layer));
042    }
043
044    @Override
045    public void setBlockFullIdAt(int x, int y, int z, int fullId) {
046        setBlockFullIdAt(x, y, z, 0, fullId);
047    }
048
049    @Override
050    public void setBlockFullIdAt(int x, int y, int z, int layer, int fullId) {
051        this.blocks.removeIf(block -> block.getFloorX() == x && block.getFloorY() == y && block.getFloorZ() == z && block.layer == layer);
052        this.blocks.add(Block.get(fullId, null, x, y, z, layer));
053    }
054
055    @Override
056    public void setBlockIdAt(int x, int y, int z, int id) {
057        setBlockIdAt(x, y, z, 0, id);
058    }
059
060    @Override
061    public void setBlockIdAt(int x, int y, int z, int layer, int id) {
062        Optional<Block> optionalBlock = this.blocks.stream().filter(block -> block.getFloorX() == x && block.getFloorY() == y && block.getFloorZ() == z && block.layer == layer).findAny();
063        Block block = optionalBlock.orElse(Block.get(this.getBlockIdAt(x, y, z, layer), this.getBlockDataAt(x, y, z, layer), new Position(x, y, z), layer));
064        this.blocks.remove(block);
065        this.blocks.add(Block.get(this.getBlockIdAt(x, y, z, layer), this.getBlockDataAt(x, y, z, layer), new Position(x, y, z), layer));
066    }
067
068    @Override
069    public void setBlockAt(int x, int y, int z, int id, int data) {
070        this.blocks.removeIf(block -> block.getFloorX() == x && block.getFloorY() == y && block.getFloorZ() == z && block.layer == 0);
071        this.blocks.add(Block.get(id, data, new Position(x, y, z), 0));
072    }
073
074    @Override
075    public boolean setBlockAtLayer(int x, int y, int z, int layer, int id, int data) {
076        boolean removed = this.blocks.removeIf(block -> block.getFloorX() == x && block.getFloorY() == y && block.getFloorZ() == z && block.layer == layer);
077        this.blocks.add(Block.get(id, data, new Position(x, y, z), layer));
078        return !removed;
079    }
080
081    @Since("1.4.0.0-PN")
082    @PowerNukkitOnly
083    @Override
084    public boolean setBlockStateAt(int x, int y, int z, int layer, BlockState state) {
085        boolean removed = this.blocks.removeIf(block -> block.getFloorX() == x && block.getFloorY() == y && block.getFloorZ() == z && block.layer == layer);
086        this.blocks.add(state.getBlock(new Position(x, y, z), layer));
087        return !removed;
088    }
089
090    @Since("1.4.0.0-PN")
091    @PowerNukkitOnly
092    @Override
093    public BlockState getBlockStateAt(int x, int y, int z, int layer) {
094        return findBlockAt(x, y, z, layer).map(Block::getCurrentState).orElseGet(()-> parent.getBlockStateAt(x, y, z, layer));
095    }
096
097    @Override
098    public int getBlockDataAt(int x, int y, int z) {
099        return getBlockIdAt(x, y, z, 0);
100    }
101
102    @Override
103    public int getBlockDataAt(int x, int y, int z, int layer) {
104        Optional<Block> optionalBlock = this.blocks.stream().filter(block -> block.getFloorX() == x && block.getFloorY() == y && block.getFloorZ() == z && block.layer == layer).findAny();
105        return optionalBlock.map(Block::getDamage).orElseGet(() -> this.parent.getBlockDataAt(x, y, z, layer));
106    }
107
108    @Override
109    public void setBlockDataAt(int x, int y, int z, int data) {
110        setBlockIdAt(x, y, z, 0, data);
111    }
112    
113    @Override
114    public void setBlockDataAt(int x, int y, int z, int layer, int data) {
115        Optional<Block> optionalBlock = this.blocks.stream().filter(block -> block.getFloorX() == x && block.getFloorY() == y && block.getFloorZ() == z && block.layer == layer).findAny();
116        Block block = optionalBlock.orElse(Block.get(this.getBlockIdAt(x, y, z, layer), this.getBlockDataAt(x, y, z, layer), new Position(x, y, z), layer));
117        this.blocks.remove(block);
118        block.setDamage(data);
119        this.blocks.add(block);
120    }
121
122    @Override
123    public BaseFullChunk getChunk(int chunkX, int chunkZ) {
124        return this.parent.getChunk(chunkX, chunkZ);
125    }
126
127    @Override
128    public void setChunk(int chunkX, int chunkZ) {
129        this.parent.setChunk(chunkX, chunkZ);
130    }
131
132    @Override
133    public void setChunk(int chunkX, int chunkZ, BaseFullChunk chunk) {
134        this.parent.setChunk(chunkX, chunkZ, chunk);
135    }
136
137    @Override
138    public long getSeed() {
139        return this.parent.getSeed();
140    }
141
142    @Since("1.4.0.0-PN")
143    public List<Block> getBlocks() {
144        return this.blocks;
145    }
146
147}
