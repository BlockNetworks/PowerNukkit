001package cn.nukkit.level.generator.noise.nukkit.f;

003import cn.nukkit.math.NukkitRandom;
004
005/**
006 * @author DaPorkchop_ (Nukkit Project)
007 */
008public class SimplexF extends PerlinF {
009    protected static float SQRT_3;
010    protected static float SQRT_5;
011    protected static float F2;
012    protected static float G2;
013    protected static float G22;
014    protected static float F3;
015    protected static float G3;
016    protected static float F4;
017    protected static float G4;
018    protected static float G42;
019    protected static float G43;
020    protected static float G44;
021    public static final int[][] grad3 = {
022            {1, 1, 0}, {-1, 1, 0}, {1, -1, 0}, {-1, -1, 0},
023            {1, 0, 1}, {-1, 0, 1}, {1, 0, -1}, {-1, 0, -1},
024            {0, 1, 1}, {0, -1, 1}, {0, 1, -1}, {0, -1, -1}
025    };
026    protected final float offsetW;
027
028    public SimplexF(NukkitRandom random, float octaves, float persistence) {
029        super(random, octaves, persistence);
030        this.offsetW = random.nextFloat() * 256;
031        SQRT_3 = (float) Math.sqrt(3);
032        SQRT_5 = (float) Math.sqrt(5);
033        F2 = 0.5f * (SQRT_3 - 1f);
034        G2 = (3f - SQRT_3) / 6f;
035        G22 = G2 * 2.0f - 1f;
036        F3 = 1.0f / 3.0f;
037        G3 = 1.0f / 6.0f;
038        F4 = (SQRT_5 - 1.0f) / 4.0f;
039        G4 = (5.0f - SQRT_5) / 20.0f;
040        G42 = G4 * 2.0f;
041        G43 = G4 * 3.0f;
042        G44 = G4 * 4.0f - 1.0f;
043    }
044
045    public SimplexF(NukkitRandom random, float octaves, float persistence, float expansion) {
046        super(random, octaves, persistence, expansion);
047        this.offsetW = random.nextFloat() * 256;
048        SQRT_3 = (float) Math.sqrt(3);
049        SQRT_5 = (float) Math.sqrt(5);
050        F2 = 0.5f * (SQRT_3 - 1f);
051        G2 = (3f - SQRT_3) / 6f;
052        G22 = G2 * 2.0f - 1f;
053        F3 = 1.0f / 3.0f;
054        G3 = 1.0f / 6.0f;
055        F4 = (SQRT_5 - 1.0f) / 4.0f;
056        G4 = (5.0f - SQRT_5) / 20.0f;
057        G42 = G4 * 2.0f;
058        G43 = G4 * 3.0f;
059        G44 = G4 * 4.0f - 1.0f;
060    }
061
062
063    protected static float dot2D(int[] g, float x, float y) {
064        return g[0] * x + g[1] * y;
065    }
066
067    protected static float dot3D(int[] g, float x, float y, float z) {
068        return g[0] * x + g[1] * y + g[2] * z;
069    }
070
071    protected static float dot4D(int[] g, float x, float y, float z, float w) {
072        return g[0] * x + g[1] * y + g[2] * z + g[3] * w;
073    }

075    @Override
076    public float getNoise3D(float x, float y, float z) {
077        x += this.offsetX;
078        y += this.offsetY;
079        z += this.offsetZ;
080
081        // Skew the input space to determine which simplex cell we're in
082        float s = (x + y + z) * F3; // Very nice and simple skew factor for 3D
083        int i = (int) (x + s);
084        int j = (int) (y + s);
085        int k = (int) (z + s);
086        float t = (i + j + k) * G3;
087        // Unskew the cell origin back to (x,y,z) space
088        float x0 = x - (i - t); // The x,y,z distances from the cell origin
089        float y0 = y - (j - t);
090        float z0 = z - (k - t);
091
092        // For the 3D case, the simplex shape is a slightly irregular tetrahedron.
093        int i1 = 0;
094        int j1 = 0;
095        int k1 = 0;
096        int i2 = 0;
097        int j2 = 0;
098        int k2 = 0;
099
100        // Determine which simplex we are in.
101        if (x0 >= y0) {
102            if (y0 >= z0) {
103                i1 = 1;
104                j1 = 0;
105                k1 = 0;
106                i2 = 1;
107                j2 = 1;
108                k2 = 0;
109            } // X Y Z order
110            else if (x0 >= z0) {
111                i1 = 1;
112                j1 = 0;
113                k1 = 0;
114                i2 = 1;
115                j2 = 0;
116                k2 = 1;
117            } // X Z Y order
118            else {
119                i1 = 0;
120                j1 = 0;
121                k1 = 1;
122                i2 = 1;
123                j2 = 0;
124                k2 = 1;
125            }
126            // Z X Y order
127        } else { // x0<y0
128            if (y0 < z0) {
129                i1 = 0;
130                j1 = 0;
131                k1 = 1;
132                i2 = 0;
133                j2 = 1;
134                k2 = 1;
135            } // Z Y X order
136            else if (x0 < z0) {
137                i1 = 0;
138                j1 = 1;
139                k1 = 0;
140                i2 = 0;
141                j2 = 1;
142                k2 = 1;
143            } // Y Z X order
144            else {
145                i1 = 0;
146                j1 = 1;
147                k1 = 0;
148                i2 = 1;
149                j2 = 1;
150                k2 = 0;
151            }
152            // Y X Z order
153        }
154
155        // A step of (1,0,0) in (i,j,k) means a step of (1-c,-c,-c) in (x,y,z),
156        // a step of (0,1,0) in (i,j,k) means a step of (-c,1-c,-c) in (x,y,z), and
157        // a step of (0,0,1) in (i,j,k) means a step of (-c,-c,1-c) in (x,y,z), where
158        // c = 1/6.
159        float x1 = x0 - i1 + G3; // Offsets for second corner in (x,y,z) coords
160        float y1 = y0 - j1 + G3;
161        float z1 = z0 - k1 + G3;
162        float x2 = x0 - i2 + 2.0f * G3; // Offsets for third corner in (x,y,z) coords
163        float y2 = y0 - j2 + 2.0f * G3;
164        float z2 = z0 - k2 + 2.0f * G3;
165        float x3 = x0 - 1.0f + 3.0f * G3; // Offsets for last corner in (x,y,z) coords
166        float y3 = y0 - 1.0f + 3.0f * G3;
167        float z3 = z0 - 1.0f + 3.0f * G3;
168
169        // Work out the hashed gradient indices of the four simplex corners
170        int ii = i & 255;
171        int jj = j & 255;
172        int kk = k & 255;
173
174        float n = 0;
175
176        // Calculate the contribution from the four corners
177        float t0 = 0.6f - x0 * x0 - y0 * y0 - z0 * z0;
178        if (t0 > 0) {
179            int[] gi0 = grad3[this.perm[ii + this.perm[jj + this.perm[kk]]] % 12];
180            n += t0 * t0 * t0 * t0 * (gi0[0] * x0 + gi0[1] * y0 + gi0[2] * z0);
181        }
182
183        float t1 = 0.6f - x1 * x1 - y1 * y1 - z1 * z1;
184        if (t1 > 0) {
185            int[] gi1 = grad3[this.perm[ii + i1 + this.perm[jj + j1 + this.perm[kk + k1]]] % 12];
186            n += t1 * t1 * t1 * t1 * (gi1[0] * x1 + gi1[1] * y1 + gi1[2] * z1);
187        }
188
189        float t2 = 0.6f - x2 * x2 - y2 * y2 - z2 * z2;
190        if (t2 > 0) {
191            int[] gi2 = grad3[this.perm[ii + i2 + this.perm[jj + j2 + this.perm[kk + k2]]] % 12];
192            n += t2 * t2 * t2 * t2 * (gi2[0] * x2 + gi2[1] * y2 + gi2[2] * z2);
193        }
194
195        float t3 = 0.6f - x3 * x3 - y3 * y3 - z3 * z3;
196        if (t3 > 0) {
197            int[] gi3 = grad3[this.perm[ii + 1 + this.perm[jj + 1 + this.perm[kk + 1]]] % 12];
198            n += t3 * t3 * t3 * t3 * (gi3[0] * x3 + gi3[1] * y3 + gi3[2] * z3);
199        }
200
201        // Add contributions from each corner to get the noise value.
202        // The result is scaled to stay just inside [-1,1]
203        return 32.0f * n;
204    }

206    @Override
207    public float getNoise2D(float x, float y) {
208        x += this.offsetX;
209        y += this.offsetY;
210
211        // Skew the input space to determine which simplex cell we're in
212        float s = (x + y) * F2; // Hairy factor for 2D
213        int i = (int) (x + s);
214        int j = (int) (y + s);
215        float t = (i + j) * G2;
216        // Unskew the cell origin back to (x,y) space
217        float x0 = x - (i - t); // The x,y distances from the cell origin
218        float y0 = y - (j - t);
219
220        // For the 2D case, the simplex shape is an equilateral triangle.
221        int i1 = 0;
222        int j1 = 0;
223        // Determine which simplex we are in.
224        if (x0 > y0) {
225            i1 = 1;
226            j1 = 0;
227        } // lower triangle, XY order: (0,0).(1,0).(1,1)
228        else {
229            i1 = 0;
230            j1 = 1;
231        }
232        // upper triangle, YX order: (0,0).(0,1).(1,1)
233
234        // A step of (1,0) in (i,j) means a step of (1-c,-c) in (x,y), and
235        // a step of (0,1) in (i,j) means a step of (-c,1-c) in (x,y), where
236        // c = (3-sqrt(3))/6
237
238        float x1 = x0 - i1 + G2; // Offsets for middle corner in (x,y) unskewed coords
239        float y1 = y0 - j1 + G2;
240        float x2 = x0 + G22; // Offsets for last corner in (x,y) unskewed coords
241        float y2 = y0 + G22;
242
243        // Work out the hashed gradient indices of the three simplex corners
244        int ii = i & 255;
245        int jj = j & 255;
246
247        float n = 0;
248
249        // Calculate the contribution from the three corners
250        float t0 = 0.5f - x0 * x0 - y0 * y0;
251        if (t0 > 0) {
252            int[] gi0 = grad3[this.perm[ii + this.perm[jj]] % 12];
253            n += t0 * t0 * t0 * t0 * (gi0[0] * x0 + gi0[1] * y0); // (x,y) of grad3 used for 2D gradient
254        }
255
256        float t1 = 0.5f - x1 * x1 - y1 * y1;
257        if (t1 > 0) {
258            int[] gi1 = grad3[this.perm[ii + i1 + this.perm[jj + j1]] % 12];
259            n += t1 * t1 * t1 * t1 * (gi1[0] * x1 + gi1[1] * y1);
260        }
261
262        float t2 = 0.5f - x2 * x2 - y2 * y2;
263        if (t2 > 0) {
264            int[] gi2 = grad3[this.perm[ii + 1 + this.perm[jj + 1]] % 12];
265            n += t2 * t2 * t2 * t2 * (gi2[0] * x2 + gi2[1] * y2);
266        }
267
268        // Add contributions from each corner to get the noise value.
269        // The result is scaled to return values in the interval [-1,1].
270        return 70.0f * n;
271    }
