001package cn.nukkit.level.generator.noise.nukkit.d;
002
003import cn.nukkit.math.NukkitRandom;
004
005/**
006 * @author MagicDroidX (Nukkit Project)
007 */
008public class PerlinD extends NoiseD {
009    public PerlinD(NukkitRandom random, double octaves, double persistence) {
010        this(random, octaves, persistence, 1);
011    }
012
013    public PerlinD(NukkitRandom random, double octaves, double persistence, double expansion) {
014        this.octaves = octaves;
015        this.persistence = persistence;
016        this.expansion = expansion;
017        this.offsetX = random.nextFloat() * 256;
018        this.offsetY = random.nextFloat() * 256;
019        this.offsetZ = random.nextFloat() * 256;
020        this.perm = new int[512];
021        for (int i = 0; i < 256; ++i) {
022            this.perm[i] = random.nextBoundedInt(256);
023        }
024        for (int i = 0; i < 256; ++i) {
025            int pos = random.nextBoundedInt(256 - i) + i;
026            int old = this.perm[i];
027            this.perm[i] = this.perm[pos];
028            this.perm[pos] = old;
029            this.perm[i + 256] = this.perm[i];
030        }
031    }
032
033    @Override
034    public double getNoise2D(double x, double y) {
035        return this.getNoise3D(x, y, 0);
036    }
037
038    @Override
039    public double getNoise3D(double x, double y, double z) {
040        x += this.offsetX;
041        y += this.offsetY;
042        z += this.offsetZ;
043
044        int floorX = (int) x;
045        int floorY = (int) y;
046        int floorZ = (int) z;
047
048        int X = floorX & 0xFF;
049        int Y = floorY & 0xFF;
050        int Z = floorZ & 0xFF;
051
052        x -= floorX;
053        y -= floorY;
054        z -= floorZ;
055
056        //Fade curves
057        //fX = fade(x);
058        //fY = fade(y);
059        //fZ = fade(z);
060        double fX = x * x * x * (x * (x * 6 - 15) + 10);
061        double fY = y * y * y * (y * (y * 6 - 15) + 10);
062        double fZ = z * z * z * (z * (z * 6 - 15) + 10);
063
064        //Cube corners
065        int A = this.perm[X] + Y;
066        int B = this.perm[X + 1] + Y;
067
068        int AA = this.perm[A] + Z;
069        int AB = this.perm[A + 1] + Z;
070        int BA = this.perm[B] + Z;
071        int BB = this.perm[B + 1] + Z;
072
073        double AA1 = grad(this.perm[AA], x, y, z);
074        double BA1 = grad(this.perm[BA], x - 1, y, z);
075        double AB1 = grad(this.perm[AB], x, y - 1, z);
076        double BB1 = grad(this.perm[BB], x - 1, y - 1, z);
077        double AA2 = grad(this.perm[AA + 1], x, y, z - 1);
078        double BA2 = grad(this.perm[BA + 1], x - 1, y, z - 1);
079        double AB2 = grad(this.perm[AB + 1], x, y - 1, z - 1);
080        double BB2 = grad(this.perm[BB + 1], x - 1, y - 1, z - 1);
081
082        double xLerp11 = AA1 + fX * (BA1 - AA1);
083
084        double zLerp1 = xLerp11 + fY * (AB1 + fX * (BB1 - AB1) - xLerp11);
085
086        double xLerp21 = AA2 + fX * (BA2 - AA2);
087
088        return zLerp1 + fZ * (xLerp21 + fY * (AB2 + fX * (BB2 - AB2) - xLerp21) - zLerp1);
089
090    }
091}
