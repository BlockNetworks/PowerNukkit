037    @Override
038    public void populate(ChunkManager level, int chunkX, int chunkZ, NukkitRandom random, FullChunk chunk) {
039        this.random = new Random();
040        this.random.setSeed(level.getSeed());
041        long worldLong1 = this.random.nextLong();
042        long worldLong2 = this.random.nextLong();
043
044        int size = this.checkAreaSize;
045
046        for (int x = chunkX - size; x <= chunkX + size; x++)
047            for (int z = chunkZ - size; z <= chunkZ + size; z++) {
048                long randomX = x * worldLong1;
049                long randomZ = z * worldLong2;
050                this.random.setSeed(randomX ^ randomZ ^ level.getSeed());
051                generateChunk(x, z, chunk);
052            }
053    }
054
055    protected void generateLargeCaveNode(long seed, FullChunk chunk, double x, double y, double z) {
056        generateCaveNode(seed, chunk, x, y, z, 1.0F + this.random.nextFloat() * 6.0F, 0.0F, 0.0F, -1, -1, 0.5D);
057    }

059    protected void generateCaveNode(long seed, FullChunk chunk, double x, double y, double z, float radius, float angelOffset, float angel, int angle, int maxAngle, double scale) {
060        int chunkX = chunk.getX();
061        int chunkZ = chunk.getZ();
062
063        double realX = chunkX * 16 + 8;
064        double realZ = chunkZ * 16 + 8;
065
066        float f1 = 0.0F;
067        float f2 = 0.0F;
068
069        Random localRandom = new Random(seed);
070
071        if (maxAngle <= 0) {
072            int checkAreaSize = this.checkAreaSize * 16 - 16;
073            maxAngle = checkAreaSize - localRandom.nextInt(checkAreaSize / 4);
074        }
075        boolean isLargeCave = false;
076
077        if (angle == -1) {
078            angle = maxAngle / 2;
079            isLargeCave = true;
080        }
081
082        int randomAngel = localRandom.nextInt(maxAngle / 2) + maxAngle / 4;
083        boolean bigAngel = localRandom.nextInt(6) == 0;
084
085        for (; angle < maxAngle; angle++) {
086            double offsetXZ = 1.5D + MathHelper.sin(angle * 3.141593F / maxAngle) * radius * 1.0F;
087            double offsetY = offsetXZ * scale;
088
089            float cos = MathHelper.cos(angel);
090            float sin = MathHelper.sin(angel);
091            x += MathHelper.cos(angelOffset) * cos;
092            y += sin;
093            z += MathHelper.sin(angelOffset) * cos;
094
095            if (bigAngel)
096                angel *= 0.92F;
097            else {
098                angel *= 0.7F;
099            }
100            angel += f2 * 0.1F;
101            angelOffset += f1 * 0.1F;
102
103            f2 *= 0.9F;
104            f1 *= 0.75F;
105            f2 += (localRandom.nextFloat() - localRandom.nextFloat()) * localRandom.nextFloat() * 2.0F;
106            f1 += (localRandom.nextFloat() - localRandom.nextFloat()) * localRandom.nextFloat() * 4.0F;
107
108            if ((!isLargeCave) && (angle == randomAngel) && (radius > 1.0F) && (maxAngle > 0)) {
109                generateCaveNode(localRandom.nextLong(), chunk, x, y, z, localRandom.nextFloat() * 0.5F + 0.5F, angelOffset - 1.570796F, angel / 3.0F, angle, maxAngle, 1.0D);
110                generateCaveNode(localRandom.nextLong(), chunk, x, y, z, localRandom.nextFloat() * 0.5F + 0.5F, angelOffset + 1.570796F, angel / 3.0F, angle, maxAngle, 1.0D);
111                return;
112            }
113            if ((!isLargeCave) && (localRandom.nextInt(4) == 0)) {
114                continue;
115            }
116
117            // Check if distance to working point (x and z) too larger than working radius (maybe ??)
118            double distanceX = x - realX;
119            double distanceZ = z - realZ;
120            double angelDiff = maxAngle - angle;
121            double newRadius = radius + 2.0F + 16.0F;
122            if (distanceX * distanceX + distanceZ * distanceZ - angelDiff * angelDiff > newRadius * newRadius) {
123                return;
124            }
125
126            //Boundaries check.
127            if ((x < realX - 16.0D - offsetXZ * 2.0D) || (z < realZ - 16.0D - offsetXZ * 2.0D) || (x > realX + 16.0D + offsetXZ * 2.0D) || (z > realZ + 16.0D + offsetXZ * 2.0D))
128                continue;
129
130
131            int xFrom = MathHelper.floor(x - offsetXZ) - chunkX * 16 - 1;
132            int xTo = MathHelper.floor(x + offsetXZ) - chunkX * 16 + 1;
133
134            int yFrom = MathHelper.floor(y - offsetY) - 1;
135            int yTo = MathHelper.floor(y + offsetY) + 1;
136
137            int zFrom = MathHelper.floor(z - offsetXZ) - chunkZ * 16 - 1;
138            int zTo = MathHelper.floor(z + offsetXZ) - chunkZ * 16 + 1;
139
140            if (xFrom < 0)
141                xFrom = 0;
142            if (xTo > 16)
143                xTo = 16;
144
145            if (yFrom < 1)
146                yFrom = 1;
147            if (yTo > this.worldHeightCap - 8) {
148                yTo = this.worldHeightCap - 8;
149            }
150            if (zFrom < 0)
151                zFrom = 0;
152            if (zTo > 16)
153                zTo = 16;
154
155            // Search for water
156            boolean waterFound = false;
157            for (int xx = xFrom; (!waterFound) && (xx < xTo); xx++) {
158                for (int zz = zFrom; (!waterFound) && (zz < zTo); zz++) {
159                    for (int yy = yTo + 1; (!waterFound) && (yy >= yFrom - 1); yy--) {
160                        if (yy >= 0 && yy < this.worldHeightCap) {
161                            int block = chunk.getBlockId(xx, yy, zz);
162                            if (block == Block.WATER || block == Block.STILL_WATER) {
163                                waterFound = true;
164                            }
165                            if ((yy != yFrom - 1) && (xx != xFrom) && (xx != xTo - 1) && (zz != zFrom) && (zz != zTo - 1))
166                                yy = yFrom;
167                        }
168                    }
169                }
170            }
171
172            if (waterFound) {
173                continue;
174            }
175
176            // Generate cave
177            for (int xx = xFrom; xx < xTo; xx++) {
178                double modX = (xx + chunkX * 16 + 0.5D - x) / offsetXZ;
179                for (int zz = zFrom; zz < zTo; zz++) {
180                    double modZ = (zz + chunkZ * 16 + 0.5D - z) / offsetXZ;
181
182                    boolean grassFound = false;
183                    if (modX * modX + modZ * modZ < 1.0D) {
184                        for (int yy = yTo; yy > yFrom; yy--) {
185                            double modY = ((yy - 1) + 0.5D - y) / offsetY;
186                            if ((modY > -0.7D) && (modX * modX + modY * modY + modZ * modZ < 1.0D)) {
187                                Biome biome = EnumBiome.getBiome(chunk.getBiomeId(xx, zz));
188                                if (!(biome instanceof CoveredBiome)) {
189                                    continue;
190                                }
191
192                                int material = chunk.getBlockId(xx, yy, zz);
193                                int materialAbove = chunk.getBlockId(xx, yy + 1, zz);
194                                if (material == Block.GRASS || material == Block.MYCELIUM) {
195                                    grassFound = true;
196                                }
197                                //TODO: check this
198//                                                              if (this.isSuitableBlock(material, materialAbove, biome))
199                                {
200                                    if (yy - 1 < 10) {
201                                        chunk.setBlock(xx, yy, zz, Block.LAVA);
202                                    } else {
203                                        chunk.setBlock(xx, yy, zz, Block.AIR);
204
205                                        // If grass was just deleted, try to
206                                        // move it down
207                                        if (grassFound && (chunk.getBlockId(xx, yy - 1, zz) == Block.DIRT)) {
208                                            chunk.setFullBlockId(xx, yy - 1, zz, ((CoveredBiome) biome).getSurfaceId(xx, yy - 1, zz));
209                                        }
210                                    }
211                                }
212                            }
213                        }
214                    }
215                }
216            }
217
218            if (isLargeCave) {
219                break;
220            }
221        }
222    }

224    protected void generateChunk(int chunkX, int chunkZ, FullChunk generatingChunkBuffer) {
225        int i = this.random.nextInt(this.random.nextInt(this.random.nextInt(caveFrequency) + 1) + 1);
226        if (evenCaveDistribution)
227            i = caveFrequency;
228        if (this.random.nextInt(100) >= caveRarity)
229            i = 0;
230
231        for (int j = 0; j < i; j++) {
232            double x = chunkX * 16 + this.random.nextInt(16);
233
234            double y;
235
236            if (evenCaveDistribution)
237                y = numberInRange(random, caveMinAltitude, caveMaxAltitude);
238            else
239                y = this.random.nextInt(this.random.nextInt(caveMaxAltitude - caveMinAltitude + 1) + 1) + caveMinAltitude;
240
241            double z = chunkZ * 16 + this.random.nextInt(16);
242
243            int count = caveSystemFrequency;
244            boolean largeCaveSpawned = false;
245            if (this.random.nextInt(100) <= individualCaveRarity) {
246                generateLargeCaveNode(this.random.nextLong(), generatingChunkBuffer, x, y, z);
247                largeCaveSpawned = true;
248            }
249
250            if ((largeCaveSpawned) || (this.random.nextInt(100) <= caveSystemPocketChance - 1)) {
251                count += numberInRange(random, caveSystemPocketMinSize, caveSystemPocketMaxSize);
252            }
253            while (count > 0) {
254                count--;
255                float f1 = this.random.nextFloat() * 3.141593F * 2.0F;
256                float f2 = (this.random.nextFloat() - 0.5F) * 2.0F / 8.0F;
257                float f3 = this.random.nextFloat() * 2.0F + this.random.nextFloat();
258
259                generateCaveNode(this.random.nextLong(), generatingChunkBuffer, x, y, z, f3, f1, f2, 0, 0, 1.0D);
260            }
261        }
262    }
263
264    public static int numberInRange(Random random, int min, int max) {
265        return min + random.nextInt(max - min + 1);
266    }
