001package cn.nukkit.level.generator.populator.impl;
002
003import cn.nukkit.api.NewRakNetOnly;
004import cn.nukkit.api.RemovedFromNewRakNet;
005import cn.nukkit.api.Since;
006import cn.nukkit.block.Block;
007import cn.nukkit.level.ChunkManager;
008import cn.nukkit.level.format.FullChunk;
009import cn.nukkit.level.generator.object.ore.OreType;
010import cn.nukkit.level.generator.populator.type.Populator;
011import cn.nukkit.math.NukkitMath;
012import cn.nukkit.math.NukkitRandom;
013
014/**
015 * @author DaPorkchop_
016 */
017public class PopulatorOre extends Populator {
018    private final int replaceId;
019    private OreType[] oreTypes = OreType.EMPTY_ARRAY;
020
021    /**
022     * @implNote Removed from the new-raknet branch
023     */
024    @RemovedFromNewRakNet
025    @Since("1.4.0.0-PN")
026    public PopulatorOre() {
027        this(Block.STONE);
028    }
029
030    /**
031     * @implNote Removed from the new-raknet branch
032     */
033    @RemovedFromNewRakNet
034    @Since("1.4.0.0-PN")
035    public PopulatorOre(int id) {
036        this.replaceId = id;
037    }
038
039    @NewRakNetOnly
040    public PopulatorOre(int replaceId, OreType[] oreTypes) {
041        this.replaceId = replaceId;
042        this.oreTypes = oreTypes;
043    }
044
045    @Override
046    public void populate(ChunkManager level, int chunkX, int chunkZ, NukkitRandom random, FullChunk chunk) {
047        int sx = chunkX << 4;
048        int ex = sx + 15;
049        int sz = chunkZ << 4;
050        int ez = sz + 15;
051        for (OreType type : this.oreTypes) {
052            for (int i = 0; i < type.clusterCount; i++) {
053                int x = NukkitMath.randomRange(random, sx, ex);
054                int z = NukkitMath.randomRange(random, sz, ez);
055                int y = NukkitMath.randomRange(random, type.minHeight, type.maxHeight);
056                if (level.getBlockIdAt(x, y, z) != replaceId) {
057                    continue;
058                }
059                type.spawn(level, random, replaceId, x, y, z);
060            }
061        }
062    }
063
064    /**
065     * @implNote Removed from the new-raknet branch
066     */
067    @RemovedFromNewRakNet
068    @Since("1.4.0.0-PN")
069    public void setOreTypes(OreType[] oreTypes) {
070        this.oreTypes = oreTypes;
071    }
072}
