001package cn.nukkit.level.generator.populator.impl;
002
003import cn.nukkit.level.format.FullChunk;
004import cn.nukkit.math.NukkitRandom;
005
006import java.util.concurrent.ThreadLocalRandom;
007
008/**
009 * @author Niall Lindsay (Niall7459, Nukkit Project)
010 */
011
012public class PopulatorTallSugarcane extends PopulatorSugarcane {
013    @Override
014    protected void placeBlock(int x, int y, int z, int id, FullChunk chunk, NukkitRandom random) {
015        int height = ThreadLocalRandom.current().nextInt(3) + 1;
016        for (int i = 0; i < height; i++)    {
017            chunk.setFullBlockId(x, y + i, z, id);
018        }
019    }
020}
