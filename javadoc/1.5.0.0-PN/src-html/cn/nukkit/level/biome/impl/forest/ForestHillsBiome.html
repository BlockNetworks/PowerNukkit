001package cn.nukkit.level.biome.impl.forest;
002
003/**
004 * @author DaPorkchop_ (Nukkit Project)
005 */
006public class ForestHillsBiome extends ForestBiome {
007    public ForestHillsBiome() {
008        this(TYPE_NORMAL);
009    }
010
011    public ForestHillsBiome(int type) {
012        super(type);
013
014        this.setBaseHeight(0.45f);
015        this.setHeightVariation(0.3f);
016    }
017
018    @Override
019    public String getName() {
020        switch (this.type) {
021            case TYPE_BIRCH:
022                return "Birch Forest Hills";
023            case TYPE_BIRCH_TALL:
024                return "Birch Forest Hills M";
025            default:
026                return "Forest Hills";
027        }
028    }
029}
