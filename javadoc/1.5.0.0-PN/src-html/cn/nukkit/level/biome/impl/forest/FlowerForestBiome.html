001package cn.nukkit.level.biome.impl.forest;
002
003import cn.nukkit.block.BlockDoublePlant;
004import cn.nukkit.block.BlockFlower;
005import cn.nukkit.level.generator.populator.impl.PopulatorFlower;
006
007/**
008 * @author DaPorkchop_ (Nukkit Project)
009 */
010public class FlowerForestBiome extends ForestBiome {
011    public FlowerForestBiome() {
012        this(TYPE_NORMAL);
013    }
014
015    public FlowerForestBiome(int type) {
016        super(type);
017
018        //see https://minecraft.gamepedia.com/Flower#Flower_biomes
019        PopulatorFlower flower = new PopulatorFlower();
020        flower.setBaseAmount(10);
021        flower.addType(DANDELION, 0);
022        flower.addType(RED_FLOWER, BlockFlower.TYPE_POPPY);
023        flower.addType(RED_FLOWER, BlockFlower.TYPE_ALLIUM);
024        flower.addType(RED_FLOWER, BlockFlower.TYPE_AZURE_BLUET);
025        flower.addType(RED_FLOWER, BlockFlower.TYPE_RED_TULIP);
026        flower.addType(RED_FLOWER, BlockFlower.TYPE_ORANGE_TULIP);
027        flower.addType(RED_FLOWER, BlockFlower.TYPE_WHITE_TULIP);
028        flower.addType(RED_FLOWER, BlockFlower.TYPE_PINK_TULIP);
029        flower.addType(RED_FLOWER, BlockFlower.TYPE_OXEYE_DAISY);
030        flower.addType(DOUBLE_PLANT, BlockDoublePlant.LILAC);
031        flower.addType(DOUBLE_PLANT, BlockDoublePlant.ROSE_BUSH);
032        flower.addType(DOUBLE_PLANT, BlockDoublePlant.PEONY);
033        this.addPopulator(flower);
034
035        this.setHeightVariation(0.4f);
036    }
037
038    @Override
039    public String getName() {
040        return this.type == TYPE_BIRCH ? "Birch Forest" : "Forest";
041    }
042}
