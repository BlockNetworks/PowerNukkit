001package cn.nukkit.event.player;
002
003import cn.nukkit.Player;
004import cn.nukkit.api.PowerNukkitOnly;
005import cn.nukkit.api.Since;
006import cn.nukkit.event.Cancellable;
007import cn.nukkit.event.HandlerList;
008import cn.nukkit.lang.TextContainer;
009
010public class PlayerKickEvent extends PlayerEvent implements Cancellable {
011    private static final HandlerList handlers = new HandlerList();
012
013    public enum Reason {
014        NEW_CONNECTION,
015        KICKED_BY_ADMIN,
016        NOT_WHITELISTED,
017        IP_BANNED,
018        NAME_BANNED,
019        INVALID_PVE,
020        LOGIN_TIMEOUT,
021        SERVER_FULL,
022        FLYING_DISABLED,
023        @PowerNukkitOnly @Since("1.4.0.0-PN") INVALID_PVP,
024        UNKNOWN;
025
026        @Override
027        public String toString() {
028            return this.name();
029        }
030    }
031
032    public static HandlerList getHandlers() {
033        return handlers;
034    }
035
036    protected TextContainer quitMessage;
037
038    protected final Reason reason;
039    protected final String reasonString;
040
041    @Deprecated
042    public PlayerKickEvent(Player player, String reason, String quitMessage) {
043        this(player, Reason.UNKNOWN, reason, new TextContainer(quitMessage));
044    }
045
046    @Deprecated
047    public PlayerKickEvent(Player player, String reason, TextContainer quitMessage) {
048        this(player, Reason.UNKNOWN, reason, quitMessage);
049    }
050
051    public PlayerKickEvent(Player player, Reason reason, TextContainer quitMessage) {
052        this(player, reason, reason.toString(), quitMessage);
053    }
054
055    public PlayerKickEvent(Player player, Reason reason, String quitMessage) {
056        this(player, reason, new TextContainer(quitMessage));
057    }
058
059    public PlayerKickEvent(Player player, Reason reason, String reasonString, TextContainer quitMessage) {
060        this.player = player;
061        this.quitMessage = quitMessage;
062        this.reason = reason;
063        this.reasonString = reason.name();
064    }
065
066    public String getReason() {
067        return reasonString;
068    }
069
070    public Reason getReasonEnum() {
071        return this.reason;
072    }
073
074    public TextContainer getQuitMessage() {
075        return quitMessage;
076    }
077
078    public void setQuitMessage(TextContainer quitMessage) {
079        this.quitMessage = quitMessage;
080    }
081
082    public void setQuitMessage(String joinMessage) {
083        this.setQuitMessage(new TextContainer(joinMessage));
084    }
085}
