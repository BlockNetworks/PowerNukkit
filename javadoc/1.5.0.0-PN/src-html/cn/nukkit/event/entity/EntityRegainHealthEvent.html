001package cn.nukkit.event.entity;
002
003import cn.nukkit.entity.Entity;
004import cn.nukkit.event.Cancellable;
005import cn.nukkit.event.HandlerList;
006
007/**
008 * @author MagicDroidX (Nukkit Project)
009 */
010public class EntityRegainHealthEvent extends EntityEvent implements Cancellable {
011    private static final HandlerList handlers = new HandlerList();
012
013    public static HandlerList getHandlers() {
014        return handlers;
015    }
016
017    public static final int CAUSE_REGEN = 0;
018    public static final int CAUSE_EATING = 1;
019    public static final int CAUSE_MAGIC = 2;
020    public static final int CAUSE_CUSTOM = 3;
021
022    private float amount;
023    private final int reason;
024
025    public EntityRegainHealthEvent(Entity entity, float amount, int regainReason) {
026        this.entity = entity;
027        this.amount = amount;
028        this.reason = regainReason;
029    }
030
031    public float getAmount() {
032        return amount;
033    }
034
035    public void setAmount(float amount) {
036        this.amount = amount;
037    }
038
039    public int getRegainReason() {
040        return reason;
041    }
042}
