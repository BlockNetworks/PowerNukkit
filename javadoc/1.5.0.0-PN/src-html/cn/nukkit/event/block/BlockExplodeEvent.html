019package cn.nukkit.event.block;
020
021import cn.nukkit.api.PowerNukkitOnly;
022import cn.nukkit.api.Since;
023import cn.nukkit.block.Block;
024import cn.nukkit.event.Cancellable;
025import cn.nukkit.event.HandlerList;
026import cn.nukkit.level.Position;
027
028import java.util.Set;
029
030/**
031 * @author joserobjr
032 * @since 2020-10-06
033 */
034@PowerNukkitOnly
035@Since("1.4.0.0-PN")
036public class BlockExplodeEvent extends BlockEvent implements Cancellable {
037
038    private static final HandlerList handlers = new HandlerList();
039
040    @PowerNukkitOnly
041    @Since("1.4.0.0-PN")
042    public static HandlerList getHandlers() {
043        return handlers;
044    }
045
046    @PowerNukkitOnly
047    @Since("1.4.0.0-PN")
048    protected final Position position;
049
050    @PowerNukkitOnly
051    @Since("1.4.0.0-PN")
052    protected Set<Block> blocks;
053
054    @PowerNukkitOnly
055    @Since("1.4.0.0-PN")
056    protected Set<Block> ignitions;
057    
058    @PowerNukkitOnly
059    @Since("1.4.0.0-PN")
060    protected double yield;
061
062    @PowerNukkitOnly
063    @Since("1.4.0.0-PN")
064    protected final double fireChance;
065
066    @PowerNukkitOnly
067    @Since("1.4.0.0-PN")
068    public BlockExplodeEvent(Block block, Position position, Set<Block> blocks, Set<Block> ignitions, double yield, double fireChance) {
069        super(block);
070        this.position = position;
071        this.blocks = blocks;
072        this.yield = yield;
073        this.ignitions = ignitions;
074        this.fireChance = fireChance;
075    }
076
077    @PowerNukkitOnly
078    @Since("1.4.0.0-PN")
079    public Position getPosition() {
080        return this.position;
081    }
082
083    @PowerNukkitOnly
084    @Since("1.4.0.0-PN")
085    public Set<Block> getAffectedBlocks() {
086        return this.blocks;
087    }
088
089    @PowerNukkitOnly
090    @Since("1.4.0.0-PN")
091    public void setAffectedBlocks(Set<Block> blocks) {
092        this.blocks = blocks;
093    }
094
095    @PowerNukkitOnly
096    @Since("1.4.0.0-PN")
097    public double getYield() {
098        return this.yield;
099    }
100
101    @PowerNukkitOnly
102    @Since("1.4.0.0-PN")
103    public void setYield(double yield) {
104        this.yield = yield;
105    }
106
107    @PowerNukkitOnly
108    @Since("1.4.0.0-PN")
109    public Set<Block> getIgnitions() {
110        return ignitions;
111    }
112
113    @PowerNukkitOnly
114    @Since("1.4.0.0-PN")
115    public void setIgnitions(Set<Block> ignitions) {
116        this.ignitions = ignitions;
117    }
118
119    @PowerNukkitOnly
120    @Since("1.4.0.0-PN")
121    public double getFireChance() {
122        return fireChance;
123    }
124}
