019package cn.nukkit.event.inventory;
020
021import cn.nukkit.Player;
022import cn.nukkit.api.PowerNukkitOnly;
023import cn.nukkit.api.Since;
024import cn.nukkit.event.Cancellable;
025import cn.nukkit.event.HandlerList;
026import cn.nukkit.inventory.SmithingInventory;
027import cn.nukkit.item.Item;
028
029import javax.annotation.Nonnull;
030
031/**
032 * @author joserobjr
033 * @since 2021-05-16
034 */
035@PowerNukkitOnly
036@Since("1.4.0.0-PN")
037public class SmithingTableEvent extends InventoryEvent implements Cancellable {
038    private static final HandlerList handlers = new HandlerList();
039
040    @PowerNukkitOnly
041    @Since("1.4.0.0-PN")
042    public static HandlerList getHandlers() {
043        return handlers;
044    }
045
046    private final @Nonnull Item equipmentItem;
047    private final @Nonnull Item resultItem;
048    private final @Nonnull Item ingredientItem;
049    private final @Nonnull Player player;
050
051    @PowerNukkitOnly
052    @Since("1.4.0.0-PN")
053    public SmithingTableEvent(SmithingInventory inventory, @Nonnull Item equipmentItem, @Nonnull Item resultItem, @Nonnull Item ingredientItem, @Nonnull Player player) {
054        super(inventory);
055        this.equipmentItem = equipmentItem;
056        this.resultItem = resultItem;
057        this.ingredientItem = ingredientItem;
058        this.player = player;
059    }
060
061    @PowerNukkitOnly
062    @Since("1.4.0.0-PN")
063    @Nonnull
064    public Item getEquipmentItem() {
065        return this.equipmentItem;
066    }
067
068    @PowerNukkitOnly
069    @Since("1.4.0.0-PN")
070    @Nonnull
071    public Item getResultItem() {
072        return this.resultItem;
073    }
074
075    @PowerNukkitOnly
076    @Since("1.4.0.0-PN")
077    @Nonnull
078    public Item getIngredientItem() {
079        return this.ingredientItem;
080    }
081
082    @PowerNukkitOnly
083    @Since("1.4.0.0-PN")
084    @Nonnull
085    public Player getPlayer() {
086        return this.player;
087    }
088}
