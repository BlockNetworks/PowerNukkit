019package cn.nukkit.event.inventory;
020
021import cn.nukkit.Player;
022import cn.nukkit.api.PowerNukkitOnly;
023import cn.nukkit.api.Since;
024import cn.nukkit.event.Cancellable;
025import cn.nukkit.event.HandlerList;
026import cn.nukkit.inventory.GrindstoneInventory;
027import cn.nukkit.item.Item;
028
029import javax.annotation.Nonnull;
030
031/**
032 * @author joserobjr
033 * @since 2021-03-21
034 */
035@PowerNukkitOnly
036@Since("1.4.0.0-PN")
037public class GrindstoneEvent extends InventoryEvent implements Cancellable {
038    private static final HandlerList handlers = new HandlerList();
039
040    @PowerNukkitOnly
041    @Since("1.4.0.0-PN")
042    public static HandlerList getHandlers() {
043        return handlers;
044    }
045
046    private final @Nonnull Item firstItem;
047    private final @Nonnull Item resultItem;
048    private final @Nonnull Item secondItem;
049    private int experienceDropped;
050    private final @Nonnull Player player;
051
052    @PowerNukkitOnly
053    @Since("1.4.0.0-PN")
054    public GrindstoneEvent(GrindstoneInventory inventory, @Nonnull Item firstItem, @Nonnull Item resultItem, @Nonnull Item secondItem, int cost, @Nonnull Player player) {
055        super(inventory);
056        this.firstItem = firstItem;
057        this.resultItem = resultItem;
058        this.secondItem = secondItem;
059        this.experienceDropped = cost;
060        this.player = player;
061    }
062
063    @PowerNukkitOnly
064    @Since("1.4.0.0-PN")
065    @Nonnull
066    public Item getFirstItem() {
067        return this.firstItem;
068    }
069
070    @PowerNukkitOnly
071    @Since("1.4.0.0-PN")
072    @Nonnull
073    public Item getResultItem() {
074        return this.resultItem;
075    }
076
077    @PowerNukkitOnly
078    @Since("1.4.0.0-PN")
079    @Nonnull
080    public Item getSecondItem() {
081        return this.secondItem;
082    }
083
084    @PowerNukkitOnly
085    @Since("1.4.0.0-PN")
086    public int getExperienceDropped() {
087        return this.experienceDropped;
088    }
089
090    @PowerNukkitOnly
091    @Since("1.4.0.0-PN")
092    public void setExperienceDropped(int experienceDropped) {
093        this.experienceDropped = experienceDropped;
094    }
095
096    @PowerNukkitOnly
097    @Since("1.4.0.0-PN")
098    @Nonnull
099    public Player getPlayer() {
100        return this.player;
101    }
102}
