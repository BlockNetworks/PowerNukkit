001package cn.nukkit.command.defaults;
002
003import cn.nukkit.command.Command;
004import cn.nukkit.command.CommandSender;
005import cn.nukkit.lang.TranslationContainer;
006
007/**
008 * @author xtypr
009 * @since 2015/11/13
010 */
011public class SaveOffCommand extends VanillaCommand {
012
013    public SaveOffCommand(String name) {
014        super(name, "%nukkit.command.saveoff.description", "%commands.save-off.usage");
015        this.setPermission("nukkit.command.save.disable");
016        this.commandParameters.clear();
017    }
018
019    @Override
020    public boolean execute(CommandSender sender, String commandLabel, String[] args) {
021        if (!this.testPermission(sender)) {
022            return true;
023        }
024        sender.getServer().setAutoSave(false);
025        Command.broadcastCommandMessage(sender, new TranslationContainer("commands.save.disabled"));
026        return true;
027    }
028}
