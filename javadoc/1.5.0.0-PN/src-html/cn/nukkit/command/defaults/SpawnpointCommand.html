001package cn.nukkit.command.defaults;
002
003import cn.nukkit.Player;
004import cn.nukkit.command.Command;
005import cn.nukkit.command.CommandSender;
006import cn.nukkit.command.data.CommandParamType;
007import cn.nukkit.command.data.CommandParameter;
008import cn.nukkit.lang.TranslationContainer;
009import cn.nukkit.level.Level;
010import cn.nukkit.level.Position;
011import cn.nukkit.utils.TextFormat;
012
013import java.text.DecimalFormat;
014
015/**
016 * @author xtypr
017 * @since 2015/12/13
018 */
019public class SpawnpointCommand extends VanillaCommand {
020    public SpawnpointCommand(String name) {
021        super(name, "%nukkit.command.spawnpoint.description", "%commands.spawnpoint.usage");
022        this.setPermission("nukkit.command.spawnpoint");
023        this.commandParameters.clear();
024        this.commandParameters.put("default", new CommandParameter[]{
025                CommandParameter.newType("player", true, CommandParamType.TARGET),
026                CommandParameter.newType("spawnPos", true, CommandParamType.POSITION),
027        });
028    }
029
030    @Override
031    public boolean execute(CommandSender sender, String commandLabel, String[] args) {
032        if (!this.testPermission(sender)) {
033            return true;
034        }
035        Player target;
036        if (args.length == 0) {
037            if (sender instanceof Player) {
038                target = (Player) sender;
039            } else {
040                sender.sendMessage(new TranslationContainer("commands.generic.ingame"));
041                return true;
042            }
043        } else {
044            target = sender.getServer().getPlayer(args[0]);
045            if (target == null) {
046                sender.sendMessage(new TranslationContainer(TextFormat.RED + "%commands.generic.player.notFound"));
047                return true;
048            }
049        }
050        Level level = target.getLevel();
051        DecimalFormat round2 = new DecimalFormat("##0.00");
052        if (args.length == 4) {
053            if (level != null) {
054                int x;
055                int y;
056                int z;
057                try {
058                    x = Integer.parseInt(args[1]);
059                    y = Integer.parseInt(args[2]);
060                    z = Integer.parseInt(args[3]);
061                } catch (NumberFormatException e1) {
062                    sender.sendMessage(new TranslationContainer("commands.generic.usage", this.usageMessage));
063                    return true;
064                }
065                if (y < 0) y = 0;
066                if (y > 256) y = 256;
067                target.setSpawn(new Position(x, y, z, level));
068                Command.broadcastCommandMessage(sender, new TranslationContainer("commands.spawnpoint.success", target.getName(),
069                        round2.format(x),
070                        round2.format(y),
071                        round2.format(z)));
072                return true;
073            }
074        } else if (args.length <= 1) {
075            if (sender instanceof Player) {
076                Position pos = (Position) sender;
077                target.setSpawn(pos);
078                Command.broadcastCommandMessage(sender, new TranslationContainer("commands.spawnpoint.success", target.getName(),
079                        round2.format(pos.x),
080                        round2.format(pos.y),
081                        round2.format(pos.z)));
082                return true;
083            } else {
084                sender.sendMessage(new TranslationContainer("commands.generic.ingame"));
085                return true;
086            }
087        }
088        sender.sendMessage(new TranslationContainer("commands.generic.usage", this.usageMessage));
089        return true;
090    }
091}
