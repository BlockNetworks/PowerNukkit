067        switch (this.sourceType) {
068        case SOURCE_CONTAINER:

070            break;
071        case SOURCE_WORLD: 
072            this.unknown = packet.getUnsignedVarInt();
073            break;
074        case SOURCE_CREATIVE: 
075            break;
076        case SOURCE_CRAFT_SLOT: 
077        case SOURCE_TODO:

080            case SOURCE_TYPE_CRAFTING_RESULT:

093                break;
094            }
095            break;
096        }
097        this.inventorySlot = (int) packet.getUnsignedVarInt();
