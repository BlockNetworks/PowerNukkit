001package cn.nukkit.utils;
002
003import cn.nukkit.api.API;
004
005import java.util.HashMap;
006import java.util.Map;
007
008/**
009 * Helper class of Minecart variants
010 * <p>
011 * By Adam Matthew
012 * Creation time: 2017/7/17 19:55.
013 */
014@API(usage = API.Usage.STABLE, definition = API.Definition.INTERNAL)
015public enum MinecartType {
016    /**
017     * Represents an empty vehicle.
018     */
019    MINECART_EMPTY(0, false, "Minecart"),
020    /**
021     * Represents a chest holder.
022     */
023    MINECART_CHEST(1, true, "Minecart with Chest"),
024    /**
025     * Represents a furnace minecart.
026     */
027    MINECART_FURNACE(2, true, "Minecart with Furnace"),
028    /**
029     * Represents a TNT minecart.
030     */
031    MINECART_TNT(3, true, "Minecart with TNT"),
032    /**
033     * Represents a mob spawner minecart.
034     */
035    MINECART_MOB_SPAWNER(4, true, "Minecart with Mob Spawner"),
036    /**
037     * Represents a hopper minecart.
038     */
039    MINECART_HOPPER(5, true, "Minecart with Hopper"),
040    /**
041     * Represents a command block minecart.
042     */
043    MINECART_COMMAND_BLOCK(6, true, "Minecart with Command Block"),
044    /**
045     * Represents an unknown minecart.
046     */
047    MINECART_UNKNOWN(-1, false, "Unknown Minecart");
048
049    private final int type;
050    private final boolean hasBlockInside;
051    private final String realName;
052    private static final Map<Integer, MinecartType> TYPES = new HashMap<>();
053
054    static {
055        for (MinecartType var3 : values()) {
056            TYPES.put(var3.getId(), var3);
057        }
058    }
059
060    MinecartType(int number, boolean hasBlockInside, String name) {
061        type = number;
062        this.hasBlockInside = hasBlockInside;
063        realName = name;
064    }
065
066    /**
067     * Get the variants of the current minecart
068     *
069     * @return Integer
070     */
071    public int getId() {
072        return type;
073    }
074
075    /**
076     * Get the name of the minecart variants
077     *
078     * @return String
079     */
080    public String getName() {
081        return realName;
082    }
083
084    /**
085     * Gets if the minecart contains block
086     *
087     * @return Boolean
088     */
089    public boolean hasBlockInside() {
090        return hasBlockInside;
091    }
092
093    /**
094     * Returns of an instance of Minecart-variants
095     *
096     * @param types The number of minecart
097     * @return Integer
098     */
099    public static MinecartType valueOf(int types) {
100        MinecartType what = TYPES.get(types);
101        return what == null ? MINECART_UNKNOWN : what;
102    }
103}
