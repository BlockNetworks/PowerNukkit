001package cn.nukkit.inventory;
002
003import cn.nukkit.Player;
004import cn.nukkit.item.Item;
005import cn.nukkit.level.Position;
006
007/**
008 * author: Rover656
009 */
010public class BeaconInventory extends FakeBlockUIComponent {
011
012    public BeaconInventory(PlayerUIInventory playerUI, Position position) {
013        super(playerUI, InventoryType.BEACON, 27, position);
014    }
015
016    @Override
017    public void onClose(Player who) {
018        super.onClose(who);
019
020        Item[] drops = who.getInventory().addItem(this.getItem(0));
021        for (Item drop : drops) {
022            if (!who.dropItem(drop)) {
023                this.getHolder().getLevel().dropItem(this.getHolder().add(0.5, 0.5, 0.5), drop);
024            }
025        }
026        
027        this.clear(0);
028    }
029}
