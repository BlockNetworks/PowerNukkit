001package cn.nukkit.inventory;
002
003import cn.nukkit.Player;
004import cn.nukkit.item.Item;
005import cn.nukkit.level.Position;
006
007public class StonecutterInventory extends FakeBlockUIComponent {
008    public StonecutterInventory(PlayerUIInventory playerUI, Position position) {
009        super(playerUI, InventoryType.STONECUTTER, 3, position);
010    }
011    
012    @Override
013    public void onOpen(Player who) {
014        who.craftingType = Player.CRAFTING_STONECUTTER;
015        sendContents(who);
016    }
017
018    @Override
019    public void onClose(Player who) {
020        super.onClose(who);
021        who.craftingType = Player.CRAFTING_SMALL;
022
023        Item[] drops = who.getInventory().addItem(this.getItem(0));
024        for (Item drop : drops) {
025            if (!who.dropItem(drop)) {
026                this.getHolder().getLevel().dropItem(this.getHolder().add(0.5, 0.5, 0.5), drop);
027            }
028        }
029
030        this.clear(0);
031        who.resetCraftingGridType();
032    }
033}
