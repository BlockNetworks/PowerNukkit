001package cn.nukkit.inventory;
002
003import cn.nukkit.Player;
004import cn.nukkit.item.Item;
005import cn.nukkit.network.protocol.InventoryContentPacket;
006import cn.nukkit.network.protocol.InventorySlotPacket;
007import cn.nukkit.network.protocol.types.ContainerIds;
008
009import java.util.HashMap;
010
011public class PlayerUIInventory extends BaseInventory {
012    private final Player player;
013
014    private final PlayerCursorInventory cursorInventory;
015    private final CraftingGrid craftingGrid;
016    private final BigCraftingGrid bigCraftingGrid;
017
018    public PlayerUIInventory(Player player) {
019        super(player, InventoryType.UI, new HashMap<>(), 51);
020        this.player = player;
021
022        this.cursorInventory = new PlayerCursorInventory(this);
023        this.craftingGrid = new CraftingGrid(this);
024        this.bigCraftingGrid = new BigCraftingGrid(this);
025    }
026
027    public PlayerCursorInventory getCursorInventory() {
028        return cursorInventory;
029    }
030
031    public CraftingGrid getCraftingGrid() {
032        return craftingGrid;
033    }
034
035    public BigCraftingGrid getBigCraftingGrid() {
036        return bigCraftingGrid;
037    }
038
039    @Override
040    public void setSize(int size) {
041        throw new UnsupportedOperationException("UI size is immutable");
042    }
043
044    @Override
045    public void sendSlot(int index, Player... target) {
046        InventorySlotPacket pk = new InventorySlotPacket();
047        pk.slot = index;
048        pk.item = this.getItem(index);
049
050        for (Player p : target) {
051            if (p == this.getHolder()) {
052                pk.inventoryId = ContainerIds.UI;
053                p.dataPacket(pk);
054            } else {
055                int id;
056
057                if ((id = p.getWindowId(this)) == ContainerIds.NONE) {
058                    this.close(p);
059                    continue;
060                }
061                pk.inventoryId = id;
062                p.dataPacket(pk);
063            }
064        }
065    }
066
067    @Override
068    public void sendContents(Player... target) {
069        InventoryContentPacket pk = new InventoryContentPacket();
070        pk.slots = new Item[this.getSize()];
071        for (int i = 0; i < this.getSize(); ++i) {
072            pk.slots[i] = this.getItem(i);
073        }
074
075        for (Player p : target) {
076            if (p == this.getHolder()) {
077                pk.inventoryId = ContainerIds.UI;
078                p.dataPacket(pk);
079            } else {
080                int id;
081
082                if ((id = p.getWindowId(this)) == ContainerIds.NONE) {
083                    this.close(p);
084                    continue;
085                }
086                pk.inventoryId = id;
087                p.dataPacket(pk);
088            }
089        }
090    }
091    
092    @Override
093    public void onSlotChange(int index, Item before, boolean send) {
094        switch (player.craftingType) {
095            case Player.CRAFTING_GRINDSTONE:
096                if (index >= GrindstoneInventory.OFFSET) {
097                    Inventory inventory = player.getWindowById(Player.ANVIL_WINDOW_ID);
098                    if (inventory instanceof GrindstoneInventory && (index == 50 || index - GrindstoneInventory.OFFSET < inventory.getSize())) {
099                        inventory.onSlotChange(index == 50 ? 2 : index - GrindstoneInventory.OFFSET, before, send);
100                    }
101                }
102            case Player.CRAFTING_ANVIL:
103                if (index >= AnvilInventory.OFFSET) {
104                    Inventory inventory = player.getWindowById(Player.ANVIL_WINDOW_ID);
105                    if (inventory instanceof AnvilInventory && (index == 50 || index - AnvilInventory.OFFSET < inventory.getSize())) {
106                        inventory.onSlotChange(index == 50 ? 2 : index - AnvilInventory.OFFSET, before, send);
107                    }
108                }
109                return;
110        }
111        super.onSlotChange(index, before, send);
112    }
113    
114    public void onSlotChangeBase(int index, Item before, boolean send) {
115        super.onSlotChange(index, before, send);
116    }
117    
118    @Override
119    public int getSize() {
120        return 51;
121    }
122
123    @Override
124    public Player getHolder() {
125        return player;
126    }
127}
