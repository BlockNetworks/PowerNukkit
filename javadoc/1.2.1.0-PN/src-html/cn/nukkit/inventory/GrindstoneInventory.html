001package cn.nukkit.inventory;
002
003import cn.nukkit.Player;
004import cn.nukkit.item.Item;
005import cn.nukkit.level.Position;
006
007public class GrindstoneInventory extends FakeBlockUIComponent {
008    public static final int OFFSET = 16;
009    private static final int SLOT_FIRST_ITEM = 0;
010    private static final int SLOT_SECOND_ITEM = 1;
011    private static final int SLOT_RESULT = 50 - OFFSET;
012
013    public GrindstoneInventory(PlayerUIInventory playerUI, Position position) {
014        super(playerUI, InventoryType.GRINDSTONE, OFFSET, position);
015    }
016
017    @Override
018    public void close(Player who) {
019        onClose(who);
020    }
021
022    @Override
023    public void onClose(Player who) {
024        super.onClose(who);
025        who.craftingType = Player.CRAFTING_SMALL;
026
027        Item[] drops = new Item[]{ getFirstItem(), getSecondItem() };
028        drops = who.getInventory().addItem(drops);
029        for (Item drop : drops) {
030            if (!who.dropItem(drop)) {
031                this.getHolder().getLevel().dropItem(this.getHolder().add(0.5, 0.5, 0.5), drop);
032            }
033        }
034
035        clear(SLOT_FIRST_ITEM);
036        clear(SLOT_SECOND_ITEM);
037
038        who.resetCraftingGridType();
039    }
040
041    @Override
042    public void onOpen(Player who) {
043        super.onOpen(who);
044        who.craftingType = Player.CRAFTING_GRINDSTONE;
045    }
046    
047    public Item getFirstItem() {
048        return getItem(SLOT_FIRST_ITEM);
049    }
050    
051    public Item getSecondItem() {
052        return getItem(SLOT_SECOND_ITEM);
053    }
054    
055    public Item getResult() {
056        return getItem(2);
057    }
058    
059    public boolean setFirstItem(Item item, boolean send) {
060        return setItem(SLOT_FIRST_ITEM, item, send);
061    }
062    
063    public boolean setFirstItem(Item item) {
064        return setFirstItem(item, true);
065    }
066    
067    public boolean setSecondItem(Item item, boolean send) {
068        return setItem(SLOT_SECOND_ITEM, item, send);
069    }
070    
071    public boolean setSecondItem(Item item) {
072        return setSecondItem(item, true);
073    }
074    
075    public boolean setResult(Item item, boolean send) {
076        return setItem(2, item, send);
077    }
078    
079    public boolean setResult(Item item) {
080        return setResult(item, true);
081    }
082    
083    @Override
084    public void onSlotChange(int index, Item before, boolean send) {
085        try {
086            if (index > 1) {
087                return;
088            }
089            Item firstItem = getFirstItem();
090            Item secondItem = getSecondItem();
091            if (!firstItem.isNull() && !secondItem.isNull() && firstItem.getId() != secondItem.getId()) {
092                setResult(Item.get(0));
093                return;
094            }
095    
096            if (firstItem.isNull()) {
097                Item air = firstItem;
098                firstItem = secondItem;
099                secondItem = air;
100            }
101    
102            if (firstItem.isNull()) {
103                setResult(Item.get(0));
104                return;
105            }
106            
107            Item result = firstItem.clone();
108            result.clearNamedTag();
109            if (!secondItem.isNull()) {
110                int first = firstItem.getMaxDurability() - firstItem.getDamage();
111                int second = secondItem.getMaxDurability() - secondItem.getDamage();
112                int reduction = first + second + firstItem.getMaxDurability() * 5 / 100;
113                int resultingDamage = Math.max(firstItem.getMaxDurability() - reduction + 1, 0);
114                result.setDamage(resultingDamage);
115            }
116            setResult(result);
117        } finally {
118            super.onSlotChange(index, before, send);
119        }
120    }
121    
122    @Override
123    public Item getItem(int index) {
124        if (index < 0 || index > 3) {
125            return Item.get(0);
126        }
127        if (index == 2) {
128            index = SLOT_RESULT;
129        }
130        
131        return super.getItem(index);
132    }
133    
134    @Override
135    public boolean setItem(int index, Item item, boolean send) {
136        if (index < 0 || index > 3) {
137            return false;
138        }
139        
140        if (index == 2) {
141            index = SLOT_RESULT;
142        }
143        
144        return super.setItem(index, item, send);
145    }
146}
