001package cn.nukkit.inventory;
002
003import cn.nukkit.Player;
004import cn.nukkit.item.Item;
005import cn.nukkit.level.Position;
006
007
008/**
009 * author: MagicDroidX
010 * Nukkit Project
011 */
012public class EnchantInventory extends FakeBlockUIComponent {
013
014    public EnchantInventory(PlayerUIInventory playerUI, Position position) {
015        super(playerUI, InventoryType.ENCHANT_TABLE, 14, position);
016    }
017
018    @Override
019    public void onOpen(Player who) {
020        super.onOpen(who);
021        who.craftingType = Player.CRAFTING_ENCHANT;
022    }
023
024    @Override
025    public void onClose(Player who) {
026        super.onClose(who);
027        who.craftingType = Player.CRAFTING_SMALL;
028        Item[] drops = new Item[]{ getItem(0), getItem(1) };
029        drops = who.getInventory().addItem(drops);
030        for (Item drop : drops) {
031            if (!who.dropItem(drop)) {
032                this.getHolder().getLevel().dropItem(this.getHolder().add(0.5, 0.5, 0.5), drop);
033            }
034        }
035
036        clear(0);
037        clear(1);
038        who.resetCraftingGridType();
039    }
040}
