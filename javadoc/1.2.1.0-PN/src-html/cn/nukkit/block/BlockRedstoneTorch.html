001package cn.nukkit.block;
002
003import cn.nukkit.Player;
004import cn.nukkit.event.redstone.RedstoneUpdateEvent;
005import cn.nukkit.item.Item;
006import cn.nukkit.level.Level;
007import cn.nukkit.math.BlockFace;
008import cn.nukkit.math.Vector3;
009import cn.nukkit.utils.BlockColor;
010
011/**
012 * author: Angelic47
013 * Nukkit Project
014 */
015public class BlockRedstoneTorch extends BlockTorch {
016
017    public BlockRedstoneTorch() {
018        this(0);
019    }
020
021    public BlockRedstoneTorch(int meta) {
022        super(meta);
023    }
024
025    @Override
026    public String getName() {
027        return "Redstone Torch";
028    }
029
030    @Override
031    public int getId() {
032        return REDSTONE_TORCH;
033    }
034
035    @Override
036    public int getLightLevel() {
037        return 7;
038    }
039
040    @Override
041    public boolean place(Item item, Block block, Block target, BlockFace face, double fx, double fy, double fz, Player player) {
042        if (!super.place(item, block, target, face, fx, fy, fz, player)) {
043            return false;
044        }
045
046        if (this.level.getServer().isRedstoneEnabled()) {
047            if (!checkState()) {
048                BlockFace facing = getBlockFace().getOpposite();
049                Vector3 pos = getLocation();
050
051                for (BlockFace side : BlockFace.values()) {
052                    if (facing == side) {
053                        continue;
054                    }
055
056                    this.level.updateAroundRedstone(pos.getSide(side), null);
057                }
058            }
059
060            checkState();
061        }
062
063        return true;
064    }
065
066    @Override
067    public int getWeakPower(BlockFace side) {
068        return getBlockFace() != side ? 15 : 0;
069    }
070
071    @Override
072    public int getStrongPower(BlockFace side) {
073        return side == BlockFace.DOWN ? this.getWeakPower(side) : 0;
074    }
075
076    @Override
077    public boolean onBreak(Item item) {
078        super.onBreak(item);
079
080        Vector3 pos = getLocation();
081
082        BlockFace face = getBlockFace().getOpposite();
083
084        if (this.level.getServer().isRedstoneEnabled()) {
085            for (BlockFace side : BlockFace.values()) {
086                if (side == face) {
087                    continue;
088                }
089
090                this.level.updateAroundRedstone(pos.getSide(side), null);
091            }
092        }
093        return true;
094    }
095
096    @Override
097    public int onUpdate(int type) {
098        if (super.onUpdate(type) == 0) {
099            if (!this.level.getServer().isRedstoneEnabled()) {
100                return 0;
101            }
102
103            if (type == Level.BLOCK_UPDATE_NORMAL || type == Level.BLOCK_UPDATE_REDSTONE) {
104                this.level.scheduleUpdate(this, tickRate());
105            } else if (type == Level.BLOCK_UPDATE_SCHEDULED) {
106                RedstoneUpdateEvent ev = new RedstoneUpdateEvent(this);
107                getLevel().getServer().getPluginManager().callEvent(ev);
108
109                if (ev.isCancelled()) {
110                    return 0;
111                }
112
113                if (checkState()) {
114                    return 1;
115                }
116            }
117        }
118
119        return 0;
120    }
121
122    private boolean checkState() {
123        if (isPoweredFromSide()) {
124            BlockFace face = getBlockFace().getOpposite();
125            Vector3 pos = getLocation();
126
127            this.level.setBlock(pos, Block.get(BlockID.UNLIT_REDSTONE_TORCH, getDamage()), false, true);
128
129            for (BlockFace side : BlockFace.values()) {
130                if (side == face) {
131                    continue;
132                }
133
134                this.level.updateAroundRedstone(pos.getSide(side), null);
135            }
136
137            return true;
138        }
139
140        return false;
141    }
142
143    protected boolean isPoweredFromSide() {
144        BlockFace face = getBlockFace().getOpposite();
145        return this.level.isSidePowered(this.getLocation().getSide(face), face);
146    }
147
148    @Override
149    public int tickRate() {
150        return 2;
151    }
152
153    @Override
154    public boolean isPowerSource() {
155        return true;
156    }
157
158    @Override
159    public BlockColor getColor() {
160        return BlockColor.AIR_BLOCK_COLOR;
161    }
162}
