001package cn.nukkit.block;
002
003import cn.nukkit.Player;
004import cn.nukkit.blockentity.BlockEntity;
005import cn.nukkit.blockentity.BlockEntityJukebox;
006import cn.nukkit.item.Item;
007import cn.nukkit.item.ItemBlock;
008import cn.nukkit.item.ItemRecord;
009import cn.nukkit.math.BlockFace;
010import cn.nukkit.nbt.tag.CompoundTag;
011import cn.nukkit.nbt.tag.ListTag;
012import cn.nukkit.utils.BlockColor;
013
014/**
015 * Created by CreeperFace on 7.8.2017.
016 */
017public class BlockJukebox extends BlockSolid {
018
019    public BlockJukebox() {
020    }
021
022    @Override
023    public String getName() {
024        return "Jukebox";
025    }
026
027    @Override
028    public int getId() {
029        return JUKEBOX;
030    }
031
032    @Override
033    public boolean canBeActivated() {
034        return true;
035    }
036
037    @Override
038    public Item toItem() {
039        return new ItemBlock(this, 0);
040    }
041
042    @Override
043    public boolean onActivate(Item item, Player player) {
044        BlockEntity blockEntity = this.getLevel().getBlockEntity(this);
045        if (!(blockEntity instanceof BlockEntityJukebox)) {
046            blockEntity = this.createBlockEntity();
047        }
048
049        BlockEntityJukebox jukebox = (BlockEntityJukebox) blockEntity;
050        if (jukebox.getRecordItem().getId() != 0) {
051            jukebox.dropItem();
052        } else if (item instanceof ItemRecord) {
053            jukebox.setRecordItem(item);
054            jukebox.play();
055            player.getInventory().decreaseCount(player.getInventory().getHeldItemIndex());
056        }
057
058        return false;
059    }
060
061    @Override
062    public boolean place(Item item, Block block, Block target, BlockFace face, double fx, double fy, double fz, Player player) {
063        if (super.place(item, block, target, face, fx, fy, fz, player)) {
064            createBlockEntity();
065            return true;
066        }
067
068        return false;
069    }
070
071    @Override
072    public boolean onBreak(Item item) {
073        if (super.onBreak(item)) {
074            BlockEntity blockEntity = this.level.getBlockEntity(this);
075
076            if (blockEntity instanceof BlockEntityJukebox) {
077                ((BlockEntityJukebox) blockEntity).dropItem();
078            }
079            return true;
080        }
081
082        return false;
083    }
084
085    private BlockEntity createBlockEntity() {
086        CompoundTag nbt = new CompoundTag()
087                .putList(new ListTag<>("Items"))
088                .putString("id", BlockEntity.JUKEBOX)
089                .putInt("x", getFloorX())
090                .putInt("y", getFloorY())
091                .putInt("z", getFloorZ());
092
093        return BlockEntity.createBlockEntity(BlockEntity.JUKEBOX, this.level.getChunk(getFloorX() >> 4, getFloorZ() >> 4), nbt);
094    }
095
096    @Override
097    public BlockColor getColor() {
098        return BlockColor.DIRT_BLOCK_COLOR;
099    }
100}
