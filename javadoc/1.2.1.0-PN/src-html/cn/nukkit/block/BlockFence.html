001package cn.nukkit.block;
002
003import cn.nukkit.item.Item;
004import cn.nukkit.item.ItemBlock;
005import cn.nukkit.item.ItemTool;
006import cn.nukkit.math.AxisAlignedBB;
007import cn.nukkit.math.SimpleAxisAlignedBB;
008import cn.nukkit.utils.BlockColor;
009
010/**
011 * Created on 2015/12/7 by xtypr.
012 * Package cn.nukkit.block in project Nukkit .
013 */
014public class BlockFence extends BlockTransparentMeta {
015
016    public static final int FENCE_OAK = 0;
017    public static final int FENCE_SPRUCE = 1;
018    public static final int FENCE_BIRCH = 2;
019    public static final int FENCE_JUNGLE = 3;
020    public static final int FENCE_ACACIA = 4;
021    public static final int FENCE_DARK_OAK = 5;
022
023    public BlockFence() {
024        this(0);
025    }
026
027    public BlockFence(int meta) {
028        super(meta);
029    }
030
031    @Override
032    public int getId() {
033        return FENCE;
034    }
035
036    @Override
037    public double getHardness() {
038        return 2;
039    }
040
041    @Override
042    public int getWaterloggingLevel() {
043        return 1;
044    }
045
046    @Override
047    public double getResistance() {
048        return 15;
049    }
050
051    @Override
052    public int getToolType() {
053        return ItemTool.TYPE_AXE;
054    }
055
056    @Override
057    public String getName() {
058        String[] names = new String[]{
059                "Oak Fence",
060                "Spruce Fence",
061                "Birch Fence",
062                "Jungle Fence",
063                "Acacia Fence",
064                "Dark Oak Fence",
065                "",
066                ""
067        };
068        return names[this.getDamage() & 0x07];
069    }
070
071    @Override
072    protected AxisAlignedBB recalculateBoundingBox() {
073        boolean north = this.canConnect(this.north());
074        boolean south = this.canConnect(this.south());
075        boolean west = this.canConnect(this.west());
076        boolean east = this.canConnect(this.east());
077        double n = north ? 0 : 0.375;
078        double s = south ? 1 : 0.625;
079        double w = west ? 0 : 0.375;
080        double e = east ? 1 : 0.625;
081        return new SimpleAxisAlignedBB(
082                this.x + w,
083                this.y,
084                this.z + n,
085                this.x + e,
086                this.y + 1.5,
087                this.z + s
088        );
089    }
090
091    @Override
092    public int getBurnChance() {
093        return 5;
094    }
095
096    @Override
097    public int getBurnAbility() {
098        return 20;
099    }
100
101    public boolean canConnect(Block block) {
102        return (block instanceof BlockFence || block instanceof BlockFenceGate) || block.isSolid() && !block.isTransparent();
103    }
104
105    @Override
106    public BlockColor getColor() {
107        switch (this.getDamage() & 0x07) {
108            default:
109            case BlockFence.FENCE_OAK: //OAK
110                return BlockColor.WOOD_BLOCK_COLOR;
111            case BlockFence.FENCE_SPRUCE: //SPRUCE
112                return BlockColor.SPRUCE_BLOCK_COLOR;
113            case BlockFence.FENCE_BIRCH: //BIRCH
114                return BlockColor.SAND_BLOCK_COLOR;
115            case BlockFence.FENCE_JUNGLE: //JUNGLE
116                return BlockColor.DIRT_BLOCK_COLOR;
117            case BlockFence.FENCE_ACACIA: //ACACIA
118                return BlockColor.ORANGE_BLOCK_COLOR;
119            case BlockFence.FENCE_DARK_OAK: //DARK OAK
120                return BlockColor.BROWN_BLOCK_COLOR;
121        }
122    }
123
124    @Override
125    public Item toItem() {
126        return new ItemBlock(this, this.getDamage());
127    }
128}
