001package cn.nukkit.block;
002
003import cn.nukkit.item.Item;
004import cn.nukkit.item.ItemBlock;
005import cn.nukkit.item.ItemTool;
006import cn.nukkit.utils.BlockColor;
007
008public class BlockDoubleSlabStone4 extends BlockDoubleSlabStone {
009    public static final int MOSSY_STONE_BRICKS = 0;
010    public static final int SMOOTH_QUARTZ = 1;
011    public static final int STONE = 2;
012    public static final int CUT_SANDSTONE = 3;
013    public static final int CUT_RED_SANDSTONE = 4;
014
015    public BlockDoubleSlabStone4() {
016        this(0);
017    }
018
019    public BlockDoubleSlabStone4(int meta) {
020        super(meta);
021    }
022
023    @Override
024    public int getId() {
025        return DOUBLE_STONE_SLAB4;
026    }
027
028    @Override
029    public double getResistance() {
030        return getToolType() > ItemTool.TIER_WOODEN ? 30 : 15;
031    }
032
033    @Override
034    public double getHardness() {
035        return 2;
036    }
037
038    @Override
039    public int getToolType() {
040        return ItemTool.TYPE_PICKAXE;
041    }
042
043    @Override
044    public String getName() {
045        String[] names = new String[]{
046                "Mossy Stone Brick",
047                "Smooth Quartz",
048                "Stone",
049                "Cut Sandstone",
050                "Cut Red Sandstone",
051                "",
052                "",
053                ""
054        };
055        return "Double " + names[this.getDamage() & 0x07] + " Slab";
056    }
057
058    @Override
059    public Item toItem() {
060        return new ItemBlock(new BlockSlabStone4(), this.getDamage() & 0x07);
061    }
062
063    @Override
064    public Item[] getDrops(Item item) {
065        if (item.isPickaxe() && item.getTier() >= ItemTool.TIER_WOODEN) {
066            Item drop = Block.get(Block.STONE_SLAB4, this.getDamage() & 0x07).toItem();
067            drop.setCount(2);
068            return new Item[]{ drop };
069        } else {
070            return new Item[0];
071        }
072    }
073
074    @Override
075    public BlockColor getColor() {
076        switch (this.getDamage() & 0x07) {
077            default:
078            case MOSSY_STONE_BRICKS:
079            case STONE:
080                return BlockColor.STONE_BLOCK_COLOR;
081            case SMOOTH_QUARTZ:
082                return BlockColor.QUARTZ_BLOCK_COLOR;
083            case CUT_SANDSTONE:
084                return BlockColor.SAND_BLOCK_COLOR;
085            case CUT_RED_SANDSTONE:
086                return BlockColor.ORANGE_BLOCK_COLOR;
087        }
088    }
089
090    @Override
091    public boolean canHarvestWithHand() {
092        return false;
093    }
094}
