001package cn.nukkit.block;
002
003import cn.nukkit.Player;
004import cn.nukkit.Server;
005import cn.nukkit.entity.Entity;
006import cn.nukkit.event.block.BlockGrowEvent;
007import cn.nukkit.event.block.BlockHarvestEvent;
008import cn.nukkit.event.entity.EntityDamageByBlockEvent;
009import cn.nukkit.event.entity.EntityDamageEvent;
010import cn.nukkit.item.Item;
011import cn.nukkit.item.ItemID;
012import cn.nukkit.item.ItemSweetBerries;
013import cn.nukkit.level.Level;
014import cn.nukkit.level.Position;
015import cn.nukkit.level.Sound;
016import cn.nukkit.level.particle.BoneMealParticle;
017import cn.nukkit.math.AxisAlignedBB;
018import cn.nukkit.math.BlockFace;
019import cn.nukkit.math.MathHelper;
020import cn.nukkit.utils.BlockColor;
021import cn.nukkit.utils.DyeColor;
022
023import java.util.concurrent.ThreadLocalRandom;
024
025public class BlockSweetBerryBush extends BlockFlowable {
026
027    public BlockSweetBerryBush() {
028        this(0);
029    }
030
031    public BlockSweetBerryBush(int meta) {
032        super(meta);
033    }
034
035    @Override
036    public int getId() {
037        return SWEET_BERRY_BUSH;
038    }
039
040    @Override
041    public String getName() {
042        return "Sweet Berry Bush";
043    }
044
045    @Override
046    public BlockColor getColor() {
047        return BlockColor.FOLIAGE_BLOCK_COLOR;
048    }
049
050    @Override
051    public int getBurnChance() {
052        return 30;
053    }
054
055    @Override
056    public int getBurnAbility() {
057        return 60;
058    }
059
060    @Override
061    public boolean canBeActivated() {
062        return true;
063    }
064
065    @Override
066    public double getHardness() {
067        return getDamage() == 0? 0 : 0.25;
068    }
069
070    @Override
071    public boolean onActivate(Item item, Player player) {
072
073        int age = MathHelper.clamp(getDamage(), 0, 3);
074
075        if (age < 3 && item.getId() == ItemID.DYE && item.getDamage() == DyeColor.WHITE.getDyeData()) {
076            BlockSweetBerryBush block = (BlockSweetBerryBush) this.clone();
077            block.setDamage(block.getDamage() + 1);
078            if (block.getDamage() > 3) {
079                block.setDamage(3);
080            }
081            BlockGrowEvent ev = new BlockGrowEvent(this, block);
082            Server.getInstance().getPluginManager().callEvent(ev);
083
084            if (ev.isCancelled()) {
085                return false;
086            }
087
088            this.getLevel().setBlock(this, ev.getNewState(), false, true);
089            this.level.addParticle(new BoneMealParticle(this));
090
091            if (player != null && (player.gamemode & 0x01) == 0) {
092                item.count--;
093            }
094
095            return true;
096        }
097
098        if (age < 2){
099            return true;
100        }
101
102        int amount = 1 + ThreadLocalRandom.current().nextInt(2);
103        if (age == 3) {
104            amount++;
105        }
106
107        BlockHarvestEvent event = new BlockHarvestEvent(this,
108                new BlockSweetBerryBush(1),
109                new Item[]{ new ItemSweetBerries(0, amount) }
110        );
111
112        getLevel().getServer().getPluginManager().callEvent(event);
113        if (!event.isCancelled()) {
114            getLevel().setBlock(this, event.getNewState(), true, true);
115            Item[] drops = event.getDrops();
116            if (drops != null) {
117                Position dropPos = add(0.5, 0.5, 0.5);
118                for (Item drop : drops) {
119                    if (drop != null) {
120                        getLevel().dropItem(dropPos, drop);
121                    }
122                }
123            }
124        }
125
126        return true;
127    }
128
129    @Override
130    public int onUpdate(int type) {
131        if (type == Level.BLOCK_UPDATE_NORMAL) {
132            if (this.down().isTransparent()) {
133                this.getLevel().useBreakOn(this);
134                return Level.BLOCK_UPDATE_NORMAL;
135            }
136        } else if (type == Level.BLOCK_UPDATE_RANDOM) {
137            if (getDamage() < 3 && ThreadLocalRandom.current().nextInt(5) == 0
138                    && getLevel().getFullLight(add(0, 1, 0)) >= BlockCrops.MINIMUM_LIGHT_LEVEL) {
139                BlockGrowEvent event = new BlockGrowEvent(this, Block.get(getId(), getDamage() + 1));
140                if (!event.isCancelled()) {
141                    getLevel().setBlock(this, event.getNewState(), true, true);
142                }
143            }
144            return type;
145        }
146        return 0;
147    }
148
149    @Override
150    public boolean place(Item item, Block block, Block target, BlockFace face, double fx, double fy, double fz, Player player) {
151        if (target.getId() == SWEET_BERRY_BUSH || block.getId() != AIR) {
152            return false;
153        }
154        Block down = this.down();
155        if (down.getId() == Block.GRASS || down.getId() == Block.DIRT || down.getId() == Block.PODZOL) {
156            this.getLevel().setBlock(block, this, true);
157            return true;
158        }
159        return false;
160    }
161
162    @Override
163    public boolean hasEntityCollision() {
164        return getDamage() > 0;
165    }
166
167    @Override
168    public void onEntityCollide(Entity entity) {
169        if (getDamage() > 0) {
170            if (entity.positionChanged && !entity.isSneaking() && ThreadLocalRandom.current().nextInt(20) == 0) {
171                if (entity.attack(new EntityDamageByBlockEvent(this, entity, EntityDamageEvent.DamageCause.CONTACT, 1))) {
172                    getLevel().addSound(entity, Sound.BLOCK_SWEET_BERRY_BUSH_HURT);
173                }
174            }
175        }
176    }
177
178    @Override
179    public AxisAlignedBB getCollisionBoundingBox() {
180        return getDamage() > 0? this : null;
181    }
182
183    @Override
184    public Item[] getDrops(Item item) {
185        int age = MathHelper.clamp(getDamage(), 0, 3);
186        int amount = 1 + ThreadLocalRandom.current().nextInt(2);
187        if (age == 3) {
188            amount++;
189        }
190
191        return new Item[]{ new ItemSweetBerries(0, amount) };
192    }
193
194    @Override
195    public Item toItem() {
196        return new ItemSweetBerries();
197    }
198}
