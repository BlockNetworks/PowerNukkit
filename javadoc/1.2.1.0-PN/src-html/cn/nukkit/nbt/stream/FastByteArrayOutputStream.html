025import java.io.IOException;
026import java.io.OutputStream;
027import java.util.Arrays;
028
029/** Simple, fast byte-array output stream that exposes the backing array.
030 *
031 * <P>{@link java.io.ByteArrayOutputStream} is nice, but to get its content you
032 * must generate each time a new object. This doesn't happen here.
033 *
034 * <P>This class will automatically enlarge the backing array, doubling its
035 * size whenever new space is needed. The {@link #reset()} method will
036 * mark the content as empty, but will not decrease the capacity
037 *
038 * @author Sebastiano Vigna
039 */
040
041public class FastByteArrayOutputStream extends OutputStream {
042
043    public static final long ONEOVERPHI = 106039;
044
045    /** The array backing the output stream. */
046    public final static int DEFAULT_INITIAL_CAPACITY = 16;
047
048    /** The array backing the output stream. */
049    public byte[] array;
050
051    /** The number of valid bytes in {@link #array}. */
052    public int length;
053
054    /** The current writing position. */
055    private int position;
056
057    /** Creates a new array output stream with an initial capacity of {@link #DEFAULT_INITIAL_CAPACITY} bytes. */
058    public FastByteArrayOutputStream() {
059        this( DEFAULT_INITIAL_CAPACITY );
060    }
061
062    /** Creates a new array output stream with a given initial capacity.
063     *
064     * @param initialCapacity the initial length of the backing array.
065     */
066    public FastByteArrayOutputStream( final int initialCapacity ) {
067        array = new byte[ initialCapacity ];
068    }
069
070    /** Creates a new array output stream wrapping a given byte array.
071     *
072     * @param a the byte array to wrap.
073     */
074    public FastByteArrayOutputStream( final byte[] a ) {
075        array = a;
076    }
077
078    /** Marks this array output stream as empty. */
079    public FastByteArrayOutputStream reset() {
080        length = 0;
081        position = 0;
082        return this;
083    }
084
085    public void write( final int b ) {
086        if ( position == length ) {
087            length++;
088            if ( position == array.length ) array = grow( array, length );
089        }
090        array[ position++ ] = (byte)b;
091    }
092
093    public static void ensureOffsetLength( final int arrayLength, final int offset, final int length ) {
094        if ( offset < 0 ) throw new ArrayIndexOutOfBoundsException( "Offset (" + offset + ") is negative" );
095        if ( length < 0 ) throw new IllegalArgumentException( "Length (" + length + ") is negative" );
096        if ( offset + length > arrayLength ) throw new ArrayIndexOutOfBoundsException( "Last index (" + ( offset + length ) + ") is greater than array length (" + arrayLength + ")" );
097    }
098
099    public static byte[] grow( final byte[] array, final int length ) {
100        if ( length > array.length ) {
101            final int newLength = (int)Math.min( Math.max( ( ONEOVERPHI * array.length ) >>> 16, length ), Integer.MAX_VALUE );
102            final byte[] t =
103                    new byte[ newLength ];
104            System.arraycopy( array, 0, t, 0, array.length );
105            return t;
106        }
107        return array;
108    }
109
110    public static byte[] grow( final byte[] array, final int length, final int preserve ) {
111        if ( length > array.length ) {
112            final int newLength = (int)Math.min( Math.max( ( ONEOVERPHI * array.length ) >>> 16, length ), Integer.MAX_VALUE );
113            final byte[] t =
114                    new byte[ newLength ];
115            System.arraycopy( array, 0, t, 0, preserve );
116            return t;
117        }
118        return array;
119    }
120
121    public void write( final byte[] b, final int off, final int len ) throws IOException {
122        if ( position + len > array.length ) array = grow( array, position + len, position );
123        System.arraycopy( b, off, array, position, len );
124        if ( position + len > length ) length = position += len;
125    }
126
127    public void position( long newPosition ) {
128        if ( position > Integer.MAX_VALUE ) throw new IllegalArgumentException( "Position too large: " + newPosition );
129        position = (int)newPosition;
130    }
131
132    public long position() {
133        return position;
134    }
135
136    public long length() throws IOException {
137        return length;
138    }
139
140    public byte[] toByteArray() {
141        if (position == array.length) return array;
142        return Arrays.copyOfRange(array, 0, position);
143    }
144}
