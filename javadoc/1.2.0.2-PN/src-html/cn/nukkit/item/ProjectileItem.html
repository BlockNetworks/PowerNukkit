001package cn.nukkit.item;
002
003import cn.nukkit.Player;
004import cn.nukkit.entity.Entity;
005import cn.nukkit.entity.projectile.EntityEnderPearl;
006import cn.nukkit.entity.projectile.EntityProjectile;
007import cn.nukkit.event.entity.ProjectileLaunchEvent;
008import cn.nukkit.math.Vector3;
009import cn.nukkit.nbt.tag.CompoundTag;
010import cn.nukkit.nbt.tag.DoubleTag;
011import cn.nukkit.nbt.tag.FloatTag;
012import cn.nukkit.nbt.tag.ListTag;
013import cn.nukkit.network.protocol.LevelSoundEventPacketV2;
014
015/**
016 * @author CreeperFace
017 */
018public abstract class ProjectileItem extends Item {
019
020    public ProjectileItem(int id, Integer meta, int count, String name) {
021        super(id, meta, count, name);
022    }
023
024    abstract public String getProjectileEntityType();
025
026    abstract public float getThrowForce();
027
028    public boolean onClickAir(Player player, Vector3 directionVector) {
029        CompoundTag nbt = new CompoundTag()
030                .putList(new ListTag<DoubleTag>("Pos")
031                        .add(new DoubleTag("", player.x))
032                        .add(new DoubleTag("", player.y + player.getEyeHeight() - 0.30000000149011612))
033                        .add(new DoubleTag("", player.z)))
034                .putList(new ListTag<DoubleTag>("Motion")
035                        .add(new DoubleTag("", directionVector.x))
036                        .add(new DoubleTag("", directionVector.y))
037                        .add(new DoubleTag("", directionVector.z)))
038                .putList(new ListTag<FloatTag>("Rotation")
039                        .add(new FloatTag("", (float) player.yaw))
040                        .add(new FloatTag("", (float) player.pitch)));
041
042        this.correctNBT(nbt);
043
044        Entity projectile = Entity.createEntity(this.getProjectileEntityType(), player.getLevel().getChunk(player.getFloorX() >> 4, player.getFloorZ() >> 4), nbt, player);
045        if (projectile != null) {
046            projectile = correctProjectile(player, projectile);
047            if (projectile == null) {
048                return false;
049            }
050
051            projectile.setMotion(projectile.getMotion().multiply(this.getThrowForce()));
052
053            if (projectile instanceof EntityProjectile) {
054                ProjectileLaunchEvent ev = new ProjectileLaunchEvent((EntityProjectile) projectile);
055
056                player.getServer().getPluginManager().callEvent(ev);
057                if (ev.isCancelled()) {
058                    projectile.kill();
059                } else {
060                    if (!player.isCreative()) {
061                        this.count--;
062                    }
063                    if (projectile instanceof EntityEnderPearl) {
064                        player.onThrowEnderPearl();
065                    }
066                    projectile.spawnToAll();
067                    addThrowSound(player);
068                }
069            }
070        } else {
071            return false;
072        }
073        return true;
074    }
075
076    protected void addThrowSound(Player player) {
077        player.getLevel().addLevelSoundEvent(player, LevelSoundEventPacketV2.SOUND_THROW, -1, "minecraft:player", false, false);
078    }
079
080    protected Entity correctProjectile(Player player, Entity projectile) {
081        return projectile;
082    }
083
084    protected void correctNBT(CompoundTag nbt) {
085
086    }
087}
