001package cn.nukkit.block;
002
003import cn.nukkit.Player;
004import cn.nukkit.blockentity.BlockEntity;
005import cn.nukkit.blockentity.BlockEntityFurnace;
006import cn.nukkit.inventory.ContainerInventory;
007import cn.nukkit.item.Item;
008import cn.nukkit.item.ItemBlock;
009import cn.nukkit.item.ItemTool;
010import cn.nukkit.level.Position;
011import cn.nukkit.math.BlockFace;
012import cn.nukkit.nbt.tag.CompoundTag;
013import cn.nukkit.nbt.tag.ListTag;
014import cn.nukkit.nbt.tag.StringTag;
015import cn.nukkit.nbt.tag.Tag;
016import cn.nukkit.utils.Faceable;
017
018import java.util.Map;
019
020/**
021 * author: Angelic47
022 * Nukkit Project
023 */
024public class BlockFurnaceBurning extends BlockSolidMeta implements Faceable {
025
026    public BlockFurnaceBurning() {
027        this(0);
028    }
029
030    public BlockFurnaceBurning(int meta) {
031        super(meta);
032    }
033
034    @Override
035    public int getId() {
036        return BURNING_FURNACE;
037    }
038
039    @Override
040    public String getName() {
041        return "Burning Furnace";
042    }
043
044    @Override
045    public boolean canBeActivated() {
046        return true;
047    }
048
049    @Override
050    public double getHardness() {
051        return 3.5;
052    }
053
054    @Override
055    public double getResistance() {
056        return 17.5;
057    }
058
059    @Override
060    public int getToolType() {
061        return ItemTool.TYPE_PICKAXE;
062    }
063
064    @Override
065    public int getLightLevel() {
066        return 13;
067    }
068
069    protected String getEntityName() {
070        return BlockEntity.FURNACE;
071    }
072
073    protected BlockEntityFurnace createEntity(Position position, CompoundTag nbt) {
074        return (BlockEntityFurnace) BlockEntity.createBlockEntity(BlockEntity.FURNACE, position, nbt);
075    }
076
077    @Override
078    public boolean place(Item item, Block block, Block target, BlockFace face, double fx, double fy, double fz, Player player) {
079        int[] faces = {2, 5, 3, 4};
080        this.setDamage(faces[player != null ? player.getDirection().getHorizontalIndex() : 0]);
081        this.getLevel().setBlock(block, this, true, true);
082        CompoundTag nbt = new CompoundTag()
083                .putList(new ListTag<>("Items"))
084                .putString("id", getEntityName())
085                .putInt("x", (int) this.x)
086                .putInt("y", (int) this.y)
087                .putInt("z", (int) this.z);
088
089        if (item.hasCustomName()) {
090            nbt.putString("CustomName", item.getCustomName());
091        }
092
093        if (item.hasCustomBlockData()) {
094            Map<String, Tag> customData = item.getCustomBlockData().getTags();
095            for (Map.Entry<String, Tag> tag : customData.entrySet()) {
096                nbt.put(tag.getKey(), tag.getValue());
097            }
098        }
099
100        createEntity(this, nbt);
101
102        return true;
103    }
104
105    @Override
106    public boolean onBreak(Item item) {
107        this.getLevel().setBlock(this, Block.get(BlockID.AIR), true, true);
108        return true;
109    }
110
111    @Override
112    public boolean onActivate(Item item, Player player) {
113        if (player != null) {
114            BlockEntity t = this.getLevel().getBlockEntity(this);
115            BlockEntityFurnace furnace;
116            if (t instanceof BlockEntityFurnace) {
117                furnace = (BlockEntityFurnace) t;
118            } else {
119                CompoundTag nbt = new CompoundTag()
120                        .putList(new ListTag<>("Items"))
121                        .putString("id", getEntityName())
122                        .putInt("x", (int) this.x)
123                        .putInt("y", (int) this.y)
124                        .putInt("z", (int) this.z);
125                furnace = createEntity(this, nbt);
126            }
127
128            if (furnace.namedTag.contains("Lock") && furnace.namedTag.get("Lock") instanceof StringTag) {
129                if (!furnace.namedTag.getString("Lock").equals(item.getCustomName())) {
130                    return true;
131                }
132            }
133
134            player.addWindow(furnace.getInventory());
135        }
136
137        return true;
138    }
139
140    @Override
141    public Item toItem() {
142        return new ItemBlock(Block.get(BlockID.FURNACE));
143    }
144
145    @Override
146    public Item[] getDrops(Item item) {
147        if (item.isPickaxe() && item.getTier() >= ItemTool.TIER_WOODEN) {
148            return new Item[]{
149                    this.toItem()
150            };
151        } else {
152            return new Item[0];
153        }
154    }
155
156    public boolean hasComparatorInputOverride() {
157        return true;
158    }
159
160    @Override
161    public int getComparatorInputOverride() {
162        BlockEntity blockEntity = this.level.getBlockEntity(this);
163
164        if (blockEntity instanceof BlockEntityFurnace) {
165            return ContainerInventory.calculateRedstone(((BlockEntityFurnace) blockEntity).getInventory());
166        }
167
168        return super.getComparatorInputOverride();
169    }
170
171    @Override
172    public boolean canHarvestWithHand() {
173        return false;
174    }
175
176    @Override
177    public BlockFace getBlockFace() {
178        return BlockFace.fromHorizontalIndex(this.getDamage() & 0x7);
179    }
180}
