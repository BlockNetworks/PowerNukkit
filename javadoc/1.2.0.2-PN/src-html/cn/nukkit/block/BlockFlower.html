001package cn.nukkit.block;
002
003import cn.nukkit.Player;
004import cn.nukkit.item.Item;
005import cn.nukkit.level.Level;
006import cn.nukkit.level.particle.BoneMealParticle;
007import cn.nukkit.math.BlockFace;
008import cn.nukkit.math.Vector3;
009import cn.nukkit.utils.BlockColor;
010
011import java.util.concurrent.ThreadLocalRandom;
012
013/**
014 * Created on 2015/11/23 by xtypr.
015 * Package cn.nukkit.block in project Nukkit .
016 */
017public class BlockFlower extends BlockFlowable {
018    public static final int TYPE_POPPY = 0;
019    public static final int TYPE_BLUE_ORCHID = 1;
020    public static final int TYPE_ALLIUM = 2;
021    public static final int TYPE_AZURE_BLUET = 3;
022    public static final int TYPE_RED_TULIP = 4;
023    public static final int TYPE_ORANGE_TULIP = 5;
024    public static final int TYPE_WHITE_TULIP = 6;
025    public static final int TYPE_PINK_TULIP = 7;
026    public static final int TYPE_OXEYE_DAISY = 8;
027
028    public BlockFlower() {
029        this(0);
030    }
031
032    public BlockFlower(int meta) {
033        super(meta);
034    }
035
036    @Override
037    public int getId() {
038        return FLOWER;
039    }
040
041    @Override
042    public String getName() {
043        String[] names = new String[]{
044                "Poppy",
045                "Blue Orchid",
046                "Allium",
047                "Azure Bluet",
048                "Red Tulip",
049                "Orange Tulip",
050                "White Tulip",
051                "Pink Tulip",
052                "Oxeye Daisy",
053                "Unknown",
054                "Unknown",
055                "Unknown",
056                "Unknown",
057                "Unknown",
058                "Unknown",
059                "Unknown"
060        };
061        return names[this.getDamage() & 0x0f];
062    }
063
064    public boolean canPlantOn(Block block) {
065        return block.getId() == Block.GRASS || block.getId() == Block.DIRT || block.getId() == Block.FARMLAND || block.getId() == Block.PODZOL;
066    }
067
068    @Override
069    public boolean place(Item item, Block block, Block target, BlockFace face, double fx, double fy, double fz, Player player) {
070        Block down = this.down();
071        if (canPlantOn(down)) {
072            this.getLevel().setBlock(block, this, true);
073
074            return true;
075        }
076        return false;
077    }
078
079    @Override
080    public int onUpdate(int type) {
081        if (type == Level.BLOCK_UPDATE_NORMAL) {
082            if (this.down().isTransparent()) {
083                this.getLevel().useBreakOn(this);
084
085                return Level.BLOCK_UPDATE_NORMAL;
086            }
087        }
088
089        return 0;
090    }
091
092    @Override
093    public BlockColor getColor() {
094        return BlockColor.FOLIAGE_BLOCK_COLOR;
095    }
096
097    @Override
098    public boolean canBeActivated() {
099        return true;
100    }
101
102    @Override
103    public boolean onActivate(Item item, Player player) {
104        if (item.getId() == Item.DYE && item.getDamage() == 0x0f) { //Bone meal
105            if (player != null && (player.gamemode & 0x01) == 0) {
106                item.count--;
107            }
108
109            this.level.addParticle(new BoneMealParticle(this));
110
111            for (int i = 0; i < 8; i++) {
112                Vector3 vec = this.add(
113                        ThreadLocalRandom.current().nextInt(-3, 4),
114                        ThreadLocalRandom.current().nextInt(-1, 2),
115                        ThreadLocalRandom.current().nextInt(-3, 4));
116
117                if (level.getBlock(vec).getId() == AIR && level.getBlock(vec.down()).getId() == GRASS && vec.getY() >= 0 && vec.getY() < 256) {
118                    if (ThreadLocalRandom.current().nextInt(10) == 0) {
119                        this.level.setBlock(vec, this.getUncommonFlower(), true);
120                    } else {
121                        this.level.setBlock(vec, get(this.getId()), true);
122                    }
123                }
124            }
125
126            return true;
127        }
128
129        return false;
130    }
131
132    protected Block getUncommonFlower() {
133        return get(DANDELION);
134    }
135}
