001package cn.nukkit.level.util;
002
003public enum BitArrayVersion {
004    V16(16, 2, null),
005    V8(8, 4, V16),
006    V6(6, 5, V8), // 2 bit padding
007    V5(5, 6, V6), // 2 bit padding
008    V4(4, 8, V5),
009    V3(3, 10, V4), // 2 bit padding
010    V2(2, 16, V3),
011    V1(1, 32, V2);
012
013    final byte bits;
014    final byte entriesPerWord;
015    final int maxEntryValue;
016    final BitArrayVersion next;
017
018    BitArrayVersion(int bits, int entriesPerWord, BitArrayVersion next) {
019        this.bits = (byte) bits;
020        this.entriesPerWord = (byte) entriesPerWord;
021        this.maxEntryValue = (1 << this.bits) - 1;
022        this.next = next;
023    }
024
025    public static BitArrayVersion get(int version, boolean read) {
026        for (BitArrayVersion ver : values()) {
027            if ((!read && ver.entriesPerWord <= version) || (read && ver.bits == version)) {
028                return ver;
029            }
030        }
031        throw new IllegalArgumentException("Invalid palette version: " + version);
032    }
033
034    public BitArray createPalette(int size) {
035        return this.createPalette(size, new int[this.getWordsForSize(size)]);
036    }
037
038    public byte getId() {
039        return bits;
040    }
041
042    public int getWordsForSize(int size) {
043        return (size / entriesPerWord) + (size % entriesPerWord == 0 ? 0 : 1);
044    }
045
046    public int getMaxEntryValue() {
047        return maxEntryValue;
048    }
049
050    public BitArrayVersion next() {
051        return next;
052    }
053
054    public BitArray createPalette(int size, int[] words) {
055        if (this == V3 || this == V5 || this == V6) {
056            // Padded palettes aren't able to use bitwise operations due to their padding.
057            return new PaddedBitArray(this, size, words);
058        } else {
059            return new Pow2BitArray(this, size, words);
060        }
061    }
062}
