001package cn.nukkit.level.generator.populator.impl.tree;
002
003import cn.nukkit.block.Block;
004import cn.nukkit.block.BlockSapling;
005import cn.nukkit.level.ChunkManager;
006import cn.nukkit.level.format.FullChunk;
007import cn.nukkit.level.generator.object.tree.NewJungleTree;
008import cn.nukkit.level.generator.populator.type.Populator;
009import cn.nukkit.math.NukkitMath;
010import cn.nukkit.math.NukkitRandom;
011import cn.nukkit.math.Vector3;
012
013/**
014 * @author DaPorkchop_
015 *
016 * Used for the dense foliage on the floor in the jungle M biome
017 */
018public class JungleFloorPopulator extends Populator {
019    private ChunkManager level;
020    private int randomAmount;
021    private int baseAmount;
022
023    private final int type;
024
025    public JungleFloorPopulator() {
026        this(BlockSapling.JUNGLE);
027    }
028
029    public JungleFloorPopulator(int type) {
030        this.type = type;
031    }
032
033    public void setRandomAmount(int randomAmount) {
034        this.randomAmount = randomAmount;
035    }
036
037    public void setBaseAmount(int baseAmount) {
038        this.baseAmount = baseAmount;
039    }
040
041    @Override
042    public void populate(ChunkManager level, int chunkX, int chunkZ, NukkitRandom random, FullChunk chunk) {
043        this.level = level;
044        int amount = random.nextBoundedInt(this.randomAmount + 1) + this.baseAmount;
045        Vector3 v = new Vector3();
046
047        for (int i = 0; i < amount; ++i) {
048            int x = NukkitMath.randomRange(random, chunkX << 4, (chunkX << 4) + 15);
049            int z = NukkitMath.randomRange(random, chunkZ << 4, (chunkZ << 4) + 15);
050            int y = this.getHighestWorkableBlock(x, z);
051            if (y == -1) {
052                continue;
053            }
054            new NewJungleTree(1, 0).generate(level, random, v.setComponents(x, y, z));
055        }
056    }
057
058    private int getHighestWorkableBlock(int x, int z) {
059        int y;
060        for (y = 255; y > 0; --y) {
061            int b = this.level.getBlockIdAt(x, y, z);
062            if (b == Block.DIRT || b == Block.GRASS) {
063                break;
064            } else if (b != Block.AIR && b != Block.SNOW_LAYER) {
065                return -1;
066            }
067        }
068
069        return ++y;
070    }
071}
