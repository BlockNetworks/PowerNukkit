001package cn.nukkit.level.generator.object.tree;
002
003import cn.nukkit.block.Block;
004import cn.nukkit.block.BlockID;
005import cn.nukkit.block.BlockLeaves2;
006import cn.nukkit.block.BlockWood2;
007import cn.nukkit.level.ChunkManager;
008import cn.nukkit.math.BlockFace;
009import cn.nukkit.math.NukkitRandom;
010import cn.nukkit.math.Vector3;
011
012public class ObjectSavannaTree extends TreeGenerator {
013    private static final Block TRUNK = Block.get(BlockID.WOOD2, BlockWood2.ACACIA);
014    private static final Block LEAF = Block.get(BlockID.LEAVES2, BlockLeaves2.ACACIA);
015
016    public boolean generate(ChunkManager level, NukkitRandom rand, Vector3 position) {
017        int i = rand.nextBoundedInt(3) + rand.nextBoundedInt(3) + 5;
018        boolean flag = true;
019
020        if (position.getY() >= 1 && position.getY() + i + 1 <= 256) {
021            for (int j = (int) position.getY(); j <= position.getY() + 1 + i; ++j) {
022                int k = 1;
023
024                if (j == position.getY()) {
025                    k = 0;
026                }
027
028                if (j >= position.getY() + 1 + i - 2) {
029                    k = 2;
030                }
031
032                Vector3 vector3 = new Vector3();
033
034                for (int l = (int) position.getX() - k; l <= position.getX() + k && flag; ++l) {
035                    for (int i1 = (int) position.getZ() - k; i1 <= position.getZ() + k && flag; ++i1) {
036                        if (j >= 0 && j < 256) {
037
038                            vector3.setComponents(l, j, i1);
039                            if (!this.canGrowInto(level.getBlockIdAt((int) vector3.x, (int) vector3.y, (int) vector3.z))) {
040                                flag = false;
041                            }
042                        } else {
043                            flag = false;
044                        }
045                    }
046                }
047            }
048
049            if (!flag) {
050                return false;
051            } else {
052                Vector3 down = position.down();
053                int block = level.getBlockIdAt(down.getFloorX(), down.getFloorY(), down.getFloorZ());
054
055                if ((block == Block.GRASS || block == Block.DIRT) && position.getY() < 256 - i - 1) {
056                    this.setDirtAt(level, position.down());
057                    BlockFace face = BlockFace.Plane.HORIZONTAL.random(rand);
058                    int k2 = i - rand.nextBoundedInt(4) - 1;
059                    int l2 = 3 - rand.nextBoundedInt(3);
060                    int i3 = position.getFloorX();
061                    int j1 = position.getFloorZ();
062                    int k1 = 0;
063
064                    for (int l1 = 0; l1 < i; ++l1) {
065                        int i2 = position.getFloorY() + l1;
066
067                        if (l1 >= k2 && l2 > 0) {
068                            i3 += face.getXOffset();
069                            j1 += face.getZOffset();
070                            --l2;
071                        }
072
073                        Vector3 blockpos = new Vector3(i3, i2, j1);
074                        int material = level.getBlockIdAt(blockpos.getFloorX(), blockpos.getFloorY(), blockpos.getFloorZ());
075
076                        if (material == Block.AIR || material == Block.LEAVES) {
077                            this.placeLogAt(level, blockpos);
078                            k1 = i2;
079                        }
080                    }
081
082                    Vector3 blockpos2 = new Vector3(i3, k1, j1);
083
084                    for (int j3 = -3; j3 <= 3; ++j3) {
085                        for (int i4 = -3; i4 <= 3; ++i4) {
086                            if (Math.abs(j3) != 3 || Math.abs(i4) != 3) {
087                                this.placeLeafAt(level, blockpos2.add(j3, 0, i4));
088                            }
089                        }
090                    }
091
092                    blockpos2 = blockpos2.up();
093
094                    for (int k3 = -1; k3 <= 1; ++k3) {
095                        for (int j4 = -1; j4 <= 1; ++j4) {
096                            this.placeLeafAt(level, blockpos2.add(k3, 0, j4));
097                        }
098                    }
099
100                    this.placeLeafAt(level, blockpos2.east(2));
101                    this.placeLeafAt(level, blockpos2.west(2));
102                    this.placeLeafAt(level, blockpos2.south(2));
103                    this.placeLeafAt(level, blockpos2.north(2));
104                    i3 = position.getFloorX();
105                    j1 = position.getFloorZ();
106                    BlockFace face1 = BlockFace.Plane.HORIZONTAL.random(rand);
107
108                    if (face1 != face) {
109                        int l3 = k2 - rand.nextBoundedInt(2) - 1;
110                        int k4 = 1 + rand.nextBoundedInt(3);
111                        k1 = 0;
112
113                        for (int l4 = l3; l4 < i && k4 > 0; --k4) {
114                            if (l4 >= 1) {
115                                int j2 = position.getFloorY() + l4;
116                                i3 += face1.getXOffset();
117                                j1 += face1.getZOffset();
118                                Vector3 blockpos1 = new Vector3(i3, j2, j1);
119                                int material1 = level.getBlockIdAt(blockpos1.getFloorX(), blockpos1.getFloorY(), blockpos1.getFloorZ());
120
121                                if (material1 == Block.AIR || material1 == Block.LEAVES) {
122                                    this.placeLogAt(level, blockpos1);
123                                    k1 = j2;
124                                }
125                            }
126
127                            ++l4;
128                        }
129
130                        if (k1 > 0) {
131                            Vector3 blockpos3 = new Vector3(i3, k1, j1);
132
133                            for (int i5 = -2; i5 <= 2; ++i5) {
134                                for (int k5 = -2; k5 <= 2; ++k5) {
135                                    if (Math.abs(i5) != 2 || Math.abs(k5) != 2) {
136                                        this.placeLeafAt(level, blockpos3.add(i5, 0, k5));
137                                    }
138                                }
139                            }
140
141                            blockpos3 = blockpos3.up();
142
143                            for (int j5 = -1; j5 <= 1; ++j5) {
144                                for (int l5 = -1; l5 <= 1; ++l5) {
145                                    this.placeLeafAt(level, blockpos3.add(j5, 0, l5));
146                                }
147                            }
148                        }
149                    }
150
151                    return true;
152                } else {
153                    return false;
154                }
155            }
156        } else {
157            return false;
158        }
159    }
160
161    private void placeLogAt(ChunkManager worldIn, Vector3 pos) {
162        this.setBlockAndNotifyAdequately(worldIn, pos, TRUNK);
163    }
164
165    private void placeLeafAt(ChunkManager worldIn, Vector3 pos) {
166        int material = worldIn.getBlockIdAt(pos.getFloorX(), pos.getFloorY(), pos.getFloorZ());
167
168        if (material == Block.AIR || material == Block.LEAVES) {
169            this.setBlockAndNotifyAdequately(worldIn, pos, LEAF);
170        }
171    }
172}
