015public class OreType {
016    public final int fullId;
017    public final int clusterCount;
018    public final int clusterSize;
019    public final int maxHeight;
020    public final int minHeight;
021    public final int replaceBlockId;
022
023    public OreType(Block material, int clusterCount, int clusterSize, int minHeight, int maxHeight) {
024        this(material, clusterCount, clusterSize, minHeight, maxHeight, STONE);
025    }
026
027    public OreType(Block material, int clusterCount, int clusterSize, int minHeight, int maxHeight, int replaceBlockId) {
028        this.fullId = material.getFullId();
029        this.clusterCount = clusterCount;
030        this.clusterSize = clusterSize;
031        this.maxHeight = maxHeight;
032        this.minHeight = minHeight;
033        this.replaceBlockId = replaceBlockId;
034    }
035
036    public boolean spawn(ChunkManager level, NukkitRandom rand, int replaceId, int x, int y, int z) {
037        float piScaled = rand.nextFloat() * (float) Math.PI;
038        double scaleMaxX = (double) ((float) (x + 8) + MathHelper.sin(piScaled) * (float) clusterSize / 8.0F);
039        double scaleMinX = (double) ((float) (x + 8) - MathHelper.sin(piScaled) * (float) clusterSize / 8.0F);
040        double scaleMaxZ = (double) ((float) (z + 8) + MathHelper.cos(piScaled) * (float) clusterSize / 8.0F);
041        double scaleMinZ = (double) ((float) (z + 8) - MathHelper.cos(piScaled) * (float) clusterSize / 8.0F);
042        double scaleMaxY = (double) (y + rand.nextBoundedInt(3) - 2);
043        double scaleMinY = (double) (y + rand.nextBoundedInt(3) - 2);
044
045        for (int i = 0; i < clusterSize; ++i) {
046            float sizeIncr = (float) i / (float) clusterSize;
047            double scaleX = scaleMaxX + (scaleMinX - scaleMaxX) * (double) sizeIncr;
048            double scaleY = scaleMaxY + (scaleMinY - scaleMaxY) * (double) sizeIncr;
049            double scaleZ = scaleMaxZ + (scaleMinZ - scaleMaxZ) * (double) sizeIncr;
050            double randSizeOffset = rand.nextDouble() * (double) clusterSize / 16.0D;
051            double randVec1 = (double) (MathHelper.sin((float) Math.PI * sizeIncr) + 1.0F) * randSizeOffset + 1.0D;
052            double randVec2 = (double) (MathHelper.sin((float) Math.PI * sizeIncr) + 1.0F) * randSizeOffset + 1.0D;
053            int minX = MathHelper.floor(scaleX - randVec1 / 2.0D);
054            int minY = MathHelper.floor(scaleY - randVec2 / 2.0D);
055            int minZ = MathHelper.floor(scaleZ - randVec1 / 2.0D);
056            int maxX = MathHelper.floor(scaleX + randVec1 / 2.0D);
057            int maxY = MathHelper.floor(scaleY + randVec2 / 2.0D);
058            int maxZ = MathHelper.floor(scaleZ + randVec1 / 2.0D);
059
060            for (int xSeg = minX; xSeg <= maxX; ++xSeg) {
061                double xVal = ((double) xSeg + 0.5D - scaleX) / (randVec1 / 2.0D);
062
063                if (xVal * xVal < 1.0D) {
064                    for (int ySeg = minY; ySeg <= maxY; ++ySeg) {
065                        double yVal = ((double) ySeg + 0.5D - scaleY) / (randVec2 / 2.0D);
066
067                        if (xVal * xVal + yVal * yVal < 1.0D) {
068                            for (int zSeg = minZ; zSeg <= maxZ; ++zSeg) {
069                                double zVal = ((double) zSeg + 0.5D - scaleZ) / (randVec1 / 2.0D);
070
071                                if (xVal * xVal + yVal * yVal + zVal * zVal < 1.0D) {
072                                    if (level.getBlockIdAt(xSeg, ySeg, zSeg) == replaceBlockId) {
073                                        level.setBlockFullIdAt(xSeg, ySeg, zSeg, fullId);
074                                    }
075                                }
076                            }
077                        }
078                    }
079                }
080            }
081        }
082        
083        return true;
084    }
085}
