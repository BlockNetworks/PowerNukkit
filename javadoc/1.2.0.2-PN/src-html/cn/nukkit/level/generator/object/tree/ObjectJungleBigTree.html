001package cn.nukkit.level.generator.object.tree;
002
003import cn.nukkit.block.Block;
004import cn.nukkit.block.BlockID;
005import cn.nukkit.level.ChunkManager;
006import cn.nukkit.math.BlockVector3;
007import cn.nukkit.math.MathHelper;
008import cn.nukkit.math.NukkitRandom;
009import cn.nukkit.math.Vector3;
010
011public class ObjectJungleBigTree extends HugeTreesGenerator {
012    public ObjectJungleBigTree(int baseHeightIn, int extraRandomHeight, Block woodMetadata, Block leavesMetadata) {
013        super(baseHeightIn, extraRandomHeight, woodMetadata, leavesMetadata);
014    }
015
016    public boolean generate(ChunkManager level, NukkitRandom rand, Vector3 position) {
017        int height = this.getHeight(rand);
018
019        if (!this.ensureGrowable(level, rand, position, height)) {
020            return false;
021        } else {
022            this.createCrown(level, position.up(height), 2);
023
024            for (int j = (int) position.getY() + height - 2 - rand.nextBoundedInt(4); j > position.getY() + height / 2; j -= 2 + rand.nextBoundedInt(4)) {
025                float f = rand.nextFloat() * ((float) Math.PI * 2F);
026                int k = (int) (position.getX() + (0.5F + MathHelper.cos(f) * 4.0F));
027                int l = (int) (position.getZ() + (0.5F + MathHelper.sin(f) * 4.0F));
028
029                for (int i1 = 0; i1 < 5; ++i1) {
030                    k = (int) (position.getX() + (1.5F + MathHelper.cos(f) * (float) i1));
031                    l = (int) (position.getZ() + (1.5F + MathHelper.sin(f) * (float) i1));
032                    this.setBlockAndNotifyAdequately(level, new BlockVector3(k, j - 3 + i1 / 2, l), this.woodMetadata);
033                }
034
035                int j2 = 1 + rand.nextBoundedInt(2);
036
037                for (int k1 = j - j2; k1 <= j; ++k1) {
038                    int l1 = k1 - j;
039                    this.growLeavesLayer(level, new Vector3(k, k1, l), 1 - l1);
040                }
041            }
042
043            for (int i2 = 0; i2 < height; ++i2) {
044                Vector3 blockpos = position.up(i2);
045
046                if (this.canGrowInto(level.getBlockIdAt((int) blockpos.x, (int) blockpos.y, (int) blockpos.z))) {
047                    this.setBlockAndNotifyAdequately(level, blockpos, this.woodMetadata);
048
049                    if (i2 > 0) {
050                        this.placeVine(level, rand, blockpos.west(), 8);
051                        this.placeVine(level, rand, blockpos.north(), 1);
052                    }
053                }
054
055                if (i2 < height - 1) {
056                    Vector3 blockpos1 = blockpos.east();
057
058                    if (this.canGrowInto(level.getBlockIdAt((int) blockpos1.x, (int) blockpos1.y, (int) blockpos1.z))) {
059                        this.setBlockAndNotifyAdequately(level, blockpos1, this.woodMetadata);
060
061                        if (i2 > 0) {
062                            this.placeVine(level, rand, blockpos1.east(), 2);
063                            this.placeVine(level, rand, blockpos1.north(), 1);
064                        }
065                    }
066
067                    Vector3 blockpos2 = blockpos.south().east();
068
069                    if (this.canGrowInto(level.getBlockIdAt((int) blockpos2.x, (int) blockpos2.y, (int) blockpos2.z))) {
070                        this.setBlockAndNotifyAdequately(level, blockpos2, this.woodMetadata);
071
072                        if (i2 > 0) {
073                            this.placeVine(level, rand, blockpos2.east(), 2);
074                            this.placeVine(level, rand, blockpos2.south(), 4);
075                        }
076                    }
077
078                    Vector3 blockpos3 = blockpos.south();
079
080                    if (this.canGrowInto(level.getBlockIdAt((int) blockpos3.x, (int) blockpos3.y, (int) blockpos3.z))) {
081                        this.setBlockAndNotifyAdequately(level, blockpos3, this.woodMetadata);
082
083                        if (i2 > 0) {
084                            this.placeVine(level, rand, blockpos3.west(), 8);
085                            this.placeVine(level, rand, blockpos3.south(), 4);
086                        }
087                    }
088                }
089            }
090
091            return true;
092        }
093    }
094
095    private void placeVine(ChunkManager level, NukkitRandom random, Vector3 pos, int meta) {
096        if (random.nextBoundedInt(3) > 0 && level.getBlockIdAt((int) pos.x, (int) pos.y, (int) pos.z) == 0) {
097            this.setBlockAndNotifyAdequately(level, pos, Block.get(BlockID.VINE, meta));
098        }
099    }
100
101    private void createCrown(ChunkManager level, Vector3 pos, int i1) {
102        for (int j = -2; j <= 0; ++j) {
103            this.growLeavesLayerStrict(level, pos.up(j), i1 + 1 - j);
104        }
105    }
106}
