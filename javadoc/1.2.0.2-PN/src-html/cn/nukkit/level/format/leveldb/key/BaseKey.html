001package cn.nukkit.level.format.leveldb.key;
002
003/**
004 * author: MagicDroidX
005 * Nukkit Project
006 */
007public class BaseKey {
008    private final int chunkX;
009    private final int chunkZ;
010    private final byte type;
011
012    public static final byte DATA_VERSION = 0x76;
013    public static final byte DATA_FLAGS = 0x66;
014    public static final byte DATA_EXTRA_DATA = 0x34;
015    public static final byte DATA_TICKS = 0x33;
016    public static final byte DATA_ENTITIES = 0x32;
017    public static final byte DATA_TILES = 0x31;
018    public static final byte DATA_TERRAIN = 0x30;
019
020    protected BaseKey(int chunkX, int chunkZ, byte type) {
021        this.chunkX = chunkX;
022        this.chunkZ = chunkZ;
023        this.type = type;
024    }
025
026    public byte[] toArray() {
027        return new byte[]{
028                (byte) (this.chunkX & 0xff),
029                (byte) ((this.chunkX >>> 8) & 0xff),
030                (byte) ((this.chunkX >>> 16) & 0xff),
031                (byte) ((this.chunkX >>> 24) & 0xff),
032                (byte) (this.chunkZ & 0xff),
033                (byte) ((this.chunkZ >>> 8) & 0xff),
034                (byte) ((this.chunkZ >>> 16) & 0xff),
035                (byte) ((this.chunkZ >>> 24) & 0xff),
036                this.type
037        };
038    }
039}
