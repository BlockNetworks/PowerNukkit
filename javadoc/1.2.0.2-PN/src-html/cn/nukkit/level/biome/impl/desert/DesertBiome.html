001package cn.nukkit.level.biome.impl.desert;
002
003import cn.nukkit.level.biome.type.SandyBiome;
004import cn.nukkit.level.generator.populator.impl.PopulatorCactus;
005import cn.nukkit.level.generator.populator.impl.PopulatorDeadBush;
006
007/**
008 * author: MagicDroidX
009 * Nukkit Project
010 */
011public class DesertBiome extends SandyBiome {
012    public DesertBiome() {
013        PopulatorCactus cactus = new PopulatorCactus();
014        cactus.setBaseAmount(2);
015        this.addPopulator(cactus);
016
017        PopulatorDeadBush deadbush = new PopulatorDeadBush();
018        deadbush.setBaseAmount(2);
019        this.addPopulator(deadbush);
020
021        this.setBaseHeight(0.125f);
022        this.setHeightVariation(0.05f);
023    }
024
025    @Override
026    public String getName() {
027        return "Desert";
028    }
029
030    @Override
031    public boolean canRain() {
032        return false;
033    }
034}
