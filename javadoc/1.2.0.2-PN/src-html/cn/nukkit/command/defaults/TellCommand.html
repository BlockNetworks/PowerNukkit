001package cn.nukkit.command.defaults;
002
003import cn.nukkit.Player;
004import cn.nukkit.command.CommandSender;
005import cn.nukkit.command.data.CommandParamType;
006import cn.nukkit.command.data.CommandParameter;
007import cn.nukkit.lang.TranslationContainer;
008import cn.nukkit.utils.TextFormat;
009
010import java.util.Objects;
011
012/**
013 * Created on 2015/11/12 by xtypr.
014 * Package cn.nukkit.command.defaults in project Nukkit .
015 */
016public class TellCommand extends VanillaCommand {
017
018    public TellCommand(String name) {
019        super(name, "%nukkit.command.tell.description", "%commands.message.usage", new String[]{"w", "msg"});
020        this.setPermission("nukkit.command.tell");
021        this.commandParameters.clear();
022        this.commandParameters.put("default", new CommandParameter[]{
023                new CommandParameter("player", CommandParamType.TARGET, false),
024                new CommandParameter("message")
025        });
026    }
027
028    @Override
029    public boolean execute(CommandSender sender, String commandLabel, String[] args) {
030        if (!this.testPermission(sender)) {
031            return true;
032        }
033
034        if (args.length < 2) {
035            sender.sendMessage(new TranslationContainer("commands.generic.usage", this.usageMessage));
036
037            return false;
038        }
039
040        String name = args[0].toLowerCase();
041
042        Player player = sender.getServer().getPlayer(name);
043        if (player == null) {
044            sender.sendMessage(new TranslationContainer("commands.generic.player.notFound"));
045            return true;
046        }
047
048        if (Objects.equals(player, sender)) {
049            sender.sendMessage(new TranslationContainer(TextFormat.RED + "%commands.message.sameTarget"));
050            return true;
051        }
052
053        String msg = "";
054        for (int i = 1; i < args.length; i++) {
055            msg += args[i] + " ";
056        }
057        if (msg.length() > 0) {
058            msg = msg.substring(0, msg.length() - 1);
059        }
060
061        String displayName = (sender instanceof Player ? ((Player) sender).getDisplayName() : sender.getName());
062
063        sender.sendMessage("[" + sender.getName() + " -> " + player.getDisplayName() + "] " + msg);
064        player.sendMessage("[" + displayName + " -> " + player.getName() + "] " + msg);
065
066        return true;
067    }
068}
