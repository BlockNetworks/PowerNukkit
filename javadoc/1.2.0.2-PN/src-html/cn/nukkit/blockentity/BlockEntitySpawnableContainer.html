001package cn.nukkit.blockentity;
002
003import cn.nukkit.Player;
004import cn.nukkit.block.BlockAir;
005import cn.nukkit.inventory.ContainerInventory;
006import cn.nukkit.inventory.InventoryHolder;
007import cn.nukkit.item.Item;
008import cn.nukkit.item.ItemBlock;
009import cn.nukkit.level.format.FullChunk;
010import cn.nukkit.nbt.NBTIO;
011import cn.nukkit.nbt.tag.CompoundTag;
012import cn.nukkit.nbt.tag.ListTag;
013
014import java.util.HashSet;
015
016public abstract class BlockEntitySpawnableContainer extends BlockEntitySpawnable implements InventoryHolder, BlockEntityContainer {
017    protected ContainerInventory inventory;
018
019    public BlockEntitySpawnableContainer(FullChunk chunk, CompoundTag nbt) {
020        super(chunk, nbt);
021    }
022
023    @Override
024    protected void initBlockEntity() {
025        if (!this.namedTag.contains("Items") || !(this.namedTag.get("Items") instanceof ListTag)) {
026            this.namedTag.putList(new ListTag<CompoundTag>("Items"));
027        }
028
029        ListTag<CompoundTag> list = (ListTag<CompoundTag>) this.namedTag.getList("Items");
030        for (CompoundTag compound : list.getAll()) {
031            Item item = NBTIO.getItemHelper(compound);
032            this.inventory.slots.put(compound.getByte("Slot"), item);
033        }
034
035        super.initBlockEntity();
036    }
037
038    @Override
039    public void close() {
040        if (!closed) {
041            for (Player player : new HashSet<>(this.getInventory().getViewers())) {
042                player.removeWindow(this.getInventory());
043            }
044            super.close();
045        }
046    }
047
048    @Override
049    public void onBreak() {
050        for (Item content : inventory.getContents().values()) {
051            level.dropItem(this, content);
052        }
053        inventory.clearAll(); // Stop items from being moved around by another player in the inventory
054    }
055
056    @Override
057    public void saveNBT() {
058        this.namedTag.putList(new ListTag<CompoundTag>("Items"));
059        for (int index = 0; index < this.getSize(); index++) {
060            this.setItem(index, this.inventory.getItem(index));
061        }
062    }
063
064    protected int getSlotIndex(int index) {
065        ListTag<CompoundTag> list = this.namedTag.getList("Items", CompoundTag.class);
066        for (int i = 0; i < list.size(); i++) {
067            if (list.get(i).getByte("Slot") == index) {
068                return i;
069            }
070        }
071
072        return -1;
073    }
074
075    @Override
076    public Item getItem(int index) {
077        int i = this.getSlotIndex(index);
078        if (i < 0) {
079            return new ItemBlock(new BlockAir(), 0, 0);
080        } else {
081            CompoundTag data = (CompoundTag) this.namedTag.getList("Items").get(i);
082            return NBTIO.getItemHelper(data);
083        }
084    }
085
086    @Override
087    public void setItem(int index, Item item) {
088        int i = this.getSlotIndex(index);
089
090        CompoundTag d = NBTIO.putItemHelper(item, index);
091
092        // If item is air or count less than 0, remove the item from the "Items" list
093        if (item.getId() == Item.AIR || item.getCount() <= 0) {
094            if (i >= 0) {
095                this.namedTag.getList("Items").remove(i);
096            }
097        } else if (i < 0) {
098            // If it is less than i, then it is a new item, so we are going to add it at the end of the list
099            (this.namedTag.getList("Items", CompoundTag.class)).add(d);
100        } else {
101            // If it is more than i, then it is an update on a inventorySlot, so we are going to overwrite the item in the list
102            (this.namedTag.getList("Items", CompoundTag.class)).add(i, d);
103        }
104    }
105}
