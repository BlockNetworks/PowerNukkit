001package cn.nukkit.inventory;
002
003import cn.nukkit.item.Item;
004
005import java.util.*;
006
007/**
008 * author: MagicDroidX
009 * Nukkit Project
010 */
011public class ShapelessRecipe implements CraftingRecipe {
012
013    private String recipeId;
014
015    private final Item output;
016
017    private long least,most;
018
019    private final List<Item> ingredients;
020
021    private final int priority;
022
023    public ShapelessRecipe(Item result, Collection<Item> ingredients) {
024        this(null, 10, result, ingredients);
025    }
026
027    public ShapelessRecipe(String recipeId, int priority, Item result, Collection<Item> ingredients) {
028        this.recipeId = recipeId;
029        this.priority = priority;
030        this.output = result.clone();
031        if (ingredients.size() > 9) {
032            throw new IllegalArgumentException("Shapeless recipes cannot have more than 9 ingredients");
033        }
034
035        this.ingredients = new ArrayList<>();
036
037        for (Item item : ingredients) {
038            if (item.getCount() < 1) {
039                throw new IllegalArgumentException("Recipe '" + recipeId + "' Ingredient amount was not 1 (value: " + item.getCount() + ")");
040            }
041            this.ingredients.add(item.clone());
042        }
043    }
044
045    @Override
046    public Item getResult() {
047        return this.output.clone();
048    }
049
050    @Override
051    public String getRecipeId() {
052        return this.recipeId;
053    }
054
055    @Override
056    public UUID getId() {
057        return new UUID(least, most);
058    }
059
060    @Override
061    public void setId(UUID uuid) {
062        this.least = uuid.getLeastSignificantBits();
063        this.most = uuid.getMostSignificantBits();
064
065        if (this.recipeId == null) {
066            this.recipeId = this.getId().toString();
067        }
068    }
069
070    public List<Item> getIngredientList() {
071        List<Item> ingredients = new ArrayList<>();
072        for (Item ingredient : this.ingredients) {
073            ingredients.add(ingredient.clone());
074        }
075
076        return ingredients;
077    }
078
079    public int getIngredientCount() {
080        return ingredients.size();
081    }
082
083    @Override
084    public void registerToCraftingManager(CraftingManager manager) {
085        manager.registerShapelessRecipe(this);
086    }
087
088    @Override
089    public RecipeType getType() {
090        return RecipeType.SHAPELESS;
091    }
092
093    @Override
094    public boolean requiresCraftingTable() {
095        return this.ingredients.size() > 4;
096    }
097
098    @Override
099    public List<Item> getExtraResults() {
100        return new ArrayList<>();
101    }
102
103    @Override
104    public List<Item> getAllResults() {
105        return null;
106    }
107
108    @Override
109    public int getPriority() {
110        return this.priority;
111    }
112
113    @Override
114    public boolean matchItems(Item[][] input, Item[][] output) {
115        List<Item> haveInputs = new ArrayList<>();
116        for (Item[] items : input) {
117            haveInputs.addAll(Arrays.asList(items));
118        }
119        haveInputs.sort(CraftingManager.recipeComparator);
120
121        List<Item> needInputs = this.getIngredientList();
122
123        if (!this.matchItemList(haveInputs, needInputs)) {
124            return false;
125        }
126
127        List<Item> haveOutputs = new ArrayList<>();
128        for (Item[] items : output) {
129            haveOutputs.addAll(Arrays.asList(items));
130        }
131        haveOutputs.sort(CraftingManager.recipeComparator);
132        List<Item> needOutputs = this.getExtraResults();
133
134        return this.matchItemList(haveOutputs, needOutputs);
135    }
136
137
138    private boolean matchItemList(List<Item> haveItems, List<Item> needItems) {
139        // Remove any air blocks that may have gotten through.
140        haveItems.removeIf(Item::isNull);
141
142        if (haveItems.size() != needItems.size()) {
143            return false;
144        }
145
146        int size = needItems.size();
147        int completed = 0;
148        for (int i = 0; i < size; i++) {
149            Item haveItem = haveItems.get(i);
150            Item needItem = needItems.get(i);
151
152            if (needItem.equals(haveItem, needItem.hasMeta(), needItem.hasCompoundTag())) {
153                completed++;
154            }
155        }
156
157        return completed == size;
158    }
159}
