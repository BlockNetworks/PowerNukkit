001package cn.nukkit.inventory.transaction.action;
002
003import cn.nukkit.Player;
004import cn.nukkit.block.Block;
005import cn.nukkit.block.BlockAnvil;
006import cn.nukkit.block.BlockID;
007import cn.nukkit.event.block.AnvilDamageEvent;
008import cn.nukkit.inventory.AnvilInventory;
009import cn.nukkit.inventory.transaction.CraftingTransaction;
010import cn.nukkit.item.Item;
011import cn.nukkit.level.Sound;
012
013public class DamageAnvilAction extends InventoryAction {
014    
015    private final AnvilInventory anvil;
016    private boolean shouldDamage;
017    private CraftingTransaction transaction;
018    
019    public DamageAnvilAction(AnvilInventory anvil, boolean shouldDamage, CraftingTransaction transaction) {
020        super(Item.get(0), Item.get(0));
021        this.anvil = anvil;
022        this.shouldDamage = shouldDamage;
023        this.transaction = transaction;
024    }
025    
026    @Override
027    public boolean isValid(Player source) {
028        return true;
029    }
030    
031    @Override
032    public boolean execute(Player source) {
033        Block levelBlock = anvil.getHolder().getLevelBlock();
034        if (!(levelBlock instanceof BlockAnvil)) {
035            return false;
036        }
037        Block newState = levelBlock.clone();
038        int damage = (newState.getDamage() >> 2 & 0x3) + 1;
039        if (damage >= 3) {
040            newState = Block.get(0, 0, newState, newState.layer);
041        } else {
042            newState.setDamage(newState.getDamage() & (Block.DATA_MASK ^ 0b1100) | (damage << 2));
043        }
044        AnvilDamageEvent ev = new AnvilDamageEvent(levelBlock, newState, source, transaction, AnvilDamageEvent.Cause.USE);
045        ev.setCancelled(!shouldDamage);
046        source.getServer().getPluginManager().callEvent(ev);
047        if (ev.isCancelled()) {
048            levelBlock.getLevel().addSound(levelBlock, Sound.RANDOM_ANVIL_USE);
049            return true;
050        } else {
051            if (newState.getId() == BlockID.AIR) {
052                levelBlock.getLevel().addSound(levelBlock, Sound.RANDOM_ANVIL_BREAK);
053            } else {
054                levelBlock.getLevel().addSound(levelBlock, Sound.RANDOM_ANVIL_USE);
055            }
056            return levelBlock.getLevel().setBlock(levelBlock, newState, true, true);
057        }
058    }
059    
060    @Override
061    public void onExecuteSuccess(Player source) {
062    
063    }
064    
065    @Override
066    public void onExecuteFail(Player source) {
067    
068    }
069}
