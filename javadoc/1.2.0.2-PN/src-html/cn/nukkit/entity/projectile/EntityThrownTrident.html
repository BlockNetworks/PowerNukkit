025/**
026 * Created by PetteriM1
027 */
028public class EntityThrownTrident extends EntityProjectile {
029
030    public static final int NETWORK_ID = 73;
031
032    public static final int DATA_SOURCE_ID = 17;
033
034    protected Item trident;
035
036    @Override
037    public int getNetworkId() {
038        return NETWORK_ID;
039    }
040
041    @Override
042    public float getWidth() {
043        return 0.05f;
044    }
045
046    @Override
047    public float getLength() {
048        return 0.5f;
049    }
050
051    @Override
052    public float getHeight() {
053        return 0.05f;
054    }
055
056    @Override
057    public float getGravity() {
058        return 0.04f;
059    }
060
061    @Override
062    public float getDrag() {
063        return 0.01f;
064    }
065
066    protected float gravity = 0.04f;
067    protected float drag = 0.01f;
068
069    public EntityThrownTrident(FullChunk chunk, CompoundTag nbt) {
070        this(chunk, nbt, null);
071    }
072
073    public EntityThrownTrident(FullChunk chunk, CompoundTag nbt, Entity shootingEntity) {
074        this(chunk, nbt, shootingEntity, false);
075    }
076
077    public EntityThrownTrident(FullChunk chunk, CompoundTag nbt, Entity shootingEntity, boolean critical) {
078        super(chunk, nbt, shootingEntity);
079    }
080
081    @Override
082    protected void initEntity() {
083        super.initEntity();
084
085        this.damage = namedTag.contains("damage") ? namedTag.getDouble("damage") : 8;
086        this.trident = namedTag.contains("Trident") ? NBTIO.getItemHelper(namedTag.getCompound("Trident")) : Item.get(0);
087
088        closeOnCollide = false;
089    }
090
091    @Override
092    public void saveNBT() {
093        super.saveNBT();
094
095        this.namedTag.put("Trident", NBTIO.putItemHelper(this.trident));
096    }
097
098    public Item getItem() {
099        return this.trident != null ? this.trident.clone() : Item.get(0);
100    }
101
102    public void setItem(Item item) {
103        this.trident = item.clone();
104    }
105
106    public void setCritical() {
107        this.setCritical(true);
108    }
109
110    public void setCritical(boolean value) {
111        this.setDataFlag(DATA_FLAGS, DATA_FLAG_CRITICAL, value);
112    }
113
114    public boolean isCritical() {
115        return this.getDataFlag(DATA_FLAGS, DATA_FLAG_CRITICAL);
116    }
117
118    @Override
119    public int getResultDamage() {
120        int base = super.getResultDamage();
121
122        if (this.isCritical()) {
123            base += ThreadLocalRandom.current().nextInt(base / 2 + 2);
124        }
125
126        return base;
127    }
128
129    @Override
130    protected double getBaseDamage() {
131        return 8;
132    }
133
134    @Override
135    public boolean onUpdate(int currentTick) {
136        if (this.closed) {
137            return false;
138        }
139
140        this.timing.startTiming();
141
142        if (this.isCollided && !this.hadCollision) {
143            this.getLevel().addLevelSoundEvent(this, LevelSoundEventPacket.SOUND_ITEM_TRIDENT_HIT_GROUND);
144        }
145
146        boolean hasUpdate = super.onUpdate(currentTick);
147
148        if (this.onGround || this.hadCollision) {
149            this.setCritical(false);
150        }
151
152        if (this.age > 1200) {
153            this.close();
154            hasUpdate = true;
155        }
156
157        this.timing.stopTiming();
158
159        return hasUpdate;
160    }
161
162    @Override
163    public void spawnTo(Player player) {
164        AddEntityPacket pk = new AddEntityPacket();
165        pk.type = NETWORK_ID;
166        pk.entityUniqueId = this.getId();
167        pk.entityRuntimeId = this.getId();
168        pk.x = (float) this.x;
169        pk.y = (float) this.y;
170        pk.z = (float) this.z;
171        pk.speedX = (float) this.motionX;
172        pk.speedY = (float) this.motionY;
173        pk.speedZ = (float) this.motionZ;
174        pk.yaw = (float) this.yaw;
175        pk.pitch = (float) this.pitch;
176        pk.metadata = this.dataProperties;
177        player.dataPacket(pk);
178
179        super.spawnTo(player);
180    }
181
182    @Override
183    public void onCollideWithEntity(Entity entity) {
184        this.server.getPluginManager().callEvent(new ProjectileHitEvent(this, MovingObjectPosition.fromEntity(entity)));
185        float damage = this.getResultDamage();
186
187        EntityDamageEvent ev;
188        if (this.shootingEntity == null) {
189            ev = new EntityDamageByEntityEvent(this, entity, DamageCause.PROJECTILE, damage);
190        } else {
191            ev = new EntityDamageByChildEntityEvent(this.shootingEntity, this, entity, DamageCause.PROJECTILE, damage);
192        }
193        entity.attack(ev);
194        this.getLevel().addLevelSoundEvent(this, LevelSoundEventPacket.SOUND_ITEM_TRIDENT_HIT);
195        this.hadCollision = true;
196        this.close();
197        Entity newTrident = create("ThrownTrident", this);
198        ((EntityThrownTrident) newTrident).setItem(this.trident);
199        newTrident.spawnToAll();
200    }
201
202    public Entity create(Object type, Position source, Object... args) {
203        FullChunk chunk = source.getLevel().getChunk((int) source.x >> 4, (int) source.z >> 4);
204        if (chunk == null) return null;
205
206        CompoundTag nbt = new CompoundTag()
207                .putList(new ListTag<DoubleTag>("Pos")
208                        .add(new DoubleTag("", source.x + 0.5))
209                        .add(new DoubleTag("", source.y))
210                        .add(new DoubleTag("", source.z + 0.5)))
211                .putList(new ListTag<DoubleTag>("Motion")
212                        .add(new DoubleTag("", 0))
213                        .add(new DoubleTag("", 0))
214                        .add(new DoubleTag("", 0)))
215                .putList(new ListTag<FloatTag>("Rotation")
216                        .add(new FloatTag("", new Random().nextFloat() * 360))
217                        .add(new FloatTag("", 0)));
218
219        return Entity.createEntity(type.toString(), chunk, nbt, args);
220    }
