001package cn.nukkit.level.biome.impl.extremehills;
002
003import cn.nukkit.block.Block;
004import cn.nukkit.level.generator.noise.nukkit.f.SimplexF;
005import cn.nukkit.math.NukkitRandom;
006
007/**
008 * author: DaPorkchop_
009 * Nukkit Project
010 * <p>
011 * make sure this is touching another extreme hills type or it'll look dumb
012 *
013 * very smooth hills with flat areas between
014 */
015public class ExtremeHillsMBiome extends ExtremeHillsPlusBiome {
016    private static final SimplexF gravelNoise = new SimplexF(new NukkitRandom(0), 1f, 1 / 4f, 1 / 64f);
017    private boolean isGravel = false;
018
019    public ExtremeHillsMBiome() {
020        this(true);
021    }
022
023    public ExtremeHillsMBiome(boolean tree) {
024        super(tree);
025
026        this.setBaseHeight(1f);
027        this.setHeightVariation(0.5f);
028    }
029
030    @Override
031    public String getName() {
032        return "Extreme Hills M";
033    }
034
035    @Override
036    public int getSurfaceId(int x, int y, int z) {
037        return gravelNoise.noise2D(x, z, true) < -0.75f ? GRAVEL << Block.DATA_BITS : super.getSurfaceId(x, y, z);
038    }
039
040    @Override
041    public int getSurfaceDepth(int x, int y, int z) {
042        return gravelNoise.noise2D(x, z, true) < -0.75f ? 4 : super.getSurfaceDepth(x, y, z);
043    }
044
045    @Override
046    public int getGroundDepth(int x, int y, int z) {
047        return gravelNoise.noise2D(x, z, true) < -0.75f ? 0 : super.getGroundDepth(x, y, z);
048    }
049
050    @Override
051    public boolean doesOverhang() {
052        return false;
053    }
054}
