084    private AxisAlignedBB getRelativeBoundingBox() {
085        int damage = this.getDamage();
086        if (damage > 11) {
087            this.setDamage(damage = 11);
088        }
089        AxisAlignedBB boundingBox = ALL[damage];
090        if (boundingBox != null) return boundingBox;
091
092        AxisAlignedBB[] bbs;
093
094        switch (getDamage()) {
095            case 1:
096            case 5:
097            case 9:
098                bbs = EAST;
099                break;
100            case 2:
101            case 6:
102            case 10:
103                bbs = SOUTH;
104                break;
105            case 3:
106            case 7:
107            case 11:
108                bbs = WEST;
109                break;
110            default:
111                bbs = NORTH;
112                break;
113        }
114
115        return ALL[damage] = bbs[this.getDamage() >> 2];
116    }

118    @Override
119    public boolean place(Item item, Block block, Block target, BlockFace face, double fx, double fy, double fz, Player player) {
120        if (target.getId() == Block.WOOD && (target.getDamage() & 0x03) == BlockWood.JUNGLE) {
121            if (face != BlockFace.DOWN && face != BlockFace.UP) {
122                int[] faces = new int[]{
123                        0,
124                        0,
125                        0,
126                        2,
127                        3,
128                        1,
129                };
130
131                this.setDamage(faces[face.getIndex()]);
132                this.level.setBlock(block, this, true, true);
133                return true;
134            }
135        }
136        return false;
137    }

139    @Override
140    public int onUpdate(int type) {
141        if (type == Level.BLOCK_UPDATE_NORMAL) {
142            int[] faces = new int[]{
143                    3, 4, 2, 5, 3, 4, 2, 5, 3, 4, 2, 5
144            };
145
146            Block side = this.getSide(BlockFace.fromIndex(faces[this.getDamage()]));
147
148            if (side.getId() != Block.WOOD && side.getDamage() != BlockWood.JUNGLE) {
149                this.getLevel().useBreakOn(this);
150                return Level.BLOCK_UPDATE_NORMAL;
151            }
152        } else if (type == Level.BLOCK_UPDATE_RANDOM) {
153            if (ThreadLocalRandom.current().nextInt(2) == 1) {
154                if (this.getGrowthStage() < 2) {
155                    if (!this.grow()) {
156                        return Level.BLOCK_UPDATE_RANDOM;
157                    }
158                }
159            } else {
160                return Level.BLOCK_UPDATE_RANDOM;
161            }
162        }
163
164        return 0;
165    }

172    @Override
173    public boolean onActivate(Item item, Player player) {
174        if (item.getId() == Item.DYE && item.getDamage() == 0x0f) {
175            if (this.getGrowthStage() < 2) {
176                if (!this.grow()) {
177                    return false;
178                }
179                this.level.addParticle(new BoneMealParticle(this));
180
181                if (player != null && (player.gamemode & 0x01) == 0) {
182                    item.count--;
183                }
184            }
185
186            return true;
187        }
188
189        return false;
190    }
191
192    private int getGrowthStage() {
193        return this.getDamage() / 4;
194    }
195
196    private boolean grow() {
197        Block block = this.clone();
198        block.setDamage(block.getDamage() + 4);
199        BlockGrowEvent ev = new BlockGrowEvent(this, block);
200        Server.getInstance().getPluginManager().callEvent(ev);
201        return !ev.isCancelled() && this.getLevel().setBlock(this, ev.getNewState(), true, true);
202    }
