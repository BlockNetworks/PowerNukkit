001package cn.nukkit.block;
002
003import cn.nukkit.api.PowerNukkitOnly;
004import cn.nukkit.api.Since;
005import cn.nukkit.math.BlockFace;
006
007import java.util.EnumSet;
008import java.util.Iterator;
009import java.util.Set;
010
011@PowerNukkitOnly
012@Since("1.3.0.0-PN")
013public interface BlockConnectable {
014    @PowerNukkitOnly
015    @Since("1.3.0.0-PN")
016    Block getSideAtLayer(int layer, BlockFace face);
017    
018    @PowerNukkitOnly
019    @Since("1.3.0.0-PN")
020    boolean canConnect(Block block);
021    
022    @PowerNukkitOnly
023    @Since("1.3.0.0-PN")
024    default boolean isStraight() {
025        Set<BlockFace> connections = getConnections();
026        if (connections.size() != 2) {
027            return false;
028        }
029
030        Iterator<BlockFace> iterator = connections.iterator();
031        BlockFace a = iterator.next();
032        BlockFace b = iterator.next();
033        return a.getOpposite() == b;
034    }
035
036    @PowerNukkitOnly
037    @Since("1.3.0.0-PN")
038    default Set<BlockFace> getConnections() {
039        EnumSet<BlockFace> connections = EnumSet.noneOf(BlockFace.class);
040        for (BlockFace blockFace : BlockFace.Plane.HORIZONTAL) {
041            if (isConnected(blockFace)) {
042                connections.add(blockFace);
043            }
044        }
045        return connections;
046    }
047
048    @PowerNukkitOnly
049    @Since("1.3.0.0-PN")
050    default boolean isConnected(BlockFace face) {
051        return canConnect(getSideAtLayer(0, face));
052    }
053}
