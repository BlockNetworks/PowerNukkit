102    protected int getFlowDecay(Block block) {
103        if (block.getId() != this.getId()) {
104            Block layer1 = block.getLevelBlockAtLayer(1);
105            if (layer1.getId() != this.getId()) {
106                return -1;
107            } else {
108                return layer1.getDamage();
109            }
110        }
111        return block.getDamage();
112    }

114    protected int getEffectiveFlowDecay(Block block) {
115        if (block.getId() != this.getId()) {
116            block = block.getLevelBlockAtLayer(1);
117            if (block.getId() != this.getId()) {
118                return -1;
119            }
120        }
121        int decay = block.getDamage();
122        if (decay >= 8) {
123            decay = 0;
124        }
125        return decay;
126    }

133    public Vector3 getFlowVector() {
134        if (this.flowVector != null) {
135            return this.flowVector;
136        }
137        Vector3 vector = new Vector3(0, 0, 0);
138        int decay = this.getEffectiveFlowDecay(this);
139        for (int j = 0; j < 4; ++j) {
140            int x = (int) this.x;
141            int y = (int) this.y;
142            int z = (int) this.z;
143            switch (j) {
144                case 0:
145                    --x;
146                    break;
147                case 1:
148                    x++;
149                    break;
150                case 2:
151                    z--;
152                    break;
153                default:
154                    z++;
155            }
156            Block sideBlock = this.level.getBlock(x, y, z);
157            int blockDecay = this.getEffectiveFlowDecay(sideBlock);
158            if (blockDecay < 0) {
159                if (!sideBlock.canBeFlowedInto()) {
160                    continue;
161                }
162                blockDecay = this.getEffectiveFlowDecay(this.level.getBlock(x, y - 1, z));
163                if (blockDecay >= 0) {
164                    int realDecay = blockDecay - (decay - 8);
165                    vector.x += (sideBlock.x - this.x) * realDecay;
166                    vector.y += (sideBlock.y - this.y) * realDecay;
167                    vector.z += (sideBlock.z - this.z) * realDecay;
168                }
169            } else {
170                int realDecay = blockDecay - decay;
171                vector.x += (sideBlock.x - this.x) * realDecay;
172                vector.y += (sideBlock.y - this.y) * realDecay;
173                vector.z += (sideBlock.z - this.z) * realDecay;
174            }
175        }
176        if (this.getDamage() >= 8) {
177            if (!this.canFlowInto(this.level.getBlock((int) this.x, (int) this.y, (int) this.z - 1)) ||
178                    !this.canFlowInto(this.level.getBlock((int) this.x, (int) this.y, (int) this.z + 1)) ||
179                    !this.canFlowInto(this.level.getBlock((int) this.x - 1, (int) this.y, (int) this.z)) ||
180                    !this.canFlowInto(this.level.getBlock((int) this.x + 1, (int) this.y, (int) this.z)) ||
181                    !this.canFlowInto(this.level.getBlock((int) this.x, (int) this.y + 1, (int) this.z - 1)) ||
182                    !this.canFlowInto(this.level.getBlock((int) this.x, (int) this.y + 1, (int) this.z + 1)) ||
183                    !this.canFlowInto(this.level.getBlock((int) this.x - 1, (int) this.y + 1, (int) this.z)) ||
184                    !this.canFlowInto(this.level.getBlock((int) this.x + 1, (int) this.y + 1, (int) this.z))) {
185                vector = vector.normalize().add(0, -6, 0);
186            }
187        }
188        return this.flowVector = vector.normalize();
189    }

205    @Override
206    public int onUpdate(int type) {
207        if (type == Level.BLOCK_UPDATE_NORMAL) {
208            this.checkForHarden();
209            if (usesWaterLogging() && layer > 0) {
210                Block layer0 = this.level.getBlock(this, 0);
211                if (layer0.getId() == 0) {
212                    this.level.setBlock(this, 1, Block.get(Block.AIR), false, false);
213                    this.level.setBlock(this, 0, this, false, false);
214                } else if (layer0.getWaterloggingLevel() <= 0 || layer0.getWaterloggingLevel() == 1 && getDamage() > 0) {
215                    this.level.setBlock(this, 1, Block.get(Block.AIR), true, true);
216                }
217            }
218            this.level.scheduleUpdate(this, this.tickRate());
219            return 0;
220        } else if (type == Level.BLOCK_UPDATE_SCHEDULED) {
221            int decay = this.getFlowDecay(this);
222            int multiplier = this.getFlowDecayPerBlock();
223            if (decay > 0) {
224                int smallestFlowDecay = -100;
225                this.adjacentSources = 0;
226                smallestFlowDecay = this.getSmallestFlowDecay(this.level.getBlock((int) this.x, (int) this.y, (int) this.z - 1), smallestFlowDecay);
227                smallestFlowDecay = this.getSmallestFlowDecay(this.level.getBlock((int) this.x, (int) this.y, (int) this.z + 1), smallestFlowDecay);
228                smallestFlowDecay = this.getSmallestFlowDecay(this.level.getBlock((int) this.x - 1, (int) this.y, (int) this.z), smallestFlowDecay);
229                smallestFlowDecay = this.getSmallestFlowDecay(this.level.getBlock((int) this.x + 1, (int) this.y, (int) this.z), smallestFlowDecay);
230                int newDecay = smallestFlowDecay + multiplier;
231                if (newDecay >= 8 || smallestFlowDecay < 0) {
232                    newDecay = -1;
233                }
234                int topFlowDecay = this.getFlowDecay(this.level.getBlock((int) this.x, (int) this.y + 1, (int) this.z));
235                if (topFlowDecay >= 0) {
236                    newDecay = topFlowDecay | 0x08;
237                }
238                if (this.adjacentSources >= 2 && this instanceof BlockWater) {
239                    Block bottomBlock = this.level.getBlock((int) this.x, (int) this.y - 1, (int) this.z);
240                    if (bottomBlock.isSolid()) {
241                        newDecay = 0;
242                    } else if (bottomBlock instanceof BlockWater && bottomBlock.getDamage() == 0) {
243                        newDecay = 0;
244                    } else {
245                        bottomBlock = bottomBlock.getLevelBlockAtLayer(1);
246                        if (bottomBlock instanceof BlockWater && bottomBlock.getDamage() == 0) {
247                            newDecay = 0;
248                        }
249                    }
250                }
251                if (newDecay != decay) {
252                    decay = newDecay;
253                    boolean decayed = decay < 0;
254                    Block to;
255                    if (decayed) {
256                        to = Block.get(BlockID.AIR);
257                    } else {
258                        to = getBlock(decay);
259                    }
260                    BlockFromToEvent event = new BlockFromToEvent(this, to);
261                    level.getServer().getPluginManager().callEvent(event);
262                    if (!event.isCancelled()) {
263                        this.level.setBlock(this, layer, event.getTo(), true, true);
264                        if (!decayed) {
265                            this.level.scheduleUpdate(this, this.tickRate());
266                        }
267                    }
268                }
269            }
270            if (decay >= 0) {
271                Block bottomBlock = this.level.getBlock((int) this.x, (int) this.y - 1, (int) this.z);
272                this.flowIntoBlock(bottomBlock, decay | 0x08);
273                if (decay == 0 || !(usesWaterLogging()? bottomBlock.canWaterloggingFlowInto(): bottomBlock.canBeFlowedInto())) {
274                    int adjacentDecay;
275                    if (decay >= 8) {
276                        adjacentDecay = 1;
277                    } else {
278                        adjacentDecay = decay + multiplier;
279                    }
280                    if (adjacentDecay < 8) {
281                        boolean[] flags = this.getOptimalFlowDirections();
282                        if (flags[0]) {
283                            this.flowIntoBlock(this.level.getBlock((int) this.x - 1, (int) this.y, (int) this.z), adjacentDecay);
284                        }
285                        if (flags[1]) {
286                            this.flowIntoBlock(this.level.getBlock((int) this.x + 1, (int) this.y, (int) this.z), adjacentDecay);
287                        }
288                        if (flags[2]) {
289                            this.flowIntoBlock(this.level.getBlock((int) this.x, (int) this.y, (int) this.z - 1), adjacentDecay);
290                        }
291                        if (flags[3]) {
292                            this.flowIntoBlock(this.level.getBlock((int) this.x, (int) this.y, (int) this.z + 1), adjacentDecay);
293                        }
294                    }
295                }
296                this.checkForHarden();
297            }
298        }
299        return 0;
300    }

302    protected void flowIntoBlock(Block block, int newFlowDecay) {
303        if (this.canFlowInto(block) && !(block instanceof BlockLiquid)) {
304            if (usesWaterLogging()) {
305                Block layer1 = block.getLevelBlockAtLayer(1);
306                if (layer1 instanceof BlockLiquid) {
307                    return;
308                }
309
310                if (block.getWaterloggingLevel() > 1) {
311                    block = layer1;
312                }
313            }
314
315            LiquidFlowEvent event = new LiquidFlowEvent(block, this, newFlowDecay);
316            level.getServer().getPluginManager().callEvent(event);
317            if (!event.isCancelled()) {
318                if (block.layer == 0 && block.getId() > 0) {
319                    this.level.useBreakOn(block);
320                }
321                this.level.setBlock(block, block.layer, getBlock(newFlowDecay), true, true);
322                this.level.scheduleUpdate(block, this.tickRate());
323            }
324        }
325    }

327    private int calculateFlowCost(int blockX, int blockY, int blockZ, int accumulatedCost, int maxCost, int originOpposite, int lastOpposite) {
328        int cost = 1000;
329        for (int j = 0; j < 4; ++j) {
330            if (j == originOpposite || j == lastOpposite) {
331                continue;
332            }
333            int x = blockX;
334            int y = blockY;
335            int z = blockZ;
336            if (j == 0) {
337                --x;
338            } else if (j == 1) {
339                ++x;
340            } else if (j == 2) {
341                --z;
342            } else if (j == 3) {
343                ++z;
344            }
345            long hash = Level.blockHash(x, y, z);
346            if (!this.flowCostVisited.containsKey(hash)) {
347                Block blockSide = this.level.getBlock(x, y, z);
348                if (!this.canFlowInto(blockSide)) {
349                    this.flowCostVisited.put(hash, BLOCKED);
350                } else if (usesWaterLogging()?
351                        this.level.getBlock(x, y - 1, z).canWaterloggingFlowInto() :
352                        this.level.getBlock(x, y - 1, z).canBeFlowedInto()) {
353                    this.flowCostVisited.put(hash, CAN_FLOW_DOWN);
354                } else {
355                    this.flowCostVisited.put(hash, CAN_FLOW);
356                }
357            }
358            byte status = this.flowCostVisited.get(hash);
359            if (status == BLOCKED) {
360                continue;
361            } else if (status == CAN_FLOW_DOWN) {
362                return accumulatedCost;
363            }
364            if (accumulatedCost >= maxCost) {
365                continue;
366            }
367            int realCost = this.calculateFlowCost(x, y, z, accumulatedCost + 1, maxCost, originOpposite, j ^ 0x01);
368            if (realCost < cost) {
369                cost = realCost;
370            }
371        }
372        return cost;
373    }

385    private boolean[] getOptimalFlowDirections() {
386        int[] flowCost = new int[]{
387                1000,
388                1000,
389                1000,
390                1000
391        };
392        int maxCost = 4 / this.getFlowDecayPerBlock();
393        for (int j = 0; j < 4; ++j) {
394            int x = (int) this.x;
395            int y = (int) this.y;
396            int z = (int) this.z;
397            if (j == 0) {
398                --x;
399            } else if (j == 1) {
400                ++x;
401            } else if (j == 2) {
402                --z;
403            } else {
404                ++z;
405            }
406            Block block = this.level.getBlock(x, y, z);
407            if (!this.canFlowInto(block)) {
408                this.flowCostVisited.put(Level.blockHash(x, y, z), BLOCKED);
409            } else if (usesWaterLogging()?
410                    this.level.getBlock(x, y - 1, z).canWaterloggingFlowInto():
411                    this.level.getBlock(x, y - 1, z).canBeFlowedInto()) {
412                this.flowCostVisited.put(Level.blockHash(x, y, z), CAN_FLOW_DOWN);
413                flowCost[j] = maxCost = 0;
414            } else if (maxCost > 0) {
415                this.flowCostVisited.put(Level.blockHash(x, y, z), CAN_FLOW);
416                flowCost[j] = this.calculateFlowCost(x, y, z, 1, maxCost, j ^ 0x01, j ^ 0x01);
417                maxCost = Math.min(maxCost, flowCost[j]);
418            }
419        }
420        this.flowCostVisited.clear();
421        double minCost = Double.MAX_VALUE;
422        for (int i = 0; i < 4; i++) {
423            double d = flowCost[i];
424            if (d < minCost) {
425                minCost = d;
426            }
427        }
428        boolean[] isOptimalFlowDirection = new boolean[4];
429        for (int i = 0; i < 4; ++i) {
430            isOptimalFlowDirection[i] = (flowCost[i] == minCost);
431        }
432        return isOptimalFlowDirection;
433    }

435    private int getSmallestFlowDecay(Block block, int decay) {
436        int blockDecay = this.getFlowDecay(block);
437        if (blockDecay < 0) {
438            return decay;
439        } else if (blockDecay == 0) {
440            ++this.adjacentSources;
441        } else if (blockDecay >= 8) {
442            blockDecay = 0;
443        }
444        return (decay >= 0 && blockDecay >= decay) ? decay : blockDecay;
445    }

471        BlockFromToEvent event = new BlockFromToEvent(this, result);
472        this.level.getServer().getPluginManager().callEvent(event);
473        if (event.isCancelled()) {
474            return false;
475        }
476        this.level.setBlock(this, event.getTo(), true, true);
477        this.level.setBlock(this, 1, Block.get(Block.AIR), true, true);
478        this.getLevel().addLevelSoundEvent(this.add(0.5, 0.5, 0.5), LevelSoundEventPacket.SOUND_FIZZ);
479        return true;

482    protected boolean canFlowInto(Block block) {
483        if (usesWaterLogging()) {
484            if (block.canWaterloggingFlowInto()) {
485                Block blockLayer1 = block.getLevelBlockAtLayer(1);
486                return !(block instanceof BlockLiquid && block.getDamage() == 0) && !(blockLayer1 instanceof BlockLiquid && blockLayer1.getDamage() == 0);
487            }
488        }
489        return block.canBeFlowedInto() && !(block instanceof BlockLiquid && block.getDamage() == 0);
490    }
