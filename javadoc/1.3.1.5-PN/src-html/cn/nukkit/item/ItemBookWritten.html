001package cn.nukkit.item;
002
003import cn.nukkit.nbt.tag.CompoundTag;
004import cn.nukkit.nbt.tag.ListTag;
005
006public class ItemBookWritten extends ItemBookWritable {
007
008    public static final int GENERATION_ORIGINAL = 0;
009    public static final int GENERATION_COPY = 1;
010    public static final int GENERATION_COPY_OF_COPY = 2;
011    public static final int GENERATION_TATTERED = 3;
012
013    public ItemBookWritten() {
014        this(0, 1);
015    }
016
017    public ItemBookWritten(Integer meta) {
018        this(meta, 1);
019    }
020
021    public ItemBookWritten(Integer meta, int count) {
022        super(Item.WRITTEN_BOOK, 0, count, "Written Book");
023    }
024
025    @Override
026    public int getMaxStackSize() {
027        return 16;
028    }
029
030    public Item writeBook(String author, String title, String[] pages) {
031        ListTag<CompoundTag> pageList = new ListTag<>("pages");
032        for (String page : pages) {
033            pageList.add(createPageTag(page));
034        }
035        return writeBook(author, title, pageList);
036    }
037
038    public Item writeBook(String author, String title, ListTag<CompoundTag> pages) {
039        if (pages.size() > 50 || pages.size() <= 0) return this; //Minecraft does not support more than 50 pages
040        CompoundTag tag = this.hasCompoundTag() ? this.getNamedTag() : new CompoundTag();
041
042        tag.putString("author", author);
043        tag.putString("title", title);
044        tag.putList(pages);
045
046        tag.putInt("generation", GENERATION_ORIGINAL);
047        tag.putString("xuid", "");
048
049        return this.setNamedTag(tag);
050    }
051
052    public boolean signBook(String title, String author, String xuid, int generation) {
053        this.setNamedTag((this.hasCompoundTag() ? this.getNamedTag() : new CompoundTag())
054                .putString("title", title)
055                .putString("author", author)
056                .putInt("generation", generation)
057                .putString("xuid", xuid));
058        return true;
059    }
060
061    /**
062     * Returns the generation of the book.
063     * Generations higher than 1 can not be copied.
064     */
065    public int getGeneration() {
066        return this.hasCompoundTag() ? this.getNamedTag().getInt("generation") : -1;
067    }
068
069    /**
070     * Sets the generation of a book.
071     */
072    public void setGeneration(int generation) {
073        this.setNamedTag((this.hasCompoundTag() ? this.getNamedTag() : new CompoundTag()).putInt("generation", generation));
074    }
075
076    /**
077     * Returns the author of this book.
078     * This is not a reliable way to get the name of the player who signed this book.
079     * The author can be set to anything when signing a book.
080     */
081    public String getAuthor() {
082        return this.hasCompoundTag() ? this.getNamedTag().getString("author") : "";
083    }
084
085    /**
086     * Sets the author of this book.
087     */
088    public void setAuthor(String author) {
089        this.setNamedTag((this.hasCompoundTag() ? this.getNamedTag() : new CompoundTag()).putString("author", author));
090    }
091
092    /**
093     * Returns the title of this book.
094     */
095    public String getTitle() {
096        return this.hasCompoundTag() ? this.getNamedTag().getString("title") : "Written Book";
097    }
098
099    /**
100     * Sets the title of this book.
101     */
102    public void setTitle(String title) {
103        this.setNamedTag((this.hasCompoundTag() ? this.getNamedTag() : new CompoundTag()).putString("title", title));
104    }
105
106    /**
107     * Returns the author's XUID of this book.
108     */
109    public String getXUID() {
110        return this.hasCompoundTag() ? this.getNamedTag().getString("xuid") : "";
111    }
112
113    /**
114     * Sets the author's XUID of this book.
115     */
116    public void setXUID(String title) {
117        this.setNamedTag((this.hasCompoundTag() ? this.getNamedTag() : new CompoundTag()).putString("xuid", title));
118    }
119}
