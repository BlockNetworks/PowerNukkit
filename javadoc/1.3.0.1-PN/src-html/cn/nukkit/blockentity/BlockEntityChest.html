001package cn.nukkit.blockentity;
002
003import cn.nukkit.Player;
004import cn.nukkit.block.Block;
005import cn.nukkit.inventory.BaseInventory;
006import cn.nukkit.inventory.ChestInventory;
007import cn.nukkit.inventory.DoubleChestInventory;
008import cn.nukkit.level.format.FullChunk;
009import cn.nukkit.math.Vector3;
010import cn.nukkit.nbt.tag.CompoundTag;
011
012import java.util.HashSet;
013
014/**
015 * author: MagicDroidX
016 * Nukkit Project
017 */
018public class BlockEntityChest extends BlockEntitySpawnableContainer implements BlockEntityNameable {
019
020    protected DoubleChestInventory doubleInventory = null;
021
022    public BlockEntityChest(FullChunk chunk, CompoundTag nbt) {
023        super(chunk, nbt);
024    }
025
026    @Override
027    protected void initBlockEntity() {
028        this.inventory = new ChestInventory(this);
029        super.initBlockEntity();
030    }
031
032    @Override
033    public void close() {
034        if (!closed) {
035            unpair();
036
037            for (Player player : new HashSet<>(this.getInventory().getViewers())) {
038                player.removeWindow(this.getInventory());
039            }
040
041            for (Player player : new HashSet<>(this.getInventory().getViewers())) {
042                player.removeWindow(this.getRealInventory());
043            }
044            super.close();
045        }
046    }
047
048    @Override
049    public boolean isBlockEntityValid() {
050        int blockID = this.getBlock().getId();
051        return blockID == Block.CHEST || blockID == Block.TRAPPED_CHEST;
052    }
053
054    @Override
055    public int getSize() {
056        return 27;
057    }
058
059    @Override
060    public BaseInventory getInventory() {
061        if (this.doubleInventory == null && this.isPaired()) {
062            this.checkPairing();
063        }
064
065        return this.doubleInventory != null ? this.doubleInventory : this.inventory;
066    }
067
068    public ChestInventory getRealInventory() {
069        return (ChestInventory) inventory;
070    }
071
072    protected void checkPairing() {
073        BlockEntityChest pair = this.getPair();
074
075        if (pair != null) {
076            if (!pair.isPaired()) {
077                pair.createPair(this);
078                pair.checkPairing();
079            }
080
081            if (pair.doubleInventory != null) {
082                this.doubleInventory = pair.doubleInventory;
083            } else if (this.doubleInventory == null) {
084                if ((pair.x + ((int) pair.z << 15)) > (this.x + ((int) this.z << 15))) { //Order them correctly
085                    this.doubleInventory = new DoubleChestInventory(pair, this);
086                } else {
087                    this.doubleInventory = new DoubleChestInventory(this, pair);
088                }
089            }
090        } else {
091            if (level.isChunkLoaded(this.namedTag.getInt("pairx") >> 4, this.namedTag.getInt("pairz") >> 4)) {
092                this.doubleInventory = null;
093                this.namedTag.remove("pairx");
094                this.namedTag.remove("pairz");
095            }
096        }
097    }
098
099    public boolean isPaired() {
100        return this.namedTag.contains("pairx") && this.namedTag.contains("pairz");
101    }
102
103    public BlockEntityChest getPair() {
104        if (this.isPaired()) {
105            BlockEntity blockEntity = this.getLevel().getBlockEntityIfLoaded(new Vector3(this.namedTag.getInt("pairx"), this.y, this.namedTag.getInt("pairz")));
106            if (blockEntity instanceof BlockEntityChest) {
107                return (BlockEntityChest) blockEntity;
108            }
109        }
110
111        return null;
112    }
113
114    public boolean pairWith(BlockEntityChest chest) {
115        if (this.isPaired() || chest.isPaired() || this.getBlock().getId() != chest.getBlock().getId()) {
116            return false;
117        }
118
119        this.createPair(chest);
120
121        chest.spawnToAll();
122        this.spawnToAll();
123        this.checkPairing();
124
125        return true;
126    }
127
128    public void createPair(BlockEntityChest chest) {
129        this.namedTag.putInt("pairx", (int) chest.x);
130        this.namedTag.putInt("pairz", (int) chest.z);
131        chest.namedTag.putInt("pairx", (int) this.x);
132        chest.namedTag.putInt("pairz", (int) this.z);
133    }
134
135    public boolean unpair() {
136        if (!this.isPaired()) {
137            return false;
138        }
139
140        BlockEntityChest chest = this.getPair();
141
142        this.doubleInventory = null;
143        this.namedTag.remove("pairx");
144        this.namedTag.remove("pairz");
145
146        this.spawnToAll();
147
148        if (chest != null) {
149            chest.namedTag.remove("pairx");
150            chest.namedTag.remove("pairz");
151            chest.doubleInventory = null;
152            chest.checkPairing();
153            chest.spawnToAll();
154        }
155        this.checkPairing();
156
157        return true;
158    }
159
160    @Override
161    public CompoundTag getSpawnCompound() {
162        CompoundTag c;
163        if (this.isPaired()) {
164            c = new CompoundTag()
165                    .putString("id", BlockEntity.CHEST)
166                    .putInt("x", (int) this.x)
167                    .putInt("y", (int) this.y)
168                    .putInt("z", (int) this.z)
169                    .putInt("pairx", this.namedTag.getInt("pairx"))
170                    .putInt("pairz", this.namedTag.getInt("pairz"));
171        } else {
172            c = new CompoundTag()
173                    .putString("id", BlockEntity.CHEST)
174                    .putInt("x", (int) this.x)
175                    .putInt("y", (int) this.y)
176                    .putInt("z", (int) this.z);
177        }
178
179        if (this.hasName()) {
180            c.put("CustomName", this.namedTag.get("CustomName"));
181        }
182
183        return c;
184    }
185
186    @Override
187    public String getName() {
188        return this.hasName() ? this.namedTag.getString("CustomName") : "Chest";
189    }
190
191    @Override
192    public boolean hasName() {
193        return this.namedTag.contains("CustomName");
194    }
195
196    @Override
197    public void setName(String name) {
198        if (name == null || name.equals("")) {
199            this.namedTag.remove("CustomName");
200            return;
201        }
202
203        this.namedTag.putString("CustomName", name);
204    }
205}
