001package cn.nukkit.dispenser;
002
003import cn.nukkit.block.Block;
004import cn.nukkit.block.BlockDispenser;
005import cn.nukkit.block.BlockID;
006import cn.nukkit.block.BlockShulkerBox;
007import cn.nukkit.blockentity.BlockEntity;
008import cn.nukkit.item.Item;
009import cn.nukkit.math.BlockFace;
010import cn.nukkit.nbt.tag.CompoundTag;
011
012public class ShulkerBoxDispenseBehavior extends DefaultDispenseBehavior {
013
014    @Override
015    public Item dispense(BlockDispenser block, BlockFace face, Item item) {
016        Block shulkerBox = new BlockShulkerBox();
017        Block target = block.getSide(face);
018
019        this.success = block.level.getCollidingEntities(shulkerBox.getBoundingBox()).length == 0;
020
021        if (this.success) {
022            BlockFace shulkerBoxFace = target.down().getId() == BlockID.AIR ? face : BlockFace.UP;
023
024            CompoundTag nbt = BlockEntity.getDefaultCompound(target, BlockEntity.SHULKER_BOX);
025            nbt.putByte("facing", shulkerBoxFace.getIndex());
026
027            if (item.hasCustomName()) {
028                nbt.putString("CustomName", item.getCustomName());
029            }
030
031            CompoundTag tag = item.getNamedTag();
032
033            if (tag != null) {
034                if (tag.contains("Items")) {
035                    nbt.putList(tag.getList("Items"));
036                }
037            }
038
039            BlockEntity.createBlockEntity(BlockEntity.SHULKER_BOX, target, nbt);
040            block.level.updateComparatorOutputLevel(target);
041        }
042
043        return null;
044    }
045}
