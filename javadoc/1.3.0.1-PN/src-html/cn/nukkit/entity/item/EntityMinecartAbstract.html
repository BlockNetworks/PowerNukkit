127    @Override
128    public boolean onUpdate(int currentTick) {
129        if (this.closed) {
130            return false;
131        }
132
133        if (!this.isAlive()) {
134            ++this.deadTicks;
135            if (this.deadTicks >= 10) {
136                this.despawnFromAll();
137                this.close();
138            }
139            return this.deadTicks < 10;
140        }
141
142        int tickDiff = currentTick - this.lastUpdate;
143
144        if (tickDiff <= 0) {
145            return false;
146        }
147
148        this.lastUpdate = currentTick;
149
150        if (isAlive()) {
151            super.onUpdate(currentTick);
152
153            // The damage token
154            if (getHealth() < 20) {
155                setHealth(getHealth() + 1);
156            }
157
158            // Entity variables
159            lastX = x;
160            lastY = y;
161            lastZ = z;
162            motionY -= 0.03999999910593033D;
163            int dx = MathHelper.floor(x);
164            int dy = MathHelper.floor(y);
165            int dz = MathHelper.floor(z);
166
167            // Some hack to check rails
168            if (Rail.isRailBlock(level.getBlockIdAt(dx, dy - 1, dz))) {
169                --dy;
170            }
171
172            Block block = level.getBlock(new Vector3(dx, dy, dz));
173
174            // Ensure that the block is a rail
175            if (Rail.isRailBlock(block)) {
176                processMovement(dx, dy, dz, (BlockRail) block);
177                // Activate the minecart/TNT
178                if (block instanceof BlockRailActivator && ((BlockRailActivator) block).isActive()) {
179                    activate(dx, dy, dz, (block.getDamage() & 0x8) != 0);
180                }
181            } else {
182                setFalling();
183            }
184            checkBlockCollision();
185
186            // Minecart head
187            pitch = 0;
188            double diffX = this.lastX - this.x;
189            double diffZ = this.lastZ - this.z;
190            double yawToChange = yaw;
191            if (diffX * diffX + diffZ * diffZ > 0.001D) {
192                yawToChange = (Math.atan2(diffZ, diffX) * 180 / Math.PI);
193            }
194
195            // Reverse yaw if yaw is below 0
196            if (yawToChange < 0) {
197                // -90-(-90)-(-90) = 90
198                yawToChange -= yawToChange - yawToChange;
199            }
200
201            setRotation(yawToChange, pitch);
202
203            Location from = new Location(lastX, lastY, lastZ, lastYaw, lastPitch, level);
204            Location to = new Location(this.x, this.y, this.z, this.yaw, this.pitch, level);
205
206            this.getServer().getPluginManager().callEvent(new VehicleUpdateEvent(this));
207
208            if (!from.equals(to)) {
209                this.getServer().getPluginManager().callEvent(new VehicleMoveEvent(this, from, to));
210            }
211
212            // Collisions
213            for (cn.nukkit.entity.Entity entity : level.getNearbyEntities(boundingBox.grow(0.2D, 0, 0.2D), this)) {
214                if (!passengers.contains(entity) && entity instanceof EntityMinecartAbstract) {
215                    entity.applyEntityCollision(this);
216                }
217            }
218
219            Iterator<cn.nukkit.entity.Entity> linkedIterator = this.passengers.iterator();
220
221            while (linkedIterator.hasNext()) {
222                cn.nukkit.entity.Entity linked = linkedIterator.next();
223
224                if (!linked.isAlive()) {
225                    if (linked.riding == this) {
226                        linked.riding = null;
227                    }
228
229                    linkedIterator.remove();
230                }
231            }
232
233            // No need to onGround or Motion diff! This always have an update
234            return true;
235        }
236
237        return false;
238    }

298    public void applyEntityCollision(cn.nukkit.entity.Entity entity) {
299        if (entity != riding && !(entity instanceof Player && ((Player) entity).getGamemode() == Player.SPECTATOR)) {
300            if (entity instanceof EntityLiving
301                    && !(entity instanceof EntityHuman)
302                    && motionX * motionX + motionZ * motionZ > 0.01D
303                    && passengers.isEmpty()
304                    && entity.riding == null
305                    && blockInside == null) {
306                if (riding == null && devs) {
307                    mountEntity(entity);// TODO: rewrite (weird riding)
308                }
309            }
310
311            double motiveX = entity.x - x;
312            double motiveZ = entity.z - z;
313            double square = motiveX * motiveX + motiveZ * motiveZ;
314
315            if (square >= 9.999999747378752E-5D) {
316                square = Math.sqrt(square);
317                motiveX /= square;
318                motiveZ /= square;
319                double next = 1 / square;
320
321                if (next > 1) {
322                    next = 1;
323                }
324
325                motiveX *= next;
326                motiveZ *= next;
327                motiveX *= 0.10000000149011612D;
328                motiveZ *= 0.10000000149011612D;
329                motiveX *= 1 + entityCollisionReduction;
330                motiveZ *= 1 + entityCollisionReduction;
331                motiveX *= 0.5D;
332                motiveZ *= 0.5D;
333                if (entity instanceof EntityMinecartAbstract) {
334                    EntityMinecartAbstract mine = (EntityMinecartAbstract) entity;
335                    double desinityX = mine.x - x;
336                    double desinityZ = mine.z - z;
337                    Vector3 vector = new Vector3(desinityX, 0, desinityZ).normalize();
338                    Vector3 vec = new Vector3((double) MathHelper.cos((float) yaw * 0.017453292F), 0, (double) MathHelper.sin((float) yaw * 0.017453292F)).normalize();
339                    double desinityXZ = Math.abs(vector.dot(vec));
340
341                    if (desinityXZ < 0.800000011920929D) {
342                        return;
343                    }
344
345                    double motX = mine.motionX + motionX;
346                    double motZ = mine.motionZ + motionZ;
347
348                    if (mine.getType().getId() == 2 && getType().getId() != 2) {
349                        motionX *= 0.20000000298023224D;
350                        motionZ *= 0.20000000298023224D;
351                        motionX += mine.motionX - motiveX;
352                        motionZ += mine.motionZ - motiveZ;
353                        mine.motionX *= 0.949999988079071D;
354                        mine.motionZ *= 0.949999988079071D;
355                    } else if (mine.getType().getId() != 2 && getType().getId() == 2) {
356                        mine.motionX *= 0.20000000298023224D;
357                        mine.motionZ *= 0.20000000298023224D;
358                        motionX += mine.motionX + motiveX;
359                        motionZ += mine.motionZ + motiveZ;
360                        motionX *= 0.949999988079071D;
361                        motionZ *= 0.949999988079071D;
362                    } else {
363                        motX /= 2;
364                        motZ /= 2;
365                        motionX *= 0.20000000298023224D;
366                        motionZ *= 0.20000000298023224D;
367                        motionX += motX - motiveX;
368                        motionZ += motZ - motiveZ;
369                        mine.motionX *= 0.20000000298023224D;
370                        mine.motionZ *= 0.20000000298023224D;
371                        mine.motionX += motX + motiveX;
372                        mine.motionZ += motZ + motiveZ;
373                    }
374                } else {
375                    motionX -= motiveX;
376                    motionZ -= motiveZ;
377                }
378            }
379        }

398    private void setFalling() {
399        motionX = NukkitMath.clamp(motionX, -getMaxSpeed(), getMaxSpeed());
400        motionZ = NukkitMath.clamp(motionZ, -getMaxSpeed(), getMaxSpeed());
401
402        if (!hasUpdated) {
403            for (cn.nukkit.entity.Entity linked : passengers) {
404                linked.setSeatPosition(getMountedOffset(linked).add(0, 0.35f));
405                updatePassengerPosition(linked);
406            }
407
408            hasUpdated = true;
409        }
410
411        if (onGround) {
412            motionX *= derailedX;
413            motionY *= derailedY;
414            motionZ *= derailedZ;
415        }
416
417        move(motionX, motionY, motionZ);
418        if (!onGround) {
419            motionX *= flyingX;
420            motionY *= flyingY;
421            motionZ *= flyingZ;
422        }
423    }

425    private void processMovement(int dx, int dy, int dz, BlockRail block) {
426        fallDistance = 0.0F;
427        Vector3 vector = getNextRail(x, y, z);
428
429        y = (double) dy;
430        boolean isPowered = false;
431        boolean isSlowed = false;
432
433        if (block instanceof BlockRailPowered) {
434            isPowered = block.isActive();
435            isSlowed = !block.isActive();
436        }
437
438        switch (Orientation.byMetadata(block.getRealMeta())) {
439            case ASCENDING_NORTH:
440                motionX -= 0.0078125D;
441                y += 1;
442                break;
443            case ASCENDING_SOUTH:
444                motionX += 0.0078125D;
445                y += 1;
446                break;
447            case ASCENDING_EAST:
448                motionZ += 0.0078125D;
449                y += 1;
450                break;
451            case ASCENDING_WEST:
452                motionZ -= 0.0078125D;
453                y += 1;
454                break;
455        }
456
457        int[][] facing = matrix[block.getRealMeta()];
458        double facing1 = (double) (facing[1][0] - facing[0][0]);
459        double facing2 = (double) (facing[1][2] - facing[0][2]);
460        double speedOnTurns = Math.sqrt(facing1 * facing1 + facing2 * facing2);
461        double realFacing = motionX * facing1 + motionZ * facing2;
462
463        if (realFacing < 0) {
464            facing1 = -facing1;
465            facing2 = -facing2;
466        }
467
468        double squareOfFame = Math.sqrt(motionX * motionX + motionZ * motionZ);
469
470        if (squareOfFame > 2) {
471            squareOfFame = 2;
472        }
473
474        motionX = squareOfFame * facing1 / speedOnTurns;
475        motionZ = squareOfFame * facing2 / speedOnTurns;
476        double expectedSpeed;
477        double playerYawNeg; // PlayerYawNegative
478        double playerYawPos; // PlayerYawPositive
479        double motion;
480
481        cn.nukkit.entity.Entity linked = getPassenger();
482
483        if (linked instanceof EntityLiving) {
484            expectedSpeed = currentSpeed;
485            if (expectedSpeed > 0) {
486                // This is a trajectory (Angle of elevation)
487                playerYawNeg = -Math.sin(linked.yaw * Math.PI / 180.0F);
488                playerYawPos = Math.cos(linked.yaw * Math.PI / 180.0F);
489                motion = motionX * motionX + motionZ * motionZ;
490                if (motion < 0.01D) {
491                    motionX += playerYawNeg * 0.1D;
492                    motionZ += playerYawPos * 0.1D;
493
494                    isSlowed = false;
495                }
496            }
497        }
498
499        //http://minecraft.gamepedia.com/Powered_Rail#Rail
500        if (isSlowed) {
501            expectedSpeed = Math.sqrt(motionX * motionX + motionZ * motionZ);
502            if (expectedSpeed < 0.03D) {
503                motionX *= 0;
504                motionY *= 0;
505                motionZ *= 0;
506            } else {
507                motionX *= 0.5D;
508                motionY *= 0;
509                motionZ *= 0.5D;
510            }
511        }
512
513        playerYawNeg = (double) dx + 0.5D + (double) facing[0][0] * 0.5D;
514        playerYawPos = (double) dz + 0.5D + (double) facing[0][2] * 0.5D;
515        motion = (double) dx + 0.5D + (double) facing[1][0] * 0.5D;
516        double wallOfFame = (double) dz + 0.5D + (double) facing[1][2] * 0.5D;
517
518        facing1 = motion - playerYawNeg;
519        facing2 = wallOfFame - playerYawPos;
520        double motX;
521        double motZ;
522
523        if (facing1 == 0) {
524            x = (double) dx + 0.5D;
525            expectedSpeed = z - (double) dz;
526        } else if (facing2 == 0) {
527            z = (double) dz + 0.5D;
528            expectedSpeed = x - (double) dx;
529        } else {
530            motX = x - playerYawNeg;
531            motZ = z - playerYawPos;
532            expectedSpeed = (motX * facing1 + motZ * facing2) * 2;
533        }
534
535        x = playerYawNeg + facing1 * expectedSpeed;
536        z = playerYawPos + facing2 * expectedSpeed;
537        setPosition(new Vector3(x, y, z)); // Hehe, my minstake :3
538
539        motX = motionX;
540        motZ = motionZ;
541        if (!passengers.isEmpty()) {
542            motX *= 0.75D;
543            motZ *= 0.75D;
544        }
545        motX = NukkitMath.clamp(motX, -getMaxSpeed(), getMaxSpeed());
546        motZ = NukkitMath.clamp(motZ, -getMaxSpeed(), getMaxSpeed());
547
548        move(motX, 0, motZ);
549        if (facing[0][1] != 0 && MathHelper.floor(x) - dx == facing[0][0] && MathHelper.floor(z) - dz == facing[0][2]) {
550            setPosition(new Vector3(x, y + (double) facing[0][1], z));
551        } else if (facing[1][1] != 0 && MathHelper.floor(x) - dx == facing[1][0] && MathHelper.floor(z) - dz == facing[1][2]) {
552            setPosition(new Vector3(x, y + (double) facing[1][1], z));
553        }
554
555        applyDrag();
556        Vector3 vector1 = getNextRail(x, y, z);
557
558        if (vector1 != null && vector != null) {
559            double d14 = (vector.y - vector1.y) * 0.05D;
560
561            squareOfFame = Math.sqrt(motionX * motionX + motionZ * motionZ);
562            if (squareOfFame > 0) {
563                motionX = motionX / squareOfFame * (squareOfFame + d14);
564                motionZ = motionZ / squareOfFame * (squareOfFame + d14);
565            }
566
567            setPosition(new Vector3(x, vector1.y, z));
568        }
569
570        int floorX = MathHelper.floor(x);
571        int floorZ = MathHelper.floor(z);
572
573        if (floorX != dx || floorZ != dz) {
574            squareOfFame = Math.sqrt(motionX * motionX + motionZ * motionZ);
575            motionX = squareOfFame * (double) (floorX - dx);
576            motionZ = squareOfFame * (double) (floorZ - dz);
577        }
578
579        if (isPowered) {
580            double newMovie = Math.sqrt(motionX * motionX + motionZ * motionZ);
581
582            if (newMovie > 0.01D) {
583                double nextMovie = 0.06D;
584
585                motionX += motionX / newMovie * nextMovie;
586                motionZ += motionZ / newMovie * nextMovie;
587            } else if (block.getOrientation() == Orientation.STRAIGHT_NORTH_SOUTH) {
588                if (level.getBlock(new Vector3(dx - 1, dy, dz)).isNormalBlock()) {
589                    motionX = 0.02D;
590                } else if (level.getBlock(new Vector3(dx + 1, dy, dz)).isNormalBlock()) {
591                    motionX = -0.02D;
592                }
593            } else if (block.getOrientation() == Orientation.STRAIGHT_EAST_WEST) {
594                if (level.getBlock(new Vector3(dx, dy, dz - 1)).isNormalBlock()) {
595                    motionZ = 0.02D;
596                } else if (level.getBlock(new Vector3(dx, dy, dz + 1)).isNormalBlock()) {
597                    motionZ = -0.02D;
598                }
599            }
600        }
601
602    }

604    private void applyDrag() {
605        if (!passengers.isEmpty() || !slowWhenEmpty) {
606            motionX *= 0.996999979019165D;
607            motionY *= 0.0D;
608            motionZ *= 0.996999979019165D;
609        } else {
610            motionX *= 0.9599999785423279D;
611            motionY *= 0.0D;
612            motionZ *= 0.9599999785423279D;
613        }
614    }

616    private Vector3 getNextRail(double dx, double dy, double dz) {
617        int checkX = MathHelper.floor(dx);
618        int checkY = MathHelper.floor(dy);
619        int checkZ = MathHelper.floor(dz);
620
621        if (Rail.isRailBlock(level.getBlockIdAt(checkX, checkY - 1, checkZ))) {
622            --checkY;
623        }
624
625        Block block = level.getBlock(new Vector3(checkX, checkY, checkZ));
626
627        if (Rail.isRailBlock(block)) {
628            int[][] facing = matrix[((BlockRail) block).getRealMeta()];
629            double rail;
630            // Genisys mistake (Doesn't check surrounding more exactly)
631            double nextOne = (double) checkX + 0.5D + (double) facing[0][0] * 0.5D;
632            double nextTwo = (double) checkY + 0.5D + (double) facing[0][1] * 0.5D;
633            double nextThree = (double) checkZ + 0.5D + (double) facing[0][2] * 0.5D;
634            double nextFour = (double) checkX + 0.5D + (double) facing[1][0] * 0.5D;
635            double nextFive = (double) checkY + 0.5D + (double) facing[1][1] * 0.5D;
636            double nextSix = (double) checkZ + 0.5D + (double) facing[1][2] * 0.5D;
637            double nextSeven = nextFour - nextOne;
638            double nextEight = (nextFive - nextTwo) * 2;
639            double nextMax = nextSix - nextThree;
640
641            if (nextSeven == 0) {
642                rail = dz - (double) checkZ;
643            } else if (nextMax == 0) {
644                rail = dx - (double) checkX;
645            } else {
646                double whatOne = dx - nextOne;
647                double whatTwo = dz - nextThree;
648
649                rail = (whatOne * nextSeven + whatTwo * nextMax) * 2;
650            }
651
652            dx = nextOne + nextSeven * rail;
653            dy = nextTwo + nextEight * rail;
654            dz = nextThree + nextMax * rail;
655            if (nextEight < 0) {
656                ++dy;
657            }
658
659            if (nextEight > 0) {
660                dy += 0.5D;
661            }
662
663            return new Vector3(dx, dy, dz);
664        } else {
665            return null;
666        }
667    }

678    private void prepareDataProperty() {
679        setRollingAmplitude(0);
680        setRollingDirection(1);
681        if (namedTag.contains("CustomDisplayTile")) {
682            if (namedTag.getBoolean("CustomDisplayTile")) {
683                int display = namedTag.getInt("DisplayTile");
684                int offSet = namedTag.getInt("DisplayOffset");
685                setDataProperty(new ByteEntityData(DATA_HAS_DISPLAY, 1));
686                setDataProperty(new IntEntityData(DATA_DISPLAY_ITEM, display));
687                setDataProperty(new IntEntityData(DATA_DISPLAY_OFFSET, offSet));
688            }
689        } else {
690            int display = blockInside == null ? 0
691                    : blockInside.getId()
692                    | blockInside.getDamage() << 16;
693            if (display == 0) {
694                setDataProperty(new ByteEntityData(DATA_HAS_DISPLAY, 0));
695                return;
696            }
697            setDataProperty(new ByteEntityData(DATA_HAS_DISPLAY, 1));
698            setDataProperty(new IntEntityData(DATA_DISPLAY_ITEM, display));
699            setDataProperty(new IntEntityData(DATA_DISPLAY_OFFSET, 6));
700        }
701    }
702
703    private void saveEntityData() {
704        boolean hasDisplay = super.getDataPropertyByte(DATA_HAS_DISPLAY) == 1
705                || blockInside != null;
706        int display;
707        int offSet;
708        namedTag.putBoolean("CustomDisplayTile", hasDisplay);
709        if (hasDisplay) {
710            display = blockInside.getId()
711                    | blockInside.getDamage() << 16;
712            offSet = getDataPropertyInt(DATA_DISPLAY_OFFSET);
713            namedTag.putInt("DisplayTile", display);
714            namedTag.putInt("DisplayOffset", offSet);
715        }
716    }

728    /**
729     * Set the minecart display block
730     *
731     * @param block The block that will changed. Set {@code null} for BlockAir
732     * @param update Do update for the block. (This state changes if you want to show the block)
733     * @return {@code true} if the block is normal block
734     */
735    @API(usage = Usage.MAINTAINED, definition = Definition.UNIVERSAL)
736    public boolean setDisplayBlock(Block block, boolean update) {
737        if(!update){
738            if (block.isNormalBlock()) {
739                blockInside = block;
740            } else {
741                blockInside = null;
742            }
743            return true;
744        }
745        if (block != null) {
746            if (block.isNormalBlock()) {
747                blockInside = block;
748                int display = blockInside.getId()
749                        | blockInside.getDamage() << 16;
750                setDataProperty(new ByteEntityData(DATA_HAS_DISPLAY, 1));
751                setDataProperty(new IntEntityData(DATA_DISPLAY_ITEM, display));
752                setDisplayBlockOffset(6);
753            }
754        } else {
755            // Set block to air (default).
756            blockInside = null;
757            setDataProperty(new ByteEntityData(DATA_HAS_DISPLAY, 0));
758            setDataProperty(new IntEntityData(DATA_DISPLAY_ITEM, 0));
759            setDisplayBlockOffset(0);
760        }
761        return true;
762    }
