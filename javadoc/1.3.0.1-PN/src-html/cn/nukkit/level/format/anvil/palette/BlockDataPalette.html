001package cn.nukkit.level.format.anvil.palette;
002
003import cn.nukkit.Server;
004import cn.nukkit.block.Block;
005import cn.nukkit.math.MathHelper;
006import cn.nukkit.utils.ThreadCache;
007import com.google.common.base.Preconditions;
008
009import java.util.Arrays;
010
011/**
012 * @author https://github.com/boy0001/
013 */
014public final class BlockDataPalette implements Cloneable {
015    private static final int BLOCK_SIZE = 4096;
016    private volatile char[] rawData;
017
018    private volatile BitArray4096 encodedData;
019    private volatile CharPalette palette;
020
021    // TODO compress unused sections
022    // private byte[] compressedData;
023
024    public BlockDataPalette() {
025        this(new char[BLOCK_SIZE]);
026    }
027
028    public BlockDataPalette(char[] rawData) {
029        Preconditions.checkArgument(rawData.length == BLOCK_SIZE, "Data is not 4096");
030        this.rawData = rawData;
031    }
032
033    private char[] getCachedRaw() {
034        char[] raw = rawData;
035        if (raw != null) {
036            return raw;
037        } else if (!Server.getInstance().isPrimaryThread()) {
038            return getRaw();
039        }
040        return rawData;
041    }
042
043    public synchronized char[] getRaw() {
044        CharPalette palette = this.palette;
045        BitArray4096 encodedData = this.encodedData;
046        this.encodedData = null;
047        this.palette = null;
048
049        char[] raw = rawData;
050        if (raw == null && palette != null) {
051            if (encodedData != null) {
052                raw = encodedData.toRaw();
053            } else {
054                raw = new char[BLOCK_SIZE];
055            }
056            for (int i = 0; i < BLOCK_SIZE; i++) {
057                raw[i] = palette.getKey(raw[i]);
058            }
059        } else {
060            raw = new char[BLOCK_SIZE];
061        }
062        rawData = raw;
063        return rawData;
064    }
065
066    private int getIndex(int x, int y, int z) {
067        return (x << 8) + (z << 4) + y; // XZY = Bedrock format
068    }
069
070    public int getBlockData(int x, int y, int z) {
071        return getFullBlock(x, y, z) & 0xF;
072    }
073
074    public int getBlockId(int x, int y, int z) {
075        return getFullBlock(x, y, z) >> Block.DATA_BITS;
076    }
077
078    public void setBlockId(int x, int y, int z, int id) {
079        setFullBlock(x, y, z, (char) (id << Block.DATA_BITS));
080    }
081
082    public synchronized void setBlockData(int x, int y, int z, int data) {
083        int index = getIndex(x, y, z);
084        char[] raw = getCachedRaw();
085
086        if (raw != null) {
087            int fullId = raw[index];
088            raw[index] = (char) ((fullId & (0xFFF << Block.DATA_BITS)) | data);
089        } if (palette != null && encodedData != null) {
090            char fullId = palette.getKey(encodedData.getAt(index));
091            if ((fullId & Block.DATA_MASK) != data) {
092                setPaletteFullBlock(index, (char) ((fullId & (0xFFF << Block.DATA_BITS)) | data));
093            }
094        } else {
095            throw new IllegalStateException("Raw data and pallete was null");
096        }
097    }
098
099    public int getFullBlock(int x, int y, int z) {
100        return getFullBlock(getIndex(x, y, z));
101    }
102
103    public void setFullBlock(int x, int y, int z, int value) {
104        this.setFullBlock(getIndex(x, y, z), (char) value);
105    }
106
107    public int getAndSetFullBlock(int x, int y, int z, int value) {
108        return getAndSetFullBlock(getIndex(x, y, z), (char) value);
109    }
110
111    private int getAndSetFullBlock(int index, char value) {
112        char[] raw = getCachedRaw();
113        if (raw != null) {
114            char result = raw[index];
115            raw[index] = value;
116            return result;
117        } else if (palette != null && encodedData != null) {
118            char result = palette.getKey(encodedData.getAt(index));
119            if (result != value) {
120                setPaletteFullBlock(index, value);
121            }
122            return result;
123        } else {
124            throw new IllegalStateException("Raw data and pallete was null");
125        }
126    }
127
128    private int getFullBlock(int index) {
129        char[] raw = getCachedRaw();
130        if (raw != null) {
131            return raw[index];
132        } else if (palette != null && encodedData != null) {
133            return palette.getKey(encodedData.getAt(index));
134        } else {
135            throw new IllegalStateException("Raw data and pallete was null");
136        }
137    }
138
139    private void setFullBlock(int index, char value) {
140        char[] raw = getCachedRaw();
141        if (raw != null) {
142            raw[index] = value;
143        } else if (!setPaletteFullBlock(index, value)) {
144            throw new IllegalStateException("Raw data and pallete was null");
145        }
146    }
147
148    private synchronized boolean setPaletteFullBlock(int index, char value) {
149        CharPalette palette = this.palette;
150        BitArray4096 encodedData = this.encodedData;
151        if (palette != null && encodedData != null) {
152            char encodedValue = palette.getValue(value);
153            if (encodedValue != Character.MAX_VALUE) {
154                encodedData.setAt(index, encodedValue);
155            } else {
156                char[] raw = encodedData.toRaw();
157                for (int i = 0; i < BLOCK_SIZE; i++) {
158                    raw[i] = palette.getKey(raw[i]);
159                }
160                raw[index] = value;
161                this.rawData = raw;
162                this.encodedData = null;
163                this.palette = null;
164            }
165            return true;
166        }
167        return false;
168    }
169
170    public synchronized boolean compress() {
171        char[] raw = rawData;
172        if (raw != null) {
173            char unique = 0;
174
175            boolean[] countTable = ThreadCache.boolCache4096.get();
176            char[] mapFullTable = ThreadCache.charCache4096.get();
177            char[] mapBitTable = ThreadCache.charCache4096v2.get();
178            Arrays.fill(countTable, false);
179            for (char c : raw) {
180                if (!countTable[c]) {
181                    mapBitTable[unique] = c;
182                    countTable[c] = true;
183                    unique++;
184                }
185            }
186
187            char[] keys = Arrays.copyOfRange(mapBitTable, 0, unique);
188            if (keys.length > 1) {
189                Arrays.sort(keys);
190                for (char c = 0; c < keys.length; c++) {
191                    mapFullTable[keys[c]] = c;
192                }
193            } else {
194                mapFullTable[keys[0]] = 0;
195            }
196
197            CharPalette palette = new CharPalette();
198            palette.set(keys);
199
200            int bits = MathHelper.log2(unique - 1);
201            BitArray4096 encodedData = new BitArray4096(bits);
202
203            for (int i = 0; i < raw.length; i++) {
204                mapBitTable[i] = mapFullTable[raw[i]];
205            }
206
207            encodedData.fromRaw(mapBitTable);
208
209            this.palette = palette;
210            this.encodedData = encodedData;
211            rawData = null;
212            return true;
213        }
214        return false;
215    }
216
217    public synchronized BlockDataPalette clone() {
218        char[] raw = getRaw();
219        return new BlockDataPalette(raw.clone());
220    }
221}
