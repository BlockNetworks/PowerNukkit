001package cn.nukkit.math;
002
003import com.google.common.collect.Iterators;
004
005import java.util.Iterator;
006import java.util.Random;
007import java.util.function.Predicate;
008
009public enum BlockFace {
010    DOWN(0, 1, -1, "down", AxisDirection.NEGATIVE, new Vector3(0, -1, 0)),
011    UP(1, 0, -1, "up", AxisDirection.POSITIVE, new Vector3(0, 1, 0)),
012    NORTH(2, 3, 2, "north", AxisDirection.NEGATIVE, new Vector3(0, 0, -1)),
013    SOUTH(3, 2, 0, "south", AxisDirection.POSITIVE, new Vector3(0, 0, 1)),
014    WEST(4, 5, 1, "west", AxisDirection.NEGATIVE, new Vector3(-1, 0, 0)),
015    EAST(5, 4, 3, "east", AxisDirection.POSITIVE, new Vector3(1, 0, 0));
016
017    /**
018     * All faces in D-U-N-S-W-E order
019     */
020    private static final BlockFace[] VALUES = new BlockFace[6];
021
022    /**
023     * All faces with horizontal axis in order S-W-N-E
024     */
025    private static final BlockFace[] HORIZONTALS = new BlockFace[4];
026
027    static {
028        //Circular dependency
029        DOWN.axis = Axis.Y;
030        UP.axis = Axis.Y;
031        NORTH.axis = Axis.Z;
032        SOUTH.axis = Axis.Z;
033        WEST.axis = Axis.X;
034        EAST.axis = Axis.X;
035
036        for (BlockFace face : values()) {
037            VALUES[face.index] = face;
038
039            if (face.getAxis().isHorizontal()) {
040                HORIZONTALS[face.horizontalIndex] = face;
041            }
042        }
043    }
044
045    /**
046     * Ordering index for D-U-N-S-W-E
047     */
048    private final int index;
049
050    /**
051     * Index of the opposite BlockFace in the VALUES array
052     */
053    private final int opposite;
054
055    /**
056     * Ordering index for the HORIZONTALS field (S-W-N-E)
057     */
058    private final int horizontalIndex;
059
060    /**
061     * The name of this BlockFace (up, down, north, etc.)
062     */
063    private final String name;
064
065
066    private Axis axis;
067    private final AxisDirection axisDirection;
068
069    /**
070     * Normalized vector that points in the direction of this BlockFace
071     */
072    private final Vector3 unitVector;
073
074    BlockFace(int index, int opposite, int horizontalIndex, String name, AxisDirection axisDirection, Vector3 unitVector) {
075        this.index = index;
076        this.opposite = opposite;
077        this.horizontalIndex = horizontalIndex;
078        this.name = name;
079        this.axisDirection = axisDirection;
080        this.unitVector = unitVector;
081    }
082
083    /**
084     * Get a BlockFace by it's index (0-5). The order is D-U-N-S-W-E
085     *
086     * @param index BlockFace index
087     * @return block face
088     */
089    public static BlockFace fromIndex(int index) {
090        return VALUES[MathHelper.abs(index % VALUES.length)];
091    }
092
093    /**
094     * Get a BlockFace by it's horizontal index (0-3). The order is S-W-N-E
095     *
096     * @param index BlockFace index
097     * @return block face
098     */
099    public static BlockFace fromHorizontalIndex(int index) {
100        return HORIZONTALS[MathHelper.abs(index % HORIZONTALS.length)];
101    }
102
103    /**
104     * Get the BlockFace corresponding to the given angle (0-360). An angle of 0 is SOUTH, an angle of 90 would be WEST
105     *
106     * @param angle horizontal angle
107     * @return block face
108     */
109    public static BlockFace fromHorizontalAngle(double angle) {
110        return fromHorizontalIndex(NukkitMath.floorDouble(angle / 90.0D + 0.5D) & 3);
111    }
112
113    public static BlockFace fromAxis(AxisDirection axisDirection, Axis axis) {
114        for (BlockFace face : VALUES) {
115            if (face.getAxisDirection() == axisDirection && face.getAxis() == axis) {
116                return face;
117            }
118        }
119
120        throw new RuntimeException("Unable to get face from axis: " + axisDirection + " " + axis);
121    }
122
123    /**
124     * Choose a random BlockFace using the given Random
125     *
126     * @param rand random number generator
127     * @return block face
128     */
129    public static BlockFace random(Random rand) {
130        return VALUES[rand.nextInt(VALUES.length)];
131    }
132
133    /**
134     * Get the index of this BlockFace (0-5). The order is D-U-N-S-W-E
135     *
136     * @return index
137     */
138    public int getIndex() {
139        return index;
140    }
141
142    /**
143     * Get the horizontal index of this BlockFace (0-3). The order is S-W-N-E
144     *
145     * @return horizontal index
146     */
147    public int getHorizontalIndex() {
148        return horizontalIndex;
149    }
150
151    /**
152     * Get the angle of this BlockFace (0-360)
153     *
154     * @return horizontal angle
155     */
156    public float getHorizontalAngle() {
157        return (float) ((horizontalIndex & 3) * 90);
158    }
159
160    /**
161     * Get the name of this BlockFace (up, down, north, etc.)
162     *
163     * @return name
164     */
165    public String getName() {
166        return name;
167    }
168
169    /**
170     * Get the Axis of this BlockFace
171     *
172     * @return axis
173     */
174    public Axis getAxis() {
175        return axis;
176    }
177
178    /**
179     * Get the AxisDirection of this BlockFace
180     *
181     * @return axis direction
182     */
183    public AxisDirection getAxisDirection() {
184        return axisDirection;
185    }
186
187    /**
188     * Get the unit vector of this BlockFace
189     *
190     * @return vector
191     */
192    public Vector3 getUnitVector() {
193        return unitVector;
194    }
195
196    /**
197     * Returns an offset that addresses the block in front of this BlockFace
198     *
199     * @return x offset
200     */
201    public int getXOffset() {
202        return axis == Axis.X ? axisDirection.getOffset() : 0;
203    }
204
205    /**
206     * Returns an offset that addresses the block in front of this BlockFace
207     *
208     * @return y offset
209     */
210    public int getYOffset() {
211        return axis == Axis.Y ? axisDirection.getOffset() : 0;
212    }
213
214    /**
215     * Returns an offset that addresses the block in front of this BlockFace
216     *
217     * @return x offset
218     */
219    public int getZOffset() {
220        return axis == Axis.Z ? axisDirection.getOffset() : 0;
221    }
222
223    /**
224     * Get the opposite BlockFace (e.g. DOWN ==&gt; UP)
225     *
226     * @return block face
227     */
228    public BlockFace getOpposite() {
229        return fromIndex(opposite);
230    }
231
232    /**
233     * Rotate this BlockFace around the Y axis clockwise (NORTH =&gt; EAST =&gt; SOUTH =&gt; WEST =&gt; NORTH)
234     *
235     * @return block face
236     */
237    public BlockFace rotateY() {
238        switch (this) {
239            case NORTH:
240                return EAST;
241            case EAST:
242                return SOUTH;
243            case SOUTH:
244                return WEST;
245            case WEST:
246                return NORTH;
247            default:
248                throw new RuntimeException("Unable to get Y-rotated face of " + this);
249        }
250    }
251
252    /**
253     * Rotate this BlockFace around the Y axis counter-clockwise (NORTH =&gt; WEST =&gt; SOUTH =&gt; EAST =&gt; NORTH)
254     *
255     * @return block face
256     */
257    public BlockFace rotateYCCW() {
258        switch (this) {
259            case NORTH:
260                return WEST;
261            case EAST:
262                return NORTH;
263            case SOUTH:
264                return EAST;
265            case WEST:
266                return SOUTH;
267            default:
268                throw new RuntimeException("Unable to get counter-clockwise Y-rotated face of " + this);
269        }
270    }
271
272    public String toString() {
273        return name;
274    }

276    public enum Axis implements Predicate<BlockFace> {
277        X("x"),
278        Y("y"),
279        Z("z");
280
281        private final String name;
282        private Plane plane;
283
284        static {
285            //Circular dependency
286            X.plane = Plane.HORIZONTAL;
287            Y.plane = Plane.VERTICAL;
288            Z.plane = Plane.HORIZONTAL;
289        }
290
291        Axis(String name) {
292            this.name = name;
293        }
294
295        public boolean isVertical() {
296            return plane == Plane.VERTICAL;
297        }
298
299        public boolean isHorizontal() {
300            return plane == Plane.HORIZONTAL;
301        }
302
303        public Plane getPlane() {
304            return plane;
305        }
306
307        public String getName() {
308            return name;
309        }
310
311        public boolean test(BlockFace face) {
312            return face != null && face.getAxis() == this;
313        }
314
315        public String toString() {
316            return name;
317        }
318    }
319
320    public enum AxisDirection {
321        POSITIVE(1, "Towards positive"),
322        NEGATIVE(-1, "Towards negative");
323
324        private final int offset;
325        private final String description;
326
327        AxisDirection(int offset, String description) {
328            this.offset = offset;
329            this.description = description;
330        }
331
332        public int getOffset() {
333            return offset;
334        }
335
336        public String toString() {
337            return description;
338        }
339    }
340
341    public enum Plane implements Predicate<BlockFace>, Iterable<BlockFace> {
342        HORIZONTAL,
343        VERTICAL;
344
345        static {
346            //Circular dependency
347            HORIZONTAL.faces = new BlockFace[]{NORTH, EAST, SOUTH, WEST};
348            VERTICAL.faces = new BlockFace[]{UP, DOWN};
349        }
350
351        private BlockFace[] faces;
352
353        public BlockFace random(NukkitRandom rand) { //todo Default Random?
354            return faces[rand.nextBoundedInt(faces.length)];
355        }
356
357        public boolean test(BlockFace face) {
358            return face != null && face.getAxis().getPlane() == this;
359        }
360
361        public Iterator<BlockFace> iterator() {
362            return Iterators.forArray(faces);
363        }
364    }
