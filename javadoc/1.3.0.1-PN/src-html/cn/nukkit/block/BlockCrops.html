001package cn.nukkit.block;
002
003import cn.nukkit.Player;
004import cn.nukkit.Server;
005import cn.nukkit.event.block.BlockGrowEvent;
006import cn.nukkit.item.Item;
007import cn.nukkit.level.Level;
008import cn.nukkit.level.particle.BoneMealParticle;
009import cn.nukkit.math.BlockFace;
010import cn.nukkit.utils.BlockColor;
011
012import java.util.concurrent.ThreadLocalRandom;
013
014/**
015 * author: MagicDroidX
016 * Nukkit Project
017 */
018public abstract class BlockCrops extends BlockFlowable {
019
020    public static final int MINIMUM_LIGHT_LEVEL = 9;
021
022    protected BlockCrops(int meta) {
023        super(meta);
024    }
025
026    @Override
027    public boolean canBeActivated() {
028        return true;
029    }
030
031
032    @Override
033    public boolean place(Item item, Block block, Block target, BlockFace face, double fx, double fy, double fz, Player player) {
034        if (block.down().getId() == FARMLAND) {
035            this.getLevel().setBlock(block, this, true, true);
036            return true;
037        }
038        return false;
039    }
040
041    @Override
042    public boolean onActivate(Item item, Player player) {
043        //Bone meal
044        if (item.getId() == Item.DYE && item.getDamage() == 0x0f) {
045            if (this.getDamage() < 7) {
046                BlockCrops block = (BlockCrops) this.clone();
047                block.setDamage(block.getDamage() + ThreadLocalRandom.current().nextInt(3) + 2);
048                if (block.getDamage() > 7) {
049                    block.setDamage(7);
050                }
051                BlockGrowEvent ev = new BlockGrowEvent(this, block);
052                Server.getInstance().getPluginManager().callEvent(ev);
053
054                if (ev.isCancelled()) {
055                    return false;
056                }
057
058                this.getLevel().setBlock(this, ev.getNewState(), false, true);
059                this.level.addParticle(new BoneMealParticle(this));
060
061                if (player != null && (player.gamemode & 0x01) == 0) {
062                    item.count--;
063                }
064            }
065
066            return true;
067        }
068
069        return false;
070    }
071
072    @Override
073    public int onUpdate(int type) {
074        if (type == Level.BLOCK_UPDATE_NORMAL) {
075            if (this.down().getId() != FARMLAND) {
076                this.getLevel().useBreakOn(this);
077                return Level.BLOCK_UPDATE_NORMAL;
078            }
079        } else if (type == Level.BLOCK_UPDATE_RANDOM) {
080            if (ThreadLocalRandom.current().nextInt(2) == 1 && getLevel().getFullLight(this) >= MINIMUM_LIGHT_LEVEL) {
081                if (this.getDamage() < 0x07) {
082                    BlockCrops block = (BlockCrops) this.clone();
083                    block.setDamage(block.getDamage() + 1);
084                    BlockGrowEvent ev = new BlockGrowEvent(this, block);
085                    Server.getInstance().getPluginManager().callEvent(ev);
086
087                    if (!ev.isCancelled()) {
088                        this.getLevel().setBlock(this, ev.getNewState(), false, true);
089                    } else {
090                        return Level.BLOCK_UPDATE_RANDOM;
091                    }
092                }
093            } else {
094                return Level.BLOCK_UPDATE_RANDOM;
095            }
096        }
097
098        return 0;
099    }
100
101    @Override
102    public BlockColor getColor() {
103        return BlockColor.FOLIAGE_BLOCK_COLOR;
104    }
105}
