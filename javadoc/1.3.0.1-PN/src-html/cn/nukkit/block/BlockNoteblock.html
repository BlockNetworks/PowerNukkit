212    public void emitSound() {
213        if (this.up().getId() != AIR) return;
214
215        Instrument instrument = this.getInstrument();
216
217        this.level.addLevelSoundEvent(this, LevelSoundEventPacket.SOUND_NOTE, instrument.ordinal() << 8 | this.getStrength());
218
219        BlockEventPacket pk = new BlockEventPacket();
220        pk.x = this.getFloorX();
221        pk.y = this.getFloorY();
222        pk.z = this.getFloorZ();
223        pk.case1 = instrument.ordinal();
224        pk.case2 = this.getStrength();
225        this.getLevel().addChunkPacket(this.getFloorX() >> 4, this.getFloorZ() >> 4, pk);
226    }
227
228    @Override
229    public boolean onActivate(Item item, Player player) {
230        this.increaseStrength();
231        this.emitSound();
232        return true;
233    }
234
235    @Override
236    public int onUpdate(int type) {
237        if (type == Level.BLOCK_UPDATE_REDSTONE) {
238            BlockEntityMusic blockEntity = this.getBlockEntity();
239            if (blockEntity != null) {
240                if (this.getLevel().isBlockPowered(this)) {
241                    if (!blockEntity.isPowered()) {
242                        this.emitSound();
243                    }
244                    blockEntity.setPowered(true);
245                } else {
246                    blockEntity.setPowered(false);
247                }
248            }
249        }
250        return super.onUpdate(type);
251    }
252
253    private BlockEntityMusic getBlockEntity() {
254        BlockEntity blockEntity = this.getLevel().getBlockEntity(this);
255        if (blockEntity instanceof BlockEntityMusic) {
256            return (BlockEntityMusic) blockEntity;
257        }
258        return null;
259    }
260
261    private BlockEntityMusic createBlockEntity() {
262        return (BlockEntityMusic) BlockEntity.createBlockEntity(BlockEntity.MUSIC, this.getLevel().getChunk(this.getFloorX() >> 4, this.getFloorZ() >> 4),
263                                        BlockEntity.getDefaultCompound(this, BlockEntity.MUSIC));
264    }
