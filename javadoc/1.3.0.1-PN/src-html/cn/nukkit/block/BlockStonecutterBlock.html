001package cn.nukkit.block;
002
003import cn.nukkit.Player;
004import cn.nukkit.inventory.StonecutterInventory;
005import cn.nukkit.item.Item;
006import cn.nukkit.item.ItemBlock;
007import cn.nukkit.item.ItemTool;
008import cn.nukkit.math.BlockFace;
009import cn.nukkit.network.protocol.types.ContainerIds;
010import cn.nukkit.utils.BlockColor;
011import cn.nukkit.utils.Faceable;
012
013public class BlockStonecutterBlock extends BlockTransparentMeta implements Faceable {
014
015    public BlockStonecutterBlock() {
016        this(0);
017    }
018
019    public BlockStonecutterBlock(int meta) {
020        super(meta);
021    }
022
023    @Override
024    public int getId() {
025        return STONECUTTER_BLOCK;
026    }
027
028    @Override
029    public String getName() {
030        return "Stonecutter";
031    }
032
033    public void setBlockFace(BlockFace face) {
034        int horizontalIndex = face.getHorizontalIndex();
035        if (horizontalIndex > -1) {
036            setDamage(horizontalIndex);
037        }
038    }
039
040    @Override
041    public BlockFace getBlockFace() {
042        return BlockFace.fromHorizontalIndex(this.getDamage() & 0x7);
043    }
044
045    @Override
046    public boolean place(Item item, Block block, Block target, BlockFace face, double fx, double fy, double fz, Player player) {
047        int[] faces = {2, 5, 3, 4};
048        this.setDamage(faces[player != null ? player.getDirection().getHorizontalIndex() : 0]);
049        this.getLevel().setBlock(block, this, true, true);
050        return true;
051    }
052
053    @Override
054    public boolean canBeActivated() {
055        return true;
056    }
057
058    @Override
059    public boolean onActivate(Item item, Player player) {
060        if (player != null) {
061            player.addWindow(new StonecutterInventory(player.getUIInventory(), this), ContainerIds.NONE);
062            player.craftingType = Player.CRAFTING_STONECUTTER;
063        }
064        return true;
065    }
066
067    @Override
068    public BlockColor getColor() {
069        return BlockColor.STONE_BLOCK_COLOR;
070    }
071
072    @Override
073    public double getHardness() {
074        return 3.5;
075    }
076
077    @Override
078    public double getResistance() {
079        return 17.5;
080    }
081
082    @Override
083    public boolean canHarvestWithHand() {
084        return false;
085    }
086
087    @Override
088    public int getWaterloggingLevel() {
089        return 1;
090    }
091
092    @Override
093    public int getToolType() {
094        return ItemTool.TYPE_PICKAXE;
095    }
096
097    @Override
098    public Item[] getDrops(Item item) {
099        return new Item[] {  toItem() };
100    }
101
102    @Override
103    public Item toItem() {
104        return new ItemBlock(new BlockStonecutterBlock());
105    }
106}
