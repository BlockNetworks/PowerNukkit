001package cn.nukkit.block;
002
003import cn.nukkit.Player;
004import cn.nukkit.event.redstone.RedstoneUpdateEvent;
005import cn.nukkit.item.Item;
006import cn.nukkit.item.ItemBlock;
007import cn.nukkit.item.ItemTool;
008import cn.nukkit.level.Level;
009import cn.nukkit.math.BlockFace;
010import cn.nukkit.utils.BlockColor;
011
012/**
013 * @author Nukkit Project Team
014 */
015public class BlockRedstoneLamp extends BlockSolid {
016
017    public BlockRedstoneLamp() {
018    }
019
020    @Override
021    public String getName() {
022        return "Redstone Lamp";
023    }
024
025    @Override
026    public int getId() {
027        return REDSTONE_LAMP;
028    }
029
030    @Override
031    public double getHardness() {
032        return 0.3D;
033    }
034
035    @Override
036    public double getResistance() {
037        return 1.5D;
038    }
039
040    @Override
041    public int getToolType() {
042        return ItemTool.TYPE_PICKAXE;
043    }
044
045    @Override
046    public boolean place(Item item, Block block, Block target, BlockFace face, double fx, double fy, double fz, Player player) {
047        if (this.level.isBlockPowered(this.getLocation())) {
048            this.level.setBlock(this, Block.get(BlockID.LIT_REDSTONE_LAMP), false, true);
049        } else {
050            this.level.setBlock(this, this, false, true);
051        }
052        return true;
053    }
054
055    @Override
056    public int onUpdate(int type) {
057        if (type == Level.BLOCK_UPDATE_NORMAL || type == Level.BLOCK_UPDATE_REDSTONE) {
058            if (!this.level.getServer().isRedstoneEnabled()) {
059                return 0;
060            }
061            // Redstone event
062            RedstoneUpdateEvent ev = new RedstoneUpdateEvent(this);
063            getLevel().getServer().getPluginManager().callEvent(ev);
064            if (ev.isCancelled()) {
065                return 0;
066            }
067            if (this.level.isBlockPowered(this.getLocation())) {
068                this.level.setBlock(this, Block.get(BlockID.LIT_REDSTONE_LAMP), false, false);
069                return 1;
070            }
071        }
072
073        return 0;
074    }
075
076    @Override
077    public Item[] getDrops(Item item) {
078        return new Item[]{
079                new ItemBlock(Block.get(BlockID.REDSTONE_LAMP))
080        };
081    }
082
083    @Override
084    public BlockColor getColor() {
085        return BlockColor.AIR_BLOCK_COLOR;
086    }
087}
