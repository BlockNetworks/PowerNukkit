001package cn.nukkit.inventory;
002
003import cn.nukkit.Player;
004import cn.nukkit.block.Block;
005import cn.nukkit.block.BlockBarrel;
006import cn.nukkit.blockentity.BlockEntityBarrel;
007import cn.nukkit.level.Level;
008import cn.nukkit.level.Sound;
009
010public class BarrelInventory extends ContainerInventory {
011
012    public BarrelInventory(BlockEntityBarrel barrel) {
013        super(barrel, InventoryType.BARREL);
014    }
015
016    @Override
017    public BlockEntityBarrel getHolder() {
018        return (BlockEntityBarrel) this.holder;
019    }
020
021    @Override
022    public void onOpen(Player who) {
023        super.onOpen(who);
024
025        if (this.getViewers().size() == 1) {
026            BlockEntityBarrel barrel = this.getHolder();
027            Level level = barrel.getLevel();
028            if (level != null) {
029                Block block = barrel.getBlock();
030                if (block instanceof BlockBarrel) {
031                    BlockBarrel blockBarrel = (BlockBarrel) block;
032                    if (!blockBarrel.isOpen()) {
033                        blockBarrel.setOpen(true);
034                        level.setBlock(blockBarrel, blockBarrel, true, true);
035                        level.addSound(blockBarrel, Sound.BLOCK_BARREL_OPEN);
036                    }
037                }
038            }
039        }
040    }
041
042    @Override
043    public void onClose(Player who) {
044        super.onClose(who);
045
046        if (this.getViewers().isEmpty()) {
047            BlockEntityBarrel barrel = this.getHolder();
048            Level level = barrel.getLevel();
049            if (level != null) {
050                Block block = barrel.getBlock();
051                if (block instanceof BlockBarrel) {
052                    BlockBarrel blockBarrel = (BlockBarrel) block;
053                    if (blockBarrel.isOpen()) {
054                        blockBarrel.setOpen(false);
055                        level.setBlock(blockBarrel, blockBarrel, true, true);
056                        level.addSound(blockBarrel, Sound.BLOCK_BARREL_CLOSE);
057                    }
058                }
059            }
060        }
061    }
062}
