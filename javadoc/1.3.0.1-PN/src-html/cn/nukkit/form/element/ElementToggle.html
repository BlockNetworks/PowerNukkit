001package cn.nukkit.form.element;
002
003import com.google.gson.annotations.SerializedName;
004
005public class ElementToggle extends Element {
006
007    private final String type = "toggle"; //This variable is used for JSON import operations. Do NOT delete :) -- @Snake1999
008    private String text;
009    @SerializedName("default")
010    private boolean defaultValue;
011
012    public ElementToggle(String text) {
013        this(text, false);
014    }
015
016    public ElementToggle(String text, boolean defaultValue) {
017        this.text = text;
018        this.defaultValue = defaultValue;
019    }
020
021    public String getText() {
022        return text;
023    }
024
025    public void setText(String text) {
026        this.text = text;
027    }
028
029    public boolean isDefaultValue() {
030        return defaultValue;
031    }
032
033    public void setDefaultValue(boolean defaultValue) {
034        this.defaultValue = defaultValue;
035    }
036}
