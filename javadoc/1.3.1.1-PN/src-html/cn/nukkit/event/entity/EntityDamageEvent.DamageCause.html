001package cn.nukkit.event.entity;
002
003import cn.nukkit.entity.Entity;
004import cn.nukkit.event.Cancellable;
005import cn.nukkit.event.HandlerList;
006import cn.nukkit.potion.Effect;
007import cn.nukkit.utils.EventException;
008import com.google.common.collect.ImmutableMap;
009
010import java.util.EnumMap;
011import java.util.Map;
012
013/**
014 * author: MagicDroidX
015 * Nukkit Project
016 */
017public class EntityDamageEvent extends EntityEvent implements Cancellable {
018    private static final HandlerList handlers = new HandlerList();
019
020    public static HandlerList getHandlers() {
021        return handlers;
022    }
023
024    private int attackCooldown = 10;
025    private final DamageCause cause;
026
027    private final Map<DamageModifier, Float> modifiers;
028    private final Map<DamageModifier, Float> originals;
029
030    public EntityDamageEvent(Entity entity, DamageCause cause, float damage) {
031        this(entity, cause, new EnumMap<DamageModifier, Float>(DamageModifier.class) {
032            {
033                put(DamageModifier.BASE, damage);
034            }
035        });
036    }
037
038    public EntityDamageEvent(Entity entity, DamageCause cause, Map<DamageModifier, Float> modifiers) {
039        this.entity = entity;
040        this.cause = cause;
041        this.modifiers = new EnumMap<>(modifiers);
042
043        this.originals = ImmutableMap.copyOf(this.modifiers);
044
045        if (!this.modifiers.containsKey(DamageModifier.BASE)) {
046            throw new EventException("BASE Damage modifier missing");
047        }
048
049        if (entity.hasEffect(Effect.DAMAGE_RESISTANCE)) {
050            this.setDamage((float) -(this.getDamage(DamageModifier.BASE) * 0.20 * (entity.getEffect(Effect.DAMAGE_RESISTANCE).getAmplifier() + 1)), DamageModifier.RESISTANCE);
051        }
052    }
053
054    public DamageCause getCause() {
055        return cause;
056    }
057
058    public float getOriginalDamage() {
059        return this.getOriginalDamage(DamageModifier.BASE);
060    }
061
062    public float getOriginalDamage(DamageModifier type) {
063        if (this.originals.containsKey(type)) {
064            return this.originals.get(type);
065        }
066
067        return 0;
068    }
069
070    public float getDamage() {
071        return this.getDamage(DamageModifier.BASE);
072    }
073
074    public float getDamage(DamageModifier type) {
075        if (this.modifiers.containsKey(type)) {
076            return this.modifiers.get(type);
077        }
078
079        return 0;
080    }
081
082    public void setDamage(float damage) {
083        this.setDamage(damage, DamageModifier.BASE);
084    }
085
086    public void setDamage(float damage, DamageModifier type) {
087        this.modifiers.put(type, damage);
088    }
089
090    public boolean isApplicable(DamageModifier type) {
091        return this.modifiers.containsKey(type);
092    }
093
094    public float getFinalDamage() {
095        float damage = 0;
096        for (Float d : this.modifiers.values()) {
097            if (d != null) {
098                damage += d;
099            }
100        }
101
102        return damage;
103    }
104    
105    public int getAttackCooldown() {
106        return this.attackCooldown;
107    }
108    
109    public void setAttackCooldown(int attackCooldown) {
110        this.attackCooldown = attackCooldown;
111    }
112
113    public boolean canBeReducedByArmor() {
114        switch (this.cause) {
115            case FIRE_TICK:
116            case SUFFOCATION:
117            case DROWNING:
118            case HUNGER:
119            case FALL:
120            case VOID:
121            case MAGIC:
122            case SUICIDE:
123                return false;
124        }
125        return true;
126    }
127
128    public enum DamageModifier {
129        /**
130         * Raw amount of damage
131         */
132        BASE,
133        /**
134         * Damage reduction caused by wearing armor
135         */
136        ARMOR,
137        /**
138         * Additional damage caused by damager's Strength potion effect
139         */
140        STRENGTH,
141        /**
142         * Damage reduction caused by damager's Weakness potion effect
143         */
144        WEAKNESS,
145        /**
146         * Damage reduction caused by the Resistance potion effect
147         */
148        RESISTANCE,
149        /**
150         * Damage reduction caused by the Damage absorption effect
151         */
152        ABSORPTION,
153        /**
154         * Damage reduction caused by the armor enchantments worn.
155         */
156        ARMOR_ENCHANTMENTS
157    }
158
159    public enum DamageCause {
160        /**
161         * Damage caused by contact with a block such as a Cactus
162         */
163        CONTACT,
164        /**
165         * Damage caused by being attacked by another entity
166         */
167        ENTITY_ATTACK,
168        /**
169         * Damage caused by being hit by a projectile such as an Arrow
170         */
171        PROJECTILE,
172        /**
173         * Damage caused by being put in a block
174         */
175        SUFFOCATION,
176        /**
177         * Fall damage
178         */
179        FALL,
180        /**
181         * Damage caused by standing in fire
182         */
183        FIRE,
184        /**
185         * Burn damage
186         */
187        FIRE_TICK,
188        /**
189         * Damage caused by standing in lava
190         */
191        LAVA,
192        /**
193         * Damage caused by running out of air underwater
194         */
195        DROWNING,
196        /**
197         * Block explosion damage
198         */
199        BLOCK_EXPLOSION,
200        /**
201         * Entity explosion damage
202         */
203        ENTITY_EXPLOSION,
204        /**
205         * Damage caused by falling into the void
206         */
207        VOID,
208        /**
209         * Player commits suicide
210         */
211        SUICIDE,
212        /**
213         * Potion or spell damage
214         */
215        MAGIC,
216        /**
217         * Plugins
218         */
219        CUSTOM,
220        /**
221         * Damage caused by being struck by lightning
222         */
223        LIGHTNING,
224        /**
225         * Damage caused by hunger
226         */
227        HUNGER
228    }
229}
