001package cn.nukkit.command;
002
003import cn.nukkit.Server;
004import cn.nukkit.lang.TextContainer;
005import cn.nukkit.permission.Permissible;
006
007/**
008 * 能发送命令的人。<br>
009 * Who sends commands.
010 *
011 * <p>可以是一个玩家或者一个控制台。<br>
012 * That can be a player or a console.</p>
013 *
014 * @author MagicDroidX(code) @ Nukkit Project
015 * @author 粉鞋大妈(javadoc) @ Nukkit Project
016 * @see cn.nukkit.command.CommandExecutor#onCommand
017 * @since Nukkit 1.0 | Nukkit API 1.0.0
018 */
019public interface CommandSender extends Permissible {
020
021    /**
022     * 给命令发送者返回信息。<br>
023     * Sends a message to the command sender.
024     *
025     * @param message 要发送的信息。<br>Message to send.
026     * @see cn.nukkit.utils.TextFormat
027     * @since Nukkit 1.0 | Nukkit API 1.0.0
028     */
029    void sendMessage(String message);
030
031    /**
032     * 给命令发送者返回信息。<br>
033     * Sends a message to the command sender.
034     *
035     * @param message 要发送的信息。<br>Message to send.
036     * @since Nukkit 1.0 | Nukkit API 1.0.0
037     */
038    void sendMessage(TextContainer message);
039
040    /**
041     * 返回命令发送者所在的服务器。<br>
042     * Returns the server of the command sender.
043     *
044     * @return 命令发送者所在的服务器。<br>the server of the command sender.
045     * @since Nukkit 1.0 | Nukkit API 1.0.0
046     */
047    Server getServer();
048
049    /**
050     * 返回命令发送者的名称。<br>
051     * Returns the name of the command sender.
052     *
053     * <p>如果命令发送者是一个玩家，将会返回他的玩家名字(name)不是显示名字(display name)。
054     * 如果命令发送者是控制台，将会返回{@code "CONSOLE"}。<br>
055     * If this command sender is a player, will return his/her player name(not display name).
056     * If it is a console, will return {@code "CONSOLE"}.</p>
057     * <p>当你需要判断命令的执行者是不是控制台时，可以用这个：<br>
058     * When you need to determine if the sender is a console, use this:<br>
059     * {@code if(sender instanceof ConsoleCommandSender) .....;}</p>
060     *
061     * @return 命令发送者的名称。<br>the name of the command sender.
062     * @see cn.nukkit.Player#getName()
063     * @see cn.nukkit.command.ConsoleCommandSender#getName()
064     * @see cn.nukkit.plugin.PluginDescription
065     * @since Nukkit 1.0 | Nukkit API 1.0.0
066     */
067    String getName();
068
069    boolean isPlayer();
070
071}
