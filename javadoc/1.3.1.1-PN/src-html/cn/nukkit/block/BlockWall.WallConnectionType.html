119    private boolean shouldBeTall(Block above, BlockFace face) {
120        switch (above.getId()) {
121        case AIR: 
122        case SKULL_BLOCK: 
123            return false;
124        // If the bell is standing and follow the path, make it tall
125        case BELL: 
126            BlockBell bell = (BlockBell) above;
127            return bell.getAttachmentType() == BlockBell.TYPE_ATTACHMENT_STANDING && bell.getBlockFace().getAxis() != face.getAxis();
128        case COBBLE_WALL: 
129            return ((BlockWall) above).getConnectionType(face) != WallConnectionType.NONE;
130        default: 
131            if (above instanceof BlockConnectable) {
132                return ((BlockConnectable) above).isConnected(face);
133            } else if (above instanceof BlockPressurePlateBase || above instanceof BlockStairs) {
134                return true;
135            }
136            return above.isSolid() && !above.isTransparent() || shouldBeTallBasedOnBoundingBox(above, face);

140    private boolean shouldBeTallBasedOnBoundingBox(Block above, BlockFace face) {
141        AxisAlignedBB boundingBox = above.getBoundingBox();
142        if (boundingBox == null) {
143            return false;
144        }
145        boundingBox = boundingBox.getOffsetBoundingBox(-above.x, -above.y, -above.z);
146        if (boundingBox.getMinY() > 0) {
147            return false;
148        }
149        int offset = face.getXOffset();
150        if (offset < 0) {
151            return  /*&& MIN_POST_BB <= boundingBox.getMaxX()*/
152            boundingBox.getMinX() < MIN_POST_BB && boundingBox.getMinZ() < MIN_POST_BB && MAX_POST_BB < boundingBox.getMaxZ();
153        } else if (offset > 0) {
154            return  /*boundingBox.getMinX() <= MAX_POST_BB &&*/ MAX_POST_BB < boundingBox.getMaxX() && MAX_POST_BB < boundingBox.getMaxZ() && boundingBox.getMinZ() < MAX_POST_BB;
155        } else {
156            offset = face.getZOffset();
157            if (offset < 0) {
158                return  /*&& MIN_POST_BB <= boundingBox.getMaxZ()*/
159                boundingBox.getMinZ() < MIN_POST_BB && boundingBox.getMinX() < MIN_POST_BB && MIN_POST_BB < boundingBox.getMaxX();
160            } else if (offset > 0) {
161                return  /*boundingBox.getMinZ() <= MAX_POST_BB &&*/ MAX_POST_BB < boundingBox.getMaxZ() && MAX_POST_BB < boundingBox.getMaxX() && boundingBox.getMinX() < MAX_POST_BB;
162            }
163        }
164        return false;
165    }

275        if (blockFace.getHorizontalIndex() < 0) {
276            return false;
277        }
278        // Locate the 2 bit position
279        int bitIndex = getBitIndex(blockFace);
280        // Clear the 2 bits
281        int damage = getDamage() & ~(3 << bitIndex);
282        // Set the 2 bits values based on the connection type

284        // Save in memory
285        setDamage(damage);

298    private boolean recheckPostConditions(Block above) {
299        // If nothing is connected, it should be a post
300        if ((getDamage() & 4080) == 0) {
301            return true;
302        }
303        // If it's not straight, it should be a post
304        Map<BlockFace, WallConnectionType> connections = getWallConnections();
305        if (connections.size() != 2) {
306            return true;
307        }
308        Iterator<Map.Entry<BlockFace, WallConnectionType>> iterator = connections.entrySet().iterator();
309        Map.Entry<BlockFace, WallConnectionType> entryA = iterator.next();
310        Map.Entry<BlockFace, WallConnectionType> entryB = iterator.next();
311        if (entryA.getValue() != entryB.getValue() || entryA.getKey().getOpposite() != entryB.getKey()) {
312            return true;
313        }
314        Axis axis = entryA.getKey().getAxis();
315        switch (above.getId()) {
316        // These special blocks forces walls to become a post
317        case FLOWER_POT_BLOCK: 
318        case SKULL_BLOCK: 
319        case CONDUIT: 
320        case STANDING_BANNER: 
321            return true;
322        // End rods make it become a post if it's placed on the wall
323        case END_ROD: 
324            if (((Faceable) above).getBlockFace() == BlockFace.UP) {
325                return true;
326            }
327            break;
328        // If the wall above is a post, it should also be a post
329        case COBBLE_WALL: 
330            if (((BlockWall) above).isWallPost()) {
331                return true;
332            }
333            break;
334        // If the bell is standing and don't follow the path, make it a post
335        case BELL: 
336            BlockBell bell = (BlockBell) above;
337            if (bell.getAttachmentType() == BlockBell.TYPE_ATTACHMENT_STANDING && bell.getBlockFace().getAxis() == axis) {
338                return true;
339            }
340            break;
341        default: 
342            if (above instanceof BlockLantern) {
343                // Lanterns makes this become a post if they are not hanging
344                if (above.getDamage() == 0) {
345                    // Not hanging
346                    return true;
347                }
348            } else if (above.getId() == LEVER || above instanceof BlockTorch || above instanceof BlockButton) {
349                // These blocks make this become a post if they are placed down (facing up)
350                if (((Faceable) above).getBlockFace() == BlockFace.UP) {
351                    return true;
352                }
353            } else if (above instanceof BlockFenceGate) {
354                // If the gate don't follow the path, make it a post
355                if (((Faceable) above).getBlockFace().getAxis() == axis) {
356                    return true;
357                }
358            } else if (above instanceof BlockConnectable) {
359                // If the connectable block above don't share 2 equal connections, then this should be a post
360                int shared = 0;
361                for (BlockFace connection : ((BlockConnectable) above).getConnections()) {
362                    if (connections.containsKey(connection) && ++shared == 2) {
363                        break;
364                    }
365                }
366                if (shared < 2) {
367                    return true;
368                }
369            }
370        }
371        // Sign posts always makes the wall become a post
372        return above instanceof BlockSignPost;
373    }

404    private boolean connect(BlockFace blockFace, Block above, boolean recheckPost) {
405        WallConnectionType type = shouldBeTall(above, blockFace) ? WallConnectionType.TALL : WallConnectionType.SHORT;
406        if (setConnection(blockFace, type)) {
407            if (recheckPost) {
408                recheckPostConditions(above);
409            }
410            return true;
411        }
412        return false;
413    }

437    protected AxisAlignedBB recalculateBoundingBox() {
438        boolean north = this.canConnect(this.getSide(BlockFace.NORTH));
439        boolean south = this.canConnect(this.getSide(BlockFace.SOUTH));
440        boolean west = this.canConnect(this.getSide(BlockFace.WEST));
441        boolean east = this.canConnect(this.getSide(BlockFace.EAST));
442        double n = north ? 0 : 0.25;
443        double s = south ? 1 : 0.75;
444        double w = west ? 0 : 0.25;
445        double e = east ? 1 : 0.75;
446        if (north && south && !west && !east) {
447            w = 0.3125;
448            e = 0.6875;
449        } else if (!north && !south && west && east) {
450            n = 0.3125;
451            s = 0.6875;
452        }
453        return new SimpleAxisAlignedBB(this.x + w, this.y, this.z + n, this.x + e, this.y + 1.5, this.z + s);
454    }

481    private BlockFace calculateFace(Block side) {
482        Position vector = side.subtract(this);
483        Axis axis = vector.x != 0 ? Axis.X : vector.z != 0 ? Axis.Z : Axis.Y;
484        double direction = axis == Axis.X ? vector.x : axis == Axis.Y ? vector.y : vector.z;
485        return BlockFace.fromAxis(direction < 0 ? AxisDirection.NEGATIVE : AxisDirection.POSITIVE, axis);
486    }
