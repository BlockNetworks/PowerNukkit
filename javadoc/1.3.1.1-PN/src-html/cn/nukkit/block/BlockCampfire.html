026public class BlockCampfire extends BlockTransparentMeta implements Faceable {

028        this(0);
029    }

032        super(meta);

051    public double getHardness() {

053    }
054

056    public int getToolType() {
057        return ItemTool.TYPE_AXE;
058    }
059
060    @Override
061    public boolean canHarvestWithHand() {
062        return false;

065    @Override
066    public Item[] getDrops(Item item) {
067        return new Item[] { new ItemCoal(0, 1 + ThreadLocalRandom.current().nextInt(1)) };
068    }

070    @Override
071    public boolean canSilkTouch() {
072        return true;
073    }

075    @Override
076    public boolean place(Item item, Block block, Block target, BlockFace face, double fx, double fy, double fz, Player player) {
077        if (down().getId() == CAMPFIRE_BLOCK) {
078            return false;
079        }
080
081        this.setDamage(player != null ? player.getDirection().getOpposite().getHorizontalIndex() : 0);
082        Block layer1 = block.getLevelBlockAtLayer(1);
083        if (block instanceof BlockWater || block instanceof BlockIceFrosted || layer1 instanceof BlockWater || layer1 instanceof BlockIceFrosted) {
084            setExtinguished(true);
085            this.level.addSound(this, Sound.RANDOM_FIZZ, 0.5f, 2.2f);
086        }
087
088        this.level.setBlock(block, this, true, true);
089        createBlockEntity(item);
090        return true;
091    }
092
093    private BlockEntityCampfire createBlockEntity(Item item) {
094        CompoundTag nbt = new CompoundTag()
095                .putString("id", BlockEntity.CAMPFIRE)
096                .putInt("x", (int) this.x)
097                .putInt("y", (int) this.y)
098                .putInt("z", (int) this.z);
099
100        if (item.hasCustomBlockData()) {
101            Map<String, Tag> customData = item.getCustomBlockData().getTags();
102            for (Map.Entry<String, Tag> tag : customData.entrySet()) {
103                nbt.put(tag.getKey(), tag.getValue());
104            }
105        }
106
107        return (BlockEntityCampfire) BlockEntity.createBlockEntity(BlockEntity.CAMPFIRE, this.getLevel().getChunk((int) (this.x) >> 4, (int) (this.z) >> 4), nbt);
108    }
109
110    @Override
111    public boolean hasEntityCollision() {
112        return true;
113    }
114
115    @Override
116    public void onEntityCollide(Entity entity) {
117        if (!isExtinguished() && !entity.isSneaking()) {
118            entity.attack(new EntityDamageByBlockEvent(this, entity, EntityDamageEvent.DamageCause.FIRE, 1));
119        }
120    }
121
122    @Override
123    public boolean canBeActivated() {
124        return true;
125    }
126
127    @Override
128    public int onUpdate(int type) {
129        if (type == Level.BLOCK_UPDATE_NORMAL) {
130            if (!isExtinguished()) {
131                Block layer1 = getLevelBlockAtLayer(1);
132                if (layer1 instanceof BlockWater || layer1 instanceof BlockIceFrosted) {
133                    setExtinguished(true);
134                    this.level.setBlock(this, this, true, true);
135                    this.level.addSound(this, Sound.RANDOM_FIZZ, 0.5f, 2.2f);
136                }
137            }
138            return type;
139        }
140        return 0;
141    }
142
143    @Override
144    public boolean onActivate(Item item, Player player) {
145        if (item.getId() == BlockID.AIR || item.getCount() <= 0) {
146            return false;
147        }
148
149        BlockEntity entity = level.getBlockEntity(this);
150        if (!(entity instanceof BlockEntityCampfire)) {
151            entity = createBlockEntity(Item.get(0));
152        }
153
154        boolean itemUsed = false;
155        if (item.isShovel() && !isExtinguished()) {
156            setExtinguished(true);
157            this.level.setBlock(this, this, true, true);
158            this.level.addSound(this, Sound.RANDOM_FIZZ, 0.5f, 2.2f);
159            itemUsed = true;
160        } else if (item.getId() == ItemID.FLINT_AND_STEEL) {
161            item.useOn(this);
162            setExtinguished(false);
163            this.level.setBlock(this, this, true, true);
164            if (entity != null) {
165                entity.scheduleUpdate();
166            }
167            this.level.addSound(this, Sound.FIRE_IGNITE);
168            itemUsed = true;
169        }
170
171        if (entity == null) {
172            return itemUsed;
173        }
174
175        BlockEntityCampfire campfire = (BlockEntityCampfire) entity;
176        Item cloned = item.clone();
177        cloned.setCount(1);
178        CampfireInventory inventory = campfire.getInventory();
179        if(inventory.canAddItem(cloned)) {
180            CampfireRecipe recipe = this.level.getServer().getCraftingManager().matchCampfireRecipe(cloned);
181            if (recipe != null) {
182                inventory.addItem(cloned);
183                item.setCount(item.getCount() - 1);
184                return true;
185            }
186        }
187
188        return itemUsed;
189    }

211    public boolean isExtinguished() {
212        return (getDamage() & 0x4) == 0x4;
213    }
214
215    public void setExtinguished(boolean extinguished) {
216        setDamage((getDamage() & 0x3) | (extinguished? 0x4 : 0x0));
217    }
218
219    @Override
220    public BlockFace getBlockFace() {
221        return BlockFace.fromHorizontalIndex(getDamage() & 0x3);
222    }
223
224    public void setBlockFace(BlockFace face) {
225        if (face == BlockFace.UP || face == BlockFace.DOWN) {
226            return;
227        }
228
229        setDamage((getDamage() & 0x4) | face.getHorizontalIndex());
230    }

246    @Override
247    public int getComparatorInputOverride() {
248        BlockEntity blockEntity = this.level.getBlockEntity(this);
249
250        if (blockEntity instanceof BlockEntityCampfire) {
251            return ContainerInventory.calculateRedstone(((BlockEntityCampfire) blockEntity).getInventory());
252        }
253
254        return super.getComparatorInputOverride();
255    }
