056    public boolean process() {
057        Iterator<NukkitSessionListener> iterator = this.sessionListeners.iterator();
058        while (iterator.hasNext()) {
059            NukkitSessionListener listener = iterator.next();
060            Player player = listener.player;
061            if (listener.disconnectReason != null) {
062                player.close(player.getLeaveMessage(), listener.disconnectReason, false);
063                iterator.remove();
064                continue;
065            }
066            DataPacket packet;
067            while ((packet = listener.packets.poll()) != null) {
068                listener.player.handleDataPacket(packet);
069            }
070        }
071        return true;
072    }

143    @Override
144    public Integer putPacket(Player player, DataPacket packet, boolean needACK, boolean immediate) {
145        RakNetServerSession session = this.raknet.getSession(player.getSocketAddress());
146        if (session == null) {
147            return null;
148        }
149        byte[] buffer;
150        if (packet.pid() == ProtocolInfo.BATCH_PACKET) {
151            buffer = ((BatchPacket) packet).payload;
152            if (buffer == null) {
153                return null;
154            }
155        } else {
156            this.server.batchPackets(new Player[] {player}, new DataPacket[] {packet}, true);
157            return null;
158        }
159        ByteBuf byteBuf = ByteBufAllocator.DEFAULT.ioBuffer(1 + buffer.length);
160        byteBuf.writeByte(254);
161        byteBuf.writeBytes(buffer);
162        byteBuf.readerIndex(0);
163        session.send(byteBuf, immediate ? RakNetPriority.IMMEDIATE : RakNetPriority.MEDIUM, packet.reliability, packet.getChannel());
164        return null;
165    }
