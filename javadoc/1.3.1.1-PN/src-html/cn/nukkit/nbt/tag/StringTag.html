001package cn.nukkit.nbt.tag;
002
003import cn.nukkit.nbt.stream.NBTInputStream;
004import cn.nukkit.nbt.stream.NBTOutputStream;
005
006import java.io.IOException;
007import java.util.Objects;
008
009public class StringTag extends Tag {
010    public String data;
011
012    public StringTag(String name) {
013        super(name);
014    }
015
016    public StringTag(String name, String data) {
017        super(name);
018        this.data = data;
019        if (data == null) throw new IllegalArgumentException("Empty string not allowed");
020    }
021
022    @Override
023    void write(NBTOutputStream dos) throws IOException {
024        dos.writeUTF(data);
025    }
026
027    @Override
028    void load(NBTInputStream dis) throws IOException {
029        data = dis.readUTF();
030    }
031
032    @Override
033    public String parseValue() {
034        return this.data;
035    }
036
037    @Override
038    public byte getId() {
039        return TAG_String;
040    }
041
042    @Override
043    public String toString() {
044        return "StringTag " + this.getName() + " (data: " + data + ")";
045    }
046
047    @Override
048    public Tag copy() {
049        return new StringTag(getName(), data);
050    }
051
052    @Override
053    public boolean equals(Object obj) {
054        if (super.equals(obj)) {
055            StringTag o = (StringTag) obj;
056            return ((data == null && o.data == null) || (data != null && data.equals(o.data)));
057        }
058        return false;
059    }
060    
061    @Override
062    public int hashCode() {
063        return Objects.hash(super.hashCode(), data);
064    }
065}
