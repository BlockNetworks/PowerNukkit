001package cn.nukkit.utils;
002
003import cn.nukkit.Server;
004
005import java.lang.management.ManagementFactory;
006import java.lang.management.MonitorInfo;
007import java.lang.management.ThreadInfo;
008
009public class Watchdog extends Thread {
010
011    private final Server server;
012    private final long time;
013    public boolean running;
014    private boolean responding = true;
015
016    public Watchdog(Server server, long time) {
017        this.server = server;
018        this.time = time;
019        this.running = true;
020        this.setName("Watchdog");
021    }
022
023    public void kill() {
024        running = false;
025        synchronized (this) {
026            this.notifyAll();
027        }
028    }
029
030    @Override
031    public void run() {
032        while (this.running) {
033            long current = server.getNextTick();
034            if (current != 0) {
035                long diff = System.currentTimeMillis() - current;
036                if (!responding && diff > time * 2) {
037                    System.exit(1); // Kill the server if it gets stuck on shutdown
038                }
039                if (server.isRunning() && diff > time) {
040                    if (responding) {
041                        MainLogger logger = this.server.getLogger();
042                        logger.emergency("--------- Server stopped responding --------- (" + Math.round(diff / 1000d) + "s)");
043                        logger.emergency("Please report this to PowerNukkit:");
044                        logger.emergency(" - https://github.com/PowerNukkit/PowerNukkit/issues/new");
045                        logger.emergency("---------------- Main thread ----------------");
046
047                        dumpThread(ManagementFactory.getThreadMXBean().getThreadInfo(this.server.getPrimaryThread().getId(), Integer.MAX_VALUE), logger);
048
049                        logger.emergency("---------------- All threads ----------------");
050                        ThreadInfo[] threads = ManagementFactory.getThreadMXBean().dumpAllThreads(true, true);
051                        for (int i = 0; i < threads.length; i++) {
052                            if (i != 0) logger.emergency("------------------------------");
053                            dumpThread(threads[i], logger);
054                        }
055                        logger.emergency("---------------------------------------------");
056                        responding = false;
057                        this.server.forceShutdown();
058                    }
059                } else {
060                    responding = true;
061                }
062            }
063            try {
064                synchronized (this) {
065                    this.wait(Math.max(time / 4, 1000));
066                }
067            } catch (InterruptedException ignore) {}
068        }
069    }
070
071    private static void dumpThread(ThreadInfo thread, Logger logger) {
072        logger.emergency("Current Thread: " + thread.getThreadName());
073        logger.emergency("\tPID: " + thread.getThreadId() + " | Suspended: " + thread.isSuspended() + " | Native: " + thread.isInNative() + " | State: " + thread.getThreadState());
074        // Monitors
075        if (thread.getLockedMonitors().length != 0) {
076            logger.emergency("\tThread is waiting on monitor(s):");
077            for (MonitorInfo monitor : thread.getLockedMonitors()) {
078                logger.emergency("\t\tLocked on:" + monitor.getLockedStackFrame());
079            }
080        }
081
082        logger.emergency("\tStack:");
083        for (StackTraceElement stack : thread.getStackTrace()) {
084            logger.emergency("\t\t" + stack);
085        }
086    }
087}
