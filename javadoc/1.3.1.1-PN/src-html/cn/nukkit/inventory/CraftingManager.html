041    public static final Comparator<Item> recipeComparator = (i1, i2) -> {
042        if (i1.getId() > i2.getId()) {
043            return 1;
044        } else if (i1.getId() < i2.getId()) {
045            return -1;
046        } else if (i1.getDamage() > i2.getDamage()) {
047            return 1;
048        } else if (i1.getDamage() < i2.getDamage()) {
049            return -1;
050        } else return Integer.compare(i1.getCount(), i2.getCount());

199    public void rebuildPacket() {
200        CraftingDataPacket pk = new CraftingDataPacket();
201        pk.cleanRecipes = true;
202        for (Recipe recipe : this.getRecipes()) {
203            if (recipe instanceof ShapedRecipe) {
204                pk.addShapedRecipe((ShapedRecipe) recipe);
205            } else if (recipe instanceof ShapelessRecipe) {
206                pk.addShapelessRecipe((ShapelessRecipe) recipe);
207            }
208        }
209        for (Map<UUID, CartographyRecipe> map : cartographyRecipes.values()) {
210            for (CartographyRecipe recipe : map.values()) {
211                pk.addCartographyRecipe(recipe);
212            }
213        }
214        for (FurnaceRecipe recipe : this.getFurnaceRecipes().values()) {
215            pk.addFurnaceRecipe(recipe);
216        }
217        for (SmokerRecipe recipe : smokerRecipes.values()) {
218            pk.addSmokerRecipe(recipe);
219        }
220        for (BlastFurnaceRecipe recipe : blastFurnaceRecipes.values()) {
221            pk.addBlastFurnaceRecipe(recipe);
222        }
223        for (CampfireRecipe recipe : campfireRecipes.values()) {
224            pk.addCampfireRecipeRecipe(recipe);
225        }
226        for (BrewingRecipe recipe : brewingRecipes.values()) {
227            pk.addBrewingRecipe(recipe);
228        }
229        for (ContainerRecipe recipe : containerRecipes.values()) {
230            pk.addContainerRecipe(recipe);
231        }
232        for (StonecutterRecipe recipe : stonecutterRecipes.values()) {
233            pk.addStonecutterRecipe(recipe);

236        packet = pk.compress(Deflater.BEST_COMPRESSION);

322    public void registerRecipe(Recipe recipe) {
323        UUID id = null;
324        if (recipe instanceof CraftingRecipe || recipe instanceof StonecutterRecipe) {
325            id = Utils.dataToUUID(String.valueOf(++RECIPE_COUNT), String.valueOf(recipe.getResult().getId()), String.valueOf(recipe.getResult().getDamage()), String.valueOf(recipe.getResult().getCount()), Arrays.toString(recipe.getResult().getCompoundTag()));
326        }
327        if (recipe instanceof CraftingRecipe) {
328            ((CraftingRecipe) recipe).setId(id);
329            this.recipes.add(recipe);
330        } else if (recipe instanceof StonecutterRecipe) {
331            ((StonecutterRecipe) recipe).setId(id);
332        }
333        recipe.registerToCraftingManager(this);
334    }

389    public CartographyRecipe matchCartographyRecipe(List<Item> inputList, Item primaryOutput, List<Item> extraOutputList) {
390        int outputHash = getItemHash(primaryOutput);
391        if (cartographyRecipes.containsKey(outputHash)) {
392            inputList.sort(recipeComparator);
393            UUID inputHash = getMultiItemHash(inputList);
394            Map<UUID, CartographyRecipe> recipes = cartographyRecipes.get(outputHash);
395            if (recipes == null) {
396                return null;
397            }
398            CartographyRecipe recipe = recipes.get(inputHash);
399            if (recipe != null && recipe.matchItems(inputList, extraOutputList) || matchItemsAccumulation(recipe, inputList, primaryOutput, extraOutputList)) {
400                return recipe;
401            }
402            for (CartographyRecipe cartographyRecipe : recipes.values()) {
403                if (cartographyRecipe.matchItems(inputList, extraOutputList) || matchItemsAccumulation(cartographyRecipe, inputList, primaryOutput, extraOutputList)) {
404                    return cartographyRecipe;
405                }
406            }
407        }
408        return null;
409    }

411    public CraftingRecipe matchRecipe(List<Item> inputList, Item primaryOutput, List<Item> extraOutputList) {
412        //TODO: try to match special recipes before anything else (first they need to be implemented!)
413        int outputHash = getItemHash(primaryOutput);
414        if (this.shapedRecipes.containsKey(outputHash)) {
415            inputList.sort(recipeComparator);
416            UUID inputHash = getMultiItemHash(inputList);
417            Map<UUID, ShapedRecipe> recipeMap = shapedRecipes.get(outputHash);
418            if (recipeMap != null) {
419                ShapedRecipe recipe = recipeMap.get(inputHash);
420                if (recipe != null && (recipe.matchItems(inputList, extraOutputList) || matchItemsAccumulation(recipe, inputList, primaryOutput, extraOutputList))) {
421                    return recipe;
422                }
423                for (ShapedRecipe shapedRecipe : recipeMap.values()) {
424                    if (shapedRecipe.matchItems(inputList, extraOutputList) || matchItemsAccumulation(shapedRecipe, inputList, primaryOutput, extraOutputList)) {
425                        return shapedRecipe;
426                    }
427                }
428            }
429        }
430        if (shapelessRecipes.containsKey(outputHash)) {
431            inputList.sort(recipeComparator);
432            UUID inputHash = getMultiItemHash(inputList);
433            Map<UUID, ShapelessRecipe> recipes = shapelessRecipes.get(outputHash);
434            if (recipes == null) {
435                return null;
436            }
437            ShapelessRecipe recipe = recipes.get(inputHash);
438            if (recipe != null && (recipe.matchItems(inputList, extraOutputList) || matchItemsAccumulation(recipe, inputList, primaryOutput, extraOutputList))) {
439                return recipe;
440            }
441            for (ShapelessRecipe shapelessRecipe : recipes.values()) {
442                if (shapelessRecipe.matchItems(inputList, extraOutputList) || matchItemsAccumulation(shapelessRecipe, inputList, primaryOutput, extraOutputList)) {
443                    return shapelessRecipe;
444                }
445            }
446        }
447        return null;

450    private boolean matchItemsAccumulation(CraftingRecipe recipe, List<Item> inputList, Item primaryOutput, List<Item> extraOutputList) {
451        Item recipeResult = recipe.getResult();
452        if (primaryOutput.equals(recipeResult, recipeResult.hasMeta(), recipeResult.hasCompoundTag()) && primaryOutput.getCount() % recipeResult.getCount() == 0) {
453            int multiplier = primaryOutput.getCount() / recipeResult.getCount();
454            return recipe.matchItems(inputList, extraOutputList, multiplier);
455        }
456        return false;
457    }
