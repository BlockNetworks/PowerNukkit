024package co.aikar.timings;
025
026import cn.nukkit.Server;
027
028import java.util.*;
029
030public class TimingsManager {
031    static final Map<TimingIdentifier, Timing> TIMING_MAP = Collections.synchronizedMap(new HashMap<>(256, 0.5f));
032
033    static final Queue<Timing> TIMINGS = new ArrayDeque<>();
034    static final ArrayDeque<TimingsHistory.MinuteReport> MINUTE_REPORTS = new ArrayDeque<>();
035
036    static Queue<TimingsHistory> HISTORY = new BoundedQueue<>(12);
037
038    static Timing CURRENT;
039
040    static long timingStart = 0;
041    static long historyStart = 0;
042    static boolean needsFullReset = false;
043    static boolean needsRecheckEnabled = false;
044
045    static void reset() {
046        needsFullReset = true;
047    }
048
049    /**
050     * Called every tick to count the number of times a timer caused TPS loss.
051     */
052    static void tick() {
053        if (Timings.isTimingsEnabled()) {
054            boolean violated = Timings.fullServerTickTimer.isViolated();
055
056            synchronized (TIMINGS) {
057                for (Timing timing : TIMINGS) {
058                    if (timing.isSpecial()) {
059                        // Called manually
060                        continue;
061                    }
062
063                    timing.tick(violated);
064                }
065            }
066
067            TimingsHistory.playerTicks += Server.getInstance().getOnlinePlayers().size();
068            TimingsHistory.timedTicks++;
069        }
070    }
071
072    static void recheckEnabled() {
073        synchronized (TIMING_MAP) {
074            TIMING_MAP.values().forEach(Timing::checkEnabled);
075        }
076
077        needsRecheckEnabled = false;
078    }
079
080    static void resetTimings() {
081        if (needsFullReset) {
082            // Full resets need to re-check every handlers enabled state
083            // Timing map can be modified from async so we must sync on it.
084            synchronized (TIMING_MAP) {
085                for (Timing timing : TIMING_MAP.values()) {
086                    timing.reset(true);
087                }
088            }
089
090            HISTORY.clear();
091            needsFullReset = false;
092            needsRecheckEnabled = false;
093            timingStart = System.currentTimeMillis();
094        } else {
095            // Soft resets only need to act on timings that have done something
096            // Handlers can only be modified on main thread.
097            for (Timing timing : TIMINGS) {
098                timing.reset(false);
099            }
100        }
101
102        TIMINGS.clear();
103        MINUTE_REPORTS.clear();
104
105        TimingsHistory.resetTicks(true);
106        historyStart = System.currentTimeMillis();
107    }
108
109    public static Timing getTiming(String name) {
110        return getTiming(null, name, null);
111    }
112
113    static Timing getTiming(String group, String name, Timing groupTiming) {
114        TimingIdentifier id = new TimingIdentifier(group, name, groupTiming);
115        return TIMING_MAP.computeIfAbsent(id, k -> new Timing(id));
116    }
117
118    static final class BoundedQueue<E> extends LinkedList<E> {
119        final int maxSize;
120
121        BoundedQueue(int maxSize) {
122            if (maxSize <= 0) {
123                throw new IllegalArgumentException("maxSize must be greater than zero");
124            }
125
126            this.maxSize = maxSize;
127        }
128
129        @Override
130        public boolean add(E e) {
131            if (this.size() == maxSize) {
132                this.remove();
133            }
134
135            return super.add(e);
136        }
137    }
138}
